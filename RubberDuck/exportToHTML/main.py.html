61       # alarm sound effect

73       # behavior for 'break' action
74       def timedBreak(self, instance):
75           self.greeting.text = "See you in 30 minutes!"
76           App.get_running_app().root_window.minimize()
77           Clock.schedule_once(self.breakOver, 5) # time set to 5 seconds for testing and demoing
78   
79       # restoring app on break finish
80       def breakOver(self, instance):
81           self.playSound_quack()
82           self.playSound_alarm()
83           App.get_running_app().root_window.restore()
84           self.greeting.text = "Welcome back!"

86       # 'dismiss' handling
87       def checkBackLater(self, instance):
