6    class Object:
7        def __init__(self, initObj=None):
8            self.__dict__["_obj_"] = initObj
9            #       self._obj_ = initObj
10           if initObj is not None:
11               initObj.AttachObject(self)
12   
13       def __del__(self):
14           self.close()
15   
16       def __getattr__(
17           self, attr
18       ):  # Make this object look like the underlying win32ui one.
19           # During cleanup __dict__ is not available, causing recursive death.
20           if not attr.startswith("__"):
21               try:
22                   o = self.__dict__["_obj_"]
23                   if o is not None:
24                       return getattr(o, attr)
25                   # Only raise this error for non "internal" names -
26                   # Python may be calling __len__, __nonzero__, etc, so
27                   # we dont want this exception
28                   if attr[0] != "_" and attr[-1] != "_":
29                       raise win32ui.error("The MFC object has died.")
30               except KeyError:
31                   # No _obj_ at all - dont report MFC object died when there isnt one!
32                   pass
33           raise AttributeError(attr)
34   
35       def OnAttachedObjectDeath(self):
36           #       print "object", self.__class__.__name__, "dieing"
37           self._obj_ = None
38   
39       def close(self):
40           if "_obj_" in self.__dict__:
41               if self._obj_ is not None:
42                   self._obj_.AttachObject(None)
43                   self._obj_ = None
