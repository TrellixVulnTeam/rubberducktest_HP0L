1    ''' 
2    RecycleView Data Model 
3    ====================== 
4     
5    .. versionadded:: 1.10.0 
6     
7    The data model part of the RecycleView model-view-controller pattern. 
8     
9    It defines the models (classes) that store the data associated with a 
10   :class:`~kivy.uix.recycleview.RecycleViewBehavior`. Each model (class) 
11   determines how the data is stored and emits requests to the controller 
12   (:class:`~kivy.uix.recycleview.RecycleViewBehavior`) when the data is 
13   modified. 
14   '''

17   from kivy.event import EventDispatcher
18   from functools import partial

23   def recondition_slice_assign(val, last_len, new_len):
24       if not isinstance(val, slice):
25           return slice(val, val + 1)
26   
27       diff = new_len - last_len
28   
29       start, stop, step = val.start, val.stop, val.step
30       if stop <= start:
31           return slice(0, 0)
32   
33       if step is not None and step != 1:
34           assert last_len == new_len
35           if stop < 0:
36               stop = max(0, last_len + stop)
37           stop = min(last_len, stop)
38   
39           if start < 0:
40               start = max(0, last_len + start)
41           start = min(last_len, start)
42   
43           return slice(start, stop, step)
44   
45       if start < 0:
46           start = last_len + start
47       if stop < 0:
48           stop = last_len + stop
49   
50       # whatever, too complicated don't try to compute it
51       if (start < 0 or stop < 0 or start > last_len or stop > last_len or
52               new_len != last_len):
53           return None
54   
55       return slice(start, stop)

58   class RecycleDataModelBehavior(object):
59       """:class:`RecycleDataModelBehavior` is the base class for the models 
60       that describes and provides the data for the 
61       :class:`~kivy.uix.recycleview.RecycleViewBehavior`. 
62    
63       :Events: 
64           `on_data_changed`: 
65               Fired when the data changes. The event may dispatch 
66               keyword arguments specific to each implementation of the data 
67               model. 
68               When dispatched, the event and keyword arguments are forwarded to 
69               :meth:`~kivy.uix.recycleview.RecycleViewBehavior.\ 
70   refresh_from_data`. 
71       """
72   
73       __events__ = ("on_data_changed", )
74   
75       recycleview = ObjectProperty(None, allownone=True)
76       '''The 
77       :class:`~kivy.uix.recycleview.RecycleViewBehavior` instance 
78       associated with this data model. 
79       '''
80   
81       def attach_recycleview(self, rv):
82           '''Associates a 
83           :class:`~kivy.uix.recycleview.RecycleViewBehavior` with 
84           this data model. 
85           '''
86           self.recycleview = rv
87           if rv:
88               self.fbind('on_data_changed', rv.refresh_from_data)
89   
90       def detach_recycleview(self):
91           '''Removes the 
92           :class:`~kivy.uix.recycleview.RecycleViewBehavior` 
93           associated with this data model. 
94           '''
95           rv = self.recycleview
96           if rv:
97               self.funbind('on_data_changed', rv.refresh_from_data)
98           self.recycleview = None
99   
100      def on_data_changed(self, *largs, **kwargs):
101          pass

104  class RecycleDataModel(RecycleDataModelBehavior, EventDispatcher):
105      '''An implementation of :class:`RecycleDataModelBehavior` that keeps the 
106      data in a indexable list. See :attr:`data`. 
107   
108      When data changes this class currently dispatches `on_data_changed`  with 
109      one of the following additional keyword arguments. 
110   
111      `none`: no keyword argument 
112          With no additional argument it means a generic data change. 
113      `removed`: a slice or integer 
114          The value is a slice or integer indicating the indices removed. 
115      `appended`: a slice 
116          The slice in :attr:`data` indicating the first and last new items 
117          (i.e. the slice pointing to the new items added at the end). 
118      `inserted`: a integer 
119          The index in :attr:`data` where a new data item was inserted. 
120      `modified`: a slice 
121          The slice with the indices where the data has been modified. 
122          This currently does not allow changing of size etc. 
123      '''
124  
125      data = ListProperty([])
126      '''Stores the model's data using a list. 
127   
128      The data for a item at index `i` can also be accessed with 
129      :class:`RecycleDataModel` `[i]`. 
130      '''
131  
132      _last_len = 0
133  
134      def __init__(self, **kwargs):
135          self.fbind('data', self._on_data_callback)
136          super(RecycleDataModel, self).__init__(**kwargs)
137  
138      def __getitem__(self, index):
139          return self.data[index]
140  
141      @property
142      def observable_dict(self):
143          '''A dictionary instance, which when modified will trigger a `data` and 
144          consequently an `on_data_changed` dispatch. 
145          '''
146          return partial(ObservableDict, self.__class__.data, self)
147  
148      def attach_recycleview(self, rv):
149          super(RecycleDataModel, self).attach_recycleview(rv)
150          if rv:
151              self.fbind('data', rv._dispatch_prop_on_source, 'data')
152  
153      def detach_recycleview(self):
154          rv = self.recycleview
155          if rv:
156              self.funbind('data', rv._dispatch_prop_on_source, 'data')
157          super(RecycleDataModel, self).detach_recycleview()
158  
159      def _on_data_callback(self, instance, value):
160          last_len = self._last_len
161          new_len = self._last_len = len(self.data)
162          op, val = value.last_op
163  
164          if op == '__setitem__':
165              val = recondition_slice_assign(val, last_len, new_len)
166              if val is not None:
167                  self.dispatch('on_data_changed', modified=val)
168              else:
169                  self.dispatch('on_data_changed')
170          elif op == '__delitem__':
171              self.dispatch('on_data_changed', removed=val)
172          elif op == '__setslice__':
173              val = recondition_slice_assign(slice(*val), last_len, new_len)
174              if val is not None:
175                  self.dispatch('on_data_changed', modified=val)
176              else:
177                  self.dispatch('on_data_changed')
178          elif op == '__delslice__':
179              self.dispatch('on_data_changed', removed=slice(*val))
180          elif op == '__iadd__' or op == '__imul__':
181              self.dispatch('on_data_changed', appended=slice(last_len, new_len))
182          elif op == 'append':
183              self.dispatch('on_data_changed', appended=slice(last_len, new_len))
184          elif op == 'insert':
185              self.dispatch('on_data_changed', inserted=val)
186          elif op == 'pop':
187              if val:
188                  self.dispatch('on_data_changed', removed=val[0])
189              else:
190                  self.dispatch('on_data_changed', removed=last_len - 1)
191          elif op == 'extend':
192              self.dispatch('on_data_changed', appended=slice(last_len, new_len))
193          else:
194              self.dispatch('on_data_changed')
