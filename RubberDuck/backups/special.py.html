22   class TextLexer(Lexer):
23       """ 
24       "Null" lexer, doesn't highlight anything. 
25       """
26       name = 'Text only'
27       aliases = ['text']
28       filenames = ['*.txt']
29       mimetypes = ['text/plain']
30       priority = 0.01
31   
32       def get_tokens_unprocessed(self, text):
33           yield 0, Text, text
34   
35       def analyse_text(text):
36           return TextLexer.priority

57   class RawTokenLexer(Lexer):
58       """ 
59       Recreate a token stream formatted with the `RawTokenFormatter`. 
60    
61       Additional options accepted: 
62    
63       `compress` 
64           If set to ``"gz"`` or ``"bz2"``, decompress the token stream with 
65           the given compression algorithm before lexing (default: ``""``). 
66       """
67       name = 'Raw token data'
68       aliases = []
69       filenames = []
70       mimetypes = ['application/x-pygments-tokens']
71   
72       def __init__(self, **options):
73           self.compress = get_choice_opt(options, 'compress',
74                                          ['', 'none', 'gz', 'bz2'], '')
75           Lexer.__init__(self, **options)
76   
77       def get_tokens(self, text):
78           if self.compress:
79               if isinstance(text, str):
80                   text = text.encode('latin1')
81               try:
82                   if self.compress == 'gz':
83                       import gzip
84                       text = gzip.decompress(text)
85                   elif self.compress == 'bz2':
86                       import bz2
87                       text = bz2.decompress(text)
88               except OSError:
89                   yield Error, text.decode('latin1')
90           if isinstance(text, bytes):
91               text = text.decode('latin1')
92   
93           # do not call Lexer.get_tokens() because stripping is not optional.
94           text = text.strip('\n') + '\n'
95           for i, t, v in self.get_tokens_unprocessed(text):
96               yield t, v
97   
98       def get_tokens_unprocessed(self, text):
99           length = 0
100          for match in line_re.finditer(text):
101              try:
102                  ttypestr, val = match.group().rstrip().split('\t', 1)
103                  ttype = _ttype_cache.get(ttypestr)
104                  if not ttype:
105                      ttype = Token
106                      ttypes = ttypestr.split('.')[1:]
107                      for ttype_ in ttypes:
108                          if not ttype_ or not ttype_[0].isupper():
109                              raise ValueError('malformed token name')
110                          ttype = getattr(ttype, ttype_)
111                      _ttype_cache[ttypestr] = ttype
112                  val = ast.literal_eval(val)
113                  if not isinstance(val, str):
114                      raise ValueError('expected str')
115              except (SyntaxError, ValueError):
116                  val = match.group()
117                  ttype = Error
118              yield length, ttype, val
119              length += len(val)
