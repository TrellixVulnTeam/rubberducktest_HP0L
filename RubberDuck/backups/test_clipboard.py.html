30   class TestBitmap(unittest.TestCase):
31       def setUp(self):
32           self.bmp_handle = None
33           try:
34               this_file = __file__
35           except NameError:
36               this_file = sys.argv[0]
37           this_dir = os.path.dirname(this_file)
38           self.bmp_name = os.path.join(
39               os.path.abspath(this_dir), "..", "Demos", "images", "smiley.bmp"
40           )
41           self.failUnless(os.path.isfile(self.bmp_name), self.bmp_name)
42           flags = win32con.LR_DEFAULTSIZE | win32con.LR_LOADFROMFILE
43           self.bmp_handle = win32gui.LoadImage(
44               0, self.bmp_name, win32con.IMAGE_BITMAP, 0, 0, flags
45           )
46           self.failUnless(self.bmp_handle, "Failed to get a bitmap handle")
47   
48       def tearDown(self):
49           if self.bmp_handle:
50               win32gui.DeleteObject(self.bmp_handle)
51   
52       def test_bitmap_roundtrip(self):
53           OpenClipboard()
54           try:
55               SetClipboardData(win32con.CF_BITMAP, self.bmp_handle)
56               got_handle = GetClipboardDataHandle(win32con.CF_BITMAP)
57               self.failUnlessEqual(got_handle, self.bmp_handle)
58           finally:
59               CloseClipboard()

70           val = "test-\a9har"

74       def test_unicode_text(self):
75           val = "test-val"
76           SetClipboardText(val)
77           # GetClipboardData doesn't to auto string conversions - so on py3k,
78           # CF_TEXT returns bytes.
79           expected = str2bytes(val)
80           self.failUnlessEqual(GetClipboardData(win32con.CF_TEXT), expected)
81           SetClipboardText(val, win32con.CF_UNICODETEXT)
82           self.failUnlessEqual(GetClipboardData(win32con.CF_UNICODETEXT), val)
83

85           val = str2bytes("test")

90   class TestGlobalMemory(unittest.TestCase):
91       def setUp(self):
92           OpenClipboard()
93   
94       def tearDown(self):
95           CloseClipboard()
96   
97       def test_mem(self):
98           val = str2bytes("test")
99           expected = str2bytes("test\0")
100          SetClipboardData(win32con.CF_TEXT, val)
101          # Get the raw data - this will include the '\0'
102          raw_data = GetGlobalMemory(GetClipboardDataHandle(win32con.CF_TEXT))
103          self.failUnlessEqual(expected, raw_data)
104  
105      def test_bad_mem(self):
106          self.failUnlessRaises(pywintypes.error, GetGlobalMemory, 0)
107          self.failUnlessRaises(pywintypes.error, GetGlobalMemory, -1)
108          if sys.getwindowsversion()[0] <= 5:
109              # For some reason, the value '1' dies from a 64bit process, but
110              # "works" (ie, gives the correct exception) from a 32bit process.
111              # just silently skip this value on Vista.
112              self.failUnlessRaises(pywintypes.error, GetGlobalMemory, 1)
113  
114      def test_custom_mem(self):
115          test_data = str2bytes("hello\x00\xff")
116          test_buffer = array.array("b", test_data)
117          cf = RegisterClipboardFormat(custom_format_name)
118          self.failUnlessEqual(custom_format_name, GetClipboardFormatName(cf))
119          SetClipboardData(cf, test_buffer)
120          hglobal = GetClipboardDataHandle(cf)
121          data = GetGlobalMemory(hglobal)
122          self.failUnlessEqual(data, test_data)
