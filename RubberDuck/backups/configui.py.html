47   class ScintillaFormatPropertyPage(dialog.PropertyPage):
48       def __init__(self, scintillaClass=None, caption=0):
49           self.scintillaClass = scintillaClass
50           dialog.PropertyPage.__init__(self, win32ui.IDD_PP_FORMAT, caption=caption)
51   
52       def OnInitDialog(self):
53           try:
54               if self.scintillaClass is None:
55                   from . import control
56   
57                   sc = control.CScintillaEdit
58               else:
59                   sc = self.scintillaClass
60   
61               self.scintilla = sc()
62               style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.ES_MULTILINE
63               # Convert the rect size
64               rect = self.MapDialogRect((5, 5, 120, 75))
65               self.scintilla.CreateWindow(style, rect, self, 111)
66               self.HookNotify(self.OnBraceMatch, scintillacon.SCN_CHECKBRACE)
67               self.scintilla.HookKeyStroke(self.OnEsc, 27)
68               self.scintilla.SCISetViewWS(1)
69               self.pos_bstart = self.pos_bend = self.pos_bbad = 0
70   
71               colorizer = self.scintilla._GetColorizer()
72               text = colorizer.GetSampleText()
73               items = text.split("|", 2)
74               pos = len(items[0])
75               self.scintilla.SCIAddText("".join(items))
76               self.scintilla.SetSel(pos, pos)
77               self.scintilla.ApplyFormattingStyles()
78               self.styles = self.scintilla._GetColorizer().styles
79   
80               self.cbo = self.GetDlgItem(win32ui.IDC_COMBO1)
81               for c in paletteVGA:
82                   self.cbo.AddString(c[0])
83   
84               self.cboBoldItalic = self.GetDlgItem(win32ui.IDC_COMBO2)
85               for item in ["Bold Italic", "Bold", "Italic", "Regular"]:
86                   self.cboBoldItalic.InsertString(0, item)
87   
88               self.butIsDefault = self.GetDlgItem(win32ui.IDC_CHECK1)
89               self.butIsDefaultBackground = self.GetDlgItem(win32ui.IDC_CHECK2)
90               self.listbox = self.GetDlgItem(win32ui.IDC_LIST1)
91               self.HookCommand(self.OnListCommand, win32ui.IDC_LIST1)
92               names = list(self.styles.keys())
93               names.sort()
94               for name in names:
95                   if self.styles[name].aliased is None:
96                       self.listbox.AddString(name)
97               self.listbox.SetCurSel(0)
98   
99               idc = win32ui.IDC_RADIO1
100              if not self.scintilla._GetColorizer().bUseFixed:
101                  idc = win32ui.IDC_RADIO2
102              self.GetDlgItem(idc).SetCheck(1)
103              self.UpdateUIForStyle(self.styles[names[0]])
104  
105              self.scintilla.HookFormatter(self)
106              self.HookCommand(self.OnButDefaultFixedFont, win32ui.IDC_BUTTON1)
107              self.HookCommand(self.OnButDefaultPropFont, win32ui.IDC_BUTTON2)
108              self.HookCommand(self.OnButThisFont, win32ui.IDC_BUTTON3)
109              self.HookCommand(self.OnButUseDefaultFont, win32ui.IDC_CHECK1)
110              self.HookCommand(self.OnButThisBackground, win32ui.IDC_BUTTON4)
111              self.HookCommand(self.OnButUseDefaultBackground, win32ui.IDC_CHECK2)
112              self.HookCommand(self.OnStyleUIChanged, win32ui.IDC_COMBO1)
113              self.HookCommand(self.OnStyleUIChanged, win32ui.IDC_COMBO2)
114              self.HookCommand(self.OnButFixedOrDefault, win32ui.IDC_RADIO1)
115              self.HookCommand(self.OnButFixedOrDefault, win32ui.IDC_RADIO2)
116          except:
117              import traceback
118  
119              traceback.print_exc()

129      def GetSelectedStyle(self):
130          return self.styles[self.listbox.GetText(self.listbox.GetCurSel())]

132      def _DoButDefaultFont(self, extra_flags, attr):
133          baseFormat = getattr(self.scintilla._GetColorizer(), attr)
134          flags = (
135              extra_flags
136              | win32con.CF_SCREENFONTS
137              | win32con.CF_EFFECTS
138              | win32con.CF_FORCEFONTEXIST
139          )
140          d = win32ui.CreateFontDialog(baseFormat, flags, None, self)
141          if d.DoModal() == win32con.IDOK:
142              setattr(self.scintilla._GetColorizer(), attr, d.GetCharFormat())
143              self.OnStyleUIChanged(0, win32con.BN_CLICKED)
144  
145      def OnButDefaultFixedFont(self, id, code):
146          if code == win32con.BN_CLICKED:
147              self._DoButDefaultFont(win32con.CF_FIXEDPITCHONLY, "baseFormatFixed")
148              return 1
149  
150      def OnButDefaultPropFont(self, id, code):
151          if code == win32con.BN_CLICKED:
152              self._DoButDefaultFont(win32con.CF_SCALABLEONLY, "baseFormatProp")
153              return 1

155      def OnButFixedOrDefault(self, id, code):
156          if code == win32con.BN_CLICKED:
157              bUseFixed = id == win32ui.IDC_RADIO1
158              self.GetDlgItem(win32ui.IDC_RADIO1).GetCheck() != 0
159              self.scintilla._GetColorizer().bUseFixed = bUseFixed
160              self.scintilla.ApplyFormattingStyles(0)
161              return 1
162  
163      def OnButThisFont(self, id, code):
164          if code == win32con.BN_CLICKED:
165              flags = (
166                  win32con.CF_SCREENFONTS
167                  | win32con.CF_EFFECTS
168                  | win32con.CF_FORCEFONTEXIST
169              )
170              style = self.GetSelectedStyle()
171              # If the selected style is based on the default, we need to apply
172              # the default to it.
173              def_format = self.scintilla._GetColorizer().GetDefaultFormat()
174              format = style.GetCompleteFormat(def_format)
175              d = win32ui.CreateFontDialog(format, flags, None, self)
176              if d.DoModal() == win32con.IDOK:
177                  style.format = d.GetCharFormat()
178                  self.scintilla.ApplyFormattingStyles(0)
179              return 1
180  
181      def OnButUseDefaultFont(self, id, code):
182          if code == win32con.BN_CLICKED:
183              isDef = self.butIsDefault.GetCheck()
184              self.GetDlgItem(win32ui.IDC_BUTTON3).EnableWindow(not isDef)
185              if isDef:  # Being reset to the default font.
186                  style = self.GetSelectedStyle()
187                  style.ForceAgainstDefault()
188                  self.UpdateUIForStyle(style)
189                  self.scintilla.ApplyFormattingStyles(0)
190              else:
191                  # User wants to override default -
192                  # do nothing!
193                  pass
194  
195      def OnButThisBackground(self, id, code):
196          if code == win32con.BN_CLICKED:
197              style = self.GetSelectedStyle()
198              bg = win32api.RGB(0xFF, 0xFF, 0xFF)
199              if style.background != CLR_INVALID:
200                  bg = style.background
201              d = win32ui.CreateColorDialog(bg, 0, self)
202              if d.DoModal() == win32con.IDOK:
203                  style.background = d.GetColor()
204                  self.scintilla.ApplyFormattingStyles(0)
205              return 1
206  
207      def OnButUseDefaultBackground(self, id, code):
208          if code == win32con.BN_CLICKED:
209              isDef = self.butIsDefaultBackground.GetCheck()
210              self.GetDlgItem(win32ui.IDC_BUTTON4).EnableWindow(not isDef)
211              if isDef:  # Being reset to the default color
212                  style = self.GetSelectedStyle()
213                  style.background = style.default_background
214                  self.UpdateUIForStyle(style)
215                  self.scintilla.ApplyFormattingStyles(0)
216              else:
217                  # User wants to override default -
218                  # do nothing!
219                  pass

221      def OnListCommand(self, id, code):
222          if code == win32con.LBN_SELCHANGE:
223              style = self.GetSelectedStyle()
224              self.UpdateUIForStyle(style)
225          return 1

227      def UpdateUIForStyle(self, style):
228          format = style.format
229          sel = 0
230          for c in paletteVGA:
231              if format[4] == c[1]:
232                  #               print "Style", style.name, "is", c[0]
233                  break
234              sel = sel + 1
235          else:
236              sel = -1
237          self.cbo.SetCurSel(sel)
238          self.butIsDefault.SetCheck(style.IsBasedOnDefault())
239          self.GetDlgItem(win32ui.IDC_BUTTON3).EnableWindow(not style.IsBasedOnDefault())
240  
241          self.butIsDefaultBackground.SetCheck(
242              style.background == style.default_background
243          )
244          self.GetDlgItem(win32ui.IDC_BUTTON4).EnableWindow(
245              style.background != style.default_background
246          )
247  
248          bold = format[1] & win32con.CFE_BOLD != 0
249          italic = format[1] & win32con.CFE_ITALIC != 0
250          self.cboBoldItalic.SetCurSel(bold * 2 + italic)
251  
252      def OnStyleUIChanged(self, id, code):
253          if code in [win32con.BN_CLICKED, win32con.CBN_SELCHANGE]:
254              style = self.GetSelectedStyle()
255              self.ApplyUIFormatToStyle(style)
256              self.scintilla.ApplyFormattingStyles(0)
257              return 0
258          return 1
259  
260      def ApplyUIFormatToStyle(self, style):
261          format = style.format
262          color = paletteVGA[self.cbo.GetCurSel()]
263          effect = 0
264          sel = self.cboBoldItalic.GetCurSel()
265          if sel == 0:
266              effect = 0
267          elif sel == 1:
268              effect = win32con.CFE_ITALIC
269          elif sel == 2:
270              effect = win32con.CFE_BOLD
271          else:
272              effect = win32con.CFE_BOLD | win32con.CFE_ITALIC
273          maskFlags = (
274              format[0] | win32con.CFM_COLOR | win32con.CFM_BOLD | win32con.CFM_ITALIC
275          )
276          style.format = (
277              maskFlags,
278              effect,
279              style.format[2],
280              style.format[3],
281              color[1],
282          ) + style.format[5:]

284      def OnOK(self):
285          self.scintilla._GetColorizer().SavePreferences()
