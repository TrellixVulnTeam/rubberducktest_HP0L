7    import ffpyplayer
8    from ffpyplayer.pic import ImageLoader as ffImageLoader, SWScale
9    from ffpyplayer.tools import set_log_callback, get_log_callback

11   from kivy.logger import Logger
12   from kivy.core.image import ImageLoaderBase, ImageData, ImageLoader
13   
14   
15   Logger.info('ImageLoaderFFPy: Using ffpyplayer {}'.format(ffpyplayer.version))
16   
17   
18   logger_func = {'quiet': Logger.critical, 'panic': Logger.critical,
19                  'fatal': Logger.critical, 'error': Logger.error,
20                  'warning': Logger.warning, 'info': Logger.info,
21                  'verbose': Logger.debug, 'debug': Logger.debug}
22   
23   
24   def _log_callback(message, level):
25       message = message.strip()
26       if message:
27           logger_func[level]('ffpyplayer: {}'.format(message))
28   
29   
30   if not get_log_callback():
31       set_log_callback(_log_callback)
32   
33   
34   class ImageLoaderFFPy(ImageLoaderBase):
35       '''Image loader based on the ffpyplayer library. 
36    
37       .. versionadded:: 1.9.0 
38    
39       .. note: 
40           This provider may support more formats than what is listed in 
41           :meth:`extensions`. 
42       '''
43   
44       @staticmethod
45       def extensions():
46           '''Return accepted extensions for this loader'''
47           # See https://www.ffmpeg.org/general.html#Image-Formats
48           return ('bmp', 'dpx', 'exr', 'gif', 'ico', 'jpeg', 'jpg2000', 'jpg',
49                   'jls', 'pam', 'pbm', 'pcx', 'pgm', 'pgmyuv', 'pic', 'png',
50                   'ppm', 'ptx', 'sgi', 'ras', 'tga', 'tiff', 'webp', 'xbm',
51                   'xface', 'xwd')
52   
53       def load(self, filename):
54           try:
55               loader = ffImageLoader(filename)
56           except:
57               Logger.warning('Image: Unable to load image <%s>' % filename)
58               raise
59   
60           # update internals
61           self.filename = filename
62           images = []
63   
64           while True:
65               frame, t = loader.next_frame()
66               if frame is None:
67                   break
68               images.append(frame)
69           if not len(images):
70               raise Exception('No image found in {}'.format(filename))
71   
72           w, h = images[0].get_size()
73           ifmt = images[0].get_pixel_format()
74           if ifmt != 'rgba' and ifmt != 'rgb24':
75               fmt = 'rgba'
76               sws = SWScale(w, h, ifmt, ofmt=fmt)
77               for i, image in enumerate(images):
78                   images[i] = sws.scale(image)
79           else:
80               fmt = ifmt if ifmt == 'rgba' else 'rgb'
81   
82           return [ImageData(w, h, fmt, img.to_memoryview()[0], source_image=img)
83                   for img in images]
84

86   # register
87   ImageLoader.register(ImageLoaderFFPy)
