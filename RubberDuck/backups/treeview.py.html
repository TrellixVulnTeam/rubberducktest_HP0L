1    ''' 
2    Tree View 
3    ========= 
4     
5    .. image:: images/treeview.png 
6        :align: right 
7     
8    .. versionadded:: 1.0.4 
9     
10    
11   :class:`TreeView` is a widget used to represent a tree structure. It is 
12   currently very basic, supporting a minimal feature set. 
13    
14   Introduction 
15   ------------ 
16    
17   A :class:`TreeView` is populated with :class:`TreeViewNode` instances, but you 
18   cannot use a :class:`TreeViewNode` directly. You must combine it with another 
19   widget, such as :class:`~kivy.uix.label.Label`, 
20   :class:`~kivy.uix.button.Button` or even your own widget. The TreeView 
21   always creates a default root node, based on :class:`TreeViewLabel`. 
22    
23   :class:`TreeViewNode` is a class object containing needed properties for 
24   serving as a tree node. Extend :class:`TreeViewNode` to create custom node 
25   types for use with a :class:`TreeView`. 
26    
27   For constructing your own subclass, follow the pattern of TreeViewLabel which 
28   combines a Label and a TreeViewNode, producing a :class:`TreeViewLabel` for 
29   direct use in a TreeView instance. 
30    
31   To use the TreeViewLabel class, you could create two nodes directly attached 
32   to root:: 
33    
34       tv = TreeView() 
35       tv.add_node(TreeViewLabel(text='My first item')) 
36       tv.add_node(TreeViewLabel(text='My second item')) 
37    
38   Or, create two nodes attached to a first:: 
39    
40       tv = TreeView() 
41       n1 = tv.add_node(TreeViewLabel(text='Item 1')) 
42       tv.add_node(TreeViewLabel(text='SubItem 1'), n1) 
43       tv.add_node(TreeViewLabel(text='SubItem 2'), n1) 
44    
45   If you have a large tree structure, perhaps you would need a utility function 
46   to populate the tree view:: 
47    
48       def populate_tree_view(tree_view, parent, node): 
49           if parent is None: 
50               tree_node = tree_view.add_node(TreeViewLabel(text=node['node_id'], 
51                                                            is_open=True)) 
52           else: 
53               tree_node = tree_view.add_node(TreeViewLabel(text=node['node_id'], 
54                                                            is_open=True), parent) 
55    
56           for child_node in node['children']: 
57               populate_tree_view(tree_view, tree_node, child_node) 
58    
59    
60       tree = {'node_id': '1', 
61               'children': [{'node_id': '1.1', 
62                             'children': [{'node_id': '1.1.1', 
63                                           'children': [{'node_id': '1.1.1.1', 
64                                                         'children': []}]}, 
65                                          {'node_id': '1.1.2', 
66                                           'children': []}, 
67                                          {'node_id': '1.1.3', 
68                                           'children': []}]}, 
69                             {'node_id': '1.2', 
70                              'children': []}]} 
71    
72    
73       class TreeWidget(FloatLayout): 
74           def __init__(self, **kwargs): 
75               super(TreeWidget, self).__init__(**kwargs) 
76    
77               tv = TreeView(root_options=dict(text='Tree One'), 
78                             hide_root=False, 
79                             indent_level=4) 
80    
81               populate_tree_view(tv, None, tree) 
82    
83               self.add_widget(tv) 
84    
85   The root widget in the tree view is opened by default and has text set as 
86   'Root'. If you want to change that, you can use the 
87   :attr:`TreeView.root_options` 
88   property. This will pass options to the root widget:: 
89    
90       tv = TreeView(root_options=dict(text='My root label')) 
91    
92    
93   Creating Your Own Node Widget 
94   ----------------------------- 
95    
96   For a button node type, combine a :class:`~kivy.uix.button.Button` and a 
97   :class:`TreeViewNode` as follows:: 
98    
99       class TreeViewButton(Button, TreeViewNode): 
100          pass 
101   
102  You must know that, for a given node, only the 
103  :attr:`~kivy.uix.widget.Widget.size_hint_x` will be honored. The allocated 
104  width for the node will depend of the current width of the TreeView and the 
105  level of the node. For example, if a node is at level 4, the width 
106  allocated will be: 
107   
108      treeview.width - treeview.indent_start - treeview.indent_level * node.level 
109   
110  You might have some trouble with that. It is the developer's responsibility to 
111  correctly handle adapting the graphical representation nodes, if needed. 
112  '''

268  class TreeView(Widget):
269      '''TreeView class. See module documentation for more information. 
270   
271      :Events: 
272          `on_node_expand`: (node, ) 
273              Fired when a node is being expanded 
274          `on_node_collapse`: (node, ) 
275              Fired when a node is being collapsed 
276      '''
277  
278      __events__ = ('on_node_expand', 'on_node_collapse')
279  
280      def __init__(self, **kwargs):
281          self._trigger_layout = Clock.create_trigger(self._do_layout, -1)
282          super(TreeView, self).__init__(**kwargs)
283          tvlabel = TreeViewLabel(text='Root', is_open=True, level=0)
284          for key, value in self.root_options.items():
285              setattr(tvlabel, key, value)
286          self._root = self.add_node(tvlabel, None)
287  
288          trigger = self._trigger_layout
289          fbind = self.fbind
290          fbind('pos', trigger)
291          fbind('size', trigger)
292          fbind('indent_level', trigger)
293          fbind('indent_start', trigger)
294          trigger()
295  
296      def add_node(self, node, parent=None):
297          '''Add a new node to the tree. 
298   
299          :Parameters: 
300              `node`: instance of a :class:`TreeViewNode` 
301                  Node to add into the tree 
302              `parent`: instance of a :class:`TreeViewNode`, defaults to None 
303                  Parent node to attach the new node. If `None`, it is added to 
304                  the :attr:`root` node. 
305   
306          :returns: 
307              the node `node`. 
308          '''
309          # check if the widget is "ok" for a node
310          if not isinstance(node, TreeViewNode):
311              raise TreeViewException(
312                  'The node must be a subclass of TreeViewNode')
313          # create node
314          if parent is None and self._root:
315              parent = self._root
316          if parent:
317              parent.is_leaf = False
318              parent.nodes.append(node)
319              node.parent_node = parent
320              node.level = parent.level + 1
321          node.fbind('size', self._trigger_layout)
322          self._trigger_layout()
323          return node
324  
325      def remove_node(self, node):
326          '''Removes a node from the tree. 
327   
328          .. versionadded:: 1.0.7 
329   
330          :Parameters: 
331              `node`: instance of a :class:`TreeViewNode` 
332                  Node to remove from the tree. If `node` is :attr:`root`, it is 
333                  not removed. 
334          '''
335          # check if the widget is "ok" for a node
336          if not isinstance(node, TreeViewNode):
337              raise TreeViewException(
338                  'The node must be a subclass of TreeViewNode')
339          parent = node.parent_node
340          if parent is not None:
341              if node == self._selected_node:
342                  node.is_selected = False
343                  self._selected_node = None
344              nodes = parent.nodes
345              if node in nodes:
346                  nodes.remove(node)
347              parent.is_leaf = not bool(len(nodes))
348              node.parent_node = None
349              node.funbind('size', self._trigger_layout)
350              self._trigger_layout()
351  
352      def on_node_expand(self, node):
353          pass
354  
355      def on_node_collapse(self, node):
356          pass
357  
358      def select_node(self, node):
359          '''Select a node in the tree. 
360          '''
361          if node.no_selection:
362              return
363          if self._selected_node:
364              self._selected_node.is_selected = False
365          node.is_selected = True
366          self._selected_node = node
367  
368      def deselect_node(self, *args):
369          '''Deselect any selected node. 
370   
371          .. versionadded:: 1.10.0 
372          '''
373          if self._selected_node:
374              self._selected_node.is_selected = False
375              self._selected_node = None
376  
377      def toggle_node(self, node):
378          '''Toggle the state of the node (open/collapsed). 
379          '''
380          node.is_open = not node.is_open
381          if node.is_open:
382              if self.load_func and not node.is_loaded:
383                  self._do_node_load(node)
384              self.dispatch('on_node_expand', node)
385          else:
386              self.dispatch('on_node_collapse', node)
387          self._trigger_layout()
388  
389      def get_node_at_pos(self, pos):
390          '''Get the node at the position (x, y). 
391          '''
392          x, y = pos
393          for node in self.iterate_open_nodes(self.root):
394              if self.x <= x <= self.right and \
395                 node.y <= y <= node.top:
396                  return node
397  
398      def iterate_open_nodes(self, node=None):
399          '''Generator to iterate over all the expended nodes starting from 
400          `node` and down. If `node` is `None`, the generator start with 
401          :attr:`root`. 
402   
403          To get all the open nodes:: 
404   
405              treeview = TreeView() 
406              # ... add nodes ... 
407              for node in treeview.iterate_open_nodes(): 
408                  print(node) 
409   
410          '''
411          if not node:
412              node = self.root
413          if self.hide_root and node is self.root:
414              pass
415          else:
416              yield node
417          if not node.is_open:
418              return
419          f = self.iterate_open_nodes
420          for cnode in node.nodes:
421              for ynode in f(cnode):
422                  yield ynode
423  
424      def iterate_all_nodes(self, node=None):
425          '''Generator to iterate over all nodes from `node` and down whether 
426          expanded or not. If `node` is `None`, the generator start with 
427          :attr:`root`. 
428          '''
429          if not node:
430              node = self.root
431          yield node
432          f = self.iterate_all_nodes
433          for cnode in node.nodes:
434              for ynode in f(cnode):
435                  yield ynode

437      #
438      # Private
439      #
440      def on_load_func(self, instance, value):
441          if value:
442              Clock.schedule_once(self._do_initial_load)
443  
444      def _do_initial_load(self, *largs):
445          if not self.load_func:
446              return
447          self._do_node_load(None)
448  
449      def _do_node_load(self, node):
450          gen = self.load_func(self, node)
451          if node:
452              node.is_loaded = True
453          if not gen:
454              return
455          for cnode in gen:
456              self.add_node(cnode, node)
457  
458      def on_root_options(self, instance, value):
459          if not self.root:
460              return
461          for key, value in value.items():
462              setattr(self.root, key, value)
463  
464      def _do_layout(self, *largs):
465          self.clear_widgets()
466          # display only the one who are is_open
467          self._do_open_node(self.root)
468          # now do layout
469          self._do_layout_node(self.root, 0, self.top)
470          # now iterate for calculating minimum size
471          min_width = min_height = 0
472          for count, node in enumerate(self.iterate_open_nodes(self.root)):
473              node.odd = False if count % 2 else True
474              min_width = max(min_width, node.right - self.x)
475              min_height += node.height
476          self.minimum_size = (min_width, min_height)
477  
478      def _do_open_node(self, node):
479          if self.hide_root and node is self.root:
480              height = 0
481          else:
482              self.add_widget(node)
483              height = node.height
484              if not node.is_open:
485                  return height
486          for cnode in node.nodes:
487              height += self._do_open_node(cnode)
488          return height
489  
490      def _do_layout_node(self, node, level, y):
491          if self.hide_root and node is self.root:
492              level -= 1
493          else:
494              node.x = self.x + self.indent_start + level * self.indent_level
495              node.top = y
496              if node.size_hint_x:
497                  node.width = (self.width - (node.x - self.x)) \
498                      * node.size_hint_x
499              y -= node.height
500              if not node.is_open:
501                  return y
502          for cnode in node.nodes:
503              y = self._do_layout_node(cnode, level + 1, y)
504          return y
505  
506      def on_touch_down(self, touch):
507          node = self.get_node_at_pos(touch.pos)
508          if not node:
509              return
510          if node.disabled:
511              return
512          # toggle node or selection ?
513          if node.x - self.indent_start <= touch.x < node.x:
514              self.toggle_node(node)
515          elif node.x <= touch.x:
516              self.select_node(node)
517              node.dispatch('on_touch_down', touch)
518          return True
