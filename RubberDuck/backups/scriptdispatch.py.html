22   def _is_callable(obj):
23       return type(obj) in [types.FunctionType, types.MethodType]
24       # ignore hasattr(obj, "__call__") as this means all COM objects!
25   
26   
27   class ScriptDispatch:
28       _public_methods_ = []
29   
30       def __init__(self, engine, scriptNamespace):
31           self.engine = engine
32           self.scriptNamespace = scriptNamespace
33   
34       def _dynamic_(self, name, lcid, wFlags, args):
35           # Ensure any newly added items are available.
36           self.engine.RegisterNewNamedItems()
37           self.engine.ProcessNewNamedItemsConnections()
38           if wFlags & pythoncom.INVOKE_FUNC:
39               # attempt to call a function
40               try:
41                   func = getattr(self.scriptNamespace, name)
42                   if not _is_callable(func):
43                       raise AttributeError(name)  # Not a function.
44                   realArgs = []
45                   for arg in args:
46                       if type(arg) == PyIDispatchType:
47                           realArgs.append(Dispatch(arg))
48                       else:
49                           realArgs.append(arg)
50                   # xxx - todo - work out what code block to pass???
51                   return self.engine.ApplyInScriptedSection(None, func, tuple(realArgs))
52   
53               except AttributeError:
54                   if not wFlags & pythoncom.DISPATCH_PROPERTYGET:
55                       raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
56           if wFlags & pythoncom.DISPATCH_PROPERTYGET:
57               # attempt to get a property
58               try:
59                   ret = getattr(self.scriptNamespace, name)
60                   if _is_callable(ret):
61                       raise AttributeError(name)  # Not a property.
62               except AttributeError:
63                   raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
64               except COMException as instance:
65                   raise
66               except:
67                   ret = self.engine.HandleException()
68               return ret
69   
70           raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)

73   class StrictDynamicPolicy(win32com.server.policy.DynamicPolicy):
74       def _wrap_(self, object):
75           win32com.server.policy.DynamicPolicy._wrap_(self, object)
76           if hasattr(self._obj_, "scriptNamespace"):
77               for name in dir(self._obj_.scriptNamespace):
78                   self._dyn_dispid_to_name_[self._getdispid_(name, 0)] = name
79   
80       def _getmembername_(self, dispid):
81           try:
82               return str(self._dyn_dispid_to_name_[dispid])
83           except KeyError:
84               raise COMException(scode=winerror.DISP_E_UNKNOWNNAME, desc="Name not found")
85   
86       def _getdispid_(self, name, fdex):
87           try:
88               func = getattr(self._obj_.scriptNamespace, str(name))
89           except AttributeError:
90               raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
91           #       if not _is_callable(func):
92           return win32com.server.policy.DynamicPolicy._getdispid_(self, name, fdex)

95   def _wrap_debug(obj):
96       return win32com.server.util.wrap(
97           obj,
98           usePolicy=StrictDynamicPolicy,
99           useDispatcher=win32com.server.policy.DispatcherWin32trace,
100      )

103  def _wrap_nodebug(obj):
104      return win32com.server.util.wrap(obj, usePolicy=StrictDynamicPolicy)
