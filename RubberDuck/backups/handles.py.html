17               import win32event
18   
19               h = win32event.CreateEvent(None, 0, 0, None)
20               if invalidate:
21                   win32api.CloseHandle(int(h))
22               1 / 0
23               # If we invalidated, then the object destruction code will attempt
24               # to close an invalid handle.  We don't wan't an exception in
25               # this case

27           def f2(invalidate):
28               """This function should throw an IOError."""

30                   f1(invalidate)

32                   raise IOError("raise 2")

138      def testLong(self):
139          # sys.maxint+1 should always be a 'valid' handle, treated as an
140          # unsigned int, even though it is a long. Although pywin32 should not
141          # directly create such longs, using struct.unpack() with a P format
142          # may well return them. eg:
143          # >>> struct.unpack("P", struct.pack("P", -1))
144          # (4294967295L,)
145          try:
146              big = sys.maxsize
147          except AttributeError:
148              big = sys.maxint
149          pywintypes.HANDLE(big + 1)
