260  def annotate(pre, post, callable, words):
261      state = False
262      i = random.randint(0, 4)
263      while i < len(words):
264          if ' ' in words[i] or '\n' in words[i]:  # skip spaces
265              i += 1
266              continue
267          if not state:
268              words[i] = pre.format(callable(), words[i])
269          else:
270              words[i] = post.format(words[i])
271          state = not state
272          i += random.randint(1, 7)
273  
274  
275  annotate('[size={0}]{1}', '{0}[/size]', partial(random.randint, 8, 24), words)
276  annotate('[b]{1}', '{0}[/b]', str, words)
277  annotate('[i]{1}', '{0}[/i]', str, words)
278  annotate('[color={0}]{1}', '{0}[/color]',
279           lambda: get_hex_from_color(get_random_color()), words)
280  annotated_text = ''.join(words)
