23   class TestDataObject:
24       _com_interfaces_ = [pythoncom.IID_IDataObject]
25       _public_methods_ = IDataObject_Methods
26   
27       def __init__(self, bytesval):
28           global num_do_objects
29           num_do_objects += 1
30           self.bytesval = bytesval
31           self.supported_fe = []
32           for cf in (win32con.CF_TEXT, win32con.CF_UNICODETEXT):
33               fe = cf, None, pythoncom.DVASPECT_CONTENT, -1, pythoncom.TYMED_HGLOBAL
34               self.supported_fe.append(fe)
35   
36       def __del__(self):
37           global num_do_objects
38           num_do_objects -= 1
39   
40       def _query_interface_(self, iid):
41           if iid == pythoncom.IID_IEnumFORMATETC:
42               return NewEnum(self.supported_fe, iid=iid)
43   
44       def GetData(self, fe):
45           ret_stg = None
46           cf, target, aspect, index, tymed = fe
47           if aspect & pythoncom.DVASPECT_CONTENT and tymed == pythoncom.TYMED_HGLOBAL:
48               if cf == win32con.CF_TEXT:
49                   ret_stg = pythoncom.STGMEDIUM()
50                   ret_stg.set(pythoncom.TYMED_HGLOBAL, self.bytesval)
51               elif cf == win32con.CF_UNICODETEXT:
52                   ret_stg = pythoncom.STGMEDIUM()
53                   ret_stg.set(pythoncom.TYMED_HGLOBAL, self.bytesval.decode("latin1"))
54   
55           if ret_stg is None:
56               raise COMException(hresult=winerror.E_NOTIMPL)
57           return ret_stg
58   
59       def GetDataHere(self, fe):
60           raise COMException(hresult=winerror.E_NOTIMPL)
61   
62       def QueryGetData(self, fe):
63           cf, target, aspect, index, tymed = fe
64           if aspect & pythoncom.DVASPECT_CONTENT == 0:
65               raise COMException(hresult=winerror.DV_E_DVASPECT)
66           if tymed != pythoncom.TYMED_HGLOBAL:
67               raise COMException(hresult=winerror.DV_E_TYMED)
68           return None  # should check better

70       def GetCanonicalFormatEtc(self, fe):
71           RaiseCOMException(winerror.DATA_S_SAMEFORMATETC)
72           # return fe

74       def SetData(self, fe, medium):
75           raise COMException(hresult=winerror.E_NOTIMPL)

77       def EnumFormatEtc(self, direction):
78           if direction != pythoncom.DATADIR_GET:
79               raise COMException(hresult=winerror.E_NOTIMPL)
80           return NewEnum(self.supported_fe, iid=pythoncom.IID_IEnumFORMATETC)

103      def testIsCurrentClipboard(self):
104          do = TestDataObject(b"Hello from Python")

106          pythoncom.OleSetClipboard(do)

109      def testComToWin32(self):
110          # Set the data via our DataObject
111          do = TestDataObject(b"Hello from Python")
112          do = WrapCOMObject(do, iid=pythoncom.IID_IDataObject)
113          pythoncom.OleSetClipboard(do)
114          # Then get it back via the standard win32 clipboard functions.
115          win32clipboard.OpenClipboard()
116          got = win32clipboard.GetClipboardData(win32con.CF_TEXT)
117          # CF_TEXT gives bytes.
118          expected = b"Hello from Python"
119          self.assertEqual(got, expected)
120          # Now check unicode
121          got = win32clipboard.GetClipboardData(win32con.CF_UNICODETEXT)
122          self.assertEqual(got, "Hello from Python")
123          win32clipboard.CloseClipboard()
124  
125      def testWin32ToCom(self):
126          # Set the data via the std win32 clipboard functions.
127          val = b"Hello again!"  # always bytes
128          win32clipboard.OpenClipboard()
129          win32clipboard.SetClipboardData(win32con.CF_TEXT, val)
130          win32clipboard.CloseClipboard()
131          # and get it via an IDataObject provided by COM
132          do = pythoncom.OleGetClipboard()
133          cf = (
134              win32con.CF_TEXT,
135              None,
136              pythoncom.DVASPECT_CONTENT,
137              -1,
138              pythoncom.TYMED_HGLOBAL,
139          )
140          stg = do.GetData(cf)
141          got = stg.data
142          # The data we get back has the \0, as our STGMEDIUM has no way of
143          # knowing if it meant to be a string, or a binary buffer, so
144          # it must return it too.
145          self.assertTrue(got, b"Hello again!\0")
