weakmethod.py 
1    """ 
2    Weak Method 
3    =========== 
4     
5    The :class:`WeakMethod` is used by the :class:`~kivy.clock.Clock` class to 
6    allow references to a bound method that permits the associated object to 
7    be garbage collected. Please refer to 
8    `examples/core/clock_method.py` for more information. 
9     
10   This WeakMethod class is taken from the recipe 
11   http://code.activestate.com/recipes/81253/, based on the nicodemus version. 
12   Many thanks nicodemus! 
13   """
14   
15   import weakref
16   
17   
18   class WeakMethod:
19       '''Implementation of a 
20       `weakref <http://en.wikipedia.org/wiki/Weak_reference>`_ 
21       for functions and bound methods. 
22       '''
23       def __init__(self, method):
24           self.method = None
25           self.method_name = None
26           try:
27               if method.__self__ is not None:
28                   self.method_name = method.__func__.__name__
29                   self.proxy = weakref.proxy(method.__self__)
30               else:
31                   self.method = method
32                   self.proxy = None
33           except AttributeError:
34               self.method = method
35               self.proxy = None
36   
37       def __call__(self):
38           '''Return a new bound-method like the original, or the 
39           original function if it was just a function or unbound 
40           method. 
41           Returns None if the original object doesn't exist. 
42           '''
43           if self.proxy is not None:
44               try:
45                   return getattr(self.proxy, self.method_name)
46               except ReferenceError:
47                   return None
48   
49           return self.method
50   
51       def is_dead(self):
52           '''Returns True if the referenced callable was a bound method and 
53           the instance no longer exists. Otherwise, return False. 
54           '''
55           if self.proxy is None:
56               return False
57   
58           try:
59               getattr(self.proxy, self.method_name)
60               return False
61           except ReferenceError:
62               return True
63   
64       def __eq__(self, other):
65           if type(self) is not type(other):
66               return False
67           s = self()
68           return s is not None and s == other()
69   
70       def __repr__(self):
71           return '<WeakMethod proxy={} method={} method_name={}>'.format(
72                  self.proxy, self.method, self.method_name)
