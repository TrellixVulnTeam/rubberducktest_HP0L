15   class DDESystemTopic(object.Object):
16       def __init__(self, app):
17           self.app = app
18           object.Object.__init__(self, CreateServerSystemTopic())
19   
20       def Exec(self, data):
21           try:
22               #           print "Executing", cmd
23               self.app.OnDDECommand(data)
24           except:
25               t, v, tb = sys.exc_info()
26               # The DDE Execution failed.
27               print("Error executing DDE command.")
28               traceback.print_exception(t, v, tb)
29               return 0

32   class DDEServer(object.Object):
33       def __init__(self, app):
34           self.app = app
35           object.Object.__init__(self, CreateServer())
36           self.topic = self.item = None
37   
38       def CreateSystemTopic(self):
39           return DDESystemTopic(self.app)
40   
41       def Shutdown(self):
42           self._obj_.Shutdown()
43           self._obj_.Destroy()
44           if self.topic is not None:
45               self.topic.Destroy()
46               self.topic = None
47           if self.item is not None:
48               self.item.Destroy()
49               self.item = None
50   
51       def OnCreate(self):
52           return 1
53   
54       def Status(self, msg):
55           try:
56               win32ui.SetStatusText(msg)
57           except win32ui.error:
58               pass
