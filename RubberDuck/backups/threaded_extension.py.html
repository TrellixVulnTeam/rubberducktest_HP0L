21   ISAPI_REQUEST = 1
22   ISAPI_SHUTDOWN = 2

25   class WorkerThread(threading.Thread):
26       def __init__(self, extension, io_req_port):
27           self.running = False
28           self.io_req_port = io_req_port
29           self.extension = extension
30           threading.Thread.__init__(self)
31           # We wait 15 seconds for a thread to terminate, but if it fails to,
32           # we don't want the process to hang at exit waiting for it...
33           self.setDaemon(True)
34   
35       def run(self):
36           self.running = True
37           while self.running:
38               errCode, bytes, key, overlapped = GetQueuedCompletionStatus(
39                   self.io_req_port, INFINITE
40               )
41               if key == ISAPI_SHUTDOWN and overlapped is None:
42                   break
43   
44               # Let the parent extension handle the command.
45               dispatcher = self.extension.dispatch_map.get(key)
46               if dispatcher is None:
47                   raise RuntimeError("Bad request '%s'" % (key,))
48   
49               dispatcher(errCode, bytes, key, overlapped)
50   
51       def call_handler(self, cblock):
52           self.extension.Dispatch(cblock)

55   # A generic thread-pool based extension, using IO Completion Ports.
56   # Sub-classes can override one method to implement a simple extension, or
57   # may leverage the CompletionPort to queue their own requests, and implement a
58   # fully asynch extension.
59   class ThreadPoolExtension(isapi.simple.SimpleExtension):
60       "Base class for an ISAPI extension based around a thread-pool"
61       max_workers = 20
62       worker_shutdown_wait = 15000  # 15 seconds for workers to quit...
63   
64       def __init__(self):
65           self.workers = []
66           # extensible dispatch map, for sub-classes that need to post their
67           # own requests to the completion port.
68           # Each of these functions is called with the result of
69           # GetQueuedCompletionStatus for our port.
70           self.dispatch_map = {
71               ISAPI_REQUEST: self.DispatchConnection,
72           }
73   
74       def GetExtensionVersion(self, vi):
75           isapi.simple.SimpleExtension.GetExtensionVersion(self, vi)
76           # As per Q192800, the CompletionPort should be created with the number
77           # of processors, even if the number of worker threads is much larger.
78           # Passing 0 means the system picks the number.
79           self.io_req_port = CreateIoCompletionPort(-1, None, 0, 0)
80           # start up the workers
81           self.workers = []
82           for i in range(self.max_workers):
83               worker = WorkerThread(self, self.io_req_port)
84               worker.start()
85               self.workers.append(worker)
86   
87       def HttpExtensionProc(self, control_block):
88           overlapped = OVERLAPPED()
89           overlapped.object = control_block
90           PostQueuedCompletionStatus(self.io_req_port, 0, ISAPI_REQUEST, overlapped)
91           return isapicon.HSE_STATUS_PENDING
92   
93       def TerminateExtension(self, status):
94           for worker in self.workers:
95               worker.running = False
96           for worker in self.workers:
97               PostQueuedCompletionStatus(self.io_req_port, 0, ISAPI_SHUTDOWN, None)
98           # wait for them to terminate - pity we aren't using 'native' threads
99           # as then we could do a smart wait - but now we need to poll....
100          end_time = time.time() + self.worker_shutdown_wait / 1000
101          alive = self.workers
102          while alive:
103              if time.time() > end_time:
104                  # xxx - might be nice to log something here.
105                  break
106              time.sleep(0.2)
107              alive = [w for w in alive if w.is_alive()]
108          self.dispatch_map = {}  # break circles
109          CloseHandle(self.io_req_port)

111      # This is the one operation the base class supports - a simple
112      # Connection request.  We setup the thread-token, and dispatch to the
113      # sub-class's 'Dispatch' method.
114      def DispatchConnection(self, errCode, bytes, key, overlapped):
115          control_block = overlapped.object
116          # setup the correct user for this request
117          hRequestToken = control_block.GetImpersonationToken()
118          SetThreadToken(None, hRequestToken)
119          try:
120              try:
121                  self.Dispatch(control_block)
122              except:
123                  self.HandleDispatchError(control_block)
124          finally:
125              # reset the security context
126              SetThreadToken(None, None)
127  
128      def Dispatch(self, ecb):
129          """Overridden by the sub-class to handle connection requests. 
130   
131          This class creates a thread-pool using a Windows completion port, 
132          and dispatches requests via this port.  Sub-classes can generally 
133          implement each connection request using blocking reads and writes, and 
134          the thread-pool will still provide decent response to the end user. 
135   
136          The sub-class can set a max_workers attribute (default is 20).  Note 
137          that this generally does *not* mean 20 threads will all be concurrently 
138          running, via the magic of Windows completion ports. 
139   
140          There is no default implementation - sub-classes must implement this. 
141          """
142          raise NotImplementedError("sub-classes should override Dispatch")
143  
144      def HandleDispatchError(self, ecb):
145          """Handles errors in the Dispatch method. 
146   
147          When a Dispatch method call fails, this method is called to handle 
148          the exception.  The default implementation formats the traceback 
149          in the browser. 
150          """
151          ecb.HttpStatusCode = isapicon.HSE_STATUS_ERROR
152          # control_block.LogData = "we failed!"
153          exc_typ, exc_val, exc_tb = sys.exc_info()
154          limit = None
155          try:
156              try:
157                  import cgi
158  
159                  ecb.SendResponseHeaders(
160                      "200 OK", "Content-type: text/html\r\n\r\n", False
161                  )
162                  print(file=ecb)
163                  print("<H3>Traceback (most recent call last):</H3>", file=ecb)
164                  list = traceback.format_tb(
165                      exc_tb, limit
166                  ) + traceback.format_exception_only(exc_typ, exc_val)
167                  print(
168                      "<PRE>%s<B>%s</B></PRE>"
169                      % (
170                          cgi.escape("".join(list[:-1])),
171                          cgi.escape(list[-1]),
172                      ),
173                      file=ecb,
174                  )
175              except ExtensionError:
176                  # The client disconnected without reading the error body -
177                  # its probably not a real browser at the other end, ignore it.
178                  pass
179              except:
180                  print("FAILED to render the error message!")
181                  traceback.print_exc()
182                  print("ORIGINAL extension error:")
183                  traceback.print_exception(exc_typ, exc_val, exc_tb)
184          finally:
185              # holding tracebacks in a local of a frame that may itself be
186              # part of a traceback used to be evil and cause leaks!
187              exc_tb = None
188              ecb.DoneWithSession()
