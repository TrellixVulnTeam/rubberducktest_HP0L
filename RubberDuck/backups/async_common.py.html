44   class WidgetResolver(object):
45       """It assumes that the widget tree strictly forms a DAG. 
46       """
47   
48       base_widget = None
49   
50       matched_widget = None
51   
52       _kwargs_filter = {}
53   
54       _funcs_filter = []
55   
56       def __init__(self, base_widget, **kwargs):
57           self.base_widget = base_widget
58           self._kwargs_filter = {}
59           self._funcs_filter = []
60           super(WidgetResolver, self).__init__(**kwargs)
61   
62       def __call__(self):
63           if self.matched_widget is not None:
64               return self.matched_widget
65   
66           if not self._kwargs_filter and not self._funcs_filter:
67               return self.base_widget
68           return None
69   
70       def match(self, **kwargs_filter):
71           self._kwargs_filter.update(kwargs_filter)
72   
73       def match_funcs(self, funcs_filter=()):
74           self._funcs_filter.extend(funcs_filter)
75   
76       def check_widget(self, widget):
77           if not all(func(widget) for func in self._funcs_filter):
78               return False
79   
80           for attr, val in self._kwargs_filter.items():
81               if getattr(widget, attr, _unique_value) != val:
82                   return False
83   
84           return True
85   
86       def not_found(self, op):
87           raise ValueError(
88               'Cannot find widget matching <{}, {}> starting from base '
89               'widget "{}" doing "{}" traversal'.format(
90                   self._kwargs_filter, self._funcs_filter, self.base_widget, op))
91   
92       def down(self, **kwargs_filter):
93           self.match(**kwargs_filter)
94           check = self.check_widget
95   
96           fifo = deque([self.base_widget])
97           while fifo:
98               widget = fifo.popleft()
99               if check(widget):
100                  return WidgetResolver(base_widget=widget)
101  
102              fifo.extend(widget.children)
103  
104          self.not_found('down')
105  
106      def up(self, **kwargs_filter):
107          self.match(**kwargs_filter)
108          check = self.check_widget
109  
110          parent = self.base_widget
111          while parent is not None:
112              if check(parent):
113                  return WidgetResolver(base_widget=parent)
114  
115              new_parent = parent.parent
116              # Window is its own parent oO
117              if new_parent is parent:
118                  break
119              parent = new_parent
120  
121          self.not_found('up')
122  
123      def family_up(self, **kwargs_filter):
124          self.match(**kwargs_filter)
125          check = self.check_widget
126  
127          base_widget = self.base_widget
128          already_checked_base = None
129          while base_widget is not None:
130              fifo = deque([base_widget])
131              while fifo:
132                  widget = fifo.popleft()
133                  # don't check the child we checked before moving up
134                  if widget is already_checked_base:
135                      continue
136  
137                  if check(widget):
138                      return WidgetResolver(base_widget=widget)
139  
140                  fifo.extend(widget.children)
141  
142              already_checked_base = base_widget
143              new_base_widget = base_widget.parent
144              # Window is its own parent oO
145              if new_base_widget is base_widget:
146                  break
147              base_widget = new_base_widget
148  
149          self.not_found('family_up')

195      async def wait_clock_frames(self, n, sleep_time=1 / 60.):
196          from kivy.clock import Clock
197          frames_start = Clock.frames
198          while Clock.frames < frames_start + n:
199              await self.async_sleep(sleep_time)

201      def get_widget_pos_pixel(self, widget, positions):
202          from kivy.graphics import Fbo, ClearColor, ClearBuffers
203  
204          canvas_parent_index = -2
205          if widget.parent is not None:
206              canvas_parent_index = widget.parent.canvas.indexof(widget.canvas)
207              if canvas_parent_index > -1:
208                  widget.parent.canvas.remove(widget.canvas)
209  
210          w, h = int(widget.width), int(widget.height)
211          fbo = Fbo(size=(w, h), with_stencilbuffer=True)
212  
213          with fbo:
214              ClearColor(0, 0, 0, 0)
215              ClearBuffers()
216  
217          fbo.add(widget.canvas)
218          fbo.draw()
219          pixels = fbo.pixels
220          fbo.remove(widget.canvas)
221  
222          if widget.parent is not None and canvas_parent_index > -1:
223              widget.parent.canvas.insert(canvas_parent_index, widget.canvas)
224  
225          values = []
226          for x, y in positions:
227              x = int(x)
228              y = int(y)
229              i = y * w * 4 + x * 4
230              values.append(tuple(pixels[i:i + 4]))
231  
232          return values

234      async def do_touch_down_up(
235              self, pos=None, widget=None, duration=.2, pos_jitter=None,
236              widget_jitter=False, jitter_dt=1 / 15., end_on_pos=False):
237          if widget is None:
238              x, y = pos
239          else:
240              if pos is None:
241                  x, y = widget.to_window(*widget.center)
242              else:
243                  x, y = widget.to_window(*pos, initial=False)
244          touch = AsyncUnitTestTouch(x, y)
245  
246          ts = time.perf_counter()
247          touch.touch_down()
248          await self.wait_clock_frames(1)
249          yield 'down', touch.pos
250  
251          if not pos_jitter and not widget_jitter:
252              await self.async_sleep(duration)
253              touch.touch_up()
254              await self.wait_clock_frames(1)
255              yield 'up', touch.pos
256  
257              return
258  
259          moved = False
260          if pos_jitter:
261              dx, dy = pos_jitter
262          else:
263              dx = widget.width / 2.
264              dy = widget.height / 2.
265  
266          while time.perf_counter() - ts < duration:
267              moved = True
268              await self.async_sleep(jitter_dt)
269  
270              touch.touch_move(
271                  x + (random.random() * 2 - 1) * dx,
272                  y + (random.random() * 2 - 1) * dy
273              )
274              await self.wait_clock_frames(1)
275              yield 'move', touch.pos
276  
277          if end_on_pos and moved:
278              touch.touch_move(x, y)
279              await self.wait_clock_frames(1)
280              yield 'move', touch.pos
281  
282          touch.touch_up()
283          await self.wait_clock_frames(1)
284          yield 'up', touch.pos

286      async def do_touch_drag(
287              self, pos=None, widget=None,
288              widget_loc=('center_x', 'center_y'), dx=0, dy=0,
289              target_pos=None, target_widget=None, target_widget_offset=(0, 0),
290              target_widget_loc=('center_x', 'center_y'), long_press=0,
291              duration=.2, drag_n=5):
292          """Initiates a touch down, followed by some dragging to a target 
293          location, ending with a touch up. 
294   
295          `origin`: These parameters specify where the drag starts. 
296          - If ``widget`` is None, it starts at ``pos`` (in window coordinates). 
297            If ``dx``/``dy`` is used, it is in the window coordinate system also. 
298          - If ``pos`` is None, it starts on the ``widget`` as specified by 
299            ``widget_loc``. If ``dx``/``dy`` is used, it is in the ``widget`` 
300            coordinate system. 
301          - If neither is None, it starts at ``pos``, but in the ``widget``'s 
302            coordinate system (:meth:`~kivy.uix.widget.Widget.to_window` is used 
303            on it). If ``dx``/``dy`` is used, it is in the ``widget`` 
304            coordinate system. 
305   
306          `target`: These parameters specify where the drag ends. 
307          - If ``target_pos`` and ``target_widget`` is None, then ``dx`` and 
308            ``dy`` is used relative to the position where the drag started. 
309          - If ``target_widget`` is None, it ends at ``target_pos`` 
310            (in window coordinates). 
311          - If ``target_pos`` is None, it ends on the ``target_widget`` as 
312            specified by ``target_widget_loc``. ``target_widget_offset``, is an 
313            additional ``(x, y)`` offset relative to ``target_widget_loc``. 
314          - If neither is None, it starts at ``target_pos``, but in the 
315            ``target_widget``'s coordinate system 
316            (:meth:`~kivy.uix.widget.Widget.to_window` is used on it). 
317   
318          When ``widget`` and/or ``target_widget`` are specified, ``widget_loc`` 
319          and ``target_widget_loc``, respectively, indicate where on the widget 
320          the drag starts/ends. It is a a tuple with property names of the widget 
321          to loop up to get the value. The default is 
322          ``('center_x', 'center_y')`` so the drag would start/end in the 
323          widget's center. 
324          """
325          if widget is None:
326              x, y = pos
327              tx, ty = x + dx, y + dy
328          else:
329              if pos is None:
330                  w_x = getattr(widget, widget_loc[0])
331                  w_y = getattr(widget, widget_loc[1])
332                  x, y = widget.to_window(w_x, w_y)
333                  tx, ty = widget.to_window(w_x + dx, w_y + dy)
334              else:
335                  x, y = widget.to_window(*pos, initial=False)
336                  tx, ty = widget.to_window(
337                      pos[0] + dx, pos[1] + dy, initial=False)
338  
339          if target_pos is not None:
340              if target_widget is None:
341                  tx, ty = target_pos
342              else:
343                  tx, ty = target_pos = target_widget.to_window(
344                      *target_pos, initial=False)
345          elif target_widget is not None:
346              x_off, y_off = target_widget_offset
347              w_x = getattr(target_widget, target_widget_loc[0]) + x_off
348              w_y = getattr(target_widget, target_widget_loc[1]) + y_off
349              tx, ty = target_pos = target_widget.to_window(w_x, w_y)
350          else:
351              target_pos = tx, ty
352  
353          touch = AsyncUnitTestTouch(x, y)
354  
355          touch.touch_down()
356          await self.wait_clock_frames(1)
357          if long_press:
358              await self.async_sleep(long_press)
359          yield 'down', touch.pos
360  
361          dx = (tx - x) / drag_n
362          dy = (ty - y) / drag_n
363  
364          ts0 = time.perf_counter()
365          for i in range(drag_n):
366              await self.async_sleep(
367                  max(0., duration - (time.perf_counter() - ts0)) / (drag_n - i))
368  
369              touch.touch_move(x + (i + 1) * dx, y + (i + 1) * dy)
370              await self.wait_clock_frames(1)
371              yield 'move', touch.pos
372  
373          if touch.pos != target_pos:
374              touch.touch_move(*target_pos)
375              await self.wait_clock_frames(1)
376              yield 'move', touch.pos
377  
378          touch.touch_up()
379          await self.wait_clock_frames(1)
380          yield 'up', touch.pos

382      async def do_touch_drag_follow(
383              self, pos=None, widget=None,
384              widget_loc=('center_x', 'center_y'),
385              target_pos=None, target_widget=None, target_widget_offset=(0, 0),
386              target_widget_loc=('center_x', 'center_y'), long_press=0,
387              duration=.2, drag_n=5, max_n=25):
388          """Very similar to :meth:`do_touch_drag`, except it follows the target 
389          widget, even if the target widget moves as a result of the drag, the 
390          drag will follow it until it's on the target widget. 
391   
392          `origin`: These parameters specify where the drag starts. 
393          - If ``widget`` is None, it starts at ``pos`` (in window coordinates). 
394          - If ``pos`` is None, it starts on the ``widget`` as specified by 
395            ``widget_loc``. 
396          - If neither is None, it starts at ``pos``, but in the ``widget``'s 
397            coordinate system (:meth:`~kivy.uix.widget.Widget.to_window` is used 
398            on it). 
399   
400          `target`: These parameters specify where the drag ends. 
401          - If ``target_pos`` is None, it ends on the ``target_widget`` as 
402            specified by ``target_widget_loc``. ``target_widget_offset``, is an 
403            additional ``(x, y)`` offset relative to ``target_widget_loc``. 
404          - If ``target_pos`` is not None, it starts at ``target_pos``, but in 
405            the ``target_widget``'s coordinate system 
406            (:meth:`~kivy.uix.widget.Widget.to_window` is used on it). 
407   
408          When ``widget`` and/or ``target_widget`` are specified, ``widget_loc`` 
409          and ``target_widget_loc``, respectively, indicate where on the widget 
410          the drag starts/ends. It is a a tuple with property names of the widget 
411          to loop up to get the value. The default is 
412          ``('center_x', 'center_y')`` so the drag would start/end in the 
413          widget's center. 
414          """
415          if widget is None:
416              x, y = pos
417          else:
418              if pos is None:
419                  w_x = getattr(widget, widget_loc[0])
420                  w_y = getattr(widget, widget_loc[1])
421                  x, y = widget.to_window(w_x, w_y)
422              else:
423                  x, y = widget.to_window(*pos, initial=False)
424  
425          if target_widget is None:
426              raise ValueError('target_widget must be specified')
427  
428          def get_target():
429              if target_pos is not None:
430                  return target_widget.to_window(*target_pos, initial=False)
431              else:
432                  x_off, y_off = target_widget_offset
433                  wt_x = getattr(target_widget, target_widget_loc[0]) + x_off
434                  wt_y = getattr(target_widget, target_widget_loc[1]) + y_off
435                  return target_widget.to_window(wt_x, wt_y)
436  
437          touch = AsyncUnitTestTouch(x, y)
438  
439          touch.touch_down()
440          await self.wait_clock_frames(1)
441          if long_press:
442              await self.async_sleep(long_press)
443          yield 'down', touch.pos
444  
445          ts0 = time.perf_counter()
446          tx, ty = get_target()
447          i = 0
448          while not (math.isclose(touch.x, tx) and math.isclose(touch.y, ty)):
449              if i >= max_n:
450                  raise Exception(
451                      'Exceeded the maximum number of iterations, '
452                      'but {} != {}'.format(touch.pos, (tx, ty)))
453  
454              rem_i = max(1, drag_n - i)
455              rem_t = max(0., duration - (time.perf_counter() - ts0)) / rem_i
456              i += 1
457              await self.async_sleep(rem_t)
458  
459              x, y = touch.pos
460              touch.touch_move(x + (tx - x) / rem_i, y + (ty - y) / rem_i)
461              await self.wait_clock_frames(1)
462              yield 'move', touch.pos
463              tx, ty = get_target()
464  
465          touch.touch_up()
466          await self.wait_clock_frames(1)
467          yield 'up', touch.pos

469      async def do_touch_drag_path(
470              self, path, axis_widget=None, long_press=0, duration=.2):
471          """Drags the touch along the specified path. 
472   
473          :parameters: 
474   
475              `path`: list 
476                  A list of position tuples the touch will follow. The first 
477                  item is used for the touch down and the rest for the move. 
478              `axis_widget`: a Widget 
479                  If None, the path coordinates is in window coordinates, 
480                  otherwise, we will first transform the path coordinates 
481                  to window coordinates using 
482                  :meth:`~kivy.uix.widget.Widget.to_window` of the specified 
483                  widget. 
484          """
485          if axis_widget is not None:
486              path = [axis_widget.to_window(*p, initial=False) for p in path]
487          x, y = path[0]
488          path = path[1:]
489  
490          touch = AsyncUnitTestTouch(x, y)
491  
492          touch.touch_down()
493          await self.wait_clock_frames(1)
494          if long_press:
495              await self.async_sleep(long_press)
496          yield 'down', touch.pos
497  
498          ts0 = time.perf_counter()
499          n = len(path)
500          for i, (x2, y2) in enumerate(path):
501              await self.async_sleep(
502                  max(0., duration - (time.perf_counter() - ts0)) / (n - i))
503  
504              touch.touch_move(x2, y2)
505              await self.wait_clock_frames(1)
506              yield 'move', touch.pos
507  
508          touch.touch_up()
509          await self.wait_clock_frames(1)
510          yield 'up', touch.pos

512      async def do_keyboard_key(
513              self, key, modifiers=(), duration=.05, num_press=1):
514          from kivy.core.window import Window
515          if key == ' ':
516              key = 'spacebar'
517          key_lower = key.lower()
518          key_code = Window._system_keyboard.string_to_keycode(key_lower)
519  
520          known_modifiers = {'shift', 'alt', 'ctrl', 'meta'}
521          if set(modifiers) - known_modifiers:
522              raise ValueError('Unknown modifiers "{}"'.
523                               format(set(modifiers) - known_modifiers))
524  
525          special_keys = {
526              27: 'escape',
527              9: 'tab',
528              8: 'backspace',
529              13: 'enter',
530              127: 'del',
531              271: 'enter',
532              273: 'up',
533              274: 'down',
534              275: 'right',
535              276: 'left',
536              278: 'home',
537              279: 'end',
538              280: 'pgup',
539              281: 'pgdown',
540              300: 'numlock',
541              301: 'capslock',
542              145: 'screenlock',
543          }
544  
545          text = None
546          try:
547              text = chr(key_code)
548              if key_lower != key:
549                  text = key
550          except ValueError:
551              pass
552  
553          dt = duration / num_press
554          for i in range(num_press):
555              await self.async_sleep(dt)
556  
557              Window.dispatch('on_key_down', key_code, 0, text, modifiers)
558              if (key not in known_modifiers and
559                      key_code not in special_keys and
560                      not (known_modifiers & set(modifiers))):
561                  Window.dispatch('on_textinput', text)
562  
563              await self.wait_clock_frames(1)
564              yield 'down', (key, key_code, 0, text, modifiers)
565  
566          Window.dispatch('on_key_up', key_code, 0)
567          await self.wait_clock_frames(1)
568          yield 'up', (key, key_code, 0, text, modifiers)
