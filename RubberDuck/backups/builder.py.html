60   def call_fn(args, instance, v):
61       element, key, value, rule, idmap = args
62       if __debug__:
63           trace('Lang: call_fn %s, key=%s, value=%r, %r' % (
64               element, key, value, rule.value))
65       rule.count += 1
66       e_value = eval(value, idmap)
67       if __debug__:
68           trace('Lang: call_fn => value=%r' % (e_value, ))
69       setattr(element, key, e_value)

72   def delayed_call_fn(args, instance, v):
73       # it's already on the list
74       if args[-1] is not None:
75           return
76   
77       global _delayed_start
78       if _delayed_start is None:
79           _delayed_start = args
80           args[-1] = StopIteration
81       else:
82           args[-1] = _delayed_start
83           _delayed_start = args

86   def update_intermediates(base, keys, bound, s, fn, args, instance, value):
87       ''' Function that is called when an intermediate property is updated 
88       and `rebind` of that property is True. In that case, we unbind 
89       all bound funcs that were bound to attrs of the old value of the 
90       property and rebind to the new value of the property. 
91    
92       For example, if the rule is `self.a.b.c.d`, then when b is changed, we 
93       unbind from `b`, `c` and `d`, if they were bound before (they were not 
94       None and `rebind` of the respective properties was True) and we rebind 
95       to the new values of the attrs `b`, `c``, `d` that are not None and 
96       `rebind` is True. 
97    
98       :Parameters: 
99           `base` 
100              A (proxied) ref to the base widget, `self` in the example 
101              above. 
102          `keys` 
103              A list of the name off the attrs of `base` being watched. In 
104              the example above it'd be `['a', 'b', 'c', 'd']`. 
105          `bound` 
106              A list 4-tuples, each tuple being (widget, attr, callback, uid) 
107              representing callback functions bound to the attributed `attr` 
108              of `widget`. `uid` is returned by `fbind` when binding. 
109              The callback may be None, in which case the attr 
110              was not bound, but is there to be able to walk the attr tree. 
111              E.g. in the example above, if `b` was not an eventdispatcher, 
112              `(_b_ref_, `c`, None)` would be added to the list so we can get 
113              to `c` and `d`, which may be eventdispatchers and their attrs. 
114          `s` 
115              The index in `keys` of the of the attr that needs to be 
116              updated. That is all the keys from `s` and further will be 
117              rebound, since the `s` key was changed. In bound, the 
118              corresponding index is `s - 1`. If `s` is None, we start from 
119              1 (first attr). 
120          `fn` 
121              The function to be called args, `args` on bound callback. 
122      '''
123      # first remove all the old bound functions from `s` and down.
124      for f, k, fun, uid in bound[s:]:
125          if fun is None:
126              continue
127          try:
128              f.unbind_uid(k, uid)
129          except ReferenceError:
130              pass
131      del bound[s:]
132  
133      # find the first attr from which we need to start rebinding.
134      f = getattr(*bound[-1][:2])
135      if f is None:
136          fn(args, None, None)
137          return
138      s += 1
139      append = bound.append
140  
141      # bind all attrs, except last to update_intermediates
142      for val in keys[s:-1]:
143          # if we need to dynamically rebind, bindm otherwise just
144          # add the attr to the list
145          if isinstance(f, (EventDispatcher, Observable)):
146              prop = f.property(val, True)
147              if prop is not None and getattr(prop, 'rebind', False):
148                  # fbind should not dispatch, otherwise
149                  # update_intermediates might be called in the middle
150                  # here messing things up
151                  uid = f.fbind(
152                      val, update_intermediates, base, keys, bound, s, fn, args)
153                  append([f.proxy_ref, val, update_intermediates, uid])
154              else:
155                  append([f.proxy_ref, val, None, None])
156          else:
157              append([getattr(f, 'proxy_ref', f), val, None, None])
158  
159          f = getattr(f, val, None)
160          if f is None:
161              break
162          s += 1
163  
164      # for the last attr we bind directly to the setting function,
165      # because that attr sets the value of the rule.
166      if isinstance(f, (EventDispatcher, Observable)):
167          uid = f.fbind(keys[-1], fn, args)
168          if uid:
169              append([f.proxy_ref, keys[-1], fn, uid])
170      # when we rebind we have to update the
171      # rule with the most recent value, otherwise, the value might be wrong
172      # and wouldn't be updated since we might not have tracked it before.
173      # This only happens for a callback when rebind was True for the prop.
174      fn(args, None, None)

177  def create_handler(iself, element, key, value, rule, idmap, delayed=False):
178      idmap = copy(idmap)
179      idmap.update(global_idmap)
180      idmap['self'] = iself.proxy_ref
181      bound_list = _handlers[iself.uid][key]
182      handler_append = bound_list.append
183  
184      # we need a hash for when delayed, so we don't execute duplicate canvas
185      # callbacks from the same handler during a sync op
186      if delayed:
187          fn = delayed_call_fn
188          args = [element, key, value, rule, idmap, None]  # see _delayed_start
189      else:
190          fn = call_fn
191          args = (element, key, value, rule, idmap)
192  
193      # bind every key.value
194      if rule.watched_keys is not None:
195          for keys in rule.watched_keys:
196              base = idmap.get(keys[0])
197              if base is None:
198                  continue
199              f = base = getattr(base, 'proxy_ref', base)
200              bound = []
201              was_bound = False
202              append = bound.append
203  
204              # bind all attrs, except last to update_intermediates
205              k = 1
206              for val in keys[1:-1]:
207                  # if we need to dynamically rebind, bindm otherwise
208                  # just add the attr to the list
209                  if isinstance(f, (EventDispatcher, Observable)):
210                      prop = f.property(val, True)
211                      if prop is not None and getattr(prop, 'rebind', False):
212                          # fbind should not dispatch, otherwise
213                          # update_intermediates might be called in the middle
214                          # here messing things up
215                          uid = f.fbind(
216                              val, update_intermediates, base, keys, bound, k,
217                              fn, args)
218                          append([f.proxy_ref, val, update_intermediates, uid])
219                          was_bound = True
220                      else:
221                          append([f.proxy_ref, val, None, None])
222                  elif not isinstance(f, type):
223                      append([getattr(f, 'proxy_ref', f), val, None, None])
224                  else:
225                      append([f, val, None, None])
226                  f = getattr(f, val, None)
227                  if f is None:
228                      break
229                  k += 1
230  
231              # for the last attr we bind directly to the setting
232              # function, because that attr sets the value of the rule.
233              if isinstance(f, (EventDispatcher, Observable)):
234                  uid = f.fbind(keys[-1], fn, args)  # f is not None
235                  if uid:
236                      append([f.proxy_ref, keys[-1], fn, uid])
237                      was_bound = True
238              if was_bound:
239                  handler_append(bound)
240  
241      try:
242          return eval(value, idmap), bound_list
243      except Exception as e:
244          tb = sys.exc_info()[2]
245          raise BuilderException(rule.ctx, rule.line,
246                                 '{}: {}'.format(e.__class__.__name__, e),
247                                 cause=tb)

250  class BuilderBase(object):
251      '''The Builder is responsible for creating a :class:`Parser` for parsing a 
252      kv file, merging the results into its internal rules, templates, etc. 
253   
254      By default, :class:`Builder` is a global Kivy instance used in widgets 
255      that you can use to load other kv files in addition to the default ones. 
256      '''
257  
258      def __init__(self):
259          super(BuilderBase, self).__init__()
260          self._match_cache = {}
261          self._match_name_cache = {}
262          self.files = []
263          self.dynamic_classes = {}
264          self.templates = {}
265          self.rules = []
266          self.rulectx = {}

268      @classmethod
269      def create_from(cls, builder):
270          """Creates a instance of the class, and initializes to the state of 
271          ``builder``. 
272   
273          :param builder: The builder to initialize from. 
274          :return: A new instance of this class. 
275          """
276          obj = cls()
277          obj._match_cache = copy(builder._match_cache)
278          obj._match_name_cache = copy(builder._match_name_cache)
279          obj.files = copy(builder.files)
280          obj.dynamic_classes = copy(builder.dynamic_classes)
281          obj.templates = copy(builder.templates)
282          obj.rules = list(builder.rules)
283          assert not builder.rulectx
284          obj.rulectx = dict(builder.rulectx)
285  
286          return obj

288      def load_file(self, filename, encoding='utf8', **kwargs):
289          '''Insert a file into the language builder and return the root widget 
290          (if defined) of the kv file. 
291   
292          :parameters: 
293              `rulesonly`: bool, defaults to False 
294                  If True, the Builder will raise an exception if you have a root 
295                  widget inside the definition. 
296   
297              `encoding`: File charcter encoding. Defaults to utf-8, 
298          '''
299          filename = resource_find(filename) or filename
300          if __debug__:
301              trace('Lang: load file %s, using %s encoding', filename, encoding)
302  
303          kwargs['filename'] = filename
304          with open(filename, 'r', encoding=encoding) as fd:
305              data = fd.read()
306              return self.load_string(data, **kwargs)

308      def unload_file(self, filename):
309          '''Unload all rules associated with a previously imported file. 
310   
311          .. versionadded:: 1.0.8 
312   
313          .. warning:: 
314   
315              This will not remove rules or templates already applied/used on 
316              current widgets. It will only effect the next widgets creation or 
317              template invocation. 
318          '''
319          # remove rules and templates
320          filename = resource_find(filename) or filename
321          self.rules = [x for x in self.rules if x[1].ctx.filename != filename]
322          self._clear_matchcache()
323          templates = {}
324          for x, y in self.templates.items():
325              if y[2] != filename:
326                  templates[x] = y
327          self.templates = templates
328  
329          if filename in self.files:
330              self.files.remove(filename)
331  
332          # unregister all the dynamic classes
333          Factory.unregister_from_filename(filename)

335      def load_string(self, string, **kwargs):
336          '''Insert a string into the Language Builder and return the root widget 
337          (if defined) of the kv string. 
338   
339          :Parameters: 
340              `rulesonly`: bool, defaults to False 
341                  If True, the Builder will raise an exception if you have a root 
342                  widget inside the definition. 
343              `filename`: str, defaults to None 
344                  If specified, the filename used to index the kv rules. 
345   
346          The filename parameter can be used to unload kv strings in the same way 
347          as you unload kv files. This can be achieved using pseudo file names 
348          e.g.:: 
349   
350              Build.load_string(""" 
351                  <MyRule>: 
352                      Label: 
353                          text="Hello" 
354              """, filename="myrule.kv") 
355   
356          can be unloaded via:: 
357   
358              Build.unload_file("myrule.kv") 
359   
360          '''
361  
362          kwargs.setdefault('rulesonly', False)
363          self._current_filename = fn = kwargs.get('filename', None)
364  
365          # put a warning if a file is loaded multiple times
366          if fn in self.files:
367              Logger.warning(
368                  'Lang: The file {} is loaded multiples times, '
369                  'you might have unwanted behaviors.'.format(fn))
370  
371          try:
372              # parse the string
373              parser = Parser(content=string, filename=fn)
374  
375              # merge rules with our rules
376              self.rules.extend(parser.rules)
377              self._clear_matchcache()
378  
379              # add the template found by the parser into ours
380              for name, cls, template in parser.templates:
381                  self.templates[name] = (cls, template, fn)
382                  Factory.register(name,
383                                   cls=partial(self.template, name),
384                                   is_template=True, warn=True)
385  
386              # register all the dynamic classes
387              for name, baseclasses in parser.dynamic_classes.items():
388                  Factory.register(name, baseclasses=baseclasses, filename=fn,
389                                   warn=True)
390  
391              # create root object is exist
392              if kwargs['rulesonly'] and parser.root:
393                  filename = kwargs.get('rulesonly', '<string>')
394                  raise Exception('The file <%s> contain also non-rules '
395                                  'directives' % filename)
396  
397              # save the loaded files only if there is a root without
398              # template/dynamic classes
399              if fn and (parser.templates or
400                         parser.dynamic_classes or parser.rules):
401                  self.files.append(fn)
402  
403              if parser.root:
404                  widget = Factory.get(parser.root.name)(__no_builder=True)
405                  rule_children = []
406                  widget.apply_class_lang_rules(
407                      root=widget, rule_children=rule_children)
408                  self._apply_rule(
409                      widget, parser.root, parser.root,
410                      rule_children=rule_children)
411  
412                  for child in rule_children:
413                      child.dispatch('on_kv_post', widget)
414                  widget.dispatch('on_kv_post', widget)
415                  return widget
416          finally:
417              self._current_filename = None

419      def template(self, *args, **ctx):
420          '''Create a specialized template using a specific context. 
421   
422          .. versionadded:: 1.0.5 
423   
424          With templates, you can construct custom widgets from a kv lang 
425          definition by giving them a context. Check :ref:`Template usage 
426          <template_usage>`. 
427          '''
428          # Prevent naming clash with whatever the user might be putting into the
429          # ctx as key.
430          name = args[0]
431          if name not in self.templates:
432              raise Exception('Unknown <%s> template name' % name)
433          baseclasses, rule, fn = self.templates[name]
434          key = '%s|%s' % (name, baseclasses)
435          cls = Cache.get('kv.lang', key)
436          if cls is None:
437              rootwidgets = []
438              for basecls in baseclasses.split('+'):
439                  rootwidgets.append(Factory.get(basecls))
440              cls = type(name, tuple(rootwidgets), {})
441              Cache.append('kv.lang', key, cls)
442          widget = cls()
443          # in previous versions, ``ctx`` is passed as is as ``template_ctx``
444          # preventing widgets in it from be collected by the GC. This was
445          # especially relevant to AccordionItem's title_template.
446          proxy_ctx = {k: get_proxy(v) for k, v in ctx.items()}
447          self._apply_rule(widget, rule, rule, template_ctx=proxy_ctx)
448          return widget

450      def apply_rules(
451              self, widget, rule_name, ignored_consts=set(), rule_children=None,
452              dispatch_kv_post=False):
453          '''Search all the rules that match the name `rule_name` 
454          and apply them to `widget`. 
455   
456          .. versionadded:: 1.10.0 
457   
458          :Parameters: 
459   
460              `widget`: :class:`~kivy.uix.widget.Widget` 
461                  The widget to whom the matching rules should be applied to. 
462              `ignored_consts`: set 
463                  A set or list type whose elements are property names for which 
464                  constant KV rules (i.e. those that don't create bindings) of 
465                  that widget will not be applied. This allows e.g. skipping 
466                  constant rules that overwrite a value initialized in python. 
467              `rule_children`: list 
468                  If not ``None``, it should be a list that will be populated 
469                  with all the widgets created by the kv rules being applied. 
470   
471                  .. versionchanged:: 1.11.0 
472   
473              `dispatch_kv_post`: bool 
474                  Normally the class `Widget` dispatches the `on_kv_post` event 
475                  to widgets created during kv rule application. 
476                  But if the rules are manually applied by calling :meth:`apply`, 
477                  that may not happen, so if this is `True`, we will dispatch the 
478                  `on_kv_post` event where needed after applying the rules to 
479                  `widget` (we won't dispatch it for `widget` itself). 
480   
481                  Defaults to False. 
482   
483                  .. versionchanged:: 1.11.0 
484          '''
485          rules = self.match_rule_name(rule_name)
486          if __debug__:
487              trace('Lang: Found %d rules for %s' % (len(rules), rule_name))
488          if not rules:
489              return
490  
491          if dispatch_kv_post:
492              rule_children = rule_children if rule_children is not None else []
493          for rule in rules:
494              self._apply_rule(
495                  widget, rule, rule, ignored_consts=ignored_consts,
496                  rule_children=rule_children)
497          if dispatch_kv_post:
498              for w in rule_children:
499                  w.dispatch('on_kv_post', widget)

501      def apply(self, widget, ignored_consts=set(), rule_children=None,
502                dispatch_kv_post=False):
503          '''Search all the rules that match the widget and apply them. 
504   
505          :Parameters: 
506   
507              `widget`: :class:`~kivy.uix.widget.Widget` 
508                  The widget whose class rules should be applied to this widget. 
509              `ignored_consts`: set 
510                  A set or list type whose elements are property names for which 
511                  constant KV rules (i.e. those that don't create bindings) of 
512                  that widget will not be applied. This allows e.g. skipping 
513                  constant rules that overwrite a value initialized in python. 
514              `rule_children`: list 
515                  If not ``None``, it should be a list that will be populated 
516                  with all the widgets created by the kv rules being applied. 
517   
518                  .. versionchanged:: 1.11.0 
519   
520              `dispatch_kv_post`: bool 
521                  Normally the class `Widget` dispatches the `on_kv_post` event 
522                  to widgets created during kv rule application. 
523                  But if the rules are manually applied by calling :meth:`apply`, 
524                  that may not happen, so if this is `True`, we will dispatch the 
525                  `on_kv_post` event where needed after applying the rules to 
526                  `widget` (we won't dispatch it for `widget` itself). 
527   
528                  Defaults to False. 
529   
530                  .. versionchanged:: 1.11.0 
531          '''
532          rules = self.match(widget)
533          if __debug__:
534              trace('Lang: Found %d rules for %s' % (len(rules), widget))
535          if not rules:
536              return
537  
538          if dispatch_kv_post:
539              rule_children = rule_children if rule_children is not None else []
540          for rule in rules:
541              self._apply_rule(
542                  widget, rule, rule, ignored_consts=ignored_consts,
543                  rule_children=rule_children)
544          if dispatch_kv_post:
545              for w in rule_children:
546                  w.dispatch('on_kv_post', widget)

548      def _clear_matchcache(self):
549          self._match_cache.clear()
550          self._match_name_cache.clear()

552      def _apply_rule(self, widget, rule, rootrule, template_ctx=None,
553                      ignored_consts=set(), rule_children=None):
554          # widget: the current instantiated widget
555          # rule: the current rule
556          # rootrule: the current root rule (for children of a rule)
557  
558          # will collect reference to all the id in children
559          assert(rule not in self.rulectx)
560          self.rulectx[rule] = rctx = {
561              'ids': {'root': widget.proxy_ref},
562              'set': [], 'hdl': []}
563  
564          # extract the context of the rootrule (not rule!)
565          assert(rootrule in self.rulectx)
566          rctx = self.rulectx[rootrule]
567  
568          # if a template context is passed, put it as "ctx"
569          if template_ctx is not None:
570              rctx['ids']['ctx'] = QueryDict(template_ctx)
571  
572          # if we got an id, put it in the root rule for a later global usage
573          if rule.id:
574              # use only the first word as `id` discard the rest.
575              rule.id = rule.id.split('#', 1)[0].strip()
576              rctx['ids'][rule.id] = widget.proxy_ref
577              # set id name as a attribute for root widget so one can in python
578              # code simply access root_widget.id_name
579              _ids = dict(rctx['ids'])
580              _root = _ids.pop('root')
581              _new_ids = _root.ids
582              for _key in _ids.keys():
583                  if _ids[_key] == _root:
584                      # skip on self
585                      continue
586                  _new_ids[_key] = _ids[_key]
587              _root.ids = _new_ids
588  
589          # first, ensure that the widget have all the properties used in
590          # the rule if not, they will be created as ObjectProperty.
591          rule.create_missing(widget)
592  
593          # build the widget canvas
594          if rule.canvas_before:
595              with widget.canvas.before:
596                  self._build_canvas(widget.canvas.before, widget,
597                                     rule.canvas_before, rootrule)
598          if rule.canvas_root:
599              with widget.canvas:
600                  self._build_canvas(widget.canvas, widget,
601                                     rule.canvas_root, rootrule)
602          if rule.canvas_after:
603              with widget.canvas.after:
604                  self._build_canvas(widget.canvas.after, widget,
605                                     rule.canvas_after, rootrule)
606  
607          # create children tree
608          Factory_get = Factory.get
609          Factory_is_template = Factory.is_template
610          for crule in rule.children:
611              cname = crule.name
612  
613              if cname in ('canvas', 'canvas.before', 'canvas.after'):
614                  raise ParserException(
615                      crule.ctx, crule.line,
616                      'Canvas instructions added in kv must '
617                      'be declared before child widgets.')
618  
619              # depending if the child rule is a template or not, we are not
620              # having the same approach
621              cls = Factory_get(cname)
622  
623              if Factory_is_template(cname):
624                  # we got a template, so extract all the properties and
625                  # handlers, and push them in a "ctx" dictionary.
626                  ctx = {}
627                  idmap = copy(global_idmap)
628                  idmap.update({'root': rctx['ids']['root']})
629                  if 'ctx' in rctx['ids']:
630                      idmap.update({'ctx': rctx['ids']['ctx']})
631                  try:
632                      for prule in crule.properties.values():
633                          value = prule.co_value
634                          if type(value) is CodeType:
635                              value = eval(value, idmap)
636                          ctx[prule.name] = value
637                      for prule in crule.handlers:
638                          value = eval(prule.value, idmap)
639                          ctx[prule.name] = value
640                  except Exception as e:
641                      tb = sys.exc_info()[2]
642                      raise BuilderException(
643                          prule.ctx, prule.line,
644                          '{}: {}'.format(e.__class__.__name__, e), cause=tb)
645  
646                  # create the template with an explicit ctx
647                  child = cls(**ctx)
648                  widget.add_widget(child)
649  
650                  # reference it on our root rule context
651                  if crule.id:
652                      rctx['ids'][crule.id] = child
653  
654              else:
655                  # we got a "normal" rule, construct it manually
656                  # we can't construct it without __no_builder=True, because the
657                  # previous implementation was doing the add_widget() before
658                  # apply(), and so, we could use "self.parent".
659                  child = cls(__no_builder=True)
660                  widget.add_widget(child)
661                  child.apply_class_lang_rules(
662                      root=rctx['ids']['root'], rule_children=rule_children)
663                  self._apply_rule(
664                      child, crule, rootrule, rule_children=rule_children)
665  
666                  if rule_children is not None:
667                      rule_children.append(child)
668  
669          # append the properties and handlers to our final resolution task
670          if rule.properties:
671              rctx['set'].append((widget.proxy_ref,
672                                  list(rule.properties.values())))
673              for key, crule in rule.properties.items():
674                  # clear previously applied rules if asked
675                  if crule.ignore_prev:
676                      Builder.unbind_property(widget, key)
677          if rule.handlers:
678              rctx['hdl'].append((widget.proxy_ref, rule.handlers))
679  
680          # if we are applying another rule that the root one, then it's done for
681          # us!
682          if rootrule is not rule:
683              del self.rulectx[rule]
684              return
685  
686          # normally, we can apply a list of properties with a proper context
687          try:
688              rule = None
689              for widget_set, rules in reversed(rctx['set']):
690                  for rule in rules:
691                      assert(isinstance(rule, ParserRuleProperty))
692                      key = rule.name
693                      value = rule.co_value
694                      if type(value) is CodeType:
695                          value, bound = create_handler(
696                              widget_set, widget_set, key, value, rule,
697                              rctx['ids'])
698                          # if there's a rule
699                          if (widget_set != widget or bound or
700                                  key not in ignored_consts):
701                              setattr(widget_set, key, value)
702                      else:
703                          if (widget_set != widget or
704                                  key not in ignored_consts):
705                              setattr(widget_set, key, value)
706  
707          except Exception as e:
708              if rule is not None:
709                  tb = sys.exc_info()[2]
710                  raise BuilderException(rule.ctx, rule.line,
711                                         '{}: {}'.format(e.__class__.__name__,
712                                                         e), cause=tb)
713              raise e
714  
715          # build handlers
716          try:
717              crule = None
718              for widget_set, rules in rctx['hdl']:
719                  for crule in rules:
720                      assert(isinstance(crule, ParserRuleProperty))
721                      assert(crule.name.startswith('on_'))
722                      key = crule.name
723                      if not widget_set.is_event_type(key):
724                          key = key[3:]
725                      idmap = copy(global_idmap)
726                      idmap.update(rctx['ids'])
727                      idmap['self'] = widget_set.proxy_ref
728                      if not widget_set.fbind(key, custom_callback, crule,
729                                              idmap):
730                          raise AttributeError(key)
731                      # hack for on_parent
732                      if crule.name == 'on_parent':
733                          Factory.Widget.parent.dispatch(widget_set.__self__)
734          except Exception as e:
735              if crule is not None:
736                  tb = sys.exc_info()[2]
737                  raise BuilderException(
738                      crule.ctx, crule.line,
739                      '{}: {}'.format(e.__class__.__name__, e), cause=tb)
740              raise e
741  
742          # rule finished, forget it
743          del self.rulectx[rootrule]

745      def match(self, widget):
746          '''Return a list of :class:`ParserRule` objects matching the widget. 
747          '''
748          cache = self._match_cache
749          k = (widget.__class__, tuple(widget.cls))
750          if k in cache:
751              return cache[k]
752          rules = []
753          for selector, rule in self.rules:
754              if selector.match(widget):
755                  if rule.avoid_previous_rules:
756                      del rules[:]
757                  rules.append(rule)
758          cache[k] = rules
759          return rules

761      def match_rule_name(self, rule_name):
762          '''Return a list of :class:`ParserRule` objects matching the widget. 
763          '''
764          cache = self._match_name_cache
765          rule_name = str(rule_name)
766          k = rule_name.lower()
767          if k in cache:
768              return cache[k]
769          rules = []
770          for selector, rule in self.rules:
771              if selector.match_rule_name(rule_name):
772                  if rule.avoid_previous_rules:
773                      del rules[:]
774                  rules.append(rule)
775          cache[k] = rules
776          return rules

778      def sync(self):
779          '''Execute all the waiting operations, such as the execution of all the 
780          expressions related to the canvas. 
781   
782          .. versionadded:: 1.7.0 
783          '''
784          global _delayed_start
785          next_args = _delayed_start
786          if next_args is None:
787              return
788  
789          while next_args is not StopIteration:
790              # is this try/except still needed? yes, in case widget died in this
791              # frame after the call was scheduled
792              try:
793                  call_fn(next_args[:-1], None, None)
794              except ReferenceError:
795                  pass
796              args = next_args
797              next_args = args[-1]
798              args[-1] = None
799          _delayed_start = None

801      def unbind_widget(self, uid):
802          '''Unbind all the handlers created by the KV rules of the 
803          widget. The :attr:`kivy.uix.widget.Widget.uid` is passed here 
804          instead of the widget itself, because Builder is using it in the 
805          widget destructor. 
806   
807          This effectively clears all the KV rules associated with this widget. 
808          For example: 
809   
810          .. code-block:: python 
811   
812              >>> w = Builder.load_string(\''' 
813              ... Widget: 
814              ...     height: self.width / 2. if self.disabled else self.width 
815              ...     x: self.y + 50 
816              ... \''') 
817              >>> w.size 
818              [100, 100] 
819              >>> w.pos 
820              [50, 0] 
821              >>> w.width = 500 
822              >>> w.size 
823              [500, 500] 
824              >>> Builder.unbind_widget(w.uid) 
825              >>> w.width = 222 
826              >>> w.y = 500 
827              >>> w.size 
828              [222, 500] 
829              >>> w.pos 
830              [50, 500] 
831   
832          .. versionadded:: 1.7.2 
833          '''
834          if uid not in _handlers:
835              return
836          for prop_callbacks in _handlers[uid].values():
837              for callbacks in prop_callbacks:
838                  for f, k, fn, bound_uid in callbacks:
839                      if fn is None:  # it's not a kivy prop.
840                          continue
841                      try:
842                          f.unbind_uid(k, bound_uid)
843                      except ReferenceError:
844                          # proxy widget is already gone, that's cool :)
845                          pass
846          del _handlers[uid]

848      def unbind_property(self, widget, name):
849          '''Unbind the handlers created by all the rules of the widget that set 
850          the name. 
851   
852          This effectively clears all the rules of widget that take the form:: 
853   
854              name: rule 
855   
856          For example: 
857   
858          .. code-block:: python 
859   
860              >>> w = Builder.load_string(\''' 
861              ... Widget: 
862              ...     height: self.width / 2. if self.disabled else self.width 
863              ...     x: self.y + 50 
864              ... \''') 
865              >>> w.size 
866              [100, 100] 
867              >>> w.pos 
868              [50, 0] 
869              >>> w.width = 500 
870              >>> w.size 
871              [500, 500] 
872              >>> Builder.unbind_property(w, 'height') 
873              >>> w.width = 222 
874              >>> w.size 
875              [222, 500] 
876              >>> w.y = 500 
877              >>> w.pos 
878              [550, 500] 
879   
880          .. versionadded:: 1.9.1 
881          '''
882          uid = widget.uid
883          if uid not in _handlers:
884              return
885  
886          prop_handlers = _handlers[uid]
887          if name not in prop_handlers:
888              return
889  
890          for callbacks in prop_handlers[name]:
891              for f, k, fn, bound_uid in callbacks:
892                  if fn is None:  # it's not a kivy prop.
893                      continue
894                  try:
895                      f.unbind_uid(k, bound_uid)
896                  except ReferenceError:
897                      # proxy widget is already gone, that's cool :)
898                      pass
899          del prop_handlers[name]
900          if not prop_handlers:
901              del _handlers[uid]

903      def _build_canvas(self, canvas, widget, rule, rootrule):
904          global Instruction
905          if Instruction is None:
906              Instruction = Factory.get('Instruction')
907          idmap = copy(self.rulectx[rootrule]['ids'])
908          for crule in rule.children:
909              name = crule.name
910              if name == 'Clear':
911                  canvas.clear()
912                  continue
913              instr = Factory.get(name)()
914              if not isinstance(instr, Instruction):
915                  raise BuilderException(
916                      crule.ctx, crule.line,
917                      'You can add only graphics Instruction in canvas.')
918              try:
919                  for prule in crule.properties.values():
920                      key = prule.name
921                      value = prule.co_value
922                      if type(value) is CodeType:
923                          value, _ = create_handler(
924                              widget, instr.proxy_ref,
925                              key, value, prule, idmap, True)
926                      setattr(instr, key, value)
927              except Exception as e:
928                  tb = sys.exc_info()[2]
929                  raise BuilderException(
930                      prule.ctx, prule.line,
931                      '{}: {}'.format(e.__class__.__name__, e), cause=tb)

942      def match_rule(fn, index, rule):
943          if rule.ctx.filename != fn:
944              return
945          for prop, prp in rule.properties.items():
946              if prp.line != index:
947                  continue
948              yield prp
949          for child in rule.children:
950              for r in match_rule(fn, index, child):
951                  yield r
952          if rule.canvas_root:
953              for r in match_rule(fn, index, rule.canvas_root):
954                  yield r
955          if rule.canvas_before:
956              for r in match_rule(fn, index, rule.canvas_before):
957                  yield r
958          if rule.canvas_after:
959              for r in match_rule(fn, index, rule.canvas_after):
960                  yield r

962      def dump_builder_stats():
963          html = [
964              '<!doctype html>'
965              '<html><body>',
966              '<style type="text/css">\n',
967              'pre { margin: 0; }\n',
968              '</style>']
969          files = set([x[1].ctx.filename for x in Builder.rules])
970          for fn in files:
971              try:
972                  with open(fn) as f:
973                      lines = f.readlines()
974              except (IOError, TypeError) as e:
975                  continue
976              html += ['<h2>', fn, '</h2>', '<table>']
977              count = 0
978              for index, line in enumerate(lines):
979                  line = line.rstrip()
980                  line = cgi.escape(line)
981                  matched_prp = []
982                  for psn, rule in Builder.rules:
983                      matched_prp += list(match_rule(fn, index, rule))
984  
985                  count = sum(set([x.count for x in matched_prp]))
986  
987                  color = (255, 155, 155) if count else (255, 255, 255)
988                  html += ['<tr style="background-color: rgb{}">'.format(color),
989                           '<td>', str(index + 1), '</td>',
990                           '<td>', str(count), '</td>',
991                           '<td><pre>', line, '</pre></td>',
992                           '</tr>']
993              html += ['</table>']
994          html += ['</body></html>']
995          with open('builder_stats.html', 'w') as fd:
996              fd.write(''.join(html))
997  
998          print('Profiling written at builder_stats.html')
