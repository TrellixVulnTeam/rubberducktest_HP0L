169      def test_nextset(self):
170          con = self._connect()
171          try:
172              cur = con.cursor()
173  
174              stmts = [self.ddl1] + self._populate()
175              for sql in stmts:
176                  cur.execute(sql)
177  
178              self.help_nextset_setUp(cur)
179  
180              cur.callproc("deleteme")
181              numberofrows = cur.fetchone()
182              assert numberofrows[0] == 6
183              assert cur.nextset()
184              names = cur.fetchall()
185              assert len(names) == len(self.samples)
186              s = cur.nextset()
187              assert s == None, "No more return sets, should return None"
188          finally:
189              try:
