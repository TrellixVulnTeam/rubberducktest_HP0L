1    ''' 
2    Popup 
3    ===== 
4     
5    .. versionadded:: 1.0.7 
6     
7    .. image:: images/popup.jpg 
8        :align: right 
9     
10   The :class:`Popup` widget is used to create modal popups. By default, the popup 
11   will cover the whole "parent" window. When you are creating a popup, you 
12   must at least set a :attr:`Popup.title` and :attr:`Popup.content`. 
13    
14   Remember that the default size of a Widget is size_hint=(1, 1). If you don't 
15   want your popup to be fullscreen, either use size hints with values less than 1 
16   (for instance size_hint=(.8, .8)) or deactivate the size_hint and use 
17   fixed size attributes. 
18    
19    
20   .. versionchanged:: 1.4.0 
21       The :class:`Popup` class now inherits from 
22       :class:`~kivy.uix.modalview.ModalView`. The :class:`Popup` offers a default 
23       layout with a title and a separation bar. 
24    
25   Examples 
26   -------- 
27    
28   Example of a simple 400x400 Hello world popup:: 
29    
30       popup = Popup(title='Test popup', 
31           content=Label(text='Hello world'), 
32           size_hint=(None, None), size=(400, 400)) 
33    
34   By default, any click outside the popup will dismiss/close it. If you don't 
35   want that, you can set 
36   :attr:`~kivy.uix.modalview.ModalView.auto_dismiss` to False:: 
37    
38       popup = Popup(title='Test popup', content=Label(text='Hello world'), 
39                     auto_dismiss=False) 
40       popup.open() 
41    
42   To manually dismiss/close the popup, use 
43   :attr:`~kivy.uix.modalview.ModalView.dismiss`:: 
44    
45       popup.dismiss() 
46    
47   Both :meth:`~kivy.uix.modalview.ModalView.open` and 
48   :meth:`~kivy.uix.modalview.ModalView.dismiss` are bindable. That means you 
49   can directly bind the function to an action, e.g. to a button's on_press:: 
50    
51       # create content and add to the popup 
52       content = Button(text='Close me!') 
53       popup = Popup(content=content, auto_dismiss=False) 
54    
55       # bind the on_press event of the button to the dismiss function 
56       content.bind(on_press=popup.dismiss) 
57    
58       # open the popup 
59       popup.open() 
60    
61   Same thing in KV language only with :class:`Factory`: 
62    
63   .. code-block:: kv 
64    
65       #:import Factory kivy.factory.Factory 
66       <MyPopup@Popup>: 
67           auto_dismiss: False 
68           Button: 
69               text: 'Close me!' 
70               on_release: root.dismiss() 
71    
72       Button: 
73           text: 'Open popup' 
74           on_release: Factory.MyPopup().open() 
75    
76   .. note:: 
77    
78       Popup is a special widget. Don't try to add it as a child to any other 
79       widget. If you do, Popup will be handled like an ordinary widget and 
80       won't be created hidden in the background. 
81    
82       .. code-block:: kv 
83    
84           BoxLayout: 
85               MyPopup:  # bad! 
86    
87   Popup Events 
88   ------------ 
89    
90   There are two events available: `on_open` which is raised when the popup is 
91   opening, and `on_dismiss` which is raised when the popup is closed. 
92   For `on_dismiss`, you can prevent the 
93   popup from closing by explictly returning True from your callback:: 
94    
95       def my_callback(instance): 
96           print('Popup', instance, 'is being dismissed but is prevented!') 
97           return True 
98       popup = Popup(content=Label(text='Hello world')) 
99       popup.bind(on_dismiss=my_callback) 
100      popup.open() 
101   
102  '''

104  __all__ = ('Popup', 'PopupException')

106  from kivy.core.text import DEFAULT_FONT
107  from kivy.uix.modalview import ModalView
108  from kivy.properties import (StringProperty, ObjectProperty, OptionProperty,
109                               NumericProperty, ColorProperty)

111  
112  class PopupException(Exception):
113      '''Popup exception, fired when multiple content widgets are added to the 
114      popup. 
115   
116      .. versionadded:: 1.4.0 
117      '''
118  
119  
120  class Popup(ModalView):
121      '''Popup class. See module documentation for more information. 
122   
123      :Events: 
124          `on_open`: 
125              Fired when the Popup is opened. 
126          `on_dismiss`: 
127              Fired when the Popup is closed. If the callback returns True, the 
128              dismiss will be canceled. 
129      '''
130  
131      title = StringProperty('No title')
132      '''String that represents the title of the popup. 
133   
134      :attr:`title` is a :class:`~kivy.properties.StringProperty` and defaults to 
135      'No title'. 
136      '''
137  
138      title_size = NumericProperty('14sp')
139      '''Represents the font size of the popup title. 
140   
141      .. versionadded:: 1.6.0 
142   
143      :attr:`title_size` is a :class:`~kivy.properties.NumericProperty` and 
144      defaults to '14sp'. 
145      '''
146  
147      title_align = OptionProperty(
148          'left', options=['left', 'center', 'right', 'justify'])
149      '''Horizontal alignment of the title. 
150   
151      .. versionadded:: 1.9.0 
152   
153      :attr:`title_align` is a :class:`~kivy.properties.OptionProperty` and 
154      defaults to 'left'. Available options are left, center, right and justify. 
155      '''
156  
157      title_font = StringProperty(DEFAULT_FONT)
158      '''Font used to render the title text. 
159   
160      .. versionadded:: 1.9.0 
161   
162      :attr:`title_font` is a :class:`~kivy.properties.StringProperty` and 
163      defaults to 'Roboto'. This value is taken 
164      from :class:`~kivy.config.Config`. 
165      '''
166  
167      content = ObjectProperty(None)
168      '''Content of the popup that is displayed just under the title. 
169   
170      :attr:`content` is an :class:`~kivy.properties.ObjectProperty` and defaults 
171      to None. 
172      '''
173  
174      title_color = ColorProperty([1, 1, 1, 1])
175      '''Color used by the Title. 
176   
177      .. versionadded:: 1.8.0 
178   
179      :attr:`title_color` is a :class:`~kivy.properties.ColorProperty` and 
180      defaults to [1, 1, 1, 1]. 
181   
182      .. versionchanged:: 2.0.0 
183          Changed from :class:`~kivy.properties.ListProperty` to 
184          :class:`~kivy.properties.ColorProperty`. 
185      '''
186  
187      separator_color = ColorProperty([47 / 255., 167 / 255., 212 / 255., 1.])
188      '''Color used by the separator between title and content. 
189   
190      .. versionadded:: 1.1.0 
191   
192      :attr:`separator_color` is a :class:`~kivy.properties.ColorProperty` and 
193      defaults to [47 / 255., 167 / 255., 212 / 255., 1.]. 
194   
195      .. versionchanged:: 2.0.0 
196          Changed from :class:`~kivy.properties.ListProperty` to 
197          :class:`~kivy.properties.ColorProperty`. 
198      '''
199  
200      separator_height = NumericProperty('2dp')
201      '''Height of the separator. 
202   
203      .. versionadded:: 1.1.0 
204   
205      :attr:`separator_height` is a :class:`~kivy.properties.NumericProperty` and 
206      defaults to 2dp. 
207      '''
208  
209      # Internal properties used for graphical representation.
210  
211      _container = ObjectProperty(None)
212  
213      def add_widget(self, widget):
214          if self._container:
215              if self.content:
216                  raise PopupException(
217                      'Popup can have only one widget as content')
218              self.content = widget
219          else:
220              super(Popup, self).add_widget(widget)
221  
222      def on_content(self, instance, value):
223          if self._container:
224              self._container.clear_widgets()
225              self._container.add_widget(value)
226  
227      def on__container(self, instance, value):
228          if value is None or self.content is None:
229              return
230          self._container.clear_widgets()
231          self._container.add_widget(self.content)
232  
233      def on_touch_down(self, touch):
234          if self.disabled and self.collide_point(*touch.pos):
235              return True
236          return super(Popup, self).on_touch_down(touch)

240      from kivy.base import runTouchApp
241      from kivy.uix.button import Button
242      from kivy.uix.label import Label
243      from kivy.uix.gridlayout import GridLayout
244      from kivy.core.window import Window
245  
246      # add popup
247      content = GridLayout(cols=1)
248      content_cancel = Button(text='Cancel', size_hint_y=None, height=40)
249      content.add_widget(Label(text='This is a hello world'))
250      content.add_widget(content_cancel)
251      popup = Popup(title='Test popup',
252                    size_hint=(None, None), size=(256, 256),
253                    content=content, disabled=True)
254      content_cancel.bind(on_release=popup.dismiss)
255  
256      layout = GridLayout(cols=3)
257      for x in range(9):
258          btn = Button(text=str(x))
259          btn.bind(on_release=popup.open)
260          layout.add_widget(btn)
261  
262      Window.add_widget(layout)
263  
264      popup.open()
265  
266      runTouchApp()
