1    # SGrepMDI is by Gordon McMillan (gmcm@hypernet.com)
2    # It does basically what Find In Files does in MSVC with a couple enhancements.
3    # - It saves any directories in the app's ini file (if you want to get rid
4    #   of them you'll have to edit the file)
5    # - "Directories" can be directories,
6    #  -    semicolon separated lists of "directories",
7    #  -    environment variables that evaluate to "directories",
8    #  -    registry path names that evaluate to "directories",
9    #  -    all of which is recursive, so you can mix them all up.
10   # - It is MDI, so you can 'nest' greps and return to earlier ones,
11   #   (ie, have multiple results open at the same time)
12   # - Like FIF, double clicking a line opens an editor and takes you to the line.
13   # - You can highlight text, right click and start a new grep with the selected
14   #   text as search pattern and same directories etc as before.
15   # - You can save grep parameters (so you don't lose your hardearned pattern)
16   #   from File|Save
17   # - You can save grep results by right clicking in the result window.
18   # Hats off to Mark Hammond for providing an environment where I could cobble
19   # something like this together in a couple evenings!

34   def getsubdirs(d):
35       dlist = []
36       flist = glob.glob(d + "\\*")
37       for f in flist:
38           if os.path.isdir(f):
39               dlist.append(f)
40               dlist = dlist + getsubdirs(f)
41       return dlist

44   class dirpath:
45       def __init__(self, str, recurse=0):
46           dp = str.split(";")
47           dirs = {}
48           for d in dp:
49               if os.path.isdir(d):
50                   d = d.lower()
51                   if d not in dirs:
52                       dirs[d] = None
53                       if recurse:
54                           subdirs = getsubdirs(d)
55                           for sd in subdirs:
56                               sd = sd.lower()
57                               if sd not in dirs:
58                                   dirs[sd] = None
59               elif os.path.isfile(d):
60                   pass
61               else:
62                   x = None
63                   if d in os.environ:
64                       x = dirpath(os.environ[d])
65                   elif d[:5] == "HKEY_":
66                       keystr = d.split("\\")
67                       try:
68                           root = eval("win32con." + keystr[0])
69                       except:
70                           win32ui.MessageBox(
71                               "Can't interpret registry key name '%s'" % keystr[0]
72                           )
73                       try:
74                           subkey = "\\".join(keystr[1:])
75                           val = win32api.RegQueryValue(root, subkey)
76                           if val:
77                               x = dirpath(val)
78                           else:
79                               win32ui.MessageBox(
80                                   "Registry path '%s' did not return a path entry" % d
81                               )
82                       except:
83                           win32ui.MessageBox(
84                               "Can't interpret registry key value: %s" % keystr[1:]
85                           )
86                   else:
87                       win32ui.MessageBox("Directory '%s' not found" % d)
88                   if x:
89                       for xd in x:
90                           if xd not in dirs:
91                               dirs[xd] = None
92                               if recurse:
93                                   subdirs = getsubdirs(xd)
94                                   for sd in subdirs:
95                                       sd = sd.lower()
96                                       if sd not in dirs:
97                                           dirs[sd] = None
98           self.dirs = []
99           for d in list(dirs.keys()):
100              self.dirs.append(d)
101

103          return self.dirs[key]

109          self.dirs[key] = value

112          del self.dirs[key]

115          return self.dirs[lo:hi]

118          self.dirs[lo:hi] = seq

121          del self.dirs[lo:hi]

124          if type(other) == type(self) or type(other) == type([]):

128          if type(other) == type(self) or type(other) == type([]):
129              return other.dirs + self.dirs

180  class GrepDocument(docview.RichEditDoc):
181      def __init__(self, template):
182          docview.RichEditDoc.__init__(self, template)
183          self.dirpattern = ""
184          self.filpattern = ""
185          self.greppattern = ""

188          self.verbose = 0

190      def OnOpenDocument(self, fnm):
191          # this bizarre stuff with params is so right clicking in a result window
192          # and starting a new grep can communicate the default parameters to the
193          # new grep.
194          try:
195              params = open(fnm, "r").read()
196          except:
197              params = None
198          self.setInitParams(params)
199          return self.OnNewDocument()

201      def OnCloseDocument(self):
202          try:
203              win32ui.GetApp().DeleteIdleHandler(self.SearchFile)
204          except:
205              pass
206          return self._obj_.OnCloseDocument()

208      def saveInitParams(self):
209          # Only save the flags, not the text boxes.
210          paramstr = "\t%s\t\t%d\t%d" % (
211              self.filpattern,
212              self.casesensitive,
213              self.recurse,
214          )
215          win32ui.WriteProfileVal("Grep", "Params", paramstr)
216  
217      def setInitParams(self, paramstr):
218          if paramstr is None:
219              paramstr = win32ui.GetProfileVal("Grep", "Params", "\t\t\t1\t0\t0")
220          params = paramstr.split("\t")
221          if len(params) < 3:
222              params = params + [""] * (3 - len(params))
223          if len(params) < 6:
224              params = params + [0] * (6 - len(params))
225          self.dirpattern = params[0]
226          self.filpattern = params[1]
227          self.greppattern = params[2]
228          self.casesensitive = int(params[3])
229          self.recurse = int(params[4])
230          self.verbose = int(params[5])
231          # setup some reasonable defaults.
232          if not self.dirpattern:
233              try:
234                  editor = win32ui.GetMainFrame().MDIGetActive()[0].GetEditorView()
235                  self.dirpattern = os.path.abspath(
236                      os.path.dirname(editor.GetDocument().GetPathName())
237                  )
238              except (AttributeError, win32ui.error):
239                  self.dirpattern = os.getcwd()
240          if not self.filpattern:
241              self.filpattern = "*.py"
242  
243      def OnNewDocument(self):
244          if self.dirpattern == "":
245              self.setInitParams(greptemplate.readParams())
246          d = GrepDialog(
247              self.dirpattern,
248              self.filpattern,
249              self.greppattern,
250              self.casesensitive,
251              self.recurse,
252              self.verbose,
253          )
254          if d.DoModal() == win32con.IDOK:
255              self.dirpattern = d["dirpattern"]
256              self.filpattern = d["filpattern"]
257              self.greppattern = d["greppattern"]
258              self.casesensitive = d["casesensitive"]
259              self.recurse = d["recursive"]
260              self.verbose = d["verbose"]
261              self.doSearch()
262              self.saveInitParams()
263              return 1
264          return 0  # cancelled - return zero to stop frame creation.

266      def doSearch(self):
267          self.dp = dirpath(self.dirpattern, self.recurse)
268          self.SetTitle("Grep for %s in %s" % (self.greppattern, self.filpattern))
269          # self.text = []
270          self.GetFirstView().Append("#Search " + self.dirpattern + "\n")
271          if self.verbose:
272              self.GetFirstView().Append("#   =" + repr(self.dp.dirs) + "\n")
273          self.GetFirstView().Append("# Files " + self.filpattern + "\n")
274          self.GetFirstView().Append("#   For " + self.greppattern + "\n")
275          self.fplist = self.filpattern.split(";")
276          if self.casesensitive:
277              self.pat = re.compile(self.greppattern)
278          else:
279              self.pat = re.compile(self.greppattern, re.IGNORECASE)
280          win32ui.SetStatusText("Searching.  Please wait...", 0)
281          self.dpndx = self.fpndx = 0
282          self.fndx = -1
283          if not self.dp:
284              self.GetFirstView().Append(
285                  "# ERROR: '%s' does not resolve to any search locations"
286                  % self.dirpattern
287              )
288              self.SetModifiedFlag(0)
289          else:
290              self.flist = glob.glob(self.dp[0] + "\\" + self.fplist[0])
291              win32ui.GetApp().AddIdleHandler(self.SearchFile)
292  
293      def SearchFile(self, handler, count):
294          self.fndx = self.fndx + 1
295          if self.fndx < len(self.flist):
296              f = self.flist[self.fndx]
297              if self.verbose:
298                  self.GetFirstView().Append("# .." + f + "\n")
299              # Directories may match the file type pattern, and files may be removed
300              #  while grep is running
301              if os.path.isfile(f):
302                  win32ui.SetStatusText("Searching " + f, 0)
303                  lines = open(f, "r").readlines()
304                  for i in range(len(lines)):
305                      line = lines[i]
306                      if self.pat.search(line) != None:
307                          self.GetFirstView().Append(f + "(" + repr(i + 1) + ") " + line)
308          else:
309              self.fndx = -1
310              self.fpndx = self.fpndx + 1
311              if self.fpndx < len(self.fplist):
312                  self.flist = glob.glob(
313                      self.dp[self.dpndx] + "\\" + self.fplist[self.fpndx]
314                  )
315              else:
316                  self.fpndx = 0
317                  self.dpndx = self.dpndx + 1
318                  if self.dpndx < len(self.dp):
319                      self.flist = glob.glob(
320                          self.dp[self.dpndx] + "\\" + self.fplist[self.fpndx]
321                      )
322                  else:
323                      win32ui.SetStatusText("Search complete.", 0)
324                      self.SetModifiedFlag(0)  # default to not modified.
325                      try:
326                          win32ui.GetApp().DeleteIdleHandler(self.SearchFile)
327                      except:
328                          pass
329                      return 0
330          return 1
331  
332      def GetParams(self):
333          return (
334              self.dirpattern
335              + "\t"
336              + self.filpattern
337              + "\t"
338              + self.greppattern
339              + "\t"
340              + repr(self.casesensitive)
341              + "\t"
342              + repr(self.recurse)
343              + "\t"
344              + repr(self.verbose)
345          )

347      def OnSaveDocument(self, filename):
348          #       print 'OnSaveDocument() filename=',filename
349          savefile = open(filename, "wb")
350          txt = self.GetParams() + "\n"
351          #       print 'writing',txt
352          savefile.write(txt)
353          savefile.close()
354          self.SetModifiedFlag(0)
355          return 1

384      def OnLDblClick(self, params):
385          line = self.GetLine()
386          regexGrepResult = regexGrep.match(line)
387          if regexGrepResult:
388              fname = regexGrepResult.group(1)
389              line = int(regexGrepResult.group(2))
390              scriptutils.JumpToDocument(fname, line)
391              return 0  # dont pass on
392          return 1  # pass it on by default.
393  
394      def OnRClick(self, params):
395          menu = win32ui.CreatePopupMenu()
396          flags = win32con.MF_STRING | win32con.MF_ENABLED
397          lineno = self._obj_.LineFromChar(-1)  # selection or current line
398          line = self._obj_.GetLine(lineno)
399          regexGrepResult = regexGrep.match(line)
400          if regexGrepResult:
401              self.fnm = regexGrepResult.group(1)
402              self.lnnum = int(regexGrepResult.group(2))
403              menu.AppendMenu(flags, ID_OPEN_FILE, "&Open " + self.fnm)
404              menu.AppendMenu(win32con.MF_SEPARATOR)
405          menu.AppendMenu(flags, ID_TRYAGAIN, "&Try Again")
406          charstart, charend = self._obj_.GetSel()
407          if charstart != charend:
408              linestart = self._obj_.LineIndex(lineno)
409              self.sel = line[charstart - linestart : charend - linestart]
410              menu.AppendMenu(flags, ID_GREP, "&Grep for " + self.sel)
411              menu.AppendMenu(win32con.MF_SEPARATOR)
412          menu.AppendMenu(flags, win32ui.ID_EDIT_CUT, "Cu&t")
413          menu.AppendMenu(flags, win32ui.ID_EDIT_COPY, "&Copy")
414          menu.AppendMenu(flags, win32ui.ID_EDIT_PASTE, "&Paste")
415          menu.AppendMenu(flags, win32con.MF_SEPARATOR)
416          menu.AppendMenu(flags, win32ui.ID_EDIT_SELECT_ALL, "&Select all")
417          menu.AppendMenu(flags, win32con.MF_SEPARATOR)
418          menu.AppendMenu(flags, ID_SAVERESULTS, "Sa&ve results")
419          menu.TrackPopupMenu(params[5])
420          return 0

425              vw = doc.GetFirstView()
426              # hope you have an editor that implements GotoLine()!
427              try:
428                  vw.GotoLine(int(self.lnnum))
429              except:
430                  pass
431          return 0

602      def getMore(self, section, key):
603          self.UpdateData(1)
604          # get the items out of the ini file
605          ini = win32ui.GetProfileFileName()
606          secitems = win32api.GetProfileSection(section, ini)
607          items = []
608          for secitem in secitems:
609              items.append(secitem.split("=")[1])
610          dlg = GrepParamsDialog(items)
611          if dlg.DoModal() == win32con.IDOK:
612              itemstr = ";".join(dlg.getItems())
613              self._obj_.data[key] = itemstr

615              i = 0
616              newitems = dlg.getNew()
617              if newitems:
618                  items = items + newitems
619                  for item in items:
620                      win32api.WriteProfileVal(section, repr(i), item, ini)
621                      i = i + 1
622              self.UpdateData(0)
