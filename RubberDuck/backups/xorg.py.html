11   from pygments.lexer import RegexLexer, bygroups
12   from pygments.token import Comment, String, Name, Text
13   
14   __all__ = ['XorgLexer']
15   
16   
17   class XorgLexer(RegexLexer):
18       """Lexer for xorg.conf file."""
19       name = 'Xorg'
20       aliases = ['xorg.conf']
21       filenames = ['xorg.conf']
22       mimetypes = []
23   
24       tokens = {
25           'root': [
26               (r'\s+', Text),
27               (r'#.*$', Comment),
28   
29               (r'((?:Sub)?Section)(\s+)("\w+")',
30                bygroups(String.Escape, Text, String.Escape)),
31               (r'(End(?:Sub)?Section)', String.Escape),
32   
33               (r'(\w+)(\s+)([^\n#]+)',
34                bygroups(Name.Builtin, Text, Name.Constant)),
35           ],
36       }
