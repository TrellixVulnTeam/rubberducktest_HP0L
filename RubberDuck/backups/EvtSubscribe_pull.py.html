EvtSubscribe_pull.py 
1    ## Demonstrates how to create a "pull" subscription
2    import win32evtlog, win32event, win32con
3    
4    query_text = '*[System[Provider[@Name="Microsoft-Windows-Winlogon"]]]'
5    
6    h = win32event.CreateEvent(None, 0, 0, None)
7    s = win32evtlog.EvtSubscribe(
8        "System",
9        win32evtlog.EvtSubscribeStartAtOldestRecord,
10       SignalEvent=h,
11       Query=query_text,
12   )
13   
14   while 1:
15       while 1:
16           events = win32evtlog.EvtNext(s, 10)
17           if len(events) == 0:
18               break
19           ##for event in events:
20           ##  print(win32evtlog.EvtRender(event, win32evtlog.EvtRenderEventXml))
21           print("retrieved %s events" % len(events))
22       while 1:
23           print("waiting...")
24           w = win32event.WaitForSingleObjectEx(h, 2000, True)
25           if w == win32con.WAIT_OBJECT_0:
26               break
