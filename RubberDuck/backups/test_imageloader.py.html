49   def bytes_per_pixel(fmt):
50       if fmt in ('rgb', 'bgr'):
51           return 3
52       if fmt in ('rgba', 'bgra', 'argb', 'abgr'):
53           return 4
54       raise Exception('bytes_per_pixel: unknown format {}'.format(fmt))

57   def get_pixel_alpha(pix, fmt):
58       if fmt in ('rgba', 'bgra'):
59           return pix[3]
60       elif fmt in ('abgr', 'argb'):
61           return pix[0]
62       return 0xFF

65   # Converts (predicted) rgba pixels to the format claimed by image loader
66   def rgba_to(pix_in, target_fmt, w, h, pitch=None):
67       if not isinstance(pix_in, (bytes, bytearray)):
68           pix_in = bytearray(pix_in)
69       assert w > 0 and h > 0, "Must specify w and h"
70       assert len(pix_in) == w * h * 4, "Invalid rgba data {}".format(pix_in)
71       assert target_fmt in ('rgba', 'bgra', 'argb', 'abgr', 'rgb', 'bgr')
72   
73       if target_fmt == 'rgba':
74           return pix_in
75   
76       pixels = [pix_in[i:i + 4] for i in range(0, len(pix_in), 4)]
77       if target_fmt == 'bgra':
78           return b''.join([bytes(p[:3][::-1] + p[3:]) for p in pixels])
79       elif target_fmt == 'abgr':
80           return b''.join([bytes(p[3:] + p[:3][::-1]) for p in pixels])
81       elif target_fmt == 'argb':
82           return b''.join([bytes(p[3:] + p[:3]) for p in pixels])
83   
84       # rgb/bgr, default to 4 byte alignment
85       if pitch is None:
86           pitch = ((3 * w) + 3) & ~3
87       # Assume pitch 0 == unaligned
88       elif pitch == 0:
89           pitch = 3 * w
90   
91       out = b''
92       padding = b'\x00' * (pitch - w * 3)
93       for row in [pix_in[i:i + w * 4] for i in range(0, len(pix_in), w * 4)]:
94           pixelrow = [row[i:i + 4] for i in range(0, len(row), 4)]
95           if target_fmt == 'rgb':
96               out += b''.join([bytes(p[:3]) for p in pixelrow])
97           elif target_fmt == 'bgr':
98               out += b''.join([bytes(p[:3][::-1]) for p in pixelrow])
99           out += padding
100      return out

103  def match_prediction(pixels, fmt, fd, pitch):
104      assert len(fd['alpha']) == 2
105      assert len(fd['pattern']) > 0
106  
107      bpp = bytes_per_pixel(fmt)
108      rowlen = fd['w'] * bpp
109      if pitch is None:
110          pitch = (rowlen + 3) & ~3
111      elif pitch == 0:
112          pitch = fd['w'] * bpp
113      pitchalign = pitch - rowlen
114  
115      errors = []
116      fail = errors.append
117  
118      if len(pixels) != pitch * fd['h']:
119          fail("Pitch errror: pitch {} * {} height != {} pixelbytes"
120               .format(pitch, fd['h'], len(pixels)))
121  
122      ptr = 0
123      pixnum = 0
124      for char in fd['pattern']:
125          pix = list(bytearray(pixels[ptr:ptr + bpp]))
126  #        print("PIXNUM {} ptr={} bpp={} : {}".format(pixnum, ptr, bpp, pix))
127          if len(pix) != bpp:
128              fail("Want {} bytes per pixel, got {}: {}"
129                   .format(bpp, len(pix), pix))
130              break
131  
132          if char == 't':
133              if get_pixel_alpha(pix, fmt) != 0:
134                  fail("pixel {} nonzero 't' pixel alpha {:02X}: {}".format(
135                        pixnum, get_pixel_alpha(pix, fmt), pix))
136          else:
137              srcpix = v0_PIXELS[char] + list(bytearray.fromhex(fd['alpha']))
138              predict = rgba_to(srcpix, fmt, 1, 1, pitch=0)
139              predict = list(bytearray(predict))
140              if not predict or not pix or predict != pix:
141                  fail("pixel {} {} format mismatch: want {} ({}) -- got {}"
142                       .format(pixnum, fmt, predict, char, pix))
143  
144          if pitchalign and (pixnum + 1) % fd['w'] == 0:
145              check = list(bytearray(pixels[ptr + bpp:ptr + bpp + pitchalign]))
146              if check != [0] * pitchalign:
147                  fail("Want {} 0x00 pitch align pixnum={}, pos={} got: {}"
148                       .format(pitchalign, pixnum, ptr + bpp, check))
149              ptr += pitchalign
150          ptr += bpp
151          pixnum += 1
152  
153      if ptr != len(pixels):
154          fail("Excess data: pixnum={} ptr={} bytes={}, bpp={} pitchalign={}"
155               .format(pixnum, ptr, len(pixels), bpp, pitchalign))
156      return (len(errors) == 0, errors)

159  class _TestContext(object):
160      def __init__(self, loadercls):
161          self.loadercls = loadercls
162          self._fd = None
163          self._fn = None
164          self._ok = 0
165          self._skip = 0
166          self._fail = 0
167          self._stats = defaultdict(dict)
168  
169      @property
170      def stats(self):
171          return self._stats
172  
173      @property
174      def results(self):
175          return (self._ok, self._skip, self._fail, self._stats)
176  
177      def start(self, fn, fd):
178          assert not self._fn, "unexpected ctx.start(), already started"
179          assert isinstance(fd, dict)
180          self._fn = fn
181          self._fd = fd
182  
183      def end(self, fn=None):
184          assert not fn or self._fn == fn, "unexpected ctx.end(), fn mismatch"
185          self._fn = None
186          self._fd = None
187  
188      def ok(self, info):
189          assert self._fn, "unexpected ctx.ok(), fn=None"
190          self._ok += 1
191          self.dbg('PASS', info)
192          self._incstat('ok')
193          self.end(self._fn)
194  
195      def skip(self, info):
196          assert self._fn, "unexpected ctx.skip(), fn=None"
197          self._skip += 1
198          self.dbg('SKIP', info)
199          self._incstat('skip')
200          self.end(self._fn)
201  
202      def fail(self, info):
203          assert self._fn, "unexpected ctx.fail(), fn=None"
204          self._fail += 1
205          self.dbg('FAIL', info)
206          self._incstat('fail')
207          self.end(self._fn)
208  
209      def dbg(self, msgtype, info):
210          assert self._fn, "unexpected ctx.dbg(), fn=None"
211          if DEBUG:
212              print("{} {} {}: {}"
213                    .format(self.loadercls.__name__, msgtype, self._fn, info))
214  
215      def _incstat(self, s):
216          assert self._fd, "unexpected ctx._incstat(), fd=None"
217          fd = self._fd
218  
219          def IS(key):
220              self._stats.setdefault(s, defaultdict(int))[key] += 1
221  
222          IS('total')
223          IS('extension:{}'.format(fd['ext']))
224          IS('encoder:{}'.format(fd['encoder']))
225          IS('fmtinfo:{}'.format(fd['fmtinfo']))
226          IS('testname:{}'.format(fd['testname']))
227          IS('testname+ext:{}+{}'.format(fd['testname'], fd['ext']))
228          IS('encoder+ext:{}+{}'.format(fd['encoder'], fd['ext']))
229          IS('encoder+testname:{}+{}'.format(fd['encoder'], fd['testname']))
230          IS('fmtinfo+ext:{}+{}'.format(fd['fmtinfo'], fd['ext']))

235  class ImageLoaderTestCase(unittest.TestCase):
236      def setUp(self):

240      def tearDown(self):
241          if not DEBUG or not self._context:
242              return
243          ctx = self._context
244          il = ctx.loadercls.__name__
245          stats = ctx.stats
246          keys = set([k for x in stats.values() for k in x.keys()])
247          sg = stats.get
248          for k in sorted(keys):
249              ok, skip, fail = sg('ok', {}), sg('skip', {}), sg('fail', {})
250              print("REPORT {} {}: ok={}, skip={}, fail={}".format(
251                    il, k, ok.get(k, 0), skip.get(k, 0), fail.get(k, 0)))

253      def _prepare_images(self):
254          if hasattr(self, '_image_files'):
255              return
256          self._image_files = {}
257          for filename in os.listdir(asset(ASSETDIR)):
258              matches = v0_FILE_RE.match(filename)
259              if not matches:
260                  continue
261              w, h, pat, alpha, fmtinfo, tst, encoder, ext = matches.groups()
262              self._image_files[filename] = {
263                  'filename': filename,
264                  'w': int(w),
265                  'h': int(h),
266                  'pattern': pat,
267                  'alpha': alpha,
268                  'fmtinfo': fmtinfo,
269                  'testname': tst,
270                  'encoder': encoder,
271                  'ext': ext,
272                  'require_alpha': 'BINARY' in tst or 'ALPHA' in tst,
273              }
274  
275      def _test_imageloader(self, loadercls, extensions=None):
276          if not loadercls:
277              return
278          if not extensions:
279              extensions = loadercls.extensions()
280  
281          ctx = _TestContext(loadercls)
282          self._context = ctx
283          for filename in sorted(self._image_files.keys()):
284              filedata = self._image_files[filename]
285  
286              if filedata['ext'] not in extensions:
287                  continue
288              try:
289                  ctx.start(filename, filedata)
290                  result = loadercls(asset(ASSETDIR, filename), keep_data=True)
291                  if not result:
292                      raise Exception('invalid result')
293              except:
294                  ctx.skip('Error loading file, result=None')
295                  continue
296              self._test_image(filedata, ctx, loadercls, result)
297              ctx.end()
298  
299          ok, skip, fail, stats = ctx.results
300          if fail:
301              self.fail('{}: {} passed, {} skipped, {} failed'
302                        .format(loadercls.__name__, ok, skip, fail))
303          return ctx

305      def _test_image(self, fd, ctx, loadercls, imgdata):
306          w, h, pixels, pitch = imgdata._data[0].get_mipmap(0)
307          fmt = imgdata._data[0].fmt
308  
309          # required for FFPy memview
310          # FIXME: bytearray() for py2 compat, I can't be bothered to research
311          if not isinstance(pixels, bytes):
312              pixels = bytearray(pixels)
313  
314          def debug():
315              if not DEBUG:
316                  return
317              print("    format: {}x{} {}".format(w, h, fmt))
318              print("     pitch: got {}, want {}".format(pitch, want_pitch))
319              print("      want: {} in {}".format(fd['pattern'], fmt))
320              print("       got: {}".format(bytearray(pixels)))
321  
322          # Assume pitch 0 = unaligned
323          want_pitch = (pitch == 0) and bytes_per_pixel(fmt) * w or pitch
324          if pitch == 0 and bytes_per_pixel(fmt) * w * h != len(pixels):
325              ctx.dbg("PITCH", "pitch=0, expected fmt={} to be "
326                               "unaligned @ ({}bpp) = {} bytes, got {}"
327                               .format(fmt, bytes_per_pixel(fmt),
328                                       bytes_per_pixel(fmt) * w * h,
329                                       len(pixels)))
330          elif pitch and want_pitch != pitch:
331              ctx.dbg("PITCH", "fmt={}, pitch={}, expected {}"
332                               .format(fmt, pitch, want_pitch))
333  
334          success, msgs = match_prediction(pixels, fmt, fd, pitch)
335          if not success:
336              if not msgs:
337                  ctx.fail("Unknown error")
338              elif len(msgs) == 1:
339                  ctx.fail(msgs[0])
340              else:
341                  for m in msgs:
342                      ctx.dbg('PREDICT', m)
343                  ctx.fail('{} errors, see debug output: {}'
344                           .format(len(msgs), msgs[-1]))
345              debug()
346          elif fd['require_alpha'] and not has_alpha(fmt):
347              ctx.fail('Missing expected alpha channel')
348              debug()
349          elif fd['w'] != w:
350              ctx.fail('Width mismatch, want {} got {}'
351                       .format(fd['w'], w))
352              debug()
353          elif fd['h'] != h:
354              ctx.fail('Height mismatch, want {} got {}'
355                       .format(fd['h'], h))
356              debug()
357          elif w != 1 and h != 1:
358              ctx.fail('v0 test protocol mandates w=1 or h=1')
359              debug()
360          else:
361              ctx.ok("Passed test as {}x{} {}".format(w, h, fmt))
362          sys.stdout.flush()
