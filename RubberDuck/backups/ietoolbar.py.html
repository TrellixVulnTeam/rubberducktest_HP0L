70   class WIN32STRUCT:
71       def __init__(self, **kw):
72           full_fmt = ""
73           for name, fmt, default in self._struct_items_:
74               self.__dict__[name] = None
75               if fmt == "z":
76                   full_fmt += "pi"
77               else:
78                   full_fmt += fmt
79           for name, val in kw.items():
80               self.__dict__[name] = val
81   
82       def __setattr__(self, attr, val):
83           if not attr.startswith("_") and attr not in self.__dict__:
84               raise AttributeError(attr)
85           self.__dict__[attr] = val
86   
87       def toparam(self):
88           self._buffs = []
89           full_fmt = ""
90           vals = []
91           for name, fmt, default in self._struct_items_:
92               val = self.__dict__[name]
93               if fmt == "z":
94                   fmt = "Pi"
95                   if val is None:
96                       vals.append(0)
97                       vals.append(0)
98                   else:
99                       str_buf = array.array("c", val + "\0")
100                      vals.append(str_buf.buffer_info()[0])
101                      vals.append(len(val))
102                      self._buffs.append(str_buf)  # keep alive during the call.
103              else:
104                  if val is None:
105                      val = default
106                  vals.append(val)
107              full_fmt += fmt
108          return struct.pack(*(full_fmt,) + tuple(vals))

137  class IEToolbarCtrl:
138      """ 
139      a tiny wrapper for our winapi-based 
140      toolbar control implementation. 
141      """
142  
143      def __init__(self, hwndparent):
144          styles = (
145              win32con.WS_CHILD
146              | win32con.WS_VISIBLE
147              | win32con.WS_CLIPSIBLINGS
148              | win32con.WS_CLIPCHILDREN
149              | commctrl.TBSTYLE_LIST
150              | commctrl.TBSTYLE_FLAT
151              | commctrl.TBSTYLE_TRANSPARENT
152              | commctrl.CCS_TOP
153              | commctrl.CCS_NODIVIDER
154              | commctrl.CCS_NORESIZE
155              | commctrl.CCS_NOPARENTALIGN
156          )
157          self.hwnd = win32gui.CreateWindow(
158              "ToolbarWindow32",
159              None,
160              styles,
161              0,
162              0,
163              100,
164              100,
165              hwndparent,
166              0,
167              win32gui.dllhandle,
168              None,
169          )
170          win32gui.SendMessage(self.hwnd, commctrl.TB_BUTTONSTRUCTSIZE, 20, 0)
171  
172      def ShowWindow(self, mode):
173          win32gui.ShowWindow(self.hwnd, mode)
174  
175      def AddButtons(self, *buttons):
176          tbbuttons = ""
177          for button in buttons:
178              tbbuttons += button.toparam()
179          return win32gui.SendMessage(
180              self.hwnd, commctrl.TB_ADDBUTTONS, len(buttons), tbbuttons
181          )
182  
183      def GetSafeHwnd(self):
184          return self.hwnd

188      """ 
189      The actual COM server class 
190      """
191  
192      _com_interfaces_ = _ietoolbar_com_interfaces_
193      _public_methods_ = _ietoolbar_methods_
194      _reg_clsctx_ = pythoncom.CLSCTX_INPROC_SERVER
195      # if you copy and modify this example, be sure to change the clsid below
196      _reg_clsid_ = "{F21202A2-959A-4149-B1C3-68B9013F3335}"
197      _reg_progid_ = "PyWin32.IEToolbar"
198      _reg_desc_ = "PyWin32 IE Toolbar"
199  
200      def __init__(self):
201          # put stubs for non-implemented methods
202          for method in self._public_methods_:
203              if not hasattr(self, method):
204                  print("providing default stub for %s" % method)
205                  setattr(self, method, Stub(method))
206  
207      def GetWindow(self):
208          return self.toolbar.GetSafeHwnd()
209  
210      def Load(self, stream):
211          # called when the toolbar is loaded
212          pass
213  
214      def Save(self, pStream, fClearDirty):
215          # called when the toolbar shall save its information
216          pass
217  
218      def CloseDW(self, dwReserved):
219          del self.toolbar
220  
221      def ShowDW(self, bShow):
222          if bShow:
223              self.toolbar.ShowWindow(win32con.SW_SHOW)
224          else:
225              self.toolbar.ShowWindow(win32con.SW_HIDE)
226  
227      def on_first_button(self):
228          print("first!")
229          self.webbrowser.Navigate2("http://starship.python.net/crew/mhammond/")
230  
231      def on_second_button(self):
232          print("second!")
233  
234      def on_third_button(self):
235          print("third!")
236  
237      def toolbar_command_handler(self, args):
238          hwnd, message, wparam, lparam, time, point = args
239          if lparam == self.toolbar.GetSafeHwnd():
240              self._command_map[wparam]()
241  
242      def SetSite(self, unknown):
243          if unknown:
244              # retrieve the parent window interface for this site
245              olewindow = unknown.QueryInterface(pythoncom.IID_IOleWindow)
246              # ask the window for its handle
247              hwndparent = olewindow.GetWindow()
248  
249              # first get a command target
250              cmdtarget = unknown.QueryInterface(axcontrol.IID_IOleCommandTarget)
251              # then travel over to a service provider
252              serviceprovider = cmdtarget.QueryInterface(pythoncom.IID_IServiceProvider)
253              # finally ask for the internet explorer application, returned as a dispatch object
254              self.webbrowser = win32com.client.Dispatch(
255                  serviceprovider.QueryService(
256                      "{0002DF05-0000-0000-C000-000000000046}", pythoncom.IID_IDispatch
257                  )
258              )
259  
260              # now create and set up the toolbar
261              self.toolbar = IEToolbarCtrl(hwndparent)
262  
263              buttons = [
264                  ("Visit PyWin32 Homepage", self.on_first_button),
265                  ("Another Button", self.on_second_button),
266                  ("Yet Another Button", self.on_third_button),
267              ]
268  
269              self._command_map = {}
270              # wrap our parent window so we can hook message handlers
271              window = win32ui.CreateWindowFromHandle(hwndparent)
272  
273              # add the buttons
274              for i in range(len(buttons)):
275                  button = TBBUTTON()
276                  name, func = buttons[i]
277                  id = 0x4444 + i
278                  button.iBitmap = -2
279                  button.idCommand = id
280                  button.fsState = commctrl.TBSTATE_ENABLED
281                  button.fsStyle = commctrl.TBSTYLE_BUTTON
282                  button.iString = name
283                  self._command_map[0x4444 + i] = func
284                  self.toolbar.AddButtons(button)
285                  window.HookMessage(self.toolbar_command_handler, win32con.WM_COMMAND)
286          else:
287              # lose all references
288              self.webbrowser = None
289

318  def DllRegisterServer():
319      comclass = IEToolbar
320  
321      # register toolbar with IE
322      try:
323          print("Trying to register Toolbar.\n")
324          hkey = winreg.CreateKey(
325              winreg.HKEY_LOCAL_MACHINE, "SOFTWARE\\Microsoft\\Internet Explorer\\Toolbar"
326          )
327          subKey = winreg.SetValueEx(
328              hkey, comclass._reg_clsid_, 0, winreg.REG_BINARY, "\0"
329          )
330      except WindowsError:
331          print(
332              "Couldn't set registry value.\nhkey: %d\tCLSID: %s\n"
333              % (hkey, comclass._reg_clsid_)
334          )
335      else:
336          print(
337              "Set registry value.\nhkey: %d\tCLSID: %s\n" % (hkey, comclass._reg_clsid_)
338          )

343  def DllUnregisterServer():
344      comclass = IEToolbar
345  
346      # unregister toolbar from internet explorer
347      try:
348          print("Trying to unregister Toolbar.\n")
349          hkey = winreg.CreateKey(
350              winreg.HKEY_LOCAL_MACHINE, "SOFTWARE\\Microsoft\\Internet Explorer\\Toolbar"
351          )
352          winreg.DeleteValue(hkey, comclass._reg_clsid_)

354          print(
355              "Couldn't delete registry value.\nhkey: %d\tCLSID: %s\n"
356              % (hkey, comclass._reg_clsid_)

359          print("Deleting reg key succeeded.\n")
