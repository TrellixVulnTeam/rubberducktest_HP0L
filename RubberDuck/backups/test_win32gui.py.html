18   class TestPyGetMemory(unittest.TestCase):
19       def test_ob(self):
20           # Check the PyGetMemory result and a bytes string can be compared
21           test_data = b"\0\1\2\3\4\5\6"
22           c = array.array("b", test_data)
23           addr, buflen = c.buffer_info()
24           got = win32gui.PyGetMemory(addr, buflen)
25           self.assertEqual(len(got), len(test_data))
26           self.assertEqual(bytes(got), test_data)
27   
28       def test_memory_index(self):
29           # Check we can index into the buffer object returned by PyGetMemory
30           test_data = b"\0\1\2\3\4\5\6"
31           c = array.array("b", test_data)
32           addr, buflen = c.buffer_info()
33           got = win32gui.PyGetMemory(addr, buflen)
34           self.assertEqual(got[0], 0)
35   
36       def test_memory_slice(self):
37           # Check we can slice the buffer object returned by PyGetMemory
38           test_data = b"\0\1\2\3\4\5\6"
39           c = array.array("b", test_data)
40           addr, buflen = c.buffer_info()
41           got = win32gui.PyGetMemory(addr, buflen)
42           self.assertEqual(list(got[0:3]), [0, 1, 2])
43   
44       def test_real_view(self):
45           # Do the PyGetMemory, then change the original memory, then ensure
46           # the initial object we fetched sees the new value.
47           test_data = b"\0\1\2\3\4\5\6"
48           c = array.array("b", test_data)
49           addr, buflen = c.buffer_info()
50           got = win32gui.PyGetMemory(addr, buflen)
51           self.assertEqual(got[0], 0)
52           c[0] = 1
53           self.assertEqual(got[0], 1)
54   
55       def test_memory_not_writable(self):
56           # Check the buffer object fetched by PyGetMemory isn't writable.
57           test_data = b"\0\1\2\3\4\5\6"
58           c = array.array("b", test_data)
59           addr, buflen = c.buffer_info()
60           got = win32gui.PyGetMemory(addr, buflen)
61           self.assertRaises(TypeError, operator.setitem, got, 0, 1)
