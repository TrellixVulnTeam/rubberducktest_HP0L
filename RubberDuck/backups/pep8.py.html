135  def tabs_or_spaces(physical_line, indent_char):
136      r"""Never mix tabs and spaces. 
137   
138      The most popular way of indenting Python is with spaces only.  The 
139      second-most popular way is with tabs only.  Code indented with a mixture 
140      of tabs and spaces should be converted to using spaces exclusively.  When 
141      invoking the Python command line interpreter with the -t option, it issues 
142      warnings about code that illegally mixes tabs and spaces.  When using -tt 
143      these warnings become errors.  These options are highly recommended! 
144   
145      Okay: if a == 0:\n        a = 1\n        b = 1 
146      E101: if a == 0:\n        a = 1\n\tb = 1 
147      """
148      indent = INDENT_REGEX.match(physical_line).group(1)
149      for offset, char in enumerate(indent):
150          if char != indent_char:
151              return offset, "E101 indentation contains mixed spaces and tabs"

165  def trailing_whitespace(physical_line):
166      r"""Trailing whitespace is superfluous. 
167   
168      The warning returned varies on whether the line itself is blank, for easier 
169      filtering for those who want to indent their blank lines. 
170   
171      Okay: spam(1)\n# 
172      W291: spam(1) \n# 
173      W293: class Foo(object):\n    \n    bang = 12 
174      """
175      physical_line = physical_line.rstrip('\n')    # chr(10), newline
176      physical_line = physical_line.rstrip('\r')    # chr(13), carriage return
177      physical_line = physical_line.rstrip('\x0c')  # chr(12), form feed, ^L
178      stripped = physical_line.rstrip(' \t\v')
179      if physical_line != stripped:
180          if stripped:
181              return len(stripped), "W291 trailing whitespace"
182          else:
183              return 0, "W293 blank line contains whitespace"

186  def trailing_blank_lines(physical_line, lines, line_number, total_lines):
187      r"""Trailing blank lines are superfluous. 
188   
189      Okay: spam(1) 
190      W391: spam(1)\n 
191   
192      However the last line should end with a new line (warning W292). 
193      """
194      if line_number == total_lines:
195          stripped_last_line = physical_line.rstrip()
196          if not stripped_last_line:
197              return 0, "W391 blank line at end of file"
198          if stripped_last_line == physical_line:
199              return len(physical_line), "W292 no newline at end of file"

202  def maximum_line_length(physical_line, max_line_length, multiline, noqa):
203      r"""Limit all lines to a maximum of 79 characters. 
204   
205      There are still many devices around that are limited to 80 character 
206      lines; plus, limiting windows to 80 characters makes it possible to have 
207      several windows side-by-side.  The default wrapping on such devices looks 
208      ugly.  Therefore, please limit all lines to a maximum of 79 characters. 
209      For flowing long blocks of text (docstrings or comments), limiting the 
210      length to 72 characters is recommended. 
211   
212      Reports error E501. 
213      """
214      line = physical_line.rstrip()
215      length = len(line)
216      if length > max_line_length and not noqa:
217          # Special case for long URLs in multi-line docstrings or comments,
218          # but still report the error when the 72 first chars are whitespaces.
219          chunks = line.split()
220          if ((len(chunks) == 1 and multiline) or
221              (len(chunks) == 2 and chunks[0] == '#')) and \
222                  len(line) - len(chunks[-1]) < max_line_length - 7:
223              return
224          if hasattr(line, 'decode'):   # Python 2
225              # The line could contain multi-byte characters
226              try:
227                  length = len(line.decode('utf-8'))
228              except UnicodeError:
229                  pass
230          if length > max_line_length:
231              return (max_line_length, "E501 line too long "
232                      "(%d > %d characters)" % (length, max_line_length))

240  def blank_lines(logical_line, blank_lines, indent_level, line_number,
241                  blank_before, previous_logical,
242                  previous_unindented_logical_line, previous_indent_level,
243                  lines):
244      r"""Separate top-level function and class definitions with two blank lines. 
245   
246      Method definitions inside a class are separated by a single blank line. 
247   
248      Extra blank lines may be used (sparingly) to separate groups of related 
249      functions.  Blank lines may be omitted between a bunch of related 
250      one-liners (e.g. a set of dummy implementations). 
251   
252      Use blank lines in functions, sparingly, to indicate logical sections. 
253   
254      Okay: def a():\n    pass\n\n\ndef b():\n    pass 
255      Okay: def a():\n    pass\n\n\nasync def b():\n    pass 
256      Okay: def a():\n    pass\n\n\n# Foo\n# Bar\n\ndef b():\n    pass 
257      Okay: default = 1\nfoo = 1 
258      Okay: classify = 1\nfoo = 1 
259   
260      E301: class Foo:\n    b = 0\n    def bar():\n        pass 
261      E302: def a():\n    pass\n\ndef b(n):\n    pass 
262      E302: def a():\n    pass\n\nasync def b(n):\n    pass 
263      E303: def a():\n    pass\n\n\n\ndef b(n):\n    pass 
264      E303: def a():\n\n\n\n    pass 
265      E304: @decorator\n\ndef a():\n    pass 
266      E305: def a():\n    pass\na() 
267      """
268      if line_number < 3 and not previous_logical:
269          return  # Don't expect blank lines before the first line
270      if previous_logical.startswith('@'):
271          if blank_lines:
272              yield 0, "E304 blank lines found after function decorator"
273      elif blank_lines > 2 or (indent_level and blank_lines == 2):
274          yield 0, "E303 too many blank lines (%d)" % blank_lines
275      elif logical_line.startswith(('def ', 'async def', 'class ', '@')):
276          if indent_level:
277              if not (blank_before or previous_indent_level < indent_level or
278                      DOCSTRING_REGEX.match(previous_logical)):
279                  ancestor_level = indent_level
280                  nested = False
281                  # Search backwards for a def ancestor or tree root (top level).
282                  for line in lines[line_number - 2::-1]:
283                      if line.strip() and expand_indent(line) < ancestor_level:
284                          ancestor_level = expand_indent(line)
285                          nested = line.lstrip().startswith('def ')
286                          if nested or ancestor_level == 0:
287                              break
288                  if nested:
289                      yield 0, "E306 expected 1 blank line before a " \
290                          "nested definition, found 0"
291                  else:
292                      yield 0, "E301 expected 1 blank line, found 0"
293          elif blank_before != 2:
294              yield 0, "E302 expected 2 blank lines, found %d" % blank_before
295      elif (logical_line and not indent_level and blank_before != 2 and
296            previous_unindented_logical_line.startswith(('def ', 'class '))):
297          yield 0, "E305 expected 2 blank lines after " \
298              "class or function definition, found %d" % blank_before

301  def extraneous_whitespace(logical_line):
302      r"""Avoid extraneous whitespace. 
303   
304      Avoid extraneous whitespace in these situations: 
305      - Immediately inside parentheses, brackets or braces. 
306      - Immediately before a comma, semicolon, or colon. 
307   
308      Okay: spam(ham[1], {eggs: 2}) 
309      E201: spam( ham[1], {eggs: 2}) 
310      E201: spam(ham[ 1], {eggs: 2}) 
311      E201: spam(ham[1], { eggs: 2}) 
312      E202: spam(ham[1], {eggs: 2} ) 
313      E202: spam(ham[1 ], {eggs: 2}) 
314      E202: spam(ham[1], {eggs: 2 }) 
315   
316      E203: if x == 4: print x, y; x, y = y , x 
317      E203: if x == 4: print x, y ; x, y = y, x 
318      E203: if x == 4 : print x, y; x, y = y, x 
319      """
320      line = logical_line
321      for match in EXTRANEOUS_WHITESPACE_REGEX.finditer(line):
322          text = match.group()
323          char = text.strip()
324          found = match.start()
325          if text == char + ' ':
326              # assert char in '([{'
327              yield found + 1, "E201 whitespace after '%s'" % char
328          elif line[found - 1] != ',':
329              code = ('E202' if char in '}])' else 'E203')  # if char in ',;:'
330              yield found, "%s whitespace before '%s'" % (code, char)

333  def whitespace_around_keywords(logical_line):
334      r"""Avoid extraneous whitespace around keywords. 
335   
336      Okay: True and False 
337      E271: True and  False 
338      E272: True  and False 
339      E273: True and\tFalse 
340      E274: True\tand False 
341      """
342      for match in KEYWORD_REGEX.finditer(logical_line):
343          before, after = match.groups()
344  
345          if '\t' in before:
346              yield match.start(1), "E274 tab before keyword"
347          elif len(before) > 1:
348              yield match.start(1), "E272 multiple spaces before keyword"
349  
350          if '\t' in after:
351              yield match.start(2), "E273 tab after keyword"
352          elif len(after) > 1:
353              yield match.start(2), "E271 multiple spaces after keyword"

356  def missing_whitespace_after_import_keyword(logical_line):
357      r"""Multiple imports in form from x import (a, b, c) should have space 
358      between import statement and parenthesised name list. 
359   
360      Okay: from foo import (bar, baz) 
361      E275: from foo import(bar, baz) 
362      E275: from importable.module import(bar, baz) 
363      """
364      line = logical_line
365      indicator = ' import('
366      if line.startswith('from '):
367          found = line.find(indicator)
368          if -1 < found:
369              pos = found + len(indicator) - 1
370              yield pos, "E275 missing whitespace after keyword"

373  def missing_whitespace(logical_line):
374      r"""Each comma, semicolon or colon should be followed by whitespace. 
375   
376      Okay: [a, b] 
377      Okay: (3,) 
378      Okay: a[1:4] 
379      Okay: a[:4] 
380      Okay: a[1:] 
381      Okay: a[1:4:2] 
382      E231: ['a','b'] 
383      E231: foo(bar,baz) 
384      E231: [{'a':'b'}] 
385      """
386      line = logical_line
387      for index in range(len(line) - 1):
388          char = line[index]
389          if char in ',;:' and line[index + 1] not in WHITESPACE:
390              before = line[:index]
391              if char == ':' and before.count('[') > before.count(']') and \
392                      before.rfind('{') < before.rfind('['):
393                  continue  # Slice syntax, no space required
394              if char == ',' and line[index + 1] == ')':
395                  continue  # Allow tuple with only one element: (3,)
396              yield index, "E231 missing whitespace after '%s'" % char

399  def indentation(logical_line, previous_logical, indent_char,
400                  indent_level, previous_indent_level):
401      r"""Use 4 spaces per indentation level. 
402   
403      For really old code that you don't want to mess up, you can continue to 
404      use 8-space tabs. 
405   
406      Okay: a = 1 
407      Okay: if a == 0:\n    a = 1 
408      E111:   a = 1 
409      E114:   # a = 1 
410   
411      Okay: for item in items:\n    pass 
412      E112: for item in items:\npass 
413      E115: for item in items:\n# Hi\n    pass 
414   
415      Okay: a = 1\nb = 2 
416      E113: a = 1\n    b = 2 
417      E116: a = 1\n    # b = 2 
418      """
419      c = 0 if logical_line else 3
420      tmpl = "E11%d %s" if logical_line else "E11%d %s (comment)"
421      if indent_level % 4:
422          yield 0, tmpl % (1 + c, "indentation is not a multiple of four")
423      indent_expect = previous_logical.endswith(':')
424      if indent_expect and indent_level <= previous_indent_level:
425          yield 0, tmpl % (2 + c, "expected an indented block")
426      elif not indent_expect and indent_level > previous_indent_level:
427          yield 0, tmpl % (3 + c, "unexpected indentation")

430  def continued_indentation(logical_line, tokens, indent_level, hang_closing,
431                            indent_char, noqa, verbose):
432      r"""Continuation lines indentation. 
433   
434      Continuation lines should align wrapped elements either vertically 
435      using Python's implicit line joining inside parentheses, brackets 
436      and braces, or using a hanging indent. 
437   
438      When using a hanging indent these considerations should be applied: 
439      - there should be no arguments on the first line, and 
440      - further indentation should be used to clearly distinguish itself as a 
441        continuation line. 
442   
443      Okay: a = (\n) 
444      E123: a = (\n    ) 
445   
446      Okay: a = (\n    42) 
447      E121: a = (\n   42) 
448      E122: a = (\n42) 
449      E123: a = (\n    42\n    ) 
450      E124: a = (24,\n     42\n) 
451      E125: if (\n    b):\n    pass 
452      E126: a = (\n        42) 
453      E127: a = (24,\n      42) 
454      E128: a = (24,\n    42) 
455      E129: if (a or\n    b):\n    pass 
456      E131: a = (\n    42\n 24) 
457      """
458      first_row = tokens[0][2][0]
459      nrows = 1 + tokens[-1][2][0] - first_row
460      if noqa or nrows == 1:
461          return
462  
463      # indent_next tells us whether the next block is indented; assuming
464      # that it is indented by 4 spaces, then we should not allow 4-space
465      # indents on the final continuation line; in turn, some other
466      # indents are allowed to have an extra 4 spaces.
467      indent_next = logical_line.endswith(':')
468  
469      row = depth = 0
470      valid_hangs = (4,) if indent_char != '\t' else (4, 8)
471      # remember how many brackets were opened on each line
472      parens = [0] * nrows
473      # relative indents of physical lines
474      rel_indent = [0] * nrows
475      # for each depth, collect a list of opening rows
476      open_rows = [[0]]
477      # for each depth, memorize the hanging indentation
478      hangs = [None]
479      # visual indents
480      indent_chances = {}
481      last_indent = tokens[0][2]
482      visual_indent = None
483      last_token_multiline = False
484      # for each depth, memorize the visual indent column
485      indent = [last_indent[1]]
486      if verbose >= 3:
487          print(">>> " + tokens[0][4].rstrip())
488  
489      for token_type, text, start, end, line in tokens:
490  
491          newline = row < start[0] - first_row
492          if newline:
493              row = start[0] - first_row
494              newline = not last_token_multiline and token_type not in NEWLINE
495  
496          if newline:
497              # this is the beginning of a continuation line.
498              last_indent = start
499              if verbose >= 3:
500                  print("... " + line.rstrip())
501  
502              # record the initial indent.
503              rel_indent[row] = expand_indent(line) - indent_level
504  
505              # identify closing bracket
506              close_bracket = (token_type == tokenize.OP and text in ']})')
507  
508              # is the indent relative to an opening bracket line?
509              for open_row in reversed(open_rows[depth]):
510                  hang = rel_indent[row] - rel_indent[open_row]
511                  hanging_indent = hang in valid_hangs
512                  if hanging_indent:
513                      break
514              if hangs[depth]:
515                  hanging_indent = (hang == hangs[depth])
516              # is there any chance of visual indent?
517              visual_indent = (not close_bracket and hang > 0 and
518                               indent_chances.get(start[1]))
519  
520              if close_bracket and indent[depth]:
521                  # closing bracket for visual indent
522                  if start[1] != indent[depth]:
523                      yield (start, "E124 closing bracket does not match "
524                             "visual indentation")
525              elif close_bracket and not hang:
526                  # closing bracket matches indentation of opening bracket's line
527                  if hang_closing:
528                      yield start, "E133 closing bracket is missing indentation"
529              elif indent[depth] and start[1] < indent[depth]:
530                  if visual_indent is not True:
531                      # visual indent is broken
532                      yield (start, "E128 continuation line "
533                             "under-indented for visual indent")
534              elif hanging_indent or (indent_next and rel_indent[row] == 8):
535                  # hanging indent is verified
536                  if close_bracket and not hang_closing:
537                      yield (start, "E123 closing bracket does not match "
538                             "indentation of opening bracket's line")
539                  hangs[depth] = hang
540              elif visual_indent is True:
541                  # visual indent is verified
542                  indent[depth] = start[1]
543              elif visual_indent in (text, str):
544                  # ignore token lined up with matching one from a previous line
545                  pass
546              else:
547                  # indent is broken
548                  if hang <= 0:
549                      error = "E122", "missing indentation or outdented"
550                  elif indent[depth]:
551                      error = "E127", "over-indented for visual indent"
552                  elif not close_bracket and hangs[depth]:
553                      error = "E131", "unaligned for hanging indent"
554                  else:
555                      hangs[depth] = hang
556                      if hang > 4:
557                          error = "E126", "over-indented for hanging indent"
558                      else:
559                          error = "E121", "under-indented for hanging indent"
560                  yield start, "%s continuation line %s" % error
561  
562          # look for visual indenting
563          if (parens[row] and
564                  token_type not in (tokenize.NL, tokenize.COMMENT) and
565                  not indent[depth]):
566              indent[depth] = start[1]
567              indent_chances[start[1]] = True
568              if verbose >= 4:
569                  print("bracket depth %s indent to %s" % (depth, start[1]))
570          # deal with implicit string concatenation
571          elif (token_type in (tokenize.STRING, tokenize.COMMENT) or
572                text in ('u', 'ur', 'b', 'br')):
573              indent_chances[start[1]] = str
574          # special case for the "if" statement because len("if (") == 4
575          elif not indent_chances and not row and not depth and text == 'if':
576              indent_chances[end[1] + 1] = True
577          elif text == ':' and line[end[1]:].isspace():

1344 def mute_string(text):
1345     """Replace contents with 'xxx' to prevent syntax matching. 
1346

1364 def parse_udiff(diff, patterns=None, parent='.'):
1365     """Return a dictionary of matching lines."""
1366     # For each file of the diff, the entry key is the filename,
1367     # and the value is a set of row numbers to consider.

1370     for line in diff.splitlines():
1371         if nrows:
1372             if line[:1] != '-':
1373                 nrows -= 1
1374             continue
1375         if line[:3] == '@@ ':
1376             hunk_match = HUNK_REGEX.match(line)
1377             (row, nrows) = [int(g or '1') for g in hunk_match.groups()]
1378             rv[path].update(range(row, row + nrows))
1379         elif line[:3] == '+++':
1380             path = line[4:].split('\t', 1)[0]
1381             if path[:2] == 'b/':
1382                 path = path[2:]
1383             rv[path] = set()
1384     return dict([(os.path.join(parent, path), rows)
1385                  for (path, rows) in rv.items()
1386                  if rows and filename_match(path, patterns)])

1670     def maybe_check_physical(self, token):
1671         """If appropriate (based on token), check current physical line(s)."""
1672         # Called after every token, but act only on end of line.
1673         if _is_eol_token(token):
1674             # Obviously, a newline token ends a single physical line.
1675             self.check_physical(token[4])
1676         elif token[0] == tokenize.STRING and '\n' in token[1]:
1677             # Less obviously, a string that contains newlines is a
1678             # multiline string, either triple-quoted or with internal
1679             # newlines backslash-escaped. Check every physical line in the
1680             # string *except* for the last one: its newline is outside of
1681             # the multiline string, so we consider it a regular physical
1682             # line, and will check it like any other physical line.
1683             #
1684             # Subtleties:
1685             # - we don't *completely* ignore the last line; if it contains
1686             #   the magical "# noqa" comment, we disable all physical
1687             #   checks for the entire multiline string
1688             # - have to wind self.line_number back because initially it
1689             #   points to the last line of the string, and we want
1690             #   check_physical() to give accurate feedback
1691             if noqa(token[4]):
1692                 return

1906                 'row': self.line_offset + line_number, 'col': offset + 1,

2005     def input_file(self, filename, lines=None, expected=None, line_offset=0):

2013     def input_dir(self, dirname):
2014         """Check all files in this directory and all subdirectories."""
2015         dirname = dirname.rstrip('/')
2016         if self.excluded(dirname):
2017             return 0
2018         counters = self.options.report.counters
2019         verbose = self.options.verbose
2020         filepatterns = self.options.filename
2021         runner = self.runner
2022         for root, dirs, files in os.walk(dirname):
2023             if verbose:
2024                 print('directory ' + root)
2025             counters['directories'] += 1

2027                 if self.excluded(subdir, root):
2028                     dirs.remove(subdir)

2037

2041             return False

2043         if filename_match(basename, self.options.exclude):
2044             return True

2046             filename = os.path.join(parent, filename)
2047         filename = os.path.abspath(filename)

2050     def ignore_code(self, code):
2051         """Check if the error code should be ignored. 
2052

2055         the error code, return True.

2057         if len(code) < 4 and any(s.startswith(code)
2058                                  for s in self.options.select):
