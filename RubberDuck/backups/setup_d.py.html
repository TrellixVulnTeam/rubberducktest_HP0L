13   def usage_and_die(rc):
14       print()
15       print("This script is designed to copy and register the Python debug")
16       print("binaries.  It looks for pythonxx_d.dll, pythoncomxx_d.dll etc,")
17       print("and installs them to work correctly with Python debug builds.")
18       print()
19       print("You will generally find this script in the. zip file that")
20       print("included these _d files.  Please run this script from")
21       print("that directory")
22       sys.exit(rc)

26       print("This scripts appears to be running a DEBUG version of Python.")
27       print("Please run it using a normal release build (python.exe)")
28       usage_and_die(1)

47   def _docopy(src, dest):
48       orig_src = src
49       if not os.path.isfile(src):
50           src = os.path.join(os.path.split(sys.argv[0])[0], src)
51           print(
52               "Can not find %s or %s to copy"
53               % (os.path.abspath(orig_src), os.path.abspath(src))
54           )
55           return 0
56       try:
57           shutil.copy(src, dest)
58           print("Copied %s -> %s" % (src, dest))
59           return 1
60       except:
61           print("Error copying '%s' -> '%s'" % (src, dest))
62           print(str(sys.exc_info[1]))
63           usage_and_die(3)

66   def _doregister(mod_name, dll_name):
67       assert os.path.isfile(dll_name), "Shouldn't get here if the file doesn't exist!"
68       try:
69           key = winreg.OpenKey(
70               winreg.HKEY_LOCAL_MACHINE,
71               "Software\\Python\\PythonCore\\%s\\Modules\\%s" % (sys.winver, mod_name),
72           )
73       except winreg.error:
74           try:
75               key = winreg.OpenKey(
76                   winreg.HKEY_LOCAL_MACHINE,
77                   "Software\\Python\\PythonCore\\%s\\Modules\\%s"
78                   % (sys.winver, mod_name),
79               )
80           except winreg.error:
81               print(
82                   "Could not find the existing '%s' module registered in the registry"
83                   % (mod_name,)
84               )
85               usage_and_die(4)
86       # Create the debug key.
87       sub_key = winreg.CreateKey(key, "Debug")
88       winreg.SetValue(sub_key, None, winreg.REG_SZ, dll_name)
89       print("Registered '%s' in the registry" % (dll_name,))

92   def _domodule(mod_name, release_mod_filename):
93       path, fname = os.path.split(release_mod_filename)
94       base, ext = os.path.splitext(fname)
95       new_fname = base + "_d" + ext
96       if _docopy(new_fname, path):
97           _doregister(mod_name, os.path.abspath(os.path.join(path, new_fname)))

100  # First the main Python DLL.
101  path, fname = path, fname = os.path.split(win32api.GetModuleFileName(sys.dllhandle))
102  base, ext = os.path.splitext(fname)
103  _docopy(base + "_d" + ext, path)
104  
105  # Then pythoncom and pywintypes.
106  _domodule("pythoncom", pythoncom.__file__)
107  _domodule("pywintypes", pywintypes.__file__)
108  
109  print("System _d files were setup.")
