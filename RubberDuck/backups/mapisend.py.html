1    #!/usr/bin/env python

3    """module to send mail with Extended MAPI using the pywin32 mapi wrappers..."""
4    
5    # this was based on Jason Hattingh's C++ code at http://www.codeproject.com/internet/mapadmin.asp
6    # written by David Fraser <davidf at sjsoft.com> and Stephen Emslie <stephene at sjsoft.com>
7    # you can test this by changing the variables at the bottom and running from the command line
8    
9    from win32com.mapi import mapi
10   from win32com.mapi import mapitags
11   
12   
13   def SendEMAPIMail(
14       Subject="", Message="", SendTo=None, SendCC=None, SendBCC=None, MAPIProfile=None
15   ):
16       """Sends an email to the recipient using the extended MAPI interface 
17       Subject and Message are strings 
18       Send{To,CC,BCC} are comma-separated address lists 
19       MAPIProfile is the name of the MAPI profile"""
20   
21       # initialize and log on
22       mapi.MAPIInitialize(None)
23       session = mapi.MAPILogonEx(
24           0, MAPIProfile, None, mapi.MAPI_EXTENDED | mapi.MAPI_USE_DEFAULT
25       )
26       messagestorestable = session.GetMsgStoresTable(0)
27       messagestorestable.SetColumns(
28           (mapitags.PR_ENTRYID, mapitags.PR_DISPLAY_NAME_A, mapitags.PR_DEFAULT_STORE), 0
29       )
30   
31       while True:
32           rows = messagestorestable.QueryRows(1, 0)
33           # if this is the last row then stop
34           if len(rows) != 1:
35               break
36           row = rows[0]
37           # if this is the default store then stop
38           if (mapitags.PR_DEFAULT_STORE, True) in row:
39               break
40   
41       # unpack the row and open the message store
42       (eid_tag, eid), (name_tag, name), (def_store_tag, def_store) = row
43       msgstore = session.OpenMsgStore(
44           0, eid, None, mapi.MDB_NO_DIALOG | mapi.MAPI_BEST_ACCESS
45       )
46   
47       # get the outbox
48       hr, props = msgstore.GetProps((mapitags.PR_IPM_OUTBOX_ENTRYID), 0)
49       (tag, eid) = props[0]
50       # check for errors
51       if mapitags.PROP_TYPE(tag) == mapitags.PT_ERROR:
52           raise TypeError("got PT_ERROR instead of PT_BINARY: %s" % eid)
53       outboxfolder = msgstore.OpenEntry(eid, None, mapi.MAPI_BEST_ACCESS)
54   
55       # create the message and the addrlist
56       message = outboxfolder.CreateMessage(None, 0)
57       # note: you can use the resolveaddress functions for this. but you may get headaches
58       pal = []
59   
60       def makeentry(recipient, recipienttype):
61           return (
62               (mapitags.PR_RECIPIENT_TYPE, recipienttype),
63               (mapitags.PR_SEND_RICH_INFO, False),
64               (mapitags.PR_DISPLAY_TYPE, 0),
65               (mapitags.PR_OBJECT_TYPE, 6),
66               (mapitags.PR_EMAIL_ADDRESS_A, recipient),
67               (mapitags.PR_ADDRTYPE_A, "SMTP"),
68               (mapitags.PR_DISPLAY_NAME_A, recipient),
69           )
70   
71       if SendTo:
72           pal.extend(
73               [makeentry(recipient, mapi.MAPI_TO) for recipient in SendTo.split(",")]
74           )
75       if SendCC:
76           pal.extend(
77               [makeentry(recipient, mapi.MAPI_CC) for recipient in SendCC.split(",")]
78           )
79       if SendBCC:
80           pal.extend(
81               [makeentry(recipient, mapi.MAPI_BCC) for recipient in SendBCC.split(",")]
82           )
83   
84       # add the resolved recipients to the message
85       message.ModifyRecipients(mapi.MODRECIP_ADD, pal)
86       message.SetProps([(mapitags.PR_BODY_A, Message), (mapitags.PR_SUBJECT_A, Subject)])
87   
88       # save changes and submit
89       outboxfolder.SaveChanges(0)
90       message.SubmitMessage(0)
91   
92   
93   if __name__ == "__main__":
94       MAPIProfile = ""
95       # Change this to a valid email address to test
96       SendTo = "an.invalid at address"
97       SendMessage = "testing one two three"
98       SendSubject = "Testing Extended MAPI!!"
99       SendEMAPIMail(SendSubject, SendMessage, SendTo, MAPIProfile=MAPIProfile)
