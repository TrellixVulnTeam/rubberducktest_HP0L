3    # Features:
4    # * Uses a "dynamic dialog resource" to build the dialog.
5    # * Uses a ListView control.
6    # * Dynamically resizes content.

8    # * Demostrates support for windows XP themes.

40   class _WIN32MASKEDSTRUCT:
41       def __init__(self, **kw):
42           full_fmt = ""
43           for name, fmt, default, mask in self._struct_items_:
44               self.__dict__[name] = None
45               if fmt == "z":
46                   full_fmt += "pi"
47               else:
48                   full_fmt += fmt
49           for name, val in kw.items():
50               if name not in self.__dict__:
51                   raise ValueError("LVITEM structures do not have an item '%s'" % (name,))
52               self.__dict__[name] = val
53   
54       def __setattr__(self, attr, val):
55           if not attr.startswith("_") and attr not in self.__dict__:
56               raise AttributeError(attr)
57           self.__dict__[attr] = val
58   
59       def toparam(self):
60           self._buffs = []
61           full_fmt = ""
62           vals = []
63           mask = 0
64           # calc the mask
65           for name, fmt, default, this_mask in self._struct_items_:
66               if this_mask is not None and self.__dict__.get(name) is not None:
67                   mask |= this_mask
68           self.mask = mask
69           for name, fmt, default, this_mask in self._struct_items_:
70               val = self.__dict__[name]
71               if fmt == "z":
72                   fmt = "Pi"
73                   if val is None:
74                       vals.append(0)
75                       vals.append(0)
76                   else:
77                       # Note this demo still works with byte strings.  An
78                       # alternate strategy would be to use unicode natively
79                       # and use the 'W' version of the messages - eg,
80                       # LVM_SETITEMW etc.
81                       val = val + "\0"
82                       if isinstance(val, str):
83                           val = val.encode("mbcs")
84                       str_buf = array.array("b", val)
85                       vals.append(str_buf.buffer_info()[0])
86                       vals.append(len(val))
87                       self._buffs.append(str_buf)  # keep alive during the call.
88               else:
89                   if val is None:
90                       val = default
91                   vals.append(val)
92               full_fmt += fmt
93           return struct.pack(*(full_fmt,) + tuple(vals))

124  class DemoWindowBase:
125      def __init__(self):
126          win32gui.InitCommonControls()
127          self.hinst = win32gui.dllhandle
128          self.list_data = {}

130      def _RegisterWndClass(self):
131          className = "PythonDocSearch"
132          message_map = {}
133          wc = win32gui.WNDCLASS()
134          wc.SetDialogProc()  # Make it a dialog class.
135          wc.hInstance = self.hinst
136          wc.lpszClassName = className
137          wc.style = win32con.CS_VREDRAW | win32con.CS_HREDRAW
138          wc.hCursor = win32gui.LoadCursor(0, win32con.IDC_ARROW)
139          wc.hbrBackground = win32con.COLOR_WINDOW + 1
140          wc.lpfnWndProc = message_map  # could also specify a wndproc.
141          # C code: wc.cbWndExtra = DLGWINDOWEXTRA + sizeof(HBRUSH) + (sizeof(COLORREF));
142          wc.cbWndExtra = win32con.DLGWINDOWEXTRA + struct.calcsize("Pi")
143          icon_flags = win32con.LR_LOADFROMFILE | win32con.LR_DEFAULTSIZE
144  
145          ## py.ico went away in python 2.5, load from executable instead
146          this_app = win32api.GetModuleHandle(None)
147          try:
148              wc.hIcon = win32gui.LoadIcon(this_app, 1)  ## python.exe and pythonw.exe
149          except win32gui.error:
150              wc.hIcon = win32gui.LoadIcon(this_app, 135)  ## pythonwin's icon
151          try:
152              classAtom = win32gui.RegisterClass(wc)
153          except win32gui.error as err_info:
154              if err_info.winerror != winerror.ERROR_CLASS_ALREADY_EXISTS:
155                  raise
156          return className

211      def _DoCreate(self, fn):
212          message_map = {
213              win32con.WM_SIZE: self.OnSize,
214              win32con.WM_COMMAND: self.OnCommand,
215              win32con.WM_NOTIFY: self.OnNotify,
216              win32con.WM_INITDIALOG: self.OnInitDialog,
217              win32con.WM_CLOSE: self.OnClose,
218              win32con.WM_DESTROY: self.OnDestroy,
219              WM_SEARCH_RESULT: self.OnSearchResult,
220              WM_SEARCH_FINISHED: self.OnSearchFinished,
221          }
222          dlgClassName = self._RegisterWndClass()
223          template = self._GetDialogTemplate(dlgClassName)
224          return fn(self.hinst, template, 0, message_map)

226      def _SetupList(self):
227          child_style = (
228              win32con.WS_CHILD
229              | win32con.WS_VISIBLE
230              | win32con.WS_BORDER
231              | win32con.WS_HSCROLL
232              | win32con.WS_VSCROLL
233          )
234          child_style |= (
235              commctrl.LVS_SINGLESEL | commctrl.LVS_SHOWSELALWAYS | commctrl.LVS_REPORT
236          )
237          self.hwndList = win32gui.CreateWindow(
238              "SysListView32",
239              None,
240              child_style,
241              0,
242              0,
243              100,
244              100,
245              self.hwnd,
246              IDC_LISTBOX,
247              self.hinst,
248              None,
249          )
250  
251          child_ex_style = win32gui.SendMessage(
252              self.hwndList, commctrl.LVM_GETEXTENDEDLISTVIEWSTYLE, 0, 0
253          )
254          child_ex_style |= commctrl.LVS_EX_FULLROWSELECT
255          win32gui.SendMessage(
256              self.hwndList, commctrl.LVM_SETEXTENDEDLISTVIEWSTYLE, 0, child_ex_style
257          )
258  
259          # Add an image list - use the builtin shell folder icon - this
260          # demonstrates the problem with alpha-blending of icons on XP if
261          # winxpgui is not used in place of win32gui.
262          il = win32gui.ImageList_Create(
263              win32api.GetSystemMetrics(win32con.SM_CXSMICON),
264              win32api.GetSystemMetrics(win32con.SM_CYSMICON),
265              commctrl.ILC_COLOR32 | commctrl.ILC_MASK,
266              1,  # initial size
267              0,
268          )  # cGrow
269  
270          shell_dll = os.path.join(win32api.GetSystemDirectory(), "shell32.dll")
271          large, small = win32gui.ExtractIconEx(shell_dll, 4, 1)
272          win32gui.ImageList_ReplaceIcon(il, -1, small[0])
273          win32gui.DestroyIcon(small[0])
274          win32gui.DestroyIcon(large[0])
275          win32gui.SendMessage(
276              self.hwndList, commctrl.LVM_SETIMAGELIST, commctrl.LVSIL_SMALL, il
277          )
278  
279          # Setup the list control columns.
280          lvc = LVCOLUMN(
281              mask=commctrl.LVCF_FMT
282              | commctrl.LVCF_WIDTH
283              | commctrl.LVCF_TEXT
284              | commctrl.LVCF_SUBITEM
285          )
286          lvc.fmt = commctrl.LVCFMT_LEFT
287          lvc.iSubItem = 1
288          lvc.text = "Title"
289          lvc.cx = 200
290          win32gui.SendMessage(self.hwndList, commctrl.LVM_INSERTCOLUMN, 0, lvc.toparam())
291          lvc.iSubItem = 0
292          lvc.text = "Order"
293          lvc.cx = 50
294          win32gui.SendMessage(self.hwndList, commctrl.LVM_INSERTCOLUMN, 0, lvc.toparam())
295  
296          win32gui.UpdateWindow(self.hwnd)

298      def ClearListItems(self):
299          win32gui.SendMessage(self.hwndList, commctrl.LVM_DELETEALLITEMS)
300          self.list_data = {}
301  
302      def AddListItem(self, data, *columns):
303          num_items = win32gui.SendMessage(self.hwndList, commctrl.LVM_GETITEMCOUNT)
304          item = LVITEM(text=columns[0], iItem=num_items)
305          new_index = win32gui.SendMessage(
306              self.hwndList, commctrl.LVM_INSERTITEM, 0, item.toparam()
307          )
308          col_no = 1
309          for col in columns[1:]:
310              item = LVITEM(text=col, iItem=new_index, iSubItem=col_no)
311              win32gui.SendMessage(self.hwndList, commctrl.LVM_SETITEM, 0, item.toparam())
312              col_no += 1
313          self.list_data[new_index] = data

315      def OnInitDialog(self, hwnd, msg, wparam, lparam):
316          self.hwnd = hwnd
317          # centre the dialog
318          desktop = win32gui.GetDesktopWindow()
319          l, t, r, b = win32gui.GetWindowRect(self.hwnd)
320          dt_l, dt_t, dt_r, dt_b = win32gui.GetWindowRect(desktop)
321          centre_x, centre_y = win32gui.ClientToScreen(
322              desktop, ((dt_r - dt_l) // 2, (dt_b - dt_t) // 2)
323          )
324          win32gui.MoveWindow(
325              hwnd, centre_x - (r // 2), centre_y - (b // 2), r - l, b - t, 0
326          )
327          self._SetupList()
328          l, t, r, b = win32gui.GetClientRect(self.hwnd)
329          self._DoSize(r - l, b - t, 1)

331      def _DoSize(self, cx, cy, repaint=1):
332          # right-justify the textbox.
333          ctrl = win32gui.GetDlgItem(self.hwnd, IDC_SEARCHTEXT)
334          l, t, r, b = win32gui.GetWindowRect(ctrl)
335          l, t = win32gui.ScreenToClient(self.hwnd, (l, t))
336          r, b = win32gui.ScreenToClient(self.hwnd, (r, b))
337          win32gui.MoveWindow(ctrl, l, t, cx - l - 5, b - t, repaint)
338          # The button.
339          ctrl = win32gui.GetDlgItem(self.hwnd, IDC_BUTTON_DISPLAY)
340          l, t, r, b = win32gui.GetWindowRect(ctrl)
341          l, t = win32gui.ScreenToClient(self.hwnd, (l, t))
342          r, b = win32gui.ScreenToClient(self.hwnd, (r, b))
343          list_y = b + 10
344          w = r - l
345          win32gui.MoveWindow(ctrl, cx - 5 - w, t, w, b - t, repaint)
346  
347          # The list control
348          win32gui.MoveWindow(self.hwndList, 0, list_y, cx, cy - list_y, repaint)
349          # The last column of the list control.
350          new_width = cx - win32gui.SendMessage(
351              self.hwndList, commctrl.LVM_GETCOLUMNWIDTH, 0
352          )
353          win32gui.SendMessage(self.hwndList, commctrl.LVM_SETCOLUMNWIDTH, 1, new_width)
354  
355      def OnSize(self, hwnd, msg, wparam, lparam):
356          x = win32api.LOWORD(lparam)
357          y = win32api.HIWORD(lparam)
358          self._DoSize(x, y)
359          return 1

361      def OnSearchResult(self, hwnd, msg, wparam, lparam):
362          try:
363              while 1:
364                  params = self.result_queue.get(0)
365                  self.AddListItem(*params)
366          except queue.Empty:
367              pass

369      def OnSearchFinished(self, hwnd, msg, wparam, lparam):

372      def OnNotify(self, hwnd, msg, wparam, lparam):
373          info = win32gui_struct.UnpackNMITEMACTIVATE(lparam)
374          if info.code == commctrl.NM_DBLCLK:
375              print("Double click on item", info.iItem + 1)
376          return 1

378      def OnCommand(self, hwnd, msg, wparam, lparam):
379          id = win32api.LOWORD(wparam)
380          if id == IDC_BUTTON_SEARCH:
381              self.ClearListItems()
382  
383              def fill_slowly(q, hwnd):
384                  import time
385  
386                  for i in range(20):
387                      q.put(("whatever", str(i + 1), "Search result " + str(i)))
388                      win32gui.PostMessage(hwnd, WM_SEARCH_RESULT, 0, 0)
389                      time.sleep(0.25)
390                  win32gui.PostMessage(hwnd, WM_SEARCH_FINISHED, 0, 0)
391  
392              import threading
393  
394              self.result_queue = queue.Queue()
395              thread = threading.Thread(
396                  target=fill_slowly, args=(self.result_queue, self.hwnd)
397              )
398              thread.start()
399          elif id == IDC_BUTTON_DISPLAY:
400              print("Display button selected")
401              sel = win32gui.SendMessage(
402                  self.hwndList, commctrl.LVM_GETNEXTITEM, -1, commctrl.LVNI_SELECTED
403              )
404              print("The selected item is", sel + 1)

414  # An implementation suitable for use with the Win32 Window functions (ie, not
415  # a true dialog)
416  class DemoWindow(DemoWindowBase):
417      def CreateWindow(self):
418          # Create the window via CreateDialogBoxIndirect - it can then
419          # work as a "normal" window, once a message loop is established.
420          self._DoCreate(win32gui.CreateDialogIndirect)
421  
422      def OnClose(self, hwnd, msg, wparam, lparam):
423          win32gui.DestroyWindow(hwnd)
424  
425      # We need to arrange to a WM_QUIT message to be sent to our
426      # PumpMessages() loop.
427      def OnDestroy(self, hwnd, msg, wparam, lparam):
428          win32gui.PostQuitMessage(0)  # Terminate the app.
