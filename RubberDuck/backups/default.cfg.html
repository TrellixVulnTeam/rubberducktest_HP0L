9    # Simply create a new .CFG file, and add an entry:
10   # [General]

14   # from the Pythonwin View/Options/Editor dialog.

135  # Add a simple file/class/function simple banner
136  def AddBanner(editor_window, event): 
137  
138  	text = editor_window.text 
139  	big_line = "#" * 70 
140  	banner = "%s\n## \n## \n## \n%s\n" % (big_line, big_line) 
141  
142  	# Insert at the start of the current line.
143  	pos = text.index("insert linestart") 
144  
145  	text.undo_block_start() # Allow action to be undone as a single unit.
146  	text.insert(pos, banner) 
147  	text.undo_block_stop() 
148  
149  	# Now set the insert point to the middle of the banner.
150  	line, col = [int(s) for s in pos.split(".")] 
151  	text.mark_set("insert", "%d.1 lineend" % (line+2, ) )

159  def InteractiveHomeExtend(editor_window, event): 
160  	return _DoInteractiveHome(editor_window.text, 1)

162  def _DoInteractiveHome(text, extend): 
163  	import sys 
164  	# If Scintilla has an autocomplete window open, then let Scintilla handle it.
165  	if text.edit.SCIAutoCActive(): 
166  		return 1 
167  	of_interest = "insert linestart + %d c" % len(sys.ps1) 
168  	if not text.compare("insert", "==", of_interest) and \ 
169  	   text.get("insert linestart", of_interest) in [sys.ps1, sys.ps2]: # Not sys.ps? line
170  		end = of_interest 
171  	else: 
172  		end = "insert linestart"
173  
174  	if extend: start = "insert"
175  	else: start = end 
176  	text.tag_add("sel", start, end)

179  def AutoFindNext(editor_window, event): 
180      "find selected text or word under cursor"
181  
182      from pywin.scintilla import find 
183      from pywin.scintilla import scintillacon 
184  
185      try: 
186          sci = editor_window.edit 
187          word = sci.GetSelText() 
188          if word: 
189              find.lastSearch.findText = word 
190              find.lastSearch.sel = sci.GetSel() 
191          else: 
192              pos = sci.SendScintilla( scintillacon.SCI_GETCURRENTPOS ) 
193              start = sci.SendScintilla( scintillacon.SCI_WORDSTARTPOSITION, pos, 1 ) 
194              end = sci.SendScintilla( scintillacon.SCI_WORDENDPOSITION, pos, 1 ) 
195              word = sci.GetTextRange( start, end ) 
196              if word: 
197                  find.lastSearch.findText = word 
198                  find.lastSearch.sel = (start,end) 
199      except Exception: 
200          import traceback 
201          traceback.print_exc() 
202      find.FindNext()
