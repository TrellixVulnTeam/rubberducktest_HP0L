43   def DispExTest(ob):
44       if not __debug__:
45           print("WARNING: Tests dressed up as assertions are being skipped!")
46       assert ob.GetDispID("Add", 0) == 10, "Policy did not honour the dispid"
47       # Not impl
48       #       assert ob.GetMemberName(10, 0)=="add", "Policy did not give me the correct function for the dispid"
49       assert ob.GetDispID("Remove", 0) == 11, "Policy did not honour the dispid"
50       assert ob.GetDispID("In", 0) == 1000, "Allocated dispid unexpected value"
51       assert (
52           ob.GetDispID("_NewEnum", 0) == pythoncom.DISPID_NEWENUM
53       ), "_NewEnum() got unexpected DISPID"
54       dispids = []
55       dispid = -1
56       while 1:
57           try:
58               dispid = ob.GetNextDispID(0, dispid)
59               dispids.append(dispid)
60           except pythoncom.com_error as xxx_todo_changeme:
61               (hr, desc, exc, arg) = xxx_todo_changeme.args
62               assert hr == winerror.S_FALSE, "Bad result at end of enum"
63               break
64       dispids.sort()
65       if dispids != [pythoncom.DISPID_EVALUATE, pythoncom.DISPID_NEWENUM, 10, 11, 1000]:
66           raise Error("Got back the wrong dispids: %s" % dispids)
