1    ''' 
2    Inspector 
3    ========= 
4     
5    .. versionadded:: 1.0.9 
6     
7    .. warning:: 
8     
9        This module is highly experimental, use it with care. 
10    
11   The Inspector is a tool for finding a widget in the widget tree by clicking or 
12   tapping on it. 
13   Some keyboard shortcuts are activated: 
14    
15       * "Ctrl + e": activate / deactivate the inspector view 
16       * "Escape": cancel widget lookup first, then hide the inspector view 
17    
18   Available inspector interactions: 
19    
20       * tap once on a widget to select it without leaving inspect mode 
21       * double tap on a widget to select and leave inspect mode (then you can 
22         manipulate the widget again) 
23    
24   Some properties can be edited live. However, due to the delayed usage of 
25   some properties, it might crash if you don't handle all the cases. 
26    
27   Usage 
28   ----- 
29    
30   For normal module usage, please see the :mod:`~kivy.modules` documentation. 
31    
32   The Inspector, however, can also be imported and used just like a normal 
33   python module. This has the added advantage of being able to activate and 
34   deactivate the module programmatically:: 
35    
36       from kivy.core.window import Window 
37       from kivy.app import App 
38       from kivy.uix.button import Button 
39       from kivy.modules import inspector 
40    
41       class Demo(App): 
42           def build(self): 
43               button = Button(text="Test") 
44               inspector.create_inspector(Window, button) 
45               return button 
46    
47       Demo().run() 
48    
49   To remove the Inspector, you can do the following:: 
50    
51       inspector.stop(Window, button) 
52    
53   '''

266  class WidgetTree(TreeView):
267      selected_widget = ObjectProperty(None, allownone=True)
268  
269      __events__ = ('on_select_widget',)
270  
271      def __init__(self, **kwargs):
272          super(WidgetTree, self).__init__(**kwargs)
273          self.update_scroll = Clock.create_trigger(self._update_scroll)
274  
275      def find_node_by_widget(self, widget):
276          for node in self.iterate_all_nodes():
277              if not node.parent_node:
278                  continue
279              try:
280                  if node.widget == widget:
281                      return node
282              except ReferenceError:
283                  pass
284          return
285  
286      def update_selected_widget(self, widget):
287          if widget:
288              node = self.find_node_by_widget(widget)
289              if node:
290                  self.select_node(node, False)
291                  while node and isinstance(node, TreeViewWidget):
292                      if not node.is_open:
293                          self.toggle_node(node)
294                      node = node.parent_node
295  
296      def on_selected_widget(self, inst, widget):
297          if widget:
298              self.update_selected_widget(widget)
299              self.update_scroll()
300  
301      def select_node(self, node, select_widget=True):
302          super(WidgetTree, self).select_node(node)
303          if select_widget:
304              try:
305                  self.dispatch('on_select_widget', node.widget.__self__)
306              except ReferenceError:
307                  pass
308  
309      def on_select_widget(self, widget):
310          pass
311  
312      def _update_scroll(self, *args):
313          node = self._selected_node
314          if not node:
315              return
316  
317          self.parent.scroll_to(node)

320  class Inspector(FloatLayout):
321  
322      widget = ObjectProperty(None, allownone=True)
323  
324      layout = ObjectProperty(None)

326      widgettree = ObjectProperty(None)

328      treeview = ObjectProperty(None)

332      activated = BooleanProperty(False)

334      widget_info = BooleanProperty(False)

342      def __init__(self, **kwargs):
343          self.win = kwargs.pop('win', None)
344          super(Inspector, self).__init__(**kwargs)
345          self.avoid_bring_to_top = False
346          with self.canvas.before:
347              self.gcolor = Color(1, 0, 0, .25)
348              PushMatrix()
349              self.gtransform = Transform(Matrix())
350              self.grect = Rectangle(size=(0, 0))
351              PopMatrix()
352          Clock.schedule_interval(self.update_widget_graphics, 0)

354      def on_touch_down(self, touch):
355          ret = super(Inspector, self).on_touch_down(touch)
356          if (('button' not in touch.profile or touch.button == 'left') and
357                  not ret and self.inspect_enabled):
358              self.highlight_at(*touch.pos)
359              if touch.is_double_tap:
360                  self.inspect_enabled = False
361                  self.show_widget_info()
362              ret = True
363          return ret
364  
365      def on_touch_move(self, touch):
366          ret = super(Inspector, self).on_touch_move(touch)
367          if not ret and self.inspect_enabled:
368              self.highlight_at(*touch.pos)
369              ret = True
370          return ret
371  
372      def on_touch_up(self, touch):
373          ret = super(Inspector, self).on_touch_up(touch)
374          if not ret and self.inspect_enabled:
375              ret = True
376          return ret

378      def on_window_children(self, win, children):
379          if self.avoid_bring_to_top or not self.activated:
380              return
381          self.avoid_bring_to_top = True
382          win.remove_widget(self)
383          win.add_widget(self)
384          self.avoid_bring_to_top = False
385  
386      def highlight_at(self, x, y):
387          widget = None
388          # reverse the loop - look at children on top first and
389          # modalviews before others
390          win_children = self.win.children
391          children = chain(
392              (c for c in win_children if isinstance(c, ModalView)),
393              (c for c in reversed(win_children) if not isinstance(c, ModalView))
394          )
395          for child in children:
396              if child is self:
397                  continue
398              widget = self.pick(child, x, y)
399              if widget:
400                  break
401          self.highlight_widget(widget)
402  
403      def highlight_widget(self, widget, info=True, *largs):
404          # no widget to highlight, reduce rectangle to 0, 0
405          self.widget = widget
406          if not widget:
407              self.grect.size = 0, 0
408          if self.widget_info and info:
409              self.show_widget_info()
410  
411      def update_widget_graphics(self, *l):
412          if not self.activated:
413              return
414          if self.widget is None:
415              self.grect.size = 0, 0
416              return
417          self.grect.size = self.widget.size
418          matrix = self.widget.get_window_matrix()
419          if self.gtransform.matrix.get() != matrix.get():
420              self.gtransform.matrix = matrix
421  
422      def toggle_position(self, button):
423          to_bottom = button.text == 'Move to Bottom'
424  
425          if to_bottom:
426              button.text = 'Move to Top'
427              if self.widget_info:
428                  Animation(top=250, t='out_quad', d=.3).start(self.layout)
429              else:
430                  Animation(top=60, t='out_quad', d=.3).start(self.layout)
431  
432              bottom_bar = self.layout.children[1]
433              self.layout.remove_widget(bottom_bar)
434              self.layout.add_widget(bottom_bar)
435          else:
436              button.text = 'Move to Bottom'
437              if self.widget_info:
438                  Animation(top=self.height, t='out_quad', d=.3).start(
439                      self.layout)
440              else:
441                  Animation(y=self.height - 60, t='out_quad', d=.3).start(
442                      self.layout)
443  
444              bottom_bar = self.layout.children[1]
445              self.layout.remove_widget(bottom_bar)
446              self.layout.add_widget(bottom_bar)
447          self.at_bottom = to_bottom
448  
449      def pick(self, widget, x, y):
450          ret = None
451          # try to filter widgets that are not visible (invalid inspect target)
452          if (hasattr(widget, 'visible') and not widget.visible):
453              return ret
454          if widget.collide_point(x, y):
455              ret = widget
456              x2, y2 = widget.to_local(x, y)
457              # reverse the loop - look at children on top first
458              for child in reversed(widget.children):
459                  ret = self.pick(child, x2, y2) or ret
460          return ret
461  
462      def on_activated(self, instance, activated):
463          if not activated:
464              self.grect.size = 0, 0
465              if self.at_bottom:
466                  anim = Animation(top=0, t='out_quad', d=.3)
467              else:
468                  anim = Animation(y=self.height, t='out_quad', d=.3)
469              anim.bind(on_complete=self.animation_close)
470              anim.start(self.layout)
471              self.widget = None
472              self.widget_info = False
473          else:
474              self.win.add_widget(self)
475              Logger.info('Inspector: inspector activated')
476              if self.at_bottom:
477                  Animation(top=60, t='out_quad', d=.3).start(self.layout)
478              else:
479                  Animation(y=self.height - 60, t='out_quad', d=.3).start(
480                      self.layout)
481              ev = self._update_widget_tree_ev
482              if ev is None:
483                  ev = self._update_widget_tree_ev = Clock.schedule_interval(
484                      self.update_widget_tree, 1)
485              else:
486                  ev()
487              self.update_widget_tree()
488  
489      def animation_close(self, instance, value):
490          if not self.activated:
491              self.inspect_enabled = False
492              self.win.remove_widget(self)
493              self.content.clear_widgets()
494              treeview = self.treeview
495              for node in list(treeview.iterate_all_nodes()):
496                  node.widget_ref = None
497                  treeview.remove_node(node)
498  
499              self._window_node = None
500              if self._update_widget_tree_ev is not None:
501                  self._update_widget_tree_ev.cancel()
502  
503              widgettree = self.widgettree
504              for node in list(widgettree.iterate_all_nodes()):
505                  widgettree.remove_node(node)
506              Logger.info('Inspector: inspector deactivated')
507  
508      def show_widget_info(self):
509          self.content.clear_widgets()
510          widget = self.widget
511          treeview = self.treeview
512          for node in list(treeview.iterate_all_nodes())[:]:
513              node.widget_ref = None
514              treeview.remove_node(node)
515          if not widget:
516              if self.at_bottom:
517                  Animation(top=60, t='out_quad', d=.3).start(self.layout)
518              else:
519                  Animation(y=self.height - 60, t='out_quad', d=.3).start(
520                      self.layout)
521              self.widget_info = False
522              return
523          self.widget_info = True
524          if self.at_bottom:
525              Animation(top=250, t='out_quad', d=.3).start(self.layout)
526          else:
527              Animation(top=self.height, t='out_quad', d=.3).start(self.layout)
528          for node in list(treeview.iterate_all_nodes())[:]:
529              treeview.remove_node(node)
530  
531          keys = list(widget.properties().keys())
532          keys.sort()
533          node = None
534          if type(widget) is WeakProxy:
535              wk_widget = widget.__ref__
536          else:
537              wk_widget = weakref.ref(widget)
538          for key in keys:
539              node = TreeViewProperty(key=key, widget_ref=wk_widget)
540              node.bind(is_selected=self.show_property)
541              try:
542                  widget.bind(**{key: partial(
543                      self.update_node_content, weakref.ref(node))})
544              except:
545                  pass
546              treeview.add_node(node)

548      def update_node_content(self, node, *l):
549          node = node()
550          if node is None:
551              return
552          node.refresh = True
553          node.refresh = False

555      def keyboard_shortcut(self, win, scancode, *largs):
556          modifiers = largs[-1]
557          if scancode == 101 and set(modifiers) & {'ctrl'} and not set(
558                  modifiers) & {'shift', 'alt', 'meta'}:
559              self.activated = not self.activated
560              if self.activated:
561                  self.inspect_enabled = True
562              return True
563          elif scancode == 27:
564              if self.inspect_enabled:
565                  self.inspect_enabled = False
566                  return True
567              if self.activated:
568                  self.activated = False
569                  return True

571      def show_property(self, instance, value, key=None, index=-1, *l):
572          # normal call: (tree node, focus, )
573          # nested call: (widget, prop value, prop key, index in dict/list)
574          if value is False:
575              return
576  
577          content = None
578          if key is None:
579              # normal call
580              nested = False
581              widget = instance.widget
582              key = instance.key
583              prop = widget.property(key)
584              value = getattr(widget, key)
585          else:
586              # nested call, we might edit subvalue
587              nested = True
588              widget = instance
589              prop = None
590  
591          dtype = None
592  
593          if isinstance(prop, AliasProperty) or nested:
594              # trying to resolve type dynamically
595              if type(value) in (str, str):
596                  dtype = 'string'
597              elif type(value) in (int, float):
598                  dtype = 'numeric'
599              elif type(value) in (tuple, list):
600                  dtype = 'list'
601  
602          if isinstance(prop, NumericProperty) or dtype == 'numeric':
603              content = TextInput(text=str(value) or '', multiline=False)
604              content.bind(text=partial(
605                  self.save_property_numeric, widget, key, index))
606          elif isinstance(prop, StringProperty) or dtype == 'string':
607              content = TextInput(text=value or '', multiline=True)
608              content.bind(text=partial(
609                  self.save_property_text, widget, key, index))
610          elif (isinstance(prop, ListProperty) or
611                isinstance(prop, ReferenceListProperty) or
612                isinstance(prop, VariableListProperty) or
613                dtype == 'list'):
614              content = GridLayout(cols=1, size_hint_y=None)
615              content.bind(minimum_height=content.setter('height'))
616              for i, item in enumerate(value):
617                  button = Button(text=repr(item), size_hint_y=None, height=44)
618                  if isinstance(item, Widget):
619                      button.bind(on_release=partial(self.highlight_widget, item,
620                                                     False))
621                  else:
622                      button.bind(on_release=partial(self.show_property, widget,
623                                                     item, key, i))
624                  content.add_widget(button)
625          elif isinstance(prop, OptionProperty):
626              content = GridLayout(cols=1, size_hint_y=None)
627              content.bind(minimum_height=content.setter('height'))
628              for option in prop.options:
629                  button = ToggleButton(
630                      text=option,
631                      state='down' if option == value else 'normal',
632                      group=repr(content.uid), size_hint_y=None,
633                      height=44)
634                  button.bind(on_press=partial(
635                      self.save_property_option, widget, key))
636                  content.add_widget(button)
637          elif isinstance(prop, ObjectProperty):
638              if isinstance(value, Widget):
639                  content = Button(text=repr(value))
640                  content.bind(on_release=partial(self.highlight_widget, value))
641              elif isinstance(value, Texture):
642                  content = Image(texture=value)
643              else:
644                  content = Label(text=repr(value))
645  
646          elif isinstance(prop, BooleanProperty):
647              state = 'down' if value else 'normal'
648              content = ToggleButton(text=key, state=state)
649              content.bind(on_release=partial(self.save_property_boolean, widget,
650                                              key, index))
651  
652          self.content.clear_widgets()
653          if content:
654              self.content.add_widget(content)
655

690      def _update_widget_tree_node(self, node, widget, is_open=False):
691          tree = self.widgettree
692          update_nodes = []
693          nodes = {}
694          for cnode in node.nodes[:]:
695              try:
696                  nodes[cnode.widget] = cnode
697              except ReferenceError:
698                  # widget no longer exists, just remove it
699                  pass
700              tree.remove_node(cnode)
701          for child in widget.children:
702              if child is self:
703                  continue
704              if child in nodes:
705                  cnode = tree.add_node(nodes[child], node)
706              else:
707                  cnode = tree.add_node(TreeViewWidget(
708                      text=child.__class__.__name__, widget=child.proxy_ref,
709                      is_open=is_open), node)
710              update_nodes.append((cnode, child))
711          return update_nodes
712  
713      def update_widget_tree(self, *args):
714          if not hasattr(self, '_window_node') or not self._window_node:
715              self._window_node = self.widgettree.add_node(
716                  TreeViewWidget(text='Window', widget=self.win, is_open=True))
717  
718          nodes = self._update_widget_tree_node(self._window_node, self.win,
719                                                is_open=True)
720          while nodes:
721              ntmp = nodes[:]
722              nodes = []
723              for node in ntmp:
724                  nodes += self._update_widget_tree_node(*node)
725  
726          self.widgettree.update_selected_widget(self.widget)

729  def create_inspector(win, ctx, *l):
730      '''Create an Inspector instance attached to the *ctx* and bound to the 
731      Window's :meth:`~kivy.core.window.WindowBase.on_keyboard` event for 
732      capturing the keyboard shortcut. 
733   
734          :Parameters: 
735              `win`: A :class:`Window <kivy.core.window.WindowBase>` 
736                  The application Window to bind to. 
737              `ctx`: A :class:`~kivy.uix.widget.Widget` or subclass 
738                  The Widget to be inspected. 
739   
740      '''
741      # Dunno why, but if we are creating inspector within the start(), no lang
742      # rules are applied.
743      ctx.inspector = Inspector(win=win)
744      win.bind(children=ctx.inspector.on_window_children,
745               on_keyboard=ctx.inspector.keyboard_shortcut)
746  
747  
748  def start(win, ctx):
749      ctx.ev_late_create = Clock.schedule_once(
750          partial(create_inspector, win, ctx))
751  
752  
753  def stop(win, ctx):
754      '''Stop and unload any active Inspectors for the given *ctx*.'''
755      if hasattr(ctx, 'ev_late_create'):
756          ctx.ev_late_create.cancel()
757          del ctx.ev_late_create
758      if hasattr(ctx, 'inspector'):
759          win.unbind(children=ctx.inspector.on_window_children,
760                     on_keyboard=ctx.inspector.keyboard_shortcut)
761          win.remove_widget(ctx.inspector)
762          del ctx.inspector
