gencache.py 
1    """Manages the cache of generated Python code. 
2     
3    Description 
4      This file manages the cache of generated Python code.  When run from the 
5      command line, it also provides a number of options for managing that cache. 
6     
7    Implementation 
8      Each typelib is generated into a filename of format "{guid}x{lcid}x{major}x{minor}.py" 
9     
10     An external persistant dictionary maps from all known IIDs in all known type libraries 
11     to the type library itself. 
12    
13     Thus, whenever Python code knows the IID of an object, it can find the IID, LCID and version of 
14     the type library which supports it.  Given this information, it can find the Python module 
15     with the support. 
16    
17     If necessary, this support can be generated on the fly. 
18    
19   Hacks, to do, etc 
20     Currently just uses a pickled dictionary, but should used some sort of indexed file. 
21     Maybe an OLE2 compound file, or a bsddb file? 
22   """
23   import pywintypes, os, sys

68   def _SaveDicts():
69       if is_readonly:
70           raise RuntimeError(
71               "Trying to write to a readonly gencache ('%s')!" % win32com.__gen_path__
72           )
73       f = open(os.path.join(GetGeneratePath(), "dicts.dat"), "wb")
74       try:
75           p = pickle.Pickler(f)
76           p.dump(pickleVersion)
77           p.dump(clsidToTypelib)
78       finally:
79           f.close()

82   def _LoadDicts():
83       # Load the dictionary from a .zip file if that is where we live.
84       if is_zip:
85           import io as io
86   
87           loader = win32com.__loader__
88           arc_path = loader.archive
89           dicts_path = os.path.join(win32com.__gen_path__, "dicts.dat")
90           if dicts_path.startswith(arc_path):
91               dicts_path = dicts_path[len(arc_path) + 1 :]
92           else:
93               # Hm. See below.
94               return
95           try:
96               data = loader.get_data(dicts_path)
97           except AttributeError:
98               # The __loader__ has no get_data method.  See below.
99               return
100          except IOError:
101              # Our gencache is in a .zip file (and almost certainly readonly)
102              # but no dicts file.  That actually needn't be fatal for a frozen
103              # application.  Assuming they call "EnsureModule" with the same
104              # typelib IDs they have been frozen with, that EnsureModule will
105              # correctly re-build the dicts on the fly.  However, objects that
106              # rely on the gencache but have not done an EnsureModule will
107              # fail (but their apps are likely to fail running from source
108              # with a clean gencache anyway, as then they would be getting
109              # Dynamic objects until the cache is built - so the best answer
110              # for these apps is to call EnsureModule, rather than freezing
111              # the dict)
112              return
113          f = io.BytesIO(data)
114      else:
115          # NOTE: IOError on file open must be caught by caller.
116          f = open(os.path.join(win32com.__gen_path__, "dicts.dat"), "rb")
117      try:
118          p = pickle.Unpickler(f)
119          version = p.load()
120          global clsidToTypelib
121          clsidToTypelib = p.load()
122          versionRedirectMap.clear()
123      finally:
124          f.close()

127  def GetGeneratedFileName(clsid, lcid, major, minor):
128      """Given the clsid, lcid, major and  minor for a type lib, return 
129      the file name (no extension) providing this support. 
130      """
131      return str(clsid).upper()[1:-1] + "x%sx%sx%s" % (lcid, major, minor)

134  def SplitGeneratedFileName(fname):
135      """Reverse of GetGeneratedFileName()"""
136      return tuple(fname.split("x", 4))

139  def GetGeneratePath():
140      """Returns the name of the path to generate to. 
141      Checks the directory is OK. 
142      """
143      assert not is_readonly, "Why do you want the genpath for a readonly store?"
144      try:
145          os.makedirs(win32com.__gen_path__)
146          # os.mkdir(win32com.__gen_path__)
147      except os.error:
148          pass
149      try:
150          fname = os.path.join(win32com.__gen_path__, "__init__.py")
151          os.stat(fname)
152      except os.error:
153          f = open(fname, "w")
154          f.write(
155              "# Generated file - this directory may be deleted to reset the COM cache...\n"
156          )
157          f.write("import win32com\n")
158          f.write(
159              "if __path__[:-1] != win32com.__gen_path__: __path__.append(win32com.__gen_path__)\n"
160          )
161          f.close()
162  
163      return win32com.__gen_path__

172      Given a Program ID, return a Python class which wraps the COM object

183  def GetClassForCLSID(clsid):
184      """Get a Python class for a CLSID 
185   
186      Given a CLSID, return a Python class which wraps the COM object 
187   
188      Returns the Python class, or None if no module is available. 
189   
190      Params 
191      clsid -- A COM CLSID (or string repr of one) 
192      """
193      # first, take a short-cut - we may already have generated support ready-to-roll.
194      clsid = str(clsid)
195      if CLSIDToClass.HasClass(clsid):
196          return CLSIDToClass.GetClass(clsid)
197      mod = GetModuleForCLSID(clsid)
198      if mod is None:
199          return None
200      try:
201          return CLSIDToClass.GetClass(clsid)
202      except KeyError:
203          return None

207      """Get a Python module for a Program ID

209      Given a Program ID, return a Python module which contains the 
210      class which wraps the COM object.

213

220          return None
221      return GetModuleForCLSID(iid)

224  def GetModuleForCLSID(clsid):
225      """Get a Python module for a CLSID 
226   
227      Given a CLSID, return a Python module which contains the 
228      class which wraps the COM object. 
229   
230      Returns the Python module, or None if no module is available. 
231   
232      Params 
233      progid -- A COM CLSID (ie, not the description) 
234      """
235      clsid_str = str(clsid)
236      try:
237          typelibCLSID, lcid, major, minor = clsidToTypelib[clsid_str]
238      except KeyError:
239          return None
240  
241      try:
242          mod = GetModuleForTypelib(typelibCLSID, lcid, major, minor)
243      except ImportError:
244          mod = None
245      if mod is not None:
246          sub_mod = mod.CLSIDToPackageMap.get(clsid_str)
247          if sub_mod is None:
248              sub_mod = mod.VTablesToPackageMap.get(clsid_str)
249          if sub_mod is not None:
250              sub_mod_name = mod.__name__ + "." + sub_mod
251              try:
252                  __import__(sub_mod_name)
253              except ImportError:
254                  info = typelibCLSID, lcid, major, minor
255                  # Force the generation.  If this typelibrary has explicitly been added,
256                  # use it (it may not be registered, causing a lookup by clsid to fail)
257                  if info in demandGeneratedTypeLibraries:
258                      info = demandGeneratedTypeLibraries[info]
259                  from . import makepy
260  
261                  makepy.GenerateChildFromTypeLibSpec(sub_mod, info)
262                  # Generate does an import...
263              mod = sys.modules[sub_mod_name]
264      return mod

267  def GetModuleForTypelib(typelibCLSID, lcid, major, minor):
268      """Get a Python module for a type library ID 
269   
270      Given the CLSID of a typelibrary, return an imported Python module, 
271      else None 
272   
273      Params 
274      typelibCLSID -- IID of the type library. 
275      major -- Integer major version. 
276      minor -- Integer minor version 
277      lcid -- Integer LCID for the library. 
278      """
279      modName = GetGeneratedFileName(typelibCLSID, lcid, major, minor)
280      mod = _GetModule(modName)
281      # If the import worked, it doesn't mean we have actually added this
282      # module to our cache though - check that here.
283      if "_in_gencache_" not in mod.__dict__:
284          AddModuleToCache(typelibCLSID, lcid, major, minor)
285          assert "_in_gencache_" in mod.__dict__
286      return mod

289  def MakeModuleForTypelib(
290      typelibCLSID,
291      lcid,
292      major,
293      minor,
294      progressInstance=None,
295      bForDemand=bForDemandDefault,
296      bBuildHidden=1,
297  ):
298      """Generate support for a type library. 
299   
300      Given the IID, LCID and version information for a type library, generate 
301      and import the necessary support files.

306      typelibCLSID -- IID of the type library.

308      minor -- Integer minor version.

312      """
313      from . import makepy

324  def MakeModuleForTypelibInterface(
325      typelib_ob, progressInstance=None, bForDemand=bForDemandDefault, bBuildHidden=1
326  ):
327      """Generate support for a type library. 
328   
329      Given a PyITypeLib interface generate and import the necessary support files.  This is useful 
330      for getting makepy support for a typelibrary that is not registered - the caller can locate 
331      and load the type library itself, rather than relying on COM to find it. 
332   
333      Returns the Python module. 
334   
335      Params 
336      typelib_ob -- The type library itself 
337      progressInstance -- Instance to use as progress indicator, or None to 
338                          use the GUI progress bar. 
339      """
340      from . import makepy
341  
342      try:
343          makepy.GenerateFromTypeLibSpec(
344              typelib_ob,
345              progressInstance=progressInstance,
346              bForDemand=bForDemandDefault,
347              bBuildHidden=bBuildHidden,
348          )
349      except pywintypes.com_error:
350          return None
351      tla = typelib_ob.GetLibAttr()
352      guid = tla[0]
353      lcid = tla[1]
354      major = tla[3]
355      minor = tla[4]
356      return GetModuleForTypelib(guid, lcid, major, minor)

359  def EnsureModuleForTypelibInterface(
360      typelib_ob, progressInstance=None, bForDemand=bForDemandDefault, bBuildHidden=1
361  ):
362      """Check we have support for a type library, generating if not. 
363   
364      Given a PyITypeLib interface generate and import the necessary 
365      support files if necessary. This is useful for getting makepy support 
366      for a typelibrary that is not registered - the caller can locate and 
367      load the type library itself, rather than relying on COM to find it. 
368   
369      Returns the Python module. 
370   
371      Params 
372      typelib_ob -- The type library itself 
373      progressInstance -- Instance to use as progress indicator, or None to 
374                          use the GUI progress bar. 
375      """
376      tla = typelib_ob.GetLibAttr()
377      guid = tla[0]
378      lcid = tla[1]
379      major = tla[3]
380      minor = tla[4]
381  
382      # If demand generated, save the typelib interface away for later use
383      if bForDemand:
384          demandGeneratedTypeLibraries[(str(guid), lcid, major, minor)] = typelib_ob
385  
386      try:
387          return GetModuleForTypelib(guid, lcid, major, minor)
388      except ImportError:
389          pass
390      # Generate it.
391      return MakeModuleForTypelibInterface(
392          typelib_ob, progressInstance, bForDemand, bBuildHidden
393      )

396  def ForgetAboutTypelibInterface(typelib_ob):
397      """Drop any references to a typelib previously added with EnsureModuleForTypelibInterface and forDemand"""
398      tla = typelib_ob.GetLibAttr()
399      guid = tla[0]
400      lcid = tla[1]
401      major = tla[3]
402      minor = tla[4]
403      info = str(guid), lcid, major, minor
404      try:
405          del demandGeneratedTypeLibraries[info]
406      except KeyError:
407          # Not worth raising an exception - maybe they dont know we only remember for demand generated, etc.
408          print(
409              "ForgetAboutTypelibInterface:: Warning - type library with info %s is not being remembered!"
410              % (info,)
411          )
412      # and drop any version redirects to it
413      for key, val in list(versionRedirectMap.items()):
414          if val == info:
415              del versionRedirectMap[key]

418  def EnsureModule(
419      typelibCLSID,
420      lcid,
421      major,
422      minor,
423      progressInstance=None,
424      bValidateFile=not is_readonly,
425      bForDemand=bForDemandDefault,
426      bBuildHidden=1,
427  ):
428      """Ensure Python support is loaded for a type library, generating if necessary. 
429   
430      Given the IID, LCID and version information for a type library, check and if 
431      necessary (re)generate, then import the necessary support files. If we regenerate the file, there 
432      is no way to totally snuff out all instances of the old module in Python, and thus we will regenerate the file more than necessary, 
433      unless makepy/genpy is modified accordingly. 
434   
435   
436      Returns the Python module.  No exceptions are caught during the generate process. 
437   
438      Params 
439      typelibCLSID -- IID of the type library. 
440      major -- Integer major version. 
441      minor -- Integer minor version 
442      lcid -- Integer LCID for the library. 
443      progressInstance -- Instance to use as progress indicator, or None to 
444                          use the GUI progress bar. 
445      bValidateFile -- Whether or not to perform cache validation or not 
446      bForDemand -- Should a complete generation happen now, or on demand? 
447      bBuildHidden -- Should hidden members/attributes etc be generated? 
448      """
449      bReloadNeeded = 0
450      try:
451          try:
452              module = GetModuleForTypelib(typelibCLSID, lcid, major, minor)
453          except ImportError:
454              # If we get an ImportError
455              # We may still find a valid cache file under a different MinorVersion #
456              # (which windows will search out for us)
457              # print "Loading reg typelib", typelibCLSID, major, minor, lcid
458              module = None
459              try:
460                  tlbAttr = pythoncom.LoadRegTypeLib(
461                      typelibCLSID, major, minor, lcid
462                  ).GetLibAttr()
463                  # if the above line doesn't throw a pythoncom.com_error, check if
464                  # it is actually a different lib than we requested, and if so, suck it in
465                  if tlbAttr[1] != lcid or tlbAttr[4] != minor:
466                      # print "Trying 2nd minor #", tlbAttr[1], tlbAttr[3], tlbAttr[4]
467                      try:
468                          module = GetModuleForTypelib(
469                              typelibCLSID, tlbAttr[1], tlbAttr[3], tlbAttr[4]
470                          )
471                      except ImportError:
472                          # We don't have a module, but we do have a better minor
473                          # version - remember that.
474                          minor = tlbAttr[4]
475                  # else module remains None
476              except pythoncom.com_error:
477                  # couldn't load any typelib - mod remains None
478                  pass
479          if module is not None and bValidateFile:
480              assert not is_readonly, "Can't validate in a read-only gencache"
481              try:
482                  typLibPath = pythoncom.QueryPathOfRegTypeLib(
483                      typelibCLSID, major, minor, lcid
484                  )
485                  # windows seems to add an extra \0 (via the underlying BSTR)
486                  # The mainwin toolkit does not add this erroneous \0
487                  if typLibPath[-1] == "\0":
488                      typLibPath = typLibPath[:-1]
489                  suf = getattr(os.path, "supports_unicode_filenames", 0)
490                  if not suf:
491                      # can't pass unicode filenames directly - convert
492                      try:
493                          typLibPath = typLibPath.encode(sys.getfilesystemencoding())
494                      except AttributeError:  # no sys.getfilesystemencoding
495                          typLibPath = str(typLibPath)
496                  tlbAttributes = pythoncom.LoadRegTypeLib(
497                      typelibCLSID, major, minor, lcid
498                  ).GetLibAttr()
499              except pythoncom.com_error:
500                  # We have a module, but no type lib - we should still
501                  # run with what we have though - the typelib may not be
502                  # deployed here.
503                  bValidateFile = 0
504          if module is not None and bValidateFile:
505              assert not is_readonly, "Can't validate in a read-only gencache"
506              filePathPrefix = "%s\\%s" % (
507                  GetGeneratePath(),
508                  GetGeneratedFileName(typelibCLSID, lcid, major, minor),
509              )
510              filePath = filePathPrefix + ".py"
511              filePathPyc = filePathPrefix + ".py"
512              if __debug__:
513                  filePathPyc = filePathPyc + "c"
514              else:
515                  filePathPyc = filePathPyc + "o"
516              # Verify that type library is up to date.
517              # If we have a differing MinorVersion or genpy has bumped versions, update the file
518              from . import genpy
519  
520              if (
521                  module.MinorVersion != tlbAttributes[4]
522                  or genpy.makepy_version != module.makepy_version
523              ):
524                  # print "Version skew: %d, %d" % (module.MinorVersion, tlbAttributes[4])
525                  # try to erase the bad file from the cache
526                  try:
527                      os.unlink(filePath)
528                  except os.error:
529                      pass
530                  try:
531                      os.unlink(filePathPyc)
532                  except os.error:
533                      pass
534                  if os.path.isdir(filePathPrefix):
535                      import shutil
536  
537                      shutil.rmtree(filePathPrefix)
538                  minor = tlbAttributes[4]
539                  module = None
540                  bReloadNeeded = 1
541              else:
542                  minor = module.MinorVersion
543                  filePathPrefix = "%s\\%s" % (
544                      GetGeneratePath(),
545                      GetGeneratedFileName(typelibCLSID, lcid, major, minor),
546                  )
547                  filePath = filePathPrefix + ".py"
548                  filePathPyc = filePathPrefix + ".pyc"
549                  # print "Trying py stat: ", filePath
550                  fModTimeSet = 0
551                  try:
552                      pyModTime = os.stat(filePath)[8]
553                      fModTimeSet = 1
554                  except os.error as e:
555                      # If .py file fails, try .pyc file
556                      # print "Trying pyc stat", filePathPyc
557                      try:
558                          pyModTime = os.stat(filePathPyc)[8]
559                          fModTimeSet = 1
560                      except os.error as e:
561                          pass
562                  # print "Trying stat typelib", pyModTime
563                  # print str(typLibPath)
564                  typLibModTime = os.stat(typLibPath)[8]
565                  if fModTimeSet and (typLibModTime > pyModTime):
566                      bReloadNeeded = 1
567                      module = None
568      except (ImportError, os.error):
569          module = None
570      if module is None:
571          # We need to build an item.  If we are in a read-only cache, we
572          # can't/don't want to do this - so before giving up, check for
573          # a different minor version in our cache - according to COM, this is OK
574          if is_readonly:
575              key = str(typelibCLSID), lcid, major, minor
576              # If we have been asked before, get last result.
577              try:
578                  return versionRedirectMap[key]
579              except KeyError:
580                  pass
581              # Find other candidates.
582              items = []
583              for desc in GetGeneratedInfos():
584                  if key[0] == desc[0] and key[1] == desc[1] and key[2] == desc[2]:
585                      items.append(desc)
586              if items:
587                  # Items are all identical, except for last tuple element
588                  # We want the latest minor version we have - so just sort and grab last
589                  items.sort()
590                  new_minor = items[-1][3]
591                  ret = GetModuleForTypelib(typelibCLSID, lcid, major, new_minor)
592              else:
593                  ret = None
594              # remember and return
595              versionRedirectMap[key] = ret
596              return ret
597          # print "Rebuilding: ", major, minor
598          module = MakeModuleForTypelib(
599              typelibCLSID,
600              lcid,
601              major,
602              minor,
603              progressInstance,
604              bForDemand=bForDemand,
605              bBuildHidden=bBuildHidden,
606          )
607          # If we replaced something, reload it
608          if bReloadNeeded:
609              module = reload(module)
610              AddModuleToCache(typelibCLSID, lcid, major, minor)
611      return module

614  def EnsureDispatch(
615      prog_id, bForDemand=1
616  ):  # New fn, so we default the new demand feature to on!
617      """Given a COM prog_id, return an object that is using makepy support, building if necessary"""
618      disp = win32com.client.Dispatch(prog_id)
619      if not disp.__dict__.get("CLSID"):  # Eeek - no makepy support - try and build it.
620          try:
621              ti = disp._oleobj_.GetTypeInfo()
622              disp_clsid = ti.GetTypeAttr()[0]
623              tlb, index = ti.GetContainingTypeLib()
624              tla = tlb.GetLibAttr()
625              mod = EnsureModule(tla[0], tla[1], tla[3], tla[4], bForDemand=bForDemand)
626              GetModuleForCLSID(disp_clsid)
627              # Get the class from the module.
628              from . import CLSIDToClass
629  
630              disp_class = CLSIDToClass.GetClass(str(disp_clsid))
631              disp = disp_class(disp._oleobj_)
632          except pythoncom.com_error:
633              raise TypeError(
634                  "This COM object can not automate the makepy process - please run makepy manually for this object"
635              )
636      return disp

639  def AddModuleToCache(
640      typelibclsid, lcid, major, minor, verbose=1, bFlushNow=not is_readonly
641  ):
642      """Add a newly generated file to the cache dictionary."""
643      fname = GetGeneratedFileName(typelibclsid, lcid, major, minor)
644      mod = _GetModule(fname)
645      # if mod._in_gencache_ is already true, then we are reloading this
646      # module - this doesn't mean anything special though!
647      mod._in_gencache_ = 1
648      info = str(typelibclsid), lcid, major, minor
649      dict_modified = False
650  
651      def SetTypelibForAllClsids(dict):
652          nonlocal dict_modified
653          for clsid, cls in dict.items():
654              if clsidToTypelib.get(clsid) != info:
655                  clsidToTypelib[clsid] = info
656                  dict_modified = True
657  
658      SetTypelibForAllClsids(mod.CLSIDToClassMap)
659      SetTypelibForAllClsids(mod.CLSIDToPackageMap)
660      SetTypelibForAllClsids(mod.VTablesToClassMap)
661      SetTypelibForAllClsids(mod.VTablesToPackageMap)
662  
663      # If this lib was previously redirected, drop it
664      if info in versionRedirectMap:
665          del versionRedirectMap[info]
666      if bFlushNow and dict_modified:
667          _SaveDicts()

670  def GetGeneratedInfos():
671      zip_pos = win32com.__gen_path__.find(".zip\\")
672      if zip_pos >= 0:
673          import zipfile
674  
675          zip_file = win32com.__gen_path__[: zip_pos + 4]
676          zip_path = win32com.__gen_path__[zip_pos + 5 :].replace("\\", "/")
677          zf = zipfile.ZipFile(zip_file)
678          infos = {}
679          for n in zf.namelist():
680              if not n.startswith(zip_path):
681                  continue
682              base = n[len(zip_path) + 1 :].split("/")[0]
683              try:
684                  iid, lcid, major, minor = base.split("x")
685                  lcid = int(lcid)
686                  major = int(major)
687                  minor = int(minor)
688                  iid = pywintypes.IID("{" + iid + "}")
689              except ValueError:
690                  continue
691              except pywintypes.com_error:
692                  # invalid IID
693                  continue
694              infos[(iid, lcid, major, minor)] = 1
695          zf.close()
696          return list(infos.keys())
697      else:
698          # on the file system
699          files = glob.glob(win32com.__gen_path__ + "\\*")
700          ret = []
701          for file in files:
702              if not os.path.isdir(file) and not os.path.splitext(file)[1] == ".py":
703                  continue
704              name = os.path.splitext(os.path.split(file)[1])[0]
705              try:
706                  iid, lcid, major, minor = name.split("x")
707                  iid = pywintypes.IID("{" + iid + "}")
708                  lcid = int(lcid)
709                  major = int(major)
710                  minor = int(minor)
711              except ValueError:
712                  continue
713              except pywintypes.com_error:
714                  # invalid IID
715                  continue
716              ret.append((iid, lcid, major, minor))
717          return ret

720  def _GetModule(fname):
721      """Given the name of a module in the gen_py directory, import and return it."""
722      mod_name = "win32com.gen_py.%s" % fname
723      mod = __import__(mod_name)
724      return sys.modules[mod_name]

727  def Rebuild(verbose=1):
728      """Rebuild the cache indexes from the file system."""
729      clsidToTypelib.clear()
730      infos = GetGeneratedInfos()
731      if verbose and len(infos):  # Dont bother reporting this when directory is empty!
732          print("Rebuilding cache of generated files for COM support...")
733      for info in infos:
734          iid, lcid, major, minor = info
735          if verbose:
736              print("Checking", GetGeneratedFileName(*info))
737          try:
738              AddModuleToCache(iid, lcid, major, minor, verbose, 0)
739          except:
740              print(
741                  "Could not add module %s - %s: %s"
742                  % (info, sys.exc_info()[0], sys.exc_info()[1])
743              )
744      if verbose and len(infos):  # Dont bother reporting this when directory is empty!
745          print("Done.")
746      _SaveDicts()

751      # Build a unique dir
752      d = {}
753      for clsid, (typelibCLSID, lcid, major, minor) in clsidToTypelib.items():
