84   def IsOnPythonPath(path):
85       "Given a path only, see if it is on the Pythonpath.  Assumes path is a full path spec."
86       # must check that the command line arg's path is in sys.path
87       for syspath in sys.path:
88           try:
89               # Python 1.5 and later allows an empty sys.path entry.
90               if syspath and win32ui.FullPath(syspath) == path:
91                   return 1
92           except win32ui.error as details:
93               print(
94                   "Warning: The sys.path entry '%s' is invalid\n%s" % (syspath, details)
95               )
96       return 0

99   def GetPackageModuleName(fileName):
100      """Given a filename, return (module name, new path). 
101      eg - given "c:\a\b\c\my.py", return ("b.c.my",None) if "c:\a" is on sys.path. 
102      If no package found, will return ("my", "c:\a\b\c") 
103      """
104      path, fname = os.path.split(fileName)
105      path = origPath = win32ui.FullPath(path)
106      fname = os.path.splitext(fname)[0]
107      modBits = []
108      newPathReturn = None
109      if not IsOnPythonPath(path):
110          # Module not directly on the search path - see if under a package.
111          while len(path) > 3:  # ie 'C:\'
112              path, modBit = os.path.split(path)
113              modBits.append(modBit)
114              # If on path, _and_ existing package of that name loaded.
115              if (
116                  IsOnPythonPath(path)
117                  and modBit in sys.modules
118                  and (
119                      os.path.exists(os.path.join(path, modBit, "__init__.py"))
120                      or os.path.exists(os.path.join(path, modBit, "__init__.pyc"))
121                      or os.path.exists(os.path.join(path, modBit, "__init__.pyo"))
122                  )
123              ):
124                  modBits.reverse()
125                  return ".".join(modBits) + "." + fname, newPathReturn
126              # Not found - look a level higher
127          else:
128              newPathReturn = origPath
129  
130      return fname, newPathReturn

159      """Returns the active editor document and view, or (None,None) if no 
160      active document or its not an editor document. 
161      """
162      view = GetActiveView()
163      if view is None or isinstance(view, TreeView):

171  def GetActiveFileName(bAutoSave=1):
172      """Gets the file name for the active frame, saving it if necessary. 
173   
174      Returns None if it cant be found, or raises KeyboardInterrupt. 
175      """
176      pathName = None
177      active = GetActiveView()
178      if active is None:
179          return None
180      try:
181          doc = active.GetDocument()
182          pathName = doc.GetPathName()
183  
184          if bAutoSave and (
185              len(pathName) > 0
186              or doc.GetTitle()[:8] == "Untitled"
187              or doc.GetTitle()[:6] == "Script"
188          ):  # if not a special purpose window
189              if doc.IsModified():
190                  try:
191                      doc.OnSaveDocument(pathName)
192                      pathName = doc.GetPathName()
193  
194                      # clear the linecache buffer
195                      linecache.clearcache()
196  
197                  except win32ui.error:
198                      raise KeyboardInterrupt
199  
200      except (win32ui.error, AttributeError):
201          pass
202      if not pathName:
203          return None
204      return pathName

212  def RunScript(defName=None, defArgs=None, bShowDialog=1, debuggingType=None):
213      global lastScript, lastArgs, lastDebuggingType
214      _debugger_stop_frame_ = 1  # Magic variable so the debugger will hide me!
215  
216      # Get the debugger - may be None!
217      debugger = GetDebugger()
218  
219      if defName is None:
220          try:
221              pathName = GetActiveFileName()
222          except KeyboardInterrupt:
223              return  # User cancelled save.
224      else:
225          pathName = defName
226      if not pathName:
227          pathName = lastScript
228      if defArgs is None:
229          args = ""
230          if pathName == lastScript:
231              args = lastArgs
232      else:
233          args = defArgs
234      if debuggingType is None:
235          debuggingType = lastDebuggingType
236  
237      if not pathName or bShowDialog:
238          dlg = DlgRunScript(debugger is not None)
239          dlg["script"] = pathName
240          dlg["args"] = args
241          dlg["debuggingType"] = debuggingType
242          if dlg.DoModal() != win32con.IDOK:
243              return
244          script = dlg["script"]
245          args = dlg["args"]
246          debuggingType = dlg["debuggingType"]
247          if not script:
248              return
249          if debuggingType == RS_DEBUGGER_GO and debugger is not None:
250              # This may surprise users - they select "Run under debugger", but
251              # it appears not to!  Only warn when they pick from the dialog!
252              # First - ensure the debugger is activated to pickup any break-points
253              # set in the editor.
254              try:
255                  # Create the debugger, but _dont_ init the debugger GUI.
256                  rd = debugger._GetCurrentDebugger()
257              except AttributeError:
258                  rd = None
259              if rd is not None and len(rd.breaks) == 0:
260                  msg = "There are no active break-points.\r\n\r\nSelecting this debug option without any\r\nbreak-points is unlikely to have the desired effect\r\nas the debugger is unlikely to be invoked..\r\n\r\nWould you like to step-through in the debugger instead?"
261                  rc = win32ui.MessageBox(
262                      msg,
263                      win32ui.LoadString(win32ui.IDR_DEBUGGER),
264                      win32con.MB_YESNOCANCEL | win32con.MB_ICONINFORMATION,
265                  )
266                  if rc == win32con.IDCANCEL:
267                      return
268                  if rc == win32con.IDYES:
269                      debuggingType = RS_DEBUGGER_STEP
270  
271          lastDebuggingType = debuggingType
272          lastScript = script
273          lastArgs = args
274      else:
275          script = pathName
276  
277      # try and open the script.
278      if (
279          len(os.path.splitext(script)[1]) == 0
280      ):  # check if no extension supplied, and give one.
281          script = script + ".py"
282      # If no path specified, try and locate the file
283      path, fnameonly = os.path.split(script)
284      if len(path) == 0:
285          try:
286              os.stat(fnameonly)  # See if it is OK as is...
287              script = fnameonly
288          except os.error:
289              fullScript = LocatePythonFile(script)
290              if fullScript is None:
291                  win32ui.MessageBox("The file '%s' can not be located" % script)
292                  return
293              script = fullScript
294      else:
295          path = win32ui.FullPath(path)
296          if not IsOnPythonPath(path):
297              sys.path.append(path)
298  
299      # py3k fun: If we use text mode to open the file, we get \r\n
300      # translated so Python allows the syntax (good!), but we get back
301      # text already decoded from the default encoding (bad!) and Python
302      # ignores any encoding decls (bad!).  If we use binary mode we get
303      # the raw bytes and Python looks at the encoding (good!) but \r\n
304      # chars stay in place so Python throws a syntax error (bad!).
305      # So: so the binary thing and manually normalize \r\n.
306      try:
307          f = open(script, "rb")
308      except IOError as exc:
309          win32ui.MessageBox(
310              "The file could not be opened - %s (%d)" % (exc.strerror, exc.errno)
311          )
312          return
313  
314      # Get the source-code - as above, normalize \r\n
315      code = f.read().replace(byte_crlf, byte_lf).replace(byte_cr, byte_lf) + byte_lf
316  
317      # Remember and hack sys.argv for the script.
318      oldArgv = sys.argv
319      sys.argv = ParseArgs(args)
320      sys.argv.insert(0, script)
321      # sys.path[0] is the path of the script
322      oldPath0 = sys.path[0]
323      newPath0 = os.path.split(script)[0]
324      if not oldPath0:  # if sys.path[0] is empty
325          sys.path[0] = newPath0
326          insertedPath0 = 0
327      else:
328          sys.path.insert(0, newPath0)
329          insertedPath0 = 1
330      bWorked = 0
331      win32ui.DoWaitCursor(1)
332      base = os.path.split(script)[1]
333      # Allow windows to repaint before starting.
334      win32ui.PumpWaitingMessages()
335      win32ui.SetStatusText("Running script %s..." % base, 1)
336      exitCode = 0
337      from pywin.framework import interact
338  
339      # Check the debugger flags
340      if debugger is None and (debuggingType != RS_DEBUGGER_NONE):
341          win32ui.MessageBox(
342              "No debugger is installed.  Debugging options have been ignored!"
343          )
344          debuggingType = RS_DEBUGGER_NONE
345  
346      # Get a code object - ignore the debugger for this, as it is probably a syntax error
347      # at this point
348      try:
349          codeObject = compile(code, script, "exec")
350      except:
351          # Almost certainly a syntax error!
352          _HandlePythonFailure("run script", script)
353          # No code object which to run/debug.
354          return
355      __main__.__file__ = script
356      try:
357          if debuggingType == RS_DEBUGGER_STEP:
358              debugger.run(codeObject, __main__.__dict__, start_stepping=1)
359          elif debuggingType == RS_DEBUGGER_GO:
360              debugger.run(codeObject, __main__.__dict__, start_stepping=0)
361          else:
362              # Post mortem or no debugging
363              exec(codeObject, __main__.__dict__)
364          bWorked = 1
365      except bdb.BdbQuit:
366          # Dont print tracebacks when the debugger quit, but do print a message.
367          print("Debugging session cancelled.")
368          exitCode = 1
369          bWorked = 1
370      except SystemExit as code:
371          exitCode = code
372          bWorked = 1
373      except KeyboardInterrupt:
374          # Consider this successful, as we dont want the debugger.
375          # (but we do want a traceback!)
376          if interact.edit and interact.edit.currentView:
377              interact.edit.currentView.EnsureNoPrompt()
378          traceback.print_exc()
379          if interact.edit and interact.edit.currentView:
380              interact.edit.currentView.AppendToPrompt([])
381          bWorked = 1
382      except:
383          if interact.edit and interact.edit.currentView:
384              interact.edit.currentView.EnsureNoPrompt()
385          traceback.print_exc()
386          if interact.edit and interact.edit.currentView:
387              interact.edit.currentView.AppendToPrompt([])
388          if debuggingType == RS_DEBUGGER_PM:
389              debugger.pm()
390      del __main__.__file__
391      sys.argv = oldArgv
392      if insertedPath0:
393          del sys.path[0]
394      else:
395          sys.path[0] = oldPath0
396      f.close()
397      if bWorked:
398          win32ui.SetStatusText("Script '%s' returned exit code %s" % (script, exitCode))
399      else:
400          win32ui.SetStatusText("Exception raised while running script  %s" % base)
401      try:
402          sys.stdout.flush()
403      except AttributeError:
404          pass
405  
406      win32ui.DoWaitCursor(0)

409  def ImportFile():
410      """This code looks for the current window, and determines if it can be imported.  If not, 
411      it will prompt for a file name, and allow it to be imported."""
412      try:
413          pathName = GetActiveFileName()
414      except KeyboardInterrupt:
415          pathName = None
416  
417      if pathName is not None:
418          if os.path.splitext(pathName)[1].lower() not in (".py", ".pyw", ".pyx"):
419              pathName = None
420  
421      if pathName is None:
422          openFlags = win32con.OFN_OVERWRITEPROMPT | win32con.OFN_FILEMUSTEXIST
423          dlg = win32ui.CreateFileDialog(
424              1, None, None, openFlags, "Python Scripts (*.py;*.pyw)|*.py;*.pyw;*.pyx||"
425          )
426          dlg.SetOFNTitle("Import Script")
427          if dlg.DoModal() != win32con.IDOK:
428              return 0
429  
430          pathName = dlg.GetPathName()
431  
432      # If already imported, dont look for package
433      path, modName = os.path.split(pathName)
434      modName, modExt = os.path.splitext(modName)
435      newPath = None
436      # note that some packages (*cough* email *cough*) use "lazy importers"
437      # meaning sys.modules can change as a side-effect of looking at
438      # module.__file__ - so we must take a copy (ie, items() in py2k,
439      # list(items()) in py3k)
440      for key, mod in list(sys.modules.items()):
441          if getattr(mod, "__file__", None):
442              fname = mod.__file__
443              base, ext = os.path.splitext(fname)
444              if ext.lower() in [".pyo", ".pyc"]:
445                  ext = ".py"
446              fname = base + ext
447              if win32ui.ComparePath(fname, pathName):
448                  modName = key
449                  break
450      else:  # for not broken
451          modName, newPath = GetPackageModuleName(pathName)
452          if newPath:
453              sys.path.append(newPath)
454  
455      if modName in sys.modules:
456          bNeedReload = 1
457          what = "reload"
458      else:
459          what = "import"
460          bNeedReload = 0
461  
462      win32ui.SetStatusText(what.capitalize() + "ing module...", 1)
463      win32ui.DoWaitCursor(1)
464      #   win32ui.GetMainFrame().BeginWaitCursor()
465  
466      try:
467          # always do an import, as it is cheap if it's already loaded.  This ensures
468          # it is in our name space.
469          codeObj = compile("import " + modName, "<auto import>", "exec")
470      except SyntaxError:
471          win32ui.SetStatusText('Invalid filename for import: "' + modName + '"')
472          return
473      try:
474          exec(codeObj, __main__.__dict__)
475          mod = sys.modules.get(modName)
476          if bNeedReload:
477              from importlib import reload
478  
479              mod = reload(sys.modules[modName])
480          win32ui.SetStatusText(
481              "Successfully "
482              + what
483              + "ed module '"
484              + modName
485              + "': %s" % getattr(mod, "__file__", "<unkown file>")
486          )
487      except:
488          _HandlePythonFailure(what)
489      win32ui.DoWaitCursor(0)

492  def CheckFile():
493      """This code looks for the current window, and gets Python to check it 
494      without actually executing any code (ie, by compiling only) 
495      """
496      try:
497          pathName = GetActiveFileName()
498      except KeyboardInterrupt:
499          return
500  
501      what = "check"
502      win32ui.SetStatusText(what.capitalize() + "ing module...", 1)
503      win32ui.DoWaitCursor(1)
504      try:
505          f = open(pathName)
506      except IOError as details:
507          print("Cant open file '%s' - %s" % (pathName, details))
508          return
509      try:
510          code = f.read() + "\n"
511      finally:
512          f.close()
513      try:
514          codeObj = compile(code, pathName, "exec")
515          if RunTabNanny(pathName):
516              win32ui.SetStatusText(
517                  "Python and the TabNanny successfully checked the file '"
518                  + os.path.basename(pathName)
519                  + "'"
520              )
521      except SyntaxError:
522          _HandlePythonFailure(what, pathName)
523      except:
524          traceback.print_exc()
525          _HandlePythonFailure(what)
526      win32ui.DoWaitCursor(0)

529  def RunTabNanny(filename):
530      import io as io
531  
532      tabnanny = FindTabNanny()
533      if tabnanny is None:
534          win32ui.MessageBox("The TabNanny is not around, so the children can run amok!")
535          return
536  
537      # Capture the tab-nanny output
538      newout = io.StringIO()
539      old_out = sys.stderr, sys.stdout
540      sys.stderr = sys.stdout = newout
541      try:
542          tabnanny.check(filename)
543      finally:
544          # Restore output
545          sys.stderr, sys.stdout = old_out
546      data = newout.getvalue()
547      if data:
548          try:
549              lineno = data.split()[1]
550              lineno = int(lineno)
551              _JumpToPosition(filename, lineno)
552              try:  # Try and display whitespace
553                  GetActiveEditControl().SCISetViewWS(1)
554              except:
555                  pass
556              win32ui.SetStatusText("The TabNanny found trouble at line %d" % lineno)
557          except (IndexError, TypeError, ValueError):
558              print("The tab nanny complained, but I cant see where!")
559              print(data)
560          return 0
561      return 1

568  def JumpToDocument(fileName, lineno=0, col=1, nChars=0, bScrollToTop=0):
569      # Jump to the position in a file.
570      # If lineno is <= 0, dont move the position - just open/restore.
571      # if nChars > 0, select that many characters.
572      # if bScrollToTop, the specified line will be moved to the top of the window
573      #  (eg, bScrollToTop should be false when jumping to an error line to retain the
574      #  context, but true when jumping to a method defn, where we want the full body.
575      # Return the view which is editing the file, or None on error.
576      doc = win32ui.GetApp().OpenDocumentFile(fileName)
577      if doc is None:
578          return None
579      frame = doc.GetFirstView().GetParentFrame()
580      try:
581          view = frame.GetEditorView()
582          if frame.GetActiveView() != view:
583              frame.SetActiveView(view)
584          frame.AutoRestore()
585      except AttributeError:  # Not an editor frame??
586          view = doc.GetFirstView()
587      if lineno > 0:
588          charNo = view.LineIndex(lineno - 1)
589          start = charNo + col - 1
590          size = view.GetTextLength()
591          try:
592              view.EnsureCharsVisible(charNo)
593          except AttributeError:
594              print("Doesnt appear to be one of our views?")
595          view.SetSel(min(start, size), min(start + nChars, size))
596      if bScrollToTop:
597          curTop = view.GetFirstVisibleLine()
598          nScroll = (lineno - 1) - curTop
599          view.LineScroll(nScroll, 0)
600      view.SetFocus()
601      return view

604  def _HandlePythonFailure(what, syntaxErrorPathName=None):
605      typ, details, tb = sys.exc_info()
606      if isinstance(details, SyntaxError):
607          try:
608              msg, (fileName, line, col, text) = details
609              if (not fileName or fileName == "<string>") and syntaxErrorPathName:
610                  fileName = syntaxErrorPathName
611              _JumpToPosition(fileName, line, col)
612          except (TypeError, ValueError):
613              msg = str(details)
614          win32ui.SetStatusText("Failed to " + what + " - syntax error - %s" % msg)
615      else:
616          traceback.print_exc()
617          win32ui.SetStatusText("Failed to " + what + " - " + str(details))
618      tb = None  # Clean up a cycle.

621  # Find the Python TabNanny in either the standard library or the Python Tools/Scripts directory.
622  def FindTabNanny():
623      try:
624          return __import__("tabnanny")
625      except ImportError:
626          pass
627      # OK - not in the standard library - go looking.
628      filename = "tabnanny.py"
629      try:
630          path = win32api.RegQueryValue(
631              win32con.HKEY_LOCAL_MACHINE,
632              "SOFTWARE\\Python\\PythonCore\\%s\\InstallPath" % (sys.winver),
633          )
634      except win32api.error:
635          print("WARNING - The Python registry does not have an 'InstallPath' setting")
636          print("          The file '%s' can not be located" % (filename))
637          return None
638      fname = os.path.join(path, "Tools\\Scripts\\%s" % filename)
639      try:
640          os.stat(fname)
641      except os.error:
642          print(
643              "WARNING - The file '%s' can not be located in path '%s'" % (filename, path)
644          )
645          return None
646  
647      tabnannyhome, tabnannybase = os.path.split(fname)
648      tabnannybase = os.path.splitext(tabnannybase)[0]
649      # Put tab nanny at the top of the path.
650      sys.path.insert(0, tabnannyhome)
651      try:
652          return __import__(tabnannybase)
653      finally:
654          # remove the tab-nanny from the path
655          del sys.path[0]

658  def LocatePythonFile(fileName, bBrowseIfDir=1):
659      "Given a file name, return a fully qualified file name, or None"
660      # first look for the exact file as specified
661      if not os.path.isfile(fileName):
662          # Go looking!
663          baseName = fileName
664          for path in sys.path:
