16   def CreateTestAccessDatabase(dbname=None):
17       # Creates a test access database - returns the filename.
18       if dbname is None:
19           dbname = os.path.join(win32api.GetTempPath(), "COMTestSuiteTempDatabase.mdb")
20   
21       access = Dispatch("Access.Application")
22       dbEngine = access.DBEngine
23       workspace = dbEngine.Workspaces(0)
24   
25       try:
26           os.unlink(dbname)
27       except os.error:
28           print(
29               "WARNING - Unable to delete old test database - expect a COM exception RSN!"
30           )
31   
32       newdb = workspace.CreateDatabase(
33           dbname, constants.dbLangGeneral, constants.dbEncrypt
34       )
35   
36       # Create one test table.
37       table = newdb.CreateTableDef("Test Table 1")
38       table.Fields.Append(table.CreateField("First Name", constants.dbText))
39       table.Fields.Append(table.CreateField("Last Name", constants.dbText))
40   
41       index = table.CreateIndex("UniqueIndex")
42       index.Fields.Append(index.CreateField("First Name"))
43       index.Fields.Append(index.CreateField("Last Name"))
44       index.Unique = -1
45       table.Indexes.Append(index)
46   
47       newdb.TableDefs.Append(table)
48   
49       # Create a second test table.
50       table = newdb.CreateTableDef("Test Table 2")
51       table.Fields.Append(table.CreateField("First Name", constants.dbText))
52       table.Fields.Append(table.CreateField("Last Name", constants.dbText))
53   
54       newdb.TableDefs.Append(table)
55   
56       # Create a relationship between them
57       relation = newdb.CreateRelation("TestRelationship")
58       relation.Table = "Test Table 1"
59       relation.ForeignTable = "Test Table 2"
60   
61       field = relation.CreateField("First Name")
62       field.ForeignName = "First Name"
63       relation.Fields.Append(field)
64   
65       field = relation.CreateField("Last Name")
66       field.ForeignName = "Last Name"
67       relation.Fields.Append(field)
68   
69       relation.Attributes = (
70           constants.dbRelationDeleteCascade + constants.dbRelationUpdateCascade
71       )
72   
73       newdb.Relations.Append(relation)
74   
75       # Finally we can add some data to the table.
76       tab1 = newdb.OpenRecordset("Test Table 1")
77       tab1.AddNew()
78       tab1.Fields("First Name").Value = "Mark"
79       tab1.Fields("Last Name").Value = "Hammond"
80       tab1.Update()
81   
82       tab1.MoveFirst()
83       # We do a simple bookmark test which tests our optimized VT_SAFEARRAY|VT_UI1 support.
84       # The bookmark will be a buffer object - remember it for later.
85       bk = tab1.Bookmark
86   
87       # Add a second record.
88       tab1.AddNew()
89       tab1.Fields("First Name").Value = "Second"
90       tab1.Fields("Last Name").Value = "Person"
91       tab1.Update()
92   
93       # Reset the bookmark to the one we saved.
94       # But first check the test is actually doing something!
95       tab1.MoveLast()
96       if tab1.Fields("First Name").Value != "Second":
97           raise RuntimeError("Unexpected record is last - makes bookmark test pointless!")
98   
99       tab1.Bookmark = bk
100      if tab1.Bookmark != bk:
101          raise RuntimeError("The bookmark data is not the same")
102  
103      if tab1.Fields("First Name").Value != "Mark":
104          raise RuntimeError("The bookmark did not reset the record pointer correctly")
105  
106      return dbname

137  def GenerateSupport():
138      # dao
139      gencache.EnsureModule("{00025E01-0000-0000-C000-000000000046}", 0, 4, 0)
140      # Access
141      #       gencache.EnsureModule("{4AFFC9A0-5F99-101B-AF4E-00AA003F0F07}", 0, 8, 0)
142      gencache.EnsureDispatch("Access.Application")

145  def DumpAccessInfo(dbname):
146      amod = gencache.GetModuleForProgID("Access.Application")
147      dmod = gencache.GetModuleForProgID("DAO.DBEngine.35")
148      if amod is None and dmod is None:
149          DoDumpAccessInfo(dbname)
150          # Now generate all the support we can.
151          GenerateSupport()
152      else:
153          sys.stderr.write(
154              "testAccess not doing dynamic test, as generated code already exists\n"
155          )
156      # Now a generated version.
157      DoDumpAccessInfo(dbname)

160  def test(dbname=None):
161      if dbname is None:
162          # We need makepy support to create a database (just for the constants!)
163          try:
164              GenerateSupport()
165          except pythoncom.com_error:
166              print("*** Can not import the MSAccess type libraries - tests skipped")
167              return
168          dbname = CreateTestAccessDatabase()
169          print("A test database at '%s' was created" % dbname)
170  
171      DumpAccessInfo(dbname)
