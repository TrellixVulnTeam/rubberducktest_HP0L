13   def GetDefaultProfileName():
14       import win32api, win32con
15   
16       try:
17           key = win32api.RegOpenKey(
18               win32con.HKEY_CURRENT_USER,
19               "Software\\Microsoft\\Windows NT\\CurrentVersion\\Windows Messaging Subsystem\\Profiles",
20           )
21           try:
22               return win32api.RegQueryValueEx(key, "DefaultProfile")[0]
23           finally:
24               key.Close()
25       except win32api.error:
26           return None

30   # Recursive dump of folders.

32   def DumpFolder(folder, indent=0):
33       print(" " * indent, folder.Name)
34       folders = folder.Folders
35       folder = folders.GetFirst()
36       while folder:
37           DumpFolder(folder, indent + 1)
38           folder = folders.GetNext()

41   def DumpFolders(session):
42       try:
43           infostores = session.InfoStores
44       except AttributeError:
45           # later outlook?
46           store = session.DefaultStore
47           folder = store.GetRootFolder()
48           DumpFolder(folder)
49           return
50   
51       print(infostores)
52       print("There are %d infostores" % infostores.Count)
53       for i in range(infostores.Count):
54           infostore = infostores[i + 1]
55           print("Infostore = ", infostore.Name)
56           try:
57               folder = infostore.RootFolder
58           except pythoncom.com_error as details:
59               hr, msg, exc, arg = details
60               # -2147221219 == MAPI_E_FAILONEPROVIDER - a single provider temporarily not available.
61               if exc and exc[-1] == -2147221219:
62                   print("This info store is currently not available")
63                   continue
64           DumpFolder(folder)
