29   class TypoScriptCssDataLexer(RegexLexer):
30       """ 
31       Lexer that highlights markers, constants and registers within css blocks. 
32    
33       .. versionadded:: 2.2 
34       """
35   
36       name = 'TypoScriptCssData'
37       aliases = ['typoscriptcssdata']
38   
39       tokens = {
40           'root': [
41               # marker: ###MARK###
42               (r'(.*)(###\w+###)(.*)', bygroups(String, Name.Constant, String)),
43               # constant: {$some.constant}
44               (r'(\{)(\$)((?:[\w\-]+\.)*)([\w\-]+)(\})',
45                bygroups(String.Symbol, Operator, Name.Constant,
46                         Name.Constant, String.Symbol)),  # constant
47               # constant: {register:somevalue}
48               (r'(.*)(\{)([\w\-]+)(\s*:\s*)([\w\-]+)(\})(.*)',
49                bygroups(String, String.Symbol, Name.Constant, Operator,
50                         Name.Constant, String.Symbol, String)),  # constant
51               # whitespace
52               (r'\s+', Text),
53               # comments
54               (r'/\*(?:(?!\*/).)*\*/', Comment),
55               (r'(?<!(#|\'|"))(?:#(?!(?:[a-fA-F0-9]{6}|[a-fA-F0-9]{3}))[^\n#]+|//[^\n]*)',
56                Comment),
57               # other
58               (r'[<>,:=.*%+|]', String),
59               (r'[\w"\-!/&;(){}]+', String),
60           ]
61       }

64   class TypoScriptHtmlDataLexer(RegexLexer):
65       """ 
66       Lexer that highlights markers, constants and registers within html tags. 
67    
68       .. versionadded:: 2.2 
69       """
70   
71       name = 'TypoScriptHtmlData'
72       aliases = ['typoscripthtmldata']
73   
74       tokens = {
75           'root': [
76               # INCLUDE_TYPOSCRIPT
77               (r'(INCLUDE_TYPOSCRIPT)', Name.Class),
78               # Language label or extension resource FILE:... or LLL:... or EXT:...
79               (r'(EXT|FILE|LLL):[^}\n"]*', String),
80               # marker: ###MARK###
81               (r'(.*)(###\w+###)(.*)', bygroups(String, Name.Constant, String)),
82               # constant: {$some.constant}
83               (r'(\{)(\$)((?:[\w\-]+\.)*)([\w\-]+)(\})',
84                bygroups(String.Symbol, Operator, Name.Constant,
85                         Name.Constant, String.Symbol)),  # constant
86               # constant: {register:somevalue}
87               (r'(.*)(\{)([\w\-]+)(\s*:\s*)([\w\-]+)(\})(.*)',
88                bygroups(String, String.Symbol, Name.Constant, Operator,
89                         Name.Constant, String.Symbol, String)),  # constant
90               # whitespace
91               (r'\s+', Text),
92               # other
93               (r'[<>,:=.*%+|]', String),
94               (r'[\w"\-!/&;(){}#]+', String),
95           ]
96       }

99   class TypoScriptLexer(RegexLexer):
100      """ 
101      Lexer for TypoScript code. 
102   
103      http://docs.typo3.org/typo3cms/TyposcriptReference/ 
104   
105      .. versionadded:: 2.2 
106      """
107  
108      name = 'TypoScript'
109      aliases = ['typoscript']
110      filenames = ['*.typoscript']
111      mimetypes = ['text/x-typoscript']
112  
113      flags = re.DOTALL | re.MULTILINE
114  
115      tokens = {
116          'root': [
117              include('comment'),
118              include('constant'),
119              include('html'),
120              include('label'),
121              include('whitespace'),
122              include('keywords'),
123              include('punctuation'),
124              include('operator'),
125              include('structure'),
126              include('literal'),
127              include('other'),
128          ],
129          'keywords': [
130              # Conditions
131              (r'(?i)(\[)(browser|compatVersion|dayofmonth|dayofweek|dayofyear|'
132               r'device|ELSE|END|GLOBAL|globalString|globalVar|hostname|hour|IP|'
133               r'language|loginUser|loginuser|minute|month|page|PIDinRootline|'
134               r'PIDupinRootline|system|treeLevel|useragent|userFunc|usergroup|'
135               r'version)([^\]]*)(\])',
136               bygroups(String.Symbol, Name.Constant, Text, String.Symbol)),
137              # Functions
138              (r'(?=[\w\-])(HTMLparser|HTMLparser_tags|addParams|cache|encapsLines|'
139               r'filelink|if|imageLinkWrap|imgResource|makelinks|numRows|numberFormat|'
140               r'parseFunc|replacement|round|select|split|stdWrap|strPad|tableStyle|'
141               r'tags|textStyle|typolink)(?![\w\-])', Name.Function),
142              # Toplevel objects and _*
143              (r'(?:(=?\s*<?\s+|^\s*))(cObj|field|config|content|constants|FEData|'
144               r'file|frameset|includeLibs|lib|page|plugin|register|resources|sitemap|'
145               r'sitetitle|styles|temp|tt_[^:.\s]*|types|xmlnews|INCLUDE_TYPOSCRIPT|'
146               r'_CSS_DEFAULT_STYLE|_DEFAULT_PI_VARS|_LOCAL_LANG)(?![\w\-])',
147               bygroups(Operator, Name.Builtin)),
148              # Content objects
149              (r'(?=[\w\-])(CASE|CLEARGIF|COA|COA_INT|COBJ_ARRAY|COLUMNS|CONTENT|'
150               r'CTABLE|EDITPANEL|FILE|FILES|FLUIDTEMPLATE|FORM|HMENU|HRULER|HTML|'
151               r'IMAGE|IMGTEXT|IMG_RESOURCE|LOAD_REGISTER|MEDIA|MULTIMEDIA|OTABLE|'
152               r'PAGE|QTOBJECT|RECORDS|RESTORE_REGISTER|SEARCHRESULT|SVG|SWFOBJECT|'
153               r'TEMPLATE|TEXT|USER|USER_INT)(?![\w\-])', Name.Class),
154              # Menu states
155              (r'(?=[\w\-])(ACTIFSUBRO|ACTIFSUB|ACTRO|ACT|CURIFSUBRO|CURIFSUB|CURRO|'
156               r'CUR|IFSUBRO|IFSUB|NO|SPC|USERDEF1RO|USERDEF1|USERDEF2RO|USERDEF2|'
157               r'USRRO|USR)', Name.Class),
158              # Menu objects
159              (r'(?=[\w\-])(GMENU_FOLDOUT|GMENU_LAYERS|GMENU|IMGMENUITEM|IMGMENU|'
160               r'JSMENUITEM|JSMENU|TMENUITEM|TMENU_LAYERS|TMENU)', Name.Class),
161              # PHP objects
162              (r'(?=[\w\-])(PHP_SCRIPT(_EXT|_INT)?)', Name.Class),
163              (r'(?=[\w\-])(userFunc)(?![\w\-])', Name.Function),
164          ],
165          'whitespace': [
166              (r'\s+', Text),
167          ],
168          'html': [
169              (r'<\S[^\n>]*>', using(TypoScriptHtmlDataLexer)),
170              (r'&[^;\n]*;', String),
171              (r'(?s)(_CSS_DEFAULT_STYLE)(\s*)(\()(.*(?=\n\)))',
172               bygroups(Name.Class, Text, String.Symbol, using(TypoScriptCssDataLexer))),
173          ],
174          'literal': [
175              (r'0x[0-9A-Fa-f]+t?', Number.Hex),
176              # (r'[0-9]*\.[0-9]+([eE][0-9]+)?[fd]?\s*(?:[^=])', Number.Float),
177              (r'[0-9]+', Number.Integer),
178              (r'(###\w+###)', Name.Constant),
179          ],
180          'label': [
181              # Language label or extension resource FILE:... or LLL:... or EXT:...
182              (r'(EXT|FILE|LLL):[^}\n"]*', String),
183              # Path to a resource
184              (r'(?![^\w\-])([\w\-]+(?:/[\w\-]+)+/?)(\S*\n)',
185               bygroups(String, String)),
186          ],
187          'punctuation': [
188              (r'[,.]', Punctuation),
189          ],
190          'operator': [
191              (r'[<>,:=.*%+|]', Operator),
192          ],
193          'structure': [
194              # Brackets and braces
195              (r'[{}()\[\]\\]', String.Symbol),
196          ],
197          'constant': [
198              # Constant: {$some.constant}
199              (r'(\{)(\$)((?:[\w\-]+\.)*)([\w\-]+)(\})',
200                  bygroups(String.Symbol, Operator, Name.Constant,
201                           Name.Constant, String.Symbol)),  # constant
202              # Constant: {register:somevalue}
203              (r'(\{)([\w\-]+)(\s*:\s*)([\w\-]+)(\})',
204                  bygroups(String.Symbol, Name.Constant, Operator,
205                           Name.Constant, String.Symbol)),  # constant
206              # Hex color: #ff0077
207              (r'(#[a-fA-F0-9]{6}\b|#[a-fA-F0-9]{3}\b)', String.Char)
208          ],
209          'comment': [
210              (r'(?<!(#|\'|"))(?:#(?!(?:[a-fA-F0-9]{6}|[a-fA-F0-9]{3}))[^\n#]+|//[^\n]*)',
211               Comment),
212              (r'/\*(?:(?!\*/).)*\*/', Comment),
213              (r'(\s*#\s*\n)', Comment),
214          ],
215          'other': [
216              (r'[\w"\-!/&;]+', Text),
217          ],
218      }
