37   class ExceptionHandler(object):
38       '''Base handler that catches exceptions in :func:`runTouchApp`. 
39       You can subclass and extend it as follows:: 
40    
41           class E(ExceptionHandler): 
42               def handle_exception(self, inst): 
43                   Logger.exception('Exception caught by ExceptionHandler') 
44                   return ExceptionManager.PASS 
45    
46           ExceptionManager.add_handler(E()) 
47    
48       Then, all exceptions will be set to PASS, and logged to the console! 
49       '''
50   
51       def handle_exception(self, exception):
52           '''Called by :class:`ExceptionManagerBase` to handle a exception. 
53    
54           Defaults to returning :attr:`ExceptionManager.RAISE` that re-raises the 
55           exception. Return :attr:`ExceptionManager.PASS` to indicate that the 
56           exception was handled and should be ignored. 
57    
58           This may be called multiple times with the same exception, if 
59           :attr:`ExceptionManager.RAISE` is returned as the exception bubbles 
60           through multiple kivy exception handling levels. 
61           '''
62           return ExceptionManager.RAISE

65   class ExceptionManagerBase:
66       '''ExceptionManager manages exceptions handlers.'''
67   
68       RAISE = 0
69       """The exception should be re-raised. 
70       """
71       PASS = 1
72       """The exception should be ignored as it was handled by the handler. 
73       """
74   
75       def __init__(self):
76           self.handlers = []
77           self.policy = ExceptionManagerBase.RAISE
78   
79       def add_handler(self, cls):
80           '''Add a new exception handler to the stack.'''
81           if cls not in self.handlers:
82               self.handlers.append(cls)
83   
84       def remove_handler(self, cls):
85           '''Remove the exception handler from the stack.'''
86           if cls in self.handlers:
87               self.handlers.remove(cls)
88   
89       def handle_exception(self, inst):
90           '''Called when an exception occurred in the :func:`runTouchApp` 
91           main loop.'''
92           ret = self.policy
93           for handler in self.handlers:
94               r = handler.handle_exception(inst)
95               if r == ExceptionManagerBase.PASS:
96                   ret = r
97           return ret

107  class EventLoopBase(EventDispatcher):
108      '''Main event loop. This loop handles the updating of input and 
109      dispatching events. 
110      '''
111  
112      __events__ = ('on_start', 'on_pause', 'on_stop')
113  
114      def __init__(self):
115          super(EventLoopBase, self).__init__()
116          self.quit = False
117          self.input_events = []
118          self.postproc_modules = []
119          self.status = 'idle'
120          self.stopping = False
121          self.input_providers = []
122          self.input_providers_autoremove = []
123          self.event_listeners = []
124          self.window = None
125          self.me_list = []
126  
127      @property
128      def touches(self):
129          '''Return the list of all touches currently in down or move states. 
130          '''
131          return self.me_list
132  
133      def ensure_window(self):
134          '''Ensure that we have a window. 
135          '''
136          import kivy.core.window  # NOQA
137          if not self.window:
138              Logger.critical('App: Unable to get a Window, abort.')
139              sys.exit(1)
140  
141      def set_window(self, window):
142          '''Set the window used for the event loop. 
143          '''
144          self.window = window
145  
146      def add_input_provider(self, provider, auto_remove=False):
147          '''Add a new input provider to listen for touch events. 
148          '''
149          if provider not in self.input_providers:
150              self.input_providers.append(provider)
151              if auto_remove:
152                  self.input_providers_autoremove.append(provider)
153  
154      def remove_input_provider(self, provider):
155          '''Remove an input provider. 
156          '''
157          if provider in self.input_providers:
158              self.input_providers.remove(provider)
159  
160      def add_event_listener(self, listener):
161          '''Add a new event listener for getting touch events. 
162          '''
163          if listener not in self.event_listeners:
164              self.event_listeners.append(listener)
165  
166      def remove_event_listener(self, listener):
167          '''Remove an event listener from the list. 
168          '''
169          if listener in self.event_listeners:
170              self.event_listeners.remove(listener)
171  
172      def start(self):
173          '''Must be called only once before :meth:`EventLoopBase.run()`. 
174          This starts all configured input providers.'''
175          self.status = 'started'
176          self.quit = False
177          Clock.start_clock()
178          for provider in self.input_providers:
179              provider.start()
180          self.dispatch('on_start')
181  
182      def close(self):
183          '''Exit from the main loop and stop all configured 
184          input providers.'''
185          self.quit = True
186          self.stop()
187          self.status = 'closed'
188  
189      def stop(self):
190          '''Stop all input providers and call callbacks registered using 
191          `EventLoop.add_stop_callback()`.'''
192  
193          # XXX stop in reverse order that we started them!! (like push
194          # pop), very important because e.g. wm_touch and WM_PEN both
195          # store old window proc and the restore, if order is messed big
196          # problem happens, crashing badly without error
197          for provider in reversed(self.input_providers[:]):
198              provider.stop()
199              if provider in self.input_providers_autoremove:
200                  self.input_providers_autoremove.remove(provider)
201                  self.input_providers.remove(provider)
202  
203          # ensure any restart will not break anything later.
204          self.input_events = []
205  
206          Clock.stop_clock()
207          self.stopping = False
208          self.status = 'stopped'
209          self.dispatch('on_stop')
210  
211      def add_postproc_module(self, mod):
212          '''Add a postproc input module (DoubleTap, TripleTap, DeJitter 
213          RetainTouch are defaults).'''
214          if mod not in self.postproc_modules:
215              self.postproc_modules.append(mod)
216  
217      def remove_postproc_module(self, mod):
218          '''Remove a postproc module.'''
219          if mod in self.postproc_modules:
220              self.postproc_modules.remove(mod)
221  
222      def remove_android_splash(self, *args):
223          '''Remove android presplash in SDL2 bootstrap.'''
224          try:
225              from android import remove_presplash
226              remove_presplash()
227          except ImportError:
228              Logger.warning(
229                  'Base: Failed to import "android" module. '
230                  'Could not remove android presplash.')
231              return

233      def post_dispatch_input(self, etype, me):
234          '''This function is called by :meth:`EventLoopBase.dispatch_input()` 
235          when we want to dispatch an input event. The event is dispatched to 
236          all listeners and if grabbed, it's dispatched to grabbed widgets. 
237          '''
238          # update available list
239          if etype == 'begin':
240              self.me_list.append(me)
241          elif etype == 'end':
242              if me in self.me_list:
243                  self.me_list.remove(me)
244  
245          # dispatch to listeners
246          if not me.grab_exclusive_class:
247              for listener in self.event_listeners:
248                  listener.dispatch('on_motion', etype, me)
249  
250          # dispatch grabbed touch
251          me.grab_state = True
252          for _wid in me.grab_list[:]:
253  
254              # it's a weakref, call it!
255              wid = _wid()
256              if wid is None:
257                  # object is gone, stop.
258                  me.grab_list.remove(_wid)
259                  continue
260  
261              root_window = wid.get_root_window()
262              if wid != root_window and root_window is not None:
263                  me.push()
264                  w, h = root_window.system_size
265                  if platform == 'ios' or root_window._density != 1:
266                      w, h = root_window.size
267                  kheight = root_window.keyboard_height
268                  smode = root_window.softinput_mode
269                  me.scale_for_screen(w, h, rotation=root_window.rotation,
270                                      smode=smode, kheight=kheight)
271                  parent = wid.parent
272                  # and do to_local until the widget
273                  try:
274                      if parent:
275                          me.apply_transform_2d(parent.to_widget)
276                      else:
277                          me.apply_transform_2d(wid.to_widget)
278                          me.apply_transform_2d(wid.to_parent)
279                  except AttributeError:
280                      # when using inner window, an app have grab the touch
281                      # but app is removed. the touch can't access
282                      # to one of the parent. (i.e, self.parent will be None)
283                      # and BAM the bug happen.
284                      me.pop()
285                      continue
286  
287              me.grab_current = wid
288  
289              wid._context.push()
290  
291              if etype == 'begin':
292                  # don't dispatch again touch in on_touch_down
293                  # a down event are nearly uniq here.
294                  # wid.dispatch('on_touch_down', touch)
295                  pass
296              elif etype == 'update':
297                  if wid._context.sandbox:
298                      with wid._context.sandbox:
299                          wid.dispatch('on_touch_move', me)
300                  else:
301                      wid.dispatch('on_touch_move', me)
302  
303              elif etype == 'end':
304                  if wid._context.sandbox:
305                      with wid._context.sandbox:
306                          wid.dispatch('on_touch_up', me)
307                  else:
308                      wid.dispatch('on_touch_up', me)
309  
310              wid._context.pop()
311  
312              me.grab_current = None
313  
314              if wid != root_window and root_window is not None:
315                  me.pop()
316          me.grab_state = False
317  
318      def _dispatch_input(self, *ev):
319          # remove the save event for the touch if exist
320          if ev in self.input_events:
321              self.input_events.remove(ev)
322          self.input_events.append(ev)
323  
324      def dispatch_input(self):
325          '''Called by :meth:`EventLoopBase.idle()` to read events from input 
326          providers, pass events to postproc, and dispatch final events. 
327          '''
328  
329          # first, acquire input events
330          for provider in self.input_providers:
331              provider.update(dispatch_fn=self._dispatch_input)
332  
333          # execute post-processing modules
334          for mod in self.postproc_modules:
335              self.input_events = mod.process(events=self.input_events)
336  
337          # real dispatch input
338          input_events = self.input_events
339          pop = input_events.pop
340          post_dispatch_input = self.post_dispatch_input
341          while input_events:
342              post_dispatch_input(*pop(0))
343  
344      def mainloop(self):
345          while not self.quit and self.status == 'started':
346              try:
347                  self.idle()
348                  if self.window:
349                      self.window.mainloop()
350              except BaseException as inst:
351                  # use exception manager first
352                  r = ExceptionManager.handle_exception(inst)
353                  if r == ExceptionManager.RAISE:
354                      stopTouchApp()
355                      raise
356                  else:
357                      pass
358  
359      async def async_mainloop(self):
360          while not self.quit and self.status == 'started':
361              try:
362                  await self.async_idle()
363                  if self.window:
364                      self.window.mainloop()
365              except BaseException as inst:
366                  # use exception manager first
367                  r = ExceptionManager.handle_exception(inst)
368                  if r == ExceptionManager.RAISE:
369                      stopTouchApp()
370                      raise
371                  else:
372                      pass
373  
374          Logger.info("Window: exiting mainloop and closing.")
375          self.close()

377      def idle(self):
378          '''This function is called after every frame. By default: 
379   
380             * it "ticks" the clock to the next frame. 
381             * it reads all input and dispatches events. 
382             * it dispatches `on_update`, `on_draw` and `on_flip` events to the 
383               window. 
384          '''
385  
386          # update dt
387          Clock.tick()
388  
389          # read and dispatch input from providers
390          if not self.quit:
391              self.dispatch_input()
392  
393          # flush all the canvas operation
394          if not self.quit:
395              Builder.sync()
396  
397          # tick before draw
398          if not self.quit:
399              Clock.tick_draw()
400  
401          # flush all the canvas operation
402          if not self.quit:
403              Builder.sync()
404  
405          if not self.quit:
406              window = self.window
407              if window and window.canvas.needs_redraw:
408                  window.dispatch('on_draw')
409                  window.dispatch('on_flip')
410  
411          # don't loop if we don't have listeners !
412          if len(self.event_listeners) == 0:
413              Logger.error('Base: No event listeners have been created')
414              Logger.error('Base: Application will leave')
415              self.exit()
416              return False
417  
418          return self.quit
419  
420      async def async_idle(self):
421          '''Identical to :meth:`idle`, but instead used when running 
422          within an async event loop. 
423          '''
424  
425          # update dt
426          await Clock.async_tick()
427  
428          # read and dispatch input from providers
429          if not self.quit:
430              self.dispatch_input()
431  
432          # flush all the canvas operation
433          if not self.quit:
434              Builder.sync()
435  
436          # tick before draw
437          if not self.quit:
438              Clock.tick_draw()
439  
440          # flush all the canvas operation
441          if not self.quit:
442              Builder.sync()
443  
444          if not self.quit:
445              window = self.window
446              if window and window.canvas.needs_redraw:
447                  window.dispatch('on_draw')
448                  window.dispatch('on_flip')
449  
450          # don't loop if we don't have listeners !
451          if len(self.event_listeners) == 0:
452              Logger.error('Base: No event listeners have been created')
453              Logger.error('Base: Application will leave')
454              self.exit()
455              return False
456  
457          return self.quit
458  
459      def run(self):
460          '''Main loop'''
461          while not self.quit:
462              self.idle()
463          self.exit()
464  
465      def exit(self):
466          '''Close the main loop and close the window.'''
467          self.close()
468          if self.window:
469              self.window.close()

491  def _runTouchApp_prepare(widget=None):
492      from kivy.input import MotionEventFactory, kivy_postproc_modules
493  
494      # Ok, we got one widget, and we are not in embedded mode
495      # so, user don't create the window, let's create it for him !
496      if widget:
497          EventLoop.ensure_window()
498  
499      # Instance all configured input
500      for key, value in Config.items('input'):
501          Logger.debug('Base: Create provider from %s' % (str(value)))
502  
503          # split value
504          args = str(value).split(',', 1)
505          if len(args) == 1:
506              args.append('')
507          provider_id, args = args
508          provider = MotionEventFactory.get(provider_id)
509          if provider is None:
510              Logger.warning('Base: Unknown <%s> provider' % str(provider_id))
511              continue
512  
513          # create provider
514          p = provider(key, args)
515          if p:
516              EventLoop.add_input_provider(p, True)
517  
518      # add postproc modules
519      for mod in list(kivy_postproc_modules.values()):
520          EventLoop.add_postproc_module(mod)
521  
522      # add main widget
523      if widget and EventLoop.window:
524          if widget not in EventLoop.window.children:
525              EventLoop.window.add_widget(widget)
526  
527      # start event loop
528      Logger.info('Base: Start application main loop')
529      EventLoop.start()
530  
531      # remove presplash on the next frame
532      if platform == 'android':
533          Clock.schedule_once(EventLoop.remove_android_splash)
534  
535      # in non-embedded mode, there are 2 issues
536      #
537      # 1. if user created a window, call the mainloop from window.
538      #    This is due to glut, it need to be called with
539      #    glutMainLoop(). Only FreeGLUT got a gluMainLoopEvent().
540      #    So, we are executing the dispatching function inside
541      #    a redisplay event.
542      #
543      # 2. if no window is created, we are dispatching event loop
544      #    ourself (previous behavior.)
545      #
546

548  def runTouchApp(widget=None, embedded=False):
549      '''Static main function that starts the application loop. 
550      You can access some magic via the following arguments: 
551   
552      See :mod:`kivy.app` for example usage. 
553   
554      :Parameters: 
555          `<empty>` 
556              To make dispatching work, you need at least one 
557              input listener. If not, application will leave. 
558              (MTWindow act as an input listener) 
559   
560          `widget` 
561              If you pass only a widget, a MTWindow will be created 
562              and your widget will be added to the window as the root 
563              widget. 
564   
565          `embedded` 
566              No event dispatching is done. This will be your job. 
567   
568          `widget + embedded` 
569              No event dispatching is done. This will be your job but 
570              we try to get the window (must be created by you beforehand) 
571              and add the widget to it. Very useful for embedding Kivy 
572              in another toolkit. (like Qt, check kivy-designed) 
573   
574      '''
575      _runTouchApp_prepare(widget=widget)
576  
577      # we are in embedded mode, don't do dispatching.
578      if embedded:
579          return
580  
581      try:
582          EventLoop.mainloop()
583      finally:
584          stopTouchApp()

587  async def async_runTouchApp(widget=None, embedded=False, async_lib=None):
588      '''Identical to :func:`runTouchApp` but instead it is a coroutine 
589      that can be run in an existing async event loop. 
590   
591      ``async_lib`` is the async library to use. See :mod:`kivy.app` for details 
592      and example usage. 
593   
594      .. versionadded:: 2.0.0 
595      '''
596      if async_lib is not None:
597          Clock.init_async_lib(async_lib)
598      _runTouchApp_prepare(widget=widget)
599  
600      # we are in embedded mode, don't do dispatching.
601      if embedded:
602          return
603  
604      try:
605          await EventLoop.async_mainloop()
606      finally:
607          stopTouchApp()

610  def stopTouchApp():
611      '''Stop the current application by leaving the main loop. 
612   
613      See :mod:`kivy.app` for example usage. 
614      '''
615      if EventLoop is None:
616          return
617      if EventLoop.status in ('stopped', 'closed'):
618          return
619      if EventLoop.status != 'started':
620          if not EventLoop.stopping:
621              EventLoop.stopping = True
622              Clock.schedule_once(lambda dt: stopTouchApp(), 0)
623          return
624      Logger.info('Base: Leaving application in progress...')
625      EventLoop.close()
