1    ''' 
2    Ignore list 
3    =========== 
4     
5    Ignore touch on some areas of the screen 
6    '''
7    
8    __all__ = ('InputPostprocIgnoreList', )
9    
10   from kivy.config import Config
11   from kivy.utils import strtotuple
12   
13   
14   class InputPostprocIgnoreList(object):
15       ''' 
16       InputPostprocIgnoreList is a post-processor which removes touches in the 
17       Ignore list. The Ignore list can be configured in the Kivy config file:: 
18    
19           [postproc] 
20           # Format: [(xmin, ymin, xmax, ymax), ...] 
21           ignore = [(0.1, 0.1, 0.15, 0.15)] 
22    
23       The Ignore list coordinates are in the range 0-1, not in screen pixels. 
24       '''
25   
26       def __init__(self):
27           self.ignore_list = strtotuple(Config.get('postproc', 'ignore'))
28   
29       def collide_ignore(self, touch):
30           x, y = touch.sx, touch.sy
31           for l in self.ignore_list:
32               xmin, ymin, xmax, ymax = l
33               if x > xmin and x < xmax and y > ymin and y < ymax:
34                   return True
35   
36       def process(self, events):
37           if not len(self.ignore_list):
38               return events
39           for etype, touch in events:
40               if not touch.is_touch:
41                   continue
42               if etype != 'begin':
43                   continue
44               if self.collide_ignore(touch):
45                   touch.ud.__pp_ignore__ = True
46           return [(etype, touch) for etype, touch in events
47                   if '__pp_ignore__' not in touch.ud]
