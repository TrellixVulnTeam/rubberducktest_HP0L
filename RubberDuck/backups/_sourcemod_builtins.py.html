1101     from pygments.util import format_lines

1103     # urllib ends up wanting to import a module called 'math' -- if
1104     # pygments/lexers is in the path, this ends badly.
1105     for i in range(len(sys.path)-1, -1, -1):
1106         if sys.path[i].endswith('/lexers'):
1107             del sys.path[i]
1108 
1109     class Opener(FancyURLopener):
1110         version = 'Mozilla/5.0 (Pygments Sourcemod Builtins Update)'
1111 
1112     opener = Opener()
1113 
1114     def get_version():
1115         f = opener.open('http://docs.sourcemod.net/api/index.php')
1116         r = re.compile(r'SourceMod v\.<b>([\d\.]+(?:-\w+)?)</td>')
1117         for line in f:
1118             m = r.search(line)
1119             if m is not None:
1120                 return m.groups()[0]
1121         raise ValueError('No version in api docs')
1122 
1123     def get_sm_functions():
1124         f = opener.open('http://docs.sourcemod.net/api/SMfuncs.js')
1125         r = re.compile(r'SMfunctions\[\d+\] = Array \("(?:public )?([^,]+)",".+"\);')
1126         functions = []
1127         for line in f:
1128             m = r.match(line)
1129             if m is not None:
1130                 functions.append(m.groups()[0])
1131         return functions
1132 
1133     def regenerate(filename, natives):
1134         with open(filename) as fp:
1135             content = fp.read()
1136 
1137         header = content[:content.find('FUNCTIONS = (')]
1138         footer = content[content.find("if __name__ == '__main__':")-1:]
1139 
1140 
1141         with open(filename, 'w') as fp:
1142             fp.write(header)
1143             fp.write(format_lines('FUNCTIONS', natives))
1144             fp.write(footer)
1145 
1146     def run():
1147         version = get_version()
1148         print('> Downloading function index for SourceMod %s' % version)
1149         functions = get_sm_functions()
1150         print('> %d functions found:' % len(functions))
1151 
1152         functionlist = []
1153         for full_function_name in functions:
1154             print('>> %s' % full_function_name)
1155             functionlist.append(full_function_name)
1156 
1157         regenerate(__file__, functionlist)
