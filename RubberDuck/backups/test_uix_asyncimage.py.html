22       def setUp(self):
23           self.maxfps = Config.getint('graphics', 'maxfps')
24           assert(self.maxfps > 0)
25           super(AsyncImageTestCase, self).setUp()

27       def zip_frames(self, path):
28           with ZipFile(path) as zipf:
29               return len(zipf.namelist())

31       def load_zipimage(self, source, frames):
32           # keep running the test until loaded (or timeout)
33           maxfps = self.maxfps
34           timeout = 30 * maxfps
35   
36           # load ZIP with images named: 000.png, 001.png, ...
37           image = AsyncImage(
38               source=source,
39               anim_delay=0.0333333333333333
40           )
41           image.test_loaded = False
42           self.render(image)
43   
44           # bind to 'on_load' because there are various
45           # steps where the image is (re)loaded, but
46           # the event is triggered only at the end
47           image.bind(on_load=lambda *_, **__: setattr(
48               image, 'test_loaded', True
49           ))
50   
51           while timeout and not image.test_loaded:
52               self.advance_frames(1)
53               timeout -= 1
54   
55           proxyimg = image._coreimage
56           self.assertTrue(proxyimg.anim_available)
57           self.assertEqual(len(proxyimg.image.textures), frames)
58           return image

60       def test_remote_zipsequence(self):
61           # cube ZIP has 63 PNGs used for animation
62           ZIP = (
63               'http://localhost:8000/widgets/'
64               'sequenced_images/data/images/cube.zip'
65           )
66   
67           # ref Loader._load_urllib
68           tempf, headers = urlretrieve(ZIP)
69           ZIP_pngs = self.zip_frames(tempf)
70           remove(tempf)
71   
72           image = self.load_zipimage(ZIP, ZIP_pngs)
73           # pure delay * fps isn't enough and
74           # just +1 isn't either (index collisions)
75           self.assertTrue(self.check_sequence_frames(
76               image._coreimage,
77               int(image._coreimage.anim_delay * self.maxfps + 3)
78           ))

96       def check_sequence_frames(self, img, frames, slides=5):
97           # check whether it really changes the images
98           # in the anim_delay interval
99           old = None
100  
101          while slides:
102              # different frames, sequence is changing
103              self.assertNotEqual(img.anim_index, old)
104  
105              old = img.anim_index
106              self.advance_frames(
107                  frames
108              )
109              slides -= 1
110  
111          return True

113      def test_reload_asyncimage(self):
114          from kivy.resources import resource_find
115          from tempfile import mkdtemp
116          from os import remove
117          from shutil import copyfile, rmtree
118  
119          fn = resource_find('data/logo/kivy-icon-16.png')
120          t = mkdtemp()
121          source = join(t, 'source.png')
122          copyfile(fn, source)
123          image = AsyncImage(source=source)
124          self.render(image, framecount=2)
125          self.assertEqual(image.texture_size, [16, 16])
126          remove(source)
127  
128          fn = resource_find('data/logo/kivy-icon-32.png')
129          copyfile(fn, source)
130          image.reload()
131          self.render(image, framecount=2)
132          self.assertEqual(image.texture_size, [32, 32])
133          remove(source)
