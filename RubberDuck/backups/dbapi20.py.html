427          self.executeDDL2(cur)

440          elif self.driver.paramstyle == "numeric":
441              cur.execute(
442                  "insert into %sbarflys values (:1, 'thi%%s :may ca%%(u)se? troub:1e')"

447              cur.execute(

453              cur.execute(
454                  "insert into %sbarflys values (%%s, 'thi%%s :may ca%%(u)se? troub:1e')"
