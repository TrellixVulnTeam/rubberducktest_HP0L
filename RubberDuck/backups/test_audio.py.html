6    import unittest
7    import os
8    import pytest
9    if os.environ.get('KIVY_TEST_AUDIO') == '0':
10       pytestmark = pytest.mark.skip("Audio is not available")
11   
12   SAMPLE_FILE = os.path.join(os.path.dirname(__file__), 'sample1.ogg')
13   SAMPLE_LENGTH = 1.402
14   DELTA = SAMPLE_LENGTH * 0.01
15   DELAY = 0.2
16   
17   
18   class AudioTestCase(unittest.TestCase):
19   
20       def get_sound(self):
21           import os
22           assert os.path.exists(SAMPLE_FILE)
23           from kivy.core import audio
24           return audio.SoundLoader.load(SAMPLE_FILE)
25   
26       def test_length_simple(self):
27           sound = self.get_sound()
28           volume = sound.volume = 0.75
29           length = sound.length
30           self.assertAlmostEqual(SAMPLE_LENGTH, length, delta=DELTA)
31           # ensure that the gstreamer play/stop doesn't mess up the volume
32           assert volume == sound.volume
33   
34       def test_length_playing(self):
35           import time
36           sound = self.get_sound()
37           sound.play()
38           try:
39               time.sleep(DELAY)
40               length = sound.length
41               self.assertAlmostEqual(SAMPLE_LENGTH, length, delta=DELTA)
42           finally:
43               sound.stop()
44           self.assertAlmostEqual(SAMPLE_LENGTH, length, delta=DELTA)
45   
46       def test_length_stopped(self):
47           import time
48           sound = self.get_sound()
49           sound.play()
50           try:
51               time.sleep(DELAY)
52           finally:
53               sound.stop()
54           length = sound.length
55           self.assertAlmostEqual(SAMPLE_LENGTH, length, delta=DELTA)
56   
57   
58   class AudioGstreamerTestCase(AudioTestCase):
59   
60       def make_sound(self, source):
61           from kivy.core.audio import audio_gstreamer
62           return audio_gstreamer.SoundGstreamer(source)
63   
64   
65   class AudioPygameTestCase(AudioTestCase):
66
