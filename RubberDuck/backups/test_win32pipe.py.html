15   class PipeTests(unittest.TestCase):
16       pipename = "\\\\.\\pipe\\python_test_pipe"

18       def _serverThread(self, pipe_handle, event, wait_time):
19           # just do one connection and terminate.
20           hr = win32pipe.ConnectNamedPipe(pipe_handle)
21           self.failUnless(
22               hr in (0, winerror.ERROR_PIPE_CONNECTED), "Got error code 0x%x" % (hr,)
23           )
24           hr, got = win32file.ReadFile(pipe_handle, 100)
25           self.failUnlessEqual(got, str2bytes("foo\0bar"))
26           time.sleep(wait_time)
27           win32file.WriteFile(pipe_handle, str2bytes("bar\0foo"))
28           pipe_handle.Close()
29           event.set()
30   
31       def startPipeServer(self, event, wait_time=0):
32           openMode = win32pipe.PIPE_ACCESS_DUPLEX
33           pipeMode = win32pipe.PIPE_TYPE_MESSAGE | win32pipe.PIPE_WAIT
34   
35           sa = pywintypes.SECURITY_ATTRIBUTES()
36           sa.SetSecurityDescriptorDacl(1, None, 0)
37   
38           pipe_handle = win32pipe.CreateNamedPipe(
39               self.pipename,
40               openMode,
41               pipeMode,
42               win32pipe.PIPE_UNLIMITED_INSTANCES,
43               0,
44               0,
45               2000,
46               sa,
47           )
48   
49           threading.Thread(
50               target=self._serverThread, args=(pipe_handle, event, wait_time)
51           ).start()
52   
53       def testCallNamedPipe(self):
54           event = threading.Event()
55           self.startPipeServer(event)
56   
57           got = win32pipe.CallNamedPipe(
58               self.pipename, str2bytes("foo\0bar"), 1024, win32pipe.NMPWAIT_WAIT_FOREVER
59           )
60           self.failUnlessEqual(got, str2bytes("bar\0foo"))
61           event.wait(5)
62           self.failUnless(event.isSet(), "Pipe server thread didn't terminate")

64       def testTransactNamedPipeBlocking(self):
65           event = threading.Event()
66           self.startPipeServer(event)
67           open_mode = win32con.GENERIC_READ | win32con.GENERIC_WRITE
68   
69           hpipe = win32file.CreateFile(
70               self.pipename,
71               open_mode,
72               0,  # no sharing
73               None,  # default security
74               win32con.OPEN_EXISTING,
75               0,  # win32con.FILE_FLAG_OVERLAPPED,
76               None,
77           )
78   
79           # set to message mode.
80           win32pipe.SetNamedPipeHandleState(
81               hpipe, win32pipe.PIPE_READMODE_MESSAGE, None, None
82           )
83   
84           hr, got = win32pipe.TransactNamedPipe(hpipe, str2bytes("foo\0bar"), 1024, None)
85           self.failUnlessEqual(got, str2bytes("bar\0foo"))
86           event.wait(5)
87           self.failUnless(event.isSet(), "Pipe server thread didn't terminate")

119      def testTransactNamedPipeAsync(self):
120          event = threading.Event()
121          overlapped = pywintypes.OVERLAPPED()
122          overlapped.hEvent = win32event.CreateEvent(None, 0, 0, None)
123          self.startPipeServer(event, 0.5)
124          open_mode = win32con.GENERIC_READ | win32con.GENERIC_WRITE
125  
126          hpipe = win32file.CreateFile(
127              self.pipename,
128              open_mode,
129              0,  # no sharing
130              None,  # default security
131              win32con.OPEN_EXISTING,
132              win32con.FILE_FLAG_OVERLAPPED,
133              None,
134          )
135  
136          # set to message mode.
137          win32pipe.SetNamedPipeHandleState(
138              hpipe, win32pipe.PIPE_READMODE_MESSAGE, None, None
139          )
140  
141          buffer = win32file.AllocateReadBuffer(1024)
142          hr, got = win32pipe.TransactNamedPipe(
143              hpipe, str2bytes("foo\0bar"), buffer, overlapped
144          )
145          self.failUnlessEqual(hr, winerror.ERROR_IO_PENDING)
146          nbytes = win32file.GetOverlappedResult(hpipe, overlapped, True)
147          got = buffer[:nbytes]
148          self.failUnlessEqual(got, str2bytes("bar\0foo"))
149          event.wait(5)
150          self.failUnless(event.isSet(), "Pipe server thread didn't terminate")
