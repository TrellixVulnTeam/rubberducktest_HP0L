267  class CssLexer(RegexLexer):
268      """ 
269      For CSS (Cascading Style Sheets). 
270      """
271  
272      name = 'CSS'
273      aliases = ['css']
274      filenames = ['*.css']
275      mimetypes = ['text/css']
276  
277      tokens = {
278          'root': [
279              include('basics'),
280          ],
281          'basics': [
282              (r'\s+', Text),
283              (r'/\*(?:.|\n)*?\*/', Comment),
284              (r'\{', Punctuation, 'content'),
285              (r'(\:{1,2})([\w-]+)', bygroups(Punctuation, Name.Decorator)),
286              (r'(\.)([\w-]+)', bygroups(Punctuation, Name.Class)),
287              (r'(\#)([\w-]+)', bygroups(Punctuation, Name.Namespace)),
288              (r'(@)([\w-]+)', bygroups(Punctuation, Keyword), 'atrule'),
289              (r'[\w-]+', Name.Tag),
290              (r'[~^*!%&$\[\]()<>|+=@:;,./?-]', Operator),
291              (r'"(\\\\|\\[^\\]|[^"\\])*"', String.Double),
292              (r"'(\\\\|\\[^\\]|[^'\\])*'", String.Single),
293          ],
294          'atrule': [
295              (r'\{', Punctuation, 'atcontent'),
296              (r';', Punctuation, '#pop'),
297              include('basics'),
298          ],
299          'atcontent': [
300              include('basics'),
301              (r'\}', Punctuation, '#pop:2'),
302          ],
303          'content': [
304              (r'\s+', Text),
305              (r'\}', Punctuation, '#pop'),
306              (r';', Punctuation),
307              (r'^@.*?$', Comment.Preproc),
308  
309              (words(_vendor_prefixes,), Keyword.Pseudo),
310              (r'('+r'|'.join(_css_properties)+r')(\s*)(\:)',
311               bygroups(Keyword, Text, Punctuation), 'value-start'),
312              (r'([-]+[a-zA-Z_][\w-]*)(\s*)(\:)', bygroups(Name.Variable, Text, Punctuation),
313               'value-start'),
314              (r'([a-zA-Z_][\w-]*)(\s*)(\:)', bygroups(Name, Text, Punctuation),
315               'value-start'),
316  
317              (r'/\*(?:.|\n)*?\*/', Comment),
318          ],
319          'value-start': [
320              (r'\s+', Text),
321              (words(_vendor_prefixes,), Name.Builtin.Pseudo),
322              include('urls'),
323              (r'('+r'|'.join(_functional_notation_keyword_values)+r')(\()',
324               bygroups(Name.Builtin, Punctuation), 'function-start'),
325              (r'([a-zA-Z_][\w-]+)(\()',
326               bygroups(Name.Function, Punctuation), 'function-start'),
327              (words(_keyword_values, suffix=r'\b'), Keyword.Constant),
328              (words(_other_keyword_values, suffix=r'\b'), Keyword.Constant),
329              (words(_color_keywords, suffix=r'\b'), Keyword.Constant),
330              # for transition-property etc.
331              (words(_css_properties, suffix=r'\b'), Keyword),
332              (r'\!important', Comment.Preproc),
333              (r'/\*(?:.|\n)*?\*/', Comment),
334  
335              include('numeric-values'),
336  
337              (r'[~^*!%&<>|+=@:./?-]+', Operator),
338              (r'[\[\](),]+', Punctuation),
339              (r'"(\\\\|\\[^\\]|[^"\\])*"', String.Double),
340              (r"'(\\\\|\\[^\\]|[^'\\])*'", String.Single),
341              (r'[a-zA-Z_][\w-]*', Name),
342              (r';', Punctuation, '#pop'),
343              (r'\}', Punctuation, '#pop:2'),
344          ],
345          'function-start': [
346              (r'\s+', Text),
347              (r'[-]+([\w+]+[-]*)+', Name.Variable),
348              include('urls'),
349              (words(_vendor_prefixes,), Keyword.Pseudo),
350              (words(_keyword_values, suffix=r'\b'), Keyword.Constant),
351              (words(_other_keyword_values, suffix=r'\b'), Keyword.Constant),
352              (words(_color_keywords, suffix=r'\b'), Keyword.Constant),
353  
354              # function-start may be entered recursively
355              (r'(' + r'|'.join(_functional_notation_keyword_values) + r')(\()',
356               bygroups(Name.Builtin, Punctuation), 'function-start'),
357              (r'([a-zA-Z_][\w-]+)(\()',
358               bygroups(Name.Function, Punctuation), 'function-start'),
359  
360              (r'/\*(?:.|\n)*?\*/', Comment),
361              include('numeric-values'),
362              (r'[*+/-]', Operator),
363              (r',', Punctuation),
364              (r'"(\\\\|\\[^\\]|[^"\\])*"', String.Double),
365              (r"'(\\\\|\\[^\\]|[^'\\])*'", String.Single),
366              (r'[a-zA-Z_-]\w*', Name),
367              (r'\)', Punctuation, '#pop'),
368          ],
369          'urls': [
370              (r'(url)(\()(".*?")(\))', bygroups(Name.Builtin, Punctuation,
371                                                 String.Double, Punctuation)),
372              (r"(url)(\()('.*?')(\))", bygroups(Name.Builtin, Punctuation,
373                                                 String.Single, Punctuation)),
374              (r'(url)(\()(.*?)(\))', bygroups(Name.Builtin, Punctuation,
375                                               String.Other, Punctuation)),
376          ],
377          'numeric-values': [
378              (r'\#[a-zA-Z0-9]{1,6}', Number.Hex),
379              (r'[+\-]?[0-9]*[.][0-9]+', Number.Float, 'numeric-end'),
380              (r'[+\-]?[0-9]+', Number.Integer, 'numeric-end'),
381          ],
382          'numeric-end': [
383              (words(_all_units, suffix=r'\b'), Keyword.Type),
384              (r'%', Keyword.Type),
385              default('#pop'),
386          ],
387      }

509  def _indentation(lexer, match, ctx):
510      indentation = match.group(0)
511      yield match.start(), Text, indentation
512      ctx.last_indentation = indentation
513      ctx.pos = match.end()
514  
515      if hasattr(ctx, 'block_state') and ctx.block_state and \
516              indentation.startswith(ctx.block_indentation) and \
517              indentation != ctx.block_indentation:
518          ctx.stack.append(ctx.block_state)
519      else:
520          ctx.block_state = None
521          ctx.block_indentation = None
522          ctx.stack.append('content')
523  
524  
525  def _starts_block(token, state):
526      def callback(lexer, match, ctx):
527          yield match.start(), token, match.group(0)
528  
529          if hasattr(ctx, 'last_indentation'):
530              ctx.block_indentation = ctx.last_indentation
531          else:
532              ctx.block_indentation = ''
533  
534          ctx.block_state = state
535          ctx.pos = match.end()
536  
537      return callback
