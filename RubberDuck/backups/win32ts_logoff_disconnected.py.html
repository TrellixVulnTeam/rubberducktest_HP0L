win32ts_logoff_disconnected.py 
1    """ Finds any disconnected terminal service sessions and logs them off"""
2    import win32ts
3    import pywintypes
4    import winerror
5    
6    sessions = win32ts.WTSEnumerateSessions(win32ts.WTS_CURRENT_SERVER_HANDLE)
7    for session in sessions:
8        """ 
9        WTS_CONNECTSTATE_CLASS: WTSActive,WTSConnected,WTSConnectQuery,WTSShadow,WTSDisconnected, 
10             WTSIdle,WTSListen,WTSReset,WTSDown,WTSInit 
11       """
12       if session["State"] == win32ts.WTSDisconnected:
13           sessionid = session["SessionId"]
14           username = win32ts.WTSQuerySessionInformation(
15               win32ts.WTS_CURRENT_SERVER_HANDLE, sessionid, win32ts.WTSUserName
16           )
17           print("Logging off disconnected user:", username)
18           try:
19               win32ts.WTSLogoffSession(win32ts.WTS_CURRENT_SERVER_HANDLE, sessionid, True)
20           except pywintypes.error as e:
21               if e.winerror == winerror.ERROR_ACCESS_DENIED:
22                   print("Can't kill that session:", e.strerror)
23               else:
24                   raise
