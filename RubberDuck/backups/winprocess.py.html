1    """ 
2    Windows Process Control 
3     
4    winprocess.run launches a child process and returns the exit code. 
5    Optionally, it can: 
6      redirect stdin, stdout & stderr to files 
7      run the command as another user 
8      limit the process's running time 
9      control the process window (location, size, window state, desktop) 
10   Works on Windows NT, 2000 & XP. Requires Mark Hammond's win32 
11   extensions.

14   -- John B. Dell'Aquila <jbd@alum.mit.edu> 
15   """

17   import win32api, win32process, win32security
18   import win32event, win32con, msvcrt, win32gui
19   import os

21   
22   def logonUser(loginString):
23       """ 
24       Login as specified user and return handle. 
25       loginString:  'Domain\nUser\nPassword'; for local 
26           login use . or empty string as domain 
27           e.g. '.\nadministrator\nsecret_password' 
28       """
29       domain, user, passwd = loginString.split("\n")
30       return win32security.LogonUser(
31           user,
32           domain,
33           passwd,
34           win32con.LOGON32_LOGON_INTERACTIVE,
35           win32con.LOGON32_PROVIDER_DEFAULT,
36       )
37   
38   
39   class Process:
40       """ 
41       A Windows process. 
42       """
43   
44       def __init__(
45           self,
46           cmd,
47           login=None,
48           hStdin=None,
49           hStdout=None,
50           hStderr=None,
51           show=1,
52           xy=None,
53           xySize=None,
54           desktop=None,
55       ):
56           """ 
57           Create a Windows process. 
58           cmd:     command to run 
59           login:   run as user 'Domain\nUser\nPassword' 
60           hStdin, hStdout, hStderr: 
61                    handles for process I/O; default is caller's stdin, 
62                    stdout & stderr 
63           show:    wShowWindow (0=SW_HIDE, 1=SW_NORMAL, ...) 
64           xy:      window offset (x, y) of upper left corner in pixels 
65           xySize:  window size (width, height) in pixels 
66           desktop: lpDesktop - name of desktop e.g. 'winsta0\\default' 
67                    None = inherit current desktop 
68                    '' = create new desktop if necessary 
69    
70           User calling login requires additional privileges: 
71             Act as part of the operating system [not needed on Windows XP] 
72             Increase quotas 
73             Replace a process level token 
74           Login string must EITHER be an administrator's account 
75           (ordinary user can't access current desktop - see Microsoft 
76           Q165194) OR use desktop='' to run another desktop invisibly 
77           (may be very slow to startup & finalize). 
78           """
79           si = win32process.STARTUPINFO()
80           si.dwFlags = win32con.STARTF_USESTDHANDLES ^ win32con.STARTF_USESHOWWINDOW
81           if hStdin is None:
82               si.hStdInput = win32api.GetStdHandle(win32api.STD_INPUT_HANDLE)
83           else:
84               si.hStdInput = hStdin
85           if hStdout is None:
86               si.hStdOutput = win32api.GetStdHandle(win32api.STD_OUTPUT_HANDLE)
87           else:
88               si.hStdOutput = hStdout
89           if hStderr is None:
90               si.hStdError = win32api.GetStdHandle(win32api.STD_ERROR_HANDLE)
91           else:
92               si.hStdError = hStderr
93           si.wShowWindow = show
94           if xy is not None:
95               si.dwX, si.dwY = xy
96               si.dwFlags ^= win32con.STARTF_USEPOSITION
97           if xySize is not None:
98               si.dwXSize, si.dwYSize = xySize
99               si.dwFlags ^= win32con.STARTF_USESIZE
100          if desktop is not None:
101              si.lpDesktop = desktop
102          procArgs = (
103              None,  # appName
104              cmd,  # commandLine
105              None,  # processAttributes
106              None,  # threadAttributes
107              1,  # bInheritHandles
108              win32process.CREATE_NEW_CONSOLE,  # dwCreationFlags
109              None,  # newEnvironment
110              None,  # currentDirectory
111              si,
112          )  # startupinfo
113          if login is not None:
114              hUser = logonUser(login)
115              win32security.ImpersonateLoggedOnUser(hUser)
116              procHandles = win32process.CreateProcessAsUser(hUser, *procArgs)
117              win32security.RevertToSelf()
118          else:
119              procHandles = win32process.CreateProcess(*procArgs)
120          self.hProcess, self.hThread, self.PId, self.TId = procHandles
121  
122      def wait(self, mSec=None):
123          """ 
124          Wait for process to finish or for specified number of 
125          milliseconds to elapse. 
126          """
127          if mSec is None:
128              mSec = win32event.INFINITE
129          return win32event.WaitForSingleObject(self.hProcess, mSec)
130  
131      def kill(self, gracePeriod=5000):
132          """ 
133          Kill process. Try for an orderly shutdown via WM_CLOSE.  If 
134          still running after gracePeriod (5 sec. default), terminate. 
135          """
136          win32gui.EnumWindows(self.__close__, 0)
137          if self.wait(gracePeriod) != win32event.WAIT_OBJECT_0:
138              win32process.TerminateProcess(self.hProcess, 0)
139              win32api.Sleep(100)  # wait for resources to be released
140  
141      def __close__(self, hwnd, dummy):
142          """ 
143          EnumWindows callback - sends WM_CLOSE to any window 
144          owned by this process. 
145          """
146          TId, PId = win32process.GetWindowThreadProcessId(hwnd)
147          if PId == self.PId:
148              win32gui.PostMessage(hwnd, win32con.WM_CLOSE, 0, 0)
149  
150      def exitCode(self):
151          """ 
152          Return process exit code. 
153          """
154          return win32process.GetExitCodeProcess(self.hProcess)
155

157  def run(cmd, mSec=None, stdin=None, stdout=None, stderr=None, **kw):
158      """ 
159      Run cmd as a child process and return exit code. 
160      mSec:  terminate cmd after specified number of milliseconds 
161      stdin, stdout, stderr: 
162             file objects for child I/O (use hStdin etc. to attach 
163             handles instead of files); default is caller's stdin, 
164             stdout & stderr; 
165      kw:    see Process.__init__ for more keyword options 
166      """
167      if stdin is not None:
168          kw["hStdin"] = msvcrt.get_osfhandle(stdin.fileno())
169      if stdout is not None:
170          kw["hStdout"] = msvcrt.get_osfhandle(stdout.fileno())
171      if stderr is not None:
172          kw["hStderr"] = msvcrt.get_osfhandle(stderr.fileno())
173      child = Process(cmd, **kw)
174      if child.wait(mSec) != win32event.WAIT_OBJECT_0:
175          child.kill()
176          raise WindowsError("process timeout exceeded")
177      return child.exitCode()

205          cmd.seek(0)

207              "CMD.EXE exit code:",

212              "NOTEPAD exit code:",
213              run(
214                  "notepad.exe %s" % out.name,
215                  show=win32con.SW_MAXIMIZE,
216                  mSec=timeoutSeconds * 1000,
217              ),
218          )
219          out.close()
220      finally:
221          for n in (cmd_name, out_name):
