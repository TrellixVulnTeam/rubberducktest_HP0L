3    import win32con
4    import win32ui
5    import sys
6    import string
7    import os
8    from pywin.framework import intpyapp
9    
10   version = "0.3.0"
11   
12   
13   class DebuggerPythonApp(intpyapp.InteractivePythonApp):
14       def LoadMainFrame(self):
15           "Create the main applications frame"
16           self.frame = self.CreateMainFrame()
17           self.SetMainFrame(self.frame)
18           self.frame.LoadFrame(win32ui.IDR_DEBUGGER, win32con.WS_OVERLAPPEDWINDOW)
19           self.frame.DragAcceptFiles()  # we can accept these.
20           self.frame.ShowWindow(win32con.SW_HIDE)
21           self.frame.UpdateWindow()
22   
23           # but we do rehook, hooking the new code objects.
24           self.HookCommands()
25   
26       def InitInstance(self):
27           # Use a registry path of "Python\Pythonwin Debugger
28           win32ui.SetAppName(win32ui.LoadString(win32ui.IDR_DEBUGGER))
29           win32ui.SetRegistryKey("Python %s" % (sys.winver,))
30           # We _need_ the Scintilla color editor.
31           # (and we _always_ get it now :-)
32   
33           numMRU = win32ui.GetProfileVal("Settings", "Recent File List Size", 10)
34           win32ui.LoadStdProfileSettings(numMRU)
35   
36           self.LoadMainFrame()
37   
38           # Display the interactive window if the user wants it.
39           from pywin.framework import interact
40   
41           interact.CreateInteractiveWindowUserPreference()
42   
43           # Load the modules we use internally.
44           self.LoadSystemModules()
45           # Load additional module the user may want.
46           self.LoadUserModules()
47   
48           #       win32ui.CreateDebuggerThread()
49           win32ui.EnableControlContainer()
