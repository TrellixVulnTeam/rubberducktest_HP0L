21   class BrainfuckLexer(RegexLexer):
22       """ 
23       Lexer for the esoteric `BrainFuck <http://www.muppetlabs.com/~breadbox/bf/>`_ 
24       language. 
25       """
26   
27       name = 'Brainfuck'
28       aliases = ['brainfuck', 'bf']
29       filenames = ['*.bf', '*.b']
30       mimetypes = ['application/x-brainfuck']
31   
32       tokens = {
33           'common': [
34               # use different colors for different instruction types
35               (r'[.,]+', Name.Tag),
36               (r'[+-]+', Name.Builtin),
37               (r'[<>]+', Name.Variable),
38               (r'[^.,+\-<>\[\]]+', Comment),
39           ],
40           'root': [
41               (r'\[', Keyword, 'loop'),
42               (r'\]', Error),
43               include('common'),
44           ],
45           'loop': [
46               (r'\[', Keyword, '#push'),
47               (r'\]', Keyword, '#pop'),
48               include('common'),
49           ]
50       }
51   
52       def analyse_text(text):
53           """It's safe to assume that a program which mostly consists of + - 
54           and < > is brainfuck."""
55           plus_minus_count = 0
56           greater_less_count = 0
57   
58           range_to_check = max(256, len(text))
59   
60           for c in text[:range_to_check]:
61               if c == '+' or c == '-':
62                   plus_minus_count += 1
63               if c == '<' or c == '>':
64                   greater_less_count += 1
65   
66           if plus_minus_count > (0.25 * range_to_check):
67               return 1.0
68           if greater_less_count > (0.25 * range_to_check):
69               return 1.0
70   
71           result = 0
72           if '[-]' in text:
73               result += 0.5
74   
75           return result

78   class BefungeLexer(RegexLexer):
79       """ 
80       Lexer for the esoteric `Befunge <http://en.wikipedia.org/wiki/Befunge>`_ 
81       language. 
82    
83       .. versionadded:: 0.7 
84       """
85       name = 'Befunge'
86       aliases = ['befunge']
87       filenames = ['*.befunge']
88       mimetypes = ['application/x-befunge']
89   
90       tokens = {
91           'root': [
92               (r'[0-9a-f]', Number),
93               (r'[+*/%!`-]', Operator),             # Traditional math
94               (r'[<>^v?\[\]rxjk]', Name.Variable),  # Move, imperatives
95               (r'[:\\$.,n]', Name.Builtin),         # Stack ops, imperatives
96               (r'[|_mw]', Keyword),
97               (r'[{}]', Name.Tag),                  # Befunge-98 stack ops
98               (r'".*?"', String.Double),            # Strings don't appear to allow escapes
99               (r'\'.', String.Single),              # Single character
100              (r'[#;]', Comment),                   # Trampoline... depends on direction hit
101              (r'[pg&~=@iotsy]', Keyword),          # Misc
102              (r'[()A-Z]', Comment),                # Fingerprints
103              (r'\s+', Text),                       # Whitespace doesn't matter
104          ],
105      }

108  class CAmkESLexer(RegexLexer):
109      """ 
110      Basic lexer for the input language for the 
111      `CAmkES <https://sel4.systems/CAmkES/>`_ component platform. 
112   
113      .. versionadded:: 2.1 
114      """
115      name = 'CAmkES'
116      aliases = ['camkes', 'idl4']
117      filenames = ['*.camkes', '*.idl4']
118  
119      tokens = {
120          'root': [
121              # C pre-processor directive
122              (r'^\s*#.*\n', Comment.Preproc),
123  
124              # Whitespace, comments
125              (r'\s+', Text),
126              (r'/\*(.|\n)*?\*/', Comment),
127              (r'//.*\n', Comment),
128  
129              (r'[\[(){},.;\]]', Punctuation),
130              (r'[~!%^&*+=|?:<>/-]', Operator),
131  
132              (words(('assembly', 'attribute', 'component', 'composition',
133                      'configuration', 'connection', 'connector', 'consumes',
134                      'control', 'dataport', 'Dataport', 'Dataports', 'emits',
135                      'event', 'Event', 'Events', 'export', 'from', 'group',
136                      'hardware', 'has', 'interface', 'Interface', 'maybe',
137                      'procedure', 'Procedure', 'Procedures', 'provides',
138                      'template', 'thread', 'threads', 'to', 'uses', 'with'),
139                     suffix=r'\b'), Keyword),
140  
141              (words(('bool', 'boolean', 'Buf', 'char', 'character', 'double',
142                      'float', 'in', 'inout', 'int', 'int16_6', 'int32_t',
143                      'int64_t', 'int8_t', 'integer', 'mutex', 'out', 'real',
144                      'refin', 'semaphore', 'signed', 'string', 'struct',
145                      'uint16_t', 'uint32_t', 'uint64_t', 'uint8_t', 'uintptr_t',
146                      'unsigned', 'void'),
147                     suffix=r'\b'), Keyword.Type),
148  
149              # Recognised attributes
150              (r'[a-zA-Z_]\w*_(priority|domain|buffer)', Keyword.Reserved),
151              (words(('dma_pool', 'from_access', 'to_access'), suffix=r'\b'),
152                  Keyword.Reserved),
153  
154              # CAmkES-level include
155              (r'import\s+(<[^>]*>|"[^"]*");', Comment.Preproc),
156  
157              # C-level include
158              (r'include\s+(<[^>]*>|"[^"]*");', Comment.Preproc),
159  
160              # Literals
161              (r'0[xX][\da-fA-F]+', Number.Hex),
162              (r'-?[\d]+', Number),
163              (r'-?[\d]+\.[\d]+', Number.Float),
164              (r'"[^"]*"', String),
165              (r'[Tt]rue|[Ff]alse', Name.Builtin),
166  
167              # Identifiers
168              (r'[a-zA-Z_]\w*', Name),
169          ],
170      }

173  class CapDLLexer(RegexLexer):
174      """ 
175      Basic lexer for 
176      `CapDL <https://ssrg.nicta.com.au/publications/nictaabstracts/Kuz_KLW_10.abstract.pml>`_. 
177   
178      The source of the primary tool that reads such specifications is available 
179      at https://github.com/seL4/capdl/tree/master/capDL-tool. Note that this 
180      lexer only supports a subset of the grammar. For example, identifiers can 
181      shadow type names, but these instances are currently incorrectly 
182      highlighted as types. Supporting this would need a stateful lexer that is 
183      considered unnecessarily complex for now. 
184   
185      .. versionadded:: 2.2 
186      """
187      name = 'CapDL'
188      aliases = ['capdl']
189      filenames = ['*.cdl']
190  
191      tokens = {
192          'root': [
193              # C pre-processor directive
194              (r'^\s*#.*\n', Comment.Preproc),
195  
196              # Whitespace, comments
197              (r'\s+', Text),
198              (r'/\*(.|\n)*?\*/', Comment),
199              (r'(//|--).*\n', Comment),
200  
201              (r'[<>\[(){},:;=\]]', Punctuation),
202              (r'\.\.', Punctuation),
203  
204              (words(('arch', 'arm11', 'caps', 'child_of', 'ia32', 'irq', 'maps',
205                      'objects'), suffix=r'\b'), Keyword),
206  
207              (words(('aep', 'asid_pool', 'cnode', 'ep', 'frame', 'io_device',
208                      'io_ports', 'io_pt', 'notification', 'pd', 'pt', 'tcb',
209                      'ut', 'vcpu'), suffix=r'\b'), Keyword.Type),
210  
211              # Properties
212              (words(('asid', 'addr', 'badge', 'cached', 'dom', 'domainID', 'elf',
213                      'fault_ep', 'G', 'guard', 'guard_size', 'init', 'ip',
214                      'prio', 'sp', 'R', 'RG', 'RX', 'RW', 'RWG', 'RWX', 'W',
215                      'WG', 'WX', 'level', 'masked', 'master_reply', 'paddr',
216                      'ports', 'reply', 'uncached'), suffix=r'\b'),
217               Keyword.Reserved),
218  
219              # Literals
220              (r'0[xX][\da-fA-F]+', Number.Hex),
221              (r'\d+(\.\d+)?(k|M)?', Number),
222              (words(('bits',), suffix=r'\b'), Number),
223              (words(('cspace', 'vspace', 'reply_slot', 'caller_slot',
224                      'ipc_buffer_slot'), suffix=r'\b'), Number),
225  
226              # Identifiers
227              (r'[a-zA-Z_][-@\.\w]*', Name),
228          ],
229      }

232  class RedcodeLexer(RegexLexer):
233      """ 
234      A simple Redcode lexer based on ICWS'94. 
235      Contributed by Adam Blinkinsop <blinks@acm.org>. 
236   
237      .. versionadded:: 0.8 
238      """
239      name = 'Redcode'
240      aliases = ['redcode']
241      filenames = ['*.cw']
242  
243      opcodes = ('DAT', 'MOV', 'ADD', 'SUB', 'MUL', 'DIV', 'MOD',
244                 'JMP', 'JMZ', 'JMN', 'DJN', 'CMP', 'SLT', 'SPL',
245                 'ORG', 'EQU', 'END')
246      modifiers = ('A', 'B', 'AB', 'BA', 'F', 'X', 'I')
247  
248      tokens = {
249          'root': [
250              # Whitespace:
251              (r'\s+', Text),
252              (r';.*$', Comment.Single),
253              # Lexemes:
254              #  Identifiers
255              (r'\b(%s)\b' % '|'.join(opcodes), Name.Function),
256              (r'\b(%s)\b' % '|'.join(modifiers), Name.Decorator),
257              (r'[A-Za-z_]\w+', Name),
258              #  Operators
259              (r'[-+*/%]', Operator),
260              (r'[#$@<>]', Operator),  # mode
261              (r'[.,]', Punctuation),  # mode
262              #  Numbers
263              (r'[-+]?\d+', Number.Integer),
264          ],
265      }
