1    '''Screen 
2    ====== 
3     
4    This module changes some environment and configuration variables 
5    to match the density / dpi / screensize of a specific device. 
6     
7    To see a list of the available screenid's, just run:: 
8     
9        python main.py -m screen 
10    
11   To simulate a medium-density screen such as the Motorola Droid 2:: 
12    
13       python main.py -m screen:droid2 
14    
15   To simulate a high-density screen such as HTC One X, in portrait:: 
16    
17       python main.py -m screen:onex,portrait 
18    
19   To simulate the iPad 2 screen:: 
20    
21       python main.py -m screen:ipad 
22    
23   If the generated window is too large, you can specify a scale:: 
24    
25       python main.py -m screen:note2,portrait,scale=.75 
26    
27   Note that to display your contents correctly on a scaled window you 
28   must consistently use units 'dp' and 'sp' throughout your app. See 
29   :mod:`~kiv.metrics` for more details. 
30    
31   '''

121  def apply_device(device, scale, orientation):
122      name, width, height, dpi, density = devices[device]
123      if orientation == 'portrait':
124          width, height = height, width
125      Logger.info('Screen: Apply screen settings for {0}'.format(name))
126      Logger.info('Screen: size={0}x{1} dpi={2} density={3} '
127                  'orientation={4}'.format(width, height, dpi, density,
128                                           orientation))
129      try:
130          scale = float(scale)
131      except:
132          scale = 1
133      environ['KIVY_METRICS_DENSITY'] = str(density * scale)
134      environ['KIVY_DPI'] = str(dpi * scale)
135      Config.set('graphics', 'width', str(int(width * scale)))
136      # simulate with the android bar
137      # FIXME should be configurable
138      Config.set('graphics', 'height', str(int(height * scale - 25 * density)))
139      Config.set('graphics', 'fullscreen', '0')
140      Config.set('graphics', 'show_mousecursor', '1')

143  def usage(device=None):
144      if device:
145          Logger.error('Screen: The specified device ({0}) is unknown.',
146                       device)
147      print('\nModule usage: python main.py -m screen:deviceid[,orientation]\n')
148      print('Available devices:\n')
149      print('{0:12} {1:<22} {2:<8} {3:<8} {4:<5} {5:<8}'.format(
150          'Device ID', 'Name', 'Width', 'Height', 'DPI', 'Density'))
151      for device, info in devices.items():
152          print('{0:12} {1:<22} {2:<8} {3:<8} {4:<5} {5:<8}'.format(
153              device, *info))
154      print('\n')
155      print('Simulate a medium-density screen such as Motorola Droid 2:\n')
156      print('    python main.py -m screen:droid2\n')
157      print('Simulate a high-density screen such as HTC One X, in portrait:\n')
158      print('    python main.py -m screen:onex,portrait\n')
159      print('Simulate the iPad 2 screen\n')
160      print('    python main.py -m screen:ipad\n')
161      print('If the generated window is too large, you can specify a scale:\n')
162      print('    python main.py -m screen:note2,portrait,scale=.75\n')
163      sys.exit(1)

166  def configure(ctx):
167      scale = ctx.pop('scale', None)
168      orientation = 'landscape'
169      ctx.pop('landscape', None)
170      if ctx.pop('portrait', None):
171          orientation = 'portrait'
172      if not ctx:
173          return usage(None)
174      device = list(ctx.keys())[0]
175      if device not in devices:
176          return usage('')
177      apply_device(device, scale, orientation)
