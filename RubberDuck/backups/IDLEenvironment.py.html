24   def GetIDLEModule(module):
25       try:
26           # First get it from Pythonwin it is exists.
27           modname = "pywin.idle." + module
28           __import__(modname)
29       except ImportError as details:
30           msg = (
31               "The IDLE extension '%s' can not be located.\r\n\r\n"
32               "Please correct the installation and restart the"
33               " application.\r\n\r\n%s" % (module, details)
34           )
35           win32ui.MessageBox(msg)
36           return None
37       mod = sys.modules[modname]
38       mod.TclError = TextError  # A hack that can go soon!
39       return mod

42   # A class that is injected into the IDLE auto-indent extension.
43   # It allows for decent performance when opening a new file,
44   # as auto-indent uses the tokenizer module to determine indents.
45   # The default AutoIndent readline method works OK, but it goes through
46   # this layer of Tk index indirection for every single line.  For large files
47   # without indents (and even small files with indents :-) it was pretty slow!
48   def fast_readline(self):
49       if self.finished:
50           val = ""
51       else:
52           if "_scint_lines" not in self.__dict__:
53               # XXX - note - assumes this is only called once the file is loaded!
54               self._scint_lines = self.text.edit.GetTextRange().split("\n")
55           sl = self._scint_lines
56           i = self.i = self.i + 1
57           if i >= len(sl):
58               val = ""
59           else:
60               val = sl[i] + "\n"
61       return val.encode(default_scintilla_encoding)

72       def __init__(self, edit):
73           self.edit = edit
74           self.text = TkText(edit)
75           self.extensions = {}
76           self.extension_menus = {}

78       def close(self):
79           self.edit = self.text = None
80           self.extension_menus = None
81           try:
82               for ext in self.extensions.values():
83                   closer = getattr(ext, "close", None)
84                   if closer is not None:
85                       closer()
86           finally:
87               self.extensions = {}
88   
89       def IDLEExtension(self, extension):
90           ext = self.extensions.get(extension)
91           if ext is not None:
92               return ext
93           mod = GetIDLEModule(extension)
94           if mod is None:
95               return None
96           klass = getattr(mod, extension)
97           ext = self.extensions[extension] = klass(self)
98           # Find and bind all the events defined in the extension.
99           events = [item for item in dir(klass) if item[-6:] == "_event"]
100          for event in events:
101              name = "<<%s>>" % (event[:-6].replace("_", "-"),)
102              self.edit.bindings.bind(name, getattr(ext, event))
103          return ext

105      def GetMenuItems(self, menu_name):
106          # Get all menu items for the menu name (eg, "edit")
107          bindings = self.edit.bindings
108          ret = []
109          for ext in self.extensions.values():
110              menudefs = getattr(ext, "menudefs", [])
111              for name, items in menudefs:
112                  if name == menu_name:
113                      for text, event in [item for item in items if item is not None]:
114                          text = text.replace("&", "&&")
115                          text = text.replace("_", "&")
116                          ret.append((text, event))
117          return ret

122      def askinteger(
123          self, caption, prompt, parent=None, initialvalue=0, minvalue=None, maxvalue=None
124      ):
125          while 1:
126              rc = GetSimpleInput(prompt, str(initialvalue), caption)
127              if rc is None:
128                  return 0  # Correct "cancel" semantics?
129              err = None
130              try:
131                  rc = int(rc)
132              except ValueError:
133                  err = "Please enter an integer"
134              if not err and minvalue is not None and rc < minvalue:
135                  err = "Please enter an integer greater then or equal to %s" % (
136                      minvalue,
137                  )
138              if not err and maxvalue is not None and rc > maxvalue:
139                  err = "Please enter an integer less then or equal to %s" % (maxvalue,)
140              if err:
141                  win32ui.MessageBox(err, caption, win32con.MB_OK)
142                  continue
143              return rc

197  def TkOffsetToIndex(offset, edit):
198      lineoff = 0
199      # May be 1 > actual end if we pretended there was a trailing '\n'
200      offset = min(offset, edit.GetTextLength())
201      line = edit.LineFromChar(offset)
202      lineIndex = edit.LineIndex(line)
203      return "%d.%d" % (line + 1, offset - lineIndex)

206  def _NextTok(str, pos):
207      # Returns (token, endPos)
208      end = len(str)
209      if pos >= end:
210          return None, 0
211      while pos < end and str[pos] in string.whitespace:
212          pos = pos + 1
213      # Special case for +-
214      if str[pos] in "+-":
215          return str[pos], pos + 1
216      # Digits also a special case.
217      endPos = pos
218      while endPos < end and str[endPos] in string.digits + ".":
219          endPos = endPos + 1
220      if pos != endPos:
221          return str[pos:endPos], endPos
222      endPos = pos
223      while endPos < end and str[endPos] not in string.whitespace + string.digits + "+-":
224          endPos = endPos + 1
225      if pos != endPos:
226          return str[pos:endPos], endPos
227      return None, 0

230  def TkIndexToOffset(bm, edit, marks):
231      base, nextTokPos = _NextTok(bm, 0)
232      if base is None:
233          raise ValueError("Empty bookmark ID!")
234      if base.find(".") > 0:
235          try:
236              line, col = base.split(".", 2)
237              if col == "first" or col == "last":
238                  # Tag name
239                  if line != "sel":
240                      raise ValueError("Tags arent here!")
241                  sel = edit.GetSel()
242                  if sel[0] == sel[1]:
243                      raise EmptyRange
244                  if col == "first":
245                      pos = sel[0]
246                  else:
247                      pos = sel[1]
248              else:
249                  # Lines are 1 based for tkinter
250                  line = int(line) - 1
251                  if line > edit.GetLineCount():
252                      pos = edit.GetTextLength() + 1
253                  else:
254                      pos = edit.LineIndex(line)
255                      if pos == -1:
256                          pos = edit.GetTextLength()
257                      pos = pos + int(col)
258          except (ValueError, IndexError):
259              raise ValueError("Unexpected literal in '%s'" % base)
260      elif base == "insert":
261          pos = edit.GetSel()[0]
262      elif base == "end":
263          pos = edit.GetTextLength()
264          # Pretend there is a trailing '\n' if necessary
265          if pos and edit.SCIGetCharAt(pos - 1) != "\n":
266              pos = pos + 1
267      else:
268          try:
269              pos = marks[base]
270          except KeyError:
271              raise ValueError("Unsupported base offset or undefined mark '%s'" % base)
272  
273      while 1:
274          word, nextTokPos = _NextTok(bm, nextTokPos)
275          if word is None:
276              break
277          if word in ["+", "-"]:
278              num, nextTokPos = _NextTok(bm, nextTokPos)
279              if num is None:
280                  raise ValueError("+/- operator needs 2 args")
281              what, nextTokPos = _NextTok(bm, nextTokPos)
282              if what is None:
283                  raise ValueError("+/- operator needs 2 args")
284              if what[0] != "c":
285                  raise ValueError("+/- only supports chars")
286              if word == "+":
287                  pos = pos + int(num)
288              else:
289                  pos = pos - int(num)
290          elif word == "wordstart":
291              while pos > 0 and edit.SCIGetCharAt(pos - 1) in wordchars:
292                  pos = pos - 1
293          elif word == "wordend":
294              end = edit.GetTextLength()
295              while pos < end and edit.SCIGetCharAt(pos) in wordchars:
296                  pos = pos + 1
297          elif word == "linestart":
298              while pos > 0 and edit.SCIGetCharAt(pos - 1) not in "\n\r":
299                  pos = pos - 1
300          elif word == "lineend":
301              end = edit.GetTextLength()
302              while pos < end and edit.SCIGetCharAt(pos) not in "\n\r":
303                  pos = pos + 1
304          else:
305              raise ValueError("Unsupported relative offset '%s'" % word)
306      return max(pos, 0)  # Tkinter is tollerant of -ve indexes - we aren't

340      def _fix_indexes(self, start, end):
341          # first some magic to handle skipping over utf8 extended chars.
342          while start > 0 and ord(self.edit.SCIGetCharAt(start)) & 0xC0 == 0x80:
343              start -= 1
344          while (
345              end < self.edit.GetTextLength()
346              and ord(self.edit.SCIGetCharAt(end)) & 0xC0 == 0x80
347          ):
348              end += 1
349          # now handling fixing \r\n->\n disparities...
350          if (
351              start > 0
352              and self.edit.SCIGetCharAt(start) == "\n"
353              and self.edit.SCIGetCharAt(start - 1) == "\r"
354          ):
355              start = start - 1
356          if (
357              end < self.edit.GetTextLength()
358              and self.edit.SCIGetCharAt(end - 1) == "\r"
359              and self.edit.SCIGetCharAt(end) == "\n"
360          ):
361              end = end + 1
362          return start, end

380      def get(self, start, end=None):
381          try:
382              start = self._getoffset(start)
383              if end is None:
384                  end = start + 1
385              else:
386                  end = self._getoffset(end)
387          except EmptyRange:
388              return ""
389          # Simple semantic checks to conform to the Tk text interface
390          if end <= start:
391              return ""
392          max = self.edit.GetTextLength()
393          checkEnd = 0
394          if end > max:
395              end = max
396              checkEnd = 1
397          start, end = self._fix_indexes(start, end)
398          ret = self.edit.GetTextRange(start, end)
399          # pretend a trailing '\n' exists if necessary.
400          if checkEnd and (not ret or ret[-1] != "\n"):
401              ret = ret + "\n"
402          return ret.replace("\r", "")

404      def index(self, spec):
405          try:
406              return self._getindex(self._getoffset(spec))
407          except EmptyRange:
408              return ""

410      def insert(self, pos, text):
411          try:
412              pos = self._getoffset(pos)
413          except EmptyRange:
414              raise TextError("Empty range")
415          self.edit.SetSel((pos, pos))
416          # IDLE only deals with "\n" - we will be nicer
417  
418          bits = text.split("\n")
419          self.edit.SCIAddText(bits[0])
420          for bit in bits[1:]:
421              self.edit.SCINewline()
422              self.edit.SCIAddText(bit)
423  
424      def delete(self, start, end=None):
425          try:
426              start = self._getoffset(start)
427              if end is not None:
428                  end = self._getoffset(end)
429          except EmptyRange:
430              raise TextError("Empty range")
431          # If end is specified and == start, then we must delete nothing.
432          if start == end:
433              return
434          # If end is not specified, delete one char
435          if end is None:
436              end = start + 1
437          else:
438              # Tk says not to delete in this case, but our control would.
439              if end < start:
440                  return
441          if start == self.edit.GetTextLength():
442              return  # Nothing to delete.
443          old = self.edit.GetSel()[0]  # Lose a selection
444          # Hack for partial '\r\n' and UTF-8 char removal
445          start, end = self._fix_indexes(start, end)
446          self.edit.SetSel((start, end))
447          self.edit.Clear()
448          if old >= start and old < end:
449              old = start
450          elif old >= end:
451              old = old - (end - start)
452          self.edit.SetSel(old)
