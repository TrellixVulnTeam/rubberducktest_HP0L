58   class MySerialControl(activex.Control, serialModule.MSComm):
59       def __init__(self, parent):
60           activex.Control.__init__(self)
61           serialModule.MSComm.__init__(self)
62           self.parent = parent
63   
64       def OnComm(self):
65           self.parent.OnComm()

68   class TestSerDialog(dialog.Dialog):
69       def __init__(self, *args):
70           dialog.Dialog.__init__(*(self,) + args)
71           self.olectl = None
72   
73       def OnComm(self):
74           event = self.olectl.CommEvent
75           if event == serialModule.OnCommConstants.comEvReceive:
76               self.editwindow.ReplaceSel(self.olectl.Input)
77   
78       def OnKey(self, key):
79           if self.olectl:
80               self.olectl.Output = chr(key)
81   
82       def OnInitDialog(self):
83           rc = dialog.Dialog.OnInitDialog(self)
84           self.editwindow = self.GetDlgItem(132)
85           self.editwindow.HookAllKeyStrokes(self.OnKey)
86   
87           self.olectl = MySerialControl(self)
88           try:
89               self.olectl.CreateControl(
90                   "OCX",
91                   win32con.WS_TABSTOP | win32con.WS_VISIBLE,
92                   (7, 43, 500, 300),
93                   self._obj_,
94                   131,
95               )
96           except win32ui.error:
97               self.MessageBox("The Serial Control could not be created")
98               self.olectl = None
99               self.EndDialog(win32con.IDCANCEL)
100          if self.olectl:
101              self.olectl.Settings = SERIAL_SETTINGS
102              self.olectl.CommPort = SERIAL_PORT
103              self.olectl.RThreshold = 1
104              try:
105                  self.olectl.PortOpen = 1
106              except pythoncom.com_error as details:
107                  print(
108                      "Could not open the specified serial port - %s"
109                      % (details.excepinfo[2])
110                  )
111                  self.EndDialog(win32con.IDCANCEL)
112          return rc
113  
114      def OnDestroy(self, msg):
115          if self.olectl:
116              try:
117                  self.olectl.PortOpen = 0
118              except pythoncom.com_error as details:
119                  print("Error closing port - %s" % (details.excepinfo[2]))
120          return dialog.Dialog.OnDestroy(self, msg)
