147  def replace(s):
148      item = s.split(' ')
149      rettype = item[1]
150      item = item[2:]
151      for x in item:
152          x = x.strip()
153          if not x or x.startswith('GL'):
154              continue
155          if x.startswith('(GL'):
156              yield '('
157              continue
158          if x.startswith('gl'):
159              prefix = ''
160              if rettype != 'void':
161                  prefix = 'return '
162              yield '%scgl.%s' % (prefix, x)
163              continue
164          yield x
