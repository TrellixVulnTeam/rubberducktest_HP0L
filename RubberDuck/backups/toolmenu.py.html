30   def LoadToolMenuItems():
31       # Load from the registry.
32       items = []
33       lookNo = 1
34       while 1:
35           menu = win32ui.GetProfileVal("Tools Menu\\%s" % lookNo, "", "")
36           if menu == "":
37               break
38           cmd = win32ui.GetProfileVal("Tools Menu\\%s" % lookNo, "Command", "")
39           items.append((menu, cmd))
40           lookNo = lookNo + 1
41   
42       if len(items) == 0:
43           items = defaultToolMenuItems
44       return items

47   def WriteToolMenuItems(items):
48       # Items is a list of (menu, command)
49       # Delete the entire registry tree.
50       try:
51           mainKey = win32ui.GetAppRegistryKey()
52           toolKey = win32api.RegOpenKey(mainKey, "Tools Menu")
53       except win32ui.error:
54           toolKey = None
55       if toolKey is not None:
56           while 1:
57               try:
58                   subkey = win32api.RegEnumKey(toolKey, 0)
59               except win32api.error:
60                   break
61               win32api.RegDeleteKey(toolKey, subkey)
62       # Keys are now removed - write the new ones.
63       # But first check if we have the defaults - and if so, dont write anything!
64       if items == defaultToolMenuItems:
65           return
66       itemNo = 1
67       for menu, cmd in items:
68           win32ui.WriteProfileVal("Tools Menu\\%s" % itemNo, "", menu)
69           win32ui.WriteProfileVal("Tools Menu\\%s" % itemNo, "Command", cmd)
70           itemNo = itemNo + 1

73   def SetToolsMenu(menu, menuPos=None):
74       global tools
75       global idPos
76   
77       # todo - check the menu does not already exist.
78       # Create the new menu
79       toolsMenu = win32ui.CreatePopupMenu()
80   
81       # Load from the ini file.
82       items = LoadToolMenuItems()
83       for menuString, cmd in items:
84           tools[idPos] = (menuString, cmd, menuString)
85           toolsMenu.AppendMenu(
86               win32con.MF_ENABLED | win32con.MF_STRING, idPos, menuString
87           )
88           win32ui.GetMainFrame().HookCommand(HandleToolCommand, idPos)
89           idPos = idPos + 1
90   
91       # Find the correct spot to insert the new tools menu.
92       if menuPos is None:
93           menuPos = menu.GetMenuItemCount() - 2
94           if menuPos < 0:
95               menuPos = 0
96   
97       menu.InsertMenu(
98           menuPos,
99           win32con.MF_BYPOSITION
100          | win32con.MF_ENABLED
101          | win32con.MF_STRING
102          | win32con.MF_POPUP,
103          toolsMenu.GetHandle(),
104          "&Tools",
105      )

108  def HandleToolCommand(cmd, code):
109      import traceback
110      import re
111  
112      global tools
113      (menuString, pyCmd, desc) = tools[cmd]
114      win32ui.SetStatusText("Executing tool %s" % desc, 1)
115      pyCmd = re.sub("\\\\n", "\n", pyCmd)
116      win32ui.DoWaitCursor(1)
117      oldFlag = None
118      try:
119          oldFlag = sys.stdout.template.writeQueueing
120          sys.stdout.template.writeQueueing = 0
121      except (NameError, AttributeError):
122          pass
123  
124      try:
125          exec("%s\n" % pyCmd)
126          worked = 1
127      except SystemExit:
128          # The program raised a SystemExit - ignore it.
129          worked = 1
130      except:
131          print("Failed to execute command:\n%s" % pyCmd)
132          traceback.print_exc()
133          worked = 0
134      if oldFlag is not None:
135          sys.stdout.template.writeQueueing = oldFlag
136      win32ui.DoWaitCursor(0)
137      if worked:
138          text = "Completed successfully."
139      else:
140          text = "Error executing %s." % desc
141      win32ui.SetStatusText(text, 1)

154  class ToolMenuPropPage(dialog.PropertyPage):
155      def __init__(self):
156          self.bImChangingEditControls = 0  # Am I programatically changing the controls?
157          dialog.PropertyPage.__init__(self, win32ui.IDD_PP_TOOLMENU)
158  
159      def OnInitDialog(self):
160          self.editMenuCommand = self.GetDlgItem(win32ui.IDC_EDIT2)
161          self.butNew = self.GetDlgItem(win32ui.IDC_BUTTON3)
162  
163          # Now hook the change notification messages for the edit controls.
164          self.HookCommand(self.OnCommandEditControls, win32ui.IDC_EDIT1)
165          self.HookCommand(self.OnCommandEditControls, win32ui.IDC_EDIT2)
166  
167          self.HookNotify(self.OnNotifyListControl, commctrl.LVN_ITEMCHANGED)
168          self.HookNotify(self.OnNotifyListControlEndLabelEdit, commctrl.LVN_ENDLABELEDIT)
169  
170          # Hook the button clicks.
171          self.HookCommand(self.OnButtonNew, win32ui.IDC_BUTTON3)  # New Item
172          self.HookCommand(self.OnButtonDelete, win32ui.IDC_BUTTON4)  # Delete item
173          self.HookCommand(self.OnButtonMove, win32ui.IDC_BUTTON1)  # Move up
174          self.HookCommand(self.OnButtonMove, win32ui.IDC_BUTTON2)  # Move down
175  
176          # Setup the columns in the list control
177          lc = self.GetDlgItem(win32ui.IDC_LIST1)
178          rect = lc.GetWindowRect()
179          cx = rect[2] - rect[0]
180          colSize = cx / 2 - win32api.GetSystemMetrics(win32con.SM_CXBORDER) - 1
181  
182          item = commctrl.LVCFMT_LEFT, colSize, "Menu Text"
183          lc.InsertColumn(0, item)
184  
185          item = commctrl.LVCFMT_LEFT, colSize, "Python Command"
186          lc.InsertColumn(1, item)
187  
188          # Insert the existing tools menu
189          itemNo = 0
190          for desc, cmd in LoadToolMenuItems():
191              lc.InsertItem(itemNo, desc)
192              lc.SetItemText(itemNo, 1, cmd)
193              itemNo = itemNo + 1
194  
195          self.listControl = lc
196          return dialog.PropertyPage.OnInitDialog(self)

198      def OnOK(self):
199          # Write the menu back to the registry.
200          items = []
201          itemLook = 0
202          while 1:
203              try:
204                  text = self.listControl.GetItemText(itemLook, 0)
205                  if not text:
206                      break
207                  items.append((text, self.listControl.GetItemText(itemLook, 1)))
208              except win32ui.error:
209                  # no more items!
210                  break
211              itemLook = itemLook + 1
212          WriteToolMenuItems(items)
213          return self._obj_.OnOK()

229      def OnNotifyListControl(self, id, cmd):
230          #       print id, cmd
231          try:
232              itemNo = self.listControl.GetNextItem(-1, commctrl.LVNI_SELECTED)
233          except win32ui.error:  # No selection!
234              return
235  
236          self.bImChangingEditControls = 1
237          try:
238              item = self.listControl.GetItem(itemNo, 1)
239              self.editMenuCommand.SetWindowText(item[4])
240          finally:
241              self.bImChangingEditControls = 0
242  
243          return 0  # we have handled this!

251      def OnButtonMove(self, id, cmd):
252          if cmd == win32con.BN_CLICKED:
253              try:
254                  itemNo = self.listControl.GetNextItem(-1, commctrl.LVNI_SELECTED)
255              except win32ui.error:
256                  return
257              menu = self.listControl.GetItemText(itemNo, 0)
258              cmd = self.listControl.GetItemText(itemNo, 1)
259              if id == win32ui.IDC_BUTTON1:
260                  # Move up
261                  if itemNo > 0:
262                      self.listControl.DeleteItem(itemNo)
263                      # reinsert it.
264                      self.listControl.InsertItem(itemNo - 1, menu)
265                      self.listControl.SetItemText(itemNo - 1, 1, cmd)
266              else:
267                  # Move down.
268                  if itemNo < self.listControl.GetItemCount() - 1:
269                      self.listControl.DeleteItem(itemNo)
270                      # reinsert it.
271                      self.listControl.InsertItem(itemNo + 1, menu)
272                      self.listControl.SetItemText(itemNo + 1, 1, cmd)
273  
274      def OnButtonDelete(self, id, cmd):
275          if cmd == win32con.BN_CLICKED:
276              try:
277                  itemNo = self.listControl.GetNextItem(-1, commctrl.LVNI_SELECTED)
278              except win32ui.error:  # No selection!
279                  return
280              self.listControl.DeleteItem(itemNo)
