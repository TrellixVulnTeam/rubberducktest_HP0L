118  class ActionScript3Lexer(RegexLexer):
119      """ 
120      For ActionScript 3 source code. 
121   
122      .. versionadded:: 0.11 
123      """
124  
125      name = 'ActionScript 3'
126      aliases = ['actionscript3', 'as3']
127      filenames = ['*.as']
128      mimetypes = ['application/x-actionscript3', 'text/x-actionscript3',
129                   'text/actionscript3']
130  
131      identifier = r'[$a-zA-Z_]\w*'
132      typeidentifier = identifier + r'(?:\.<\w+>)?'
133  
134      flags = re.DOTALL | re.MULTILINE
135      tokens = {
136          'root': [
137              (r'\s+', Text),
138              (r'(function\s+)(' + identifier + r')(\s*)(\()',
139               bygroups(Keyword.Declaration, Name.Function, Text, Operator),
140               'funcparams'),
141              (r'(var|const)(\s+)(' + identifier + r')(\s*)(:)(\s*)(' +
142               typeidentifier + r')',
143               bygroups(Keyword.Declaration, Text, Name, Text, Punctuation, Text,
144                        Keyword.Type)),
145              (r'(import|package)(\s+)((?:' + identifier + r'|\.)+)(\s*)',
146               bygroups(Keyword, Text, Name.Namespace, Text)),
147              (r'(new)(\s+)(' + typeidentifier + r')(\s*)(\()',
148               bygroups(Keyword, Text, Keyword.Type, Text, Operator)),
149              (r'//.*?\n', Comment.Single),
150              (r'/\*.*?\*/', Comment.Multiline),
151              (r'/(\\\\|\\[^\\]|[^\\\n])*/[gisx]*', String.Regex),
152              (r'(\.)(' + identifier + r')', bygroups(Operator, Name.Attribute)),
153              (r'(case|default|for|each|in|while|do|break|return|continue|if|else|'
154               r'throw|try|catch|with|new|typeof|arguments|instanceof|this|'
155               r'switch|import|include|as|is)\b',
156               Keyword),
157              (r'(class|public|final|internal|native|override|private|protected|'
158               r'static|import|extends|implements|interface|intrinsic|return|super|'
159               r'dynamic|function|const|get|namespace|package|set)\b',
160               Keyword.Declaration),
161              (r'(true|false|null|NaN|Infinity|-Infinity|undefined|void)\b',
162               Keyword.Constant),
163              (r'(decodeURI|decodeURIComponent|encodeURI|escape|eval|isFinite|isNaN|'
164               r'isXMLName|clearInterval|fscommand|getTimer|getURL|getVersion|'
165               r'isFinite|parseFloat|parseInt|setInterval|trace|updateAfterEvent|'
166               r'unescape)\b', Name.Function),
167              (identifier, Name),
168              (r'[0-9][0-9]*\.[0-9]+([eE][0-9]+)?[fd]?', Number.Float),
169              (r'0x[0-9a-f]+', Number.Hex),
170              (r'[0-9]+', Number.Integer),
171              (r'"(\\\\|\\[^\\]|[^"\\])*"', String.Double),
172              (r"'(\\\\|\\[^\\]|[^'\\])*'", String.Single),
173              (r'[~^*!%&<>|+=:;,/?\\{}\[\]().-]+', Operator),
174          ],
175          'funcparams': [
176              (r'\s+', Text),
177              (r'(\s*)(\.\.\.)?(' + identifier + r')(\s*)(:)(\s*)(' +
178               typeidentifier + r'|\*)(\s*)',
179               bygroups(Text, Punctuation, Name, Text, Operator, Text,
180                        Keyword.Type, Text), 'defval'),
181              (r'\)', Operator, 'type')
182          ],
183          'type': [
184              (r'(\s*)(:)(\s*)(' + typeidentifier + r'|\*)',
185               bygroups(Text, Operator, Text, Keyword.Type), '#pop:2'),
186              (r'\s+', Text, '#pop:2'),
187              default('#pop:2')
188          ],
189          'defval': [
190              (r'(=)(\s*)([^(),]+)(\s*)(,?)',
191               bygroups(Operator, Text, using(this), Text, Operator), '#pop'),
192              (r',', Operator, '#pop'),
193              default('#pop')
194          ]
195      }
196  
197      def analyse_text(text):
198          if re.match(r'\w+\s*:\s*\w', text):
199              return 0.3
200          return 0

203  class MxmlLexer(RegexLexer):
204      """ 
205      For MXML markup. 
206      Nested AS3 in <script> tags is highlighted by the appropriate lexer. 
207   
208      .. versionadded:: 1.1 
209      """
210      flags = re.MULTILINE | re.DOTALL
211      name = 'MXML'
212      aliases = ['mxml']
213      filenames = ['*.mxml']
214      mimetimes = ['text/xml', 'application/xml']
215  
216      tokens = {
217          'root': [
218              ('[^<&]+', Text),
219              (r'&\S*?;', Name.Entity),
220              (r'(\<\!\[CDATA\[)(.*?)(\]\]\>)',
221               bygroups(String, using(ActionScript3Lexer), String)),
222              ('<!--', Comment, 'comment'),
223              (r'<\?.*?\?>', Comment.Preproc),
224              ('<![^>]*>', Comment.Preproc),
225              (r'<\s*[\w:.-]+', Name.Tag, 'tag'),
226              (r'<\s*/\s*[\w:.-]+\s*>', Name.Tag),
227          ],
228          'comment': [
229              ('[^-]+', Comment),
230              ('-->', Comment, '#pop'),
231              ('-', Comment),
232          ],
233          'tag': [
234              (r'\s+', Text),
235              (r'[\w.:-]+\s*=', Name.Attribute, 'attr'),
236              (r'/?\s*>', Name.Tag, '#pop'),
237          ],
238          'attr': [
239              (r'\s+', Text),
240              ('".*?"', String, '#pop'),
241              ("'.*?'", String, '#pop'),
242              (r'[^\s>]+', String, '#pop'),
243          ],
244      }
