25   @pytest.mark.parametrize('transition_cls_name', transition_cls_names)
26   def test_switching_does_not_affect_a_list_of_screens(transition_cls_name):
27       import kivy.uix.screenmanager as sm
28       transition_cls = getattr(sm, transition_cls_name)
29       scrmgr = sm.ScreenManager()
30       for i in range(3):
31           scrmgr.add_widget(sm.Screen(name=str(i)))
32       names = list(scrmgr.screen_names)
33       scrmgr.transition = transition_cls()
34       scrmgr.current = '1'
35       assert names == scrmgr.screen_names
36       scrmgr.transition = transition_cls()
37       scrmgr.current = '2'
38       assert names == scrmgr.screen_names
