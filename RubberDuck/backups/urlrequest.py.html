1    ''' 
2    UrlRequest 
3    ========== 
4     
5    .. versionadded:: 1.0.8 
6     
7    You can use the :class:`UrlRequest` to make asynchronous requests on the 
8    web and get the result when the request is completed. The spirit is the 
9    same as the XHR object in Javascript. 
10    
11   The content is also decoded if the Content-Type is 
12   application/json and the result automatically passed through json.loads. 
13    
14    
15   The syntax to create a request:: 
16    
17       from kivy.network.urlrequest import UrlRequest 
18       req = UrlRequest(url, on_success, on_redirect, on_failure, on_error, 
19                        on_progress, req_body, req_headers, chunk_size, 
20                        timeout, method, decode, debug, file_path, ca_file, 
21                        verify) 
22    
23    
24   Only the first argument is mandatory: the rest are optional. 
25   By default, a "GET" request will be sent. If the :attr:`UrlRequest.req_body` is 
26   not None, a "POST" request will be sent. It's up to you to adjust 
27   :attr:`UrlRequest.req_headers` to suit your requirements and the response 
28   to the request will be accessible as the parameter called "result" on 
29   the callback function of the on_success event. 
30    
31    
32   Example of fetching JSON:: 
33    
34       def got_json(req, result): 
35           for key, value in req.resp_headers.items(): 
36               print('{}: {}'.format(key, value)) 
37    
38       req = UrlRequest('https://httpbin.org/headers', got_json) 
39    
40   Example of Posting data (adapted from httplib example):: 
41    
42       import urllib 
43    
44       def bug_posted(req, result): 
45           print('Our bug is posted!') 
46           print(result) 
47    
48       params = urllib.urlencode({'@number': 12524, '@type': 'issue', 
49           '@action': 'show'}) 
50       headers = {'Content-type': 'application/x-www-form-urlencoded', 
51                 'Accept': 'text/plain'} 
52       req = UrlRequest('bugs.python.org', on_success=bug_posted, req_body=params, 
53               req_headers=headers) 
54    
55   If you want a synchronous request, you can call the wait() method. 
56    
57   '''

97   class UrlRequest(Thread):
98       '''A UrlRequest. See module documentation for usage. 
99    
100      .. versionchanged:: 1.5.1 
101          Add `debug` parameter 
102   
103      .. versionchanged:: 1.0.10 
104          Add `method` parameter 
105   
106      .. versionchanged:: 1.8.0 
107   
108          Parameter `decode` added. 
109          Parameter `file_path` added. 
110          Parameter `on_redirect` added. 
111          Parameter `on_failure` added. 
112   
113      .. versionchanged:: 1.9.1 
114   
115          Parameter `ca_file` added. 
116          Parameter `verify` added. 
117   
118      .. versionchanged:: 1.10.0 
119   
120          Parameters `proxy_host`, `proxy_port` and `proxy_headers` added. 
121   
122      .. versionchanged:: 1.11.0 
123   
124          Parameters `on_cancel` added. 
125   
126      :Parameters: 
127          `url`: str 
128              Complete url string to call. 
129          `on_success`: callback(request, result) 
130              Callback function to call when the result has been fetched. 
131          `on_redirect`: callback(request, result) 
132              Callback function to call if the server returns a Redirect. 
133          `on_failure`: callback(request, result) 
134              Callback function to call if the server returns a Client or 
135              Server Error. 
136          `on_error`: callback(request, error) 
137              Callback function to call if an error occurs. 
138          `on_progress`: callback(request, current_size, total_size) 
139              Callback function that will be called to report progression of the 
140              download. `total_size` might be -1 if no Content-Length has been 
141              reported in the http response. 
142              This callback will be called after each `chunk_size` is read. 
143          `on_cancel`: callback(request) 
144              Callback function to call if user requested to cancel the download 
145              operation via the .cancel() method. 
146          `req_body`: str, defaults to None 
147              Data to sent in the request. If it's not None, a POST will be done 
148              instead of a GET. 
149          `req_headers`: dict, defaults to None 
150              Custom headers to add to the request. 
151          `chunk_size`: int, defaults to 8192 
152              Size of each chunk to read, used only when `on_progress` callback 
153              has been set. If you decrease it too much, a lot of on_progress 
154              callbacks will be fired and will slow down your download. If you 
155              want to have the maximum download speed, increase the chunk_size 
156              or don't use ``on_progress``. 
157          `timeout`: int, defaults to None 
158              If set, blocking operations will timeout after this many seconds. 
159          `method`: str, defaults to 'GET' (or 'POST' if ``body`` is specified) 
160              The HTTP method to use. 
161          `decode`: bool, defaults to True 
162              If False, skip decoding of the response. 
163          `debug`: bool, defaults to False 
164              If True, it will use the Logger.debug to print information 
165              about url access/progression/errors. 
166          `file_path`: str, defaults to None 
167              If set, the result of the UrlRequest will be written to this path 
168              instead of in memory. 
169          `ca_file`: str, defaults to None 
170              Indicates a SSL CA certificate file path to validate HTTPS 
171              certificates against 
172          `verify`: bool, defaults to True 
173              If False, disables SSL CA certificate verification 
174          `proxy_host`: str, defaults to None 
175              If set, the proxy host to use for this connection. 
176          `proxy_port`: int, defaults to None 
177              If set, and `proxy_host` is also set, the port to use for 
178              connecting to the proxy server. 
179          `proxy_headers`: dict, defaults to None 
180              If set, and `proxy_host` is also set, the headers to send to the 
181              proxy server in the ``CONNECT`` request. 
182      '''

184      def __init__(self, url, on_success=None, on_redirect=None,
185                   on_failure=None, on_error=None, on_progress=None,
186                   req_body=None, req_headers=None, chunk_size=8192,
187                   timeout=None, method=None, decode=True, debug=False,
188                   file_path=None, ca_file=None, verify=True, proxy_host=None,
189                   proxy_port=None, proxy_headers=None, user_agent=None,
190                   on_cancel=None, cookies=None):
191          super(UrlRequest, self).__init__()
192          self._queue = deque()
193          self._trigger_result = Clock.create_trigger(self._dispatch_result, 0)
194          self.daemon = True
195          self.on_success = WeakMethod(on_success) if on_success else None
196          self.on_redirect = WeakMethod(on_redirect) if on_redirect else None
197          self.on_failure = WeakMethod(on_failure) if on_failure else None
198          self.on_error = WeakMethod(on_error) if on_error else None
199          self.on_progress = WeakMethod(on_progress) if on_progress else None
200          self.on_cancel = WeakMethod(on_cancel) if on_cancel else None
201          self.decode = decode
202          self.file_path = file_path
203          self._debug = debug
204          self._result = None
205          self._error = None
206          self._is_finished = False
207          self._resp_status = None
208          self._resp_headers = None
209          self._resp_length = -1
210          self._chunk_size = chunk_size
211          self._timeout = timeout
212          self._method = method
213          self.verify = verify
214          self._proxy_host = proxy_host
215          self._proxy_port = proxy_port
216          self._proxy_headers = proxy_headers
217          self._cancel_event = Event()
218          self._user_agent = user_agent
219          self._cookies = cookies
220  
221          if platform in ['android', 'ios']:
222              import certifi
223              self.ca_file = ca_file or certifi.where()
224          else:
225              self.ca_file = ca_file
226  
227          #: Url of the request
228          self.url = url
229  
230          #: Request body passed in __init__
231          self.req_body = req_body
232  
233          #: Request headers passed in __init__
234          self.req_headers = req_headers
235  
236          # save our request to prevent GC
237          g_requests.append(self)
238  
239          self.start()

241      def run(self):
242          q = self._queue.appendleft
243          url = self.url
244          req_body = self.req_body
245          req_headers = self.req_headers or {}
246  
247          user_agent = self._user_agent
248          cookies = self._cookies
249  
250          if user_agent:
251              req_headers.setdefault('User-Agent', user_agent)
252  
253          elif (
254              Config.has_section('network')
255              and 'useragent' in Config.items('network')
256          ):
257              useragent = Config.get('network', 'useragent')
258              req_headers.setdefault('User-Agent', useragent)
259  
260          if cookies:
261              req_headers.setdefault("Cookie", cookies)
262  
263          try:
264              result, resp = self._fetch_url(url, req_body, req_headers, q)
265              if self.decode:
266                  result = self.decode_result(result, resp)
267          except Exception as e:
268              q(('error', None, e))
269          else:
270              if not self._cancel_event.is_set():
271                  q(('success', resp, result))
272              else:
273                  q(('killed', None, None))
274  
275          # using trigger can result in a missed on_success event
276          self._trigger_result()
277  
278          # clean ourself when the queue is empty
279          while len(self._queue):
280              sleep(.1)
281              self._trigger_result()
282  
283          # ok, authorize the GC to clean us.
284          if self in g_requests:
285              g_requests.remove(self)

287      def _parse_url(self, url):
288          parse = urlparse(url)
289          host = parse.hostname
290          port = parse.port
291          userpass = None
292  
293          # append user + pass to hostname if specified
294          if parse.username and parse.password:
295              userpass = {
296                  "Authorization": "Basic {}".format(b64encode(
297                      "{}:{}".format(
298                          parse.username,
299                          parse.password
300                      ).encode('utf-8')
301                  ).decode('utf-8'))
302              }
303  
304          return host, port, userpass, parse
305  
306      def _fetch_url(self, url, body, headers, q):
307          # Parse and fetch the current url
308          trigger = self._trigger_result
309          chunk_size = self._chunk_size
310          report_progress = self.on_progress is not None
311          timeout = self._timeout
312          file_path = self.file_path
313          ca_file = self.ca_file
314          verify = self.verify
315  
316          if self._debug:
317              Logger.debug('UrlRequest: {0} Fetch url <{1}>'.format(
318                  id(self), url))
319              Logger.debug('UrlRequest: {0} - body: {1}'.format(
320                  id(self), body))
321              Logger.debug('UrlRequest: {0} - headers: {1}'.format(
322                  id(self), headers))
323  
324          # parse url
325          host, port, userpass, parse = self._parse_url(url)
326          if userpass and not headers:
327              headers = userpass
328          elif userpass and headers:
329              key = list(userpass.keys())[0]
330              headers[key] = userpass[key]
331  
332          # translate scheme to connection class
333          cls = self.get_connection_for_scheme(parse.scheme)
334  
335          # reconstruct path to pass on the request
336          path = parse.path
337          if parse.params:
338              path += ';' + parse.params
339          if parse.query:
340              path += '?' + parse.query
341          if parse.fragment:
342              path += '#' + parse.fragment
343  
344          # create connection instance
345          args = {}
346          if timeout is not None:
347              args['timeout'] = timeout
348  
349          if (ca_file is not None and hasattr(ssl, 'create_default_context') and
350                  parse.scheme == 'https'):
351              ctx = ssl.create_default_context(cafile=ca_file)
352              ctx.verify_mode = ssl.CERT_REQUIRED
353              args['context'] = ctx
354  
355          if not verify and parse.scheme == 'https' and (
356              hasattr(ssl, 'create_default_context')):
357              ctx = ssl.create_default_context()
358              ctx.check_hostname = False
359              ctx.verify_mode = ssl.CERT_NONE
360              args['context'] = ctx
361  
362          if self._proxy_host:
363              Logger.debug('UrlRequest: {0} - proxy via {1}:{2}'.format(
364                  id(self), self._proxy_host, self._proxy_port
365              ))
366              req = cls(self._proxy_host, self._proxy_port, **args)
367              if parse.scheme == 'https':
368                  req.set_tunnel(host, port, self._proxy_headers)
369              else:
370                  path = urlunparse(parse)
371          else:
372              req = cls(host, port, **args)
373  
374          # send request
375          method = self._method
376          if method is None:
377              method = 'GET' if body is None else 'POST'
378          req.request(method, path, body, headers or {})
379  
380          # read header
381          resp = req.getresponse()
382  
383          # read content
384          if report_progress or file_path is not None:
385              try:
386                  total_size = int(resp.getheader('content-length'))
387              except:
388                  total_size = -1
389  
390              # before starting the download, send a fake progress to permit the
391              # user to initialize his ui
392              if report_progress:
393                  q(('progress', resp, (0, total_size)))
394  
395              def get_chunks(fd=None):
396                  bytes_so_far = 0
397                  result = b''
398                  while 1:
399                      chunk = resp.read(chunk_size)
400                      if not chunk:
401                          break
402  
403                      if fd:
404                          fd.write(chunk)
405                      else:
406                          result += chunk
407  
408                      bytes_so_far += len(chunk)
409                      # report progress to user
410                      if report_progress:
411                          q(('progress', resp, (bytes_so_far, total_size)))
412                          trigger()
413                      if self._cancel_event.is_set():
414                          break
415                  return bytes_so_far, result
416  
417              if file_path is not None:
418                  with open(file_path, 'wb') as fd:
419                      bytes_so_far, result = get_chunks(fd)
420              else:
421                  bytes_so_far, result = get_chunks()
422  
423              # ensure that results are dispatched for the last chunk,
424              # avoid trigger
425              if report_progress:
426                  q(('progress', resp, (bytes_so_far, total_size)))
427                  trigger()
428          else:
429              result = resp.read()
430              try:
431                  if isinstance(result, bytes):
432                      result = result.decode('utf-8')
433              except UnicodeDecodeError:
434                  # if it's an image? decoding would not work
435                  pass
436          req.close()
437  
438          # return everything
439          return result, resp
440  
441      def get_connection_for_scheme(self, scheme):
442          '''Return the Connection class for a particular scheme. 
443          This is an internal function that can be expanded to support custom 
444          schemes. 
445   
446          Actual supported schemes: http, https. 
447          '''
448          if scheme == 'http':
449              return HTTPConnection
450          elif scheme == 'https' and HTTPSConnection is not None:
451              return HTTPSConnection
452          else:
453              raise Exception('No class for scheme %s' % scheme)

455      def decode_result(self, result, resp):
456          '''Decode the result fetched from url according to his Content-Type. 
457          Currently supports only application/json. 
458          '''
459          # Entry to decode url from the content type.
460          # For example, if the content type is a json, it will be automatically
461          # decoded.
462          content_type = resp.getheader('Content-Type', None)
463          if content_type is not None:
464              ct = content_type.split(';')[0]
465              if ct == 'application/json':
466                  if isinstance(result, bytes):
467                      result = result.decode('utf-8')
468                  try:
469                      return loads(result)
470                  except:
471                      return result
472  
473          return result

475      def _dispatch_result(self, dt):
476          while True:
477              # Read the result pushed on the queue, and dispatch to the client
478              try:
479                  result, resp, data = self._queue.pop()
480              except IndexError:
481                  return
482              if resp:
483                  # Small workaround in order to prevent the situation mentioned
484                  # in the comment below
485                  final_cookies = ""
486                  parsed_headers = []
487                  for key, value in resp.getheaders():
488                      if key == "Set-Cookie":
489                          final_cookies += "{};".format(value)
490                      else:
491                          parsed_headers.append((key, value))
492                  parsed_headers.append(("Set-Cookie", final_cookies[:-1]))
493  
494                  # XXX usage of dict can be dangerous if multiple headers
495                  # are set even if it's invalid. But it look like it's ok
496                  # ?  http://stackoverflow.com/questions/2454494/..
497                  # ..urllib2-multiple-set-cookie-headers-in-response
498                  self._resp_headers = dict(parsed_headers)
499                  self._resp_status = resp.status
500              if result == 'success':
501                  status_class = resp.status // 100
502  
503                  if status_class in (1, 2):
504                      if self._debug:
505                          Logger.debug('UrlRequest: {0} Download finished with'
506                                       ' {1} datalen'.format(id(self),
507                                                             len(data)))
508                      self._is_finished = True
509                      self._result = data
510                      if self.on_success:
511                          func = self.on_success()
512                          if func:
513                              func(self, data)
514  
515                  elif status_class == 3:
516                      if self._debug:
517                          Logger.debug('UrlRequest: {} Download '
518                                       'redirected'.format(id(self)))
519                      self._is_finished = True
520                      self._result = data
521                      if self.on_redirect:
522                          func = self.on_redirect()
523                          if func:
524                              func(self, data)
525  
526                  elif status_class in (4, 5):
527                      if self._debug:
528                          Logger.debug('UrlRequest: {} Download failed with '
529                                       'http error {}'.format(id(self),
530                                                              resp.status))
531                      self._is_finished = True
532                      self._result = data
533                      if self.on_failure:
534                          func = self.on_failure()
535                          if func:
536                              func(self, data)
537  
538              elif result == 'error':
539                  if self._debug:
540                      Logger.debug('UrlRequest: {0} Download error '
541                                   '<{1}>'.format(id(self), data))
542                  self._is_finished = True
543                  self._error = data
544                  if self.on_error:
545                      func = self.on_error()
546                      if func:
547                          func(self, data)
548  
549              elif result == 'progress':
550                  if self._debug:
551                      Logger.debug('UrlRequest: {0} Download progress '
552                                   '{1}'.format(id(self), data))
553                  if self.on_progress:
554                      func = self.on_progress()
555                      if func:
556                          func(self, data[0], data[1])
557  
558              elif result == 'killed':
559                  if self._debug:
560                      Logger.debug('UrlRequest: Cancelled by user')
561                  if self.on_cancel:
562                      func = self.on_cancel()
563                      if func:
564                          func(self)
565  
566              else:
567                  assert(0)

569      @property
570      def is_finished(self):
571          '''Return True if the request has finished, whether it's a 
572          success or a failure. 
573          '''
574          return self._is_finished

576      @property

578          '''Return the result of the request. 
579          This value is not determined until the request is finished.

611      def wait(self, delay=0.5):
612          '''Wait for the request to finish (until :attr:`resp_status` is not 
613          None) 
614   
615          .. note:: 
616              This method is intended to be used in the main thread, and the 
617              callback will be dispatched from the same thread 
618              from which you're calling. 
619   
620          .. versionadded:: 1.1.0 
621          '''
622          while self.resp_status is None:
623              self._dispatch_result(delay)
624              sleep(delay)
625  
626      def cancel(self):
627          '''Cancel the current request. It will be aborted, and the result 
628          will not be dispatched. Once cancelled, the callback on_cancel will 
629          be called. 
630   
631          .. versionadded:: 1.11.0 
632          '''
633          self._cancel_event.set()
