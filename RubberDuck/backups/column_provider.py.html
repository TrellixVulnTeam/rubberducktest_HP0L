1    # A sample shell column provider
2    # Mainly ported from MSDN article:
3    #  Using Shell Column Handlers for Detailed File Information,
4    #  Raymond Chen, Microsoft Corporation, February 2000
5    #
6    # To demostrate:
7    # * Execute this script to register the namespace.
8    # * Open Windows Explorer
9    # * Right-click an explorer column header - select "More"
10   # * Locate column 'pyc size' or 'pyo size', and add it to the view.
11   # This handler is providing that column data.
12   import sys, os, stat
13   import pythoncom
14   from win32com.shell import shell, shellcon
15   import commctrl
16   import winerror
17   from win32com.server.util import wrap
18   from pywintypes import IID
19   
20   IPersist_Methods = ["GetClassID"]
21   IColumnProvider_Methods = IPersist_Methods + [
22       "Initialize",
23       "GetColumnInfo",
24       "GetItemData",
25   ]
26   
27   
28   class ColumnProvider:
29       _reg_progid_ = "Python.ShellExtension.ColumnProvider"
30       _reg_desc_ = "Python Sample Shell Extension (Column Provider)"
31       _reg_clsid_ = IID("{0F14101A-E05E-4070-BD54-83DFA58C3D68}")
32       _com_interfaces_ = [
33           pythoncom.IID_IPersist,
34           shell.IID_IColumnProvider,
35       ]
36       _public_methods_ = IColumnProvider_Methods
37       # IPersist
38       def GetClassID(self):
39           return self._reg_clsid_
40   
41       # IColumnProvider
42       def Initialize(self, colInit):
43           flags, reserved, name = colInit
44           print("ColumnProvider initializing for file", name)
45   
46       def GetColumnInfo(self, index):
47           # We support exactly 2 columns - 'pyc size' and 'pyo size'
48           if index in [0, 1]:
49               # As per the MSDN sample, use our CLSID as the fmtid
50               if index == 0:
51                   ext = ".pyc"
52               else:
53                   ext = ".pyo"
54               title = ext + " size"
55               description = "Size of compiled %s file" % ext
56               col_id = (self._reg_clsid_, index)  # fmtid  # pid
57               col_info = (
58                   col_id,  # scid
59                   pythoncom.VT_I4,  # vt
60                   commctrl.LVCFMT_RIGHT,  # fmt
61                   20,  # cChars
62                   shellcon.SHCOLSTATE_TYPE_INT
63                   | shellcon.SHCOLSTATE_SECONDARYUI,  # csFlags
64                   title,
65                   description,
66               )
67               return col_info
68           return None  # Indicate no more columns.
69   
70       def GetItemData(self, colid, colData):
71           fmt_id, pid = colid
72           fmt_id == self._reg_clsid_
73           flags, attr, reserved, ext, name = colData
74           if ext.lower() not in [".py", ".pyw"]:
75               return None
76           if pid == 0:
77               ext = ".pyc"
78           else:
79               ext = ".pyo"
80           check_file = os.path.splitext(name)[0] + ext
81           try:
82               st = os.stat(check_file)
83               return st[stat.ST_SIZE]
84           except OSError:
85               # No file
86               return None

89   def DllRegisterServer():
90       import winreg
91   
92       # Special ColumnProvider key
93       key = winreg.CreateKey(
94           winreg.HKEY_CLASSES_ROOT,
95           "Folder\\ShellEx\\ColumnHandlers\\" + str(ColumnProvider._reg_clsid_),
96       )
97       winreg.SetValueEx(key, None, 0, winreg.REG_SZ, ColumnProvider._reg_desc_)
98       print(ColumnProvider._reg_desc_, "registration complete.")
99   
100  
101  def DllUnregisterServer():
102      import winreg
103  
104      try:
105          key = winreg.DeleteKey(
106              winreg.HKEY_CLASSES_ROOT,
107              "Folder\\ShellEx\\ColumnHandlers\\" + str(ColumnProvider._reg_clsid_),
108          )
109      except WindowsError as details:
110          import errno
111  
112          if details.errno != errno.ENOENT:
113              raise
114      print(ColumnProvider._reg_desc_, "unregistration complete.")
115  
116  
117  if __name__ == "__main__":
118      from win32com.server import register
119  
120      register.UseCommandLine(
121          ColumnProvider,
122          finalize_register=DllRegisterServer,
123          finalize_unregister=DllUnregisterServer,
124      )
