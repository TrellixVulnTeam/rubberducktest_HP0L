133  class TAPLexer(RegexLexer):
134      """ 
135      For Test Anything Protocol (TAP) output. 
136   
137      .. versionadded:: 2.1 
138      """
139      name = 'TAP'
140      aliases = ['tap']
141      filenames = ['*.tap']
142  
143      tokens = {
144          'root': [
145              # A TAP version may be specified.
146              (r'^TAP version \d+\n', Name.Namespace),
147  
148              # Specify a plan with a plan line.
149              (r'^1\.\.\d+', Keyword.Declaration, 'plan'),
150  
151              # A test failure
152              (r'^(not ok)([^\S\n]*)(\d*)',
153               bygroups(Generic.Error, Text, Number.Integer), 'test'),
154  
155              # A test success
156              (r'^(ok)([^\S\n]*)(\d*)',
157               bygroups(Keyword.Reserved, Text, Number.Integer), 'test'),
158  
159              # Diagnostics start with a hash.
160              (r'^#.*\n', Comment),
161  
162              # TAP's version of an abort statement.
163              (r'^Bail out!.*\n', Generic.Error),
164  
165              # TAP ignores any unrecognized lines.
166              (r'^.*\n', Text),
167          ],
168          'plan': [
169              # Consume whitespace (but not newline).
170              (r'[^\S\n]+', Text),
171  
172              # A plan may have a directive with it.
173              (r'#', Comment, 'directive'),
174  
175              # Or it could just end.
176              (r'\n', Comment, '#pop'),
177  
178              # Anything else is wrong.
179              (r'.*\n', Generic.Error, '#pop'),
180          ],
181          'test': [
182              # Consume whitespace (but not newline).
183              (r'[^\S\n]+', Text),
184  
185              # A test may have a directive with it.
186              (r'#', Comment, 'directive'),
187  
188              (r'\S+', Text),
189  
190              (r'\n', Text, '#pop'),
191          ],
192          'directive': [
193              # Consume whitespace (but not newline).
194              (r'[^\S\n]+', Comment),
195  
196              # Extract todo items.
197              (r'(?i)\bTODO\b', Comment.Preproc),
198  
199              # Extract skip items.
200              (r'(?i)\bSKIP\S*', Comment.Preproc),
201  
202              (r'\S+', Comment),
203  
204              (r'\n', Comment, '#pop:2'),
205          ],
206      }
