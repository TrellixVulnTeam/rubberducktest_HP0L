4    '''
5    These functions below, take a data element; if the data implements the
6    buffer interface, the data is returned unchanged, otherwise, a python
7    array is created from the data and returned.
8    
9    in both cases, the second parameter is initialized with a pointer to the
10   starting address of the returned buffer
11   
12   The return value is a tuple, of (original data, array), where in the first
13   case, array is None.
14   
15   The method used below (a untyped python array + array.data.as_floats pointer)
16   results in the fastest list to array creation and usage. Even malloc isn't
17   faster. Note, using memoryview (which we avoided for this case) is relatively
18   slow in cython.
19   
20   When the user passes in a memoryview type, we have no choice but to use the
21   memoryview passed in, though.
22   '''
23   cdef inline _ensure_float_view(data, float **f):
24       cdef array arr
25       cdef list src
26       cdef int i
27       cdef float [::1] memview
28       # do if/else instead of straight try/except because its faster for list
29       if not isinstance(data, (tuple, list)):
30           try:
31               memview = data
32               f[0] = &memview[0]
33               return data, None
34           except Exception as e:
35               import traceback; traceback.print_exc() 
36               src = list(data)
37               arr = clone(array('f'), len(src), False)
38               f[0] = arr.data.as_floats
39               for i in range(len(src)):
40                   f[0][i] = src[i]
41       else:
42           src = list(data)
43           arr = clone(array('f'), len(src), False)
44           f[0] = arr.data.as_floats
45           for i in range(len(src)):
46               f[0][i] = src[i]
47       return src, arr
