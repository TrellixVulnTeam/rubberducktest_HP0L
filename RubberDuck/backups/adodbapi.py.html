109  def connect(*args, **kwargs):  # --> a db-api connection object
110      """Connect to a database. 
111   
112      call using: 
113      :connection_string -- An ADODB formatted connection string, see: 
114           * http://www.connectionstrings.com 
115           * http://www.asp101.com/articles/john/connstring/default.asp 
116      :timeout -- A command timeout value, in seconds (default 30 seconds) 
117      """
118      co = Connection()  # make an empty connection object
119  
120      kwargs = process_connect_string.process(args, kwargs, True)
121  
122      try:  # connect to the database, using the connection information in kwargs
123          co.connect(kwargs)
124          return co
125      except (Exception) as e:
126          message = 'Error opening connection to "%s"' % co.connection_string
127          raise api.OperationalError(e, message)

152      """Format a collection of ADO Command Parameters. 
153   
154      Used by error reporting in _execute_command. 
155      """
156      try:
157          if show_value:
158              desc = [
159                  'Name: %s, Dir.: %s, Type: %s, Size: %s, Value: "%s", Precision: %s, NumericScale: %s'
160                  % (

172              desc = [
173                  "Name: %s, Dir.: %s, Type: %s, Size: %s, Precision: %s, NumericScale: %s"

182                  for p in ADOparameters

189  def _configure_parameter(p, value, adotype, settings_known):
190      """Configure the given ADO Parameter 'p' with the Python 'value'."""
191  
192      if adotype in api.adoBinaryTypes:
193          p.Size = len(value)
194          p.AppendChunk(value)
195  
196      elif isinstance(value, StringTypes):  # v2.1 Jevon
197          L = len(value)
198          if adotype in api.adoStringTypes:  # v2.2.1 Cole
199              if settings_known:
200                  L = min(L, p.Size)  # v2.1 Cole limit data to defined size
201              p.Value = value[:L]  # v2.1 Jevon & v2.1 Cole
202          else:
203              p.Value = value  # dont limit if db column is numeric
204          if L > 0:  # v2.1 Cole something does not like p.Size as Zero
205              p.Size = L  # v2.1 Jevon
206  
207      elif isinstance(value, decimal.Decimal):
208          if api.onIronPython:
209              s = str(value)
210              p.Value = s
211              p.Size = len(s)
212          else:
213              p.Value = value
214          exponent = value.as_tuple()[2]
215          digit_count = len(value.as_tuple()[1])
216          p.Precision = digit_count
217          if exponent == 0:
218              p.NumericScale = 0
219          elif exponent < 0:
220              p.NumericScale = -exponent
221              if p.Precision < p.NumericScale:
222                  p.Precision = p.NumericScale
223          else:  # exponent > 0:
224              p.NumericScale = 0
225              p.Precision = digit_count + exponent
226  
227      elif type(value) in dateconverter.types:
228          if settings_known and adotype in api.adoDateTimeTypes:
229              p.Value = dateconverter.COMDate(value)
230          else:  # probably a string
231              # provide the date as a string in the format 'YYYY-MM-dd'
232              s = dateconverter.DateObjectToIsoFormatString(value)
233              p.Value = s
234              p.Size = len(s)
235  
236      elif api.onIronPython and isinstance(value, longType):  # Iron Python Long
237          s = str(value)  # feature workaround for IPy 2.0
238          p.Value = s
239  
240      elif adotype == adc.adEmpty:  # ADO will not let you specify a null column
241          p.Type = (
242              adc.adInteger
243          )  # so we will fake it to be an integer (just to have something)
244          p.Value = None  # and pass in a Null *value*
245  
246          # For any other type, set the value and let pythoncom do the right thing.
247      else:
248          p.Value = value

285      def connect(self, kwargs, connection_maker=make_COM_connecter):
286          if verbose > 9:
287              print("kwargs=", repr(kwargs))
288          try:
289              self.connection_string = (
290                  kwargs["connection_string"] % kwargs
291              )  # insert keyword arguments
292          except (Exception) as e:
293              self._raiseConnectionError(
294                  KeyError, "Python string format error in connection string->"
295              )
296          self.timeout = kwargs.get("timeout", 30)
297          self.mode = kwargs.get("mode", adc.adModeUnknown)
298          self.kwargs = kwargs
299          if verbose:
300              print('%s attempting: "%s"' % (version, self.connection_string))
301          self.connector = connection_maker()
302          self.connector.ConnectionTimeout = self.timeout
303          self.connector.ConnectionString = self.connection_string
304          self.connector.Mode = self.mode
305  
306          try:
307              self.connector.Open()  # Open the ADO connection
308          except api.Error:
309              self._raiseConnectionError(
310                  api.DatabaseError,
311                  "ADO error trying to Open=%s" % self.connection_string,
312              )
313  
314          try:  # Stefan Fuchs; support WINCCOLEDBProvider
315              if getIndexedValue(self.connector.Properties, "Transaction DDL").Value != 0:
316                  self.supportsTransactions = True
317          except pywintypes.com_error:
318              pass  # Stefan Fuchs
319          self.dbms_name = getIndexedValue(self.connector.Properties, "DBMS Name").Value
320          try:  # Stefan Fuchs
321              self.dbms_version = getIndexedValue(
322                  self.connector.Properties, "DBMS Version"
323              ).Value
324          except pywintypes.com_error:
325              pass  # Stefan Fuchs
326          self.connector.CursorLocation = defaultCursorLocation  # v2.1 Rose
327          if self.supportsTransactions:
328              self.connector.IsolationLevel = defaultIsolationLevel
329              self._autocommit = bool(kwargs.get("autocommit", False))
330              if not self._autocommit:
331                  self.transaction_level = (
332                      self.connector.BeginTrans()
333                  )  # Disables autocommit & inits transaction_level
334          else:
335              self._autocommit = True
336          if "paramstyle" in kwargs:
337              self.paramstyle = kwargs["paramstyle"]  # let setattr do the error checking
338          self.messages = []
339          if verbose:
340              print("adodbapi New connection at %X" % id(self))

348      def _closeAdoConnection(self):  # all v2.1 Rose
349          """close the underlying ADO Connection object, 
350          rolling it back first if it supports transactions."""
351          if self.connector is None:
352              return
353          if not self._autocommit:
354              if self.transaction_level:
355                  try:
356                      self.connector.RollbackTrans()
357                  except:
358                      pass
359          self.connector.Close()
360          if verbose:
361              print("adodbapi Closed connection at %X" % id(self))

363      def close(self):
364          """Close the connection now (rather than whenever __del__ is called). 
365   
366          The connection will be unusable from this point forward; 
367          an Error (or subclass) exception will be raised if any operation is attempted with the connection. 
368          The same applies to all cursor objects trying to use the connection. 
369          """
370          for crsr in list(self.cursors.values())[
371              :
372          ]:  # copy the list, then close each one
373              crsr.close(dont_tell_me=True)  # close without back-link clearing
374          self.messages = []
375          try:
376              self._closeAdoConnection()  # v2.1 Rose
377          except (Exception) as e:
378              self._raiseConnectionError(sys.exc_info()[0], sys.exc_info()[1])
379  
380          self.connector = None  # v2.4.2.2 fix subtle timeout bug
381          # per M.Hammond: "I expect the benefits of uninitializing are probably fairly small,
382          #    so never uninitializing will probably not cause any problems."

384      def commit(self):
385          """Commit any pending transaction to the database. 
386   
387          Note that if the database supports an auto-commit feature, 
388          this must be initially off. An interface method may be provided to turn it back on. 
389          Database modules that do not support transactions should implement this method with void functionality. 
390          """
391          self.messages = []
392          if not self.supportsTransactions:
393              return
394  
395          try:
396              self.transaction_level = self.connector.CommitTrans()
397              if verbose > 1:
398                  print("commit done on connection at %X" % id(self))
399              if not (
400                  self._autocommit
401                  or (self.connector.Attributes & adc.adXactAbortRetaining)
402              ):
403                  # If attributes has adXactCommitRetaining it performs retaining commits that is,
404                  # calling CommitTrans automatically starts a new transaction. Not all providers support this.
405                  # If not, we will have to start a new transaction by this command:
406                  self.transaction_level = self.connector.BeginTrans()
407          except Exception as e:
408              self._raiseConnectionError(api.ProgrammingError, e)

410      def _rollback(self):
411          """In case a database does provide transactions this method causes the the database to roll back to 
412          the start of any pending transaction. Closing a connection without committing the changes first will 
413          cause an implicit rollback to be performed. 
414   
415          If the database does not support the functionality required by the method, the interface should 
416          throw an exception in case the method is used. 
417          The preferred approach is to not implement the method and thus have Python generate 
418          an AttributeError in case the method is requested. This allows the programmer to check for database 
419          capabilities using the standard hasattr() function. 
420   
421          For some dynamically configured interfaces it may not be appropriate to require dynamically making 
422          the method available. These interfaces should then raise a NotSupportedError to indicate the 
423          non-ability to perform the roll back when the method is invoked. 
424          """
425          self.messages = []
426          if (
427              self.transaction_level
428          ):  # trying to roll back with no open transaction causes an error
429              try:
430                  self.transaction_level = self.connector.RollbackTrans()
431                  if verbose > 1:
432                      print("rollback done on connection at %X" % id(self))
433                  if not self._autocommit and not (
434                      self.connector.Attributes & adc.adXactAbortRetaining
435                  ):
436                      # If attributes has adXactAbortRetaining it performs retaining aborts that is,
437                      # calling RollbackTrans automatically starts a new transaction. Not all providers support this.
438                      # If not, we will have to start a new transaction by this command:
439                      if (
440                          not self.transaction_level
441                      ):  # if self.transaction_level == 0 or self.transaction_level is None:
442                          self.transaction_level = self.connector.BeginTrans()
443              except Exception as e:
444                  self._raiseConnectionError(api.ProgrammingError, e)

446      def __setattr__(self, name, value):
447          if name == "autocommit":  # extension: allow user to turn autocommit on or off
448              if self.supportsTransactions:
449                  object.__setattr__(self, "_autocommit", bool(value))
450                  try:
451                      self._rollback()  # must clear any outstanding transactions
452                  except:
453                      pass
454              return
455          elif name == "paramstyle":
456              if value not in api.accepted_paramstyles:
457                  self._raiseConnectionError(
458                      api.NotSupportedError,
459                      'paramstyle="%s" not in:%s'
460                      % (value, repr(api.accepted_paramstyles)),
461                  )
462          elif name == "variantConversions":
463              value = copy.copy(
464                  value
465              )  # make a new copy -- no changes in the default, please
466          object.__setattr__(self, name, value)
467  
468      def __getattr__(self, item):
469          if (
470              item == "rollback"
471          ):  # the rollback method only appears if the database supports transactions
472              if self.supportsTransactions:
473                  return (
474                      self._rollback
475                  )  # return the rollback method so the caller can execute it.
476              else:
477                  raise AttributeError("this data provider does not support Rollback")
478          elif item == "autocommit":
479              return self._autocommit

481              raise AttributeError(

518      def _suggest_error_class(self):
519          """Introspect the current ADO Errors and determine an appropriate error class. 
520   
521          Error.SQLState is a SQL-defined error condition, per the SQL specification: 
522          http://www.contrib.andrew.cmu.edu/~shadow/sql/sql1992.txt 
523   
524          The 23000 class of errors are integrity errors. 
525          Error 40002 is a transactional integrity error. 
526          """
527          if self.connector is not None:
528              for e in self.connector.Errors:
529                  state = str(e.SQLState)
530                  if state.startswith("23") or state == "40002":
531                      return api.IntegrityError
532          return api.DatabaseError

563  class Cursor(object):
564      ## ** api required attributes:
565      ## description...
566      ##    This read-only attribute is a sequence of 7-item sequences.
567      ##    Each of these sequences contains information describing one result column:
568      ##        (name, type_code, display_size, internal_size, precision, scale, null_ok).
569      ##    This attribute will be None for operations that do not return rows or if the
570      ##    cursor has not had an operation invoked via the executeXXX() method yet.
571      ##    The type_code can be interpreted by comparing it to the Type Objects specified in the section below.
572      ## rowcount...
573      ##    This read-only attribute specifies the number of rows that the last executeXXX() produced
574      ##    (for DQL statements like select) or affected (for DML statements like update or insert).
575      ##    The attribute is -1 in case no executeXXX() has been performed on the cursor or
576      ##    the rowcount of the last operation is not determinable by the interface.[7]
577      ## arraysize...
578      ##    This read/write attribute specifies the number of rows to fetch at a time with fetchmany().
579      ##    It defaults to 1 meaning to fetch a single row at a time.
580      ##    Implementations must observe this value with respect to the fetchmany() method,
581      ##    but are free to interact with the database a single row at a time.
582      ##    It may also be used in the implementation of executemany().
583      ## ** extension attributes:
584      ## paramstyle...
585      ##   allows the programmer to override the connection's default paramstyle
586      ## errorhandler...
587      ##   allows the programmer to override the connection's default error handler
588  
589      def __init__(self, connection):
590          self.command = None
591          self._ado_prepared = False
592          self.messages = []
593          self.connection = connection
594          self.paramstyle = connection.paramstyle  # used for overriding the paramstyle
595          self._parameter_names = []
596          self.recordset_is_remote = False
597          self.rs = None  # the ADO recordset for this cursor
598          self.converters = []  # conversion function for each column
599          self.columnNames = {}  # names of columns {lowercase name : number,...}
600          self.numberOfColumns = 0
601          self._description = None
602          self.rowcount = -1
603          self.errorhandler = connection.errorhandler
604          self.arraysize = 1
605          connection._i_am_here(self)
606          if verbose:
607              print(
608                  "%s New cursor at %X on conn %X"
609                  % (version, id(self), id(self.connection))
610              )

640      def build_column_info(self, recordset):
641          self.converters = []  # convertion function for each column
642          self.columnNames = {}  # names of columns {lowercase name : number,...}
643          self._description = None
644  
645          # if EOF and BOF are true at the same time, there are no records in the recordset
646          if (recordset is None) or (recordset.State == adc.adStateClosed):
647              self.rs = None
648              self.numberOfColumns = 0
649              return
650          self.rs = recordset  # v2.1.1 bkline
651          self.recordset_format = api.RS_ARRAY if api.onIronPython else api.RS_WIN_32
652          self.numberOfColumns = recordset.Fields.Count
653          try:
654              varCon = self.connection.variantConversions
655          except AttributeError:
656              varCon = api.variantConversions
657          for i in range(self.numberOfColumns):
658              f = getIndexedValue(self.rs.Fields, i)
659              try:

776      def _execute_command(self):
777          # Stored procedures may have an integer return value
778          self.return_value = None
779          recordset = None
780          count = -1  # default value
781          if verbose:
782              print('Executing command="%s"' % self.commandText)
783          try:
784              # ----- the actual SQL is executed here ---
785              if api.onIronPython:
786                  ra = Reference[int]()
787                  recordset = self.cmd.Execute(ra)
788                  count = ra.Value
789              else:  # pywin32
790                  recordset, count = self.cmd.Execute()
791              # ----- ------------------------------- ---
792          except (Exception) as e:
793              _message = ""
794              if hasattr(e, "args"):
795                  _message += str(e.args) + "\n"
796              _message += "Command:\n%s\nParameters:\n%s" % (
797                  self.commandText,
798                  format_parameters(self.cmd.Parameters, True),
799              )
800              klass = self.connection._suggest_error_class()
801              self._raiseCursorError(klass, _message)
802          try:
803              self.rowcount = recordset.RecordCount
804          except:
805              self.rowcount = count
806          self.build_column_info(recordset)
807  
808          # The ADO documentation hints that obtaining the recordcount may be timeconsuming
809          #   "If the Recordset object does not support approximate positioning, this property
810          #    may be a significant drain on resources # [ekelund]
811          # Therefore, COM will not return rowcount for server-side cursors. [Cole]
812          # Client-side cursors (the default since v2.8) will force a static
813          # cursor, and rowcount will then be set accurately [Cole]

816          return self.rowcount

851      def callproc(self, procname, parameters=None):
852          """Call a stored database procedure with the given name. 
853          The sequence of parameters must contain one entry for each 
854          argument that the sproc expects. The result of the 
855          call is returned as modified copy of the input 
856          sequence.  Input parameters are left untouched, output and 
857          input/output parameters replaced with possibly new values. 
858   
859          The sproc may also provide a result set as output, 
860          which is available through the standard .fetch*() methods. 
861          Extension: A "return_value" property may be set on the 
862          cursor if the sproc defines an integer return value. 
863          """
864          self._parameter_names = []
865          self.commandText = procname
866          self._new_command(command_type=adc.adCmdStoredProc)
867          self._buildADOparameterList(parameters, sproc=True)
868          if verbose > 2:
869              print(
870                  "Calling Stored Proc with Params=",
871                  format_parameters(self.cmd.Parameters, True),
872              )
873          self._execute_command()
874          return self.get_returned_parameters()

887      def _buildADOparameterList(self, parameters, sproc=False):
888          self.parameters = parameters
889          if parameters is None:
890              parameters = []
891  
892          # Note: ADO does not preserve the parameter list, even if "Prepared" is True, so we must build every time.
893          parameters_known = False
894          if sproc:  # needed only if we are calling a stored procedure
895              try:  # attempt to use ADO's parameter list
896                  self.cmd.Parameters.Refresh()
897                  if verbose > 2:
898                      print(
899                          "ADO detected Params=",
900                          format_parameters(self.cmd.Parameters, True),
901                      )
902                      print("Program Parameters=", repr(parameters))
903                  parameters_known = True
904              except api.Error:
905                  if verbose:
906                      print("ADO Parameter Refresh failed")
907                  pass
908              else:
909                  if len(parameters) != self.cmd.Parameters.Count - 1:
910                      raise api.ProgrammingError(
911                          "You must supply %d parameters for this stored procedure"
912                          % (self.cmd.Parameters.Count - 1)
913                      )
914          if sproc or parameters != []:
915              i = 0
916              if parameters_known:  # use ado parameter list
917                  if self._parameter_names:  # named parameters
918                      for i, pm_name in enumerate(self._parameter_names):
919                          p = getIndexedValue(self.cmd.Parameters, i)
920                          try:
921                              _configure_parameter(
922                                  p, parameters[pm_name], p.Type, parameters_known
923                              )
924                          except (Exception) as e:
925                              _message = (
926                                  "Error Converting Parameter %s: %s, %s <- %s\n"
927                                  % (
928                                      p.Name,
929                                      adc.ado_type_name(p.Type),
930                                      p.Value,
931                                      repr(parameters[pm_name]),
932                                  )
933                              )
934                              self._raiseCursorError(
935                                  api.DataError, _message + "->" + repr(e.args)
936                              )
937                  else:  # regular sequence of parameters
938                      for value in parameters:
939                          p = getIndexedValue(self.cmd.Parameters, i)
940                          if (
941                              p.Direction == adc.adParamReturnValue
942                          ):  # this is an extra parameter added by ADO
943                              i += 1  # skip the extra
944                              p = getIndexedValue(self.cmd.Parameters, i)
945                          try:
946                              _configure_parameter(p, value, p.Type, parameters_known)
947                          except Exception as e:
948                              _message = (
949                                  "Error Converting Parameter %s: %s, %s <- %s\n"
950                                  % (
951                                      p.Name,
952                                      adc.ado_type_name(p.Type),
953                                      p.Value,
954                                      repr(value),
955                                  )
956                              )
957                              self._raiseCursorError(
958                                  api.DataError, _message + "->" + repr(e.args)
959                              )
960                          i += 1
961              else:  # -- build own parameter list
962                  if (
963                      self._parameter_names
964                  ):  # we expect a dictionary of parameters, this is the list of expected names
965                      for parm_name in self._parameter_names:
966                          elem = parameters[parm_name]
967                          adotype = api.pyTypeToADOType(elem)
968                          p = self.cmd.CreateParameter(
969                              parm_name, adotype, adc.adParamInput
970                          )
971                          _configure_parameter(p, elem, adotype, parameters_known)
972                          try:
973                              self.cmd.Parameters.Append(p)
974                          except Exception as e:
975                              _message = "Error Building Parameter %s: %s, %s <- %s\n" % (
976                                  p.Name,
977                                  adc.ado_type_name(p.Type),
978                                  p.Value,
979                                  repr(elem),
980                              )
981                              self._raiseCursorError(
982                                  api.DataError, _message + "->" + repr(e.args)
983                              )
984                  else:  # expecting the usual sequence of parameters
985                      if sproc:
986                          p = self.cmd.CreateParameter(
987                              "@RETURN_VALUE", adc.adInteger, adc.adParamReturnValue
988                          )
989                          self.cmd.Parameters.Append(p)
990  
991                      for elem in parameters:
992                          name = "p%i" % i
993                          adotype = api.pyTypeToADOType(elem)
994                          p = self.cmd.CreateParameter(
995                              name, adotype, adc.adParamInput
996                          )  # Name, Type, Direction, Size, Value
997                          _configure_parameter(p, elem, adotype, parameters_known)
998                          try:
999                              self.cmd.Parameters.Append(p)
1000                         except Exception as e:
1001                             _message = "Error Building Parameter %s: %s, %s <- %s\n" % (
1002                                 p.Name,
1003                                 adc.ado_type_name(p.Type),
1004                                 p.Value,
1005                                 repr(elem),
1006                             )
1007                             self._raiseCursorError(
1008                                 api.DataError, _message + "->" + repr(e.args)
1009                             )
1010                         i += 1
1011                 if self._ado_prepared == "setup":
1012                     self._ado_prepared = (
1013                         True  # parameters will be "known" by ADO next loop
1014                     )

1016     def execute(self, operation, parameters=None):
1017         """Prepare and execute a database operation (query or command). 
1018  
1019         Parameters may be provided as sequence or mapping and will be bound to variables in the operation. 
1020         Variables are specified in a database-specific notation 
1021         (see the module's paramstyle attribute for details). [5] 
1022         A reference to the operation will be retained by the cursor. 
1023         If the same operation object is passed in again, then the cursor 
1024         can optimize its behavior. This is most effective for algorithms 
1025         where the same operation is used, but different parameters are bound to it (many times). 
1026  
1027         For maximum efficiency when reusing an operation, it is best to use 
1028         the setinputsizes() method to specify the parameter types and sizes ahead of time. 
1029         It is legal for a parameter to not match the predefined information; 
1030         the implementation should compensate, possibly with a loss of efficiency. 
1031  
1032         The parameters may also be specified as list of tuples to e.g. insert multiple rows in 
1033         a single operation, but this kind of usage is depreciated: executemany() should be used instead. 
1034  
1035         Return value is not defined. 
1036  
1037         [5] The module will use the __getitem__ method of the parameters object to map either positions 
1038         (integers) or names (strings) to parameter values. This allows for both sequences and mappings 
1039         to be used as input. 
1040         The term "bound" refers to the process of binding an input value to a database execution buffer. 
1041         In practical terms, this means that the input value is directly used as a value in the operation. 
1042         The client should not be required to "escape" the value so that it can be used -- the value 
1043         should be equal to the actual database value."""
1044         if (
1045             self.command is not operation
1046             or self._ado_prepared == "setup"
1047             or not hasattr(self, "commandText")
1048         ):
1049             if self.command is not operation:
1050                 self._ado_prepared = False
1051                 self.command = operation
1052             self._parameter_names = []
1053             self.commandText = (
1054                 operation
1055                 if (self.paramstyle == "qmark" or not parameters)
1056                 else self._reformat_operation(operation, parameters)
1057             )
1058         self._new_command()
1059         self._buildADOparameterList(parameters)
1060         if verbose > 3:
1061             print("Params=", format_parameters(self.cmd.Parameters, True))
1062         self._execute_command()
1063 
1064     def executemany(self, operation, seq_of_parameters):
1065         """Prepare a database operation (query or command) 
1066         and then execute it against all parameter sequences or mappings found in the sequence seq_of_parameters. 
1067  
1068             Return values are not defined. 
1069         """
1070         self.messages = list()
1071         total_recordcount = 0
1072 
1073         self.prepare(operation)
1074         for params in seq_of_parameters:
1075             self.execute(self.command, params)
1076             if self.rowcount == -1:
1077                 total_recordcount = -1
1078             if total_recordcount != -1:
1079                 total_recordcount += self.rowcount
1080         self.rowcount = total_recordcount

1082     def _fetch(self, limit=None):
1083         """Fetch rows from the current recordset. 
1084  
1085         limit -- Number of rows to fetch, or None (default) to fetch all rows. 
1086         """
1087         if self.connection is None or self.rs is None:
1088             self._raiseCursorError(
1089                 api.FetchFailedError, "fetch() on closed connection or empty query set"
1090             )
1091             return
1092 
1093         if self.rs.State == adc.adStateClosed or self.rs.BOF or self.rs.EOF:
1094             return list()
1095         if limit:  # limit number of rows retrieved
1096             ado_results = self.rs.GetRows(limit)
1097         else:  # get all rows
1098             ado_results = self.rs.GetRows()
1099         if (
1100             self.recordset_format == api.RS_ARRAY
1101         ):  # result of GetRows is a two-dimension array
1102             length = (
1103                 len(ado_results) // self.numberOfColumns
1104             )  # length of first dimension
1105         else:  # pywin32
1106             length = len(ado_results[0])  # result of GetRows is tuples in a tuple
1107         fetchObject = api.SQLrows(
1108             ado_results, length, self
1109         )  # new object to hold the results of the fetch
1110         return fetchObject

1125     def fetchmany(self, size=None):
1126         """Fetch the next set of rows of a query result, returning a list of tuples. An empty sequence is returned when no more rows are available. 
1127  
1128         The number of rows to fetch per call is specified by the parameter. 
1129         If it is not given, the cursor's arraysize determines the number of rows to be fetched. 
1130         The method should try to fetch as many rows as indicated by the size parameter. 
1131         If this is not possible due to the specified number of rows not being available, 
1132         fewer rows may be returned. 
1133  
1134         An Error (or subclass) exception is raised if the previous call to executeXXX() 
1135         did not produce any result set or no call was issued yet. 
1136  
1137         Note there are performance considerations involved with the size parameter. 
1138         For optimal performance, it is usually best to use the arraysize attribute. 
1139         If the size parameter is used, then it is best for it to retain the same value from 
1140         one fetchmany() call to the next.
