263      def helpTestDataType(
264          self,
265          sqlDataTypeString,
266          DBAPIDataTypeString,
267          pyData,
268          pyDataInputAlternatives=None,
269          compareAlmostEqual=None,
270          allowedReturnValues=None,
271      ):
272          self.helpForceDropOnTblTemp()
273          conn = self.getConnection()
274          crsr = conn.cursor()
275          tabdef = (
276              """ 
277              CREATE TABLE xx_%s ( 
278                  fldId integer NOT NULL, 
279                  fldData """
280              % config.tmp
281              + sqlDataTypeString
282              + ")\n"
283          )
284  
285          crsr.execute(tabdef)
286  
287          # Test Null values mapped to None
288          crsr.execute("INSERT INTO xx_%s (fldId) VALUES (1)" % config.tmp)
289  
290          crsr.execute("SELECT fldId,fldData FROM xx_%s" % config.tmp)
291          rs = crsr.fetchone()
292          self.assertEqual(rs[1], None)  # Null should be mapped to None
293          assert rs[0] == 1
294  
295          # Test description related
296          descTuple = crsr.description[1]
297          assert descTuple[0] in ["fldData", "flddata"], 'was "%s" expected "%s"' % (
298              descTuple[0],
299              "fldData",
300          )
301  
302          if DBAPIDataTypeString == "STRING":
303              assert descTuple[1] == api.STRING, 'was "%s" expected "%s"' % (
304                  descTuple[1],
305                  api.STRING.values,
306              )
307          elif DBAPIDataTypeString == "NUMBER":
308              assert descTuple[1] == api.NUMBER, 'was "%s" expected "%s"' % (
309                  descTuple[1],
310                  api.NUMBER.values,
311              )
312          elif DBAPIDataTypeString == "BINARY":
313              assert descTuple[1] == api.BINARY, 'was "%s" expected "%s"' % (
314                  descTuple[1],
315                  api.BINARY.values,
316              )
317          elif DBAPIDataTypeString == "DATETIME":
318              assert descTuple[1] == api.DATETIME, 'was "%s" expected "%s"' % (
319                  descTuple[1],
320                  api.DATETIME.values,
321              )
322          elif DBAPIDataTypeString == "ROWID":
323              assert descTuple[1] == api.ROWID, 'was "%s" expected "%s"' % (
324                  descTuple[1],
325                  api.ROWID.values,
326              )
327          elif DBAPIDataTypeString == "UUID":
328              assert descTuple[1] == api.OTHER, 'was "%s" expected "%s"' % (
329                  descTuple[1],
330                  api.OTHER.values,
331              )
332          else:
333              raise NotImplementedError  # "DBAPIDataTypeString not provided"
334  
335          # Test data binding
336          inputs = [pyData]
337          if pyDataInputAlternatives:
338              inputs.extend(pyDataInputAlternatives)
339          inputs = set(inputs)  # removes redundant string==unicode tests
340          fldId = 1
341          for inParam in inputs:
342              fldId += 1
343              try:
344                  crsr.execute(
345                      "INSERT INTO xx_%s (fldId,fldData) VALUES (?,?)" % config.tmp,
346                      (fldId, inParam),
347                  )
348              except:
349                  if self.remote:
350                      for message in crsr.messages:
351                          print(message)
352                  else:
353                      conn.printADOerrors()
354                  raise
355              crsr.execute(
356                  "SELECT fldData FROM xx_%s WHERE ?=fldID" % config.tmp, [fldId]
357              )
358              rs = crsr.fetchone()
359              if allowedReturnValues:
360                  allowedTypes = tuple([type(aRV) for aRV in allowedReturnValues])
361                  assert isinstance(
362                      rs[0], allowedTypes
363                  ), 'result type "%s" must be one of %s' % (type(rs[0]), allowedTypes)
364              else:
365                  assert isinstance(
366                      rs[0], type(pyData)
367                  ), 'result type "%s" must be instance of %s' % (
368                      type(rs[0]),
369                      type(pyData),
370                  )
371  
372              if compareAlmostEqual and DBAPIDataTypeString == "DATETIME":
373                  iso1 = adodbapi.dateconverter.DateObjectToIsoFormatString(rs[0])
374                  iso2 = adodbapi.dateconverter.DateObjectToIsoFormatString(pyData)
375                  self.assertEqual(iso1, iso2)
376              elif compareAlmostEqual:
377                  s = float(pyData)
378                  v = float(rs[0])
379                  assert (
380                      abs(v - s) / s < 0.00001
381                  ), "Values not almost equal recvd=%s, expected=%f" % (rs[0], s)
382              else:
383                  if allowedReturnValues:
384                      ok = False
385                      self.assertTrue(
386                          rs[0] in allowedReturnValues,
387                          'Value "%s" not in %s' % (repr(rs[0]), allowedReturnValues),
388                      )
389                  else:
390                      self.assertEqual(
391                          rs[0],
392                          pyData,
393                          'Values are not equal recvd="%s", expected="%s"'
394                          % (rs[0], pyData),
395                      )
