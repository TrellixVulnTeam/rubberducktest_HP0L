1    # Demo RegisterDeviceNotification etc.  Creates a hidden window to receive
2    # notifications.  See serviceEvents.py for an example of a service doing
3    # that.
4    import sys, time
5    import win32gui, win32con, win32api, win32file
6    import win32gui_struct, winnt

8    # These device GUIDs are from Ioevent.h in the Windows SDK.  Ideally they
9    # could be collected somewhere for pywin32...
10   GUID_DEVINTERFACE_USB_DEVICE = "{A5DCBF10-6530-11D2-901F-00C04FB951ED}"

12   # WM_DEVICECHANGE message handler.
13   def OnDeviceChange(hwnd, msg, wp, lp):
14       # Unpack the 'lp' into the appropriate DEV_BROADCAST_* structure,
15       # using the self-identifying data inside the DEV_BROADCAST_HDR.
16       info = win32gui_struct.UnpackDEV_BROADCAST(lp)
17       print("Device change notification:", wp, str(info))
18       if (
19           wp == win32con.DBT_DEVICEQUERYREMOVE
20           and info.devicetype == win32con.DBT_DEVTYP_HANDLE
21       ):
22           # Our handle is stored away in the structure - just close it
23           print("Device being removed - closing handle")
24           win32file.CloseHandle(info.handle)
25           # and cancel our notifications - if it gets plugged back in we get
26           # the same notification and try and close the same handle...
27           win32gui.UnregisterDeviceNotification(info.hdevnotify)
28       return True

31   def TestDeviceNotifications(dir_names):
32       wc = win32gui.WNDCLASS()
33       wc.lpszClassName = "test_devicenotify"
34       wc.style = win32con.CS_GLOBALCLASS | win32con.CS_VREDRAW | win32con.CS_HREDRAW
35       wc.hbrBackground = win32con.COLOR_WINDOW + 1
36       wc.lpfnWndProc = {win32con.WM_DEVICECHANGE: OnDeviceChange}
37       class_atom = win32gui.RegisterClass(wc)
38       hwnd = win32gui.CreateWindow(
39           wc.lpszClassName,
40           "Testing some devices",
41           # no need for it to be visible.
42           win32con.WS_CAPTION,
43           100,
44           100,
45           900,
46           900,
47           0,
48           0,
49           0,
50           None,
51       )
52   
53       hdevs = []
54       # Watch for all USB device notifications
55       filter = win32gui_struct.PackDEV_BROADCAST_DEVICEINTERFACE(
56           GUID_DEVINTERFACE_USB_DEVICE
57       )
58       hdev = win32gui.RegisterDeviceNotification(
59           hwnd, filter, win32con.DEVICE_NOTIFY_WINDOW_HANDLE
60       )
61       hdevs.append(hdev)
62       # and create handles for all specified directories
63       for d in dir_names:
64           hdir = win32file.CreateFile(
65               d,
66               winnt.FILE_LIST_DIRECTORY,
67               winnt.FILE_SHARE_READ | winnt.FILE_SHARE_WRITE | winnt.FILE_SHARE_DELETE,
68               None,  # security attributes
69               win32con.OPEN_EXISTING,
70               win32con.FILE_FLAG_BACKUP_SEMANTICS
71               | win32con.FILE_FLAG_OVERLAPPED,  # required privileges: SE_BACKUP_NAME and SE_RESTORE_NAME.
72               None,
73           )
74   
75           filter = win32gui_struct.PackDEV_BROADCAST_HANDLE(hdir)
76           hdev = win32gui.RegisterDeviceNotification(
77               hwnd, filter, win32con.DEVICE_NOTIFY_WINDOW_HANDLE
78           )
79           hdevs.append(hdev)
80   
81       # now start a message pump and wait for messages to be delivered.
82       print("Watching", len(hdevs), "handles - press Ctrl+C to terminate, or")
83       print("add and remove some USB devices...")
84       if not dir_names:
85           print("(Note you can also pass paths to watch on the command-line - eg,")
86           print("pass the root of an inserted USB stick to see events specific to")
87           print("that volume)")
88       while 1:
89           win32gui.PumpWaitingMessages()
90           time.sleep(0.01)
91       win32gui.DestroyWindow(hwnd)
92       win32gui.UnregisterClass(wc.lpszClassName, None)
93

95   if __name__ == "__main__":
96       # optionally pass device/directory names to watch for notifications.
97       # Eg, plug in a USB device - assume it connects as E: - then execute:
98       # % win32gui_devicenotify.py E:
99       # Then remove and insert the device.
100      TestDeviceNotifications(sys.argv[1:])
