31   def TestText():
32       OpenClipboard()
33       try:
34           text = "Hello from Python"
35           text_bytes = str2bytes(text)
36           SetClipboardText(text)
37           got = GetClipboardData(win32con.CF_TEXT)
38           # CF_TEXT always gives us 'bytes' back .
39           assert got == text_bytes, "Didnt get the correct result back - '%r'." % (got,)
40       finally:
41           CloseClipboard()
42   
43       OpenClipboard()
44       try:
45           # CF_UNICODE text always gives unicode objects back.
46           got = GetClipboardData(win32con.CF_UNICODETEXT)
47           assert got == text, "Didnt get the correct result back - '%r'." % (got,)
48           assert type(got) == str, "Didnt get the correct result back - '%r'." % (got,)
49   
50           # CF_OEMTEXT is a bytes-based format.
51           got = GetClipboardData(win32con.CF_OEMTEXT)
52           assert got == text_bytes, "Didnt get the correct result back - '%r'." % (got,)
53   
54           # Unicode tests
55           EmptyClipboard()
56           text = "Hello from Python unicode"
57           text_bytes = str2bytes(text)
58           # Now set the Unicode value
59           SetClipboardData(win32con.CF_UNICODETEXT, text)
60           # Get it in Unicode.
61           got = GetClipboardData(win32con.CF_UNICODETEXT)
62           assert got == text, "Didnt get the correct result back - '%r'." % (got,)
63           assert type(got) == str, "Didnt get the correct result back - '%r'." % (got,)
64   
65           # Close and open the clipboard to ensure auto-conversions take place.
66       finally:
67           CloseClipboard()
68   
69       OpenClipboard()
70       try:
71   
72           # Make sure I can still get the text as bytes
73           got = GetClipboardData(win32con.CF_TEXT)
74           assert got == text_bytes, "Didnt get the correct result back - '%r'." % (got,)
75           # Make sure we get back the correct types.
76           got = GetClipboardData(win32con.CF_UNICODETEXT)
77           assert type(got) == str, "Didnt get the correct result back - '%r'." % (got,)
78           got = GetClipboardData(win32con.CF_OEMTEXT)
79           assert got == text_bytes, "Didnt get the correct result back - '%r'." % (got,)
80           print("Clipboard text tests worked correctly")
81       finally:
82           CloseClipboard()
