probesysfs.py 
1    ''' 
2    Auto Create Input Provider Config Entry for Available MT Hardware (linux only). 
3    =============================================================================== 
4     
5    Thanks to Marc Tardif for the probing code, taken from scan-for-mt-device. 
6     
7    The device discovery is done by this provider. However, the reading of 
8    input can be performed by other providers like: hidinput, mtdev and 
9    linuxwacom. mtdev is used prior to other providers. For more 
10   information about mtdev, check :py:class:`~kivy.input.providers.mtdev`. 
11    
12   Here is an example of auto creation:: 
13    
14       [input] 
15       # using mtdev 
16       device_%(name)s = probesysfs,provider=mtdev 
17       # using hidinput 
18       device_%(name)s = probesysfs,provider=hidinput 
19       # using mtdev with a match on name 
20       device_%(name)s = probesysfs,provider=mtdev,match=acer 
21    
22       # using hidinput with custom parameters to hidinput (all on one line) 
23       %(name)s = probesysfs, 
24           provider=hidinput,param=min_pressure=1,param=max_pressure=99 
25    
26       # you can also match your wacom touchscreen 
27       touch = probesysfs,match=E3 Finger,provider=linuxwacom, 
28           select_all=1,param=mode=touch 
29       # and your wacom pen 
30       pen = probesysfs,match=E3 Pen,provider=linuxwacom, 
31           select_all=1,param=mode=pen 
32    
33   By default, ProbeSysfs module will enumerate hardware from the /sys/class/input 
34   device, and configure hardware with ABS_MT_POSITION_X capability. But for 
35   example, the wacom screen doesn't support this capability. You can prevent this 
36   behavior by putting select_all=1 in your config line. Add use_mouse=1 to also 
37   include touchscreen hardware that offers core pointer functionality. 
38   '''
39   
40   __all__ = ('ProbeSysfsHardwareProbe', )

42   import os
43   from os.path import sep
44   
45   if 'KIVY_DOC' in os.environ:
46   
47       ProbeSysfsHardwareProbe = None
48   
49   else:
50       import ctypes
51       from re import match, IGNORECASE
52       from glob import glob
53       from subprocess import Popen, PIPE
54       from kivy.logger import Logger
55       from kivy.input.provider import MotionEventProvider
56       from kivy.input.providers.mouse import MouseMotionEventProvider
57       from kivy.input.factory import MotionEventFactory
58       from kivy.config import _is_rpi
59   
60       EventLoop = None
61   
62       # See linux/input.h
63       ABS_MT_POSITION_X = 0x35
64   
65       _cache_input = None
66       _cache_xinput = None
67   
68       class Input(object):
69   
70           def __init__(self, path):
71               query_xinput()
72               self.path = path
73   
74           @property
75           def device(self):
76               base = os.path.basename(self.path)
77               return os.path.join("/dev", "input", base)
78   
79           @property
80           def name(self):
81               path = os.path.join(self.path, "device", "name")
82               return read_line(path)
83   
84           def get_capabilities(self):
85               path = os.path.join(self.path, "device", "capabilities", "abs")
86               line = "0"
87               try:
88                   line = read_line(path)
89               except (IOError, OSError):
90                   return []
91   
92               capabilities = []
93               long_bit = ctypes.sizeof(ctypes.c_long) * 8
94               for i, word in enumerate(line.split(" ")):
95                   word = int(word, 16)
96                   subcapabilities = [bool(word & 1 << i)
97                                      for i in range(long_bit)]
98                   capabilities[:0] = subcapabilities
99   
100              return capabilities
101  
102          def has_capability(self, capability):
103              capabilities = self.get_capabilities()
104              return len(capabilities) > capability and capabilities[capability]
105  
106          @property
107          def is_mouse(self):
108              return self.device in _cache_xinput
109  
110      def getout(*args):
111          try:
112              return Popen(args, stdout=PIPE).communicate()[0]
113          except OSError:
114              return ''
115  
116      def query_xinput():
117          global _cache_xinput
118          if _cache_xinput is None:
119              _cache_xinput = []
120              devids = getout('xinput', '--list', '--id-only')
121              for did in devids.splitlines():
122                  devprops = getout('xinput', '--list-props', did)
123                  evpath = None
124                  for prop in devprops.splitlines():
125                      prop = prop.strip()
126                      if (prop.startswith(b'Device Enabled') and
127                              prop.endswith(b'0')):
128                          evpath = None
129                          break
130                      if prop.startswith(b'Device Node'):
131                          try:
132                              evpath = prop.split('"')[1]
133                          except Exception:
134                              evpath = None
135                  if evpath:
136                      _cache_xinput.append(evpath)
137  
138      def get_inputs(path):
139          global _cache_input
140          if _cache_input is None:
141              event_glob = os.path.join(path, "event*")
142              _cache_input = [Input(x) for x in glob(event_glob)]
143          return _cache_input
144  
145      def read_line(path):
146          f = open(path)
147          try:
148              return f.readline().strip()
149          finally:
150              f.close()
151  
152      class ProbeSysfsHardwareProbe(MotionEventProvider):
153  
154          def __new__(self, device, args):
155              # hack to not return an instance of this provider.
156              # :)
157              instance = super(ProbeSysfsHardwareProbe, self).__new__(self)
158              instance.__init__(device, args)
159  
160          def __init__(self, device, args):
161              super(ProbeSysfsHardwareProbe, self).__init__(device, args)
162              self.provider = 'mtdev'
163              self.match = None
164              self.input_path = '/sys/class/input'
165              self.select_all = True if _is_rpi else False
166              self.use_mouse = False
167              self.use_regex = False
168              self.args = []
169  
170              args = args.split(',')
171              for arg in args:
172                  if arg == '':
173                      continue
174                  arg = arg.split('=', 1)
175                  # ensure it's a key = value
176                  if len(arg) != 2:
177                      Logger.error('ProbeSysfs: invalid parameters %s, not'
178                                   ' key=value format' % arg)
179                      continue
180  
181                  key, value = arg
182                  if key == 'match':
183                      self.match = value
184                  elif key == 'provider':
185                      self.provider = value
186                  elif key == 'use_regex':
187                      self.use_regex = bool(int(value))
188                  elif key == 'select_all':
189                      self.select_all = bool(int(value))
190                  elif key == 'use_mouse':
191                      self.use_mouse = bool(int(value))
192                  elif key == 'param':
193                      self.args.append(value)
194                  else:
195                      Logger.error('ProbeSysfs: unknown %s option' % key)
196                      continue
197  
198              self.probe()
199  
200          def should_use_mouse(self):
201              return (self.use_mouse or
202                      not any(p for p in EventLoop.input_providers
203                              if isinstance(p, MouseMotionEventProvider)))
204  
205          def probe(self):
206              global EventLoop
207              from kivy.base import EventLoop
208  
209              inputs = get_inputs(self.input_path)
210              Logger.debug('ProbeSysfs: using probesysfs!')
211  
212              use_mouse = self.should_use_mouse()
213  
214              if not self.select_all:
215                  inputs = [x for x in inputs if
216                            x.has_capability(ABS_MT_POSITION_X) and
217                            (use_mouse or not x.is_mouse)]
218              for device in inputs:
219                  Logger.debug('ProbeSysfs: found device: %s at %s' % (
220                      device.name, device.device))
221  
222                  # must ignore ?
223                  if self.match:
224                      if self.use_regex:
225                          if not match(self.match, device.name, IGNORECASE):
226                              Logger.debug('ProbeSysfs: device not match the'
227                                           ' rule in config, ignoring.')
228                              continue
229                      else:
230                          if self.match not in device.name:
231                              continue
232  
233                  Logger.info('ProbeSysfs: device match: %s' % device.device)
234  
235                  d = device.device
236                  devicename = self.device % dict(name=d.split(sep)[-1])
237  
238                  provider = MotionEventFactory.get(self.provider)
239                  if provider is None:
240                      Logger.info('ProbeSysfs: unable to found provider %s' %
241                                  self.provider)
242                      Logger.info('ProbeSysfs: fallback on hidinput')
243                      provider = MotionEventFactory.get('hidinput')
244                  if provider is None:
245                      Logger.critical('ProbeSysfs: no input provider found'
246                                      ' to handle this device !')
247                      continue
248  
249                  instance = provider(devicename, '%s,%s' % (
250                      device.device, ','.join(self.args)))
251                  if instance:
252                      EventLoop.add_input_provider(instance)
253  
254      MotionEventFactory.register('probesysfs', ProbeSysfsHardwareProbe)
