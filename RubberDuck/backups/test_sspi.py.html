9    # It is quite likely that the Kerberos tests will fail due to not being

12   def applyHandlingSkips(func, *args):
13       try:
14           return func(*args)
15       except win32api.error as exc:
16           if exc.winerror == sspicon.SEC_E_NO_CREDENTIALS:
17               raise TestSkipped(exc)
18           raise

21   class TestSSPI(unittest.TestCase):
22       def assertRaisesHRESULT(self, hr, func, *args):
23           try:
24               return func(*args)
25               raise RuntimeError("expecting %s failure" % (hr,))
26           except win32security.error as exc:
27               self.failUnlessEqual(exc.winerror, hr)

29       def _doAuth(self, pkg_name):
30           sspiclient = sspi.ClientAuth(pkg_name, targetspn=win32api.GetUserName())
31           sspiserver = sspi.ServerAuth(pkg_name)
32   
33           sec_buffer = None
34           err = 1
35           while err != 0:
36               err, sec_buffer = sspiclient.authorize(sec_buffer)
37               err, sec_buffer = sspiserver.authorize(sec_buffer)
38           return sspiclient, sspiserver

44           sspiserver.ctxt.RevertSecurityContext()

52       def _doTestEncrypt(self, pkg_name):
53   
54           sspiclient, sspiserver = self._doAuth(pkg_name)
55   
56           pkg_size_info = sspiclient.ctxt.QueryContextAttributes(
57               sspicon.SECPKG_ATTR_SIZES
58           )
59           msg = str2bytes("some data to be encrypted ......")
60   
61           trailersize = pkg_size_info["SecurityTrailer"]
62           encbuf = win32security.PySecBufferDescType()
63           encbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
64           encbuf.append(
65               win32security.PySecBufferType(trailersize, sspicon.SECBUFFER_TOKEN)
66           )
67           encbuf[0].Buffer = msg
68           sspiclient.ctxt.EncryptMessage(0, encbuf, 1)
69           sspiserver.ctxt.DecryptMessage(encbuf, 1)
70           self.failUnlessEqual(msg, encbuf[0].Buffer)
71           # and test the higher-level functions
72           data_in = str2bytes("hello")
73           data, sig = sspiclient.encrypt(data_in)
74           self.assertEqual(sspiserver.decrypt(data, sig), data_in)
75   
76           data, sig = sspiserver.encrypt(data_in)
77           self.assertEqual(sspiclient.decrypt(data, sig), data_in)

79       def _doTestEncryptStream(self, pkg_name):
80           # Test out the SSPI/GSSAPI interop wrapping examples at
81           # https://docs.microsoft.com/en-us/windows/win32/secauthn/sspi-kerberos-interoperability-with-gssapi
82   
83           sspiclient, sspiserver = self._doAuth(pkg_name)
84   
85           pkg_size_info = sspiclient.ctxt.QueryContextAttributes(
86               sspicon.SECPKG_ATTR_SIZES
87           )
88           msg = str2bytes("some data to be encrypted ......")
89   
90           trailersize = pkg_size_info["SecurityTrailer"]
91           blocksize = pkg_size_info["BlockSize"]
92           encbuf = win32security.PySecBufferDescType()
93           encbuf.append(
94               win32security.PySecBufferType(trailersize, sspicon.SECBUFFER_TOKEN)
95           )
96           encbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
97           encbuf.append(
98               win32security.PySecBufferType(blocksize, sspicon.SECBUFFER_PADDING)
99           )
100          encbuf[1].Buffer = msg
101          sspiclient.ctxt.EncryptMessage(0, encbuf, 1)
102  
103          encmsg = encbuf[0].Buffer + encbuf[1].Buffer + encbuf[2].Buffer
104          decbuf = win32security.PySecBufferDescType()
105          decbuf.append(
106              win32security.PySecBufferType(len(encmsg), sspicon.SECBUFFER_STREAM)
107          )
108          decbuf.append(win32security.PySecBufferType(0, sspicon.SECBUFFER_DATA))
109          decbuf[0].Buffer = encmsg
110  
111          sspiserver.ctxt.DecryptMessage(decbuf, 1)
112          self.failUnlessEqual(msg, decbuf[1].Buffer)

126      def _doTestSign(self, pkg_name):
127  
128          sspiclient, sspiserver = self._doAuth(pkg_name)
129  
130          pkg_size_info = sspiclient.ctxt.QueryContextAttributes(
131              sspicon.SECPKG_ATTR_SIZES
132          )
133          msg = str2bytes("some data to be encrypted ......")
134  
135          sigsize = pkg_size_info["MaxSignature"]
136          sigbuf = win32security.PySecBufferDescType()
137          sigbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
138          sigbuf.append(win32security.PySecBufferType(sigsize, sspicon.SECBUFFER_TOKEN))
139          sigbuf[0].Buffer = msg
140          sspiclient.ctxt.MakeSignature(0, sigbuf, 0)
141          sspiserver.ctxt.VerifySignature(sigbuf, 0)
142          # and test the higher-level functions
143          sspiclient.next_seq_num = 1
144          sspiserver.next_seq_num = 1
145          data = str2bytes("hello")
146          key = sspiclient.sign(data)
147          sspiserver.verify(data, key)
148          key = sspiclient.sign(data)
149          self.assertRaisesHRESULT(
150              sspicon.SEC_E_MESSAGE_ALTERED, sspiserver.verify, data + data, key
151          )
152  
153          # and the other way
154          key = sspiserver.sign(data)
155          sspiclient.verify(data, key)
156          key = sspiserver.sign(data)
157          self.assertRaisesHRESULT(
158              sspicon.SEC_E_MESSAGE_ALTERED, sspiclient.verify, data + data, key
159          )

167      def _testSequenceSign(self):
168          # Only Kerberos supports sequence detection.
169          sspiclient, sspiserver = self._doAuth("Kerberos")
170          key = sspiclient.sign("hello")
171          sspiclient.sign("hello")
172          self.assertRaisesHRESULT(
173              sspicon.SEC_E_OUT_OF_SEQUENCE, sspiserver.verify, "hello", key

193      def testSecBufferRepr(self):
194          desc = win32security.PySecBufferDescType()
195          assert re.match(
196              "PySecBufferDesc\(ulVersion: 0 \| cBuffers: 0 \| pBuffers: 0x[\da-fA-F]{8,16}\)",
197              repr(desc),
198          )
199  
200          buffer1 = win32security.PySecBufferType(0, sspicon.SECBUFFER_TOKEN)
201          assert re.match(
202              "PySecBuffer\(cbBuffer: 0 \| BufferType: 2 \| pvBuffer: 0x[\da-fA-F]{8,16}\)",
203              repr(buffer1),
204          )
205          "PySecBuffer(cbBuffer: 0 | BufferType: 2 | pvBuffer: 0x000001B8CC6D8020)"
206          desc.append(buffer1)
207  
208          assert re.match(
209              "PySecBufferDesc\(ulVersion: 0 \| cBuffers: 1 \| pBuffers: 0x[\da-fA-F]{8,16}\)",
210              repr(desc),
211          )
212  
213          buffer2 = win32security.PySecBufferType(4, sspicon.SECBUFFER_DATA)
214          assert re.match(
215              "PySecBuffer\(cbBuffer: 4 \| BufferType: 1 \| pvBuffer: 0x[\da-fA-F]{8,16}\)",
216              repr(buffer2),
217          )
218          desc.append(buffer2)
219  
220          assert re.match(
221              "PySecBufferDesc\(ulVersion: 0 \| cBuffers: 2 \| pBuffers: 0x[\da-fA-F]{8,16}\)",
222              repr(desc),
223          )
