1    # Creates a task-bar icon.  Run from Python.exe to see the
2    # messages printed.
3    import win32api, win32gui
4    import win32con, winerror
5    import sys, os
6    
7    
8    class MainWindow:
9        def __init__(self):
10           msg_TaskbarRestart = win32gui.RegisterWindowMessage("TaskbarCreated")
11           message_map = {
12               msg_TaskbarRestart: self.OnRestart,
13               win32con.WM_DESTROY: self.OnDestroy,
14               win32con.WM_COMMAND: self.OnCommand,
15               win32con.WM_USER + 20: self.OnTaskbarNotify,
16           }
17           # Register the Window class.
18           wc = win32gui.WNDCLASS()
19           hinst = wc.hInstance = win32api.GetModuleHandle(None)
20           wc.lpszClassName = "PythonTaskbarDemo"
21           wc.style = win32con.CS_VREDRAW | win32con.CS_HREDRAW
22           wc.hCursor = win32api.LoadCursor(0, win32con.IDC_ARROW)
23           wc.hbrBackground = win32con.COLOR_WINDOW
24           wc.lpfnWndProc = message_map  # could also specify a wndproc.
25   
26           # Don't blow up if class already registered to make testing easier
27           try:
28               classAtom = win32gui.RegisterClass(wc)
29           except win32gui.error as err_info:
30               if err_info.winerror != winerror.ERROR_CLASS_ALREADY_EXISTS:
31                   raise
32   
33           # Create the Window.
34           style = win32con.WS_OVERLAPPED | win32con.WS_SYSMENU
35           self.hwnd = win32gui.CreateWindow(
36               wc.lpszClassName,
37               "Taskbar Demo",
38               style,
39               0,
40               0,
41               win32con.CW_USEDEFAULT,
42               win32con.CW_USEDEFAULT,
43               0,
44               0,
45               hinst,
46               None,
47           )
48           win32gui.UpdateWindow(self.hwnd)
49           self._DoCreateIcons()
50   
51       def _DoCreateIcons(self):
52           # Try and find a custom icon
53           hinst = win32api.GetModuleHandle(None)
54           iconPathName = os.path.abspath(
55               os.path.join(os.path.split(sys.executable)[0], "pyc.ico")
56           )
57           if not os.path.isfile(iconPathName):
58               # Look in DLLs dir, a-la py 2.5
59               iconPathName = os.path.abspath(
60                   os.path.join(os.path.split(sys.executable)[0], "DLLs", "pyc.ico")
61               )
62           if not os.path.isfile(iconPathName):
63               # Look in the source tree.
64               iconPathName = os.path.abspath(
65                   os.path.join(os.path.split(sys.executable)[0], "..\\PC\\pyc.ico")
66               )
67           if os.path.isfile(iconPathName):
68               icon_flags = win32con.LR_LOADFROMFILE | win32con.LR_DEFAULTSIZE
69               hicon = win32gui.LoadImage(
70                   hinst, iconPathName, win32con.IMAGE_ICON, 0, 0, icon_flags
71               )
72           else:
73               print("Can't find a Python icon file - using default")
74               hicon = win32gui.LoadIcon(0, win32con.IDI_APPLICATION)
75   
76           flags = win32gui.NIF_ICON | win32gui.NIF_MESSAGE | win32gui.NIF_TIP
77           nid = (self.hwnd, 0, flags, win32con.WM_USER + 20, hicon, "Python Demo")
78           try:
79               win32gui.Shell_NotifyIcon(win32gui.NIM_ADD, nid)
80           except win32gui.error:
81               # This is common when windows is starting, and this code is hit
82               # before the taskbar has been created.
83               print("Failed to add the taskbar icon - is explorer running?")
84               # but keep running anyway - when explorer starts, we get the
85               # TaskbarCreated message.
86   
87       def OnRestart(self, hwnd, msg, wparam, lparam):
88           self._DoCreateIcons()
89   
90       def OnDestroy(self, hwnd, msg, wparam, lparam):
91           nid = (self.hwnd, 0)
92           win32gui.Shell_NotifyIcon(win32gui.NIM_DELETE, nid)
93           win32gui.PostQuitMessage(0)  # Terminate the app.
94   
95       def OnTaskbarNotify(self, hwnd, msg, wparam, lparam):
96           if lparam == win32con.WM_LBUTTONUP:
97               print("You clicked me.")
98           elif lparam == win32con.WM_LBUTTONDBLCLK:
99               print("You double-clicked me - goodbye")
100              win32gui.DestroyWindow(self.hwnd)
101          elif lparam == win32con.WM_RBUTTONUP:
102              print("You right clicked me.")
103              menu = win32gui.CreatePopupMenu()
104              win32gui.AppendMenu(menu, win32con.MF_STRING, 1023, "Display Dialog")
105              win32gui.AppendMenu(menu, win32con.MF_STRING, 1024, "Say Hello")
106              win32gui.AppendMenu(menu, win32con.MF_STRING, 1025, "Exit program")
107              pos = win32gui.GetCursorPos()
108              # See http://msdn.microsoft.com/library/default.asp?url=/library/en-us/winui/menus_0hdi.asp
109              win32gui.SetForegroundWindow(self.hwnd)
110              win32gui.TrackPopupMenu(
111                  menu, win32con.TPM_LEFTALIGN, pos[0], pos[1], 0, self.hwnd, None
112              )
113              win32gui.PostMessage(self.hwnd, win32con.WM_NULL, 0, 0)
114          return 1
115  
116      def OnCommand(self, hwnd, msg, wparam, lparam):
117          id = win32api.LOWORD(wparam)
118          if id == 1023:
119              import win32gui_dialog
120  
121              win32gui_dialog.DemoModal()
122          elif id == 1024:
123              print("Hello")
124          elif id == 1025:
125              print("Goodbye")
126              win32gui.DestroyWindow(self.hwnd)
127          else:
128              print("Unknown command -", id)

132      w = MainWindow()
133      win32gui.PumpMessages()
