13   def circumcircle(a, b, c):
14       ''' 
15       Computes the circumcircle of a triangle defined by a, b, c. 
16       See: http://en.wikipedia.org/wiki/Circumscribed_circle 
17    
18       :Parameters: 
19           `a`: iterable containing at least 2 values (for x and y) 
20               The 1st point of the triangle. 
21           `b`: iterable containing at least 2 values (for x and y) 
22               The 2nd point of the triangle. 
23           `c`: iterable containing at least 2 values (for x and y) 
24               The 3rd point of the triangle. 
25    
26       :Return: 
27           A tuple that defines the circle : 
28            * The first element in the returned tuple is the center as (x, y) 
29            * The second is the radius (float) 
30       '''
31       P = Vector(a[0], a[1])
32       Q = Vector(b[0], b[1])
33       R = Vector(c[0], c[1])
34   
35       mPQ = (P + Q) * .5
36       mQR = (Q + R) * .5
37   
38       numer = -(- mPQ.y * R.y + mPQ.y * Q.y + mQR.y * R.y - mQR.y * Q.y -
39                 mPQ.x * R.x + mPQ.x * Q.x + mQR.x * R.x - mQR.x * Q.x)
40       denom = (-Q.x * R.y + P.x * R.y - P.x * Q.y +
41                Q.y * R.x - P.y * R.x + P.y * Q.x)
42   
43       t = numer / denom
44   
45       cx = -t * (Q.y - P.y) + mPQ.x
46       cy = t * (Q.x - P.x) + mPQ.y
47   
48       return ((cx, cy), (P - (cx, cy)).length())

51   def minimum_bounding_circle(points):
52       ''' 
53       Returns the minimum bounding circle for a set of points. 
54    
55       For a description of the problem being solved, see the `Smallest Circle 
56       Problem <http://en.wikipedia.org/wiki/Smallest_circle_problem>`_. 
57    
58       The function uses Applet's Algorithm, the runtime is ``O(h^3, *n)``, 
59       where h is the number of points in the convex hull of the set of points. 
60       **But** it runs in linear time in almost all real world cases. 
61       See: http://tinyurl.com/6e4n5yb 
62    
63       :Parameters: 
64           `points`: iterable 
65               A list of points (2 tuple with x,y coordinates) 
66    
67       :Return: 
68           A tuple that defines the circle: 
69               * The first element in the returned tuple is the center (x, y) 
70               * The second the radius (float) 
71    
72       '''
73       points = [Vector(p[0], p[1]) for p in points]
74   
75       if len(points) == 1:
76           return (points[0].x, points[0].y), 0.0
77   
78       if len(points) == 2:
79           p1, p2 = points
80           return (p1 + p2) * .5, ((p1 - p2) * .5).length()
81   
82       # determine a point P with the smallest y value
83       P = min(points, key=lambda p: p.y)
84   
85       # find a point Q such that the angle of the line segment
86       # PQ with the x axis is minimal
87       def x_axis_angle(q):
88           if q == P:
89               return 1e10  # max val if the same, to skip
90           return abs((q - P).angle((1, 0)))
91       Q = min(points, key=x_axis_angle)
92   
93       for p in points:
94           # find R such that angle PRQ is minimal
95           def angle_pq(r):
96               if r in (P, Q):
97                   return 1e10  # max val if the same, to skip
98               return abs((r - P).angle(r - Q))
99           R = min(points, key=angle_pq)
100  
101          # check for case 1 (angle PRQ is obtuse), the circle is determined
102          # by two points, P and Q. radius = |(P-Q)/2|, center = (P+Q)/2
103          if angle_pq(R) > 90.0:
104              return (P + Q) * .5, ((P - Q) * .5).length()
105  
106          # if angle RPQ is obtuse, make P = R, and try again
107          if abs((R - P).angle(Q - P)) > 90:
108              P = R
109              continue
110  
111          # if angle PQR is obtuse, make Q = R, and try again
112          if abs((P - Q).angle(R - Q)) > 90:
113              Q = R
114              continue
115  
116          # all angles were acute..we just need the circle through the
117          # two points furthest apart!
118          break
119  
120      # find the circumcenter for triangle given by P,Q,R
121      return circumcircle(P, Q, R)
