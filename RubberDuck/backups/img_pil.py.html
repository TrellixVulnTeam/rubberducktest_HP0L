23       # https://github.com/kivy/kivy/issues/5460
24       PILImage.frombytes = PILImage.frombuffer
25       PILImage.Image.tobytes = PILImage.Image.tostring

28   class ImageLoaderPIL(ImageLoaderBase):
29       '''Image loader based on the PIL library. 
30    
31       .. versionadded:: 1.0.8 
32    
33       Support for GIF animation added. 
34    
35       Gif animation has a lot of issues(transparency/color depths... etc). 
36       In order to keep it simple, what is implemented here is what is 
37       natively supported by the PIL library. 
38    
39       As a general rule, try to use gifs that have no transparency. 
40       Gif's with transparency will work but be prepared for some 
41       artifacts until transparency support is improved. 
42    
43       '''
44   
45       @staticmethod
46       def can_save(fmt, is_bytesio):
47           if is_bytesio:
48               return False
49           return fmt in ImageLoaderPIL.extensions()
50   
51       @staticmethod
52       def can_load_memory():
53           return True
54   
55       @staticmethod
56       def extensions():
57           '''Return accepted extensions for this loader'''
58           PILImage.init()
59           return tuple((ext_with_dot[1:] for ext_with_dot in PILImage.EXTENSION))
60   
61       def _img_correct(self, _img_tmp):
62           '''Convert image to the correct format and orientation. 
63           '''
64           # image loader work only with rgb/rgba image
65           if _img_tmp.mode.lower() not in ('rgb', 'rgba'):
66               try:
67                   imc = _img_tmp.convert('RGBA')
68               except:
69                   Logger.warning(
70                       'Image: Unable to convert image to rgba (was %s)' %
71                       (_img_tmp.mode.lower()))
72                   raise
73               _img_tmp = imc
74   
75           return _img_tmp
76   
77       def _img_read(self, im):
78           '''Read images from an animated file. 
79           '''
80           im.seek(0)
81   
82           # Read all images inside
83           try:
84               img_ol = None
85               while True:
86                   img_tmp = im
87                   img_tmp = self._img_correct(img_tmp)
88                   if img_ol and (hasattr(im, 'dispose') and not im.dispose):
89                       # paste new frame over old so as to handle
90                       # transparency properly
91                       img_ol.paste(img_tmp, (0, 0), img_tmp)
92                       img_tmp = img_ol
93                   img_ol = img_tmp
94                   yield ImageData(img_tmp.size[0], img_tmp.size[1],
95                                   img_tmp.mode.lower(), img_tmp.tobytes())
96                   im.seek(im.tell() + 1)
97           except EOFError:
98               pass
99   
100      def load(self, filename):
101          try:
102              im = PILImage.open(filename)
103          except:
104              Logger.warning('Image: Unable to load image <%s>' % filename)
105              raise
106          # update internals
107          if not self._inline:
108              self.filename = filename
109          # returns an array of type ImageData len 1 if not a sequence image
110          return list(self._img_read(im))
111  
112      @staticmethod
113      def save(filename, width, height, pixelfmt, pixels, flipped=False,
114               imagefmt=None):
115          image = PILImage.frombytes(pixelfmt.upper(), (width, height), pixels)
116          if flipped:
117              image = image.transpose(PILImage.FLIP_TOP_BOTTOM)
118          image.save(filename)
119          return True
