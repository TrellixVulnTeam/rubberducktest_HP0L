1    ''' 
2    Switch 
3    ====== 
4     
5    .. versionadded:: 1.0.7 
6     
7    .. image:: images/switch-on.jpg 
8        :align: right 
9     
10   .. image:: images/switch-off.jpg 
11       :align: right 
12    
13   The :class:`Switch` widget is active or inactive, like a mechanical light 
14   switch. The user can swipe to the left/right to activate/deactivate it:: 
15    
16       switch = Switch(active=True) 
17    
18   To attach a callback that listens to the activation state:: 
19    
20       def callback(instance, value): 
21           print('the switch', instance, 'is', value) 
22    
23       switch = Switch() 
24       switch.bind(active=callback) 
25    
26   By default, the representation of the widget is static. The minimum size 
27   required is 83x32 pixels (defined by the background image). The image is 
28   centered within the widget. 
29    
30   The entire widget is active, not just the part with graphics. As long as you 
31   swipe over the widget's bounding box, it will work. 
32    
33   .. note:: 
34    
35       If you want to control the state with a single touch instead of a swipe, 
36       use the :class:`ToggleButton` instead. 
37    
38   Kv Example:: 
39    
40       BoxLayout: 
41           Label: 
42               text: 'power up' 
43           Switch: 
44               id: switch 
45           Label: 
46               text: 'woooooooooooh' if switch.active else '' 
47   '''
48   
49   
50   from kivy.uix.widget import Widget
51   from kivy.animation import Animation
52   from kivy.properties import BooleanProperty, ObjectProperty, NumericProperty
53   
54   
55   class Switch(Widget):
56       '''Switch class. See module documentation for more information. 
57       '''
58       active = BooleanProperty(False)
59       '''Indicate whether the switch is active or inactive. 
60    
61       :attr:`active` is a :class:`~kivy.properties.BooleanProperty` and defaults 
62       to False. 
63       '''
64   
65       touch_control = ObjectProperty(None, allownone=True)
66       '''(internal) Contains the touch that currently interacts with the switch. 
67    
68       :attr:`touch_control` is an :class:`~kivy.properties.ObjectProperty` 
69       and defaults to None. 
70       '''
71   
72       touch_distance = NumericProperty(0)
73       '''(internal) Contains the distance between the initial position of the 
74       touch and the current position to determine if the swipe is from the left 
75       or right. 
76    
77       :attr:`touch_distance` is a :class:`~kivy.properties.NumericProperty` 
78       and defaults to 0. 
79       '''
80   
81       active_norm_pos = NumericProperty(0)
82       '''(internal) Contains the normalized position of the movable element 
83       inside the switch, in the 0-1 range. 
84    
85       :attr:`active_norm_pos` is a :class:`~kivy.properties.NumericProperty` 
86       and defaults to 0. 
87       '''
88   
89       def on_touch_down(self, touch):
90           if self.disabled or self.touch_control is not None:
91               return
92           if not self.collide_point(*touch.pos):
93               return
94           touch.grab(self)
95           self.touch_distance = 0
96           self.touch_control = touch
97           return True
98   
99       def on_touch_move(self, touch):
100          if touch.grab_current is not self:
101              return
102          self.touch_distance = touch.x - touch.ox
103          return True
104  
105      def on_touch_up(self, touch):
106          if touch.grab_current is not self:
107              return
108          touch.ungrab(self)
109          # depending of the distance, activate by norm pos or invert
110          if abs(touch.ox - touch.x) < 5:
111              self.active = not self.active
112          else:
113              self.active = self.active_norm_pos > 0.5
114          Animation(active_norm_pos=int(self.active), t='out_quad',
115                    d=.2).start(self)
116          self.touch_control = None
117          return True
118  
119  
120  if __name__ == '__main__':
121      from kivy.base import runTouchApp
122      runTouchApp(Switch())
