29   class Style:
30       """Represents a single format"""
31   
32       def __init__(self, name, format, background=CLR_INVALID):
33           self.name = name  # Name the format representes eg, "String", "Class"
34           # Default background for each style is only used when there are no
35           # saved settings (generally on first startup)
36           self.background = self.default_background = background
37           if type(format) == type(""):
38               self.aliased = format
39               self.format = None
40           else:
41               self.format = format
42               self.aliased = None
43           self.stylenum = None  # Not yet registered.
44   
45       def IsBasedOnDefault(self):
46           return len(self.format) == 5
47   
48       # If the currently extended font defintion matches the
49       # default format, restore the format to the "simple" format.
50       def NormalizeAgainstDefault(self, defaultFormat):
51           if self.IsBasedOnDefault():
52               return 0  # No more to do, and not changed.
53           bIsDefault = (
54               self.format[7] == defaultFormat[7] and self.format[2] == defaultFormat[2]
55           )
56           if bIsDefault:
57               self.ForceAgainstDefault()
58           return bIsDefault
59   
60       def ForceAgainstDefault(self):
61           self.format = self.format[:5]
62   
63       def GetCompleteFormat(self, defaultFormat):
64           # Get the complete style after applying any relevant defaults.
65           if len(self.format) == 5:  # It is a default one
66               fmt = self.format + defaultFormat[5:]
67           else:
68               fmt = self.format
69           flags = (
70               win32con.CFM_BOLD
71               | win32con.CFM_CHARSET
72               | win32con.CFM_COLOR
73               | win32con.CFM_FACE
74               | win32con.CFM_ITALIC
75               | win32con.CFM_SIZE
76           )
77           return (flags,) + fmt[1:]

95       # Used by the IDLE extensions to quickly determine if a character is a string.
96       def GetStringStyle(self, pos):
97           try:
98               style = self.styles_by_id[self.scintilla.SCIGetStyleAt(pos)]
99           except KeyError:
100              # A style we dont know about - probably not even a .py file - can't be a string

104          return None

112          self.styles_by_id[stylenum] = style

125      # Update the control with the new style format.
126      def _ReformatStyle(self, style):
127          ## Selection (background only for now)
128          ## Passing False for WPARAM to SCI_SETSELBACK is documented as resetting to scintilla default,
129          ## but does not work - selection background is not visible at all.
130          ## Default value in SPECIAL_STYLES taken from scintilla source.
131          if style.name == STYLE_SELECTION:
132              clr = style.background
133              self.scintilla.SendScintilla(scintillacon.SCI_SETSELBACK, True, clr)
134  
135              ## Can't change font for selection, but could set color
136              ## However, the font color dropbox has no option for default, and thus would
137              ## always override syntax coloring
138              ## clr = style.format[4]
139              ## self.scintilla.SendScintilla(scintillacon.SCI_SETSELFORE, clr != CLR_INVALID, clr)
140              return
141  
142          assert style.stylenum is not None, "Unregistered style."
143          # print "Reformat style", style.name, style.stylenum
144          scintilla = self.scintilla
145          stylenum = style.stylenum
146          # Now we have the style number, indirect for the actual style.
147          if style.aliased is not None:
148              style = self.styles[style.aliased]
149          f = style.format
150          if style.IsBasedOnDefault():
151              baseFormat = self.GetDefaultFormat()
152          else:
153              baseFormat = f
154          scintilla.SCIStyleSetFore(stylenum, f[4])
155          scintilla.SCIStyleSetFont(stylenum, baseFormat[7], baseFormat[5])
156          if f[1] & 1:
157              scintilla.SCIStyleSetBold(stylenum, 1)
158          else:
159              scintilla.SCIStyleSetBold(stylenum, 0)
160          if f[1] & 2:
161              scintilla.SCIStyleSetItalic(stylenum, 1)
162          else:
163              scintilla.SCIStyleSetItalic(stylenum, 0)
164          scintilla.SCIStyleSetSize(stylenum, int(baseFormat[2] / 20))
165          scintilla.SCIStyleSetEOLFilled(stylenum, 1)  # Only needed for unclosed strings.
166  
167          ## Default style background to whitespace background if set,
168          ##  otherwise use system window color
169          bg = style.background
170          if bg == CLR_INVALID:
171              bg = self.styles[STYLE_DEFAULT].background
172          if bg == CLR_INVALID:
173              bg = win32api.GetSysColor(win32con.COLOR_WINDOW)
174          scintilla.SCIStyleSetBack(stylenum, bg)

176      def GetStyleByNum(self, stylenum):
177          return self.styles_by_id[stylenum]

179      def ApplyFormattingStyles(self, bReload=1):
180          if bReload:
181              self.LoadPreferences()
182          baseFormat = self.GetDefaultFormat()
183          defaultStyle = Style("default", baseFormat)
184          defaultStyle.stylenum = scintillacon.STYLE_DEFAULT
185          self._ReformatStyle(defaultStyle)
186          for style in list(self.styles.values()):
187              if style.aliased is None:
188                  style.NormalizeAgainstDefault(baseFormat)
189              self._ReformatStyle(style)
190          self.scintilla.InvalidateRect()

192      # Some functions for loading and saving preferences.  By default
193      # an INI file (well, MFC maps this to the registry) is used.
194      def LoadPreferences(self):
195          self.baseFormatFixed = eval(
196              self.LoadPreference("Base Format Fixed", str(self.baseFormatFixed))
197          )
198          self.baseFormatProp = eval(
199              self.LoadPreference("Base Format Proportional", str(self.baseFormatProp))
200          )
201          self.bUseFixed = int(self.LoadPreference("Use Fixed", 1))
202  
203          for style in list(self.styles.values()):
204              new = self.LoadPreference(style.name, str(style.format))
205              try:
206                  style.format = eval(new)
207              except:
208                  print("Error loading style data for", style.name)
209              # Use "vanilla" background hardcoded in PYTHON_STYLES if no settings in registry
210              style.background = int(
211                  self.LoadPreference(
212                      style.name + " background", style.default_background
213                  )
214              )
215  
216      def LoadPreference(self, name, default):
217          return win32ui.GetProfileVal("Format", name, default)
218  
219      def SavePreferences(self):
220          self.SavePreference("Base Format Fixed", str(self.baseFormatFixed))
221          self.SavePreference("Base Format Proportional", str(self.baseFormatProp))
222          self.SavePreference("Use Fixed", self.bUseFixed)
223          for style in list(self.styles.values()):
224              if style.aliased is None:
225                  self.SavePreference(style.name, str(style.format))
226                  bg_name = style.name + " background"
227                  self.SavePreference(bg_name, style.background)

229      def SavePreference(self, name, value):
230          win32ui.WriteProfileVal("Format", name, value)

233  # An abstract formatter
234  # For all formatters we actually implement here.
235  # (as opposed to those formatters built in to Scintilla)
236  class Formatter(FormatterBase):
237      def __init__(self, scintilla):
238          self.bCompleteWhileIdle = 0
239          self.bHaveIdleHandler = 0  # Dont currently have an idle handle
240          self.nextstylenum = 0
241          FormatterBase.__init__(self, scintilla)
242  
243      def HookFormatter(self, parent=None):
244          if parent is None:
245              parent = self.scintilla.GetParent()  # was GetParentFrame()!?
246          parent.HookNotify(self.OnStyleNeeded, scintillacon.SCN_STYLENEEDED)
247  
248      def OnStyleNeeded(self, std, extra):
249          notify = self.scintilla.SCIUnpackNotifyMessage(extra)
250          endStyledChar = self.scintilla.SendScintilla(scintillacon.SCI_GETENDSTYLED)
251          lineEndStyled = self.scintilla.LineFromChar(endStyledChar)
252          endStyled = self.scintilla.LineIndex(lineEndStyled)
253          # print "enPosPaint %d endStyledChar %d lineEndStyled %d endStyled %d" % (endPosPaint, endStyledChar, lineEndStyled, endStyled)
254          self.Colorize(endStyled, notify.position)
255  
256      def ColorSeg(self, start, end, styleName):
257          end = end + 1
258          #       assert end-start>=0, "Can't have negative styling"
259          stylenum = self.styles[styleName].stylenum
260          while start < end:
261              self.style_buffer[start] = stylenum
262              start = start + 1
263          # self.scintilla.SCISetStyling(end - start + 1, stylenum)
264  
265      def RegisterStyle(self, style, stylenum=None):
266          if stylenum is None:
267              stylenum = self.nextstylenum
268              self.nextstylenum = self.nextstylenum + 1
269          FormatterBase.RegisterStyle(self, style, stylenum)
270  
271      def ColorizeString(self, str, charStart, styleStart):
272          raise RuntimeError("You must override this method")
273  
274      def Colorize(self, start=0, end=-1):
275          scintilla = self.scintilla
276          # scintilla's formatting is all done in terms of utf, so
277          # we work with utf8 bytes instead of unicode.  This magically
278          # works as any extended chars found in the utf8 don't change
279          # the semantics.
280          stringVal = scintilla.GetTextRange(start, end, decode=False)
281          if start > 0:
282              stylenum = scintilla.SCIGetStyleAt(start - 1)
283              styleStart = self.GetStyleByNum(stylenum).name
284          else:
285              styleStart = None
286          #       trace("Coloring", start, end, end-start, len(stringVal), styleStart, self.scintilla.SCIGetCharAt(start))
287          scintilla.SCIStartStyling(start, 31)
288          self.style_buffer = array.array("b", (0,) * len(stringVal))
289          self.ColorizeString(stringVal, styleStart)
290          scintilla.SCISetStylingEx(self.style_buffer)
291          self.style_buffer = None
292          #       trace("After styling, end styled is", self.scintilla.SCIGetEndStyled())
293          if (
294              self.bCompleteWhileIdle
295              and not self.bHaveIdleHandler
296              and end != -1
297              and end < scintilla.GetTextLength()
298          ):
299              self.bHaveIdleHandler = 1
300              win32ui.GetApp().AddIdleHandler(self.DoMoreColoring)
301              # Kicking idle makes the app seem slower when initially repainting!
302  
303      #           win32ui.GetMainFrame().PostMessage(WM_KICKIDLE, 0, 0)
304  
305      def DoMoreColoring(self, handler, count):
306          try:
307              scintilla = self.scintilla
308              endStyled = scintilla.SCIGetEndStyled()
309              lineStartStyled = scintilla.LineFromChar(endStyled)
310              start = scintilla.LineIndex(lineStartStyled)
311              end = scintilla.LineIndex(lineStartStyled + 1)
312              textlen = scintilla.GetTextLength()
313              if end < 0:
314                  end = textlen
315  
316              finished = end >= textlen
317              self.Colorize(start, end)
318          except (win32ui.error, AttributeError):
319              # Window may have closed before we finished - no big deal!
320              finished = 1
321  
322          if finished:
323              self.bHaveIdleHandler = 0
324              win32ui.GetApp().DeleteIdleHandler(handler)
325          return not finished

451      def SetStyles(self):
452          for name, format, bg, ignore in PYTHON_STYLES:
453              self.RegisterStyle(Style(name, format, bg))
454          for name, format, bg, sc_id in SPECIAL_STYLES:
455              self.RegisterStyle(Style(name, format, bg), sc_id)

457      def ClassifyWord(self, cdoc, start, end, prevWord):
458          word = cdoc[start : end + 1].decode("latin-1")
459          attr = STYLE_IDENTIFIER
460          if prevWord == "class":
461              attr = STYLE_CLASS
462          elif prevWord == "def":
463              attr = STYLE_METHOD
464          elif word[0] in string.digits:
465              attr = STYLE_NUMBER
466          elif iskeyword(word):
467              attr = STYLE_KEYWORD
468          self.ColorSeg(start, end, attr)
469          return word
470  
471      def ColorizeString(self, str, styleStart):
472          if styleStart is None:
473              styleStart = STYLE_DEFAULT
474          return self.ColorizePythonCode(str, 0, styleStart)
475  
476      def ColorizePythonCode(self, cdoc, charStart, styleStart):
477          # Straight translation of C++, should do better
478          lengthDoc = len(cdoc)
479          if lengthDoc <= charStart:
480              return
481          prevWord = ""
482          state = styleStart
483          chPrev = chPrev2 = chPrev3 = " "
484          chNext2 = chNext = cdoc[charStart : charStart + 1].decode("latin-1")
485          startSeg = i = charStart
486          while i < lengthDoc:
487              ch = chNext
488              chNext = " "
489              if i + 1 < lengthDoc:
490                  chNext = cdoc[i + 1 : i + 2].decode("latin-1")
491              chNext2 = " "
492              if i + 2 < lengthDoc:
493                  chNext2 = cdoc[i + 2 : i + 3].decode("latin-1")
494              if state == STYLE_DEFAULT:
495                  if ch in wordstarts:
496                      self.ColorSeg(startSeg, i - 1, STYLE_DEFAULT)
497                      state = STYLE_KEYWORD
498                      startSeg = i
499                  elif ch == "#":
500                      self.ColorSeg(startSeg, i - 1, STYLE_DEFAULT)
501                      if chNext == "#":
502                          state = STYLE_COMMENT_BLOCK
503                      else:
504                          state = STYLE_COMMENT
505                      startSeg = i
506                  elif ch == '"':
507                      self.ColorSeg(startSeg, i - 1, STYLE_DEFAULT)
508                      startSeg = i
509                      state = STYLE_COMMENT
510                      if chNext == '"' and chNext2 == '"':
511                          i = i + 2
512                          state = STYLE_TQDSTRING
513                          ch = " "
514                          chPrev = " "
515                          chNext = " "
516                          if i + 1 < lengthDoc:
517                              chNext = cdoc[i + 1]
518                      else:
519                          state = STYLE_STRING
520                  elif ch == "'":
521                      self.ColorSeg(startSeg, i - 1, STYLE_DEFAULT)
522                      startSeg = i
523                      state = STYLE_COMMENT
524                      if chNext == "'" and chNext2 == "'":
525                          i = i + 2
526                          state = STYLE_TQSSTRING
527                          ch = " "
528                          chPrev = " "
529                          chNext = " "
530                          if i + 1 < lengthDoc:
531                              chNext = cdoc[i + 1]
532                      else:
533                          state = STYLE_SQSTRING
534                  elif ch in operators:
535                      self.ColorSeg(startSeg, i - 1, STYLE_DEFAULT)
536                      self.ColorSeg(i, i, STYLE_OPERATOR)
537                      startSeg = i + 1
538              elif state == STYLE_KEYWORD:
539                  if ch not in wordchars:
540                      prevWord = self.ClassifyWord(cdoc, startSeg, i - 1, prevWord)
541                      state = STYLE_DEFAULT
542                      startSeg = i
543                      if ch == "#":
544                          if chNext == "#":
545                              state = STYLE_COMMENT_BLOCK
546                          else:
547                              state = STYLE_COMMENT
548                      elif ch == '"':
549                          if chNext == '"' and chNext2 == '"':
550                              i = i + 2
551                              state = STYLE_TQDSTRING
552                              ch = " "
553                              chPrev = " "
554                              chNext = " "
555                              if i + 1 < lengthDoc:
556                                  chNext = cdoc[i + 1]
557                          else:
558                              state = STYLE_STRING
559                      elif ch == "'":
560                          if chNext == "'" and chNext2 == "'":
561                              i = i + 2
562                              state = STYLE_TQSSTRING
563                              ch = " "
564                              chPrev = " "
565                              chNext = " "
566                              if i + 1 < lengthDoc:
567                                  chNext = cdoc[i + 1]
568                          else:
569                              state = STYLE_SQSTRING
570                      elif ch in operators:
571                          self.ColorSeg(startSeg, i, STYLE_OPERATOR)
572                          startSeg = i + 1
573              elif state == STYLE_COMMENT or state == STYLE_COMMENT_BLOCK:
574                  if ch == "\r" or ch == "\n":
575                      self.ColorSeg(startSeg, i - 1, state)
576                      state = STYLE_DEFAULT
577                      startSeg = i
578              elif state == STYLE_STRING:
579                  if ch == "\\":
580                      if chNext == '"' or chNext == "'" or chNext == "\\":
581                          i = i + 1
582                          ch = chNext
583                          chNext = " "
584                          if i + 1 < lengthDoc:
585                              chNext = cdoc[i + 1]
586                  elif ch == '"':
587                      self.ColorSeg(startSeg, i, STYLE_STRING)
588                      state = STYLE_DEFAULT
589                      startSeg = i + 1
590              elif state == STYLE_SQSTRING:
591                  if ch == "\\":
592                      if chNext == '"' or chNext == "'" or chNext == "\\":
593                          i = i + 1
594                          ch = chNext
595                          chNext = " "
596                          if i + 1 < lengthDoc:
597                              chNext = cdoc[i + 1]
598                  elif ch == "'":
599                      self.ColorSeg(startSeg, i, STYLE_SQSTRING)
600                      state = STYLE_DEFAULT
601                      startSeg = i + 1
602              elif state == STYLE_TQSSTRING:
603                  if ch == "'" and chPrev == "'" and chPrev2 == "'" and chPrev3 != "\\":
604                      self.ColorSeg(startSeg, i, STYLE_TQSSTRING)
605                      state = STYLE_DEFAULT
606                      startSeg = i + 1
607              elif (
608                  state == STYLE_TQDSTRING
609                  and ch == '"'
610                  and chPrev == '"'
611                  and chPrev2 == '"'
612                  and chPrev3 != "\\"
613              ):
614                  self.ColorSeg(startSeg, i, STYLE_TQDSTRING)
615                  state = STYLE_DEFAULT
616                  startSeg = i + 1
617              chPrev3 = chPrev2
618              chPrev2 = chPrev
619              chPrev = ch
620              i = i + 1
621          if startSeg < lengthDoc:
622              if state == STYLE_KEYWORD:
623                  self.ClassifyWord(cdoc, startSeg, lengthDoc - 1, prevWord)
624              else:
625                  self.ColorSeg(startSeg, lengthDoc - 1, state)

647  class BuiltinSourceFormatter(FormatterBase):
648      # A class that represents a formatter built-in to Scintilla
649      def __init__(self, scintilla, ext):
650          self.ext = ext
651          FormatterBase.__init__(self, scintilla)
652  
653      def Colorize(self, start=0, end=-1):
654          self.scintilla.SendScintilla(scintillacon.SCI_COLOURISE, start, end)
655  
656      def RegisterStyle(self, style, stylenum=None):
657          assert style.stylenum is None, "Style has already been registered"
658          if stylenum is None:
659              stylenum = self.nextstylenum
660              self.nextstylenum = self.nextstylenum + 1
661          assert self.styles.get(stylenum) is None, "We are reusing a style number!"
662          style.stylenum = stylenum
663          self.styles[style.name] = style
664          self.styles_by_id[stylenum] = style
665  
666      def HookFormatter(self, parent=None):
667          sc = self.scintilla
668          for exts, formatter in source_formatter_extensions:
669              if self.ext in exts:
670                  formatter_use = formatter
671                  break
672          else:
673              formatter_use = scintillacon.SCLEX_PYTHON
674          sc.SendScintilla(scintillacon.SCI_SETLEXER, formatter_use)
675          keywords = " ".join(kwlist)
676          sc.SCISetKeywords(keywords)
