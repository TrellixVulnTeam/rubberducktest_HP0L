28   class EmptyVolumeCache:
29       _reg_progid_ = "Python.ShellExtension.EmptyVolumeCache"
30       _reg_desc_ = "Python Sample Shell Extension (disk cleanup)"
31       _reg_clsid_ = "{EADD0777-2968-4c72-A999-2BF5F756259C}"
32       _reg_icon_ = ico
33       _com_interfaces_ = [shell.IID_IEmptyVolumeCache, shell.IID_IEmptyVolumeCache2]
34       _public_methods_ = IEmptyVolumeCache_Methods + IEmptyVolumeCache2_Methods

36       def Initialize(self, hkey, volume, flags):
37           # This should never be called, except on win98.
38           print("Unless we are on 98, Initialize call is unexpected!")
39           raise COMException(hresult=winerror.E_NOTIMPL)

41       def InitializeEx(self, hkey, volume, key_name, flags):
42           # Must return a tuple of:
43           # (display_name, description, button_name, flags)
44           print("InitializeEx called with", hkey, volume, key_name, flags)
45           self.volume = volume
46           if flags & shellcon.EVCF_SETTINGSMODE:
47               print("We are being run on a schedule")
48               # In this case, "because there is no opportunity for user
49               # feedback, only those files that are extremely safe to clean up
50               # should be touched. You should ignore the initialization
51               # method's pcwszVolume parameter and clean unneeded files
52               # regardless of what drive they are on."
53               self.volume = None  # flag as 'any disk will do'
54           elif flags & shellcon.EVCF_OUTOFDISKSPACE:
55               # In this case, "the handler should be aggressive about deleting
56               # files, even if it results in a performance loss. However, the
57               # handler obviously should not delete files that would cause an
58               # application to fail or the user to lose data."
59               print("We are being run as we are out of disk-space")
60           else:
61               # This case is not documented - we are guessing :)
62               print("We are being run because the user asked")
63   
64           # For the sake of demo etc, we tell the shell to only show us when
65           # there are > 0 bytes available.  Our GetSpaceUsed will check the
66           # volume, so will return 0 when we are on a different disk
67           flags = shellcon.EVCF_DONTSHOWIFZERO | shellcon.EVCF_ENABLEBYDEFAULT
68   
69           return (
70               "pywin32 compiled files",
71               "Removes all .pyc and .pyo files in the pywin32 directories",
72               "click me!",
73               flags,
74           )

76       def _GetDirectories(self):
77           root_dir = os.path.abspath(os.path.dirname(os.path.dirname(win32gui.__file__)))
78           if self.volume is not None and not root_dir.lower().startswith(
79               self.volume.lower()
80           ):
81               return []
82           return [
83               os.path.join(root_dir, p)
84               for p in ("win32", "win32com", "win32comext", "isapi")
85           ]
86   
87       def _WalkCallback(self, arg, directory, files):
88           # callback function for os.path.walk - no need to be member, but its
89           # close to the callers :)
90           callback, total_list = arg
91           for file in files:
92               fqn = os.path.join(directory, file).lower()
93               if file.endswith(".pyc") or file.endswith(".pyo"):
94                   # See below - total_list == None means delete files,
95                   # otherwise it is a list where the result is stored. Its a
96                   # list simply due to the way os.walk works - only [0] is
97                   # referenced
98                   if total_list is None:
99                       print("Deleting file", fqn)
100                      # Should do callback.PurgeProcess - left as an exercise :)
101                      os.remove(fqn)
102                  else:
103                      total_list[0] += os.stat(fqn)[stat.ST_SIZE]
104                      # and callback to the tool
105                      if callback:
106                          # for the sake of seeing the progress bar do its thing,
107                          # we take longer than we need to...
108                          # ACK - for some bizarre reason this screws up the XP
109                          # cleanup manager - clues welcome!! :)
110                          ## print "Looking in", directory, ", but waiting a while..."
111                          ## time.sleep(3)
112                          # now do it
113                          used = total_list[0]
114                          callback.ScanProgress(used, 0, "Looking at " + fqn)

116      def GetSpaceUsed(self, callback):
117          total = [0]  # See _WalkCallback above
118          try:
119              for d in self._GetDirectories():
120                  os.path.walk(d, self._WalkCallback, (callback, total))
121                  print("After looking in", d, "we have", total[0], "bytes")
122          except pythoncom.error as exc:
123              # This will be raised by the callback when the user selects 'cancel'.
124              if exc.hresult != winerror.E_ABORT:
125                  raise  # that's the documented error code!
126              print("User cancelled the operation")
127          return total[0]

129      def Purge(self, amt_to_free, callback):
130          print("Purging", amt_to_free, "bytes...")
131          # we ignore amt_to_free - it is generally what we returned for
132          # GetSpaceUsed
133          try:
134              for d in self._GetDirectories():
135                  os.path.walk(d, self._WalkCallback, (callback, None))
136          except pythoncom.error as exc:
137              # This will be raised by the callback when the user selects 'cancel'.
138              if exc.hresult != winerror.E_ABORT:
139                  raise  # that's the documented error code!
140              print("User cancelled the operation")

150  def DllRegisterServer():
151      # Also need to register specially in:
152      # HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\Explorer\VolumeCaches
153      # See link at top of file.
154      import winreg
155  
156      kn = r"Software\Microsoft\Windows\CurrentVersion\Explorer\VolumeCaches\%s" % (
157          EmptyVolumeCache._reg_desc_,
158      )
159      key = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, kn)
160      winreg.SetValueEx(key, None, 0, winreg.REG_SZ, EmptyVolumeCache._reg_clsid_)
161  
162  
163  def DllUnregisterServer():
164      import winreg
165  
166      kn = r"Software\Microsoft\Windows\CurrentVersion\Explorer\VolumeCaches\%s" % (
167          EmptyVolumeCache._reg_desc_,
168      )
169      try:
170          key = winreg.DeleteKey(winreg.HKEY_LOCAL_MACHINE, kn)
171      except WindowsError as details:
172          import errno
173  
174          if details.errno != errno.ENOENT:
175              raise
176      print(EmptyVolumeCache._reg_desc_, "unregistration complete.")
