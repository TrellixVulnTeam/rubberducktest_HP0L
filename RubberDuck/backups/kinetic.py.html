1    ''' 
2    Kinetic effect 
3    ============== 
4     
5    .. versionadded:: 1.7.0 
6     
7    The :class:`KineticEffect` is the base class that is used to compute the 
8    velocity out of a movement. When the movement is finished, the effect will 
9    compute the position of the movement according to the velocity, and reduce the 
10   velocity with a friction. The movement stop until the velocity is 0. 
11    
12   Conceptually, the usage could be:: 
13    
14       >>> effect = KineticEffect() 
15       >>> effect.start(10) 
16       >>> effect.update(15) 
17       >>> effect.update(30) 
18       >>> effect.stop(48) 
19    
20   Over the time, you will start a movement of a value, update it, and stop the 
21   movement. At this time, you'll get the movement value into 
22   :attr:`KineticEffect.value`. On the example i've typed manually, the computed 
23   velocity will be:: 
24    
25       >>> effect.velocity 
26       3.1619100231163046 
27    
28   After multiple clock interaction, the velocity will decrease according to 
29   :attr:`KineticEffect.friction`. The computed value will be stored in 
30   :attr:`KineticEffect.value`. The output of this `value` could be:: 
31    
32       46.30038145219605 
33       54.58302451968686 
34       61.9229016256196 
35       # ... 
36    
37   '''
38   
39   __all__ = ('KineticEffect', )
40   
41   
42   from time import time
43   from kivy.event import EventDispatcher
44   from kivy.properties import NumericProperty, BooleanProperty
45   from kivy.clock import Clock
46   
47   
48   class KineticEffect(EventDispatcher):
49       '''Kinetic effect class. See module documentation for more information. 
50       '''
51   
52       velocity = NumericProperty(0)
53       '''Velocity of the movement. 
54    
55       :attr:`velocity` is a :class:`~kivy.properties.NumericProperty` and 
56       defaults to 0. 
57       '''
58   
59       friction = NumericProperty(0.05)
60       '''Friction to apply on the velocity 
61    
62       :attr:`velocity` is a :class:`~kivy.properties.NumericProperty` and 
63       defaults to 0.05. 
64       '''
65   
66       value = NumericProperty(0)
67       '''Value (during the movement and computed) of the effect. 
68    
69       :attr:`velocity` is a :class:`~kivy.properties.NumericProperty` and 
70       defaults to 0. 
71       '''
72   
73       is_manual = BooleanProperty(False)
74       '''Indicate if a movement is in progress (True) or not (False). 
75    
76       :attr:`velocity` is a :class:`~kivy.properties.BooleanProperty` and 
77       defaults to False. 
78       '''
79   
80       max_history = NumericProperty(5)
81       '''Save up to `max_history` movement value into the history. This is used 
82       for correctly calculating the velocity according to the movement. 
83    
84       :attr:`max_history` is a :class:`~kivy.properties.NumericProperty` and 
85       defaults to 5. 
86       '''
87       min_distance = NumericProperty(.1)
88       '''The minimal distance for a movement to have nonzero velocity. 
89    
90       .. versionadded:: 1.8.0 
91    
92       :attr:`min_distance` is :class:`~kivy.properties.NumericProperty` and 
93       defaults to 0.1. 
94       '''
95   
96       min_velocity = NumericProperty(.5)
97       '''Velocity below this quantity is normalized to 0. In other words, 
98       any motion whose velocity falls below this number is stopped. 
99    
100      .. versionadded:: 1.8.0 
101   
102      :attr:`min_velocity` is a :class:`~kivy.properties.NumericProperty` and 
103      defaults to 0.5. 
104      '''
105  
106      std_dt = NumericProperty(0.017)
107      ''' std_dt 
108          correction update_velocity if dt is not constant 
109   
110      .. versionadded:: 2.0.0 
111   
112      :attr:`std_dt` is a :class:`~kivy.properties.NumericProperty` and 
113      defaults to 0.017. 
114      '''
115  
116      def __init__(self, **kwargs):
117          self.history = []
118          self.trigger_velocity_update = Clock.create_trigger(
119              self.update_velocity, 0)
120          super(KineticEffect, self).__init__(**kwargs)
121  
122      def apply_distance(self, distance):
123          if abs(distance) < self.min_distance:
124              self.velocity = 0
125          self.value += distance
126  
127      def start(self, val, t=None):
128          '''Start the movement. 
129   
130          :Parameters: 
131              `val`: float or int 
132                  Value of the movement 
133              `t`: float, defaults to None 
134                  Time when the movement happen. If no time is set, it will use 
135                  time.time() 
136          '''
137          self.is_manual = True
138          t = t or time()
139          self.velocity = 0
140          self.history = [(t, val)]
141  
142      def update(self, val, t=None):
143          '''Update the movement. 
144   
145          See :meth:`start` for the arguments. 
146          '''
147          t = t or time()
148          distance = val - self.history[-1][1]
149          self.apply_distance(distance)
150          self.history.append((t, val))
151          if len(self.history) > self.max_history:
152              self.history.pop(0)
153  
154      def stop(self, val, t=None):
155          '''Stop the movement. 
156   
157          See :meth:`start` for the arguments. 
158          '''
159          self.is_manual = False
160          t = t or time()
161          distance = val - self.history[-1][1]
162          self.apply_distance(distance)
163          newest_sample = (t, val)
164          old_sample = self.history[0]
165          for sample in self.history:
166              if (newest_sample[0] - sample[0]) < 10. / 60.:
167                  break
168              old_sample = sample
169          distance = newest_sample[1] - old_sample[1]
170          duration = abs(newest_sample[0] - old_sample[0])
171          self.velocity = (distance / max(duration, 0.0001))
172          self.trigger_velocity_update()
173  
174      def cancel(self):
175          '''Cancel a movement. This can be used in case :meth:`stop` cannot be 
176          called. It will reset :attr:`is_manual` to False, and compute the 
177          movement if the velocity is > 0. 
178          '''
179          self.is_manual = False
180          self.trigger_velocity_update()
181  
182      def update_velocity(self, dt):
183          '''(internal) Update the velocity according to the frametime and 
184          friction. 
185          '''
186          if abs(self.velocity) <= self.min_velocity:
187              self.velocity = 0
188              return
189  
190          self.velocity -= self.velocity * self.friction * dt / self.std_dt
191          self.apply_distance(self.velocity * dt)
192          self.trigger_velocity_update()
