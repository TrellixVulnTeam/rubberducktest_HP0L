25   % duclass::
26   
27   % class handling for environments (block-level elements)
28   % \begin{DUclass}{spam} tries \DUCLASSspam and
29   % \end{DUclass}{spam} tries \endDUCLASSspam
30   \ifx\DUclass\undefined % poor man's "provideenvironment"
31    \newenvironment{DUclass}[1]%
32     {% "#1" does not work in end-part of environment.
33      \def\DocutilsClassFunctionName{DUCLASS#1}
34        \csname \DocutilsClassFunctionName \endcsname}%
35     {\csname end\DocutilsClassFunctionName \endcsname}%
36   \fi

38   % providelength::
39   
40   % Provide a length variable and set default, if it is new
41   \providecommand*{\DUprovidelength}[2]{
42     \ifthenelse{\isundefined{#1}}{\newlength{#1}\setlength{#1}{#2}}{}
43   }

88   % admonition::
89   
90   % admonition environment (specially marked topic)
91   \ifx\DUadmonition\undefined % poor man's "provideenvironment"
92    \newbox{\DUadmonitionbox}
93    \newenvironment{DUadmonition}%
94     {\begin{center}
95        \begin{lrbox}{\DUadmonitionbox}
96          \begin{minipage}{0.9\linewidth}
97     }%
98     {    \end{minipage}
99        \end{lrbox}
100       \fbox{\usebox{\DUadmonitionbox}}
101     \end{center}
102    }
103  \fi

114  % footnotes::
115  
116  % numerical or symbol footnotes with hyperlinks and backlinks
117  \providecommand*{\DUfootnotemark}[3]{%
118    \raisebox{1em}{\hypertarget{#1}{}}%
119    \hyperlink{#2}{\textsuperscript{#3}}%
120  }
121  \providecommand{\DUfootnotetext}[4]{%
122    \begingroup%
123    \renewcommand{\thefootnote}{%
124      \protect\raisebox{1em}{\protect\hypertarget{#1}{}}%
125      \protect\hyperlink{#2}{#3}}%
126    \footnotetext{#4}%
127    \endgroup%
128  }

130  % inline::
131  
132  % custom inline roles: \DUrole{#1}{#2} tries \DUrole#1{#2}
133  \providecommand*{\DUrole}[2]{%
134    \ifcsname DUrole#1\endcsname%
135      \csname DUrole#1\endcsname{#2}%
136    \else%
137      #2%
138    \fi%
139  }

148  % lineblock::
149  
150  % line block environment
151  \DUprovidelength{\DUlineblockindent}{2.5em}
152  \ifthenelse{\isundefined{\DUlineblock}}{
153    \newenvironment{DUlineblock}[1]{%
154      \list{}{\setlength{\partopsep}{\parskip}
155              \addtolength{\partopsep}{\baselineskip}
156              \setlength{\topsep}{0pt}
157              \setlength{\itemsep}{0.15\baselineskip}
158              \setlength{\parsep}{0pt}
159              \setlength{\leftmargin}{#1}}
160      \raggedright
161    }
162    {\endlist}
163  }{}

165  % optionlist::
166  
167  % list of command line options
168  \providecommand*{\DUoptionlistlabel}[1]{\bfseries #1 \hfill}
169  \DUprovidelength{\DUoptionlistindent}{3cm}
170  \ifthenelse{\isundefined{\DUoptionlist}}{
171    \newenvironment{DUoptionlist}{%
172      \list{}{\setlength{\labelwidth}{\DUoptionlistindent}
173              \setlength{\rightmargin}{1cm}
174              \setlength{\leftmargin}{\rightmargin}
175              \addtolength{\leftmargin}{\labelwidth}
176              \addtolength{\leftmargin}{\labelsep}
177              \renewcommand{\makelabel}{\DUoptionlistlabel}}
178    }
179    {\endlist}
180  }{}
