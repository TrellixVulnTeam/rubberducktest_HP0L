1    ''' 
2    AppKit Spelling: Implements spelling backend based on OSX's spellchecking 
3                     features provided by the ApplicationKit. 
4     
5                     NOTE: 
6                        Requires pyobjc and setuptools to be installed! 
7                        `sudo easy_install pyobjc setuptools` 
8     
9                     Developers should read: 
10                       http://developer.apple.com/mac/library/documentation/ 
11                           Cocoa/Conceptual/SpellCheck/SpellCheck.html 
12                       http://developer.apple.com/cocoa/pyobjc.html 
13   '''
14   
15   
16   from AppKit import NSSpellChecker, NSMakeRange
17   
18   from kivy.core.spelling import SpellingBase, NoSuchLangError
19   
20   
21   class SpellingOSXAppKit(SpellingBase):
22       ''' 
23       Spelling backend based on OSX's spelling features provided by AppKit. 
24       '''
25   
26       def __init__(self, language=None):
27           self._language = NSSpellChecker.alloc().init()
28           super(SpellingOSXAppKit, self).__init__(language)
29   
30       def select_language(self, language):
31           success = self._language.setLanguage_(language)
32           if not success:
33               err = 'AppKit Backend: No language "%s" ' % (language, )
34               raise NoSuchLangError(err)
35   
36       def list_languages(self):
37           return list(self._language.availableLanguages())
38   
39       def check(self, word):
40           # TODO Implement this!
41           #      NSSpellChecker provides several functions that look like what we
42           #      need, but they're a) slooow and b) return a strange result.
43           #      Might be a snow leopard bug. Have to test further.
44           #      See: http://paste.pocoo.org/show/217968/
45           if not word:
46               return None
47           err = 'check() not currently supported by the OSX AppKit backend'
48           raise NotImplementedError(err)
49   
50       def suggest(self, fragment):
51           l = self._language
52           # XXX Both ways below work on OSX 10.6. It has not been tested on any
53           #     other version, but it should work.
54           try:
55               # This is deprecated as of OSX 10.6, hence the try-except
56               return list(l.guessesForWord_(fragment))
57           except AttributeError:
58               # From 10.6 onwards you're supposed to do it like this:
59               checkrange = NSMakeRange(0, len(fragment))
60               g = l.\
61                   guessesForWordRange_inString_language_inSpellDocumentWithTag_(
62                       checkrange, fragment, l.language(), 0)
63               # Right, this was much easier, Apple! :-)
64               return list(g)
