1    import win32print, pywintypes, win32con, win32gui, win32api
2    
3    pname = win32print.GetDefaultPrinter()
4    print(pname)
5    p = win32print.OpenPrinter(pname)
6    print("Printer handle: ", p)
7    print_processor = win32print.GetPrinter(p, 2)["pPrintProcessor"]
8    ## call with last parm set to 0 to get total size needed for printer's DEVMODE
9    dmsize = win32print.DocumentProperties(0, p, pname, None, None, 0)
10   ## dmDriverExtra should be total size - fixed size
11   driverextra = (
12       dmsize - pywintypes.DEVMODEType().Size
13   )  ## need a better way to get DEVMODE.dmSize
14   dm = pywintypes.DEVMODEType(driverextra)
15   dm.Fields = dm.Fields | win32con.DM_ORIENTATION | win32con.DM_COPIES
16   dm.Orientation = win32con.DMORIENT_LANDSCAPE
17   dm.Copies = 2
18   win32print.DocumentProperties(
19       0, p, pname, dm, dm, win32con.DM_IN_BUFFER | win32con.DM_OUT_BUFFER
20   )
21   
22   pDC = win32gui.CreateDC(print_processor, pname, dm)
23   printerwidth = win32print.GetDeviceCaps(pDC, win32con.PHYSICALWIDTH)
24   printerheight = win32print.GetDeviceCaps(pDC, win32con.PHYSICALHEIGHT)
25   
26   hwnd = win32gui.GetDesktopWindow()
27   l, t, r, b = win32gui.GetWindowRect(hwnd)
28   desktopheight = b - t
29   desktopwidth = r - l
30   dDC = win32gui.GetWindowDC(hwnd)
31   
32   dcDC = win32gui.CreateCompatibleDC(dDC)
33   dcBM = win32gui.CreateCompatibleBitmap(dDC, desktopwidth, desktopheight)
34   win32gui.SelectObject(dcDC, dcBM)
35   win32gui.StretchBlt(
36       dcDC,
37       0,
38       0,
39       desktopwidth,
40       desktopheight,
41       dDC,
42       0,
43       0,
44       desktopwidth,
45       desktopheight,
46       win32con.SRCCOPY,
47   )
48   
49   pcDC = win32gui.CreateCompatibleDC(pDC)
50   pcBM = win32gui.CreateCompatibleBitmap(pDC, printerwidth, printerheight)
51   win32gui.SelectObject(pcDC, pcBM)
52   win32gui.StretchBlt(
53       pcDC,
54       0,
55       0,
56       printerwidth,
57       printerheight,
58       dcDC,
59       0,
60       0,
61       desktopwidth,
62       desktopheight,
63       win32con.SRCCOPY,
64   )
65   
66   win32print.StartDoc(pDC, ("desktop.bmp", None, None, 0))
67   win32print.StartPage(pDC)
68   win32gui.StretchBlt(
69       pDC,
70       0,
71       0,
72       int(printerwidth * 0.9),
73       int(printerheight * 0.9),
74       pcDC,
75       0,
76       0,
77       printerwidth,
78       printerheight,
79       win32con.SRCCOPY,
80   )
81   
82   font = win32gui.LOGFONT()
83   font.lfHeight = int(printerheight / 20)
84   font.lfWidth = font.lfHeight
85   font.lfWeight = 150
86   font.lfItalic = 1
87   font.lfUnderline = 1
88   hf = win32gui.CreateFontIndirect(font)
89   win32gui.SelectObject(pDC, hf)
90   win32gui.SetBkMode(pDC, win32con.TRANSPARENT)
91   win32gui.SetTextColor(pDC, win32api.RGB(0, 255, 0))
92   win32gui.DrawText(
93       pDC,
94       "Printed by Python!",
95       -1,
96       (0, 0, int(printerwidth * 0.9), int(printerheight * 0.9)),
97       win32con.DT_RIGHT | win32con.DT_BOTTOM | win32con.DT_SINGLELINE,
98   )
99   win32print.EndPage(pDC)
100  win32print.EndDoc(pDC)
101  
102  win32print.ClosePrinter(p)
103  win32gui.DeleteObject(dcBM)
104  win32gui.DeleteObject(pcBM)
105  win32gui.DeleteObject(hf)
106  win32gui.DeleteDC(dDC)
107  win32gui.DeleteDC(dcDC)
108  win32gui.DeleteDC(pDC)
109  win32gui.DeleteDC(pcDC)
