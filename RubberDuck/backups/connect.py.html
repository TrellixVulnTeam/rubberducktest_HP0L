1    """Utilities for Server Side connections. 
2     
3      A collection of helpers for server side connection points. 
4    """
5    import pythoncom
6    from .exception import Exception
7    import winerror
8    from win32com import olectl
9    import win32com.server.util
10   
11   # Methods implemented by the interfaces.
12   IConnectionPointContainer_methods = ["EnumConnectionPoints", "FindConnectionPoint"]
13   IConnectionPoint_methods = [
14       "EnumConnections",
15       "Unadvise",
16       "Advise",
17       "GetConnectionPointContainer",
18       "GetConnectionInterface",
19   ]
20   
21   
22   class ConnectableServer:
23       _public_methods_ = IConnectionPointContainer_methods + IConnectionPoint_methods
24       _com_interfaces_ = [
25           pythoncom.IID_IConnectionPoint,
26           pythoncom.IID_IConnectionPointContainer,
27       ]
28       # Clients must set _connect_interfaces_ = [...]
29       def __init__(self):
30           self.cookieNo = 0
31           self.connections = {}
32   
33       # IConnectionPoint interfaces
34       def EnumConnections(self):
35           raise Exception(winerror.E_NOTIMPL)
36   
37       def GetConnectionInterface(self):
38           raise Exception(winerror.E_NOTIMPL)
39   
40       def GetConnectionPointContainer(self):
41           return win32com.server.util.wrap(self)
42   
43       def Advise(self, pUnk):
44           # Creates a connection to the client.  Simply allocate a new cookie,
45           # find the clients interface, and store it in a dictionary.
46           try:
47               interface = pUnk.QueryInterface(
48                   self._connect_interfaces_[0], pythoncom.IID_IDispatch
49               )
50           except pythoncom.com_error:
51               raise Exception(scode=olectl.CONNECT_E_NOCONNECTION)
52           self.cookieNo = self.cookieNo + 1
53           self.connections[self.cookieNo] = interface
54           return self.cookieNo
55   
56       def Unadvise(self, cookie):
57           # Destroy a connection - simply delete interface from the map.
58           try:
59               del self.connections[cookie]
60           except KeyError:
61               raise Exception(scode=winerror.E_UNEXPECTED)
62   
63       # IConnectionPointContainer interfaces
64       def EnumConnectionPoints(self):
65           raise Exception(winerror.E_NOTIMPL)
66   
67       def FindConnectionPoint(self, iid):
68           # Find a connection we support.  Only support the single event interface.
69           if iid in self._connect_interfaces_:
70               return win32com.server.util.wrap(self)
71   
72       def _BroadcastNotify(self, broadcaster, extraArgs):
73           # Broadcasts a notification to all connections.
74           # Ignores clients that fail.
75           for interface in self.connections.values():
76               try:
77                   broadcaster(*(interface,) + extraArgs)
78               except pythoncom.com_error as details:
79                   self._OnNotifyFail(interface, details)
80   
81       def _OnNotifyFail(self, interface, details):
82           print("Ignoring COM error to connection - %s" % (repr(details)))
