16   # font is a dictionary in which the following elements matter:
17   # (the best matching font to supplied parameters is returned)
18   #   name        string name of the font as known by Windows
19   #   size        point size of font in logical units
20   #   weight      weight of font (win32con.FW_NORMAL, win32con.FW_BOLD)
21   #   italic      boolean; true if set to anything but None
22   #   underline   boolean; true if set to anything but None

24   # This window is a child window of a frame.  It is not the frame window itself.
25   class FontWindow(window.Wnd):
26       def __init__(self, text="Python Rules!"):
27           window.Wnd.__init__(self)
28           self.text = text
29           self.index = 0
30           self.incr = 1
31           self.width = self.height = 0
32           self.ChangeAttributes()
33           # set up message handlers
34   
35       def Create(self, title, style, rect, parent):
36           classStyle = win32con.CS_HREDRAW | win32con.CS_VREDRAW
37           className = win32ui.RegisterWndClass(
38               classStyle, 0, win32con.COLOR_WINDOW + 1, 0
39           )
40           self._obj_ = win32ui.CreateWnd()
41           self._obj_.AttachObject(self)
42           self._obj_.CreateWindow(
43               className, title, style, rect, parent, win32ui.AFX_IDW_PANE_FIRST
44           )
45           self.HookMessage(self.OnSize, win32con.WM_SIZE)
46           self.HookMessage(self.OnPrepareToClose, WM_USER_PREPARE_TO_CLOSE)
47           self.HookMessage(self.OnDestroy, win32con.WM_DESTROY)
48           self.timerid = timer.set_timer(100, self.OnTimer)
49           self.InvalidateRect()
50   
51       def OnDestroy(self, msg):
52           timer.kill_timer(self.timerid)
53   
54       def OnTimer(self, id, timeVal):
55           self.index = self.index + self.incr
56           if self.index > len(self.text):
57               self.incr = -1
58               self.index = len(self.text)
59           elif self.index < 0:
60               self.incr = 1
61               self.index = 0
62           self.InvalidateRect()
63   
64       def OnPaint(self):
65           #       print "Paint message from thread", win32api.GetCurrentThreadId()
66           dc, paintStruct = self.BeginPaint()
67           self.OnPrepareDC(dc, None)
68   
69           if self.width == 0 and self.height == 0:
70               left, top, right, bottom = self.GetClientRect()
71               self.width = right - left
72               self.height = bottom - top
73           x, y = self.width // 2, self.height // 2
74           dc.TextOut(x, y, self.text[: self.index])
75           self.EndPaint(paintStruct)
76   
77       def ChangeAttributes(self):
78           font_spec = {"name": "Arial", "height": 42}
79           self.font = win32ui.CreateFont(font_spec)
80   
81       def OnPrepareToClose(self, params):
82           self.DestroyWindow()
83   
84       def OnSize(self, params):
85           lParam = params[3]
86           self.width = win32api.LOWORD(lParam)
87           self.height = win32api.HIWORD(lParam)
88   
89       def OnPrepareDC(self, dc, printinfo):
90           # Set up the DC for forthcoming OnDraw call
91           dc.SetTextColor(win32api.RGB(0, 0, 255))
92           dc.SetBkColor(win32api.GetSysColor(win32con.COLOR_WINDOW))
93           dc.SelectObject(self.font)
94           dc.SetTextAlign(win32con.TA_CENTER | win32con.TA_BASELINE)

97   class FontFrame(window.MDIChildWnd):
98       def __init__(self):
99           pass  # Dont call base class doc/view version...
100  
101      def Create(self, title, rect=None, parent=None):
102          style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.WS_OVERLAPPEDWINDOW
103          self._obj_ = win32ui.CreateMDIChild()
104          self._obj_.AttachObject(self)
105  
106          self._obj_.CreateWindow(None, title, style, rect, parent)
107          rect = self.GetClientRect()
108          rect = (0, 0, rect[2] - rect[0], rect[3] - rect[1])
109          self.child = FontWindow("Not threaded")
110          self.child.Create(
111              "FontDemo", win32con.WS_CHILD | win32con.WS_VISIBLE, rect, self

115  class TestThread(WinThread):
116      def __init__(self, parentWindow):
117          self.parentWindow = parentWindow
118          self.child = None
119          WinThread.__init__(self)
120  
121      def InitInstance(self):
122          rect = self.parentWindow.GetClientRect()
123          rect = (0, 0, rect[2] - rect[0], rect[3] - rect[1])
124  
125          self.child = FontWindow()
126          self.child.Create(
127              "FontDemo", win32con.WS_CHILD | win32con.WS_VISIBLE, rect, self.parentWindow
128          )
129          self.SetMainFrame(self.child)
130          return WinThread.InitInstance(self)
131  
132      def ExitInstance(self):
133          return 0

136  class ThreadedFontFrame(window.MDIChildWnd):
137      def __init__(self):
138          pass  # Dont call base class doc/view version...
139          self.thread = None
140  
141      def Create(self, title, rect=None, parent=None):
142          style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.WS_OVERLAPPEDWINDOW
143          self._obj_ = win32ui.CreateMDIChild()
144          self._obj_.CreateWindow(None, title, style, rect, parent)
145          self._obj_.HookMessage(self.OnDestroy, win32con.WM_DESTROY)
146          self._obj_.HookMessage(self.OnSize, win32con.WM_SIZE)
147  
148          self.thread = TestThread(self)
149          self.thread.CreateThread()
150  
151      def OnSize(self, msg):
152          pass
153  
154      def OnDestroy(self, msg):
155          win32ui.OutputDebugString("OnDestroy\n")
156          if self.thread and self.thread.child:
157              child = self.thread.child
158              child.SendMessage(WM_USER_PREPARE_TO_CLOSE, 0, 0)
159              win32ui.OutputDebugString("Destroyed\n")

167  def ThreadedDemo():
168      rect = win32ui.GetMainFrame().GetMDIClient().GetClientRect()
169      rect = rect[0], int(rect[3] * 3 / 4), int(rect[2] / 4), rect[3]
170      incr = rect[2]
171      for i in range(4):
172          if i == 0:
173              f = FontFrame()
174              title = "Not threaded"
175          else:
176              f = ThreadedFontFrame()
177              title = "Threaded GUI Demo"
178          f.Create(title, rect)
179          rect = rect[0] + incr, rect[1], rect[2] + incr, rect[3]
180      # Givem a chance to start
181      win32api.Sleep(100)
182      win32ui.PumpWaitingMessages()
