74   def RaiseNotImpl(who=None):
75       if who is not None:
76           print("********* Function %s Raising E_NOTIMPL  ************" % (who))
77   
78       # Print a sort-of "traceback", dumping all the frames leading to here.
79       try:
80           1 / 0
81       except:
82           frame = sys.exc_info()[2].tb_frame
83       while frame:
84           print("File: %s, Line: %d" % (frame.f_code.co_filename, frame.f_lineno))
85           frame = frame.f_back
86   
87       # and raise the exception for COM
88       raise Exception(scode=winerror.E_NOTIMPL)

107      def _Invoke_(self, dispid, lcid, wFlags, args):
108          print(
109              "In Invoke with",
110              dispid,
111              lcid,
112              wFlags,
113              args,
114              "with object",
115              self.policy._obj_,
116          )
117          try:
118              rc = win32com.server.policy.DispatcherBase._Invoke_(
119                  self, dispid, lcid, wFlags, args
120              )
121              #                       print "Invoke of", dispid, "returning", rc
122              return rc
123          except Exception:
124              t, v, tb = sys.exc_info()
125              tb = None  # A cycle
126              scode = v.scode
127              try:
128                  desc = " (" + str(v.description) + ")"
129              except AttributeError:
130                  desc = ""
131              print(
132                  "*** Invoke of %s raised COM exception 0x%x%s" % (dispid, scode, desc)
133              )
134          except:
135              print("*** Invoke of %s failed:" % dispid)
136              typ, val, tb = sys.exc_info()
137              import traceback
138  
139              traceback.print_exception(typ, val, tb)
140              raise
