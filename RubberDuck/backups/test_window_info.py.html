1    from kivy.tests.common import GraphicUnitTest

5    class WindowInfoTest(GraphicUnitTest):
6        def test_window_info_nonzero(self):
7            from kivy.core.window import Window
8            window_info = Window.get_window_info()
9            if window_info is None:
10               return
11   
12           if setupconfig.USE_X11:
13               from kivy.core.window.window_info import WindowInfoX11
14               if isinstance(window_info, WindowInfoX11):
15                   self.assertNotEqual(window_info.display, 0)
16                   self.assertNotEqual(window_info.window, 0)
17   
18           if setupconfig.USE_WAYLAND:
19               from kivy.core.window.window_info import WindowInfoWayland
20               if isinstance(window_info, WindowInfoWayland):
21                   self.assertNotEqual(window_info.display, 0)
22                   self.assertNotEqual(window_info.surface, 0)
23                   self.assertNotEqual(window_info.shell_surface, 0)
24   
25           if setupconfig.PLATFORM == 'win32':
26               from kivy.core.window.window_info import WindowInfoWindows
27               if isinstance(window_info, WindowInfoWindows):
28                   self.assertNotEqual(window_info.HWND, 0)
29                   self.assertNotEqual(window_info.HDC, 0)
