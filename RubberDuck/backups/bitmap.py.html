14   class BitmapDocument(docview.Document):
15       "A bitmap document.  Holds the bitmap data itself."
16   
17       def __init__(self, template):
18           docview.Document.__init__(self, template)
19           self.bitmap = None
20   
21       def OnNewDocument(self):
22           # I can not create new bitmaps.
23           win32ui.MessageBox("Bitmaps can not be created.")
24   
25       def OnOpenDocument(self, filename):
26           self.bitmap = win32ui.CreateBitmap()
27           # init data members
28           f = open(filename, "rb")
29           try:
30               try:
31                   self.bitmap.LoadBitmapFile(f)
32               except IOError:
33                   win32ui.MessageBox("Could not load the bitmap from %s" % filename)
34                   return 0
35           finally:
36               f.close()
37           self.size = self.bitmap.GetSize()
38           return 1
39   
40       def DeleteContents(self):
41           self.bitmap = None

44   class BitmapView(docview.ScrollView):
45       "A view of a bitmap.  Obtains data from document."
46   
47       def __init__(self, doc):
48           docview.ScrollView.__init__(self, doc)
49           self.width = self.height = 0
50           # set up message handlers
51           self.HookMessage(self.OnSize, win32con.WM_SIZE)
52   
53       def OnInitialUpdate(self):
54           doc = self.GetDocument()
55           if doc.bitmap:
56               bitmapSize = doc.bitmap.GetSize()
57               self.SetScrollSizes(win32con.MM_TEXT, bitmapSize)
58   
59       def OnSize(self, params):
60           lParam = params[3]
61           self.width = win32api.LOWORD(lParam)
62           self.height = win32api.HIWORD(lParam)
63   
64       def OnDraw(self, dc):
65           # set sizes used for "non stretch" mode.
66           doc = self.GetDocument()
67           if doc.bitmap is None:
68               return
69           bitmapSize = doc.bitmap.GetSize()
70           if bStretch:
71               # stretch BMP.
72               viewRect = (0, 0, self.width, self.height)
73               bitmapRect = (0, 0, bitmapSize[0], bitmapSize[1])
74               doc.bitmap.Paint(dc, viewRect, bitmapRect)
75           else:
76               # non stretch.
77               doc.bitmap.Paint(dc)
