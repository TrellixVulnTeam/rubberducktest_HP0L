14   def wait_request_is_finished(kivy_clock, request, timeout=10):
15       start_time = datetime.now()
16       timed_out = False
17       while not request.is_finished and not timed_out:
18           kivy_clock.tick()
19           sleep(.1)
20           timed_out = (datetime.now() - start_time).total_seconds() > timeout
21       assert request.is_finished

24   def ensure_called_from_thread(queue):
25       """Ensures the callback is called from this thread (main)."""
26       tid = threading.get_ident()
27       assert queue[0][0] == tid
28       assert queue[-2][0] == tid
29       assert queue[-1][0] == tid

32   def check_queue_values(queue):
33       """Helper function verifying the queue contains the expected values."""
34       # we should have 2 progress minimum and one success
35       assert len(queue) >= 3
36   
37       assert queue[0][1] == 'progress'
38       assert queue[-2][1] == 'progress'
39       assert queue[-1][1] in ('success', 'redirect')
40   
41       assert queue[0][2][0] == 0
42       assert queue[-2][2][0] == queue[-2][2][1]

45   class UrlRequestQueue:
46   
47       def __init__(self, queue):
48           self.queue = queue
49   
50       def _on_success(self, req, *args):
51           self.queue.append((threading.get_ident(), 'success', args))
52   
53       def _on_redirect(self, req, *args):
54           self.queue.append((threading.get_ident(), 'redirect', args))
55   
56       def _on_error(self, req, *args):
57           self.queue.append((threading.get_ident(), 'error', args))
58   
59       def _on_progress(self, req, *args):
60           self.queue.append((threading.get_ident(), 'progress', args))

63   @pytest.mark.skipif(os.environ.get('NONETWORK'), reason="No network")
64   def test_callbacks(kivy_clock):
65       from kivy.network.urlrequest import UrlRequest
66       obj = UrlRequestQueue([])
67       queue = obj.queue
68       req = UrlRequest('http://google.com',
69                        on_success=obj._on_success,
70                        on_progress=obj._on_progress,
71                        on_error=obj._on_error,
72                        on_redirect=obj._on_redirect,
73                        debug=True)
74       wait_request_is_finished(kivy_clock, req)
75   
76       if req.error and req.error.errno == 11001:
77           pytest.skip('Cannot connect to get address')
78   
79       ensure_called_from_thread(queue)
80       check_queue_values(queue)

83   @pytest.mark.skipif(os.environ.get('NONETWORK'), reason="No network")
84   def test_auth_header(kivy_clock):
85       from kivy.network.urlrequest import UrlRequest
86       obj = UrlRequestQueue([])
87       queue = obj.queue
88       head = {
89           "Authorization": "Basic {}".format(b64encode(
90               "{}:{}".format('user', 'passwd').encode('utf-8')
91           ).decode('utf-8'))
92       }
93       req = UrlRequest(
94           'http://httpbin.org/basic-auth/user/passwd',
95           on_success=obj._on_success,
96           on_progress=obj._on_progress,
97           on_error=obj._on_error,
98           on_redirect=obj._on_redirect,
99           req_headers=head,
100          debug=True
101      )
102      wait_request_is_finished(kivy_clock, req)
103  
104      if req.error and req.error.errno == 11001:
105          pytest.skip('Cannot connect to get address')
106  
107      ensure_called_from_thread(queue)
108      check_queue_values(queue)
109      assert queue[-1][2] == ({'authenticated': True, 'user': 'user'}, )

138      """Passing a `ca_file` should not crash on http scheme, refs #6946"""

153  
154      if req.error and req.error.errno == 11001:
155          pytest.skip('Cannot connect to get address')
