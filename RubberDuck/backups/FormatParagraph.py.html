1    # Extension to format a paragraph
2    
3    # Does basic, standard text formatting, and also understands Python
4    # comment blocks.  Thus, for editing Python source code, this
5    # extension is really only suitable for reformatting these comment
6    # blocks or triple-quoted strings.
7    
8    # Known problems with comment reformatting:
9    # * If there is a selection marked, and the first line of the
10   #   selection is not complete, the block will probably not be detected
11   #   as comments, and will have the normal "text formatting" rules
12   #   applied.
13   # * If a comment block has leading whitespace that mixes tabs and
14   #   spaces, they will not be considered part of the same block.
15   # * Fancy comments, like this bulleted list, arent handled :-)
16

18   import re

41           self.editwin = editwin

44           self.editwin = None

46       def format_paragraph_event(self, event):
47           text = self.editwin.text
48           first, last = self.editwin.get_selection_indices()
49           if first and last:
50               data = text.get(first, last)
51               comment_header = ""
52           else:
53               first, last, comment_header, data = find_paragraph(
54                   text, text.index("insert")
55               )
56           if comment_header:
57               # Reformat the comment lines - convert to text sans header.
58               lines = data.split("\n")
59               lines = map(lambda st, l=len(comment_header): st[l:], lines)
60               data = "\n".join(lines)
61               # Reformat to 70 chars or a 20 char width, whichever is greater.
62               format_width = max(70 - len(comment_header), 20)
63               newdata = reformat_paragraph(data, format_width)
64               # re-split and re-insert the comment header.
65               newdata = newdata.split("\n")
66               # If the block ends in a \n, we dont want the comment
67               # prefix inserted after it. (Im not sure it makes sense to
68               # reformat a comment block that isnt made of complete
69               # lines, but whatever!)  Can't think of a clean soltution,
70               # so we hack away
71               block_suffix = ""
72               if not newdata[-1]:
73                   block_suffix = "\n"
74                   newdata = newdata[:-1]
75               builder = lambda item, prefix=comment_header: prefix + item
76               newdata = "\n".join([builder(d) for d in newdata]) + block_suffix
77           else:
78               # Just a normal text format
79               newdata = reformat_paragraph(data)
80           text.tag_remove("sel", "1.0", "end")
81           if newdata != data:
82               text.mark_set("insert", first)
83               text.undo_block_start()
84               text.delete(first, last)
85               text.insert(first, newdata)
86               text.undo_block_stop()
87           else:
88               text.mark_set("insert", last)
89           text.see("insert")

92   def find_paragraph(text, mark):
93       lineno, col = list(map(int, mark.split(".")))
94       line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
95       while text.compare("%d.0" % lineno, "<", "end") and is_all_white(line):
96           lineno = lineno + 1
97           line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
98       first_lineno = lineno
99       comment_header = get_comment_header(line)
100      comment_header_len = len(comment_header)
101      while get_comment_header(line) == comment_header and not is_all_white(
102          line[comment_header_len:]
103      ):
104          lineno = lineno + 1
105          line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
106      last = "%d.0" % lineno
107      # Search back to beginning of paragraph
108      lineno = first_lineno - 1
109      line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
110      while (
111          lineno > 0
112          and get_comment_header(line) == comment_header
113          and not is_all_white(line[comment_header_len:])
114      ):
115          lineno = lineno - 1
116          line = text.get("%d.0" % lineno, "%d.0 lineend" % lineno)
117      first = "%d.0" % (lineno + 1)
118      return first, last, comment_header, text.get(first, last)

121  def reformat_paragraph(data, limit=70):
122      lines = data.split("\n")
123      i = 0
124      n = len(lines)
125      while i < n and is_all_white(lines[i]):
126          i = i + 1
127      if i >= n:
128          return data
129      indent1 = get_indent(lines[i])
130      if i + 1 < n and not is_all_white(lines[i + 1]):
131          indent2 = get_indent(lines[i + 1])
132      else:
133          indent2 = indent1
134      new = lines[:i]
135      partial = indent1
136      while i < n and not is_all_white(lines[i]):
137          # XXX Should take double space after period (etc.) into account
138          words = re.split("(\s+)", lines[i])
139          for j in range(0, len(words), 2):
140              word = words[j]
141              if not word:
142                  continue  # Can happen when line ends in whitespace
143              if len((partial + word).expandtabs()) > limit and partial != indent1:
144                  new.append(partial.rstrip())
145                  partial = indent2
146              partial = partial + word + " "
147              if j + 1 < len(words) and words[j + 1] != " ":
148                  partial = partial + " "
149          i = i + 1
150      new.append(partial.rstrip())
151      # XXX Should reformat remaining paragraphs as well
152      new.extend(lines[i:])
153      return "\n".join(new)

156  def is_all_white(line):
157      return re.match(r"^\s*$", line) is not None
158  
159  
160  def get_indent(line):
161      return re.match(r"^(\s*)", line).group()
162  
163  
164  def get_comment_header(line):
165      m = re.match(r"^(\s*#*)", line)
166      if m is None:
167          return ""
168      return m.group(1)
