54   class SecurityInformation(win32com.server.policy.DesignatedWrapPolicy):
55       _com_interfaces_ = [authorization.IID_ISecurityInformation]
56       _public_methods_ = [
57           "GetObjectInformation",
58           "GetSecurity",
59           "SetSecurity",
60           "GetAccessRights",
61           "GetInheritTypes",
62           "MapGeneric",
63           "PropertySheetPageCallback",
64       ]
65   
66       def __init__(self, FileName):
67           self.FileName = FileName
68           self._wrap_(self)
69   
70       def GetObjectInformation(self):
71           """Identifies object whose security will be modified, and determines options available 
72           to the end user"""
73           flags = SI_ADVANCED | SI_EDIT_ALL | SI_PAGE_TITLE | SI_RESET
74           if os.path.isdir(self.FileName):
75               flags |= SI_CONTAINER
76           hinstance = 0  ## handle to module containing string resources
77           servername = ""  ## name of authenticating server if not local machine
78           objectname = os.path.split(self.FileName)[1]
79           pagetitle = "Python ACL Editor"
80           if os.path.isdir(self.FileName):
81               pagetitle += " (dir)"
82           else:
83               pagetitle += " (file)"
84           objecttype = IID_NULL
85           return flags, hinstance, servername, objectname, pagetitle, objecttype
86   
87       def GetSecurity(self, requestedinfo, bdefault):
88           """Requests the existing permissions for object"""
89           if bdefault:
90               ## This is invoked if the 'Default' button is pressed (only present if SI_RESET is passed
91               ## with the flags in GetObjectInfo). Passing an empty SD with a NULL Dacl
92               ##  should cause inherited ACL from parent dir or default dacl from user's token to be used
93               return win32security.SECURITY_DESCRIPTOR()
94           else:
95               ## GetFileSecurity sometimes fails to return flags indicating that an ACE is inherited
96               return win32security.GetNamedSecurityInfo(
97                   self.FileName, win32security.SE_FILE_OBJECT, requestedinfo
98               )
99   
100      def SetSecurity(self, requestedinfo, sd):
101          """Applies permissions to the object"""
102          owner = sd.GetSecurityDescriptorOwner()
103          group = sd.GetSecurityDescriptorGroup()
104          dacl = sd.GetSecurityDescriptorDacl()
105          sacl = sd.GetSecurityDescriptorSacl()
106          win32security.SetNamedSecurityInfo(
107              self.FileName,
108              win32security.SE_FILE_OBJECT,
109              requestedinfo,
110              owner,
111              group,
112              dacl,
113              sacl,
114          )
115          ## should also handle recursive operations here
116  
117      def GetAccessRights(self, objecttype, flags):
118          """Returns a tuple of (AccessRights, DefaultAccess), where AccessRights is a sequence of tuples representing 
119          SI_ACCESS structs, containing (guid, access mask, Name, flags). DefaultAccess indicates which of the 
120          AccessRights will be used initially when a new ACE is added (zero based). 
121          Flags can contain SI_ACCESS_SPECIFIC,SI_ACCESS_GENERAL,SI_ACCESS_CONTAINER,SI_ACCESS_PROPERTY, 
122                CONTAINER_INHERIT_ACE,INHERIT_ONLY_ACE,OBJECT_INHERIT_ACE 
123          """
124          ## input flags: SI_ADVANCED,SI_EDIT_AUDITS,SI_EDIT_PROPERTIES indicating which property sheet is requesting the rights
125          if (objecttype is not None) and (objecttype != IID_NULL):
126              ## Should not be true for file objects.  Usually only used with DS objects that support security for
127              ## their properties
128              raise NotImplementedError("Object type is not supported")
129  
130          if os.path.isdir(self.FileName):
131              file_append_data_desc = "Create subfolders"
132              file_write_data_desc = "Create Files"
133          else:
134              file_append_data_desc = "Append data"
135              file_write_data_desc = "Write data"
136  
137          accessrights = [
138              (
139                  IID_NULL,
140                  FILE_GENERIC_READ,
141                  "Generic read",
142                  SI_ACCESS_GENERAL
143                  | SI_ACCESS_SPECIFIC
144                  | OBJECT_INHERIT_ACE
145                  | CONTAINER_INHERIT_ACE,
146              ),
147              (
148                  IID_NULL,
149                  FILE_GENERIC_WRITE,
150                  "Generic write",
151                  SI_ACCESS_GENERAL
152                  | SI_ACCESS_SPECIFIC
153                  | OBJECT_INHERIT_ACE
154                  | CONTAINER_INHERIT_ACE,
155              ),
156              (
157                  IID_NULL,
158                  win32con.DELETE,
159                  "Delete",
160                  SI_ACCESS_SPECIFIC | OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
161              ),
162              (
163                  IID_NULL,
164                  WRITE_OWNER,
165                  "Change owner",
166                  SI_ACCESS_SPECIFIC | OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
167              ),
168              (
169                  IID_NULL,
170                  READ_CONTROL,
171                  "Read Permissions",
172                  SI_ACCESS_SPECIFIC | OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
173              ),
174              (
175                  IID_NULL,
176                  WRITE_DAC,
177                  "Change permissions",
178                  SI_ACCESS_SPECIFIC | OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
179              ),
180              (
181                  IID_NULL,
182                  FILE_APPEND_DATA,
183                  file_append_data_desc,
184                  SI_ACCESS_SPECIFIC | OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
185              ),
186              (
187                  IID_NULL,
188                  FILE_WRITE_DATA,
189                  file_write_data_desc,
190                  SI_ACCESS_SPECIFIC | OBJECT_INHERIT_ACE | CONTAINER_INHERIT_ACE,
191              ),
192          ]
193          return (accessrights, 0)
194  
195      def MapGeneric(self, guid, aceflags, mask):
196          """Converts generic access rights to specific rights.  This implementation uses standard file system rights, 
197          but you can map them any way that suits your application. 
198          """
199          return win32security.MapGenericMask(
200              mask,
201              (
202                  FILE_GENERIC_READ,
203                  FILE_GENERIC_WRITE,
204                  FILE_GENERIC_EXECUTE,
205                  FILE_ALL_ACCESS,
206              ),
207          )
208  
209      def GetInheritTypes(self):
210          """Specifies which types of ACE inheritance are supported. 
211          Returns a sequence of tuples representing SI_INHERIT_TYPE structs, containing 
212          (object type guid, inheritance flags, display name).  Guid is usually only used with 
213          Directory Service objects. 
214          """
215          return (
216              (IID_NULL, 0, "Only current object"),
217              (IID_NULL, OBJECT_INHERIT_ACE, "Files inherit permissions"),
218              (IID_NULL, CONTAINER_INHERIT_ACE, "Sub Folders inherit permissions"),
219              (
220                  IID_NULL,
221                  CONTAINER_INHERIT_ACE | OBJECT_INHERIT_ACE,
222                  "Files and subfolders",
223              ),
224          )
225  
226      def PropertySheetPageCallback(self, hwnd, msg, pagetype):
227          """Invoked each time a property sheet page is created or destroyed."""
228          ## page types from SI_PAGE_TYPE enum: SI_PAGE_PERM SI_PAGE_ADVPERM SI_PAGE_AUDIT SI_PAGE_OWNER
229          ## msg: PSPCB_CREATE, PSPCB_RELEASE, PSPCB_SI_INITDIALOG
230          return None

232      def EditSecurity(self, owner_hwnd=0):
233          """Creates an ACL editor dialog based on parameters returned by interface methods"""
234          isi = pythoncom.WrapObject(
235              self, authorization.IID_ISecurityInformation, pythoncom.IID_IUnknown
236          )
237          authorization.EditSecurity(owner_hwnd, isi)
