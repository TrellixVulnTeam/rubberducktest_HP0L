305  class PostScriptLexer(RegexLexer):
306      """ 
307      Lexer for PostScript files. 
308   
309      The PostScript Language Reference published by Adobe at 
310      <http://partners.adobe.com/public/developer/en/ps/PLRM.pdf> 
311      is the authority for this. 
312   
313      .. versionadded:: 1.4 
314      """
315      name = 'PostScript'
316      aliases = ['postscript', 'postscr']
317      filenames = ['*.ps', '*.eps']
318      mimetypes = ['application/postscript']
319  
320      delimiter = r'()<>\[\]{}/%\s'
321      delimiter_end = r'(?=[%s])' % delimiter
322  
323      valid_name_chars = r'[^%s]' % delimiter
324      valid_name = r"%s+%s" % (valid_name_chars, delimiter_end)
325  
326      tokens = {
327          'root': [
328              # All comment types
329              (r'^%!.+\n', Comment.Preproc),
330              (r'%%.*\n', Comment.Special),
331              (r'(^%.*\n){2,}', Comment.Multiline),
332              (r'%.*\n', Comment.Single),
333  
334              # String literals are awkward; enter separate state.
335              (r'\(', String, 'stringliteral'),
336  
337              (r'[{}<>\[\]]', Punctuation),
338  
339              # Numbers
340              (r'<[0-9A-Fa-f]+>' + delimiter_end, Number.Hex),
341              # Slight abuse: use Oct to signify any explicit base system
342              (r'[0-9]+\#(\-|\+)?([0-9]+\.?|[0-9]*\.[0-9]+|[0-9]+\.[0-9]*)'
343               r'((e|E)[0-9]+)?' + delimiter_end, Number.Oct),
344              (r'(\-|\+)?([0-9]+\.?|[0-9]*\.[0-9]+|[0-9]+\.[0-9]*)((e|E)[0-9]+)?'
345               + delimiter_end, Number.Float),
346              (r'(\-|\+)?[0-9]+' + delimiter_end, Number.Integer),
347  
348              # References
349              (r'\/%s' % valid_name, Name.Variable),
350  
351              # Names
352              (valid_name, Name.Function),      # Anything else is executed
353  
354              # These keywords taken from
355              # <http://www.math.ubc.ca/~cass/graphics/manual/pdf/a1.pdf>
356              # Is there an authoritative list anywhere that doesn't involve
357              # trawling documentation?
358  
359              (r'(false|true)' + delimiter_end, Keyword.Constant),
360  
361              # Conditionals / flow control
362              (r'(eq|ne|g[et]|l[et]|and|or|not|if(?:else)?|for(?:all)?)'
363               + delimiter_end, Keyword.Reserved),
364  
365              (words((
366                  'abs', 'add', 'aload', 'arc', 'arcn', 'array', 'atan', 'begin',
367                  'bind', 'ceiling', 'charpath', 'clip', 'closepath', 'concat',
368                  'concatmatrix', 'copy', 'cos', 'currentlinewidth', 'currentmatrix',
369                  'currentpoint', 'curveto', 'cvi', 'cvs', 'def', 'defaultmatrix',
370                  'dict', 'dictstackoverflow', 'div', 'dtransform', 'dup', 'end',
371                  'exch', 'exec', 'exit', 'exp', 'fill', 'findfont', 'floor', 'get',
372                  'getinterval', 'grestore', 'gsave', 'gt', 'identmatrix', 'idiv',
373                  'idtransform', 'index', 'invertmatrix', 'itransform', 'length',
374                  'lineto', 'ln', 'load', 'log', 'loop', 'matrix', 'mod', 'moveto',
375                  'mul', 'neg', 'newpath', 'pathforall', 'pathbbox', 'pop', 'print',
376                  'pstack', 'put', 'quit', 'rand', 'rangecheck', 'rcurveto', 'repeat',
377                  'restore', 'rlineto', 'rmoveto', 'roll', 'rotate', 'round', 'run',
378                  'save', 'scale', 'scalefont', 'setdash', 'setfont', 'setgray',
379                  'setlinecap', 'setlinejoin', 'setlinewidth', 'setmatrix',
380                  'setrgbcolor', 'shfill', 'show', 'showpage', 'sin', 'sqrt',
381                  'stack', 'stringwidth', 'stroke', 'strokepath', 'sub', 'syntaxerror',
382                  'transform', 'translate', 'truncate', 'typecheck', 'undefined',
383                  'undefinedfilename', 'undefinedresult'), suffix=delimiter_end),
384               Name.Builtin),
385  
386              (r'\s+', Text),
387          ],
388  
389          'stringliteral': [
390              (r'[^()\\]+', String),
391              (r'\\', String.Escape, 'escape'),
392              (r'\(', String, '#push'),
393              (r'\)', String, '#pop'),
394          ],

396          'escape': [
397              (r'[0-8]{3}|n|r|t|b|f|\\|\(|\)', String.Escape, '#pop'),
398              default('#pop'),
399          ],
400      }

403  class AsymptoteLexer(RegexLexer):
404      """ 
405      For `Asymptote <http://asymptote.sf.net/>`_ source code. 
406   
407      .. versionadded:: 1.2 
408      """
409      name = 'Asymptote'
410      aliases = ['asymptote', 'asy']
411      filenames = ['*.asy']
412      mimetypes = ['text/x-asymptote']
413  
414      #: optional Comment or Whitespace
415      _ws = r'(?:\s|//.*?\n|/\*.*?\*/)+'
416  
417      tokens = {
418          'whitespace': [
419              (r'\n', Text),
420              (r'\s+', Text),
421              (r'\\\n', Text),  # line continuation
422              (r'//(\n|(.|\n)*?[^\\]\n)', Comment),
423              (r'/(\\\n)?\*(.|\n)*?\*(\\\n)?/', Comment),
424          ],
425          'statements': [
426              # simple string (TeX friendly)
427              (r'"(\\\\|\\[^\\]|[^"\\])*"', String),
428              # C style string (with character escapes)
429              (r"'", String, 'string'),
430              (r'(\d+\.\d*|\.\d+|\d+)[eE][+-]?\d+[lL]?', Number.Float),
431              (r'(\d+\.\d*|\.\d+|\d+[fF])[fF]?', Number.Float),
432              (r'0x[0-9a-fA-F]+[Ll]?', Number.Hex),
433              (r'0[0-7]+[Ll]?', Number.Oct),
434              (r'\d+[Ll]?', Number.Integer),
435              (r'[~!%^&*+=|?:<>/-]', Operator),
436              (r'[()\[\],.]', Punctuation),
437              (r'\b(case)(.+?)(:)', bygroups(Keyword, using(this), Text)),
438              (r'(and|controls|tension|atleast|curl|if|else|while|for|do|'
439               r'return|break|continue|struct|typedef|new|access|import|'
440               r'unravel|from|include|quote|static|public|private|restricted|'
441               r'this|explicit|true|false|null|cycle|newframe|operator)\b', Keyword),
442              # Since an asy-type-name can be also an asy-function-name,
443              # in the following we test if the string "  [a-zA-Z]" follows
444              # the Keyword.Type.
445              # Of course it is not perfect !
446              (r'(Braid|FitResult|Label|Legend|TreeNode|abscissa|arc|arrowhead|'
447               r'binarytree|binarytreeNode|block|bool|bool3|bounds|bqe|circle|'
448               r'conic|coord|coordsys|cputime|ellipse|file|filltype|frame|grid3|'
449               r'guide|horner|hsv|hyperbola|indexedTransform|int|inversion|key|'
450               r'light|line|linefit|marginT|marker|mass|object|pair|parabola|path|'
451               r'path3|pen|picture|point|position|projection|real|revolution|'
452               r'scaleT|scientific|segment|side|slice|splitface|string|surface|'
453               r'tensionSpecifier|ticklocate|ticksgridT|tickvalues|transform|'
454               r'transformation|tree|triangle|trilinear|triple|vector|'
455               r'vertex|void)(?=\s+[a-zA-Z])', Keyword.Type),
456              # Now the asy-type-name which are not asy-function-name
457              # except yours !
458              # Perhaps useless
459              (r'(Braid|FitResult|TreeNode|abscissa|arrowhead|block|bool|bool3|'
460               r'bounds|coord|frame|guide|horner|int|linefit|marginT|pair|pen|'
461               r'picture|position|real|revolution|slice|splitface|ticksgridT|'
462               r'tickvalues|tree|triple|vertex|void)\b', Keyword.Type),
463              (r'[a-zA-Z_]\w*:(?!:)', Name.Label),
464              (r'[a-zA-Z_]\w*', Name),
465          ],
466          'root': [
467              include('whitespace'),
468              # functions
469              (r'((?:[\w*\s])+?(?:\s|\*))'  # return arguments
470               r'([a-zA-Z_]\w*)'            # method name
471               r'(\s*\([^;]*?\))'           # signature
472               r'(' + _ws + r')(\{)',
473               bygroups(using(this), Name.Function, using(this), using(this),
474                        Punctuation),

509      def get_tokens_unprocessed(self, text):
510          from pygments.lexers._asy_builtins import ASYFUNCNAME, ASYVARNAME
511          for index, token, value in \
512                  RegexLexer.get_tokens_unprocessed(self, text):
513              if token is Name and value in ASYFUNCNAME:
514                  token = Name.Function
515              elif token is Name and value in ASYVARNAME:
516                  token = Name.Variable
517              yield index, token, value

520  def _shortened(word):
521      dpos = word.find('$')
522      return '|'.join(word[:dpos] + word[dpos+1:i] + r'\b'
523                      for i in range(len(word), dpos, -1))
524  
525  
526  def _shortened_many(*words):
527      return '|'.join(map(_shortened, words))

782      def analyse_text(text):
783          """POVRAY is similar to JSON/C, but the combination of camera and 
784          light_source is probably not very likely elsewhere. HLSL or GLSL 
785          are similar (GLSL even has #version), but they miss #declare, and 
786          light_source/camera are not keywords anywhere else -- it's fair 
787          to assume though that any POVRAY scene must have a camera and 
788          lightsource."""
789          result = 0
790          if '#version' in text:
791              result += 0.05
792          if '#declare' in text:
793              result += 0.05
794          if 'camera' in text:
795              result += 0.05
796          if 'light_source' in text:
797              result += 0.1
798  
799          return result
