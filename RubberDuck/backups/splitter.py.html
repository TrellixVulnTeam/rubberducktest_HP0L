1    '''Splitter 
2    ====== 
3     
4    .. versionadded:: 1.5.0 
5     
6    .. image:: images/splitter.jpg 
7        :align: right 
8     
9    The :class:`Splitter` is a widget that helps you re-size its child 
10   widget/layout by letting you re-size it via dragging the boundary or 
11   double tapping the boundary. This widget is similar to the 
12   :class:`~kivy.uix.scrollview.ScrollView` in that it allows only one 
13   child widget. 
14    
15   Usage:: 
16    
17       splitter = Splitter(sizable_from = 'right') 
18       splitter.add_widget(layout_or_widget_instance) 
19       splitter.min_size = 100 
20       splitter.max_size = 250 
21    
22   To change the size of the strip/border used for resizing:: 
23    
24       splitter.strip_size = '10pt' 
25    
26   To change its appearance:: 
27    
28       splitter.strip_cls = your_custom_class 
29    
30   You can also change the appearance of the `strip_cls`, which defaults to 
31   :class:`SplitterStrip`, by overriding the `kv` rule in your app: 
32    
33   .. code-block:: kv 
34    
35       <SplitterStrip>: 
36           horizontal: True if self.parent and self.parent.sizable_from[0] \ 
37   in ('t', 'b') else False 
38           background_normal: 'path to normal horizontal image' \ 
39   if self.horizontal else 'path to vertical normal image' 
40           background_down: 'path to pressed horizontal image' \ 
41   if self.horizontal else 'path to vertical pressed image' 
42    
43   '''

63   class Splitter(BoxLayout):
64       '''See module documentation. 
65    
66       :Events: 
67           `on_press`: 
68               Fired when the splitter is pressed. 
69           `on_release`: 
70               Fired when the splitter is released. 
71    
72       .. versionchanged:: 1.6.0 
73           Added `on_press` and `on_release` events. 
74

77       border = ListProperty([4, 4, 4, 4])
78       '''Border used for the 
79       :class:`~kivy.graphics.vertex_instructions.BorderImage` 
80       graphics instruction. 
81    
82       This must be a list of four values: (bottom, right, top, left). 
83       Read the BorderImage instructions for more information about how 
84       to use it. 
85    
86       :attr:`border` is a :class:`~kivy.properties.ListProperty` and 
87       defaults to (4, 4, 4, 4). 
88       '''
89   
90       strip_cls = ObjectProperty(SplitterStrip)
91       '''Specifies the class of the resize Strip. 
92    
93       :attr:`strip_cls` is an :class:`kivy.properties.ObjectProperty` and 
94       defaults to :class:`~kivy.uix.splitter.SplitterStrip`, which is of type 
95       :class:`~kivy.uix.button.Button`. 
96    
97       .. versionchanged:: 1.8.0 
98           If you set a string, the :class:`~kivy.factory.Factory` will be used to 
99           resolve the class. 
100   
101      '''
102  
103      sizable_from = OptionProperty('left', options=(
104          'left', 'right', 'top', 'bottom'))
105      '''Specifies whether the widget is resizable. Options are: 
106      `left`, `right`, `top` or `bottom` 
107   
108      :attr:`sizable_from` is an :class:`~kivy.properties.OptionProperty` 
109      and defaults to `left`. 
110      '''
111  
112      strip_size = NumericProperty('10pt')
113      '''Specifies the size of resize strip 
114   
115      :attr:`strp_size` is a :class:`~kivy.properties.NumericProperty` 
116      defaults to `10pt` 
117      '''
118  
119      min_size = NumericProperty('100pt')
120      '''Specifies the minimum size beyond which the widget is not resizable. 
121   
122      :attr:`min_size` is a :class:`~kivy.properties.NumericProperty` and 
123      defaults to `100pt`. 
124      '''
125  
126      max_size = NumericProperty('500pt')
127      '''Specifies the maximum size beyond which the widget is not resizable. 
128   
129      :attr:`max_size` is a :class:`~kivy.properties.NumericProperty` 
130      and defaults to `500pt`. 
131      '''
132  
133      _parent_proportion = NumericProperty(0.)
134      '''(internal) Specifies the distance that the slider has travelled 
135      across its parent, used to automatically maintain a sensible 
136      position if the parent is resized. 
137   
138      :attr:`_parent_proportion` is a 
139      :class:`~kivy.properties.NumericProperty` and defaults to 0. 
140   
141      .. versionadded:: 1.9.0 
142      '''
143  
144      _bound_parent = ObjectProperty(None, allownone=True)
145      '''(internal) References the widget whose size is currently being 
146      tracked by :attr:`_parent_proportion`. 
147   
148      :attr:`_bound_parent` is a 
149      :class:`~kivy.properties.ObjectProperty` and defaults to None. 
150   
151      .. versionadded:: 1.9.0 
152      '''
153  
154      keep_within_parent = BooleanProperty(False)
155      '''If True, will limit the splitter to stay within its parent widget. 
156   
157      :attr:`keep_within_parent` is a 
158      :class:`~kivy.properties.BooleanProperty` and defaults to False. 
159   
160      .. versionadded:: 1.9.0 
161      '''
162  
163      rescale_with_parent = BooleanProperty(False)
164      '''If True, will automatically change size to take up the same 
165      proportion of the parent widget when it is resized, while 
166      staying within :attr:`min_size` and :attr:`max_size`. As long as 
167      these attributes can be satisfied, this stops the 
168      :class:`Splitter` from exceeding the parent size during rescaling. 
169   
170      :attr:`rescale_with_parent` is a 
171      :class:`~kivy.properties.BooleanProperty` and defaults to False. 
172   
173      .. versionadded:: 1.9.0 
174      '''

176      __events__ = ('on_press', 'on_release')

178      def __init__(self, **kwargs):
179          self._container = None
180          self._strip = None
181          super(Splitter, self).__init__(**kwargs)
182  
183          do_size = self._do_size
184          fbind = self.fbind
185          fbind('max_size', do_size)
186          fbind('min_size', do_size)
187          fbind('parent', self._rebind_parent)
188  
189      def on_sizable_from(self, instance, sizable_from):
190          if not instance._container:
191              return
192  
193          sup = super(Splitter, instance)
194          _strp = instance._strip
195          if _strp:
196              # remove any previous binds
197              _strp.unbind(on_touch_down=instance.strip_down)
198              _strp.unbind(on_touch_move=instance.strip_move)
199              _strp.unbind(on_touch_up=instance.strip_up)
200              self.unbind(disabled=_strp.setter('disabled'))
201  
202              sup.remove_widget(instance._strip)
203          else:
204              cls = instance.strip_cls
205              if isinstance(cls, string_types):
206                  cls = Factory.get(cls)
207              instance._strip = _strp = cls()
208  
209          sz_frm = instance.sizable_from[0]
210          if sz_frm in ('l', 'r'):
211              _strp.size_hint = None, 1
212              _strp.width = instance.strip_size
213              instance.orientation = 'horizontal'
214              instance.unbind(strip_size=_strp.setter('width'))
215              instance.bind(strip_size=_strp.setter('width'))
216          else:
217              _strp.size_hint = 1, None
218              _strp.height = instance.strip_size
219              instance.orientation = 'vertical'
220              instance.unbind(strip_size=_strp.setter('height'))
221              instance.bind(strip_size=_strp.setter('height'))
222  
223          index = 1
224          if sz_frm in ('r', 'b'):
225              index = 0
226          sup.add_widget(_strp, index)
227  
228          _strp.bind(on_touch_down=instance.strip_down)
229          _strp.bind(on_touch_move=instance.strip_move)
230          _strp.bind(on_touch_up=instance.strip_up)
231          _strp.disabled = self.disabled
232          self.bind(disabled=_strp.setter('disabled'))
233  
234      def add_widget(self, widget, index=0):
235          if self._container or not widget:
236              return Exception('Splitter accepts only one Child')
237          self._container = widget
238          sz_frm = self.sizable_from[0]
239          if sz_frm in ('l', 'r'):
240              widget.size_hint_x = 1
241          else:
242              widget.size_hint_y = 1
243  
244          index = 0
245          if sz_frm in ('r', 'b'):
246              index = 1
247          super(Splitter, self).add_widget(widget, index)
248          self.on_sizable_from(self, self.sizable_from)
249  
250      def remove_widget(self, widget, *largs):
251          super(Splitter, self).remove_widget(widget)
252          if widget == self._container:
253              self._container = None
254  
255      def clear_widgets(self):
256          self.remove_widget(self._container)
257  
258      def strip_down(self, instance, touch):
259          if not instance.collide_point(*touch.pos):
260              return False
261          touch.grab(self)
262          self.dispatch('on_press')
263  
264      def on_press(self):
265          pass
266  
267      def _rebind_parent(self, instance, new_parent):
268          if self._bound_parent is not None:
269              self._bound_parent.unbind(size=self.rescale_parent_proportion)
270          if self.parent is not None:
271              new_parent.bind(size=self.rescale_parent_proportion)
272          self._bound_parent = new_parent
273          self.rescale_parent_proportion()
274  
275      def rescale_parent_proportion(self, *args):
276          if not self.parent:
277              return
278          if self.rescale_with_parent:
279              parent_proportion = self._parent_proportion
280              if self.sizable_from in ('top', 'bottom'):
281                  new_height = parent_proportion * self.parent.height
282                  self.height = max(self.min_size,
283                                   min(new_height, self.max_size))
284              else:
285                  new_width = parent_proportion * self.parent.width
286                  self.width = max(self.min_size, min(new_width, self.max_size))
287  
288      def _do_size(self, instance, value):
289          if self.sizable_from[0] in ('l', 'r'):
290              self.width = max(self.min_size, min(self.width, self.max_size))
291          else:
292              self.height = max(self.min_size, min(self.height, self.max_size))
293  
294      @staticmethod
295      def _is_moving(sz_frm, diff, pos, minpos, maxpos):
296          if sz_frm in ('l', 'b'):
297              cmp = minpos
298          else:
299              cmp = maxpos
300          if diff == 0:
301              return False
302          elif diff > 0 and pos <= cmp:
303              return False
304          elif diff < 0 and pos >= cmp:
305              return False
306          return True
307  
308      def strip_move(self, instance, touch):
309          if touch.grab_current is not instance:
310              return False
311          max_size = self.max_size
312          min_size = self.min_size
313          sz_frm = self.sizable_from[0]
314  
315          if sz_frm in ('t', 'b'):
316              diff_y = (touch.dy)
317              self_y = self.y
318              self_top = self.top
319              if not self._is_moving(sz_frm, diff_y, touch.y, self_y, self_top):
320                  return
321              if self.keep_within_parent:
322                  if sz_frm == 't' and (self_top + diff_y) > self.parent.top:
323                      diff_y = self.parent.top - self_top
324                  elif sz_frm == 'b' and (self_y + diff_y) < self.parent.y:
325                      diff_y = self.parent.y - self_y
326              if sz_frm == 'b':
327                  diff_y *= -1
328              if self.size_hint_y:
329                  self.size_hint_y = None
330              if self.height > 0:
331                  self.height += diff_y
332              else:
333                  self.height = 1
334  
335              height = self.height
336              self.height = max(min_size, min(height, max_size))
337  
338              self._parent_proportion = self.height / self.parent.height
339          else:
340              diff_x = (touch.dx)
341              self_x = self.x
342              self_right = self.right
343              if not self._is_moving(sz_frm, diff_x, touch.x, self_x, self_right):
344                  return
345              if self.keep_within_parent:
346                  if sz_frm == 'l' and (self_x + diff_x) < self.parent.x:
347                      diff_x = self.parent.x - self_x
348                  elif (sz_frm == 'r' and
349                        (self_right + diff_x) > self.parent.right):
350                      diff_x = self.parent.right - self_right
351              if sz_frm == 'l':
352                  diff_x *= -1
353              if self.size_hint_x:
354                  self.size_hint_x = None
355              if self.width > 0:
356                  self.width += diff_x
357              else:
358                  self.width = 1
359  
360              width = self.width
361              self.width = max(min_size, min(width, max_size))
362  
363              self._parent_proportion = self.width / self.parent.width
364  
365      def strip_up(self, instance, touch):
366          if touch.grab_current is not instance:
367              return
368  
369          if touch.is_double_tap:
370              max_size = self.max_size
371              min_size = self.min_size
372              sz_frm = self.sizable_from[0]
373              s = self.size
374  
375              if sz_frm in ('t', 'b'):
376                  if self.size_hint_y:
377                      self.size_hint_y = None
378                  if s[1] - min_size <= max_size - s[1]:
379                      self.height = max_size
380                  else:
381                      self.height = min_size
382              else:
383                  if self.size_hint_x:
384                      self.size_hint_x = None
385                  if s[0] - min_size <= max_size - s[0]:
386                      self.width = max_size
387                  else:
388                      self.width = min_size
389          touch.ungrab(instance)
390          self.dispatch('on_release')
391  
392      def on_release(self):
393          pass
