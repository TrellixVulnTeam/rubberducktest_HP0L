fetch_url.py 
1    """ 
2    Fetches a URL from a web-server supporting NTLM authentication 
3    eg, IIS. 
4     
5    If no arguments are specified, a default of http://localhost/localstart.asp 
6    is used.  This script does follow simple 302 redirections, so pointing at the 
7    root of an IIS server is should work. 
8    """
9    
10   import sys
11   import urllib.request, urllib.parse, urllib.error
12   import http.client
13   import urllib.parse
14   from base64 import encodestring, decodestring
15   
16   from sspi import ClientAuth
17   
18   import optparse  # sorry, this demo needs 2.3+
19   
20   options = None  # set to optparse options object
21   
22   
23   def open_url(host, url):
24       h = http.client.HTTPConnection(host)
25       #    h.set_debuglevel(9)
26       h.putrequest("GET", url)
27       h.endheaders()
28       resp = h.getresponse()
29       print("Initial response is", resp.status, resp.reason)
30       body = resp.read()
31       if resp.status == 302:  # object moved
32           url = "/" + resp.msg["location"]
33           resp.close()
34           h.putrequest("GET", url)
35           h.endheaders()
36           resp = h.getresponse()
37           print("After redirect response is", resp.status, resp.reason)
38       if options.show_headers:
39           print("Initial response headers:")
40           for name, val in list(resp.msg.items()):
41               print(" %s: %s" % (name, val))
42       if options.show_body:
43           print(body)
44       if resp.status == 401:
45           # 401: Unauthorized - here is where the real work starts
46           auth_info = None
47           if options.user or options.domain or options.password:
48               auth_info = options.user, options.domain, options.password
49           ca = ClientAuth("NTLM", auth_info=auth_info)
50           auth_scheme = ca.pkg_info["Name"]
51           data = None
52           while 1:
53               err, out_buf = ca.authorize(data)
54               data = out_buf[0].Buffer
55               # Encode it as base64 as required by HTTP
56               auth = encodestring(data).replace("\012", "")
57               h.putrequest("GET", url)
58               h.putheader("Authorization", auth_scheme + " " + auth)
59               h.putheader("Content-Length", "0")
60               h.endheaders()
61               resp = h.getresponse()
62               if options.show_headers:
63                   print("Token dance headers:")
64                   for name, val in list(resp.msg.items()):
65                       print(" %s: %s" % (name, val))
66   
67               if err == 0:
68                   break
69               else:
70                   if resp.status != 401:
71                       print("Eeek - got response", resp.status)
72                       cl = resp.msg.get("content-length")
73                       if cl:
74                           print(repr(resp.read(int(cl))))
75                       else:
76                           print("no content!")
77   
78                   assert resp.status == 401, resp.status
79   
80               assert not resp.will_close, "NTLM is per-connection - must not close"
81               schemes = [
82                   s.strip() for s in resp.msg.get("WWW-Authenticate", "").split(",")
83               ]
84               for scheme in schemes:
85                   if scheme.startswith(auth_scheme):
86                       data = decodestring(scheme[len(auth_scheme) + 1 :])
87                       break
88               else:
89                   print(
90                       "Could not find scheme '%s' in schemes %r" % (auth_scheme, schemes)
91                   )
92                   break
93   
94               resp.read()
95       print("Final response status is", resp.status, resp.reason)
96       if resp.status == 200:
97           # Worked!
98           # Check we can read it again without re-authenticating.
99           if resp.will_close:
100              print(
101                  "EEEK - response will close, but NTLM is per connection - it must stay open"
102              )
103          body = resp.read()
104          if options.show_body:
105              print("Final response body:")
106              print(body)
107          h.putrequest("GET", url)
108          h.endheaders()
109          resp = h.getresponse()
110          print("Second fetch response is", resp.status, resp.reason)
111          if options.show_headers:
112              print("Second response headers:")
113              for name, val in list(resp.msg.items()):
114                  print(" %s: %s" % (name, val))
115  
116          resp.read(int(resp.msg.get("content-length", 0)))
117      elif resp.status == 500:
118          print("Error text")
119          print(resp.read())
120      else:
121          if options.show_body:
122              cl = resp.msg.get("content-length")
123              print(resp.read(int(cl)))
124  
125  
126  if __name__ == "__main__":
127      parser = optparse.OptionParser(description=__doc__)
128  
129      parser.add_option(
130          "",
131          "--show-body",
132          action="store_true",
133          help="print the body of each response as it is received",
134      )
135  
136      parser.add_option(
137          "",
138          "--show-headers",
139          action="store_true",
140          help="print the headers of each response as it is received",
141      )
142  
143      parser.add_option("", "--user", action="store", help="The username to login with")
144  
145      parser.add_option(
146          "", "--password", action="store", help="The password to login with"
147      )
148  
149      parser.add_option("", "--domain", action="store", help="The domain to login to")
150  
151      options, args = parser.parse_args()
152      if not args:
153          print("Run with --help for usage details")
154          args = ["http://localhost/localstart.asp"]
155      for url in args:
156          scheme, netloc, path, params, query, fragment = urllib.parse.urlparse(url)
157          if (scheme != "http") or params or query or fragment:
158              parser.error("Scheme must be http, URL must be simple")
159  
160          print("Opening '%s' from '%s'" % (path, netloc))
161          r = open_url(netloc, path)
