1    # Demonstrates how to validate a password.
2    # See also MSKB article Q180548
3    #
4    # To use with Kerberos you need to jump through the 'targetspn' hoops.
5    
6    import win32security
7    import sys
8    from sspi import ClientAuth, ServerAuth
9    
10   
11   def validate(username, password, domain=""):
12       auth_info = username, domain, password
13       ca = ClientAuth("NTLM", auth_info=auth_info)
14       sa = ServerAuth("NTLM")
15   
16       data = err = None
17       while err != 0:
18           err, data = ca.authorize(data)
19           err, data = sa.authorize(data)
20       # If we get here without exception, we worked!
21   
22   
23   if __name__ == "__main__":
24       if len(sys.argv) not in [2, 3, 4]:
25           print("Usage: %s username [password [domain]]" % (__file__,))
26           sys.exit(1)
27   
28       # password and domain are optional!
29       password = None
30       if len(sys.argv) >= 3:
31           password = sys.argv[2]
32       domain = ""
33       if len(sys.argv) >= 4:
34           domain = sys.argv[3]
35       try:
36           validate(sys.argv[1], password, domain)
37           print("Validated OK")
38       except win32security.error as details:
39           hr, func, msg = details
40           print("Validation failed: %s (%d)" % (msg, hr))
