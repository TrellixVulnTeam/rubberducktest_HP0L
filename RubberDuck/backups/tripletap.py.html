1    ''' 
2    Triple Tap 
3    ========== 
4     
5    .. versionadded:: 1.7.0 
6     
7    Search touch for a triple tap 
8    '''
9    
10   __all__ = ('InputPostprocTripleTap', )
11   
12   from time import time
13   from kivy.config import Config
14   from kivy.vector import Vector
15   
16   
17   class InputPostprocTripleTap(object):
18       ''' 
19       InputPostProcTripleTap is a post-processor to check if 
20       a touch is a triple tap or not. 
21       Triple tap can be configured in the Kivy config file:: 
22    
23           [postproc] 
24           triple_tap_time = 250 
25           triple_tap_distance = 20 
26    
27       The distance parameter is in the range 0-1000 and time is in milliseconds. 
28       '''
29   
30       def __init__(self):
31           dist = Config.getint('postproc', 'triple_tap_distance')
32           self.triple_tap_distance = dist / 1000.0
33           time = Config.getint('postproc', 'triple_tap_time')
34           self.triple_tap_time = time / 1000.0
35           self.touches = {}
36   
37       def find_triple_tap(self, ref):
38           '''Find a triple tap touch within *self.touches*. 
39           The touch must be not be a previous triple tap and the distance 
40           must be be within the bounds specified. Additionally, the touch profile 
41           must be the same kind of touch. 
42           '''
43           ref_button = None
44           if 'button' in ref.profile:
45               ref_button = ref.button
46   
47           for touchid in self.touches:
48               if ref.uid == touchid:
49                   continue
50               etype, touch = self.touches[touchid]
51               if not touch.is_double_tap:
52                   continue
53               if etype != 'end':
54                   continue
55               if touch.is_triple_tap:
56                   continue
57               distance = Vector.distance(
58                   Vector(ref.sx, ref.sy),
59                   Vector(touch.osx, touch.osy))
60               if distance > self.triple_tap_distance:
61                   continue
62               if touch.is_mouse_scrolling or ref.is_mouse_scrolling:
63                   continue
64               touch_button = None
65               if 'button' in touch.profile:
66                   touch_button = touch.button
67               if touch_button != ref_button:
68                   continue
69               touch.triple_tap_distance = distance
70               return touch
71   
72       def process(self, events):
73           if self.triple_tap_distance == 0 or self.triple_tap_time == 0:
74               return events
75           # first, check if a touch down have a triple tap
76           for etype, touch in events:
77               if not touch.is_touch:
78                   continue
79               if etype == 'begin':
80                   triple_tap = self.find_triple_tap(touch)
81                   if triple_tap:
82                       touch.is_double_tap = False
83                       touch.is_triple_tap = True
84                       tap_time = touch.time_start - triple_tap.time_start
85                       touch.triple_tap_time = tap_time
86                       distance = triple_tap.triple_tap_distance
87                       touch.triple_tap_distance = distance
88   
89               # add the touch internally
90               self.touches[touch.uid] = (etype, touch)
91   
92           # second, check if up-touch is timeout for triple tap
93           time_current = time()
94           to_delete = []
95           for touchid in self.touches.keys():
96               etype, touch = self.touches[touchid]
97               if etype != 'end':
98                   continue
99               if time_current - touch.time_start < self.triple_tap_time:
100                  continue
101              to_delete.append(touchid)
102  
103          for touchid in to_delete:
104              del self.touches[touchid]
105  
106          return events
