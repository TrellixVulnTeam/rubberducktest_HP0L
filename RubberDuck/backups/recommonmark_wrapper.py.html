15   """ 
16   A parser for CommonMark MarkDown text using `recommonmark`__. 
17    
18   __ https://pypi.org/project/recommonmark/ 
19   """
20   
21   import docutils.parsers
22   from docutils import nodes, Component
23   
24   try:
25       from recommonmark.parser import CommonMarkParser
26   except ImportError as err:
27       CommonMarkParser = None
28       class Parser(docutils.parsers.Parser):
29           def parse(self, inputstring, document):
30               error = document.reporter.warning(
31                   'Missing dependency: MarkDown input is processed by a 3rd '
32                   'party parser but Python did not find the required module '
33                   '"recommonmark" (https://pypi.org/project/recommonmark/).')
34               document.append(error)
35   
36   # recommonmark 0.5.0 introduced a hard dependency on Sphinx
37   # https://github.com/readthedocs/recommonmark/issues/202
38   # There is a PR to change this to an optional dependency
39   # https://github.com/readthedocs/recommonmark/pull/218
40   try:
41       from sphinx import addnodes
42   except ImportError:
43       # create a stub
44       class addnodes(nodes.pending): pass
45   
46   
47   if CommonMarkParser:
48       class Parser(CommonMarkParser):
49           """MarkDown parser based on recommonmark."""
50           
51           supported = ('recommonmark', 'commonmark', 'markdown', 'md')
52           config_section = 'recommonmark parser'
53           config_section_dependencies = ('parsers',)
54   
55           def get_transforms(self):
56               return Component.get_transforms(self) # + [AutoStructify]
57   
58           def parse(self, inputstring, document):
59               """Use the upstream parser and clean up afterwards. 
60               """
61               # check for exorbitantly long lines
62               for i, line in enumerate(inputstring.split('\n')):
63                   if len(line) > document.settings.line_length_limit:
64                       error = document.reporter.error(
65                           'Line %d exceeds the line-length-limit.'%(i+1))
66                       document.append(error)
67                       return
68   
69               # pass to upstream parser
70               try:
71                   CommonMarkParser.parse(self, inputstring, document)
72               except Exception as err:
73                   error = document.reporter.error('Parsing with "recommonmark" '
74                                                   'returned the error:\n%s'%err)
75                   document.append(error)
76   
77               # Post-Processing
78               # ---------------
79   
80               # merge adjoining Text nodes:
81               for node in document.traverse(nodes.TextElement):
82                   children = node.children
83                   i = 0
84                   while i+1 < len(children):
85                       if (isinstance(children[i], nodes.Text)
86                           and isinstance(children[i+1], nodes.Text)):
87                           children[i] = nodes.Text(children[i]+children.pop(i+1))
88                           children[i].parent = node
89                       else:
90                           i += 1
91   
92               # add "code" class argument to inline literal (code spans)
93               for node in document.traverse(lambda n: isinstance(n,
94                                       (nodes.literal, nodes.literal_block))):
95                   node['classes'].append('code')
96               # move "language" argument to classes
97               for node in document.traverse(nodes.literal_block):
98                   if 'language' in node.attributes:
99                       node['classes'].append(node['language'])
100                      del node['language']
101  
102              # remove empty target nodes
103              for node in list(document.traverse(nodes.target)):
104                  # remove empty name
105                  node['names'] = [v for v in node['names'] if v]
106                  if node.children or [v for v in node.attributes.values() if v]:
107                      continue
108                  node.parent.remove(node)
109  
110              # replace raw nodes if raw is not allowed
111              if not document.settings.raw_enabled:
112                  for node in document.traverse(nodes.raw):
113                      warning = document.reporter.warning('Raw content disabled.')
114                      node.parent.replace(node, warning)
115  
116              # fix section nodes
117              for node in document.traverse(nodes.section):
118                  # remove spurious IDs (first may be from duplicate name)
119                  if len(node['ids']) > 1:
120                      node['ids'].pop()
121                  # fix section levels (recommonmark 0.4.0
122                  # later versions silently ignore incompatible levels)
123                  if 'level' in node:
124                      section_level = self.get_section_level(node)
125                      if node['level'] != section_level:
126                          warning = document.reporter.warning(
127                              'Title level inconsistent. Changing from %d to %d.'
128                              %(node['level'], section_level),
129                              nodes.literal_block('', node[0].astext()))
130                          node.insert(1, warning)
131                          # remove non-standard attribute "level"
132                          del node['level']
133      
134              # drop pending_xref (Sphinx cross reference extension)
135              for node in document.traverse(addnodes.pending_xref):
136                  reference = node.children[0]
137                  if 'name' not in reference:
138                      reference['name'] = nodes.fully_normalize_name(
139                                                          reference.astext())
140                  node.parent.replace(node, reference)
141  
142          def get_section_level(self, node):
143              """Auxiliary function for post-processing in self.parse()"""
144              level = 1
145              while True:
146                  node = node.parent
147                  if isinstance(node, nodes.document):
148                      return level
149                  if isinstance(node, nodes.section):
150                      level += 1
151  
152          def visit_document(self, node):
153              """Dummy function to prevent spurious warnings. 
154   
155              cf. https://github.com/readthedocs/recommonmark/issues/177 
156              """
157              pass
