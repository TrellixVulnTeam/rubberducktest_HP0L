1    """Exception Handling 
2     
3     Exceptions 
4     
5         To better support COM exceptions, the framework allows for an instance to be 
6         raised.  This instance may have a certain number of known attributes, which are 
7         translated into COM exception details. 
8         
9         This means, for example, that Python could raise a COM exception that includes details 
10        on a Help file and location, and a description for the user. 
11        
12        This module provides a class which provides the necessary attributes. 
13    
14   """
15   import sys, pythoncom
16   
17   # Note that we derive from com_error, which derives from exceptions.Exception
18   # Also note that we dont support "self.args", as we dont support tuple-unpacking
19   class COMException(pythoncom.com_error):
20       """An Exception object that is understood by the framework. 
21    
22       If the framework is presented with an exception of type class, 
23       it looks for certain known attributes on this class to provide rich 
24       error information to the caller. 
25    
26       It should be noted that the framework supports providing this error 
27       information via COM Exceptions, or via the ISupportErrorInfo interface. 
28    
29       By using this class, you automatically provide rich error information to the 
30       server. 
31       """
32   
33       def __init__(
34           self,
35           description=None,
36           scode=None,
37           source=None,
38           helpfile=None,
39           helpContext=None,
40           desc=None,
41           hresult=None,
42       ):
43           """Initialize an exception 
44           **Params** 
45    
46           description -- A string description for the exception. 
47           scode -- An integer scode to be returned to the server, if necessary. 
48           The pythoncom framework defaults this to be DISP_E_EXCEPTION if not specified otherwise. 
49           source -- A string which identifies the source of the error. 
50           helpfile -- A string which points to a help file which contains details on the error. 
51           helpContext -- An integer context in the help file. 
52           desc -- A short-cut for description. 
53           hresult -- A short-cut for scode. 
54           """
55   
56           # convert a WIN32 error into an HRESULT
57           scode = scode or hresult
58           if scode and scode != 1:  # We dont want S_FALSE mapped!
59               if scode >= -32768 and scode < 32768:
60                   # this is HRESULT_FROM_WIN32()
61                   scode = -2147024896 | (scode & 0x0000FFFF)
62           self.scode = scode
63   
64           self.description = description or desc
65           if scode == 1 and not self.description:
66               self.description = "S_FALSE"
67           elif scode and not self.description:
68               self.description = pythoncom.GetScodeString(scode)
69   
70           self.source = source
71           self.helpfile = helpfile
72           self.helpcontext = helpContext
73   
74           # todo - fill in the exception value
75           pythoncom.com_error.__init__(self, scode, self.description, None, -1)
76   
77       def __repr__(self):
78           return "<COM Exception - scode=%s, desc=%s>" % (self.scode, self.description)

87   def IsCOMException(t=None):
88       if t is None:
89           t = sys.exc_info()[0]
90       try:
91           return issubclass(t, pythoncom.com_error)
92       except TypeError:  # 1.5 in -X mode?
93           return t is pythoncon.com_error

96   def IsCOMServerException(t=None):
97       if t is None:
98           t = sys.exc_info()[0]
99       try:
100          return issubclass(t, COMException)
101      except TypeError:  # String exception
102          return 0
