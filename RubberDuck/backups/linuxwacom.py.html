1    ''' 
2    Native support of Wacom tablet from linuxwacom driver 
3    ===================================================== 
4     
5    To configure LinuxWacom, add this to your configuration:: 
6     
7        [input] 
8        pen = linuxwacom,/dev/input/event2,mode=pen 
9        finger = linuxwacom,/dev/input/event3,mode=touch 
10    
11   .. note:: 
12       You must have read access to the input event. 
13    
14   You can use a custom range for the X, Y and pressure values. 
15   On some drivers, the range reported is invalid. 
16   To fix that, you can add these options to the argument line: 
17    
18   * invert_x : 1 to invert X axis 
19   * invert_y : 1 to invert Y axis 
20   * min_position_x : X minimum 
21   * max_position_x : X maximum 
22   * min_position_y : Y minimum 
23   * max_position_y : Y maximum 
24   * min_pressure : pressure minimum 
25   * max_pressure : pressure maximum 
26   '''
27

35   class LinuxWacomMotionEvent(MotionEvent):
36   
37       def depack(self, args):
38           self.is_touch = True
39           self.sx = args['x']
40           self.sy = args['y']
41           self.profile = ['pos']
42           if 'size_w' in args and 'size_h' in args:
43               self.shape = ShapeRect()
44               self.shape.width = args['size_w']
45               self.shape.height = args['size_h']
46               self.profile.append('shape')
47           if 'pressure' in args:
48               self.pressure = args['pressure']
49               self.profile.append('pressure')
50           super(LinuxWacomMotionEvent, self).depack(args)
51   
52       def __str__(self):
53           return '<LinuxWacomMotionEvent id=%d pos=(%f, %f) device=%s>' \
54               % (self.id, self.sx, self.sy, self.device)

74       # Event types
75       EV_SYN = 0x00
76       EV_KEY = 0x01
77       EV_REL = 0x02
78       EV_ABS = 0x03
79       EV_MSC = 0x04
80       EV_SW = 0x05
81       EV_LED = 0x11
82       EV_SND = 0x12
83       EV_REP = 0x14
84       EV_FF = 0x15
85       EV_PWR = 0x16
86       EV_FF_STATUS = 0x17
87       EV_MAX = 0x1f
88       EV_CNT = (EV_MAX + 1)
89   
90       KEY_MAX = 0x2ff
91   
92       # Synchronization events
93       SYN_REPORT = 0
94       SYN_CONFIG = 1
95       SYN_MT_REPORT = 2
96   
97       # Misc events
98       MSC_SERIAL = 0x00
99       MSC_PULSELED = 0x01
100      MSC_GESTURE = 0x02
101      MSC_RAW = 0x03
102      MSC_SCAN = 0x04
103      MSC_MAX = 0x07
104      MSC_CNT = (MSC_MAX + 1)
105  
106      ABS_X = 0x00
107      ABS_Y = 0x01
108      ABS_PRESSURE = 0x18
109      ABS_MISC = 0x28  # if 0, it's touch up
110      ABS_MT_TOUCH_MAJOR = 0x30  # Major axis of touching ellipse
111      ABS_MT_TOUCH_MINOR = 0x31  # Minor axis (omit if circular)
112      ABS_MT_WIDTH_MAJOR = 0x32  # Major axis of approaching ellipse
113      ABS_MT_WIDTH_MINOR = 0x33  # Minor axis (omit if circular)
114      ABS_MT_ORIENTATION = 0x34  # Ellipse orientation
115      ABS_MT_POSITION_X = 0x35   # Center X ellipse position
116      ABS_MT_POSITION_Y = 0x36   # Center Y ellipse position
117      ABS_MT_TOOL_TYPE = 0x37    # Type of touching device
118      ABS_MT_BLOB_ID = 0x38      # Group a set of packets as a blob
119      ABS_MT_TRACKING_ID = 0x39  # Unique ID of initiated contact
120      ABS_MT_PRESSURE = 0x3a     # Pressure on contact area
121  
122      # some ioctl base (with 0 value)
123      EVIOCGNAME = 2147501318
124      EVIOCGBIT = 2147501344
125      EVIOCGABS = 2149074240
126  
127      # sizeof(struct input_event)
128      struct_input_event_sz = struct.calcsize('LLHHi')
129      struct_input_absinfo_sz = struct.calcsize('iiiiii')
130      sz_l = struct.calcsize('Q')
131  
132      class LinuxWacomMotionEventProvider(MotionEventProvider):
133  
134          options = ('min_position_x', 'max_position_x',
135                     'min_position_y', 'max_position_y',
136                     'min_pressure', 'max_pressure',
137                     'invert_x', 'invert_y')
138  
139          def __init__(self, device, args):
140              super(LinuxWacomMotionEventProvider, self).__init__(device, args)
141              self.input_fn = None
142              self.default_ranges = dict()
143              self.mode = 'touch'
144  
145              # split arguments
146              args = args.split(',')
147              if not args:
148                  Logger.error('LinuxWacom: No filename given in config')
149                  Logger.error('LinuxWacom: Use /dev/input/event0 for example')
150                  return
151  
152              # read filename
153              self.input_fn = args[0]
154              Logger.info('LinuxWacom: Read event from <%s>' % self.input_fn)
155  
156              # read parameters
157              for arg in args[1:]:
158                  if arg == '':
159                      continue
160                  arg = arg.split('=')
161  
162                  # ensure it's a key = value
163                  if len(arg) != 2:
164                      err = 'LinuxWacom: Bad parameter' \
165                          '%s: Not in key=value format.' % arg
166                      Logger.error(err)
167                      continue
168  
169                  # ensure the key exist
170                  key, value = arg
171                  if key == 'mode':
172                      self.mode = value
173                      continue
174  
175                  if key not in LinuxWacomMotionEventProvider.options:
176                      Logger.error('LinuxWacom: unknown %s option' % key)
177                      continue
178  
179                  # ensure the value
180                  try:
181                      self.default_ranges[key] = int(value)
182                  except ValueError:
183                      err = 'LinuxWacom: value %s invalid for %s' % (key, value)
184                      Logger.error(err)
185                      continue
186  
187                  # all good!
188                  msg = 'LinuxWacom: Set custom %s to %d' % (key, int(value))
189                  Logger.info(msg)
190              Logger.info('LinuxWacom: mode is <%s>' % self.mode)
191  
192          def start(self):
193              if self.input_fn is None:
194                  return
195              self.uid = 0
196              self.queue = collections.deque()
197              self.thread = threading.Thread(
198                  target=self._thread_run,
199                  kwargs=dict(
200                      queue=self.queue,
201                      input_fn=self.input_fn,
202                      device=self.device,
203                      default_ranges=self.default_ranges))
204              self.thread.daemon = True
205              self.thread.start()
206  
207          def _thread_run(self, **kwargs):
208              input_fn = kwargs.get('input_fn')
209              queue = kwargs.get('queue')
210              device = kwargs.get('device')
211              drs = kwargs.get('default_ranges').get
212              touches = {}
213              touches_sent = []
214              l_points = {}
215  
216              # prepare some vars to get limit of some component
217              range_min_position_x = 0
218              range_max_position_x = 2048
219              range_min_position_y = 0
220              range_max_position_y = 2048
221              range_min_pressure = 0
222              range_max_pressure = 255
223              invert_x = int(bool(drs('invert_x', 0)))
224              invert_y = int(bool(drs('invert_y', 0)))
225              reset_touch = False
226  
227              def process(points):
228                  actives = list(points.keys())
229                  for args in points.values():
230                      tid = args['id']
231                      try:
232                          touch = touches[tid]
233                      except KeyError:
234                          touch = LinuxWacomMotionEvent(device, tid, args)
235                          touches[touch.id] = touch
236                      if touch.sx == args['x'] \
237                              and touch.sy == args['y'] \
238                              and tid in touches_sent:
239                          continue
240                      touch.move(args)
241                      if tid not in touches_sent:
242                          queue.append(('begin', touch))
243                          touches_sent.append(tid)
244                      queue.append(('update', touch))
245  
246                  for tid in list(touches.keys())[:]:
247                      if tid not in actives:
248                          touch = touches[tid]
249                          if tid in touches_sent:
250                              touch.update_time_end()
251                              queue.append(('end', touch))
252                              touches_sent.remove(tid)
253                          del touches[tid]
254  
255              def normalize(value, vmin, vmax):
256                  return (value - vmin) / float(vmax - vmin)
257  
258              # open the input
259              try:
260                  fd = open(input_fn, 'rb')
261              except IOError:
262                  Logger.exception('Unable to open %s' % input_fn)
263                  return
264  
265              # get the controler name (EVIOCGNAME)
266              device_name = fcntl.ioctl(fd, EVIOCGNAME + (256 << 16),
267                                        " " * 256).split('\x00')[0]
268              Logger.info('LinuxWacom: using <%s>' % device_name)
269  
270              # get abs infos
271              bit = fcntl.ioctl(fd, EVIOCGBIT + (EV_MAX << 16), ' ' * sz_l)
272              bit, = struct.unpack('Q', bit)
273              for x in range(EV_MAX):
274                  # preserve this, we may want other things than EV_ABS
275                  if x != EV_ABS:
276                      continue
277                  # EV_ABS available for this device ?
278                  if (bit & (1 << x)) == 0:
279                      continue
280                  # ask abs info keys to the devices
281                  sbit = fcntl.ioctl(fd, EVIOCGBIT + x + (KEY_MAX << 16),
282                                     ' ' * sz_l)
283                  sbit, = struct.unpack('Q', sbit)
284                  for y in range(KEY_MAX):
285                      if (sbit & (1 << y)) == 0:
286                          continue
287                      absinfo = fcntl.ioctl(fd, EVIOCGABS + y +
288                                            (struct_input_absinfo_sz << 16),
289                                            ' ' * struct_input_absinfo_sz)
290                      abs_value, abs_min, abs_max, abs_fuzz, \
291                          abs_flat, abs_res = struct.unpack('iiiiii', absinfo)
292                      if y == ABS_X:
293                          range_min_position_x = drs('min_position_x', abs_min)
294                          range_max_position_x = drs('max_position_x', abs_max)
295                          Logger.info('LinuxWacom: ' +
296                                      '<%s> range position X is %d - %d' % (
297                                          device_name, abs_min, abs_max))
298                      elif y == ABS_Y:
299                          range_min_position_y = drs('min_position_y', abs_min)
300                          range_max_position_y = drs('max_position_y', abs_max)
301                          Logger.info('LinuxWacom: ' +
302                                      '<%s> range position Y is %d - %d' % (
303                                          device_name, abs_min, abs_max))
304                      elif y == ABS_PRESSURE:
305                          range_min_pressure = drs('min_pressure', abs_min)
306                          range_max_pressure = drs('max_pressure', abs_max)
307                          Logger.info('LinuxWacom: ' +
308                                      '<%s> range pressure is %d - %d' % (
309                                          device_name, abs_min, abs_max))
310  
311              # read until the end
312              changed = False
313              touch_id = 0
314              touch_x = 0
315              touch_y = 0
316              touch_pressure = 0
317              while fd:
318  
319                  data = fd.read(struct_input_event_sz)
320                  if len(data) < struct_input_event_sz:
321                      break
322  
323                  # extract each event
324                  for i in range(len(data) / struct_input_event_sz):
325                      ev = data[i * struct_input_event_sz:]
326  
327                      # extract timeval + event infos
328                      tv_sec, tv_usec, ev_type, ev_code, ev_value = \
329                          struct.unpack('LLHHi', ev[:struct_input_event_sz])
330  
331                      if ev_type == EV_SYN and ev_code == SYN_REPORT:
332                          if touch_id in l_points:
333                              p = l_points[touch_id]
334                          else:
335                              p = dict()
336                              l_points[touch_id] = p
337                          p['id'] = touch_id
338                          if not reset_touch:
339                              p['x'] = touch_x
340                              p['y'] = touch_y
341                              p['pressure'] = touch_pressure
342                          if self.mode == 'pen' \
343                                  and touch_pressure == 0 \
344                                  and not reset_touch:
345                              del l_points[touch_id]
346                          if changed:
347                              if 'x' not in p:
348                                  reset_touch = False
349                                  continue
350                              process(l_points)
351                              changed = False
352                          if reset_touch:
353                              l_points.clear()
354                              reset_touch = False
355                              process(l_points)
356                      elif ev_type == EV_MSC and ev_code == MSC_SERIAL:
357                          touch_id = ev_value
358                      elif ev_type == EV_ABS and ev_code == ABS_X:
359                          val = normalize(ev_value,
360                                          range_min_position_x,
361                                          range_max_position_x)
362                          if invert_x:
363                              val = 1. - val
364                          touch_x = val
365                          changed = True
366                      elif ev_type == EV_ABS and ev_code == ABS_Y:
367                          val = 1. - normalize(ev_value,
368                                               range_min_position_y,
369                                               range_max_position_y)
370                          if invert_y:
371                              val = 1. - val
372                          touch_y = val
373                          changed = True
374                      elif ev_type == EV_ABS and ev_code == ABS_PRESSURE:
375                          touch_pressure = normalize(ev_value,
376                                                     range_min_pressure,
377                                                     range_max_pressure)
378                          changed = True
379                      elif ev_type == EV_ABS and ev_code == ABS_MISC:
380                          if ev_value == 0:
381                              reset_touch = True
382  
383          def update(self, dispatch_fn):
384              # dispatch all event from threads
385              try:
386                  while True:
387                      event_type, touch = self.queue.popleft()
388                      dispatch_fn(event_type, touch)
389              except:
390                  pass
