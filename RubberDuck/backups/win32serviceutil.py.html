15   def LocatePythonServiceExe(exeName=None):
16       if not exeName and hasattr(sys, "frozen"):
17           # If py2exe etc calls this with no exeName, default is current exe.
18           return sys.executable
19   
20       # Try and find the specified EXE somewhere.  If specifically registered,
21       # use it.  Otherwise look down sys.path, and the global PATH environment.
22       if exeName is None:
23           if os.path.splitext(win32service.__file__)[0].endswith("_d"):
24               exeName = "PythonService_d.exe"
25           else:
26               exeName = "PythonService.exe"
27       # See if it exists as specified
28       if os.path.isfile(exeName):
29           return win32api.GetFullPathName(exeName)
30       baseName = os.path.splitext(os.path.basename(exeName))[0]
31       try:
32           exeName = win32api.RegQueryValue(
33               win32con.HKEY_LOCAL_MACHINE,
34               "Software\\Python\\%s\\%s" % (baseName, sys.winver),
35           )
36           if os.path.isfile(exeName):
37               return exeName
38           raise RuntimeError(
39               "The executable '%s' is registered as the Python "
40               "service exe, but it does not exist as specified" % exeName
41           )
42       except win32api.error:
43           # OK - not there - lets go a-searchin'
44           for path in [sys.prefix] + sys.path:
45               look = os.path.join(path, exeName)
46               if os.path.isfile(look):
47                   return win32api.GetFullPathName(look)
48           # Try the global Path.
49           try:
50               return win32api.SearchPath(None, exeName)[0]
51           except win32api.error:
52               msg = (
53                   "%s is not correctly registered\nPlease locate and run %s, and it will self-register\nThen run this service registration process again."
54                   % (exeName, exeName)
55               )
56               raise error(msg)

59   def _GetServiceShortName(longName):
60       # looks up a services name
61       # from the display name
62       # Thanks to Andy McKay for this code.
63       access = (
64           win32con.KEY_READ | win32con.KEY_ENUMERATE_SUB_KEYS | win32con.KEY_QUERY_VALUE
65       )
66       hkey = win32api.RegOpenKey(
67           win32con.HKEY_LOCAL_MACHINE, "SYSTEM\\CurrentControlSet\\Services", 0, access
68       )
69       num = win32api.RegQueryInfoKey(hkey)[0]
70       longName = longName.lower()
71       # loop through number of subkeys
72       for x in range(0, num):
73           # find service name, open subkey
74           svc = win32api.RegEnumKey(hkey, x)
75           skey = win32api.RegOpenKey(hkey, svc, 0, access)
76           try:
77               # find display name
78               thisName = str(win32api.RegQueryValueEx(skey, "DisplayName")[0])
79               if thisName.lower() == longName:
80                   return svc
81           except win32api.error:
82               # in case there is no key called DisplayName
83               pass
84       return None

88   def SmartOpenService(hscm, name, access):
89       try:
90           return win32service.OpenService(hscm, name, access)
91       except win32api.error as details:
92           if details.winerror not in [
93               winerror.ERROR_SERVICE_DOES_NOT_EXIST,
94               winerror.ERROR_INVALID_NAME,
95           ]:
96               raise
97       name = win32service.GetServiceKeyName(hscm, name)
98       return win32service.OpenService(hscm, name, access)

101  def LocateSpecificServiceExe(serviceName):
102      # Given the name of a specific service, return the .EXE name _it_ uses
103      # (which may or may not be the Python Service EXE
104      hkey = win32api.RegOpenKey(
105          win32con.HKEY_LOCAL_MACHINE,
106          "SYSTEM\\CurrentControlSet\\Services\\%s" % (serviceName),
107          0,
108          win32con.KEY_ALL_ACCESS,
109      )
110      try:
111          return win32api.RegQueryValueEx(hkey, "ImagePath")[0]
112      finally:
113          hkey.Close()

116  def InstallPerfmonForService(serviceName, iniName, dllName=None):
117      # If no DLL name, look it up in the INI file name
118      if not dllName:  # May be empty string!
119          dllName = win32api.GetProfileVal("Python", "dll", "", iniName)
120      # Still not found - look for the standard one in the same dir as win32service.pyd
121      if not dllName:
122          try:
123              tryName = os.path.join(
124                  os.path.split(win32service.__file__)[0], "perfmondata.dll"
125              )
126              if os.path.isfile(tryName):
127                  dllName = tryName
128          except AttributeError:
129              # Frozen app? - anyway, can't find it!
130              pass
131      if not dllName:
132          raise ValueError("The name of the performance DLL must be available")
133      dllName = win32api.GetFullPathName(dllName)
134      # Now setup all the required "Performance" entries.
135      hkey = win32api.RegOpenKey(
136          win32con.HKEY_LOCAL_MACHINE,
137          "SYSTEM\\CurrentControlSet\\Services\\%s" % (serviceName),
138          0,
139          win32con.KEY_ALL_ACCESS,
140      )
141      try:
142          subKey = win32api.RegCreateKey(hkey, "Performance")
143          try:
144              win32api.RegSetValueEx(subKey, "Library", 0, win32con.REG_SZ, dllName)
145              win32api.RegSetValueEx(
146                  subKey, "Open", 0, win32con.REG_SZ, "OpenPerformanceData"
147              )
148              win32api.RegSetValueEx(
149                  subKey, "Close", 0, win32con.REG_SZ, "ClosePerformanceData"
150              )
151              win32api.RegSetValueEx(
152                  subKey, "Collect", 0, win32con.REG_SZ, "CollectPerformanceData"
153              )
154          finally:
155              win32api.RegCloseKey(subKey)
156      finally:
157          win32api.RegCloseKey(hkey)
158      # Now do the "Lodctr" thang...
159  
160      try:
161          import perfmon
162  
163          path, fname = os.path.split(iniName)
164          oldPath = os.getcwd()
165          if path:
166              os.chdir(path)
167          try:
168              perfmon.LoadPerfCounterTextStrings("python.exe " + fname)
169          finally:
170              os.chdir(oldPath)
171      except win32api.error as details:
172          print("The service was installed OK, but the performance monitor")
173          print("data could not be loaded.", details)

183  def InstallService(
184      pythonClassString,
185      serviceName,
186      displayName,
187      startType=None,
188      errorControl=None,
189      bRunInteractive=0,
190      serviceDeps=None,
191      userName=None,
192      password=None,
193      exeName=None,
194      perfMonIni=None,
195      perfMonDll=None,
196      exeArgs=None,
197      description=None,
198      delayedstart=None,
199  ):
200      # Handle the default arguments.
201      if startType is None:
202          startType = win32service.SERVICE_DEMAND_START
203      serviceType = win32service.SERVICE_WIN32_OWN_PROCESS
204      if bRunInteractive:
205          serviceType = serviceType | win32service.SERVICE_INTERACTIVE_PROCESS
206      if errorControl is None:
207          errorControl = win32service.SERVICE_ERROR_NORMAL
208  
209      exeName = '"%s"' % LocatePythonServiceExe(
210          exeName
211      )  # None here means use default PythonService.exe
212      commandLine = _GetCommandLine(exeName, exeArgs)
213      hscm = win32service.OpenSCManager(None, None, win32service.SC_MANAGER_ALL_ACCESS)
214      try:
215          hs = win32service.CreateService(
216              hscm,
217              serviceName,
218              displayName,
219              win32service.SERVICE_ALL_ACCESS,  # desired access
220              serviceType,  # service type
221              startType,
222              errorControl,  # error control type
223              commandLine,
224              None,
225              0,
226              serviceDeps,
227              userName,
228              password,
229          )
230          if description is not None:
231              try:
232                  win32service.ChangeServiceConfig2(
233                      hs, win32service.SERVICE_CONFIG_DESCRIPTION, description
234                  )
235              except NotImplementedError:
236                  pass  ## ChangeServiceConfig2 and description do not exist on NT
237          if delayedstart is not None:
238              try:
239                  win32service.ChangeServiceConfig2(
240                      hs,
241                      win32service.SERVICE_CONFIG_DELAYED_AUTO_START_INFO,
242                      delayedstart,
243                  )
244              except (win32service.error, NotImplementedError):
245                  ## delayed start only exists on Vista and later - warn only when trying to set delayed to True
246                  if delayedstart:
247                      warnings.warn("Delayed Start not available on this system")
248          win32service.CloseServiceHandle(hs)
249      finally:
250          win32service.CloseServiceHandle(hscm)
251      InstallPythonClassString(pythonClassString, serviceName)
252      # If I have performance monitor info to install, do that.
253      if perfMonIni is not None:
254          InstallPerfmonForService(serviceName, perfMonIni, perfMonDll)

257  def ChangeServiceConfig(
258      pythonClassString,
259      serviceName,
260      startType=None,
261      errorControl=None,
262      bRunInteractive=0,
263      serviceDeps=None,
264      userName=None,
265      password=None,
266      exeName=None,
267      displayName=None,
268      perfMonIni=None,
269      perfMonDll=None,
270      exeArgs=None,
271      description=None,
272      delayedstart=None,
273  ):
274      # Before doing anything, remove any perfmon counters.
275      try:
276          import perfmon
277  
278          perfmon.UnloadPerfCounterTextStrings("python.exe " + serviceName)
279      except (ImportError, win32api.error):
280          pass
281  
282      # The EXE location may have changed
283      exeName = '"%s"' % LocatePythonServiceExe(exeName)
284  
285      # Handle the default arguments.
286      if startType is None:
287          startType = win32service.SERVICE_NO_CHANGE
288      if errorControl is None:
289          errorControl = win32service.SERVICE_NO_CHANGE
290  
291      hscm = win32service.OpenSCManager(None, None, win32service.SC_MANAGER_ALL_ACCESS)
292      serviceType = win32service.SERVICE_WIN32_OWN_PROCESS
293      if bRunInteractive:
294          serviceType = serviceType | win32service.SERVICE_INTERACTIVE_PROCESS
295      commandLine = _GetCommandLine(exeName, exeArgs)
296      try:
297          hs = SmartOpenService(hscm, serviceName, win32service.SERVICE_ALL_ACCESS)
298          try:
299  
300              win32service.ChangeServiceConfig(
301                  hs,
302                  serviceType,  # service type
303                  startType,
304                  errorControl,  # error control type
305                  commandLine,
306                  None,
307                  0,
308                  serviceDeps,
309                  userName,
310                  password,
311                  displayName,
312              )
313              if description is not None:
314                  try:
315                      win32service.ChangeServiceConfig2(
316                          hs, win32service.SERVICE_CONFIG_DESCRIPTION, description
317                      )
318                  except NotImplementedError:
319                      pass  ## ChangeServiceConfig2 and description do not exist on NT
320              if delayedstart is not None:
321                  try:
322                      win32service.ChangeServiceConfig2(
323                          hs,
324                          win32service.SERVICE_CONFIG_DELAYED_AUTO_START_INFO,
325                          delayedstart,
326                      )
327                  except (win32service.error, NotImplementedError):
328                      ## Delayed start only exists on Vista and later.  On Nt, will raise NotImplementedError since ChangeServiceConfig2
329                      ## doensn't exist.  On Win2k and XP, will fail with ERROR_INVALID_LEVEL
330                      ## Warn only if trying to set delayed to True
331                      if delayedstart:
332                          warnings.warn("Delayed Start not available on this system")
333          finally:
334              win32service.CloseServiceHandle(hs)
335      finally:
336          win32service.CloseServiceHandle(hscm)
337      InstallPythonClassString(pythonClassString, serviceName)
338      # If I have performance monitor info to install, do that.
339      if perfMonIni is not None:
340          InstallPerfmonForService(serviceName, perfMonIni, perfMonDll)

357  def SetServiceCustomOption(serviceName, option, value):
358      try:
359          serviceName = serviceName._svc_name_
360      except AttributeError:
361          pass
362      key = win32api.RegCreateKey(
363          win32con.HKEY_LOCAL_MACHINE,
364          "System\\CurrentControlSet\\Services\\%s\\Parameters" % serviceName,
365      )
366      try:
367          if type(value) == type(0):
368              win32api.RegSetValueEx(key, option, 0, win32con.REG_DWORD, value)
369          else:
370              win32api.RegSetValueEx(key, option, 0, win32con.REG_SZ, value)
371      finally:
372          win32api.RegCloseKey(key)

376      # First param may also be a service class/instance.
377      # This allows services to pass "self"
378      try:

384          "System\\CurrentControlSet\\Services\\%s\\Parameters" % serviceName,

387          try:

392          win32api.RegCloseKey(key)

433  def __FindSvcDeps(findName):
434      if type(findName) is pywintypes.UnicodeType:
435          findName = str(findName)
436      dict = {}
437      k = win32api.RegOpenKey(
438          win32con.HKEY_LOCAL_MACHINE, "SYSTEM\\CurrentControlSet\\Services"
439      )
440      num = 0
441      while 1:
442          try:
443              svc = win32api.RegEnumKey(k, num)
444          except win32api.error:
445              break
446          num = num + 1
447          sk = win32api.RegOpenKey(k, svc)
448          try:
449              deps, typ = win32api.RegQueryValueEx(sk, "DependOnService")
450          except win32api.error:
451              deps = ()
452          for dep in deps:
453              dep = dep.lower()
454              dep_on = dict.get(dep, [])
455              dep_on.append(svc)
456              dict[dep] = dep_on
457  
458      return __ResolveDeps(findName, dict)

461  def __ResolveDeps(findName, dict):
462      items = dict.get(findName.lower(), [])
463      retList = []
464      for svc in items:
465          retList.insert(0, svc)
466          retList = __ResolveDeps(svc, dict) + retList
467      return retList

470  def WaitForServiceStatus(serviceName, status, waitSecs, machine=None):
471      """Waits for the service to return the specified status.  You 
472      should have already requested the service to enter that state"""
473      for i in range(waitSecs * 4):
474          now_status = QueryServiceStatus(serviceName, machine)[1]
475          if now_status == status:
476              break
477          win32api.Sleep(250)
478      else:
479          raise pywintypes.error(
480              winerror.ERROR_SERVICE_REQUEST_TIMEOUT,
481              "QueryServiceStatus",
482              win32api.FormatMessage(winerror.ERROR_SERVICE_REQUEST_TIMEOUT)[:-2],
483          )

486  def __StopServiceWithTimeout(hs, waitSecs=30):
487      try:
488          status = win32service.ControlService(hs, win32service.SERVICE_CONTROL_STOP)
489      except pywintypes.error as exc:
490          if exc.winerror != winerror.ERROR_SERVICE_NOT_ACTIVE:
491              raise
492      for i in range(waitSecs):
493          status = win32service.QueryServiceStatus(hs)
494          if status[1] == win32service.SERVICE_STOPPED:
495              break
496          win32api.Sleep(1000)
497      else:
498          raise pywintypes.error(
499              winerror.ERROR_SERVICE_REQUEST_TIMEOUT,
500              "ControlService",
501              win32api.FormatMessage(winerror.ERROR_SERVICE_REQUEST_TIMEOUT)[:-2],
502          )

505  def StopServiceWithDeps(serviceName, machine=None, waitSecs=30):
506      # Stop a service recursively looking for dependant services
507      hscm = win32service.OpenSCManager(machine, None, win32service.SC_MANAGER_ALL_ACCESS)
508      try:
509          deps = __FindSvcDeps(serviceName)
510          for dep in deps:
511              hs = win32service.OpenService(hscm, dep, win32service.SERVICE_ALL_ACCESS)
512              try:
513                  __StopServiceWithTimeout(hs, waitSecs)
514              finally:
515                  win32service.CloseServiceHandle(hs)
516          # Now my service!
517          hs = win32service.OpenService(
518              hscm, serviceName, win32service.SERVICE_ALL_ACCESS
519          )
520          try:
521              __StopServiceWithTimeout(hs, waitSecs)
522          finally:
523              win32service.CloseServiceHandle(hs)
524  
525      finally:
526          win32service.CloseServiceHandle(hscm)

546  def RestartService(serviceName, args=None, waitSeconds=30, machine=None):
547      "Stop the service, and then start it again (with some tolerance for allowing it to stop.)"
548      try:
549          StopService(serviceName, machine)
550      except pywintypes.error as exc:
551          # Allow only "service not running" error
552          if exc.winerror != winerror.ERROR_SERVICE_NOT_ACTIVE:
553              raise
554      # Give it a few goes, as the service may take time to stop
555      for i in range(waitSeconds):
556          try:
557              StartService(serviceName, args, machine)
558              break
559          except pywintypes.error as exc:
560              if exc.winerror != winerror.ERROR_SERVICE_ALREADY_RUNNING:
561                  raise
562              win32api.Sleep(1000)
563      else:
564          print("Gave up waiting for the old service to stop!")

576  def DebugService(cls, argv=[]):
577      # Run a service in "debug" mode.  Re-implements what pythonservice.exe
578      # does when it sees a "-debug" param.
579      # Currently only used by "frozen" (ie, py2exe) programs (but later may
580      # end up being used for all services should we ever remove
581      # pythonservice.exe)
582      import servicemanager
583  
584      global g_debugService
585  
586      print("Debugging service %s - press Ctrl+C to stop." % (cls._svc_name_,))
587      servicemanager.Debugging(True)
588      servicemanager.PrepareToHostSingle(cls)
589      g_debugService = cls(argv)
590      # Setup a ctrl+c handler to simulate a "stop"
591      win32api.SetConsoleCtrlHandler(_DebugCtrlHandler, True)
592      try:
593          g_debugService.SvcRun()
594      finally:
595          win32api.SetConsoleCtrlHandler(_DebugCtrlHandler, False)
596          servicemanager.Debugging(False)
597          g_debugService = None

600  def GetServiceClassString(cls, argv=None):
601      if argv is None:
602          argv = sys.argv
603      import pickle
604  
605      modName = pickle.whichmodule(cls, cls.__name__)
606      if modName == "__main__":
607          try:
608              fname = win32api.GetFullPathName(argv[0])
609              path = os.path.split(fname)[0]
610              # Eaaaahhhh - sometimes this will be a short filename, which causes
611              # problems with 1.5.1 and the silly filename case rule.
612              filelist = win32api.FindFiles(fname)
613              # win32api.FindFiles will not detect files in a zip or exe. If list is empty,
614              # skip the test and hope the file really exists.
615              if len(filelist) != 0:
616                  # Get the long name
617                  fname = os.path.join(path, filelist[0][8])
618          except win32api.error:
619              raise error(
620                  "Could not resolve the path name '%s' to a full path" % (argv[0])
621              )
622          modName = os.path.splitext(fname)[0]
623      return modName + "." + cls.__name__

627      hscm = win32service.OpenSCManager(machine, None, win32service.SC_MANAGER_CONNECT)
628      try:
629  
630          hs = SmartOpenService(hscm, serviceName, win32service.SERVICE_QUERY_STATUS)
631          try:
632              status = win32service.QueryServiceStatus(hs)
633          finally:
634              win32service.CloseServiceHandle(hs)
635      finally:

637      return status

640  def usage():
641      try:
642          fname = os.path.split(sys.argv[0])[1]
643      except:
644          fname = sys.argv[0]
645      print(
646          "Usage: '%s [options] install|update|remove|start [...]|stop|restart [...]|debug [...]'"
647          % fname
648      )
649      print("Options for 'install' and 'update' commands only:")
650      print(" --username domain\\username : The Username the service is to run under")
651      print(" --password password : The password for the username")
652      print(
653          " --startup [manual|auto|disabled|delayed] : How the service starts, default = manual"
654      )
655      print(" --interactive : Allow the service to interact with the desktop.")
656      print(
657          " --perfmonini file: .ini file to use for registering performance monitor data"
658      )
659      print(" --perfmondll file: .dll file to use when querying the service for")
660      print("   performance data, default = perfmondata.dll")

663      print("                 If you specify --wait with the 'stop' option, the service")
664      print("                 and all dependent services will be stopped, each waiting")
665      print("                 the specified period.")
666      sys.exit(1)

669  def HandleCommandLine(
670      cls,
671      serviceClassString=None,
672      argv=None,
673      customInstallOptions="",
674      customOptionHandler=None,
675  ):
676      """Utility function allowing services to process the command line. 
677   
678      Allows standard commands such as 'start', 'stop', 'debug', 'install' etc. 
679   
680      Install supports 'standard' command line options prefixed with '--', such as 
681      --username, --password, etc.  In addition, 
682      the function allows custom command line options to be handled by the calling function. 
683      """
684      err = 0
685  
686      if argv is None:
687          argv = sys.argv
688  
689      if len(argv) <= 1:
690          usage()
691  
692      serviceName = cls._svc_name_
693      serviceDisplayName = cls._svc_display_name_
694      if serviceClassString is None:
695          serviceClassString = GetServiceClassString(cls)
696  
697      # Pull apart the command line
698      import getopt
699  
700      try:
701          opts, args = getopt.getopt(
702              argv[1:],
703              customInstallOptions,
704              [
705                  "password=",
706                  "username=",
707                  "startup=",
708                  "perfmonini=",
709                  "perfmondll=",
710                  "interactive",
711                  "wait=",
712              ],
713          )
714      except getopt.error as details:
715          print(details)
716          usage()
717      userName = None
718      password = None
719      perfMonIni = perfMonDll = None
720      startup = None
721      delayedstart = None
722      interactive = None
723      waitSecs = 0
724      for opt, val in opts:
725          if opt == "--username":
726              userName = val
727          elif opt == "--password":
728              password = val
729          elif opt == "--perfmonini":
730              perfMonIni = val
731          elif opt == "--perfmondll":
732              perfMonDll = val
733          elif opt == "--interactive":
734              interactive = 1
735          elif opt == "--startup":
736              map = {
737                  "manual": win32service.SERVICE_DEMAND_START,
738                  "auto": win32service.SERVICE_AUTO_START,
739                  "delayed": win32service.SERVICE_AUTO_START,  ## ChangeServiceConfig2 called later
740                  "disabled": win32service.SERVICE_DISABLED,
741              }
742              try:
743                  startup = map[val.lower()]
744              except KeyError:
745                  print("'%s' is not a valid startup option" % val)
746              if val.lower() == "delayed":
747                  delayedstart = True
748              elif val.lower() == "auto":
749                  delayedstart = False
750              ## else no change
751          elif opt == "--wait":
752              try:
753                  waitSecs = int(val)
754              except ValueError:
755                  print("--wait must specify an integer number of seconds.")
756                  usage()
757  
758      arg = args[0]
759      knownArg = 0
760      # First we process all arguments which pass additional args on
761      if arg == "start":
762          knownArg = 1
763          print("Starting service %s" % (serviceName))
764          try:
765              StartService(serviceName, args[1:])
766              if waitSecs:
767                  WaitForServiceStatus(
768                      serviceName, win32service.SERVICE_RUNNING, waitSecs
769                  )
770          except win32service.error as exc:
771              print("Error starting service: %s" % exc.strerror)
772              err = exc.winerror
773  
774      elif arg == "restart":
775          knownArg = 1
776          print("Restarting service %s" % (serviceName))
777          RestartService(serviceName, args[1:])
778          if waitSecs:
779              WaitForServiceStatus(serviceName, win32service.SERVICE_RUNNING, waitSecs)
780  
781      elif arg == "debug":
782          knownArg = 1
783          if not hasattr(sys, "frozen"):
784              # non-frozen services use pythonservice.exe which handles a
785              # -debug option
786              svcArgs = " ".join(args[1:])
787              try:
788                  exeName = LocateSpecificServiceExe(serviceName)
789              except win32api.error as exc:
790                  if exc.winerror == winerror.ERROR_FILE_NOT_FOUND:
791                      print("The service does not appear to be installed.")
792                      print("Please install the service before debugging it.")
793                      sys.exit(1)
794                  raise
795              try:
796                  os.system("%s -debug %s %s" % (exeName, serviceName, svcArgs))
797              # ^C is used to kill the debug service.  Sometimes Python also gets
798              # interrupted - ignore it...
799              except KeyboardInterrupt:
800                  pass
801          else:
802              # py2exe services don't use pythonservice - so we simulate
803              # debugging here.
804              DebugService(cls, args)
805  
806      if not knownArg and len(args) != 1:
807          usage()  # the rest of the cmds don't take addn args
808  
809      if arg == "install":
810          knownArg = 1
811          try:
812              serviceDeps = cls._svc_deps_
813          except AttributeError:
814              serviceDeps = None
815          try:
816              exeName = cls._exe_name_
817          except AttributeError:
818              exeName = None  # Default to PythonService.exe
819          try:
820              exeArgs = cls._exe_args_
821          except AttributeError:
822              exeArgs = None
823          try:
824              description = cls._svc_description_
825          except AttributeError:
826              description = None
827          print("Installing service %s" % (serviceName,))
828          # Note that we install the service before calling the custom option
829          # handler, so if the custom handler fails, we have an installed service (from NT's POV)
830          # but is unlikely to work, as the Python code controlling it failed.  Therefore
831          # we remove the service if the first bit works, but the second doesnt!
832          try:
833              InstallService(
834                  serviceClassString,
835                  serviceName,
836                  serviceDisplayName,
837                  serviceDeps=serviceDeps,
838                  startType=startup,
839                  bRunInteractive=interactive,
840                  userName=userName,
841                  password=password,
842                  exeName=exeName,
843                  perfMonIni=perfMonIni,
844                  perfMonDll=perfMonDll,
845                  exeArgs=exeArgs,
846                  description=description,
847                  delayedstart=delayedstart,
848              )
849              if customOptionHandler:
850                  customOptionHandler(*(opts,))
851              print("Service installed")
852          except win32service.error as exc:
853              if exc.winerror == winerror.ERROR_SERVICE_EXISTS:
854                  arg = "update"  # Fall through to the "update" param!
855              else:
856                  print(
857                      "Error installing service: %s (%d)" % (exc.strerror, exc.winerror)
858                  )
859                  err = exc.winerror
860          except ValueError as msg:  # Can be raised by custom option handler.
861              print("Error installing service: %s" % str(msg))
862              err = -1
863              # xxx - maybe I should remove after _any_ failed install - however,
864              # xxx - it may be useful to help debug to leave the service as it failed.
865              # xxx - We really _must_ remove as per the comments above...
866              # As we failed here, remove the service, so the next installation
867              # attempt works.
868              try:
869                  RemoveService(serviceName)
870              except win32api.error:
871                  print("Warning - could not remove the partially installed service.")
872  
873      if arg == "update":
874          knownArg = 1
875          try:
876              serviceDeps = cls._svc_deps_
877          except AttributeError:
878              serviceDeps = None
879          try:
880              exeName = cls._exe_name_
881          except AttributeError:
882              exeName = None  # Default to PythonService.exe
883          try:
884              exeArgs = cls._exe_args_
885          except AttributeError:
886              exeArgs = None
887          try:
888              description = cls._svc_description_
889          except AttributeError:
890              description = None
891          print("Changing service configuration")
892          try:
893              ChangeServiceConfig(
894                  serviceClassString,
895                  serviceName,
896                  serviceDeps=serviceDeps,
897                  startType=startup,
898                  bRunInteractive=interactive,
899                  userName=userName,
900                  password=password,
901                  exeName=exeName,
902                  displayName=serviceDisplayName,
903                  perfMonIni=perfMonIni,
904                  perfMonDll=perfMonDll,
905                  exeArgs=exeArgs,
906                  description=description,
907                  delayedstart=delayedstart,
908              )
909              if customOptionHandler:
910                  customOptionHandler(*(opts,))
911              print("Service updated")
912          except win32service.error as exc:
913              print(
914                  "Error changing service configuration: %s (%d)"
915                  % (exc.strerror, exc.winerror)
916              )
917              err = exc.winerror
918  
919      elif arg == "remove":
920          knownArg = 1
921          print("Removing service %s" % (serviceName))
922          try:
923              RemoveService(serviceName)
924              print("Service removed")
925          except win32service.error as exc:
926              print("Error removing service: %s (%d)" % (exc.strerror, exc.winerror))
927              err = exc.winerror
928      elif arg == "stop":
929          knownArg = 1
930          print("Stopping service %s" % (serviceName))
931          try:
932              if waitSecs:
933                  StopServiceWithDeps(serviceName, waitSecs=waitSecs)
934              else:
935                  StopService(serviceName)
936          except win32service.error as exc:
937              print("Error stopping service: %s (%d)" % (exc.strerror, exc.winerror))
938              err = exc.winerror
939      if not knownArg:
940          err = -1
941          print("Unknown command - '%s'" % arg)
942          usage()
943      return err

949  class ServiceFramework:
950      # Required Attributes:
951      # _svc_name_ = The service name
952      # _svc_display_name_ = The service display name
953  
954      # Optional Attributes:
955      _svc_deps_ = None  # sequence of service names on which this depends
956      _exe_name_ = None  # Default to PythonService.exe
957      _exe_args_ = None  # Default to no arguments
958      _svc_description_ = (
959          None  # Only exists on Windows 2000 or later, ignored on windows NT
960      )

962      def __init__(self, args):
963          import servicemanager
964  
965          self.ssh = servicemanager.RegisterServiceCtrlHandler(
966              args[0], self.ServiceCtrlHandlerEx, True
967          )
968          servicemanager.SetEventSourceName(self._svc_name_)
969          self.checkPoint = 0
970  
971      def GetAcceptedControls(self):
972          # Setup the service controls we accept based on our attributes. Note
973          # that if you need to handle controls via SvcOther[Ex](), you must
974          # override this.
975          accepted = 0
976          if hasattr(self, "SvcStop"):
977              accepted = accepted | win32service.SERVICE_ACCEPT_STOP
978          if hasattr(self, "SvcPause") and hasattr(self, "SvcContinue"):
979              accepted = accepted | win32service.SERVICE_ACCEPT_PAUSE_CONTINUE
980          if hasattr(self, "SvcShutdown"):
981              accepted = accepted | win32service.SERVICE_ACCEPT_SHUTDOWN
982          return accepted
983  
984      def ReportServiceStatus(
985          self, serviceStatus, waitHint=5000, win32ExitCode=0, svcExitCode=0
986      ):
987          if self.ssh is None:  # Debugging!
988              return
989          if serviceStatus == win32service.SERVICE_START_PENDING:
990              accepted = 0
991          else:
992              accepted = self.GetAcceptedControls()
993  
994          if serviceStatus in [
995              win32service.SERVICE_RUNNING,
996              win32service.SERVICE_STOPPED,
997          ]:
998              checkPoint = 0
999          else:
1000             self.checkPoint = self.checkPoint + 1
1001             checkPoint = self.checkPoint
1002 
1003         # Now report the status to the control manager
1004         status = (
1005             win32service.SERVICE_WIN32_OWN_PROCESS,
1006             serviceStatus,
1007             accepted,  # dwControlsAccepted,
1008             win32ExitCode,  # dwWin32ExitCode;
1009             svcExitCode,  # dwServiceSpecificExitCode;
1010             checkPoint,  # dwCheckPoint;
1011             waitHint,
1012         )
1013         win32service.SetServiceStatus(self.ssh, status)
1014 
1015     def SvcInterrogate(self):
1016         # Assume we are running, and everyone is happy.
1017         self.ReportServiceStatus(win32service.SERVICE_RUNNING)

1035     def ServiceCtrlHandlerEx(self, control, event_type, data):
1036         if control == win32service.SERVICE_CONTROL_STOP:
1037             return self.SvcStop()
1038         elif control == win32service.SERVICE_CONTROL_PAUSE:
1039             return self.SvcPause()
1040         elif control == win32service.SERVICE_CONTROL_CONTINUE:
1041             return self.SvcContinue()
1042         elif control == win32service.SERVICE_CONTROL_INTERROGATE:
1043             return self.SvcInterrogate()
1044         elif control == win32service.SERVICE_CONTROL_SHUTDOWN:
1045             return self.SvcShutdown()
1046         else:
1047             return self.SvcOtherEx(control, event_type, data)

1049     def SvcRun(self):
1050         # This is the entry point the C framework calls when the Service is
1051         # started. Your Service class should implement SvcDoRun().
1052         # Or you can override this method for more control over the Service
1053         # statuses reported to the SCM.
1054 
1055         # If this method raises an exception, the C framework will detect this
1056         # and report a SERVICE_STOPPED status with a non-zero error code.
1057 
1058         self.ReportServiceStatus(win32service.SERVICE_RUNNING)
1059         self.SvcDoRun()
1060         # Once SvcDoRun terminates, the service has stopped.
1061         # We tell the SCM the service is still stopping - the C framework
1062         # will automatically tell the SCM it has stopped when this returns.
1063         self.ReportServiceStatus(win32service.SERVICE_STOP_PENDING)
