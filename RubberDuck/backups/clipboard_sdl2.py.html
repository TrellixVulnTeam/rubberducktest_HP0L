1    ''' 
2    Clipboard SDL2: an implementation of the Clipboard using sdl2. 
3    '''
4    
5    __all__ = ('ClipboardSDL2', )
6    
7    from kivy.utils import platform
8    from kivy.core.clipboard import ClipboardBase
9    
10   if platform not in ('win', 'linux', 'macosx', 'android', 'ios'):
11       raise SystemError('unsupported platform for sdl2 clipboard')
12   
13   try:
14       from kivy.core.clipboard._clipboard_sdl2 import (
15           _get_text, _has_text, _set_text)
16   except ImportError:
17       from kivy.core import handle_win_lib_import_error
18       handle_win_lib_import_error(
19           'Clipboard', 'sdl2', 'kivy.core.clipboard._clipboard_sdl2')
20       raise
21   
22   
23   class ClipboardSDL2(ClipboardBase):
24   
25       def get(self, mimetype):
26           return _get_text() if _has_text() else ''
27   
28       def _ensure_clipboard(self):
29           super(ClipboardSDL2, self)._ensure_clipboard()
30           self._encoding = 'utf8'
31   
32       def put(self, data=b'', mimetype='text/plain'):
33           _set_text(data)
34   
35       def get_types(self):
36           return ['text/plain']
