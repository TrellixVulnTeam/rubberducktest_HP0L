showborder.py 
1    ''' 
2    Show border 
3    =========== 
4     
5    Shows widget's border. 
6     
7    The idea was taken from 
8    http://robertour.com/2013/10/02/easy-way-debugging-kivy-interfaces/ 
9    '''
10   
11   __all__ = ('start', 'stop')
12   
13   from kivy.lang import Builder
14   
15   
16   KV_CODE = ''' 
17   <Widget>: 
18       canvas.after: 
19           Color: 
20               rgba: 1, 1, 1, 1 
21           Line: 
22               rectangle: self.x + 1, self.y + 1, self.width - 1, self.height - 1 
23               dash_offset: 5 
24               dash_length: 3 
25   '''
26   
27   
28   def start(win, ctx):
29       Builder.load_string(KV_CODE, filename=__file__)
30   
31   
32   def stop(win, ctx):
33       Builder.unload_file(__file__)
