48   def RemoveRefCountOutput(data):
49       while 1:
50           last_line_pos = data.rfind("\n")
51           if not re.match("\[\d+ refs\]", data[last_line_pos + 1 :]):
52               break
53           if last_line_pos < 0:
54               # All the output
55               return ""
56           data = data[:last_line_pos]
57   
58       return data

61   def ExecuteSilentlyIfOK(cmd, testcase):
62       f = os.popen(cmd)
63       data = f.read().strip()
64       rc = f.close()
65       if rc:
66           print(data)
67           testcase.fail("Executing '%s' failed (%d)" % (cmd, rc))
68       # for "_d" builds, strip the '[xxx refs]' line
69       return RemoveRefCountOutput(data)

180  def get_test_mod_and_func(test_name, import_failures):
181      if test_name.find(".") > 0:
182          mod_name, func_name = test_name.split(".")
183      else:
184          mod_name = test_name
185          func_name = None
186      fq_mod_name = "win32com.test." + mod_name
187      try:
188          __import__(fq_mod_name)
189          mod = sys.modules[fq_mod_name]
190      except:
191          import_failures.append((mod_name, sys.exc_info()[:2]))
192          return None, None
193      if func_name is None:
194          func = None
195      else:
196          func = getattr(mod, func_name)
197      return mod, func

200  # Return a test suite all loaded with the tests we want to run
201  def make_test_suite(test_level=1):
202      suite = unittest.TestSuite()
203      import_failures = []
204      loader = TestLoader()
205      for i in range(testLevel):
206          for mod_name in unittest_modules[i]:
207              mod, func = get_test_mod_and_func(mod_name, import_failures)
208              if mod is None:
209                  continue
210              if func is not None:
211                  test = CapturingFunctionTestCase(func, description=mod_name)
212              else:
213                  if hasattr(mod, "suite"):
214                      test = mod.suite()
215                  else:
216                      test = loader.loadTestsFromModule(mod)
217              assert test.countTestCases() > 0, "No tests loaded from %r" % mod
218              suite.addTest(test)
219          for cmd, output in output_checked_programs[i]:
220              suite.addTest(ShellTestCase(cmd, output))
221  
222          for test_class in custom_test_cases[i]:
223              suite.addTest(unittest.defaultTestLoader.loadTestsFromTestCase(test_class))
224      # other "normal" unittest modules.
225      for i in range(testLevel):
226          for mod_name in unittest_other_modules[i]:
227              try:
228                  __import__(mod_name)
229              except:
230                  import_failures.append((mod_name, sys.exc_info()[:2]))
231                  continue
232  
233              mod = sys.modules[mod_name]
234              if hasattr(mod, "suite"):
235                  test = mod.suite()
236              else:
237                  test = loader.loadTestsFromModule(mod)
238              assert test.countTestCases() > 0, "No tests loaded from %r" % mod
239              suite.addTest(test)
240  
241      return suite, import_failures

244  def usage(why):
245      print(why)
246      print()
247      print("win32com test suite")
248      print("usage: testall [-v] test_level")
249      print("  where test_level is an integer 1-3.  Level 1 tests are quick,")

251      sys.exit(1)

254  if __name__ == "__main__":
255      try:
256          opts, args = getopt.getopt(sys.argv[1:], "v")
257      except getopt.error as why:
258          usage(why)
259      for opt, val in opts:
260          if opt == "-v":
261              verbosity += 1
262      testLevel = 2  # default to quick test with local objects
263      test_names = []
264      for arg in args:
265          try:
266              testLevel = int(arg)
267              if testLevel < 0 or testLevel > 4:
268                  raise ValueError("Only levels 1-4 are supported")
269          except ValueError:
270              test_names.append(arg)
271      if test_names:
272          usage("Test names are not supported yet")
273      CleanGenerated()
274  
275      suite, import_failures = make_test_suite(testLevel)
276      if verbosity:
277          if hasattr(sys, "gettotalrefcount"):
278              print("This is a debug build - memory leak tests will also be run.")
279              print("These tests may take *many* minutes to run - be patient!")
280              print("(running from python.exe will avoid these leak tests)")
281          print(
282              "Executing level %d tests - %d test cases will be run"
283              % (testLevel, suite.countTestCases())
284          )
285          if verbosity == 1 and suite.countTestCases() < 70:
286              # A little row of markers so the dots show how close to finished
287              print("|" * suite.countTestCases())
288      testRunner = TestRunner(verbosity=verbosity)
289      testResult = testRunner.run(suite)
290      if import_failures:
291          testResult.stream.writeln(
292              "*** The following test modules could not be imported ***"
293          )
294          for mod_name, (exc_type, exc_val) in import_failures:
295              desc = "\n".join(traceback.format_exception_only(exc_type, exc_val))
296              testResult.stream.write("%s: %s" % (mod_name, desc))
297          testResult.stream.writeln(
298              "*** %d test(s) could not be run ***" % len(import_failures)
299          )
300  
301      # re-print unit-test error here so it is noticed
302      if not testResult.wasSuccessful():
303          print("*" * 20, "- unittest tests FAILED")
304  
305      CheckClean()
306      pythoncom.CoUninitialize()
307      CleanGenerated()
308      if not testResult.wasSuccessful():
309          sys.exit(1)
