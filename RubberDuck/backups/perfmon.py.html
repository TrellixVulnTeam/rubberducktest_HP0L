1    """A COM Server which exposes the NT Performance monitor in a very rudimentary way 
2     
3    Usage from VB: 
4        set ob = CreateObject("Python.PerfmonQuery") 
5        freeBytes = ob.Query("Memory", "Available Bytes") 
6    """
7    from win32com.server import exception, register
8    import pythoncom, win32pdhutil, winerror
9    
10   
11   class PerfMonQuery:
12       _reg_verprogid_ = "Python.PerfmonQuery.1"
13       _reg_progid_ = "Python.PerfmonQuery"
14       _reg_desc_ = "Python Performance Monitor query object"
15       _reg_clsid_ = "{64cef7a0-8ece-11d1-a65a-00aa00125a98}"
16       _reg_class_spec_ = "win32com.servers.perfmon.PerfMonQuery"
17       _public_methods_ = ["Query"]
18   
19       def Query(self, object, counter, instance=None, machine=None):
20           try:
21               return win32pdhutil.GetPerformanceAttributes(
22                   object, counter, instance, machine=machine
23               )
24           except win32pdhutil.error as exc:
25               raise exception.Exception(desc=exc.strerror)
26           except TypeError as desc:
27               raise exception.Exception(desc=desc, scode=winerror.DISP_E_TYPEMISMATCH)
28   
29   
30   if __name__ == "__main__":
31       print("Registering COM server...")
32       register.UseCommandLine(PerfMonQuery)
