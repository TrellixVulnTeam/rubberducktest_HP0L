1    import os, win32api
2    
3    ver_strings = (
4        "Comments",
5        "InternalName",
6        "ProductName",
7        "CompanyName",
8        "LegalCopyright",
9        "ProductVersion",
10       "FileDescription",
11       "LegalTrademarks",
12       "PrivateBuild",
13       "FileVersion",
14       "OriginalFilename",
15       "SpecialBuild",
16   )
17   fname = os.environ["comspec"]
18   d = win32api.GetFileVersionInfo(fname, "\\")
19   ## backslash as parm returns dictionary of numeric info corresponding to VS_FIXEDFILEINFO struc
20   for n, v in d.items():
21       print(n, v)
22   
23   pairs = win32api.GetFileVersionInfo(fname, "\\VarFileInfo\\Translation")
24   ## \VarFileInfo\Translation returns list of available (language, codepage) pairs that can be used to retreive string info
25   ## any other must be of the form \StringfileInfo\%04X%04X\parm_name, middle two are language/codepage pair returned from above
26   for lang, codepage in pairs:
27       print("lang: ", lang, "codepage:", codepage)
28       for ver_string in ver_strings:
29           str_info = "\\StringFileInfo\\%04X%04X\\%s" % (lang, codepage, ver_string)
30           ## print str_info
31           print(ver_string, repr(win32api.GetFileVersionInfo(fname, str_info)))
