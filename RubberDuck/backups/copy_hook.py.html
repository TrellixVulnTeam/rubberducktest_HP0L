3    # To demostrate:
4    # * Execute this script to register the context menu.

6    # * Attempt to move or copy a directory.

8    import sys, os

10   from win32com.shell import shell, shellcon
11   import win32gui
12   import win32con
13   import winerror

15   # Our shell extension.
16   class ShellExtension:
17       _reg_progid_ = "Python.ShellExtension.CopyHook"
18       _reg_desc_ = "Python Sample Shell Extension (copy hook)"
19       _reg_clsid_ = "{1845b6ba-2bbd-4197-b930-46d8651497c1}"
20       _com_interfaces_ = [shell.IID_ICopyHook]
21       _public_methods_ = ["CopyCallBack"]
22   
23       def CopyCallBack(self, hwnd, func, flags, srcName, srcAttr, destName, destAttr):
24           # This function should return:
25           # IDYES Allows the operation.
26           # IDNO Prevents the operation on this folder but continues with any other operations that have been approved (for example, a batch copy operation).
27           # IDCANCEL Prevents the current operation and cancels any pending operations.
28           print("CopyCallBack", hwnd, func, flags, srcName, srcAttr, destName, destAttr)
29           return win32gui.MessageBox(
30               hwnd, "Allow operation?", "CopyHook", win32con.MB_YESNO
31           )
32   
33   
34   def DllRegisterServer():
35       import winreg
36   
37       key = winreg.CreateKey(
38           winreg.HKEY_CLASSES_ROOT,
39           "directory\\shellex\\CopyHookHandlers\\" + ShellExtension._reg_desc_,
40       )
41       winreg.SetValueEx(key, None, 0, winreg.REG_SZ, ShellExtension._reg_clsid_)
42       key = winreg.CreateKey(
43           winreg.HKEY_CLASSES_ROOT,
44           "*\\shellex\\CopyHookHandlers\\" + ShellExtension._reg_desc_,
45       )
46       winreg.SetValueEx(key, None, 0, winreg.REG_SZ, ShellExtension._reg_clsid_)
47       print(ShellExtension._reg_desc_, "registration complete.")
48   
49   
50   def DllUnregisterServer():
51       import winreg
52   
53       try:
54           key = winreg.DeleteKey(
55               winreg.HKEY_CLASSES_ROOT,
56               "directory\\shellex\\CopyHookHandlers\\" + ShellExtension._reg_desc_,
57           )
58       except WindowsError as details:
59           import errno
60   
61           if details.errno != errno.ENOENT:
62               raise
63       try:
64           key = winreg.DeleteKey(
65               winreg.HKEY_CLASSES_ROOT,
66               "*\\shellex\\CopyHookHandlers\\" + ShellExtension._reg_desc_,
67           )
68       except WindowsError as details:
69           import errno
70   
71           if details.errno != errno.ENOENT:
72               raise
73       print(ShellExtension._reg_desc_, "unregistration complete.")
74   
75   
76   if __name__ == "__main__":
77       from win32com.server import register
78   
79       register.UseCommandLine(
80           ShellExtension,
81           finalize_register=DllRegisterServer,
82           finalize_unregister=DllUnregisterServer,
83       )
