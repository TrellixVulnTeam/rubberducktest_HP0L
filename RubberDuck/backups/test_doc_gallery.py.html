4    def test_parse_docstring_info():
5        assert 'error' in parse_docstring_info("No Docstring")

7        assert 'error' in parse_docstring_info(
8                                "'''No Sentence\n======\nPeriods'''"
9                            )

11           "'\nSingle Quotes\n===\n\nNo singles.'")

16
