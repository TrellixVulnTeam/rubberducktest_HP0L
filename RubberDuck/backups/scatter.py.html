scatter.py 
1    ''' 
2    Scatter 
3    ======= 
4     
5    .. image:: images/scatter.gif 
6        :align: right 
7     
8    :class:`Scatter` is used to build interactive widgets that can be translated, 
9    rotated and scaled with two or more fingers on a multitouch system. 
10    
11   Scatter has its own matrix transformation: the modelview matrix is changed 
12   before the children are drawn and the previous matrix is restored when the 
13   drawing is finished. That makes it possible to perform rotation, scaling and 
14   translation over the entire children tree without changing any widget 
15   properties. That specific behavior makes the scatter unique, but there are some 
16   advantages / constraints that you should consider: 
17    
18   #. The children are positioned relative to the scatter similarly to a 
19      :mod:`~kivy.uix.relativelayout.RelativeLayout`. So when dragging the 
20      scatter, the position of the children don't change, only the position of 
21      the scatter does. 
22   #. The scatter size has no impact on the size of its children. 
23   #. If you want to resize the scatter, use scale, not size (read #2). Scale 
24      transforms both the scatter and its children, but does not change size. 
25   #. The scatter is not a layout. You must manage the size of the children 
26      yourself. 
27    
28   For touch events, the scatter converts from the parent matrix to the scatter 
29   matrix automatically in on_touch_down/move/up events. If you are doing things 
30   manually, you will need to use :meth:`~kivy.uix.widget.Widget.to_parent` and 
31   :meth:`~kivy.uix.widget.Widget.to_local`. 
32    
33   Usage 
34   ----- 
35    
36   By default, the Scatter does not have a graphical representation: it is a 
37   container only. The idea is to combine the Scatter with another widget, for 
38   example an :class:`~kivy.uix.image.Image`:: 
39    
40       scatter = Scatter() 
41       image = Image(source='sun.jpg') 
42       scatter.add_widget(image) 
43    
44   Control Interactions 
45   -------------------- 
46    
47   By default, all interactions are enabled. You can selectively disable 
48   them using the do_rotation, do_translation and do_scale properties. 
49    
50   Disable rotation:: 
51    
52       scatter = Scatter(do_rotation=False) 
53    
54   Allow only translation:: 
55    
56       scatter = Scatter(do_rotation=False, do_scale=False) 
57    
58   Allow only translation on x axis:: 
59    
60       scatter = Scatter(do_rotation=False, do_scale=False, 
61                         do_translation_y=False) 
62    
63    
64   Automatic Bring to Front 
65   ------------------------ 
66    
67   If the :attr:`Scatter.auto_bring_to_front` property is True, the scatter 
68   widget will be removed and re-added to the parent when it is touched 
69   (brought to front, above all other widgets in the parent). This is useful 
70   when you are manipulating several scatter widgets and don't want the active 
71   one to be partially hidden. 
72    
73   Scale Limitation 
74   ---------------- 
75    
76   We are using a 32-bit matrix in double representation. That means we have 
77   a limit for scaling. You cannot do infinite scaling down/up with our 
78   implementation. Generally, you don't hit the minimum scale (because you don't 
79   see it on the screen), but the maximum scale is 9.99506983235e+19 (2^66). 
80    
81   You can also limit the minimum and maximum scale allowed:: 
82    
83       scatter = Scatter(scale_min=.5, scale_max=3.) 
84    
85   Behavior 
86   -------- 
87    
88   .. versionchanged:: 1.1.0 
89       If no control interactions are enabled, then the touch handler will never 
90       return True. 
91    
92   '''

236      def _get_bbox(self):
237          xmin, ymin = xmax, ymax = self.to_parent(0, 0)
238          for point in [(self.width, 0), (0, self.height), self.size]:
239              x, y = self.to_parent(*point)
240              if x < xmin:
241                  xmin = x
242              if y < ymin:
243                  ymin = y
244              if x > xmax:
245                  xmax = x
246              if y > ymax:
247                  ymax = y
248          return (xmin, ymin), (xmax - xmin, ymax - ymin)

251      '''Bounding box of the widget in parent space:: 
252   
253          ((x, y), (w, h)) 
254          # x, y = lower left corner 
255   
256      :attr:`bbox` is an :class:`~kivy.properties.AliasProperty`. 
257      '''

259      def _get_rotation(self):
260          v1 = Vector(0, 10)
261          tp = self.to_parent
262          v2 = Vector(*tp(*self.pos)) - tp(self.x, self.y + 10)
263          return -1.0 * (v1.angle(v2) + 180) % 360
264  
265      def _set_rotation(self, rotation):
266          angle_change = self.rotation - rotation
267          r = Matrix().rotate(-radians(angle_change), 0, 0, 1)
268          self.apply_transform(r, post_multiply=True,
269                               anchor=self.to_local(*self.center))

280      def _get_scale(self):
281          p1 = Vector(*self.to_parent(0, 0))
282          p2 = Vector(*self.to_parent(1, 0))
283          scale = p1.distance(p2)
284  
285          # XXX float calculation are not accurate, and then, scale can be
286          # throwed again even with only the position change. So to
287          # prevent anything wrong with scale, just avoid to dispatch it
288          # if the scale "visually" didn't change. #947
289          # Remove this ugly hack when we'll be Python 3 only.
290          if hasattr(self, '_scale_p'):
291              if str(scale) == str(self._scale_p):
292                  return self._scale_p
293  
294          self._scale_p = scale
295          return scale
296  
297      def _set_scale(self, scale):
298          rescale = scale * 1.0 / self.scale
299          self.apply_transform(Matrix().scale(rescale, rescale, rescale),
300                               post_multiply=True,
301                               anchor=self.to_local(*self.center))

398      def collide_point(self, x, y):

411          m = self.transform.multiply(m)
412          return super(Scatter, self)._apply_transform(m, (0, 0))

414      def apply_transform(self, trans, post_multiply=False, anchor=(0, 0)):
415          ''' 
416          Transforms the scatter by applying the "trans" transformation 
417          matrix (on top of its current transformation state). The resultant 
418          matrix can be found in the :attr:`~Scatter.transform` property. 
419   
420          :Parameters: 
421              `trans`: :class:`~kivy.graphics.transformation.Matrix`. 
422                  Transformation matix to be applied to the scatter widget. 
423              `anchor`: tuple, defaults to (0, 0). 
424                  The point to use as the origin of the transformation 
425                  (uses local widget space). 
426              `post_multiply`: bool, defaults to False. 
427                  If True, the transform matrix is post multiplied 
428                  (as if applied before the current transform). 
429   
430          Usage example:: 
431   
432              from kivy.graphics.transformation import Matrix 
433              mat = Matrix().scale(3, 3, 3) 
434              scatter_instance.apply_transform(mat) 
435   
436          '''
437          t = Matrix().translate(anchor[0], anchor[1], 0)
438          t = t.multiply(trans)
439          t = t.multiply(Matrix().translate(-anchor[0], -anchor[1], 0))
440  
441          if post_multiply:
442              self.transform = self.transform.multiply(t)
443          else:
444              self.transform = t.multiply(self.transform)
445  
446      def transform_with_touch(self, touch):
447          # just do a simple one finger drag
448          changed = False
449          if len(self._touches) == self.translation_touches:
450              # _last_touch_pos has last pos in correct parent space,
451              # just like incoming touch
452              dx = (touch.x - self._last_touch_pos[touch][0]) \
453                  * self.do_translation_x
454              dy = (touch.y - self._last_touch_pos[touch][1]) \
455                  * self.do_translation_y
456              dx = dx / self.translation_touches
457              dy = dy / self.translation_touches
458              self.apply_transform(Matrix().translate(dx, dy, 0))
459              changed = True
460  
461          if len(self._touches) == 1:
462              return changed
463  
464          # we have more than one touch... list of last known pos
465          points = [Vector(self._last_touch_pos[t]) for t in self._touches
466                    if t is not touch]
467          # add current touch last
468          points.append(Vector(touch.pos))
469  
470          # we only want to transform if the touch is part of the two touches
471          # farthest apart! So first we find anchor, the point to transform
472          # around as another touch farthest away from current touch's pos
473          anchor = max(points[:-1], key=lambda p: p.distance(touch.pos))
474  
475          # now we find the touch farthest away from anchor, if its not the
476          # same as touch. Touch is not one of the two touches used to transform
477          farthest = max(points, key=anchor.distance)
478          if farthest is not points[-1]:
479              return changed
480  
481          # ok, so we have touch, and anchor, so we can actually compute the
482          # transformation
483          old_line = Vector(*touch.ppos) - anchor
484          new_line = Vector(*touch.pos) - anchor
485          if not old_line.length():   # div by zero
486              return changed
487  
488          angle = radians(new_line.angle(old_line)) * self.do_rotation
489          if angle:
490              changed = True
491          self.apply_transform(Matrix().rotate(angle, 0, 0, 1), anchor=anchor)
492  
493          if self.do_scale:
494              scale = new_line.length() / old_line.length()
495              new_scale = scale * self.scale
496              if new_scale < self.scale_min:
497                  scale = self.scale_min / self.scale
498              elif new_scale > self.scale_max:
499                  scale = self.scale_max / self.scale
500              self.apply_transform(Matrix().scale(scale, scale, scale),
501                                   anchor=anchor)
502              changed = True
503          return changed
504  
505      def _bring_to_front(self, touch):
506          # auto bring to front
507          if self.auto_bring_to_front and self.parent:
508              parent = self.parent
509              if parent.children[0] is self:
510                  return
511              parent.remove_widget(self)
512              parent.add_widget(self)
513              self.dispatch('on_bring_to_front', touch)
514  
515      def on_touch_down(self, touch):
516          x, y = touch.x, touch.y
517  
518          # if the touch isnt on the widget we do nothing
519          if not self.do_collide_after_children:
520              if not self.collide_point(x, y):
521                  return False
522  
523          # let the child widgets handle the event if they want
524          touch.push()
525          touch.apply_transform_2d(self.to_local)
526          if super(Scatter, self).on_touch_down(touch):
527              touch.pop()
528              self._bring_to_front(touch)
529              return True
530          touch.pop()
531  
532          # if our child didn't do anything, and if we don't have any active
533          # interaction control, then don't accept the touch.
534          if not self.do_translation_x and \
535                  not self.do_translation_y and \
536                  not self.do_rotation and \
537                  not self.do_scale:
538              return False
539  
540          if self.do_collide_after_children:
541              if not self.collide_point(x, y):
542                  return False
543  
544          if 'multitouch_sim' in touch.profile:
545              touch.multitouch_sim = True
546          # grab the touch so we get all it later move events for sure
547          self._bring_to_front(touch)
548          touch.grab(self)
549          self._touches.append(touch)
550          self._last_touch_pos[touch] = touch.pos
551  
552          return True
553  
554      def on_touch_move(self, touch):
555          x, y = touch.x, touch.y
556          # let the child widgets handle the event if they want
557          if self.collide_point(x, y) and not touch.grab_current == self:
558              touch.push()
559              touch.apply_transform_2d(self.to_local)
560              if super(Scatter, self).on_touch_move(touch):
561                  touch.pop()
562                  return True
563              touch.pop()
564  
565          # rotate/scale/translate
566          if touch in self._touches and touch.grab_current == self:
567              if self.transform_with_touch(touch):
568                  self.dispatch('on_transform_with_touch', touch)
569              self._last_touch_pos[touch] = touch.pos
570  
571          # stop propagating if its within our bounds
572          if self.collide_point(x, y):
573              return True

603      def on_touch_up(self, touch):
604          x, y = touch.x, touch.y
605          # if the touch isnt on the widget we do nothing, just try children
606          if not touch.grab_current == self:
607              touch.push()
608              touch.apply_transform_2d(self.to_local)
609              if super(Scatter, self).on_touch_up(touch):
610                  touch.pop()
611                  return True
612              touch.pop()
613  
614          # remove it from our saved touches
615          if touch in self._touches and touch.grab_state:
616              touch.ungrab(self)
617              del self._last_touch_pos[touch]
618              self._touches.remove(touch)
619  
620          # stop propagating if its within our bounds
621          if self.collide_point(x, y):
622              return True
