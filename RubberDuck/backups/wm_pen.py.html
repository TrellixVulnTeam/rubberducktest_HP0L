1    ''' 
2    Support for WM_PEN messages (Windows platform) 
3    ============================================== 
4    '''
5    
6    __all__ = ('WM_PenProvider', 'WM_Pen')
7    
8    import os
9    from kivy.input.providers.wm_common import RECT, PEN_OR_TOUCH_MASK, \
10       PEN_OR_TOUCH_SIGNATURE, PEN_EVENT_TOUCH_MASK, WM_LBUTTONDOWN, \
11       WM_MOUSEMOVE, WM_LBUTTONUP, WM_TABLET_QUERYSYSTEMGESTURE, \
12       QUERYSYSTEMGESTURE_WNDPROC, WNDPROC, SetWindowLong_WndProc_wrapper
13   from kivy.input.motionevent import MotionEvent
14   
15   
16   class WM_Pen(MotionEvent):
17       '''MotionEvent representing the WM_Pen event. Supports the pos profile.'''
18   
19       def depack(self, args):
20           self.is_touch = True
21           self.sx, self.sy = args[0], args[1]
22           super(WM_Pen, self).depack(args)
23   
24       def __str__(self):
25           i, u, s, d = (self.id, self.uid, str(self.spos), self.device)
26           return '<WMPen id:%d uid:%d pos:%s device:%s>' % (i, u, s, d)
27   
28   
29   if 'KIVY_DOC' in os.environ:
30       # documentation hack
31       WM_PenProvider = None
32   
33   else:
34       from collections import deque
35       from ctypes import windll, byref, c_int16, c_int
36       from kivy.input.provider import MotionEventProvider
37       from kivy.input.factory import MotionEventFactory
38   
39       win_rect = RECT()
40   
41       class WM_PenProvider(MotionEventProvider):
42   
43           def _is_pen_message(self, msg):
44               info = windll.user32.GetMessageExtraInfo()
45               # It's a touch or a pen
46               if (info & PEN_OR_TOUCH_MASK) == PEN_OR_TOUCH_SIGNATURE:
47                   if not info & PEN_EVENT_TOUCH_MASK:
48                       return True
49   
50           def _pen_handler(self, msg, wParam, lParam):
51               if msg not in (WM_LBUTTONDOWN, WM_MOUSEMOVE, WM_LBUTTONUP):
52                   return
53   
54               windll.user32.GetClientRect(self.hwnd, byref(win_rect))
55               x = c_int16(lParam & 0xffff).value / float(win_rect.w)
56               y = c_int16(lParam >> 16).value / float(win_rect.h)
57               y = abs(1.0 - y)
58   
59               if msg == WM_LBUTTONDOWN:
60                   self.pen_events.appendleft(('begin', x, y))
61                   self.pen_status = True
62   
63               if msg == WM_MOUSEMOVE and self.pen_status:
64                   self.pen_events.appendleft(('update', x, y))
65   
66               if msg == WM_LBUTTONUP:
67                   self.pen_events.appendleft(('end', x, y))
68                   self.pen_status = False
69   
70           def _pen_wndProc(self, hwnd, msg, wParam, lParam):
71               if msg == WM_TABLET_QUERYSYSTEMGESTURE:
72                   return QUERYSYSTEMGESTURE_WNDPROC
73               if self._is_pen_message(msg):
74                   self._pen_handler(msg, wParam, lParam)
75                   return 1
76               else:
77                   return windll.user32.CallWindowProcW(self.old_windProc,
78                                                        hwnd, msg, wParam, lParam)
79   
80           def start(self):
81               self.uid = 0
82               self.pen = None
83               self.pen_status = None
84               self.pen_events = deque()
85   
86               self.hwnd = windll.user32.GetActiveWindow()
87   
88               # inject our own wndProc to handle messages
89               # before window manager does
90               self.new_windProc = WNDPROC(self._pen_wndProc)
91               self.old_windProc = SetWindowLong_WndProc_wrapper(
92                   self.hwnd, self.new_windProc)
93   
94           def update(self, dispatch_fn):
95               while True:
96   
97                   try:
98                       etype, x, y = self.pen_events.pop()
99                   except:
100                      break
101  
102                  if etype == 'begin':
103                      self.uid += 1
104                      self.pen = WM_Pen(self.device, self.uid, [x, y])
105                  elif etype == 'update':
106                      self.pen.move([x, y])
107                  elif etype == 'end':
108                      self.pen.update_time_end()
109  
110                  dispatch_fn(etype, self.pen)
111  
112          def stop(self):
113              self.pen = None
114              SetWindowLong_WndProc_wrapper(self.hwnd, self.old_windProc)
115  
116      MotionEventFactory.register('wm_pen', WM_PenProvider)
