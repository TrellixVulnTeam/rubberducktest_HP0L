310  class math(object):
311      """Base class for MathML elements and root of MathML trees."""
312  
313      nchildren = None
314      """Expected number of children or None"""
315      # cf. https://www.w3.org/TR/MathML3/chapter3.html#id.3.1.3.2
316      parent = None
317      """Parent node in MathML DOM tree."""
318      _level = 0 # indentation level (static class variable)
319      xml_entities = { # for invalid and invisible characters
320                      ord('<'): u'&lt;',
321                      ord('>'): u'&gt;',
322                      ord('&'): u'&amp;',
323                      0x2061:   u'&ApplyFunction;',
324                     }
325      _boolstrings = {True: 'true', False: 'false'}
326      """String representation of boolean MathML attribute values."""
327  
328      html_tagname = 'span'
329      """Tag name for HTML representation."""
330  
331      def __init__(self, *children, **attributes):
332          """Set up node with `children` and `attributes`. 
333   
334          Attributes are downcased: Use CLASS to set "class" value. 
335          >>> math(mn(3), CLASS='test') 
336          math(mn(3), class='test') 
337          >>> math(CLASS='test').toprettyxml() 
338          '<math class="test">\n</math>' 
339   
340          """
341          self.children = []
342          self.extend(children)
343  
344          self.attributes = collections.OrderedDict()
345          # sort attributes for predictable functional tests
346          # as self.attributes.update(attributes) does not keep order in Python < 3.6
347          for key in sorted(attributes.keys()):
348              # Use .lower() to allow argument `CLASS` for attribute `class`
349              # (Python keyword). MathML uses only lowercase attributes.
350              self.attributes[key.lower()] = attributes[key]
351  
352      def __repr__(self):
353          content = [repr(item) for item in getattr(self, 'children', [])]
354          if hasattr(self, 'data'):
355              content.append(repr(self.data))
356          if isinstance(self, MathSchema) and self.switch:
357              content.append('switch=True')
358          content += ["%s=%r"%(k, v) for k, v in self.attributes.items()
359                      if v is not None]
360  
361          return self.__class__.__name__ + '(%s)' % ', '.join(content)
362  
363      def __len__(self):
364          return len(self.children)
365  
366      # emulate dictionary-like access to attributes
367      # see `docutils.nodes.Element` for dict/list interface
368      def __getitem__(self, key):
369          return self.attributes[key]
370      def __setitem__(self, key, item):
371          self.attributes[key] = item
372      def get(self, *args, **kwargs):
373          return self.attributes.get(*args, **kwargs)
374  
375      def full(self):
376          """Return boolean indicating whether children may be appended."""
377          return (self.nchildren is not None
378                  and len(self) >= self.nchildren)
379  
380      def append(self, child):
381          """Append child and return self or first non-full parent. 
382   
383          If self is full, go up the tree and return first non-full node or 
384          `None`. 
385          """
386          if self.full():
387              raise SyntaxError('Node %s already full!' % self)
388          self.children.append(child)
389          child.parent = self
390          if self.full():
391              return self.close()
392          return self
393  
394      def extend(self, children):
395          for child in children:
396              self.append(child)
397          return self
398  
399      def close(self):
400          """Close element and return first non-full parent or None."""
401          parent = self.parent
402          while parent is not None and parent.full():
403              parent = parent.parent
404          return parent
405  
406      def toprettyxml(self):
407          """Return XML representation of self as string."""
408          return ''.join(self._xml())
409  
410      def _xml(self, level=0):
411          return ([self.xml_starttag()]
412                  + self._xml_body(level)
413                  + ['</%s>' % self.__class__.__name__])
414  
415      def xml_starttag(self):
416          attrs = ['%s="%s"' % (k, str(v).replace('True', 'true').replace('False', 'false'))
417                   for k, v in self.attributes.items()
418                   if v is not None]
419          return '<%s>' % ' '.join([self.__class__.__name__] + attrs)
420  
421      def _xml_body(self, level=0):
422          xml = []
423          for child in self.children:
424              xml.extend(['\n', '  ' * (level+1)])
425              xml.extend(child._xml(level+1))
426          xml.extend(['\n', '  ' * level])
427          return xml

454  class mrow(math):
455      """Group sub-expressions as a horizontal row."""
456  
457      def close(self):
458          """Close element and return first non-full parent or None. 
459   
460          Remove <mrow>, if it is single child and the parent infers an mrow 
461          or if it has only one child element. 
462          """
463          parent = self.parent
464          if isinstance(parent, MathRowSchema) and parent.nchildren == 1:
465              parent.nchildren = len(parent.children)
466              parent.children = self.children
467              for child in self.children:
468                  child.parent = parent
469              return parent.close()
470          if len(self) == 1:
471              try:
472                  parent.children[parent.children.index(self)] = self.children[0]
473                  self.children[0].parent = parent
474              except (AttributeError, ValueError):
475                  return self.children[0]
476          return super(mrow, self).close()

528  class MathSchema(math):
529      """Base class for schemata expecting 2 or more children. 
530   
531      The special attribute `switch` indicates that the last two child 
532      elements are in reversed order and must be switched before XML-export. 
533      """
534  
535      nchildren = 2
536  
537      def __init__(self, *children, **kwargs):
538          self.switch = kwargs.pop('switch', False)
539          math.__init__(self, *children, **kwargs)
540  
541      def append(self, child):
542          current_node = super(MathSchema, self).append(child)
543          # normalize order if full
544          if self.switch and self.full():
545              self.children[-1], self.children[-2] = self.children[-2], self.children[-1]
546              self.switch = False
547          return current_node

596  def tex_cmdname(string):
597      """Return leading TeX command name and remainder of `string`. 
598   
599      >>> tex_cmdname('mymacro2') # up to first non-letter 
600      ('mymacro', '2') 
601      >>> tex_cmdname('name 2') # strip trailing whitespace 
602      ('name', '2') 
603      >>> tex_cmdname('_2') # single non-letter character 
604      ('_', '2') 
605   
606      """
607      m = re.match(r'([a-zA-Z]+) *(.*)', string)
608      if m is None:
609          m = re.match(r'(.?)(.*)', string)
610      return m.group(1), m.group(2)

623  def tex_number(string):
624      """Return leading number literal and remainder of `string`. 
625   
626      >>> tex_number('123.4') 
627      ('123.4', '') 
628   
629      """
630      m = re.match(r'([0-9.,]*[0-9]+)(.*)', string)
631      if m is None:
632          return '', string
633      return m.group(1), m.group(2)

648  def tex_token(string):
649      """Return first simple TeX token and remainder of `string`. 
650   
651      >>> tex_token('\\command{without argument}') 
652      ('\\command', '{without argument}') 
653      >>> tex_token('or first character') 
654      ('o', 'r first character') 
655   
656      """
657      m = re.match(r"""((?P<cmd>\\[a-zA-Z]+)\s* # TeX command, skip whitespace 
658                        |(?P<chcmd>\\.)          # one-character TeX command 
659                        |(?P<ch>.?))            # first character (or empty) 
660                       (?P<remainder>.*$)    # remaining part of string 
661                   """, string, re.VERBOSE)
662      cmd, chcmd, ch, remainder = m.group('cmd', 'chcmd', 'ch', 'remainder')
663      return cmd or chcmd or ch, remainder

678  def tex_group(string):
679      """Return first TeX group or token and remainder of `string`. 
680   
681      >>> tex_group('{first group} returned without brackets') 
682      ('first group', ' returned without brackets') 
683   
684      """
685      split_index = 0
686      nest_level = 0   # level of {{nested} groups}
687      escape = False   # the next character is escaped (\)
688  
689      if not string.startswith('{'):
690          # special case: there is no group, return first token and remainder
691          return string[:1], string[1:]
692      for c in string:
693          split_index += 1
694          if escape:
695              escape = False
696          elif c == '\\':
697              escape = True
698          elif c == '{':
699              nest_level += 1
700          elif c == '}':
701              nest_level -= 1
702          if nest_level == 0:
703              break
704      else:
705          raise SyntaxError('Group without closing bracket')
706      return string[1:split_index-1], string[split_index:]

728  def tex_token_or_group(string):
729      """Return first TeX group or token and remainder of `string`. 
730   
731      >>> tex_token_or_group('\\command{without argument}') 
732      ('\\command', '{without argument}') 
733      >>> tex_token_or_group('first character') 
734      ('f', 'irst character')

736      (' ', 'also whitespace') 
737      >>> tex_token_or_group('{first group} keep rest') 
738      ('first group', ' keep rest') 
739   
740      """
741      arg, remainder = tex_token(string)
742      if arg == '{':
743          arg, remainder = tex_group(string.lstrip())
744      return arg, remainder

749  def tex_optarg(string):
750      """Return optional argument and remainder. 
751   
752      >>> tex_optarg('[optional argument] returned without brackets') 
753      ('optional argument', ' returned without brackets') 
754      >>> tex_optarg('{empty string, if there is no optional arg}') 
755      ('', '{empty string, if there is no optional arg}') 
756   
757      """
758      m = re.match(r"""\s*                            # leading whitespace 
759                   \[(?P<optarg>(\\]|[^\[\]]|\\])*)\] # [group] without nested groups 
760                   (?P<remainder>.*$) 
761                   """, string, re.VERBOSE)
762      if m is None and not string.startswith('['):
763          return '', string
764      try:
765          return m.group('optarg'), m.group('remainder')
766      except AttributeError:
767          raise SyntaxError('Could not extract optional argument from %r' % string)

780  def parse_latex_math(node, string):
781      """Append MathML conversion of `string` to `node` and return it. 
782   
783      >>> parse_latex_math(math(), r'\alpha') 
784      math(mi('α')) 
785      >>> parse_latex_math(mrow(), r'x_{n}') 
786      mrow(msub(mi('x'), mi('n'))) 
787   
788      """
789      # Normalize white-space:
790      string = ' '.join(string.split())
791      tree = node
792  
793      while len(string) > 0:
794          # Take off first character:
795          c, string = string[0], string[1:]
796  
797          if c == ' ':
798              continue  # whitespace is ignored in LaTeX math mode
799          if c == '\\': # start of a LaTeX macro
800              cmdname, string = tex_cmdname(string)
801              node, string = handle_cmd(cmdname, node, string)
802          elif c in "_^":
803              node = handle_script_or_limit(node, c)
804          elif c == '{':
805              new_node = mrow()
806              node.append(new_node)
807              node = new_node
808          elif c == '}':
809              node = node.close()
810          elif c == '&':
811              new_node = mtd()
812              node.close().append(new_node)
813              node = new_node
814          elif c.isalpha():
815              node = node.append(mi(c))
816          elif c.isdigit():
817              number, string = tex_number(string)
818              node = node.append(mn(c+number))
819          elif c in anomalous_chars:
820              # characters with a special meaning in LaTeX math mode
821              # fix spacing before "unary" minus.
822              attributes = {}
823              if c == '-' and node.children:
824                  previous_node = node.children[-1]
825                  if (getattr(previous_node, 'data', '-') in '([='
826                      or previous_node.get('class') == 'mathopen'):
827                      attributes['form'] = 'prefix'
828              node = node.append(mo(anomalous_chars[c], **attributes))
829          elif c in "/()[]|":
830              node = node.append(mo(c, stretchy=False))
831          elif c in "+*=<>,.!?';@":
832              node = node.append(mo(c))
833          else:
834              raise SyntaxError(u'Unsupported character: "%s"' % c)
835      return tree

862  def handle_cmd(name, node, string):
863      """Process LaTeX command `name` followed by `string`. 
864   
865      Append result to `node`. 
866      If needed, parse `string` for command argument. 
867      Return new current node and remainder of `string`: 
868   
869      >>> handle_cmd('hbar', math(), r' \frac') 
870      (math(mi('ℏ')), ' \\frac') 
871      >>> handle_cmd('hspace', math(), r'{1ex} (x)') 
872      (math(mspace(width='1ex')), ' (x)') 
873   
874      """
875  
876      # Token elements
877      # ==============
878  
879      # identifier  ->  <mi>
880  
881      if name in letters:
882          new_node = mi(letters[name])
883          if name in greek_capitals:
884              # upright in "TeX style" but MathML sets them italic ("ISO style").
885              # CSS styling does not change the font style in Firefox 78.
886              # Use 'mathvariant="normal"'?
887              new_node['class'] = 'capital-greek'
888          node = node.append(new_node)
889          return node, string
890  
891      if name in functions:
892          # use <mi> followed by invisible function applicator character
893          # (see https://www.w3.org/TR/MathML3/chapter3.html#presm.mi)
894          if name == 'operatorname':
895              # custom function name, e.g. ``\operatorname{abs}(x)``
896              # TODO: \operatorname* -> with limits
897              arg, string = tex_token_or_group(string)
898              new_node = mi(arg, mathvariant='normal')
899          else:
900              new_node = mi(functions[name])
901          # embellished function names:
902          if name == 'varliminf':    # \underline\lim
903              new_node = munder(new_node, mo(u'_'))
904          elif name == 'varlimsup':  # \overline\lim
905              new_node = mover(new_node, mo(u'¯'), accent=False)
906          elif name == 'varprojlim': # \underleftarrow\lim
907              new_node = munder(new_node, mo(u'\u2190'))
908          elif name == 'varinjlim':  # \underrightarrow\lim
909              new_node = munder(new_node, mo(u'\u2192'))
910  
911          node = node.append(new_node)
912          # add ApplyFunction when appropriate (not \sin^2(x), say)
913          # cf. https://www.w3.org/TR/MathML3/chapter3.html#presm.mi
914          if string and string[0] not in ('^', '_'):
915              node = node.append(mo(u'\u2061')) # &ApplyFunction;
916          return node, string
917  
918      if name in math_alphabets:
919          if name == 'boldsymbol':
920              attributes = {'class': 'boldsymbol'}
921          else:
922              attributes = {'mathvariant': math_alphabets[name]}
923          if name == 'mathscr':
924              attributes['class'] = 'mathscr'
925          # Check for single symbol (letter, name, or ⅀)
926          arg, remainder = tex_token_or_group(string)
927          if arg.startswith('\\'):
928              # convert single letters (so the isalpha() test below works).
929              # TODO: convert all LICRs in a group (\matrm{\mu\Omega})
930              arg = letters.get(arg[1:], arg)
931          if name == 'mathbb':
932              # mathvariant="double-struck" is ignored for Greek letters
933              # (tested in Firefox 78). Use literal Unicode characters.
934              arg = mathbb.get(arg, arg)
935          if arg.isalpha() or arg == u'\u2140':
936              node = node.append(mi(arg, **attributes))
937              return node, remainder
938          # Wrap in <style>
939          style = mstyle(**attributes)
940          node.append(style)
941          return style, string
942  
943  
944      # operator, fence, or separator  ->  <mo>
945  
946      if name == 'colon': # trailing punctuation, not binary relation
947          node = node.append(mo(':', form='postfix', lspace='0', rspace='0.28em'))
948          return node, string
949  
950      if name == 'idotsint':
951          node = parse_latex_math(node, '\int\dotsi\int')
952          return node, string
953  
954      if name in thick_operators:
955          node = node.append(mo(thick_operators[name], style='font-weight: bold'))
956          return node, string
957  
958      if name in small_operators:
959          node = node.append(mo(small_operators[name], mathsize='75%'))
960          return node, string
961  
962      if name in operators:
963          attributes = {}
964          if name in movablelimits and string and string[0] in ' _^':
965              attributes['movablelimits'] = True
966          elif name in ('lvert', 'lVert'):
967              attributes['class'] = 'mathopen'
968          node = node.append(mo(operators[name], **attributes))
969          return node, string
970  
971      if name in bigdelimiters:
972          delimiter_attributes = {}
973          size = delimiter_sizes[bigdelimiters[name]]
974          delimiter, string = tex_token_or_group(string)
975          if delimiter not in '()[]/|.':
976              try:
977                  delimiter = stretchables[delimiter.lstrip('\\')]
978              except KeyError:
979                  raise SyntaxError(u'Unsupported "\\%s" delimiter "%s"!'
980                                    % (name, delimiter))
981          if size:
982              delimiter_attributes['maxsize'] = size
983              delimiter_attributes['minsize'] = size
984              delimiter_attributes['symmetric'] = True
985          if name == 'left' or name.endswith('l'):
986              row = mrow()
987              node.append(row)
988              node = row
989          if delimiter != '.': # '.' stands for "empty delimiter"
990              node.append(mo(delimiter, **delimiter_attributes))
991          if name == 'right' or name.endswith('r'):
992              node = node.close()
993          return node, string
994  
995      if name == 'not':
996          arg, string = tex_token(string)
997          if arg == '{':
998              return node, '{\\not ' + string
999          if arg.startswith('\\'): # LaTeX macro
1000             try:
1001                 arg = operators[arg[1:]]
1002             except KeyError:
1003                 raise SyntaxError(u'\\not: Cannot negate: "%s"!'%arg)
1004         arg = unicodedata.normalize('NFC', arg+u'\u0338')
1005         node = node.append(mo(arg))
1006         return node, string
1007 
1008     # arbitrary text (usually comments)  ->  <mtext>
1009     if name in ('text', 'mbox', 'textrm'):
1010         arg, string = tex_token_or_group(string)
1011         parts = arg.split('$') # extract inline math
1012         for i, part in enumerate(parts):
1013             if i % 2 == 0: # i is even
1014                 part = re.sub('(^ | $)', u'\u00a0', part)
1015                 node = node.append(mtext(part))
1016             else:
1017                 parse_latex_math(node, part)
1018         return node, string
1019 
1020     # horizontal space -> <mspace>
1021     if name in spaces:
1022         node = node.append(mspace(width='%s'%spaces[name]))
1023         return node, string
1024 
1025     if name in ('hspace', 'mspace'):
1026         arg, string = tex_group(string)
1027         if arg.endswith('mu'):
1028             arg = '%sem' % (float(arg[:-2])/18)
1029         node = node.append(mspace(width='%s'%arg))
1030         return node, string
1031 
1032     if name == 'phantom':
1033         new_node = mphantom()
1034         node.append(new_node)
1035         return new_node, string
1036 
1037     if name == 'boxed':
1038         new_node = menclose(notation='box')
1039         node.append(new_node)
1040         return new_node, string
1041 
1042 
1043     # Complex elements (Layout schemata)
1044     # ==================================
1045 
1046     if name == 'sqrt':
1047         radix, string = tex_optarg(string)
1048         if radix:
1049             indexnode = mrow()
1050             new_node = mroot(indexnode, switch=True)
1051             parse_latex_math(indexnode, radix)
1052             indexnode.close()
1053         else:
1054             new_node = msqrt()
1055         node.append(new_node)
1056         return new_node, string
1057 
1058     if name in fractions:
1059         (style_atts, frac_atts) = fractions[name]
1060         if name == 'cfrac':
1061             optarg, string = tex_optarg(string)
1062             optargs = {'l': 'left', 'r': 'right'}
1063             if optarg in optargs:
1064                 frac_atts = frac_atts.copy()
1065                 frac_atts['numalign'] = optargs[optarg] # "numalign" is deprecated
1066                 frac_atts['class'] = 'numalign-' + optargs[optarg]
1067         new_node = frac = mfrac(**frac_atts)
1068         if name.endswith('binom'):
1069             new_node = mrow(mo('('), new_node, mo(')'), CLASS='binom')
1070             new_node.nchildren = 3
1071         if style_atts:
1072             new_node = mstyle(new_node, **style_atts)
1073         node.append(new_node)
1074         return frac, string
1075 
1076     if name == '\\': # end of a row
1077         entry = mtd()
1078         new_node = mtr(entry)
1079         node.close().close().append(new_node)
1080         return entry, string
1081 
1082     if name in accents:
1083         new_node = mover(mo(accents[name][0], stretchy=False), switch=True)
1084         if name == 'vec':
1085             new_node.children[0]['accent'] = False # scale down arrow but drop i-dot
1086         new_node.tex_cmd = name # for HTML export
1087         node.append(new_node)
1088         return new_node, string
1089 
1090     if name in over:
1091         # set "accent" to False (otherwise dots on i and j are dropped)
1092         # but to True on accent node get "textstyle" (full size) symbols on top
1093         new_node = mover(mo(over[name][0], accent=True),
1094                          switch=True, accent=False)
1095         new_node.tex_cmd = name # for HTML export
1096         node.append(new_node)
1097         return new_node, string
1098 
1099     if name == 'overset':
1100         new_node = mover(switch=True)
1101         node.append(new_node)
1102         return new_node, string
1103 
1104     if name in under:
1105         new_node = munder(mo(under[name][0]), switch=True)
1106         new_node.tex_cmd = name # for HTML export
1107         node.append(new_node)
1108         return new_node, string
1109 
1110     if name == 'underset':
1111         new_node = munder(switch=True)
1112         node.append(new_node)
1113         return new_node, string
1114 
1115     if name in ('xleftarrow', 'xrightarrow'):
1116         subscript, string = tex_optarg(string)
1117         base = mo(operators['long'+name[1:]])
1118         if subscript:
1119             new_node = munderover(base)
1120             sub_node = parse_latex_math(mrow(), subscript)
1121             if len(sub_node) == 1:
1122                 sub_node = sub_node.children[0]
1123             new_node.append(sub_node)
1124         else:
1125             new_node = mover(base)
1126         node.append(new_node)
1127         return new_node, string
1128 
1129     if name in layout_styles: # 'displaystyle', 'textstyle', ...
1130         new_node = mstyle(**layout_styles[name])
1131         new_node.nchildren = None
1132         if isinstance(node, mrow) and len(node) == 0:
1133             # replace node with new_node
1134             node.parent.children[node.parent.children.index(node)] = new_node
1135             new_node.parent = node.parent
1136         elif node.__class__.__name__ == 'math':
1137             node.append(new_node)
1138         else:
1139             raise SyntaxError(u'Declaration "\\%s" must be first command '
1140                               u'in a group.' % name)
1141         return new_node, string
1142 
1143     if name.endswith('limits'):
1144         arg, remainder = tex_token(string)
1145         if arg in '_^': # else ignore
1146             string = remainder
1147             node = handle_script_or_limit(node, arg, limits=name)
1148         return node, string
1149 
1150     # Environments
1151 
1152     if name == 'begin':
1153         return begin_environment(node, string)
1154 
1155     if name == 'end':
1156         return end_environment(node, string)
1157 
1158     raise SyntaxError(u'Unknown LaTeX command: ' + name)

1195 def handle_script_or_limit(node, c, limits=''):
1196     """Append script or limit element to `node`."""
1197     child = node.children.pop()
1198     if limits == 'limits':
1199         child['movablelimits'] = False
1200     elif (limits == 'movablelimits'
1201           or getattr(child, 'data', '') in movablelimits):
1202         child['movablelimits'] = True
1203 
1204     if c == '_':
1205         if isinstance(child, mover):
1206             new_node = munderover(*child.children, switch=True)
1207         elif isinstance(child, msup):
1208             new_node = msubsup(*child.children, switch=True)
1209         elif (limits in ('limits', 'movablelimits')
1210               or limits == ''
1211               and child.get('movablelimits', None) == True):
1212             new_node = munder(child)
1213         else:
1214             new_node = msub(child)
1215     elif c == '^':
1216         if isinstance(child, munder):
1217             new_node = munderover(*child.children)
1218         elif isinstance(child, msub):
1219             new_node = msubsup(*child.children)
1220         elif (limits in ('limits', 'movablelimits')
1221               or limits == ''
1222               and child.get('movablelimits', None) == True):
1223             new_node = mover(child)
1224         else:
1225             new_node = msup(child)
1226     node.append(new_node)
1227     return new_node

1230 def begin_environment(node, string):
1231     name, string = tex_group(string)
1232     if name in matrices:
1233         left_delimiter = matrices[name][0]
1234         attributes = {}
1235         if left_delimiter:
1236             wrapper = mrow(mo(left_delimiter))
1237             if name == 'cases':
1238                 wrapper = mrow(mo(left_delimiter, rspace='0.17em'))
1239                 attributes['columnalign'] = 'left'
1240             node.append(wrapper)
1241             node = wrapper
1242         elif name == 'smallmatrix':
1243             attributes['rowspacing'] = '0.02em'
1244             attributes['columnspacing'] = '0.333em'
1245             wrapper = mstyle(scriptlevel='1')
1246             node.append(wrapper)
1247             node = wrapper
1248         # TODO: aligned, alignedat
1249         # take an optional [t], [b] or the default [c]
1250         entry = mtd()
1251         node.append(mtable(mtr(entry), **attributes))
1252         node = entry
1253     else:
1254         raise SyntaxError(u'Environment not supported!')
1255     return node, string

1258 def end_environment(node, string):
1259     name, string = tex_group(string)
1260     if name in matrices:
1261         node = node.close().close().close() # close: mtd, mdr, mtable
1262         right_delimiter = matrices[name][1]
1263         if right_delimiter:
1264             node = node.append(mo(right_delimiter))
1265             node = node.close()
1266         elif name == 'cases':
1267             node = node.close()
1268     else:
1269         raise SyntaxError(u'Environment not supported!')
1270     return node, string

1273 # Return the number of "equation_columns" in `code_lines`. cf. "alignat"
1274 # in http://mirror.ctan.org/macros/latex/required/amsmath/amsldoc.pdf
1275 def tex_equation_columns(rows):
1276     tabs = max(row.count('&') - row.count(r'\&') for row in rows)
1277     if tabs == 0:
1278         return 0
1279     return int(tabs/2 + 1)

1291 # Return dictionary with attributes to style an <mtable> as align environment:
1292 def align_attributes(rows):
1293     atts = {'class': 'align',
1294             'displaystyle': True}
1295     tabs = max(row.count('&') - row.count(r'\&') for row in rows)
1296     if tabs:
1297         aligns = ['right', 'left'] * tabs
1298         spacing = ['0', '2em'] * tabs
1299         atts['columnalign'] = ' '.join(aligns[:tabs+1])
1300         atts['columnspacing'] = ' '.join(spacing[:tabs])
1301     return atts

1313 def tex2mathml(tex_math, inline=True):
1314     """Return string with MathML code corresponding to `tex_math`. 
1315  
1316     Set `inline` to False for displayed math. 
1317     """
1318     # Set up tree
1319     math_tree = math(xmlns='http://www.w3.org/1998/Math/MathML')
1320     node = math_tree
1321     if not inline:
1322         math_tree['display'] = 'block'
1323         rows = toplevel_code(tex_math).split(r'\\')
1324         if len(rows) > 1:
1325             # emulate align* environment with a math table
1326             node = mtd()
1327             math_tree.append(mtable(mtr(node),
1328                                     **align_attributes(rows)))
1329     parse_latex_math(node, tex_math)
1330     return math_tree.toprettyxml()
