1    """Testing pasing object between multiple COM threads 
2     
3    Uses standard COM marshalling to pass objects between threads.  Even 
4    though Python generally seems to work when you just pass COM objects 
5    between threads, it shouldnt. 
6     
7    This shows the "correct" way to do it. 
8     
9    It shows that although we create new threads to use the Python.Interpreter, 
10   COM marshalls back all calls to that object to the main Python thread, 
11   which must be running a message loop (as this sample does). 
12    
13   When this test is run in "free threaded" mode (at this stage, you must 
14   manually mark the COM objects as "ThreadingModel=Free", or run from a 
15   service which has marked itself as free-threaded), then no marshalling 
16   is done, and the Python.Interpreter object start doing the "expected" thing 
17   - ie, it reports being on the same thread as its caller! 
18    
19   Python.exe needs a good way to mark itself as FreeThreaded - at the moment 
20   this is a pain in the but! 
21    
22   """
23   
24   import threading, traceback
25   import win32com.client
26   import win32event, win32api
27   import pythoncom
28   import unittest

30   from .testServers import InterpCase

32   freeThreaded = 1

35   class ThreadInterpCase(InterpCase):
36       def _testInterpInThread(self, stopEvent, interp):
37           try:
38               self._doTestInThread(interp)
39           finally:
40               win32event.SetEvent(stopEvent)
41   
42       def _doTestInThread(self, interp):
43           pythoncom.CoInitialize()
44           myThread = win32api.GetCurrentThreadId()
45   
46           if freeThreaded:
47               interp = pythoncom.CoGetInterfaceAndReleaseStream(
48                   interp, pythoncom.IID_IDispatch
49               )
50               interp = win32com.client.Dispatch(interp)
51   
52           interp.Exec("import win32api")
53           # print "The test thread id is %d, Python.Interpreter's thread ID is %d" % (myThread, interp.Eval("win32api.GetCurrentThreadId()"))
54           pythoncom.CoUninitialize()
55   
56       def BeginThreadsSimpleMarshal(self, numThreads):
57           """Creates multiple threads using simple (but slower) marshalling. 
58    
59           Single interpreter object, but a new stream is created per thread. 
60    
61           Returns the handles the threads will set when complete. 
62           """
63           interp = win32com.client.Dispatch("Python.Interpreter")
64           events = []
65           threads = []
66           for i in range(numThreads):
67               hEvent = win32event.CreateEvent(None, 0, 0, None)
68               events.append(hEvent)
69               interpStream = pythoncom.CoMarshalInterThreadInterfaceInStream(
70                   pythoncom.IID_IDispatch, interp._oleobj_
71               )
72               t = threading.Thread(
73                   target=self._testInterpInThread, args=(hEvent, interpStream)
74               )
75               t.setDaemon(1)  # so errors dont cause shutdown hang
76               t.start()
77               threads.append(t)
78           interp = None
79           return threads, events

85       def BeginThreadsFastMarshal(self, numThreads):
86           """Creates multiple threads using fast (but complex) marshalling. 
87    
88           The marshal stream is created once, and each thread uses the same stream 
89    
90           Returns the handles the threads will set when complete. 
91           """
92           interp = win32com.client.Dispatch("Python.Interpreter")
93           if freeThreaded:
94               interp = pythoncom.CoMarshalInterThreadInterfaceInStream(
95                   pythoncom.IID_IDispatch, interp._oleobj_
96               )
97           events = []
98           threads = []
99           for i in range(numThreads):
100              hEvent = win32event.CreateEvent(None, 0, 0, None)
101              t = threading.Thread(target=self._testInterpInThread, args=(hEvent, interp))
102              t.setDaemon(1)  # so errors dont cause shutdown hang
103              t.start()
104              events.append(hEvent)
105              threads.append(t)
106          return threads, events

108      def _DoTestMarshal(self, fn, bCoWait=0):
109          # print "The main thread is %d" % (win32api.GetCurrentThreadId())
110          threads, events = fn(2)
111          numFinished = 0
112          while 1:
113              try:
114                  if bCoWait:
115                      rc = pythoncom.CoWaitForMultipleHandles(0, 2000, events)
116                  else:
117                      # Specifying "bWaitAll" here will wait for messages *and* all events
118                      # (which is pretty useless)
119                      rc = win32event.MsgWaitForMultipleObjects(
120                          events, 0, 2000, win32event.QS_ALLINPUT
121                      )
122                  if (
123                      rc >= win32event.WAIT_OBJECT_0
124                      and rc < win32event.WAIT_OBJECT_0 + len(events)
125                  ):
126                      numFinished = numFinished + 1
127                      if numFinished >= len(events):
128                          break
129                  elif rc == win32event.WAIT_OBJECT_0 + len(events):  # a message
130                      # This is critical - whole apartment model demo will hang.
131                      pythoncom.PumpWaitingMessages()
132                  else:  # Timeout
133                      print(
134                          "Waiting for thread to stop with interfaces=%d, gateways=%d"
135                          % (pythoncom._GetInterfaceCount(), pythoncom._GetGatewayCount())
136                      )
137              except KeyboardInterrupt:
138                  break
139          for t in threads:
140              t.join(2)
141              self.failIf(t.is_alive(), "thread failed to stop!?")
142          threads = None  # threads hold references to args
143          # Seems to be a leak here I can't locate :(
144          # self.failUnlessEqual(pythoncom._GetInterfaceCount(), 0)
145          # self.failUnlessEqual(pythoncom._GetGatewayCount(), 0)
