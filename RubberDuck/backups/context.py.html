19   __all__ = ('Context', 'ProxyContext', 'register_context',
20              'get_current_context')
21   
22   _contexts = {}
23   _default_context = None
24   _context_stack = []
25   
26   
27   class ProxyContext(object):
28   
29       __slots__ = ['_obj']
30   
31       def __init__(self, obj):
32           object.__init__(self)
33           object.__setattr__(self, '_obj', obj)
34   
35       def __getattribute__(self, name):
36           return getattr(object.__getattribute__(self, '_obj'), name)
37   
38       def __delattr__(self, name):
39           delattr(object.__getattribute__(self, '_obj'), name)
40   
41       def __setattr__(self, name, value):
42           setattr(object.__getattribute__(self, '_obj'), name, value)
43   
44       def __bool__(self):
45           return bool(object.__getattribute__(self, '_obj'))
46   
47       def __str__(self):
48           return str(object.__getattribute__(self, '_obj'))
49   
50       def __repr__(self):
51           return repr(object.__getattribute__(self, '_obj'))
52   
53   
54   class Context(dict):
55   
56       def __init__(self, init=False):
57           dict.__init__(self)
58           self.sandbox = None
59           if not init:
60               return
61   
62           for name in _contexts:
63               context = _contexts[name]
64               instance = context['cls'](*context['args'], **context['kwargs'])
65               self[name] = instance
66   
67       def push(self):
68           _context_stack.append(self)
69           for name, instance in self.items():
70               object.__setattr__(_contexts[name]['proxy'], '_obj', instance)
71   
72       def pop(self):
73           # After poping context from stack. Update proxy's _obj with
74           # instances in current context
75           _context_stack.pop(-1)
76           for name, instance in get_current_context().items():
77               object.__setattr__(_contexts[name]['proxy'], '_obj', instance)
78   
79   
80   def register_context(name, cls, *args, **kwargs):
81       '''Register a new context. 
82       '''
83       instance = cls(*args, **kwargs)
84       proxy = ProxyContext(instance)
85       _contexts[name] = {
86           'cls': cls,
87           'args': args,
88           'kwargs': kwargs,
89           'proxy': proxy}
90       _default_context[name] = instance
91       return proxy
92   
93   
94   def get_current_context():
95       '''Return the current context. 
96       '''
97       if not _context_stack:
98           return _default_context
99       return _context_stack[-1]
100  
101  
102  _default_context = Context(init=False)
