1    ''' 
2    Multistroke gesture recognizer 
3    ============================== 
4     
5    .. versionadded:: 
6        1.9.0 
7     
8    .. warning:: 
9     
10       This is experimental and subject to change as long as this warning notice 
11       is present. 
12    
13   See :file:`kivy/examples/demo/multistroke/main.py` for a complete application 
14   example. 
15    
16   Conceptual Overview 
17   ------------------- 
18    
19   This module implements the Protractor gesture recognition algorithm. 
20    
21   :class:`Recognizer` is the search/database API similar to 
22   :class:`~kivy.gesture.GestureDatabase`. It maintains a list of 
23   :class:`MultistrokeGesture` objects and allows you to search for a 
24   user-input gestures among them. 
25    
26   :class:`ProgressTracker` tracks the progress of a :meth:`Recognizer.recognize` 
27   call. It can be used to interact with the running recognizer task, for example 
28   forcing it to stop half-way, or analyzing results as they arrive. 
29    
30   :class:`MultistrokeGesture` represents a gesture in the gesture database 
31   (:attr:`Recognizer.db`). It is a container for :class:`UnistrokeTemplate` 
32   objects, and implements the heap permute algorithm to automatically generate 
33   all possible stroke orders (if desired). 
34    
35   :class:`UnistrokeTemplate` represents a single stroke path. It's typically 
36   instantiated automatically by :class:`MultistrokeGesture`, but sometimes you 
37   may need to create them manually. 
38    
39   :class:`Candidate` represents a user-input gesture that is used to search 
40   the gesture database for matches. It is normally instantiated automatically 
41   by calling :meth:`Recognizer.recognize`. 
42    
43   Usage examples 
44   -------------- 
45    
46   See :file:`kivy/examples/demo/multistroke/main.py` for a complete application 
47   example. 
48    
49   You can bind to events on :class:`Recognizer` to track the state of all 
50   calls to :meth:`Recognizer.recognize`. The callback function will receive an 
51   instance of :class:`ProgressTracker` that can be used to analyze and control 
52   various aspects of the recognition process :: 
53    
54       from kivy.vector import Vector 
55       from kivy.multistroke import Recognizer 
56    
57       gdb = Recognizer() 
58    
59       def search_start(gdb, pt): 
60           print("A search is starting with %d tasks" % (pt.tasks)) 
61    
62       def search_stop(gdb, pt): 
63           # This will call max() on the result dictionary, so it's best to store 
64           # it instead of calling it 3 times consecutively 
65           best = pt.best 
66           print("Search ended (%s). Best is %s (score %f, distance %f)" % ( 
67               pt.status, best['name'], best['score'], best['dist'] )) 
68    
69       # Bind your callbacks to track all matching operations 
70       gdb.bind(on_search_start=search_start) 
71       gdb.bind(on_search_complete=search_stop) 
72    
73       # The format below is referred to as `strokes`, a list of stroke paths. 
74       # Note that each path shown here consists of two points, ie a straight 
75       # line; if you plot them it looks like a T, hence the name. 
76       gdb.add_gesture('T', [ 
77           [Vector(30, 7), Vector(103, 7)], 
78           [Vector(66, 7), Vector(66, 87)]]) 
79    
80       # Now you can search for the 'T' gesture using similar data (user input). 
81       # This will trigger both of the callbacks bound above. 
82       gdb.recognize([ 
83           [Vector(45, 8), Vector(110, 12)], 
84           [Vector(88, 9), Vector(85, 95)]]) 
85    
86   On the next :class:`~kivy.clock.Clock` tick, the matching process starts 
87   (and, in this case, completes). 
88    
89   To track individual calls to :meth:`Recognizer.recognize`, use the return 
90   value (also a :class:`ProgressTracker` instance) :: 
91    
92       # Same as above, but keep track of progress using returned value 
93       progress = gdb.recognize([ 
94           [Vector(45, 8), Vector(110, 12)], 
95           [Vector(88, 9), Vector(85, 95)]]) 
96    
97       progress.bind(on_progress=my_other_callback) 
98       print(progress.progress) # = 0 
99    
100      # [ assuming a kivy.clock.Clock.tick() here ] 
101   
102      print(result.progress) # = 1 
103   
104  Algorithm details 
105  ----------------- 
106   
107  For more information about the matching algorithm, see: 
108   
109  "Protractor: A fast and accurate gesture recognizer" by Yang Li 
110    http://yangl.org/pdf/protractor-chi2010.pdf 
111   
112  "$N-Protractor" by Lisa Anthony and Jacob O. Wobbrock 
113    http://depts.washington.edu/aimgroup/proj/dollar/ndollar-protractor.pdf 
114   
115  Some of the code is derived from the JavaScript implementation here: 
116    http://depts.washington.edu/aimgroup/proj/dollar/ndollar.html 
117  '''

157  class Recognizer(EventDispatcher):
158      ''':class:`Recognizer` provides a gesture database with matching 
159      facilities. 
160   
161      :Events: 
162          `on_search_start` 
163              Fired when a new search is started using this Recognizer. 
164   
165          `on_search_complete` 
166              Fired when a running search ends, for whatever reason. 
167              (use :data:`ProgressTracker.status` to find out) 
168   
169      :Properties: 
170          `db` 
171              A :class:`ListProperty` that contains the available 
172              :class:`MultistrokeGesture` objects. 
173   
174              :attr:`db` is a 
175              :class:`~kivy.properties.ListProperty` and defaults to [] 
176      '''
177  
178      db = ListProperty([])

181          super(Recognizer, self).__init__(**kwargs)
182          self.register_event_type('on_search_start')
183          self.register_event_type('on_search_complete')

185      def filter(self, **kwargs):
186          ''':meth:`filter` returns a subset of objects in :attr:`self.db`, 
187          according to given criteria. This is used by many other methods of 
188          the :class:`Recognizer`; the arguments below can for example be 
189          used when calling :meth:`Recognizer.recognize` or 
190          :meth:`Recognizer.export_gesture`. You normally don't need to call 
191          this directly. 
192   
193          :Arguments: 
194   
195              `name` 
196                  Limits the returned list to gestures where 
197                  :attr:`MultistrokeGesture.name` matches given regular 
198                  expression(s). If re.match(name, MultistrokeGesture.name) 
199                  tests true, the gesture is included in the returned list. 
200                  Can be a string or an array of strings :: 
201   
202                      gdb = Recognizer() 
203   
204                      # Will match all names that start with a capital N 
205                      # (ie Next, New, N, Nebraska etc, but not "n" or "next") 
206                      gdb.filter(name='N') 
207   
208                      # exactly 'N' 
209                      gdb.filter(name='N$') 
210   
211                      # Nebraska, teletubbies, France, fraggle, N, n, etc 
212                      gdb.filter(name=['[Nn]', '(?i)T', '(?i)F']) 
213   
214              `priority` 
215                  Limits the returned list to gestures with certain 
216                  :attr:`MultistrokeGesture.priority` values. If specified as an 
217                  integer, only gestures with a lower priority are returned. If 
218                  specified as a list (min/max) :: 
219   
220                      # Max priority 50 
221                      gdb.filter(priority=50) 
222   
223                      # Max priority 50 (same result as above) 
224                      gdb.filter(priority=[0, 50]) 
225   
226                      # Min priority 50, max 100 
227                      gdb.filter(priority=[50, 100]) 
228   
229                  When this option is used, :attr:`Recognizer.db` is 
230                  automatically sorted according to priority, incurring extra 
231                  cost. You can use `force_priority_sort` to override this 
232                  behavior if your gestures are already sorted according to 
233                  priority. 
234   
235              `orientation_sensitive` 
236                  Limits the returned list to gestures that are 
237                  orientation sensitive (True), gestures that are not orientation 
238                  sensitive (False) or None (ignore template sensitivity, this is 
239                  the default). 
240   
241              `numstrokes` 
242                  Limits the returned list to gestures that have the specified 
243                  number of strokes (in :attr:`MultistrokeGesture.strokes`). 
244                  Can be a single integer or a list of integers. 
245   
246              `numpoints` 
247                  Limits the returned list to gestures that have specific 
248                  :attr:`MultistrokeGesture.numpoints` values. This is provided 
249                  for flexibility, do not use it unless you understand what it 
250                  does. Can be a single integer or a list of integers. 
251   
252              `force_priority_sort` 
253                  Can be used to override the default sort behavior. Normally 
254                  :class:`MultistrokeGesture` objects are returned in priority 
255                  order if the `priority` option is used. Setting this to True 
256                  will return gestures sorted in priority order, False will 
257                  return in the order gestures were added. None means decide 
258                  automatically (the default). 
259   
260                  .. Note :: 
261                      For improved performance, you can load your gesture 
262                      database in priority order and set this to False when 
263                      calling :meth:`Recognizer.recognize` 
264   
265              `db` 
266                  Can be set if you want to filter a different list of objects 
267                  than :attr:`Recognizer.db`. You probably don't want to do this; 
268                  it is used internally by :meth:`import_gesture`. 
269          '''
270          have_filters = False
271  
272          kwargs_get = kwargs.get
273  
274          name = kwargs_get('name', None)
275          if name is not None:
276              have_filters = True
277              if not isinstance(name, list):
278                  name = [name]
279  
280          priority = kwargs_get('priority', None)
281          min_p, max_p = None, None
282          if priority is not None:
283              have_filters = True
284              if isinstance(priority, list):
285                  min_p, max_p = priority
286              elif isinstance(priority, int):
287                  min_p, max_p = None, priority
288  
289          numstrokes = kwargs_get('numstrokes', None)
290          if numstrokes is not None:
291              have_filters = True
292              if not isinstance(numstrokes, list):
293                  numstrokes = [numstrokes]
294  
295          numpoints = kwargs_get('numpoints', None)
296          if numpoints is not None:
297              have_filters = True
298              if not isinstance(numpoints, list):
299                  numpoints = [numpoints]
300  
301          orientation_sens = kwargs_get('orientation_sensitive', None)
302          if orientation_sens is not None:
303              have_filters = True
304  
305          # Prepare a correctly sorted tasklist
306          force_priority_sort = kwargs.get('force_priority_sort', None)
307          force_sort_on = force_priority_sort and True
308          force_sort_off = (force_priority_sort is False) and True
309  
310          db = kwargs.get('db', None) or self.db
311          if (force_sort_on or priority) and not force_sort_off:
312              tasklist = sorted(db, key=lambda n: n.priority)
313          else:
314              tasklist = db
315  
316          # Now test each gesture in the database against filter criteria
317          out = deque()
318          if not have_filters:
319              out.extend(tasklist)
320              return out
321  
322          out_append = out.append
323          for gesture in tasklist:
324  
325              if (orientation_sens is not None and
326                      orientation_sens != gesture.orientation_sens):
327                  continue
328  
329              if numpoints and gesture.numpoints not in numpoints:
330                  continue
331  
332              if numstrokes and len(gesture.strokes) not in numstrokes:
333                  continue
334  
335              if min_p is not None and gesture.priority < min_p:
336                  continue
337  
338              if max_p is not None and gesture.priority > max_p:
339                  return out
340  
341              if name:
342                  for f in name:
343                      if re_match(f, gesture.name):
344                          out_append(gesture)
345                          break
346              else:
347                  out_append(gesture)
348  
349          return out

351      def add_gesture(self, name, strokes, **kwargs):
352          '''Add a new gesture to the database. This will instantiate a new 
353          :class:`MultistrokeGesture` with `strokes` and append it to self.db. 
354   
355          .. Note :: 
356              If you already have instantiated a :class:`MultistrokeGesture` 
357              object and wish to add it, append it to :attr:`Recognizer.db` 
358              manually. 
359          '''
360          if not strokes:
361              return False
362          self.db.append(
363              MultistrokeGesture(name=name, strokes=strokes, **kwargs)
364          )
365          return True

367      def parse_gesture(self, data):
368          '''Parse data formatted by export_gesture(). Returns a list of 
369          :class:`MultistrokeGesture` objects. This is used internally by 
370          :meth:`import_gesture`, you normally don't need to call this 
371          directly.'''
372          io = BytesIO(zlib.decompress(base64.b64decode(data)))
373  
374          p = pickle.Unpickler(io)
375          multistrokes = []
376          ms_append = multistrokes.append
377          for multistroke in p.load():
378              strokes = multistroke['strokes']
379              multistroke['strokes'] = [[Vector(
380                  x, y) for x, y in line] for line in strokes]
381              ms_append(MultistrokeGesture(**multistroke))
382          return multistrokes

384      # FIXME: use a try block, maybe shelve or something
385      def export_gesture(self, filename=None, **kwargs):
386          '''Export a list of :class:`MultistrokeGesture` objects. Outputs a 
387          base64-encoded string that can be decoded to a Python list with 
388          the :meth:`parse_gesture` function or imported directly to 
389          :attr:`self.db` using :meth:`Recognizer.import_gesture`. If 
390          `filename` is specified, the output is written to disk, otherwise 
391          returned. 
392   
393          This method accepts optional :meth:`Recognizer.filter` arguments. 
394          '''
395          io = BytesIO()
396          p = pickle.Pickler(io, protocol=0)
397          multistrokes = []
398          defaults = {'priority': 100, 'numpoints': 16, 'stroke_sens': True,
399                      'orientation_sens': False, 'angle_similarity': 30.0}
400          dkeys = defaults.keys()
401  
402          for multistroke in self.filter(**kwargs):
403              m = dict(defaults)
404              m = {'name': multistroke.name}
405              for attr in dkeys:
406                  m[attr] = getattr(multistroke, attr)
407              m['strokes'] = tuple([(p.x, p.y) for p in line]
408                                   for line in multistroke.strokes)
409              multistrokes.append(m)
410          p.dump(multistrokes)
411  
412          if filename:
413              f = open(filename, 'wb')
414              f.write(base64.b64encode(zlib.compress(io.getvalue(), 9)))
415              f.close()
416          else:
417              return base64.b64encode(zlib.compress(io.getvalue(), 9))

437          if new:
438              self.db.extend(new)

440      def transfer_gesture(self, tgt, **kwargs):
441          '''Transfers :class:`MultistrokeGesture` objects from 
442          :attr:`Recognizer.db` to another :class:`Recognizer` instance `tgt`. 
443   
444          This method accepts optional :meth:`Recognizer.filter` arguments. 
445          '''
446          if hasattr(tgt, 'db') and isinstance(tgt.db, list):
447              send = self.filter(**kwargs)
448              if send:
449                  tgt.db.append(None)
450                  tgt.db[-1:] = send
451                  return True

453      def prepare_templates(self, **kwargs):
454          '''This method is used to prepare :class:`UnistrokeTemplate` objects 
455          within the gestures in self.db. This is useful if you want to minimize 
456          punishment of lazy resampling by preparing all vectors in advance. If 
457          you do this before a call to :meth:`Recognizer.export_gesture`, you 
458          will have the vectors computed when you load the data later. 
459   
460          This method accepts optional :meth:`Recognizer.filter` arguments. 
461   
462          `force_numpoints`, if specified, will prepare all templates to the 
463          given number of points (instead of each template's preferred n; ie 
464          :data:`UnistrokeTemplate.numpoints`). You normally don't want to 
465          do this.'''
466          for gesture in self.filter(**kwargs):
467              for tpl in gesture:
468                  n = kwargs.get('force_numpoints', tpl.numpoints)
469                  tpl.prepare(n)

471      def recognize(self, strokes, goodscore=None, timeout=0, delay=0, **kwargs):
472          '''Search for gestures matching `strokes`. Returns a 
473          :class:`ProgressTracker` instance. 
474   
475          This method accepts optional :meth:`Recognizer.filter` arguments. 
476   
477          :Arguments: 
478   
479              `strokes` 
480                  A list of stroke paths (list of lists of 
481                  :class:`~kivy.vector.Vector` objects) that will be matched 
482                  against gestures in the database. Can also be a 
483                  :class:`Candidate` instance. 
484   
485                  .. Warning :: 
486   
487                      If you manually supply a :class:`Candidate` that has a 
488                      skip-flag, make sure that the correct filter arguments 
489                      are set. Otherwise the system will attempt to load vectors 
490                      that have not been computed. For example, if you set 
491                      `skip_bounded` and do not set `orientation_sensitive` to 
492                      False, it will raise an exception if an 
493                      orientation_sensitive :class:`UnistrokeTemplate` 
494                      is encountered. 
495   
496              `goodscore` 
497                  If this is set (between 0.0 - 1.0) and a gesture score is 
498                  equal to or higher than the specified value, the search is 
499                  immediately halted and the on_search_complete event is 
500                  fired (+ the on_complete event of the associated 
501                  :class:`ProgressTracker` instance). Default is None (disabled). 
502   
503              `timeout` 
504                  Specifies a timeout (in seconds) for when the search is 
505                  aborted and the results returned. This option applies only 
506                  when `max_gpf` is not 0. Default value is 0, meaning all 
507                  gestures in the database will be tested, no matter how long 
508                  it takes. 
509   
510              `max_gpf` 
511                  Specifies the maximum number of :class:`MultistrokeGesture` 
512                  objects that can be processed per frame. When exceeded, will 
513                  cause the search to halt and resume work in the next frame. 
514                  Setting to 0 will complete the search immediately (and block 
515                  the UI). 
516   
517                  .. Warning :: 
518   
519                      This does not limit the number of 
520                      :class:`UnistrokeTemplate` objects matched! If a single 
521                      gesture has a million templates, they will all be 
522                      processed in a single frame with max_gpf=1! 
523   
524              `delay` 
525                  Sets an optional delay between each run of the recognizer 
526                  loop. Normally, a run is scheduled for the next frame until 
527                  the tasklist is exhausted. If you set this, there will be an 
528                  additional delay between each run (specified in seconds). 
529                  Default is 0, resume in the next frame. 
530   
531              `force_numpoints` 
532                  forces all templates (and candidate) to be prepared to a 
533                  certain number of points. This can be useful for example if 
534                  you are evaluating templates for optimal n (do not use this 
535                  unless you understand what it does). 
536          '''
537          GPF = kwargs.get('max_gpf', DEFAULT_GPF)
538  
539          # Obtain a list of MultistrokeGesture objects matching filter arguments
540          tasklist = self.filter(**kwargs)
541  
542          # Initialize the candidate and result objects
543          cand = self._candidate(strokes)
544          result = ProgressTracker(cand, len(tasklist))
545  
546          # This is done to inform caller if they bind to on_complete and there
547          # is nothing to do; perhaps should just return None?
548          if not tasklist:
549              result.status = 'complete'
550              self.dispatch('on_search_complete', result)
551  
552              def result_hack(dt):
553                  result.dispatch('on_complete')
554              Clock.schedule_once(result_hack)
555              return result
556  
557          # This callback is scheduled once per frame until completed
558          def _recognize_tick(dt):
559              start_gc = result._completed
560              stop_now = False
561  
562              while not stop_now and (tasklist and not result._break_flag) and \
563                      (not GPF or (result._completed - start_gc < GPF)):
564  
565                  if (timeout and
566                          Clock.get_time() - result._start_time >= timeout):
567                      result.status = 'timeout'
568                      stop_now = True
569                      break
570  
571                  # Get the best distance and number of matching operations done
572                  gesture = tasklist.popleft()
573                  tpl, d, res, mos = gesture.match_candidate(
574                      cand, **kwargs)
575  
576                  if tpl is not None:
577                      score = result._add_result(gesture, d, tpl, res)
578                      if goodscore is not None and score >= goodscore:
579                          result.status = 'goodscore'
580                          stop_now = True
581  
582                  result._match_ops += mos
583                  result._completed += 1
584                  result.dispatch('on_progress')
585  
586              # The loop has ended. Prepare to dispatch 'complete'
587              def _dispatch():
588                  result.dispatch('on_complete')
589                  self.dispatch('on_search_complete', result)
590                  return False
591  
592              # Dispatch or reschedule another run
593              if not tasklist:
594                  result.status = 'complete'
595                  return _dispatch()
596              elif result._break_flag:
597                  result.status = 'stop'
598                  return _dispatch()
599              elif stop_now:
600                  return _dispatch()
601              else:
602                  Clock.schedule_once(_recognize_tick, delay)
603                  return True
604          # End _recognize_tick()
605  
606          self.dispatch('on_search_start', result)
607          if not GPF:
608              _recognize_tick(0)
609          else:
610              Clock.schedule_once(_recognize_tick, 0)
611  
612          return result
613  
614      def _candidate(self, strokes, **kwargs):
615          # recognize() helper function, do not use directly. Set up a
616          # Candidate object from arguments. Either use a specified object
617          # or make a new one from strokes and apply safe skip_* settings to
618          # use less resources.
619          if isinstance(strokes, Candidate):
620              return strokes
621  
622          if (not isinstance(strokes, list) or not len(strokes) or not
623                  isinstance(strokes[0], list)):
624              raise MultistrokeError('recognize() needs strokes= '
625                                     'list or Candidate')
626  
627          cand = Candidate(strokes)
628          o_filter = kwargs.get('orientation_sensitive', None)
629          if o_filter is False:
630              cand.skip_bounded = True
631          elif o_filter is True:
632              cand.skip_invariant = True
633  
634          return cand

648  class ProgressTracker(EventDispatcher):
649      '''Represents an ongoing (or completed) search operation. Instantiated and 
650      returned by the :meth:`Recognizer.recognize` method when it is called. The 
651      `results` attribute is a dictionary that is  updated as the recognition 
652      operation progresses. 
653   
654      .. Note :: 
655          You do not need to instantiate this class. 
656   
657      :Arguments: 
658          `candidate` 
659              :class:`Candidate` object to be evaluated 
660          `tasks` 
661              Total number of gestures in tasklist (to test against) 
662   
663      :Events: 
664          `on_progress` 
665              Fired for every gesture that is processed 
666          `on_result` 
667              Fired when a new result is added, and it is the first match 
668              for the `name` so far, or a consecutive match with better score. 
669          `on_complete` 
670              Fired when the search is completed, for whatever reason. 
671              (use `ProgressTracker.status` to find out) 
672   
673      :Attributes: 
674          `results` 
675              A dictionary of all results (so far). The key is the name of the 
676              gesture (ie :attr:`UnistrokeTemplate.name` usually inherited from 
677              :class:`MultistrokeGesture`). Each item in the dictionary is a 
678              dict with the following entries: 
679   
680                  `name` 
681                      Name of the matched template (redundant) 
682                  `score` 
683                      Computed score from 1.0 (perfect match) to 0.0 
684                  `dist` 
685                      Cosine distance from candidate to template (low=closer) 
686                  `gesture` 
687                      The :class:`MultistrokeGesture` object that was matched 
688                  `best_template` 
689                      Index of the best matching template (in 
690                      :attr:`MultistrokeGesture.templates`) 
691                  `template_results` 
692                      List of distances for all templates. The list index 
693                      corresponds to a :class:`UnistrokeTemplate` index in 
694                      gesture.templates. 
695   
696          `status` 
697              `search` 
698                  Currently working 
699              `stop` 
700                  Was stopped by the user (:meth:`stop` called) 
701              `timeout` 
702                  A timeout occurred (specified as `timeout=` to recognize()) 
703              `goodscore` 
704                  The search was stopped early because a gesture with a high 
705                  enough score was found (specified as `goodscore=` to 
706                  recognize()) 
707              `complete` 
708                  The search is complete (all gestures matching filters were 
709                  tested) 
710      '''
711      def __init__(self, candidate, tasks, **kwargs):
712          self.status = 'search'
713          self.candidate = candidate
714          self.results = {}
715          self.tasks = tasks
716          self._start_time = Clock.get_time()
717          self._match_ops = 0
718          self._completed = 0
719          self._break_flag = False
720  
721          # fired by recognize()
722          self.register_event_type('on_complete')
723          self.register_event_type('on_progress')
724  
725          # fired locally
726          self.register_event_type('on_result')
727          super(ProgressTracker, self).__init__(**kwargs)

730      def progress(self):
731          '''Returns the progress as a float, 0 is 0% done, 1 is 100%. This 
732          is a Python property.'''
733          if not self.tasks:
734              return 1
735          return self._completed / float(self.tasks)

739          '''Return the best match found by recognize() so far. It returns a 
740          dictionary with three keys, 'name', 'dist' and 'score' representing 
741          the template's name, distance (from candidate path) and the 
742          computed score value. This is a Python property.'''
743          results = self.results  # to avoid too many self. lookups
744          if not results:
745              return {'name': None, 'dist': None, 'score': 0}
746          b = max(results, key=lambda r: results[r]['score'])
747          return {
748              'name': results[b]['name'],
749              'dist': results[b]['dist'],
750              'score': results[b]['score']
751          }

756          self._break_flag = True

758      def _add_result(self, gesture, dist, tpl, res):
759          # Add a result; used internally by the recognize() function
760          if tpl <= len(res):
761              n = gesture.templates[tpl].name
762          else:
763              return 0.
764  
765          if n not in self.results or dist < self.results[n]['dist']:
766              self.results[n] = {
767                  'name': n,
768                  'dist': dist,
769                  'gesture': gesture,
770                  'best_template': tpl,
771                  'template_results': res
772              }
773  
774              if not dist:
775                  self.results[n]['score'] = 1.0
776              else:
777                  self.results[n]['score'] = 1.0 - (dist / pi)
778  
779              self.dispatch('on_result', self.results[n])
780              return self.results[n]['score']
781          else:
782              return 0.

798  class MultistrokeGesture(object):
799      ''':class:`MultistrokeGesture` represents a gesture. It maintains a set of 
800      `strokes` and generates unistroke (ie :class:`UnistrokeTemplate`) 
801      permutations that are used for evaluating candidates against this gesture 
802      later. 
803   
804      :Arguments: 
805          `name` 
806              Identifies the name of the gesture - it is returned to you in the 
807              results of a :meth:`Recognizer.recognize` search. You can have any 
808              number of MultistrokeGesture objects with the same name; many 
809              definitions of one gesture. The same name is given to all the 
810              generated unistroke permutations. Required, no default. 
811          `strokes` 
812              A list of paths that represents the gesture. A path is a list of 
813              Vector objects:: 
814   
815                  gesture = MultistrokeGesture('my_gesture', strokes=[ 
816                    [Vector(x1, y1), Vector(x2, y2), ...... ], # stroke 1 
817                    [Vector(), Vector(), Vector(), Vector() ]  # stroke 2 
818                    #, [stroke 3], [stroke 4], ... 
819                  ]) 
820   
821              For template matching purposes, all the strokes are combined to a 
822              single list (unistroke). You should still specify the strokes 
823              individually, and set `stroke_sensitive` True (whenever possible). 
824   
825              Once you do this, unistroke permutations are immediately generated 
826              and stored in `self.templates` for later, unless you set the 
827              `permute` flag to False. 
828          `priority` 
829              Determines when :func:`Recognizer.recognize` will attempt to match 
830              this template, lower priorities are evaluated first (only if 
831              a priority `filter` is used). You should use lower priority on 
832              gestures that are more likely to match. For example, set user 
833              templates at lower number than generic templates. Default is 100. 
834          `numpoints` 
835              Determines the number of points this gesture should be resampled to 
836              (for matching purposes). The default is 16. 
837          `stroke_sensitive` 
838              Determines if the number of strokes (paths) in this gesture is 
839              required to be the same in the candidate (user input) gesture 
840              during matching. If this is False, candidates will always be 
841              evaluated, disregarding the number of strokes. Default is True. 
842          `orientation_sensitive` 
843              Determines if this gesture is orientation sensitive. If True, 
844              aligns the indicative orientation with the one of eight base 
845              orientations that requires least rotation. Default is True. 
846          `angle_similarity` 
847              This is used by the :func:`Recognizer.recognize` function when a 
848              candidate is evaluated against this gesture. If the angles between 
849              them are too far off, the template is considered a non-match. 
850              Default is 30.0 (degrees) 
851          `permute` 
852              If False, do not use Heap Permute algorithm to generate different 
853              stroke orders when instantiated. If you set this to False, a 
854              single UnistrokeTemplate built from `strokes` is used. 
855      '''
856      def __init__(self, name, strokes=None, **kwargs):
857          self.name = name
858          self.priority = kwargs.get('priority', 100)
859          self.numpoints = kwargs.get('numpoints', 16)
860          self.stroke_sens = kwargs.get('stroke_sensitive', True)
861          self.orientation_sens = kwargs.get('orientation_sensitive', True)
862          self.angle_similarity = kwargs.get('angle_similarity', 30.0)
863          self.strokes = []
864  
865          if strokes is not None:
866              self.strokes = strokes
867              if kwargs.get('permute', True):
868                  self.permute()
869              else:
870                  self.templates = [UnistrokeTemplate(name,
871                                    points=[i for sub in strokes for i in sub],
872                                    numpoints=self.numpoints,
873                                    orientation_sensitive=self.orientation_sens)]
874  
875      def angle_similarity_threshold(self):
876          return radians(self.angle_similarity)
877  
878      def add_stroke(self, stroke, permute=False):
879          '''Add a stroke to the self.strokes list. If `permute` is True, the 
880          :meth:`permute` method is called to generate new unistroke templates'''
881          self.strokes.append(stroke)
882          if permute:
883              self.permute()

885      def get_distance(self, cand, tpl, numpoints=None):
886          '''Compute the distance from this Candiate to a UnistrokeTemplate. 
887          Returns the Cosine distance between the stroke paths. 
888   
889          `numpoints` will prepare both the UnistrokeTemplate and Candidate path 
890          to n points (when necessary), you probably don't want to do this. 
891          '''
892          n = numpoints
893          if n is None or n < 2:
894              n = self.numpoints
895  
896          # optimal_cosine_distance() inlined here for performance
897          v1 = tpl.get_vector(n)
898          v2 = cand.get_protractor_vector(n, tpl.orientation_sens)
899  
900          a = 0.0
901          b = 0.0
902  
903          for i in xrange(0, len(v1), 2):
904              a += (v1[i] * v2[i]) + (v1[i + 1] * v2[i + 1])
905              b += (v1[i] * v2[i + 1]) - (v1[i + 1] * v2[i])
906  
907          angle = atan(b / a)
908  
909          # If you put the below directly into math.acos(), you will get a domain
910          # error when a=1.0 and angle=0.0 (ie math_cos(angle)=1.0). It seems to
911          # be because float representation of 1.0*1.0 is >1.0 (ie 1.00000...001)
912          # and this is problematic for math.acos().
913          # If you try math.acos(1.0*1.0) in interpreter it does not happen,
914          # only with exact match at runtime
915          result = a * math_cos(angle) + b * math_sin(angle)
916  
917          # FIXME: I'm sure there is a better way to do it but..
918          if result >= 1:
919              result = 1
920          elif result <= -1:  # has not happened to me, but I leave it here.
921              result = -1
922          return acos(result)
923  
924      def match_candidate(self, cand, **kwargs):
925          '''Match a given candidate against this MultistrokeGesture object. Will 
926          test against all templates and report results as a list of four 
927          items: 
928   
929              `index 0` 
930                  Best matching template's index (in self.templates) 
931              `index 1` 
932                  Computed distance from the template to the candidate path 
933              `index 2` 
934                  List of distances for all templates. The list index 
935                  corresponds to a :class:`UnistrokeTemplate` index in 
936                  self.templates. 
937              `index 3` 
938                  Counter for the number of performed matching operations, ie 
939                  templates matched against the candidate 
940          '''
941          best_d = float('infinity')
942          best_tpl = None
943          mos = 0
944          out = []
945  
946          if (self.stroke_sens and len(self.strokes) != len(cand.strokes)):
947              return (best_tpl, best_d, out, mos)
948  
949          skip_bounded = cand.skip_bounded
950          skip_invariant = cand.skip_invariant
951          get_distance = self.get_distance
952          ang_sim_threshold = self.angle_similarity_threshold()
953  
954          for idx, tpl in enumerate(self.templates):
955              # Handle a theoretical case where a MultistrokeGesture is composed
956              # manually and the orientation_sensitive flag is True, and contains
957              # a UnistrokeTemplate that has orientation_sensitive=False (or vice
958              # versa). This would cause KeyError - requesting nonexistant vector
959              if tpl.orientation_sens:
960                  if skip_bounded:
961                      continue
962              elif skip_invariant:
963                  continue
964  
965              # Count as a match operation now, since the call to get_
966              # angle_similarity below will force vector calculation,
967              # even if it doesn't make it to get_distance
968              mos += 1
969  
970              # Note: With this implementation, we always resample the candidate
971              # to *any* encountered UnistrokeTemplate numpoints here, the filter
972              # is only applied to MultistrokeGesture. See theoretical case
973              # above; should not matter normally.
974              n = kwargs.get('force_numpoints', tpl.numpoints)
975  
976              # Skip if candidate/gesture angles are too far off
977              ang_sim = cand.get_angle_similarity(tpl, numpoints=n)
978              if ang_sim > ang_sim_threshold:
979                  continue
980  
981              # Get the distance between cand/tpl paths
982              d = get_distance(cand, tpl, numpoints=n)
983              out.append(d)
984  
985              if d < best_d:
986                  best_d = d
987                  best_tpl = idx
988  
989          return (best_tpl, best_d, out, mos)
990  
991      def permute(self):
992          '''Generate all possible unistroke permutations from self.strokes and 
993          save the resulting list of UnistrokeTemplate objects in self.templates. 
994   
995          Quote from http://faculty.washington.edu/wobbrock/pubs/gi-10.2.pdf :: 
996   
997              We use Heap Permute [16] (p. 179) to generate all stroke orders 
998              in a multistroke gesture. Then, to generate stroke directions for 
999              each order, we treat each component stroke as a dichotomous 
1000             [0,1] variable. There are 2^N combinations for N strokes, so we 
1001             convert the decimal values 0 to 2^N-1, inclusive, to binary 
1002             representations and regard each bit as indicating forward (0) or 
1003             reverse (1). This algorithm is often used to generate truth tables 
1004             in propositional logic. 
1005  
1006         See section 4.1: "$N Algorithm" of the linked paper for details. 
1007  
1008         .. Warning :: 
1009  
1010             Using heap permute for gestures with more than 3 strokes 
1011             can result in very large number of templates (a 9-stroke 
1012             gesture = 38 million templates). If you are dealing with 
1013             these types of gestures, you should manually compose 
1014             all the desired stroke orders. 
1015         '''
1016         # Seed with index of each stroke
1017         self._order = [i for i in xrange(0, len(self.strokes))]
1018 
1019         # Prepare ._orders
1020         self._orders = []
1021         self._heap_permute(len(self.strokes))
1022         del self._order
1023 
1024         # Generate unistroke permutations
1025         self.templates = [UnistrokeTemplate(
1026             self.name,
1027             points=permutation,
1028             numpoints=self.numpoints,
1029             orientation_sensitive=self.orientation_sens
1030         ) for permutation in self._make_unistrokes()]
1031         del self._orders
1032 
1033     def _heap_permute(self, n):
1034         # Heap Permute algorithm
1035         self_order = self._order
1036         if n == 1:
1037             self._orders.append(self_order[:])
1038         else:
1039             i = 0
1040             for i in xrange(0, n):
1041                 self._heap_permute(n - 1)
1042                 if n % 2 == 1:
1043                     tmp = self_order[0]
1044                     self_order[0] = self_order[n - 1]
1045                     self_order[n - 1] = tmp
1046                 else:
1047                     tmp = self_order[i]
1048                     self_order[i] = self_order[n - 1]
1049                     self_order[n - 1] = tmp
1050 
1051     def _make_unistrokes(self):
1052         # Create unistroke permutations from self.strokes
1053         unistrokes = []
1054         unistrokes_append = unistrokes.append
1055         self_strokes = self.strokes
1056         for r in self._orders:
1057             b = 0
1058             while b < pow(2, len(r)):  # use b's bits for directions
1059                 unistroke = []
1060                 unistroke_append = unistroke.append
1061                 for i in xrange(0, len(r)):
1062                     pts = self_strokes[r[i]][:]
1063                     if (b >> i) & 1 == 1:  # is b's bit at index i 1?
1064                         pts.reverse()
1065                     unistroke_append(None)
1066                     unistroke[-1:] = pts
1067 
1068                 unistrokes_append(unistroke)
1069                 b += 1
1070         return unistrokes

1077 class UnistrokeTemplate(object):
1078     '''Represents a (uni)stroke path as a list of Vectors. Normally, this class 
1079     is instantiated by MultistrokeGesture and not by the programmer directly. 
1080     However, it is possible to manually compose UnistrokeTemplate objects. 
1081  
1082     :Arguments: 
1083         `name` 
1084             Identifies the name of the gesture. This is normally inherited from 
1085             the parent MultistrokeGesture object when a template is generated. 
1086         `points` 
1087             A list of points that represents a unistroke path. This is normally 
1088             one of the possible stroke order permutations from a 
1089             MultistrokeGesture. 
1090         `numpoints` 
1091             The number of points this template should (ideally) be resampled to 
1092             before the matching process. The default is 16, but you can use a 
1093             template-specific settings if that improves results. 
1094         `orientation_sensitive` 
1095             Determines if this template is orientation sensitive (True) or 
1096             fully rotation invariant (False). The default is True. 
1097  
1098     .. Note:: 
1099         You will get an exception if you set a skip-flag and then attempt to 
1100         retrieve those vectors. 
1101     '''
1102     def __init__(self, name, points=None, **kwargs):
1103         self.name = name
1104         self.numpoints = kwargs.get('numpoints', 16)
1105         self.orientation_sens = kwargs.get('orientation_sensitive', True)
1106 
1107         self.db = {}
1108         self.points = []
1109 
1110         if points is not None:
1111             self.points = points
1112 
1113     def add_point(self, p):
1114         '''Add a point to the unistroke/path. This invalidates all previously 
1115         computed vectors.'''
1116         self.points.append(p)
1117         # All previously computed data is now void.
1118         self.db = {}
1119 
1120     # Used to lazily prepare the template
1121     def _get_db_key(self, key, numpoints=None):
1122         n = numpoints and numpoints or self.numpoints
1123         if n not in self.db:
1124             self.prepare(n)
1125         return self.db[n][key]
1126 
1127     def get_start_unit_vector(self, numpoints=None):
1128         return self._get_db_key('startvector', numpoints)
1129 
1130     def get_vector(self, numpoints=None):
1131         return self._get_db_key('vector', numpoints)
1132 
1133     def get_points(self, numpoints=None):
1134         return self._get_db_key('points', numpoints)
1135 
1136     def prepare(self, numpoints=None):
1137         '''This function prepares the UnistrokeTemplate for matching given a 
1138         target number of points (for resample). 16 is optimal.'''
1139 
1140         if not self.points:
1141             raise MultistrokeError('prepare() called without self.points')
1142 
1143         # How many points are we resampling to?
1144         n = numpoints or self.numpoints
1145         if not n or n < 2:
1146             raise MultistrokeError('prepare() called with invalid numpoints')
1147 
1148         p = resample(self.points, n)
1149         radians = indicative_angle(p)
1150         p = rotate_by(p, -radians)
1151         p = scale_dim(p, SQUARESIZE, ONEDTHRESHOLD)
1152 
1153         if self.orientation_sens:
1154             p = rotate_by(p, +radians)  # restore
1155 
1156         p = translate_to(p, ORIGIN)
1157 
1158         # Now store it using the number of points in the resampled path as the
1159         # dict key. On the next call to get_*, it will be returned instead of
1160         # recomputed. Implicitly, you must reset self.db or call prepare() for
1161         # all the keys once you manipulate self.points.
1162         self.db[n] = {
1163             # Compute STARTANGLEINDEX as n/8:
1164             'startvector': start_unit_vector(p, (n / 8)),
1165             'vector': vectorize(p, self.orientation_sens)
1166         }

1173 class Candidate(object):
1174     '''Represents a set of unistroke paths of user input, ie data to be matched 
1175     against a :class:`UnistrokeTemplate` object using the Protractor algorithm. 
1176     By default, data is precomputed to match both rotation bounded and fully 
1177     invariant :class:`UnistrokeTemplate` objects. 
1178  
1179     :Arguments: 
1180         `strokes` 
1181             See :data:`MultistrokeGesture.strokes` for format example. The 
1182             Candidate strokes are simply combined to a unistroke in the order 
1183             given. The idea is that this will match one of the unistroke 
1184             permutations in `MultistrokeGesture.templates`. 
1185         `numpoints` 
1186             The Candidate's default N; this is only for a fallback, it is not 
1187             normally used since n is driven by the UnistrokeTemplate we are 
1188             being compared to. 
1189         `skip_bounded` 
1190             If True, do not generate/store rotation bounded vectors 
1191         `skip_invariant` 
1192             If True, do not generate/store rotation invariant vectors 
1193  
1194     Note that you WILL get errors if you set a skip-flag and then attempt to 
1195     retrieve the data.'''
1196     def __init__(self, strokes=None, numpoints=16, **kwargs):
1197         self.skip_invariant = kwargs.get('skip_invariant', False)
1198         self.skip_bounded = kwargs.get('skip_bounded', False)
1199 
1200         self.numpoints = numpoints
1201         self.db = {}
1202         self.strokes = []
1203 
1204         if strokes is not None:
1205             self.strokes = strokes
1206 
1207     def add_stroke(self, stroke):
1208         '''Add a stroke to the candidate; this will invalidate all 
1209         previously computed vectors'''
1210         self.points.append(stroke)
1211         self.db = {}
1212 
1213     # Used to lazily prepare the candidate
1214     def _get_db_key(self, key, numpoints, orientation_sens):
1215         n = numpoints and numpoints or self.numpoints
1216         if n not in self.db:
1217             self.prepare(n)
1218 
1219         prefix = orientation_sens and 'bound_' or 'inv_'
1220         return self.db[n][prefix + key]
1221 
1222     def get_start_unit_vector(self, numpoints, orientation_sens):
1223         '''(Internal use only) Get the start vector for this Candidate, 
1224         with the path resampled to `numpoints` points. This is the first 
1225         step in the matching process. It is compared to a 
1226         UnistrokeTemplate object's start vector to determine angle 
1227         similarity.'''
1228         return self._get_db_key('startvector', numpoints, orientation_sens)
1229 
1230     def get_protractor_vector(self, numpoints, orientation_sens):
1231         '''(Internal use only) Return vector for comparing to a 
1232         UnistrokeTemplate with Protractor'''
1233         return self._get_db_key('vector', numpoints, orientation_sens)
1234 
1235     def get_angle_similarity(self, tpl, **kwargs):
1236         '''(Internal use only) Compute the angle similarity between this 
1237         Candidate and a UnistrokeTemplate object. Returns a number that 
1238         represents the angle similarity (lower is more similar).'''
1239         n = kwargs.get('numpoints', self.numpoints)
1240 
1241         # angle_between_unit_vectors() inlined here for performance
1242         v1x, v1y = self.get_start_unit_vector(n, tpl.orientation_sens)
1243         v2x, v2y = tpl.get_start_unit_vector(n)
1244 
1245         n = (v1x * v2x + v1y * v2y)
1246         # FIXME: Domain error on float representation of 1.0 (exact match)
1247         # (see comments in MultistrokeGesture.get_distance())
1248         if n >= 1:
1249             return 0.0
1250         if n <= -1:
1251             return pi
1252         return acos(n)
1253 
1254     def prepare(self, numpoints=None):
1255         '''Prepare the Candidate vectors. self.strokes is combined to a single 
1256         unistroke (connected end-to-end), resampled to :attr:`numpoints` 
1257         points, and then the vectors are calculated and stored in self.db (for 
1258         use by `get_distance` and `get_angle_similarity`)'''
1259         n = numpoints and numpoints or self.numpoints
1260 
1261         # Inlined combine_strokes() for performance
1262         points = [i for sub in self.strokes for i in sub]
1263         points = resample(points, n)
1264         radians = indicative_angle(points)
1265         points = rotate_by(points, -radians)
1266         points = scale_dim(points, SQUARESIZE, ONEDTHRESHOLD)
1267 
1268         # Compute STARTANGLEINDEX as n / 8
1269         angidx = n / 8
1270         cand = {}
1271 
1272         # full rotation invariance
1273         if not self.skip_invariant:
1274             inv_points = translate_to(points, ORIGIN)
1275             cand['inv_startvector'] = start_unit_vector(inv_points, angidx)
1276             cand['inv_vector'] = vectorize(inv_points, False)
1277 
1278         # rotation bounded invariance
1279         if not self.skip_bounded:
1280             bound_points = rotate_by(points, +radians)  # restore
1281             bound_points = translate_to(bound_points, ORIGIN)
1282             cand['bound_startvector'] = start_unit_vector(bound_points, angidx)
1283             cand['bound_vector'] = vectorize(bound_points, True)
1284 
1285         self.db[n] = cand

1290 # recognition algorithm, and is almost 100% transcription from the JavaScript

1292 def resample(points, n):
1293     # Resample a path to `n` points
1294     if not len(points) or not n or n < 2:
1295         raise MultistrokeError('resample() called with invalid arguments')
1296 
1297     interval = path_length(points) / (n - 1)
1298     D = 0.0
1299     i = 1
1300     newpoints = [points[0]]
1301     workpoints = points[:]
1302     newpoints_len = 1
1303     workpoints_len = len(points)
1304 
1305     new_append = newpoints.append
1306     work_insert = workpoints.insert
1307     while i < len(workpoints):
1308         p1 = workpoints[i - 1]
1309         p2 = workpoints[i]
1310         d = distance(p1, p2)
1311 
1312         if D + d >= interval:
1313             qx = p1[0] + ((interval - D) / d) * (p2[0] - p1[0])
1314             qy = p1[1] + ((interval - D) / d) * (p2[1] - p1[1])
1315             q = Vector(qx, qy)
1316             new_append(q)
1317             work_insert(i, q)  # q is the next i
1318             newpoints_len += 1
1319             workpoints_len += 1
1320             D = 0.0
1321         else:
1322             D += d
1323 
1324         i += 1
1325 
1326     # rounding error; insert the last point
1327     if newpoints_len < n:
1328         new_append(points[-1])
1329 
1330     return newpoints

1395 def vectorize(points, use_bounded_rotation_invariance):
1396     # Helper function for the Protractor algorithm
1397     cos = 1.0
1398     sin = 0.0
1399 
1400     if use_bounded_rotation_invariance:
1401         ang = atan2(points[0][1], points[0][0])
1402         bo = (pi / 4.) * floor((ang + pi / 8.) / (pi / 4.))
1403         cos = math_cos(bo - ang)
1404         sin = math_sin(bo - ang)
1405 
1406     sum = 0.0
1407     vector = []
1408     vector_len = 0
1409     vector_append = vector.append
1410 
1411     for px, py in points:
1412         newx = px * cos - py * sin
1413         newy = py * cos + px * sin
1414         vector_append(newx)
1415         vector_append(newy)
1416         vector_len += 2
1417         sum += newx ** 2 + newy ** 2
1418 
1419     magnitude = sqrt(sum)
1420     for i in xrange(0, vector_len):
1421         vector[i] /= magnitude
1422 
1423     return vector

1426 def centroid(points):
1427     x = 0.0
1428     y = 0.0
1429     points_len = len(points)
1430 
1431     for i in xrange(0, points_len):
1432         x += points[i][0]
1433         y += points[i][1]
1434 
1435     x /= points_len
1436     y /= points_len
1437 
1438     return Vector(x, y)

1441 def bounding_box(points):
1442     minx = float('infinity')
1443     miny = float('infinity')
1444     maxx = float('-infinity')
1445     maxy = float('-infinity')
1446 
1447     for px, py in points:
1448         if px < minx:
1449             minx = px
1450         if px > maxx:
1451             maxx = px
1452         if py < miny:
1453             miny = py
1454         if py > maxy:
1455             maxy = py
1456 
1457     return (minx, miny, maxx - minx + 1, maxy - miny + 1)
