70       def OnClose(self):
71           try:
72               import pywin.debugger
73   
74               if (
75                   pywin.debugger.currentDebugger is not None
76                   and pywin.debugger.currentDebugger.pumping
77               ):
78                   try:
79                       pywin.debugger.currentDebugger.close(1)
80                   except:
81                       traceback.print_exc()
82                   return
83           except win32ui.error:
84               pass
85           self.closing = 1
86           self.SaveBarState("ToolbarDefault")
87           self.SetActiveView(None)  # Otherwise MFC's OnClose may _not_ prompt for save.
88   
89           from pywin.framework import help

91           help.FinalizeHelp()
92   
93           self.DestroyControlBar(afxres.AFX_IDW_TOOLBAR)
94           self.DestroyControlBar(win32ui.ID_VIEW_TOOLBAR_DBG)
95   
96           return self._obj_.OnClose()

105      def OnCommand(self, wparam, lparam):
106          # By default, the current MDI child frame will process WM_COMMAND
107          # messages before any docked control bars - even if the control bar
108          # has focus.  This is a problem for the interactive window when docked.
109          # Therefore, we detect the situation of a view having the main frame
110          # as its parent, and assume it must be a docked view (which it will in an MDI app)
111          try:
112              v = (
113                  self.GetActiveView()
114              )  # Raise an exception if none - good - then we want default handling
115              # Main frame _does_ have a current view (ie, a docking view) - see if it wants it.
116              if v.OnCommand(wparam, lparam):
117                  return 1
118          except (win32ui.error, AttributeError):
119              pass
120          return self._obj_.OnCommand(wparam, lparam)

150      def MakeExistingDDEConnection(self):
151          # Use DDE to connect to an existing instance
152          # Return None if no existing instance
153          try:
154              from . import intpydde
155          except ImportError:
156              # No dde support!
157              return None
158          conv = intpydde.CreateConversation(self.ddeServer)
159          try:
160              conv.ConnectTo("Pythonwin", "System")
161              return conv
162          except intpydde.error:
163              return None

165      def InitDDE(self):
166          # Do all the magic DDE handling.
167          # Returns TRUE if we have pumped the arguments to our
168          # remote DDE app, and we should terminate.
169          try:
170              from . import intpydde
171          except ImportError:
172              self.ddeServer = None
173              intpydde = None
174          if intpydde is not None:
175              self.ddeServer = intpydde.DDEServer(self)
176              self.ddeServer.Create("Pythonwin", intpydde.CBF_FAIL_SELFCONNECTIONS)
177              try:
178                  # If there is an existing instance, pump the arguments to it.
179                  connection = self.MakeExistingDDEConnection()
180                  if connection is not None:
181                      connection.Exec("self.Activate()")
182                      if self.ProcessArgs(sys.argv, connection) is None:
183                          return 1
184              except:
185                  # It is too early to 'print' an exception - we
186                  # don't have stdout setup yet!
187                  win32ui.DisplayTraceback(
188                      sys.exc_info(), " - error in DDE conversation with Pythonwin"
189                  )
190                  return 1

192      def InitInstance(self):
193          # Allow "/nodde" and "/new" to optimize this!
194          if (
195              "/nodde" not in sys.argv
196              and "/new" not in sys.argv
197              and "-nodde" not in sys.argv
198              and "-new" not in sys.argv
199          ):
200              if self.InitDDE():
201                  return 1  # A remote DDE client is doing it for us!
202          else:
203              self.ddeServer = None
204  
205          win32ui.SetRegistryKey(
206              "Python %s" % (sys.winver,)
207          )  # MFC automatically puts the main frame caption on!
208          app.CApp.InitInstance(self)
209  
210          # Create the taskbar icon
211          win32ui.CreateDebuggerThread()
212  
213          # Allow Pythonwin to host OCX controls.
214          win32ui.EnableControlContainer()
215  
216          # Display the interactive window if the user wants it.
217          from . import interact
218  
219          interact.CreateInteractiveWindowUserPreference()
220  
221          # Load the modules we use internally.
222          self.LoadSystemModules()
223  
224          # Load additional module the user may want.
225          self.LoadUserModules()
226  
227          # Load the ToolBar state near the end of the init process, as
228          # there may be Toolbar IDs created by the user or other modules.
229          # By now all these modules should be loaded, so all the toolbar IDs loaded.
230          try:
231              self.frame.LoadBarState("ToolbarDefault")
232          except win32ui.error:
233              # MFC sucks.  It does essentially "GetDlgItem(x)->Something", so if the
234              # toolbar with ID x does not exist, MFC crashes!  Pythonwin has a trap for this
235              # but I need to investigate more how to prevent it (AFAIK, ensuring all the
236              # toolbars are created by now _should_ stop it!)
237              pass
238  
239          # Finally process the command line arguments.
240          try:
241              self.ProcessArgs(sys.argv)
242          except:
243              # too early for printing anything.
244              win32ui.DisplayTraceback(
245                  sys.exc_info(), " - error processing command line args"
246              )

261      def Activate(self):
262          # Bring to the foreground.  Mainly used when another app starts up, it asks
263          # this one to activate itself, then it terminates.
264          frame = win32ui.GetMainFrame()
265          frame.SetForegroundWindow()
266          if frame.GetWindowPlacement()[1] == win32con.SW_SHOWMINIMIZED:
267              frame.ShowWindow(win32con.SW_RESTORE)

269      def ProcessArgs(self, args, dde=None):
270          # If we are going to talk to a remote app via DDE, then
271          # activate it!
272          if (
273              len(args) < 1 or not args[0]
274          ):  # argv[0]=='' when started without args, just like Python.exe!
275              return
276  
277          i = 0
278          while i < len(args):
279              argType = args[i]
280              i += 1
281              if argType.startswith("-"):
282                  # Support dash options. Slash options are misinterpreted by python init
283                  # as path and not finding usually 'C:\\' ends up in sys.path[0]
284                  argType = "/" + argType[1:]
285              if not argType.startswith("/"):
286                  argType = win32ui.GetProfileVal(
287                      "Python", "Default Arg Type", "/edit"
288                  ).lower()
289                  i -= 1  #  arg is /edit's parameter
290              par = i < len(args) and args[i] or "MISSING"
291              if argType in ["/nodde", "/new", "-nodde", "-new"]:
292                  # Already handled
293                  pass
294              elif argType.startswith("/goto:"):
295                  gotoline = int(argType[len("/goto:") :])
296                  if dde:
297                      dde.Exec(
298                          "from pywin.framework import scriptutils\n"
299                          "ed = scriptutils.GetActiveEditControl()\n"
300                          "if ed: ed.SetSel(ed.LineIndex(%s - 1))" % gotoline
301                      )
302                  else:
303                      from . import scriptutils
304  
305                      ed = scriptutils.GetActiveEditControl()
306                      if ed:
307                          ed.SetSel(ed.LineIndex(gotoline - 1))
308              elif argType == "/edit":
309                  # Load up the default application.
310                  i += 1
311                  fname = win32api.GetFullPathName(par)
312                  if not os.path.isfile(fname):
313                      # if we don't catch this, OpenDocumentFile() (actually
314                      # PyCDocument.SetPathName() in
315                      # pywin.scintilla.document.CScintillaDocument.OnOpenDocument)
316                      # segfaults Pythonwin on recent PY3 builds (b228)
317                      win32ui.MessageBox(
318                          "No such file: %s\n\nCommand Line: %s"
319                          % (fname, win32api.GetCommandLine()),
320                          "Open file for edit",
321                          win32con.MB_ICONERROR,
322                      )
323                      continue
324                  if dde:
325                      dde.Exec("win32ui.GetApp().OpenDocumentFile(%s)" % (repr(fname)))
326                  else:
327                      win32ui.GetApp().OpenDocumentFile(par)
328              elif argType == "/rundlg":
329                  if dde:
330                      dde.Exec(
331                          "from pywin.framework import scriptutils;scriptutils.RunScript(%r, %r, 1)"
332                          % (par, " ".join(args[i + 1 :]))
333                      )
334                  else:
335                      from . import scriptutils
336  
337                      scriptutils.RunScript(par, " ".join(args[i + 1 :]))
338                  return
339              elif argType == "/run":
340                  if dde:
341                      dde.Exec(
342                          "from pywin.framework import scriptutils;scriptutils.RunScript(%r, %r, 0)"
343                          % (par, " ".join(args[i + 1 :]))
344                      )
345                  else:
346                      from . import scriptutils
347  
348                      scriptutils.RunScript(par, " ".join(args[i + 1 :]), 0)
349                  return
350              elif argType == "/app":
351                  raise RuntimeError(
352                      "/app only supported for new instances of Pythonwin.exe"
353                  )
354              elif argType == "/dde":  # Send arbitary command
355                  if dde is not None:
356                      dde.Exec(par)
357                  else:
358                      win32ui.MessageBox(
359                          "The /dde command can only be used\r\nwhen Pythonwin is already running"
360                      )
361                  i += 1
362              else:
363                  raise ValueError("Command line argument not recognised: %s" % argType)

368      def LoadUserModules(self, moduleNames=None):
369          # Load the users modules.
370          if moduleNames is None:
371              default = "pywin.framework.sgrepmdi,pywin.framework.mdi_pychecker"
372              moduleNames = win32ui.GetProfileVal("Python", "Startup Modules", default)
373          self.DoLoadModules(moduleNames)

375      def DoLoadModules(self, moduleNames):  # ", sep string of module names.
376          if not moduleNames:
377              return
378          modules = moduleNames.split(",")
379          for module in modules:
380              try:
381                  __import__(module)
382              except:  # Catch em all, else the app itself dies! 'ImportError:
383                  traceback.print_exc()
384                  msg = 'Startup import of user module "%s" failed' % module
385                  print(msg)
386                  win32ui.MessageBox(msg)

402      def OnViewBrowse(self, id, code):
403          "Called when ViewBrowse message is received"
404          from pywin.tools import browser
405  
406          obName = dialog.GetSimpleInput("Object", "__builtins__", "Browse Python Object")
407          if obName is None:
408              return
409          try:
410              browser.Browse(eval(obName, __main__.__dict__, __main__.__dict__))
411          except NameError:
412              win32ui.MessageBox("This is no object with this name")
413          except AttributeError:
414              win32ui.MessageBox("The object has no attribute of that name")
415          except:
416              traceback.print_exc()
417              win32ui.MessageBox("This object can not be browsed")

443      def OnFileLocate(self, id, code):
444          from . import scriptutils
445  
446          global lastLocateFileName  # save the new version away for next time...
447  
448          name = dialog.GetSimpleInput(
449              "File name", lastLocateFileName, "Locate Python File"
450          )
451          if name is None:  # Cancelled.
452              return
453          lastLocateFileName = name
454          # if ".py" supplied, rip it off!
455          # should also check for .pys and .pyw
456          if lastLocateFileName[-3:].lower() == ".py":
457              lastLocateFileName = lastLocateFileName[:-3]
458          lastLocateFileName = lastLocateFileName.replace(".", "\\")
459          newName = scriptutils.LocatePythonFile(lastLocateFileName)
460          if newName is None:
461              win32ui.MessageBox("The file '%s' can not be located" % lastLocateFileName)
462          else:
463              win32ui.GetApp().OpenDocumentFile(newName)

465      # Display all the "options" proprety pages we can find
466      def OnViewOptions(self, id, code):
467          win32ui.InitRichEdit()
468          sheet = dialog.PropertySheet("Pythonwin Options")
469          # Add property pages we know about that need manual work.
470          from pywin.dialogs import ideoptions
471  
472          sheet.AddPage(ideoptions.OptionsPropPage())
473  
474          from . import toolmenu
475  
476          sheet.AddPage(toolmenu.ToolMenuPropPage())
477  
478          # Get other dynamic pages from templates.
479          pages = []
480          for template in self.GetDocTemplateList():
481              try:
482                  # Dont actually call the function with the exception handler.
483                  getter = template.GetPythonPropertyPages
484              except AttributeError:
485                  # Template does not provide property pages!
486                  continue
487              pages = pages + getter()
488  
489          # Debugger template goes at the end
490          try:
491              from pywin.debugger import configui
492          except ImportError:
493              configui = None
494          if configui is not None:
495              pages.append(configui.DebuggerOptionsPropPage())
496          # Now simply add the pages, and display the dialog.
497          for page in pages:
498              sheet.AddPage(page)
499  
500          if sheet.DoModal() == win32con.IDOK:
501              win32ui.SetStatusText("Applying configuration changes...", 1)
502              win32ui.DoWaitCursor(1)
503              # Tell every Window in our app that win.ini has changed!
504              win32ui.GetMainFrame().SendMessageToDescendants(
505                  win32con.WM_WININICHANGE, 0, 0
506              )
507              win32ui.DoWaitCursor(0)

524      def OnFileSaveAll(self, id, code):
525          # Only attempt to save editor documents.
526          from pywin.framework.editor import editorTemplate
527  
528          num = 0
529          for doc in editorTemplate.GetDocumentList():
530              if doc.IsModified() and doc.GetPathName():
531                  num = num = 1
532                  doc.OnSaveDocument(doc.GetPathName())
533          win32ui.SetStatusText("%d documents saved" % num, 1)
