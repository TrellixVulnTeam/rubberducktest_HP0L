5    # globals
6    Accum = 0.0	  # Previous number (operand) awaiting operation
7    FlagNewNum = 1    # Flag to indicate a new number (operand) is being entered
8    NullOp = lambda x,y: y
9    PendingOp = NullOp# Pending operation waiting for completion of second operand
10   numberButNames = ['Zero','One','Two','Three','Four','Five','Six','Seven','Eight','Nine']
11   
12   def NumPressed(Num):
13   	print "NumPressed", Num
14   	global FlagNewNum
15   	if FlagNewNum:
16   		ax.document.Keypad.ReadOut.Value = Num
17   		FlagNewNum = None
18   	else:
19   		if ax.document.Keypad.ReadOut.Value == "0":
20   			ax.document.Keypad.ReadOut.Value = str(Num)
21   		else:
22   			ax.document.Keypad.ReadOut.Value= ax.document.Keypad.ReadOut.Value + str(Num)
23   
24   # Dynamically create handlers for all the decimal buttons.
25   # (ie, this will dynamically create "One_OnClick()"... etc handlers
26   for i in range(len(numberButNames)):
27   	exec "def %s_OnClick():\tNumPressed(%d)\n" % (numberButNames[i],i)
28   
29   def Decimal_OnClick():
30   	global curReadOut, FlagNewNum
31   	curReadOut = ax.document.Keypad.ReadOut.Value
32   	if FlagNewNum:
33   		curReadOut = "0."
34   		FlagNewNum = None
35   	else:
36   		if not ("." in curReadOut):
37   			curReadOut = curReadOut + "."
38   	ax.document.Keypad.ReadOut.Value = curReadOut
39   
40   import sys, string
41   
42   def Operation(Op, fn):
43   	global FlagNewNum, PendingOp, Accum
44   	ReadOut = ax.document.Keypad.ReadOut.Value
45   	print "Operation", Op, ReadOut, PendingOp, Accum
46   	if FlagNewNum:
47   		# User is hitting op keys repeatedly, so don't do anything
48   		PendingOp = NullOp
49   	else:
50   		FlagNewNum = 1
51   		Accum = PendingOp( Accum, string.atof(ReadOut) )
52   	ax.document.Keypad.ReadOut.Value = str(Accum)
53   	PendingOp = fn
54   
55   def ClearEntry_OnClick():
56   	# Remove current number and reset state
57   	global FlagNewNum
58   	ax.document.Keypad.ReadOut.Value = "0"
59   	FlagNewNum = 1
60   
61   def Clear_OnClick():
62   	global Accum, PendingOp
63   	Accum = 0
64   	PendingOp = NullOp
65   	ClearEntry_OnClick()
66   
67   def Neg_OnClick():
68   	ax.document.Keypad.ReadOut.Value = str(-string.atof(ax.document.Keypad.ReadOut.Value))
