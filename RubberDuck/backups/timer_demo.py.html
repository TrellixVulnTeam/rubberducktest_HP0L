7    import timer
8    import time
9    
10   # Timers are based on Windows messages.  So we need
11   # to do the event-loop thing!
12   import win32event, win32gui
13   
14   # glork holds a simple counter for us.
15   
16   
17   class glork:
18       def __init__(self, delay=1000, max=10):
19           self.x = 0
20           self.max = max
21           self.id = timer.set_timer(delay, self.increment)
22           # Could use the threading module, but this is
23           # a win32 extension test after all! :-)
24           self.event = win32event.CreateEvent(None, 0, 0, None)
25   
26       def increment(self, id, time):
27           print("x = %d" % self.x)
28           self.x = self.x + 1
29           # if we've reached the max count,
30           # kill off the timer.
31           if self.x > self.max:
32               # we could have used 'self.id' here, too
33               timer.kill_timer(id)
34               win32event.SetEvent(self.event)
35   
36   
37   # create a counter that will count from '1' thru '10', incrementing
38   # once a second, and then stop.
39   
40   
41   def demo(delay=1000, stop=10):
42       g = glork(delay, stop)
43       # Timers are message based - so we need
44       # To run a message loop while waiting for our timers
45       # to expire.
46       start_time = time.time()
47       while 1:
48           # We can't simply give a timeout of 30 seconds, as
49           # we may continouusly be recieving other input messages,
50           # and therefore never expire.
51           rc = win32event.MsgWaitForMultipleObjects(
52               (g.event,),  # list of objects
53               0,  # wait all
54               500,  # timeout
55               win32event.QS_ALLEVENTS,  # type of input
56           )
57           if rc == win32event.WAIT_OBJECT_0:
58               # Event signalled.
59               break
60           elif rc == win32event.WAIT_OBJECT_0 + 1:
61               # Message waiting.
62               if win32gui.PumpWaitingMessages():
63                   raise RuntimeError("We got an unexpected WM_QUIT message!")
64           else:
65               # This wait timed-out.
66               if time.time() - start_time > 30:
67                   raise RuntimeError("We timed out waiting for the timers to expire!")
68   
69   
70   if __name__ == "__main__":
71       demo()
