226  STDMETHODIMP PyGEnum%(enumtype)s::Next(  
227              /* [in] */ ULONG celt, 
228              /* [length_is][size_is][out] */ %(argdeclare)s, 
229              /* [out] */ ULONG __RPC_FAR *pCeltFetched) 
230  { 
231      PY_GATEWAY_METHOD; 
232      PyObject *result; 
233      HRESULT hr = InvokeViaPolicy("Next", &result, "i", celt); 
234      if ( FAILED(hr) ) 
235          return hr; 
236   
237      if ( !PySequence_Check(result) ) 
238          goto error; 
239      int len; 
240      len = PyObject_Length(result); 
241      if ( len == -1 ) 
242          goto error; 
243      if ( len > (int)celt) 
244          len = celt; 
245   
246      if ( pCeltFetched ) 
247          *pCeltFetched = len; 
248   
249      int i; 
250      for ( i = 0; i < len; ++i ) 
251      { 
252          PyObject *ob = PySequence_GetItem(result, i); 
253          if ( ob == NULL ) 
254              goto error; 
255   
256          %(converter)s 
257          { 
258              Py_DECREF(result); 
259              return PyCom_SetCOMErrorFromPyException(IID_IEnum%(enumtype)s); 
260          } 
261      } 
262   
263      Py_DECREF(result); 
264   
265      return len < (int)celt ? S_FALSE : S_OK; 
266   
267    error: 
268      PyErr_Clear();  // just in case 
269      Py_DECREF(result); 
270      return PyCom_SetCOMErrorFromSimple(E_FAIL, IID_IEnum%(enumtype)s, "Next() did not return a sequence of objects");

286  STDMETHODIMP PyGEnum%(enumtype)s::Clone(  
287              /* [out] */ IEnum%(enumtype)s __RPC_FAR *__RPC_FAR *ppEnum) 
288  { 
289      PY_GATEWAY_METHOD; 
290      PyObject * result; 
291      HRESULT hr = InvokeViaPolicy("Clone", &result); 
292      if ( FAILED(hr) ) 
293          return hr; 
294   
295      /* 
296      ** Make sure we have the right kind of object: we should have some kind 
297      ** of IUnknown subclass wrapped into a PyIUnknown instance. 
298      */ 
299      if ( !PyIBase::is_object(result, &PyIUnknown::type) ) 
300      { 
301          /* the wrong kind of object was returned to us */ 
302          Py_DECREF(result); 
303          return PyCom_SetCOMErrorFromSimple(E_FAIL, IID_IEnum%(enumtype)s); 
304      } 
305   
306      /* 
307      ** Get the IUnknown out of the thing. note that the Python ob maintains 
308      ** a reference, so we don't have to explicitly AddRef() here. 
309      */ 
310      IUnknown *punk = ((PyIUnknown *)result)->m_obj; 
311      if ( !punk ) 
312      { 
313          /* damn. the object was released. */ 
314          Py_DECREF(result); 
315          return PyCom_SetCOMErrorFromSimple(E_FAIL, IID_IEnum%(enumtype)s); 
316      } 
317   
318      /* 
319      ** Get the interface we want. note it is returned with a refcount. 
320      ** This QI is actually going to instantiate a PyGEnum%(enumtype)s. 
321      */ 
322      hr = punk->QueryInterface(IID_IEnum%(enumtype)s, (LPVOID *)ppEnum); 
323   
324      /* done with the result; this DECREF is also for <punk> */ 
325      Py_DECREF(result); 
326   
327      return PyCom_SetCOMErrorFromSimple(hr, IID_IEnum%(enumtype)s, "Python could not convert the result from Next() into the required COM interface"); 
328  }
