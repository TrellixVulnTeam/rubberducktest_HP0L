1    # A demo of basic SSPI authentication.
2    # There is a 'client' context and a 'server' context - typically these will
3    # be on different machines (here they are in the same process, but the same
4    # concepts apply)
5    import sspi
6    import win32security, sspicon, win32api
7    
8    
9    def lookup_ret_code(err):
10       for k, v in list(sspicon.__dict__.items()):
11           if k[0:6] in ("SEC_I_", "SEC_E_") and v == err:
12               return k
13   
14   
15   """ 
16   pkg_name='Kerberos' 
17   sspiclient=SSPIClient(pkg_name, win32api.GetUserName(),  ## target spn is ourself 
18       None, None,   ## use none for client name and authentication information for current context 
19       ## u'username', (u'username',u'domain.com',u'passwd'), 
20       sspicon.ISC_REQ_INTEGRITY|sspicon.ISC_REQ_SEQUENCE_DETECT|sspicon.ISC_REQ_REPLAY_DETECT|    \ 
21           sspicon.ISC_REQ_DELEGATE|sspicon.ISC_REQ_CONFIDENTIALITY|sspicon.ISC_REQ_USE_SESSION_KEY) 
22   sspiserver=SSPIServer(pkg_name, None, 
23       sspicon.ASC_REQ_INTEGRITY|sspicon.ASC_REQ_SEQUENCE_DETECT|sspicon.ASC_REQ_REPLAY_DETECT|   \ 
24           sspicon.ASC_REQ_DELEGATE|sspicon.ASC_REQ_CONFIDENTIALITY|sspicon.ASC_REQ_STREAM|sspicon.ASC_REQ_USE_SESSION_KEY) 
25   """
26   
27   pkg_name = "NTLM"
28   
29   # Setup the 2 contexts.
30   sspiclient = sspi.ClientAuth(pkg_name)
31   sspiserver = sspi.ServerAuth(pkg_name)
32   
33   # Perform the authentication dance, each loop exchanging more information
34   # on the way to completing authentication.
35   sec_buffer = None
36   while 1:
37       err, sec_buffer = sspiclient.authorize(sec_buffer)
38       err, sec_buffer = sspiserver.authorize(sec_buffer)
39       if err == 0:
40           break
41   
42   # The server can now impersonate the client.  In this demo the 2 users will
43   # always be the same.
44   sspiserver.ctxt.ImpersonateSecurityContext()
45   print("Impersonated user: ", win32api.GetUserNameEx(win32api.NameSamCompatible))
46   sspiserver.ctxt.RevertSecurityContext()
47   print("Reverted to self: ", win32api.GetUserName())
48   
49   pkg_size_info = sspiclient.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
50   # Now sign some data
51   msg = "some data to be encrypted ......"
52   
53   sigsize = pkg_size_info["MaxSignature"]
54   sigbuf = win32security.PySecBufferDescType()
55   sigbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
56   sigbuf.append(win32security.PySecBufferType(sigsize, sspicon.SECBUFFER_TOKEN))
57   sigbuf[0].Buffer = msg
58   sspiclient.ctxt.MakeSignature(0, sigbuf, 1)
59   sspiserver.ctxt.VerifySignature(sigbuf, 1)
60   
61   # And finally encrypt some.
62   trailersize = pkg_size_info["SecurityTrailer"]
63   encbuf = win32security.PySecBufferDescType()
64   encbuf.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
65   encbuf.append(win32security.PySecBufferType(trailersize, sspicon.SECBUFFER_TOKEN))
66   encbuf[0].Buffer = msg
67   sspiclient.ctxt.EncryptMessage(0, encbuf, 1)
68   print("Encrypted data:", repr(encbuf[0].Buffer))
69   sspiserver.ctxt.DecryptMessage(encbuf, 1)
70   print("Unencrypted data:", encbuf[0].Buffer)
