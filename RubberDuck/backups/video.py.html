1    ''' 
2    Video 
3    ===== 
4     
5    The :class:`Video` widget is used to display video files and streams. 
6    Depending on your Video core provider, platform, and plugins, you will 
7    be able to play different formats. For example, the pygame video 
8    provider only supports MPEG1 on Linux and OSX. GStreamer is more 
9    versatile, and can read many video containers and codecs such as MKV, 
10   OGV, AVI, MOV, FLV (if the correct gstreamer plugins are installed). Our 
11   :class:`~kivy.core.video.VideoBase` implementation is used under the 
12   hood. 
13    
14   Video loading is asynchronous - many properties are not available until 
15   the video is loaded (when the texture is created):: 
16    
17       def on_position_change(instance, value): 
18           print('The position in the video is', value) 
19       def on_duration_change(instance, value): 
20           print('The duration of the video is', value) 
21       video = Video(source='PandaSneezes.avi') 
22       video.bind(position=on_position_change, 
23                  duration=on_duration_change) 
24    
25   '''
26   
27   __all__ = ('Video', )
28   
29   from kivy.clock import Clock
30   from kivy.uix.image import Image
31   from kivy.core.video import Video as CoreVideo
32   from kivy.resources import resource_find
33   from kivy.properties import (BooleanProperty, NumericProperty, ObjectProperty,
34                                OptionProperty)
35   
36   
37   class Video(Image):
38       '''Video class. See module documentation for more information. 
39       '''
40   
41       state = OptionProperty('stop', options=('play', 'pause', 'stop'))
42       '''String, indicates whether to play, pause, or stop the video:: 
43    
44           # start playing the video at creation 
45           video = Video(source='movie.mkv', state='play') 
46    
47           # create the video, and start later 
48           video = Video(source='movie.mkv') 
49           # and later 
50           video.state = 'play' 
51    
52       :attr:`state` is an :class:`~kivy.properties.OptionProperty` and defaults 
53       to 'stop'. 
54       '''
55   
56       play = BooleanProperty(False, deprecated=True)
57       ''' 
58       .. deprecated:: 1.4.0 
59           Use :attr:`state` instead. 
60    
61       Boolean, indicates whether the video is playing or not. 
62       You can start/stop the video by setting this property:: 
63    
64           # start playing the video at creation 
65           video = Video(source='movie.mkv', play=True) 
66    
67           # create the video, and start later 
68           video = Video(source='movie.mkv') 
69           # and later 
70           video.play = True 
71    
72       :attr:`play` is a :class:`~kivy.properties.BooleanProperty` and defaults to 
73       False. 
74    
75       .. deprecated:: 1.4.0 
76           Use :attr:`state` instead. 
77       '''
78   
79       eos = BooleanProperty(False)
80       '''Boolean, indicates whether the video has finished playing or not 
81       (reached the end of the stream). 
82    
83       :attr:`eos` is a :class:`~kivy.properties.BooleanProperty` and defaults to 
84       False. 
85       '''
86   
87       loaded = BooleanProperty(False)
88       '''Boolean, indicates whether the video is loaded and ready for playback 
89       or not. 
90    
91       .. versionadded:: 1.6.0 
92    
93       :attr:`loaded` is a :class:`~kivy.properties.BooleanProperty` and defaults 
94       to False. 
95       '''
96   
97       position = NumericProperty(-1)
98       '''Position of the video between 0 and :attr:`duration`. The position 
99       defaults to -1 and is set to a real position when the video is loaded. 
100   
101      :attr:`position` is a :class:`~kivy.properties.NumericProperty` and 
102      defaults to -1. 
103      '''
104  
105      duration = NumericProperty(-1)
106      '''Duration of the video. The duration defaults to -1, and is set to a real 
107      duration when the video is loaded. 
108   
109      :attr:`duration` is a :class:`~kivy.properties.NumericProperty` and 
110      defaults to -1. 
111      '''
112  
113      volume = NumericProperty(1.)
114      '''Volume of the video, in the range 0-1. 1 means full volume, 0 
115      means mute. 
116   
117      :attr:`volume` is a :class:`~kivy.properties.NumericProperty` and defaults 
118      to 1. 
119      '''
120  
121      options = ObjectProperty({})
122      '''Options to pass at Video core object creation. 
123   
124      .. versionadded:: 1.0.4 
125   
126      :attr:`options` is an :class:`kivy.properties.ObjectProperty` and defaults 
127      to {}. 
128      '''
129  
130      _video_load_event = None
131  
132      def __init__(self, **kwargs):
133          self._video = None
134          super(Video, self).__init__(**kwargs)
135          self.fbind('source', self._trigger_video_load)
136  
137          if "eos" in kwargs:
138              self.options["eos"] = kwargs["eos"]
139          if self.source:
140              self._trigger_video_load()
141  
142      def seek(self, percent, precise=True):
143          '''Change the position to a percentage (strictly, a proportion) 
144             of duration. 
145   
146          :Parameters: 
147              `percent`: float or int 
148                  Position to seek as a proportion of the total duration, 
149                  must be between 0-1. 
150              `precise`: bool, defaults to True 
151                  Precise seeking is slower, but seeks to exact requested 
152                  percent. 
153   
154          .. warning:: 
155              Calling seek() before the video is loaded has no effect. 
156   
157          .. versionadded:: 1.2.0 
158   
159          .. versionchanged:: 1.10.1 
160              The `precise` keyword argument has been added. 
161          '''
162          if self._video is None:
163              raise Exception('Video not loaded.')
164          self._video.seek(percent, precise=precise)
165  
166      def _trigger_video_load(self, *largs):
167          ev = self._video_load_event
168          if ev is None:
169              ev = self._video_load_event = Clock.schedule_once(
170                  self._do_video_load, -1)
171          ev()
172  
173      def _do_video_load(self, *largs):
174          if CoreVideo is None:
175              return
176          self.unload()
177          if not self.source:
178              self._video = None
179              self.texture = None
180          else:
181              filename = self.source
182              # Check if filename is not url
183              if '://' not in filename:
184                  filename = resource_find(filename)
185              self._video = CoreVideo(filename=filename, **self.options)
186              self._video.volume = self.volume
187              self._video.bind(on_load=self._on_load,
188                               on_frame=self._on_video_frame,
189                               on_eos=self._on_eos)
190              if self.state == 'play' or self.play:
191                  self._video.play()
192              self.duration = 1.
193              self.position = 0.
194  
195      def on_play(self, instance, value):
196          value = 'play' if value else 'stop'
197          return self.on_state(instance, value)
198  
199      def on_state(self, instance, value):
200          if not self._video:
201              return
202          if value == 'play':
203              if self.eos:
204                  self._video.stop()
205                  self._video.position = 0.
206              self.eos = False
207              self._video.play()
208          elif value == 'pause':
209              self._video.pause()
210          else:
211              self._video.stop()
212              self._video.position = 0
213  
214      def _on_video_frame(self, *largs):
215          video = self._video
216          if not video:
217              return
218          self.duration = video.duration
219          self.position = video.position
220          self.texture = video.texture
221          self.canvas.ask_update()
222  
223      def _on_eos(self, *largs):
224          if self._video.eos != 'loop':
225              self.state = 'stop'
226              self.eos = True
227  
228      def _on_load(self, *largs):
229          self.loaded = True
230          self._on_video_frame(largs)
231  
232      def on_volume(self, instance, value):
233          if self._video:
234              self._video.volume = value
235  
236      def unload(self):
237          '''Unload the video. The playback will be stopped. 
238   
239          .. versionadded:: 1.8.0 
240          '''
241          if self._video:
242              self._video.stop()
243              self._video.unload()
244              self._video = None
245          self.loaded = False

248  if __name__ == '__main__':
249      from kivy.app import App
250      import sys
251  
252      if len(sys.argv) != 2:
253          print("usage: %s file" % sys.argv[0])
254          sys.exit(1)
255  
256      class VideoApp(App):
257          def build(self):
258              self.v = Video(source=sys.argv[1], state='play')
259              self.v.bind(state=self.replay)
260              return self.v
261  
262          def replay(self, *args):
263              if self.v.state == 'stop':
264                  self.v.state = 'play'
265  
266      VideoApp().run()
