33   def OpenHelpFile(fileName, helpCmd=None, helpArg=None):
34       "Open a help file, given a full path"
35       # default help arg.
36       win32ui.DoWaitCursor(1)
37       try:
38           if helpCmd is None:
39               helpCmd = win32con.HELP_CONTENTS
40           ext = os.path.splitext(fileName)[1].lower()
41           if ext == ".hlp":
42               win32api.WinHelp(
43                   win32ui.GetMainFrame().GetSafeHwnd(), fileName, helpCmd, helpArg
44               )
45           # XXX - using the htmlhelp API wreaks havoc with keyboard shortcuts
46           # so we disable it, forcing ShellExecute, which works fine (but
47           # doesn't close the help file when Pythonwin is closed.
48           # Tom Heller also points out http://www.microsoft.com/mind/0499/faq/faq0499.asp,
49           # which may or may not be related.
50           elif 0 and ext == ".chm":
51               import win32help
52   
53               global htmlhelp_handle
54               helpCmd = html_help_command_translators.get(helpCmd, helpCmd)
55               # frame = win32ui.GetMainFrame().GetSafeHwnd()
56               frame = 0  # Dont want it overlapping ours!
57               if htmlhelp_handle is None:
58                   htmlhelp_hwnd, htmlhelp_handle = win32help.HtmlHelp(
59                       frame, None, win32help.HH_INITIALIZE
60                   )
61               win32help.HtmlHelp(frame, fileName, helpCmd, helpArg)
62           else:
63               # Hope that the extension is registered, and we know what to do!
64               win32api.ShellExecute(0, "open", fileName, None, "", win32con.SW_SHOW)
65           return fileName
66       finally:
67           win32ui.DoWaitCursor(-1)

70   def ListAllHelpFiles():
71       ret = []
72       ret = _ListAllHelpFilesInRoot(win32con.HKEY_LOCAL_MACHINE)
73       # Ensure we don't get dups.
74       for item in _ListAllHelpFilesInRoot(win32con.HKEY_CURRENT_USER):
75           if item not in ret:
76               ret.append(item)
77       return ret

80   def _ListAllHelpFilesInRoot(root):
81       """Returns a list of (helpDesc, helpFname) for all registered help files"""
82       import regutil
83   
84       retList = []
85       try:
86           key = win32api.RegOpenKey(
87               root, regutil.BuildDefaultPythonKey() + "\\Help", 0, win32con.KEY_READ
88           )
89       except win32api.error as exc:
90           import winerror
91   
92           if exc.winerror != winerror.ERROR_FILE_NOT_FOUND:
93               raise
94           return retList
95       try:
96           keyNo = 0
97           while 1:
98               try:
99                   helpDesc = win32api.RegEnumKey(key, keyNo)
100                  helpFile = win32api.RegQueryValue(key, helpDesc)
101                  retList.append((helpDesc, helpFile))
102                  keyNo = keyNo + 1
103              except win32api.error as exc:
104                  import winerror
105  
106                  if exc.winerror != winerror.ERROR_NO_MORE_ITEMS:
107                      raise
108                  break
109      finally:
110          win32api.RegCloseKey(key)
111      return retList

114  def SelectAndRunHelpFile():
115      from pywin.dialogs import list
116  
117      helpFiles = ListAllHelpFiles()
118      if len(helpFiles) == 1:
119          # only 1 help file registered - probably ours - no point asking
120          index = 0
121      else:
122          index = list.SelectFromLists("Select Help file", helpFiles, ["Title"])
123      if index is not None:
124          OpenHelpFile(helpFiles[index][1])

130  def SetHelpMenuOtherHelp(mainMenu):
131      """Modifies the main Help Menu to handle all registered help files. 
132      mainMenu -- The main menu to modify - usually from docTemplate.GetSharedMenu() 
133      """
134  
135      # Load all help files from the registry.
136      global helpIDMap
137      if helpIDMap is None:
138          helpIDMap = {}
139          cmdID = win32ui.ID_HELP_OTHER
140          excludeList = ["Main Python Documentation", "Pythonwin Reference"]
141          firstList = ListAllHelpFiles()
142          # We actually want to not only exclude these entries, but
143          # their help file names (as many entries may share the same name)
144          excludeFnames = []
145          for desc, fname in firstList:
146              if desc in excludeList:
147                  excludeFnames.append(fname)
148  
149          helpDescs = []
150          for desc, fname in firstList:
151              if fname not in excludeFnames:
152                  helpIDMap[cmdID] = (desc, fname)
153                  win32ui.GetMainFrame().HookCommand(HandleHelpOtherCommand, cmdID)
154                  cmdID = cmdID + 1
155  
156      helpMenu = mainMenu.GetSubMenu(
157          mainMenu.GetMenuItemCount() - 1
158      )  # Help menu always last.
159      otherHelpMenuPos = 2  # cant search for ID, as sub-menu has no ID.
160      otherMenu = helpMenu.GetSubMenu(otherHelpMenuPos)
161      while otherMenu.GetMenuItemCount():
162          otherMenu.DeleteMenu(0, win32con.MF_BYPOSITION)
163  
164      if helpIDMap:
165          for id, (desc, fname) in helpIDMap.items():
166              otherMenu.AppendMenu(win32con.MF_ENABLED | win32con.MF_STRING, id, desc)
167      else:
168          helpMenu.EnableMenuItem(
169              otherHelpMenuPos, win32con.MF_BYPOSITION | win32con.MF_GRAYED
170          )
