1    from kivy.tests import async_run, UnitKivyApp

4    def dropdown_app():
5        from kivy.app import App
6        from kivy.uix.button import Button
7        from kivy.uix.dropdown import DropDown
8        from kivy.uix.label import Label
9    
10       class RootButton(Button):
11   
12           dropdown = None
13   
14           def on_touch_down(self, touch):
15               assert self.dropdown.attach_to is None
16               return super(RootButton, self).on_touch_down(touch)
17   
18           def on_touch_move(self, touch):
19               assert self.dropdown.attach_to is None
20               return super(RootButton, self).on_touch_move(touch)
21   
22           def on_touch_up(self, touch):
23               assert self.dropdown.attach_to is None
24               return super(RootButton, self).on_touch_up(touch)
25   
26       class TestApp(UnitKivyApp, App):
27           def build(self):
28               root = RootButton(text='Root')
29               self.attach_widget = Label(text='Attached widget')
30               root.add_widget(self.attach_widget)
31   
32               root.dropdown = self.dropdown = DropDown(
33                   auto_dismiss=True, min_state_time=0)
34               self.inner_widget = w = Label(
35                   size_hint=(None, None), text='Dropdown')
36               root.dropdown.add_widget(w)
37               return root
38   
39       return TestApp()
40   
41   
42   @async_run(app_cls_func=dropdown_app)
43   async def test_dropdown_app(kivy_app):
44       await kivy_app.wait_clock_frames(2)
45       dropdown = kivy_app.dropdown
46       button = kivy_app.root
47       assert dropdown.attach_to is None
48   
49       kivy_app.attach_widget.size = button.width * 3 / 5, 2
50       kivy_app.attach_widget.top = button.height
51       kivy_app.inner_widget.size = button.width * 3 / 5, button.height
52   
53       # just press button
54       async for _ in kivy_app.do_touch_down_up(widget=button):
55           pass
56       async for _ in kivy_app.do_touch_drag(widget=button, dx=button.width / 4):
57           pass
58   
59       # open dropdown
60       dropdown.open(kivy_app.attach_widget)
61       dropdown.pos = 0, 0
62       await kivy_app.wait_clock_frames(2)
63       assert dropdown.attach_to is not None
64   
65       # press within dropdown area - should stay open
66       async for _ in kivy_app.do_touch_down_up(widget=button):
67           pass
68       assert dropdown.attach_to is not None
69       # start in dropdown but release outside - should stay open
70       async for _ in kivy_app.do_touch_drag(widget=button, dx=button.width / 4):
71           pass
72       assert dropdown.attach_to is not None
73   
74       # start outside but release in dropdown - should close
75       async for _ in kivy_app.do_touch_drag(
76               pos=(button.center_x + button.width / 4, button.center_y),
77               target_widget=button):
78           pass
79       assert dropdown.attach_to is None
80   
81       # open dropdown again
82       dropdown.open(kivy_app.attach_widget)
83       dropdown.pos = 0, 0
84       await kivy_app.wait_clock_frames(2)
85       assert dropdown.attach_to is not None
86   
87       # press outside dropdown area to close it - should close
88       async for _ in kivy_app.do_touch_down_up(
89               pos=(button.center_x + button.width / 4, button.center_y)):
90           pass
91       assert dropdown.attach_to is None
