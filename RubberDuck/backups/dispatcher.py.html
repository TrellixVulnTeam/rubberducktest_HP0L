14   class DispatcherBase:
15       """The base class for all Dispatchers. 
16    
17       This dispatcher supports wrapping all operations in exception handlers, 
18       and all the necessary delegation to the policy. 
19    
20       This base class supports the printing of "unexpected" exceptions.  Note, however, 
21       that exactly where the output of print goes may not be useful!  A derived class may 
22       provide additional semantics for this. 
23       """
24   
25       def __init__(self, policyClass, object):
26           self.policy = policyClass(object)
27           # The logger we should dump to.  If None, we should send to the
28           # default location (typically 'print')
29           self.logger = getattr(win32com, "logger", None)
30   
31       # Note the "return self._HandleException_()" is purely to stop pychecker
32       # complaining - _HandleException_ will itself raise an exception for the
33       # pythoncom framework, so the result will never be seen.
34       def _CreateInstance_(self, clsid, reqIID):
35           try:
36               self.policy._CreateInstance_(clsid, reqIID)
37               return pythoncom.WrapObject(self, reqIID)
38           except:
39               return self._HandleException_()
40   
41       def _QueryInterface_(self, iid):
42           try:
43               return self.policy._QueryInterface_(iid)
44           except:
45               return self._HandleException_()
46   
47       def _Invoke_(self, dispid, lcid, wFlags, args):
48           try:
49               return self.policy._Invoke_(dispid, lcid, wFlags, args)
50           except:
51               return self._HandleException_()
52   
53       def _GetIDsOfNames_(self, names, lcid):
54           try:
55               return self.policy._GetIDsOfNames_(names, lcid)
56           except:
57               return self._HandleException_()
58   
59       def _GetTypeInfo_(self, index, lcid):
60           try:
61               return self.policy._GetTypeInfo_(index, lcid)
62           except:
63               return self._HandleException_()
64   
65       def _GetTypeInfoCount_(self):
66           try:
67               return self.policy._GetTypeInfoCount_()
68           except:
69               return self._HandleException_()
70   
71       def _GetDispID_(self, name, fdex):
72           try:
73               return self.policy._GetDispID_(name, fdex)
74           except:
75               return self._HandleException_()
76   
77       def _InvokeEx_(self, dispid, lcid, wFlags, args, kwargs, serviceProvider):
78           try:
79               return self.policy._InvokeEx_(
80                   dispid, lcid, wFlags, args, kwargs, serviceProvider
81               )
82           except:
83               return self._HandleException_()
84   
85       def _DeleteMemberByName_(self, name, fdex):
86           try:
87               return self.policy._DeleteMemberByName_(name, fdex)
88           except:
89               return self._HandleException_()
90   
91       def _DeleteMemberByDispID_(self, id):
92           try:
93               return self.policy._DeleteMemberByDispID_(id)
94           except:
95               return self._HandleException_()
96   
97       def _GetMemberProperties_(self, id, fdex):
98           try:
99               return self.policy._GetMemberProperties_(id, fdex)
100          except:
101              return self._HandleException_()
102  
103      def _GetMemberName_(self, dispid):
104          try:
105              return self.policy._GetMemberName_(dispid)
106          except:
107              return self._HandleException_()
108  
109      def _GetNextDispID_(self, fdex, flags):
110          try:
111              return self.policy._GetNextDispID_(fdex, flags)
112          except:
113              return self._HandleException_()
114  
115      def _GetNameSpaceParent_(self):
116          try:
117              return self.policy._GetNameSpaceParent_()
118          except:
119              return self._HandleException_()
120  
121      def _HandleException_(self):
122          """Called whenever an exception is raised. 
123   
124          Default behaviour is to print the exception. 
125          """
126          # If not a COM exception, print it for the developer.
127          if not IsCOMServerException():
128              if self.logger is not None:
129                  self.logger.exception("pythoncom server error")
130              else:
131                  traceback.print_exc()
132          # But still raise it for the framework.
133          raise
134  
135      def _trace_(self, *args):
136          if self.logger is not None:
137              record = " ".join(map(str, args))
138              self.logger.debug(record)
139          else:
140              for arg in args[:-1]:
141                  print(arg, end=" ")
142              print(args[-1])

145  class DispatcherTrace(DispatcherBase):
146      """A dispatcher, which causes a 'print' line for each COM function called."""
147  
148      def _QueryInterface_(self, iid):
149          rc = DispatcherBase._QueryInterface_(self, iid)
150          if not rc:
151              self._trace_(
152                  "in %s._QueryInterface_ with unsupported IID %s (%s)"
153                  % (repr(self.policy._obj_), IIDToInterfaceName(iid), iid)
154              )
155          return rc
156  
157      def _GetIDsOfNames_(self, names, lcid):
158          self._trace_("in _GetIDsOfNames_ with '%s' and '%d'\n" % (names, lcid))
159          return DispatcherBase._GetIDsOfNames_(self, names, lcid)
160  
161      def _GetTypeInfo_(self, index, lcid):
162          self._trace_("in _GetTypeInfo_ with index=%d, lcid=%d\n" % (index, lcid))
163          return DispatcherBase._GetTypeInfo_(self, index, lcid)
164  
165      def _GetTypeInfoCount_(self):
166          self._trace_("in _GetTypeInfoCount_\n")
167          return DispatcherBase._GetTypeInfoCount_(self)
168  
169      def _Invoke_(self, dispid, lcid, wFlags, args):
170          self._trace_("in _Invoke_ with", dispid, lcid, wFlags, args)
171          return DispatcherBase._Invoke_(self, dispid, lcid, wFlags, args)
172  
173      def _GetDispID_(self, name, fdex):
174          self._trace_("in _GetDispID_ with", name, fdex)
175          return DispatcherBase._GetDispID_(self, name, fdex)
176  
177      def _InvokeEx_(self, dispid, lcid, wFlags, args, kwargs, serviceProvider):
178          self._trace_(
179              "in %r._InvokeEx_-%s%r [%x,%s,%r]"
180              % (self.policy._obj_, dispid, args, wFlags, lcid, serviceProvider)
181          )
182          return DispatcherBase._InvokeEx_(
183              self, dispid, lcid, wFlags, args, kwargs, serviceProvider
184          )
185  
186      def _DeleteMemberByName_(self, name, fdex):
187          self._trace_("in _DeleteMemberByName_ with", name, fdex)
188          return DispatcherBase._DeleteMemberByName_(self, name, fdex)
189  
190      def _DeleteMemberByDispID_(self, id):
191          self._trace_("in _DeleteMemberByDispID_ with", id)
192          return DispatcherBase._DeleteMemberByDispID_(self, id)
193  
194      def _GetMemberProperties_(self, id, fdex):
195          self._trace_("in _GetMemberProperties_ with", id, fdex)
196          return DispatcherBase._GetMemberProperties_(self, id, fdex)
197  
198      def _GetMemberName_(self, dispid):
199          self._trace_("in _GetMemberName_ with", dispid)
200          return DispatcherBase._GetMemberName_(self, dispid)
201  
202      def _GetNextDispID_(self, fdex, flags):
203          self._trace_("in _GetNextDispID_ with", fdex, flags)
204          return DispatcherBase._GetNextDispID_(self, fdex, flags)
205  
206      def _GetNameSpaceParent_(self):
207          self._trace_("in _GetNameSpaceParent_")
208          return DispatcherBase._GetNameSpaceParent_(self)

211  class DispatcherWin32trace(DispatcherTrace):
212      """A tracing dispatcher that sends its output to the win32trace remote collector."""
213  
214      def __init__(self, policyClass, object):
215          DispatcherTrace.__init__(self, policyClass, object)
216          if self.logger is None:
217              # If we have no logger, setup our output.
218              import win32traceutil  # Sets up everything.
219          self._trace_(
220              "Object with win32trace dispatcher created (object=%s)" % repr(object)
221          )

224  class DispatcherOutputDebugString(DispatcherTrace):
225      """A tracing dispatcher that sends its output to win32api.OutputDebugString"""
226  
227      def _trace_(self, *args):
228          for arg in args[:-1]:
229              win32api.OutputDebugString(str(arg) + " ")
230          win32api.OutputDebugString(str(args[-1]) + "\n")

233  class DispatcherWin32dbg(DispatcherBase):
234      """A source-level debugger dispatcher 
235   
236      A dispatcher which invokes the debugger as an object is instantiated, or 
237      when an unexpected exception occurs. 
238   
239      Requires Pythonwin. 
240      """
241  
242      def __init__(self, policyClass, ob):
243          # No one uses this, and it just causes py2exe to drag all of
244          # pythonwin in.
245          # import pywin.debugger
246          pywin.debugger.brk()
247          print("The DispatcherWin32dbg dispatcher is deprecated!")
248          print("Please let me know if this is a problem.")
249          print("Uncomment the relevant lines in dispatcher.py to re-enable")
250          # DEBUGGER Note - You can either:
251          # * Hit Run and wait for a (non Exception class) exception to occur!
252          # * Set a breakpoint and hit run.
253          # * Step into the object creation (a few steps away!)
254          DispatcherBase.__init__(self, policyClass, ob)
255  
256      def _HandleException_(self):
257          """Invoke the debugger post mortem capability"""
258          # Save details away.
259          typ, val, tb = exc_info()
260          # import pywin.debugger, pywin.debugger.dbgcon
261          debug = 0
262          try:
263              raise typ(val)
264          except Exception:  # AARG - What is this Exception???
265              # Use some inside knowledge to borrow a Debugger option which dictates if we
266              # stop at "expected" exceptions.
267              debug = pywin.debugger.GetDebugger().get_option(
268                  pywin.debugger.dbgcon.OPT_STOP_EXCEPTIONS
269              )
270          except:
271              debug = 1
272          if debug:
273              try:
274                  pywin.debugger.post_mortem(tb, typ, val)  # The original exception
275              except:
276                  traceback.print_exc()
277  
278          # But still raise it.
279          del tb
280          raise
