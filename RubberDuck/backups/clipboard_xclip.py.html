1    ''' 
2    Clipboard xclip: an implementation of the Clipboard using xclip 
3    command line tool. 
4    '''
5    
6    __all__ = ('ClipboardXclip', )
7    
8    from kivy.utils import platform
9    from kivy.core.clipboard._clipboard_ext import ClipboardExternalBase
10   
11   if platform != 'linux':
12       raise SystemError('unsupported platform for xclip clipboard')
13   
14   try:
15       import subprocess
16   
17       p = subprocess.Popen(['xclip', '-version'], stdout=subprocess.PIPE,
18           stderr=subprocess.DEVNULL)
19       p.communicate()
20   except:
21       raise
22   
23   
24   class ClipboardXclip(ClipboardExternalBase):
25       @staticmethod
26       def _clip(inout, selection):
27           pipe = {'std' + inout: subprocess.PIPE}
28           return subprocess.Popen(
29               ['xclip', '-' + inout, '-selection', selection], **pipe)
