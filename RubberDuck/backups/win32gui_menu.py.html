2    # This creates a taskbar icon which has some fancy menus (but note that
3    # selecting the menu items does nothing useful - see win32gui_taskbar.py
4    # for examples of this.

28       def __init__(self):
29           message_map = {
30               win32con.WM_DESTROY: self.OnDestroy,
31               win32con.WM_COMMAND: self.OnCommand,
32               win32con.WM_USER + 20: self.OnTaskbarNotify,
33               # owner-draw related handlers.
34               win32con.WM_MEASUREITEM: self.OnMeasureItem,
35               win32con.WM_DRAWITEM: self.OnDrawItem,
36           }
37           # Register the Window class.
38           wc = WNDCLASS()
39           hinst = wc.hInstance = GetModuleHandle(None)
40           wc.lpszClassName = "PythonTaskbarDemo"
41           wc.lpfnWndProc = message_map  # could also specify a wndproc.
42           classAtom = RegisterClass(wc)
43           # Create the Window.
44           style = win32con.WS_OVERLAPPED | win32con.WS_SYSMENU
45           self.hwnd = CreateWindow(
46               classAtom,
47               "Taskbar Demo",
48               style,
49               0,
50               0,
51               win32con.CW_USEDEFAULT,
52               win32con.CW_USEDEFAULT,
53               0,
54               0,
55               hinst,
56               None,
57           )
58           UpdateWindow(self.hwnd)
59           iconPathName = os.path.abspath(os.path.join(sys.prefix, "pyc.ico"))
60           # py2.5 includes the .ico files in the DLLs dir for some reason.
61           if not os.path.isfile(iconPathName):
62               iconPathName = os.path.abspath(
63                   os.path.join(os.path.split(sys.executable)[0], "DLLs", "pyc.ico")
64               )
65           if not os.path.isfile(iconPathName):
66               # Look in the source tree.
67               iconPathName = os.path.abspath(
68                   os.path.join(os.path.split(sys.executable)[0], "..\\PC\\pyc.ico")
69               )
70           if os.path.isfile(iconPathName):
71               icon_flags = win32con.LR_LOADFROMFILE | win32con.LR_DEFAULTSIZE
72               hicon = LoadImage(
73                   hinst, iconPathName, win32con.IMAGE_ICON, 0, 0, icon_flags
74               )
75           else:
76               iconPathName = None
77               print("Can't find a Python icon file - using default")
78               hicon = LoadIcon(0, win32con.IDI_APPLICATION)
79           self.iconPathName = iconPathName
80   
81           # Load up some information about menus needed by our owner-draw code.
82           # The font to use on the menu.
83           ncm = SystemParametersInfo(win32con.SPI_GETNONCLIENTMETRICS)
84           self.font_menu = CreateFontIndirect(ncm["lfMenuFont"])
85           # spacing for our ownerdraw menus - not sure exactly what constants
86           # should be used (and if you owner-draw all items on the menu, it
87           # doesn't matter!)
88           self.menu_icon_height = GetSystemMetrics(win32con.SM_CYMENU) - 4
89           self.menu_icon_width = self.menu_icon_height
90           self.icon_x_pad = 8  # space from end of icon to start of text.
91           # A map we use to stash away data we need for ownerdraw.  Keyed
92           # by integer ID - that ID will be set in dwTypeData of the menu item.
93           self.menu_item_map = {}
94   
95           # Finally, create the menu
96           self.createMenu()
97   
98           flags = NIF_ICON | NIF_MESSAGE | NIF_TIP
99           nid = (self.hwnd, 0, flags, win32con.WM_USER + 20, hicon, "Python Demo")
100          Shell_NotifyIcon(NIM_ADD, nid)
101          print("Please right-click on the Python icon in the taskbar")

103      def createMenu(self):
104          self.hmenu = menu = CreatePopupMenu()
105          # Create our 'Exit' item with the standard, ugly 'close' icon.
106          item, extras = PackMENUITEMINFO(
107              text="Exit", hbmpItem=win32con.HBMMENU_MBAR_CLOSE, wID=1000
108          )
109          InsertMenuItem(menu, 0, 1, item)
110          # Create a 'text only' menu via InsertMenuItem rather then
111          # AppendMenu, just to prove we can!
112          item, extras = PackMENUITEMINFO(text="Text only item", wID=1001)
113          InsertMenuItem(menu, 0, 1, item)
114  
115          load_bmp_flags = win32con.LR_LOADFROMFILE | win32con.LR_LOADTRANSPARENT
116          # These images are "over sized", so we load them scaled.
117          hbmp = LoadImage(
118              0,
119              os.path.join(this_dir, "images/smiley.bmp"),
120              win32con.IMAGE_BITMAP,
121              20,
122              20,
123              load_bmp_flags,
124          )
125  
126          # Create a top-level menu with a bitmap
127          item, extras = PackMENUITEMINFO(
128              text="Menu with bitmap", hbmpItem=hbmp, wID=1002
129          )
130          InsertMenuItem(menu, 0, 1, item)
131  
132          # Owner-draw menus mainly from:
133          # http://windowssdk.msdn.microsoft.com/en-us/library/ms647558.aspx
134          # and:
135          # http://www.codeguru.com/cpp/controls/menu/bitmappedmenus/article.php/c165
136  
137          # Create one with an icon - this is *lots* more work - we do it
138          # owner-draw!  The primary reason is to handle transparency better -
139          # converting to a bitmap causes the background to be incorrect when
140          # the menu item is selected.  I can't see a simpler way.
141          # First, load the icon we want to use.
142          ico_x = GetSystemMetrics(win32con.SM_CXSMICON)
143          ico_y = GetSystemMetrics(win32con.SM_CYSMICON)
144          if self.iconPathName:
145              hicon = LoadImage(
146                  0,
147                  self.iconPathName,
148                  win32con.IMAGE_ICON,
149                  ico_x,
150                  ico_y,
151                  win32con.LR_LOADFROMFILE,
152              )
153          else:
154              shell_dll = os.path.join(GetSystemDirectory(), "shell32.dll")
155              large, small = win32gui.ExtractIconEx(shell_dll, 4, 1)
156              hicon = small[0]
157              DestroyIcon(large[0])
158  
159          # Stash away the text and hicon in our map, and add the owner-draw
160          # item to the menu.
161          index = 0
162          self.menu_item_map[index] = (hicon, "Menu with owner-draw icon")
163          item, extras = PackMENUITEMINFO(
164              fType=win32con.MFT_OWNERDRAW, dwItemData=index, wID=1009
165          )
166          InsertMenuItem(menu, 0, 1, item)
167  
168          # Add another icon-based icon - but this time using HBMMENU_CALLBACK
169          # in the hbmpItem elt, so we only need to draw the icon (ie, not the
170          # text or checkmark)
171          index = 1
172          self.menu_item_map[index] = (hicon, None)
173          item, extras = PackMENUITEMINFO(
174              text="Menu with o-d icon 2",
175              dwItemData=index,
176              hbmpItem=win32con.HBMMENU_CALLBACK,
177              wID=1010,
178          )
179          InsertMenuItem(menu, 0, 1, item)
180  
181          # Add another icon-based icon - this time by converting
182          # via bitmap.  Note the icon background when selected is ugly :(
183          hdcBitmap = CreateCompatibleDC(0)
184          hdcScreen = GetDC(0)
185          hbm = CreateCompatibleBitmap(hdcScreen, ico_x, ico_y)
186          hbmOld = SelectObject(hdcBitmap, hbm)
187          SetBkMode(hdcBitmap, win32con.TRANSPARENT)
188          # Fill the background.
189          brush = GetSysColorBrush(win32con.COLOR_MENU)
190          FillRect(hdcBitmap, (0, 0, 16, 16), brush)
191          # unclear if brush needs to be freed.  Best clue I can find is:
192          # "GetSysColorBrush returns a cached brush instead of allocating a new
193          # one." - implies no DeleteObject.
194          # draw the icon
195          DrawIconEx(hdcBitmap, 0, 0, hicon, ico_x, ico_y, 0, 0, win32con.DI_NORMAL)
196          SelectObject(hdcBitmap, hbmOld)
197          DeleteDC(hdcBitmap)
198          item, extras = PackMENUITEMINFO(
199              text="Menu with icon", hbmpItem=hbm.Detach(), wID=1011
200          )
201          InsertMenuItem(menu, 0, 1, item)
202  
203          # Create a sub-menu, and put a few funky ones there.
204          self.sub_menu = sub_menu = CreatePopupMenu()
205          # A 'checkbox' menu.
206          item, extras = PackMENUITEMINFO(
207              fState=win32con.MFS_CHECKED, text="Checkbox menu", hbmpItem=hbmp, wID=1003
208          )
209          InsertMenuItem(sub_menu, 0, 1, item)
210          # A 'radio' menu.
211          InsertMenu(sub_menu, 0, win32con.MF_BYPOSITION, win32con.MF_SEPARATOR, None)
212          item, extras = PackMENUITEMINFO(
213              fType=win32con.MFT_RADIOCHECK,
214              fState=win32con.MFS_CHECKED,
215              text="Checkbox menu - bullet 1",
216              hbmpItem=hbmp,
217              wID=1004,
218          )
219          InsertMenuItem(sub_menu, 0, 1, item)
220          item, extras = PackMENUITEMINFO(
221              fType=win32con.MFT_RADIOCHECK,
222              fState=win32con.MFS_UNCHECKED,
223              text="Checkbox menu - bullet 2",
224              hbmpItem=hbmp,
225              wID=1005,
226          )
227          InsertMenuItem(sub_menu, 0, 1, item)
228          # And add the sub-menu to the top-level menu.
229          item, extras = PackMENUITEMINFO(text="Sub-Menu", hSubMenu=sub_menu)
230          InsertMenuItem(menu, 0, 1, item)
231  
232          # Set 'Exit' as the default option.
233          SetMenuDefaultItem(menu, 1000, 0)

235      def OnDestroy(self, hwnd, msg, wparam, lparam):
236          nid = (self.hwnd, 0)
237          Shell_NotifyIcon(NIM_DELETE, nid)
238          PostQuitMessage(0)  # Terminate the app.

240      def OnTaskbarNotify(self, hwnd, msg, wparam, lparam):
241          if lparam == win32con.WM_RBUTTONUP:
242              print("You right clicked me.")
243              # display the menu at the cursor pos.
244              pos = GetCursorPos()
245              SetForegroundWindow(self.hwnd)
246              TrackPopupMenu(
247                  self.hmenu, win32con.TPM_LEFTALIGN, pos[0], pos[1], 0, self.hwnd, None
248              )
249              PostMessage(self.hwnd, win32con.WM_NULL, 0, 0)
250          elif lparam == win32con.WM_LBUTTONDBLCLK:
251              print("You double-clicked me")
252              # find the default menu item and fire it.
253              cmd = GetMenuDefaultItem(self.hmenu, False, 0)
254              if cmd == -1:
255                  print("Can't find a default!")
256              # and just pretend it came from the menu
257              self.OnCommand(hwnd, win32con.WM_COMMAND, cmd, 0)
258          return 1

260      def OnCommand(self, hwnd, msg, wparam, lparam):
261          id = LOWORD(wparam)
262          if id == 1000:
263              print("Goodbye")
264              DestroyWindow(self.hwnd)
265          elif id in (1003, 1004, 1005):
266              # Our 'checkbox' and 'radio' items
267              state = GetMenuState(self.sub_menu, id, win32con.MF_BYCOMMAND)
268              if state == -1:
269                  raise RuntimeError("No item found")
270              if state & win32con.MF_CHECKED:
271                  check_flags = win32con.MF_UNCHECKED
272                  print("Menu was checked - unchecking")
273              else:
274                  check_flags = win32con.MF_CHECKED
275                  print("Menu was unchecked - checking")
276  
277              if id == 1003:
278                  # simple checkbox
279                  rc = CheckMenuItem(
280                      self.sub_menu, id, win32con.MF_BYCOMMAND | check_flags
281                  )
282              else:
283                  # radio button - must pass the first and last IDs in the
284                  # "group", and the ID in the group that is to be selected.
285                  rc = CheckMenuRadioItem(
286                      self.sub_menu, 1004, 1005, id, win32con.MF_BYCOMMAND
287                  )
288              # Get and check the new state - first the simple way...
289              new_state = GetMenuState(self.sub_menu, id, win32con.MF_BYCOMMAND)
290              if new_state & win32con.MF_CHECKED != check_flags:
291                  raise RuntimeError("The new item didn't get the new checked state!")
292              # Now the long-winded way via GetMenuItemInfo...
293              buf, extras = EmptyMENUITEMINFO()
294              win32gui.GetMenuItemInfo(self.sub_menu, id, False, buf)
295              (
296                  fType,
297                  fState,
298                  wID,
299                  hSubMenu,
300                  hbmpChecked,
301                  hbmpUnchecked,
302                  dwItemData,
303                  text,
304                  hbmpItem,
305              ) = UnpackMENUITEMINFO(buf)
306  
307              if fState & win32con.MF_CHECKED != check_flags:
308                  raise RuntimeError("The new item didn't get the new checked state!")
309          else:
310              print("OnCommand for ID", id)

312      # Owner-draw related functions.  We only have 1 owner-draw item, but
313      # we pretend we have more than that :)
314      def OnMeasureItem(self, hwnd, msg, wparam, lparam):
315          ## Last item of MEASUREITEMSTRUCT is a ULONG_PTR
316          fmt = "5iP"
317          buf = PyMakeBuffer(struct.calcsize(fmt), lparam)
318          data = struct.unpack(fmt, buf)
319          ctlType, ctlID, itemID, itemWidth, itemHeight, itemData = data
320  
321          hicon, text = self.menu_item_map[itemData]
322          if text is None:
323              # Only drawing icon due to HBMMENU_CALLBACK
324              cx = self.menu_icon_width
325              cy = self.menu_icon_height
326          else:
327              # drawing the lot!
328              dc = GetDC(hwnd)
329              oldFont = SelectObject(dc, self.font_menu)
330              cx, cy = GetTextExtentPoint32(dc, text)
331              SelectObject(dc, oldFont)
332              ReleaseDC(hwnd, dc)
333  
334              cx += GetSystemMetrics(win32con.SM_CXMENUCHECK)
335              cx += self.menu_icon_width + self.icon_x_pad
336  
337              cy = GetSystemMetrics(win32con.SM_CYMENU)
338  
339          new_data = struct.pack(fmt, ctlType, ctlID, itemID, cx, cy, itemData)
340          PySetMemory(lparam, new_data)
341          return True

343      def OnDrawItem(self, hwnd, msg, wparam, lparam):
344          ## lparam is a DRAWITEMSTRUCT
345          fmt = "5i2P4iP"
346          data = struct.unpack(fmt, PyGetMemory(lparam, struct.calcsize(fmt)))
347          (
348              ctlType,
349              ctlID,
350              itemID,
351              itemAction,
352              itemState,
353              hwndItem,
354              hDC,
355              left,
356              top,
357              right,
358              bot,
359              itemData,
360          ) = data
361  
362          rect = left, top, right, bot
363          hicon, text = self.menu_item_map[itemData]
364  
365          if text is None:
366              # This means the menu-item had HBMMENU_CALLBACK - so all we
367              # draw is the icon.  rect is the entire area we should use.
368              DrawIconEx(
369                  hDC, left, top, hicon, right - left, bot - top, 0, 0, win32con.DI_NORMAL
370              )
371          else:
372              # If the user has selected the item, use the selected
373              # text and background colors to display the item.
374              selected = itemState & win32con.ODS_SELECTED
375              if selected:
376                  crText = SetTextColor(hDC, GetSysColor(win32con.COLOR_HIGHLIGHTTEXT))
377                  crBkgnd = SetBkColor(hDC, GetSysColor(win32con.COLOR_HIGHLIGHT))
378  
379              each_pad = self.icon_x_pad // 2
380              x_icon = left + GetSystemMetrics(win32con.SM_CXMENUCHECK) + each_pad
381              x_text = x_icon + self.menu_icon_width + each_pad
382  
383              # Draw text first, specifying a complete rect to fill - this sets
384              # up the background (but overwrites anything else already there!)
385              # Select the font, draw it, and restore the previous font.
386              hfontOld = SelectObject(hDC, self.font_menu)
387              ExtTextOut(hDC, x_text, top + 2, win32con.ETO_OPAQUE, rect, text)
388              SelectObject(hDC, hfontOld)
389  
390              # Icon image next.  Icons are transparent - no need to handle
391              # selection specially.
392              DrawIconEx(
393                  hDC,
394                  x_icon,
395                  top + 2,
396                  hicon,
397                  self.menu_icon_width,
398                  self.menu_icon_height,
399                  0,
400                  0,
401                  win32con.DI_NORMAL,
402              )
403  
404              # Return the text and background colors to their
405              # normal state (not selected).
406              if selected:
407                  SetTextColor(hDC, crText)
408                  SetBkColor(hDC, crBkgnd)
