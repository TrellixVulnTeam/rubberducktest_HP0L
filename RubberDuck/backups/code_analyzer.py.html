10   from docutils import ApplicationError
11   try:
12       import pygments
13       from pygments.lexers import get_lexer_by_name
14       from pygments.formatters.html import _get_ttype_class
15       with_pygments = True
16   except ImportError:
17       with_pygments = False
18   
19   # Filter the following token types from the list of class arguments:
20   unstyled_tokens = ['token', # Token (base token type)
21                      'text',  # Token.Text
22                      '']      # short name for Token and Text
23   # (Add, e.g., Token.Punctuation with ``unstyled_tokens += 'punctuation'``.)
24   
25   class LexerError(ApplicationError):
26       pass
27   
28   class Lexer(object):
29       """Parse `code` lines and yield "classified" tokens. 
30    
31       Arguments 
32    
33         code       -- string of source code to parse, 
34         language   -- formal language the code is written in, 
35         tokennames -- either 'long', 'short', or 'none' (see below). 
36    
37       Merge subsequent tokens of the same token-type. 
38    
39       Iterating over an instance yields the tokens as ``(tokentype, value)`` 
40       tuples. The value of `tokennames` configures the naming of the tokentype: 
41    
42         'long':  downcased full token type name, 
43         'short': short name defined by pygments.token.STANDARD_TYPES 
44                  (= class argument used in pygments html output), 
45         'none':  skip lexical analysis. 
46       """
47   
48       def __init__(self, code, language, tokennames='short'):
49           """ 
50           Set up a lexical analyzer for `code` in `language`. 
51           """
52           self.code = code
53           self.language = language
54           self.tokennames = tokennames
55           self.lexer = None
56           # get lexical analyzer for `language`:
57           if language in ('', 'text') or tokennames == 'none':
58               return
59           if not with_pygments:
60               raise LexerError('Cannot analyze code. '
61                                       'Pygments package not found.')
62           try:
63               self.lexer = get_lexer_by_name(self.language)
64           except pygments.util.ClassNotFound:
65               raise LexerError('Cannot analyze code. '
66                   'No Pygments lexer found for "%s".' % language)
67           # self.lexer.add_filter('tokenmerge')
68           # Since version 1.2. (released Jan 01, 2010) Pygments has a
69           # TokenMergeFilter. # ``self.merge(tokens)`` in __iter__ could
70           # be replaced by ``self.lexer.add_filter('tokenmerge')`` in __init__.
71           # However, `merge` below also strips a final newline added by pygments.
72           #
73           # self.lexer.add_filter('tokenmerge')
74   
75       def merge(self, tokens):
76           """Merge subsequent tokens of same token-type. 
77    
78              Also strip the final newline (added by pygments). 
79           """
80           tokens = iter(tokens)
81           (lasttype, lastval) = next(tokens)
82           for ttype, value in tokens:
83               if ttype is lasttype:
84                   lastval += value
85               else:
86                   yield(lasttype, lastval)
87                   (lasttype, lastval) = (ttype, value)
88           if lastval.endswith('\n'):
89               lastval = lastval[:-1]
90           if lastval:
91               yield(lasttype, lastval)
92   
93       def __iter__(self):
94           """Parse self.code and yield "classified" tokens. 
95           """
96           if self.lexer is None:
97               yield ([], self.code)
98               return
99           tokens = pygments.lex(self.code, self.lexer)
100          for tokentype, value in self.merge(tokens):
101              if self.tokennames == 'long': # long CSS class args
102                  classes = str(tokentype).lower().split('.')
103              else: # short CSS class args
104                  classes = [_get_ttype_class(tokentype)]
105              classes = [cls for cls in classes if cls not in unstyled_tokens]
106              yield (classes, value)

109  class NumberLines(object):
110      """Insert linenumber-tokens at the start of every code line. 
111   
112      Arguments 
113   
114         tokens    -- iterable of ``(classes, value)`` tuples 
115         startline -- first line number 
116         endline   -- last line number 
117   
118      Iterating over an instance yields the tokens with a 
119      ``(['ln'], '<the line number>')`` token added for every code line. 
120      Multi-line tokens are split."""
121  
122      def __init__(self, tokens, startline, endline):
123          self.tokens = tokens
124          self.startline = startline
125          # pad linenumbers, e.g. endline == 100 -> fmt_str = '%3d '
126          self.fmt_str = '%%%dd ' % len(str(endline))
127  
128      def __iter__(self):
129          lineno = self.startline
130          yield (['ln'], self.fmt_str % lineno)
131          for ttype, value in self.tokens:
132              lines = value.split('\n')
133              for line in lines[:-1]:
134                  yield (ttype, line + '\n')
135                  lineno += 1
136                  yield (['ln'], self.fmt_str % lineno)
137              yield (ttype, lines[-1])
