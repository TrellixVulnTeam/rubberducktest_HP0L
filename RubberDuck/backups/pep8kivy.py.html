8    pep8_ignores = (
9        'E125',  # continuation line does not
10                # distinguish itself from next logical line
11       'E126',  # continuation line over-indented for hanging indent
12       'E127',  # continuation line over-indented for visual indent
13       'E128',  # continuation line under-indented for visual indent
14       'E402',  # module level import not at top of file
15       'E741',  # ambiguous variable name
16       'E731',  # do not assign a lambda expression, use a def
17       'W503',  # allow putting binary operators after line split
18   )
19   
20   
21   class KivyStyleChecker(pep8.Checker):
22   
23       def __init__(self, filename):
24           pep8.Checker.__init__(self, filename, ignore=pep8_ignores)
25   
26       def report_error(self, line_number, offset, text, check):
27           return pep8.Checker.report_error(
28               self, line_number, offset, text, check)

30   
31   if __name__ == '__main__':
32   
33       def usage():
34           print('Usage: python pep8kivy.py <file_or_folder_to_check>*')
35           print('Folders will be checked recursively.')
36           sys.exit(1)
37   
38       if len(sys.argv) < 2:
39           usage()
40       elif sys.argv == 2:
41           targets = sys.argv[-1]
42       else:
43           targets = sys.argv[-1].split()
44   
45       def check(fn):
46           try:
47               checker = KivyStyleChecker(fn)
48           except IOError:
49               # File couldn't be opened, so was deleted apparently.
50               # Don't check deleted files.
51               return 0
52           return checker.check_all()
53   
54       errors = 0
55   
56       exclude_dirs = [
57           'kivy/lib',
58           'kivy/deps',
59           'kivy/tools/pep8checker',
60           'coverage',
61           'doc'
62       ]
63       exclude_dirs = [normpath(i) for i in exclude_dirs]
64       exclude_files = [
65           'kivy/gesture.py',
66           'kivy/tools/stub-gl-debug.py',
67           'kivy/modules/webdebugger.py',
68           'kivy/modules/_webdebugger.py'
69       ]
70       exclude_files = [normpath(i) for i in exclude_files]
71   
72       for target in targets:
73           if isdir(target):
74               for dirpath, dirnames, filenames in walk(target):
75                   cont = False
76                   dpath = normpath(dirpath)
77                   for pat in exclude_dirs:
78                       if dpath.startswith(pat):
79                           cont = True
80                           break
81                   if cont:
82                       continue
83                   for filename in filenames:
84                       if not filename.endswith('.py'):
85                           continue
86                       cont = False
87                       complete_filename = join(dirpath, filename)
88                       for pat in exclude_files:
89                           if complete_filename.endswith(pat):
90                               cont = True
91                       if cont:
92                           continue
93   
94                       errors += check(complete_filename)
95   
96           else:
97               # Got a single file to check
98               for pat in exclude_dirs + exclude_files:
99                   if pat in target:
100                      break
101              else:
102                  if target.endswith('.py'):
103                      errors += check(target)
104  
105      if errors:
106          print("Error: {} style guide violation(s) encountered.".format(errors))
107          sys.exit(1)
