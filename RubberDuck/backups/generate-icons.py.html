55       def run(self):
56           parser = ArgumentParser(
57                   description='Generate icons for various stores')
58           parser.add_argument('--dir', type=str, default=None,
59                   help=('Output directory to generate all the icons,'
60                         'defaults to the directory of the source icon'))
61           parser.add_argument('--force', type=bool, default=False,
62                   help=('Generate all icons even if the source is not perfect.'))
63           parser.add_argument('icon', type=str,
64                   help='Base icon (must be 1024x1024 or 512x512)')
65   
66           args = parser.parse_args()
67           if not exists(args.icon):
68               print('Error: No such icon file')
69               sys.exit(1)
70   
71           # ensure the destination directory will be set
72           if args.dir is None:
73               args.dir = dirname(args.icon)
74   
75           # read the source image, and do some quality checks
76           base_fn = basename(args.icon).rsplit('.', 1)[0]
77           source = Image.open(args.icon)
78           self.ensure_quality(source, args.force)
79   
80           for directory_name, sizeinfo in self.iterate():
81               description, pattern_fn, size = sizeinfo
82               print('Generate {}: {}x{}'.format(description, size, size))
83               dest_dir = realpath(join(args.dir, directory_name))
84               if not exists(dest_dir):
85                   makedirs(dest_dir)
86               icon_fn = join(dest_dir, pattern_fn.format('Icon'))
87               self.convert_to(source, icon_fn, size)
88   
89       def convert_to(self, source, icon_fn, size):
90           dest = source.resize((size, size))
91           dest.save(icon_fn, 'png')
92   
93       def ensure_quality(self, image, force=False):
94           messages = []
95           w, h = image.size
96           if w != h:
97               messages.append('Width and height should be the same')
98           if w not in (512, 1024):
99               messages.append(
100                  'Source image is recommended to be 1024 (512 minimum)')
101          if not messages:
102              return
103  
104          print('Quality check failed')
105          for message in messages:
106              print('- {}'.format(message))
107          if not force:
108              sys.exit(1)
109  
110      def iterate(self):
111          for store, infos in Converter.converters.items():
112              for size in infos['sizes']:
113                  yield infos['directory_name'], size
114  
115  
116  if __name__ == '__main__':
117      Converter().run()
