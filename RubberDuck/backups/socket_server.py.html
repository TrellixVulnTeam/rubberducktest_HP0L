1    """A sample socket server and client using SSPI authentication and encryption. 
2     
3    You must run with either 'client' or 'server' as arguments.  A server must be 
4    running before a client can connect. 
5     
6    To use with Kerberos you should include in the client options 
7    --target-spn=username, where 'username' is the user under which the server is 
8    being run. 
9     
10   Running either the client or server as a different user can be informative. 
11   A command-line such as the following may be useful: 
12   `runas /user:{user} {fqp}\python.exe {fqp}\socket_server.py --wait client|server` 
13    
14   {fqp} should specify the relevant fully-qualified path names. 
15    
16   To use 'runas' with Kerberos, the client program will need to 
17   specify --target-spn with the username under which the *server* is running. 
18    
19   See the SSPI documentation for more details. 
20   """

38   def GetUserName():
39       try:
40           return win32api.GetUserName()
41       except win32api.error as details:
42           # Seeing 'access denied' errors here for non-local users (presumably
43           # without permission to login locally).  Get the fully-qualified
44           # username, although a side-effect of these permission-denied errors
45           # is a lack of Python codecs - so printing the Unicode value fails.
46           # So just return the repr(), and avoid codecs completely.
47           return repr(win32api.GetUserNameEx(win32api.NameSamCompatible))

50   # Send a simple "message" over a socket - send the number of bytes first,
51   # then the string.  Ditto for receive.
52   def _send_msg(s, m):
53       s.send(struct.pack("i", len(m)))
54       s.send(m)
55   
56   
57   def _get_msg(s):
58       size_data = s.recv(struct.calcsize("i"))
59       if not size_data:
60           return None
61       cb = struct.unpack("i", size_data)[0]
62       return s.recv(cb)
63

65   class SSPISocketServer(socketserver.TCPServer):
66       def __init__(self, *args, **kw):
67           socketserver.TCPServer.__init__(self, *args, **kw)
68           self.sa = sspi.ServerAuth(options.package)
69   
70       def verify_request(self, sock, ca):
71           # Do the sspi auth dance
72           self.sa.reset()
73           while 1:
74               data = _get_msg(sock)
75               if data is None:
76                   return False
77               try:
78                   err, sec_buffer = self.sa.authorize(data)
79               except sspi.error as details:
80                   print("FAILED to authorize client:", details)
81                   return False
82   
83               if err == 0:
84                   break
85               _send_msg(sock, sec_buffer[0].Buffer)
86           return True
87   
88       def process_request(self, request, client_address):
89           # An example using the connection once it is established.
90           print("The server is running as user", GetUserName())
91           self.sa.ctxt.ImpersonateSecurityContext()
92           try:
93               print("Having conversation with client as user", GetUserName())
94               while 1:
95                   # we need to grab 2 bits of data - the encrypted data, and the
96                   # 'key'
97                   data = _get_msg(request)
98                   key = _get_msg(request)
99                   if data is None or key is None:
100                      break
101                  data = self.sa.decrypt(data, key)
102                  print("Client sent:", repr(data))
103          finally:
104              self.sa.ctxt.RevertSecurityContext()
105          self.close_request(request)
106          print("The server is back to user", GetUserName())
107

109  def serve():
110      s = SSPISocketServer(("localhost", options.port), None)
111      print("Running test server...")
112      s.serve_forever()

115  def sspi_client():
116      c = http.client.HTTPConnection("localhost", options.port)
117      c.connect()
118      # Do the auth dance.
119      ca = sspi.ClientAuth(options.package, targetspn=options.target_spn)
120      data = None
121      while 1:
122          err, out_buf = ca.authorize(data)
123          _send_msg(c.sock, out_buf[0].Buffer)
124          if err == 0:
125              break
126          data = _get_msg(c.sock)
127      print("Auth dance complete - sending a few encryted messages")
128      # Assume out data is sensitive - encrypt the message.
129      for data in "Hello from the client".split():
130          blob, key = ca.encrypt(data)
131          _send_msg(c.sock, blob)
132          _send_msg(c.sock, key)
133      c.sock.close()
134      print("Client completed.")
