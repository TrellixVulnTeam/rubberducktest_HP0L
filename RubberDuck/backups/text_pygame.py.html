1    ''' 
2    Text Pygame: Draw text with pygame 
3     
4    .. warning:: 
5     
6        Pygame has been deprecated and will be removed in the release after Kivy 
7        1.11.0. 
8    '''
9    
10   __all__ = ('LabelPygame', )
11   
12   from kivy.compat import PY2
13   from kivy.core.text import LabelBase
14   from kivy.core.image import ImageData
15   from kivy.utils import deprecated
16   
17   try:
18       import pygame
19   except:
20       raise
21   
22   pygame_cache = {}
23   pygame_font_handles = {}
24   pygame_cache_order = []
25   
26   # init pygame font
27   try:
28       pygame.ftfont.init()
29   except:
30       pygame.font.init()
31   
32   
33   class LabelPygame(LabelBase):
34   
35       @deprecated(
36           msg='Pygame has been deprecated and will be removed after 1.11.0')
37       def __init__(self, *largs, **kwargs):
38           super(LabelPygame, self).__init__(*largs, **kwargs)
39   
40       def _get_font_id(self):
41           return '|'.join([str(self.options[x]) for x in
42                            ('font_size', 'font_name_r', 'bold', 'italic')])
43   
44       def _get_font(self):
45           fontid = self._get_font_id()
46           if fontid not in pygame_cache:
47               # try first the file if it's a filename
48               font_handle = fontobject = None
49               fontname = self.options['font_name_r']
50               ext = fontname.rsplit('.', 1)
51               if len(ext) == 2:
52                   # try to open the font if it has an extension
53                   font_handle = open(fontname, 'rb')
54                   fontobject = pygame.font.Font(font_handle,
55                                                 int(self.options['font_size']))
56   
57               # fallback to search a system font
58               if fontobject is None:
59                   # try to search the font
60                   font = pygame.font.match_font(
61                       self.options['font_name_r'].replace(' ', ''),
62                       bold=self.options['bold'],
63                       italic=self.options['italic'])
64   
65                   # fontobject
66                   fontobject = pygame.font.Font(font,
67                                                 int(self.options['font_size']))
68               pygame_cache[fontid] = fontobject
69               pygame_font_handles[fontid] = font_handle
70               pygame_cache_order.append(fontid)
71   
72           # to prevent too much file open, limit the number of opened fonts to 64
73           while len(pygame_cache_order) > 64:
74               popid = pygame_cache_order.pop(0)
75               del pygame_cache[popid]
76               font_handle = pygame_font_handles.pop(popid)
77               if font_handle is not None:
78                   font_handle.close()
79   
80           return pygame_cache[fontid]
81   
82       def get_ascent(self):
83           return self._get_font().get_ascent()
84   
85       def get_descent(self):
86           return self._get_font().get_descent()
87   
88       def get_extents(self, text):
89           return self._get_font().size(text)
90   
91       def get_cached_extents(self):
92           return self._get_font().size
93   
94       def _render_begin(self):
95           self._pygame_surface = pygame.Surface(self._size, pygame.SRCALPHA, 32)
96           self._pygame_surface.fill((0, 0, 0, 0))
97   
98       def _render_text(self, text, x, y):
99           font = self._get_font()
100          color = [c * 255 for c in self.options['color']]
101          color[0], color[2] = color[2], color[0]
102          try:
103              text = font.render(text, True, color)
104              text.set_colorkey(color)
105              self._pygame_surface.blit(text, (x, y), None,
106                                        pygame.BLEND_RGBA_ADD)
107          except pygame.error:
108              pass
109  
110      def _render_end(self):
111          w, h = self._size
112          data = ImageData(w, h,
113                           'rgba', self._pygame_surface.get_buffer().raw)
114  
115          del self._pygame_surface
116  
117          return data
