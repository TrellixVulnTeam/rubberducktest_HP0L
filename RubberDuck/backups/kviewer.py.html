kviewer.py 
1    #!/usr/bin/env python
2    ''' 
3    KViewer 
4    ======= 
5     
6    KViewer, for KV-Viewer, is a simple tool allowing you to dynamically display 
7    a KV file, taking its changes into account (thanks to watchdog). The 
8    idea is to facilitate design using the KV language. It's somewhat related to 
9    the KivyCatalog demo, except it uses an on-disc file, allowing the user to use 
10   any editor. 
11    
12   You can use the script as follows:: 
13    
14       python kviewer.py ./test.kv 
15    
16   This will display the test.kv and automatically update the display when the 
17   file changes. 
18    
19   .. note: This scripts uses watchdog to listen for file changes. To install 
20      watchdog:: 
21    
22      pip install watchdog 
23    
24   '''
25   
26   from sys import argv
27   from kivy.lang import Builder
28   from kivy.app import App
29   from kivy.core.window import Window
30   from kivy.clock import Clock, mainthread
31   from kivy.uix.label import Label
32   
33   from watchdog.observers import Observer
34   from watchdog.events import FileSystemEventHandler
35   from os.path import dirname, basename, join
36   
37   
38   if len(argv) != 2:
39       print('usage: %s filename.kv' % argv[0])
40       exit(1)
41   
42   
43   PATH = dirname(argv[1])
44   TARGET = basename(argv[1])
45   
46   
47   class KvHandler(FileSystemEventHandler):
48       def __init__(self, callback, target, **kwargs):
49           super(KvHandler, self).__init__(**kwargs)
50           self.callback = callback
51           self.target = target
52   
53       def on_any_event(self, event):
54           if basename(event.src_path) == self.target:
55               self.callback()
56   
57   
58   class KvViewerApp(App):
59       def build(self):
60           o = Observer()
61           o.schedule(KvHandler(self.update, TARGET), PATH)
62           o.start()
63           Clock.schedule_once(self.update, 1)
64           return super(KvViewerApp, self).build()
65   
66       @mainthread
67       def update(self, *args):
68           Builder.unload_file(join(PATH, TARGET))
69           for w in Window.children[:]:
70               Window.remove_widget(w)
71           try:
72               Window.add_widget(Builder.load_file(join(PATH, TARGET)))
73           except Exception as e:
74               Window.add_widget(Label(text=(
75                   e.message if getattr(e, r'message', None) else str(e)
76               )))
77   
78   
79   if __name__ == '__main__':
80       KvViewerApp().run()
