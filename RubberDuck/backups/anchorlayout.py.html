1    ''' 
2    Anchor Layout 
3    ============= 
4     
5    .. only:: html 
6     
7        .. image:: images/anchorlayout.gif 
8            :align: right 
9     
10   .. only:: latex 
11    
12       .. image:: images/anchorlayout.png 
13           :align: right 
14    
15   The :class:`AnchorLayout` aligns its children to a border (top, bottom, 
16   left, right) or center. 
17    
18    
19   To draw a button in the lower-right corner:: 
20    
21       layout = AnchorLayout( 
22           anchor_x='right', anchor_y='bottom') 
23       btn = Button(text='Hello World') 
24       layout.add_widget(btn) 
25    
26   '''
27   
28   __all__ = ('AnchorLayout', )
29   
30   from kivy.uix.layout import Layout
31   from kivy.properties import OptionProperty, VariableListProperty
32   
33   
34   class AnchorLayout(Layout):
35       '''Anchor layout class. See the module documentation for more information. 
36       '''
37   
38       padding = VariableListProperty([0, 0, 0, 0])
39       '''Padding between the widget box and its children, in pixels: 
40       [padding_left, padding_top, padding_right, padding_bottom]. 
41    
42       padding also accepts a two argument form [padding_horizontal, 
43       padding_vertical] and a one argument form [padding]. 
44    
45       :attr:`padding` is a :class:`~kivy.properties.VariableListProperty` and 
46       defaults to [0, 0, 0, 0]. 
47       '''
48   
49       anchor_x = OptionProperty('center', options=(
50           'left', 'center', 'right'))
51       '''Horizontal anchor. 
52    
53       :attr:`anchor_x` is an :class:`~kivy.properties.OptionProperty` and 
54       defaults to 'center'. It accepts values of 'left', 'center' or 
55       'right'. 
56       '''
57   
58       anchor_y = OptionProperty('center', options=(
59           'top', 'center', 'bottom'))
60       '''Vertical anchor. 
61    
62       :attr:`anchor_y` is an :class:`~kivy.properties.OptionProperty` and 
63       defaults to 'center'. It accepts values of 'top', 'center' or 
64       'bottom'. 
65       '''
66   
67       def __init__(self, **kwargs):
68           super(AnchorLayout, self).__init__(**kwargs)
69           fbind = self.fbind
70           update = self._trigger_layout
71           fbind('children', update)
72           fbind('parent', update)
73           fbind('padding', update)
74           fbind('anchor_x', update)
75           fbind('anchor_y', update)
76           fbind('size', update)
77           fbind('pos', update)
78   
79       def do_layout(self, *largs):
80           _x, _y = self.pos
81           width = self.width
82           height = self.height
83           anchor_x = self.anchor_x
84           anchor_y = self.anchor_y
85           pad_left, pad_top, pad_right, pad_bottom = self.padding
86   
87           for c in self.children:
88               x, y = _x, _y
89               cw, ch = c.size
90               shw, shh = c.size_hint
91               shw_min, shh_min = c.size_hint_min
92               shw_max, shh_max = c.size_hint_max
93   
94               if shw is not None:
95                   cw = shw * (width - pad_left - pad_right)
96                   if shw_min is not None and cw < shw_min:
97                       cw = shw_min
98                   elif shw_max is not None and cw > shw_max:
99                       cw = shw_max
100  
101              if shh is not None:
102                  ch = shh * (height - pad_top - pad_bottom)
103                  if shh_min is not None and ch < shh_min:
104                      ch = shh_min
105                  elif shh_max is not None and ch > shh_max:
106                      ch = shh_max
107  
108              if anchor_x == 'left':
109                  x = x + pad_left
110              elif anchor_x == 'right':
111                  x = x + width - (cw + pad_right)
112              else:
113                  x = x + (width - pad_right + pad_left - cw) / 2
114              if anchor_y == 'bottom':
115                  y = y + pad_bottom
116              elif anchor_y == 'top':
117                  y = y + height - (ch + pad_top)
118              else:
119                  y = y + (height - pad_top + pad_bottom - ch) / 2
120  
121              c.pos = x, y
122              c.size = cw, ch
