1    ''' 
2    Accordion 
3    ========= 
4     
5    .. versionadded:: 1.0.8 
6     
7     
8    .. image:: images/accordion.jpg 
9        :align: right 
10    
11   The Accordion widget is a form of menu where the options are stacked either 
12   vertically or horizontally and the item in focus (when touched) opens up to 
13   display its content. 
14    
15   The :class:`Accordion` should contain one or many :class:`AccordionItem` 
16   instances, each of which should contain one root content widget. You'll end up 
17   with a Tree something like this: 
18    
19   - Accordion 
20    
21     - AccordionItem 
22    
23       - YourContent 
24    
25     - AccordionItem 
26    
27       - BoxLayout 
28    
29         - Another user content 1 
30    
31         - Another user content 2 
32    
33     - AccordionItem 
34    
35       - Another user content 
36    
37    
38   The current implementation divides the :class:`AccordionItem` into two parts: 
39    
40   #. One container for the title bar 
41   #. One container for the content 
42    
43   The title bar is made from a Kv template. We'll see how to create a new 
44   template to customize the design of the title bar. 
45    
46   .. warning:: 
47    
48       If you see message like:: 
49    
50           [WARNING] [Accordion] not have enough space for displaying all children 
51           [WARNING] [Accordion] need 440px, got 100px 
52           [WARNING] [Accordion] layout aborted. 
53    
54       That means you have too many children and there is no more space to 
55       display the content. This is "normal" and nothing will be done. Try to 
56       increase the space for the accordion or reduce the number of children. You 
57       can also reduce the :attr:`Accordion.min_space`. 
58    
59   Simple example 
60   -------------- 
61    
62   .. include:: ../../examples/widgets/accordion_1.py 
63       :literal: 
64    
65   Customize the accordion 
66   ----------------------- 
67    
68   You can increase the default size of the title bar:: 
69    
70       root = Accordion(min_space=60) 
71    
72   Or change the orientation to vertical:: 
73    
74       root = Accordion(orientation='vertical') 
75    
76   The AccordionItem is more configurable and you can set your own title 
77   background when the item is collapsed or opened:: 
78    
79       item = AccordionItem(background_normal='image_when_collapsed.png', 
80           background_selected='image_when_selected.png') 
81    
82   '''

103  class AccordionItem(FloatLayout):
104      '''AccordionItem class that must be used in conjunction with the 
105      :class:`Accordion` class. See the module documentation for more 
106      information. 
107      '''
108  
109      title = StringProperty('')
110      '''Title string of the item. The title might be used in conjunction with the 
111      `AccordionItemTitle` template. If you are using a custom template, you can 
112      use that property as a text entry, or not. By default, it's used for the 
113      title text. See title_template and the example below. 
114   
115      :attr:`title` is a :class:`~kivy.properties.StringProperty` and defaults 
116      to ''. 
117      '''
118  
119      title_template = StringProperty('AccordionItemTitle')
120      '''Template to use for creating the title part of the accordion item. The 
121      default template is a simple Label, not customizable (except the text) that 
122      supports vertical and horizontal orientation and different backgrounds for 
123      collapse and selected mode. 
124   
125      It's better to create and use your own template if the default template 
126      does not suffice. 
127   
128      :attr:`title` is a :class:`~kivy.properties.StringProperty` and defaults to 
129      'AccordionItemTitle'. The current default template lives in the 
130      `kivy/data/style.kv` file. 
131   
132      Here is the code if you want to build your own template:: 
133   
134          [AccordionItemTitle@Label]: 
135              text: ctx.title 
136              canvas.before: 
137                  Color: 
138                      rgb: 1, 1, 1 
139                  BorderImage: 
140                      source: 
141                          ctx.item.background_normal \ 
142                          if ctx.item.collapse \ 
143                          else ctx.item.background_selected 
144                      pos: self.pos 
145                      size: self.size 
146                  PushMatrix 
147                  Translate: 
148                      xy: self.center_x, self.center_y 
149                  Rotate: 
150                      angle: 90 if ctx.item.orientation == 'horizontal' else 0 
151                      axis: 0, 0, 1 
152                  Translate: 
153                      xy: -self.center_x, -self.center_y 
154              canvas.after: 
155                  PopMatrix 
156   
157   
158      '''
159  
160      title_args = DictProperty({})
161      '''Default arguments that will be passed to the 
162      :meth:`kivy.lang.Builder.template` method. 
163   
164      :attr:`title_args` is a :class:`~kivy.properties.DictProperty` and defaults 
165      to {}. 
166      '''
167  
168      collapse = BooleanProperty(True)
169      '''Boolean to indicate if the current item is collapsed or not. 
170   
171      :attr:`collapse` is a :class:`~kivy.properties.BooleanProperty` and 
172      defaults to True. 
173      '''
174  
175      collapse_alpha = NumericProperty(1.)
176      '''Value between 0 and 1 to indicate how much the item is collapsed (1) or 
177      whether it is selected (0). It's mostly used for animation. 
178   
179      :attr:`collapse_alpha` is a :class:`~kivy.properties.NumericProperty` and 
180      defaults to 1. 
181      '''
182  
183      accordion = ObjectProperty(None)
184      '''Instance of the :class:`Accordion` that the item belongs to. 
185   
186      :attr:`accordion` is an :class:`~kivy.properties.ObjectProperty` and 
187      defaults to None. 
188      '''
189  
190      background_normal = StringProperty(
191          'atlas://data/images/defaulttheme/button')
192      '''Background image of the accordion item used for the default graphical 
193      representation when the item is collapsed. 
194   
195      :attr:`background_normal` is a :class:`~kivy.properties.StringProperty` and 
196      defaults to 'atlas://data/images/defaulttheme/button'. 
197      '''
198  
199      background_disabled_normal = StringProperty(
200          'atlas://data/images/defaulttheme/button_disabled')
201      '''Background image of the accordion item used for the default graphical 
202      representation when the item is collapsed and disabled. 
203   
204      .. versionadded:: 1.8.0 
205   
206      :attr:`background__disabled_normal` is a 
207      :class:`~kivy.properties.StringProperty` and defaults to 
208      'atlas://data/images/defaulttheme/button_disabled'. 
209      '''
210  
211      background_selected = StringProperty(
212          'atlas://data/images/defaulttheme/button_pressed')
213      '''Background image of the accordion item used for the default graphical 
214      representation when the item is selected (not collapsed). 
215   
216      :attr:`background_normal` is a :class:`~kivy.properties.StringProperty` and 
217      defaults to 'atlas://data/images/defaulttheme/button_pressed'. 
218      '''
219  
220      background_disabled_selected = StringProperty(
221          'atlas://data/images/defaulttheme/button_disabled_pressed')
222      '''Background image of the accordion item used for the default graphical 
223      representation when the item is selected (not collapsed) and disabled. 
224   
225      .. versionadded:: 1.8.0 
226   
227      :attr:`background_disabled_selected` is a 
228      :class:`~kivy.properties.StringProperty` and defaults to 
229      'atlas://data/images/defaulttheme/button_disabled_pressed'. 
230      '''
231  
232      orientation = OptionProperty('vertical', options=(
233          'horizontal', 'vertical'))
234      '''Link to the :attr:`Accordion.orientation` property. 
235      '''
236  
237      min_space = NumericProperty('44dp')
238      '''Link to the :attr:`Accordion.min_space` property. 
239      '''
240  
241      content_size = ListProperty([100, 100])
242      '''(internal) Set by the :class:`Accordion` to the size allocated for the 
243      content. 
244      '''
245  
246      container = ObjectProperty(None)
247      '''(internal) Property that will be set to the container of children inside 
248      the AccordionItem representation. 
249      '''
250  
251      container_title = ObjectProperty(None)
252      '''(internal) Property that will be set to the container of title inside 
253      the AccordionItem representation. 
254      '''
255  
256      def __init__(self, **kwargs):
257          self._trigger_title = Clock.create_trigger(self._update_title, -1)
258          self._anim_collapse = None
259          super(AccordionItem, self).__init__(**kwargs)
260          trigger_title = self._trigger_title
261          fbind = self.fbind
262          fbind('title', trigger_title)
263          fbind('title_template', trigger_title)
264          fbind('title_args', trigger_title)
265          trigger_title()
266  
267      def add_widget(self, widget):
268          if self.container is None:
269              return super(AccordionItem, self).add_widget(widget)
270          return self.container.add_widget(widget)
271  
272      def remove_widget(self, widget):
273          if self.container:
274              self.container.remove_widget(widget)
275          super(AccordionItem, self).remove_widget(widget)
276  
277      def on_collapse(self, instance, value):
278          accordion = self.accordion
279          if accordion is None:
280              return
281          if not value:
282              self.accordion.select(self)
283          collapse_alpha = float(value)
284          if self._anim_collapse:
285              self._anim_collapse.stop(self)
286              self._anim_collapse = None
287          if self.collapse_alpha != collapse_alpha:
288              self._anim_collapse = Animation(
289                  collapse_alpha=collapse_alpha,
290                  t=accordion.anim_func,
291                  d=accordion.anim_duration)
292              self._anim_collapse.start(self)
293  
294      def on_collapse_alpha(self, instance, value):
295          self.accordion._trigger_layout()
296  
297      def on_touch_down(self, touch):
298          if not self.collide_point(*touch.pos):
299              return
300          if self.disabled:
301              return True
302          if self.collapse:
303              self.collapse = False
304              return True
305          else:
306              return super(AccordionItem, self).on_touch_down(touch)
307  
308      def _update_title(self, dt):
309          if not self.container_title:
310              self._trigger_title()
311              return
312          c = self.container_title
313          c.clear_widgets()
314          instance = Builder.template(self.title_template,
315                                      title=self.title,
316                                      item=self,
317                                      **self.title_args)
318          c.add_widget(instance)

321  class Accordion(Widget):
322      '''Accordion class. See module documentation for more information. 
323      '''
324  
325      orientation = OptionProperty('horizontal', options=(
326          'horizontal', 'vertical'))
327      '''Orientation of the layout. 
328   
329      :attr:`orientation` is an :class:`~kivy.properties.OptionProperty` 
330      and defaults to 'horizontal'. Can take a value of 'vertical' or 
331      'horizontal'. 
332   
333      '''
334  
335      anim_duration = NumericProperty(.25)
336      '''Duration of the animation in seconds when a new accordion item is 
337      selected. 
338   
339      :attr:`anim_duration` is a :class:`~kivy.properties.NumericProperty` and 
340      defaults to .25 (250ms). 
341      '''
342  
343      anim_func = ObjectProperty('out_expo')
344      '''Easing function to use for the animation. Check 
345      :class:`kivy.animation.AnimationTransition` for more information about 
346      available animation functions. 
347   
348      :attr:`anim_func` is an :class:`~kivy.properties.ObjectProperty` and 
349      defaults to 'out_expo'. You can set a string or a function to use as an 
350      easing function. 
351      '''
352  
353      min_space = NumericProperty('44dp')
354      '''Minimum space to use for the title of each item. This value is 
355      automatically set for each child every time the layout event occurs. 
356   
357      :attr:`min_space` is a :class:`~kivy.properties.NumericProperty` and 
358      defaults to 44 (px). 
359      '''
360  
361      def __init__(self, **kwargs):
362          super(Accordion, self).__init__(**kwargs)
363          update = self._trigger_layout = \
364              Clock.create_trigger(self._do_layout, -1)
365          fbind = self.fbind
366          fbind('orientation', update)
367          fbind('children', update)
368          fbind('size', update)
369          fbind('pos', update)
370          fbind('min_space', update)
371  
372      def add_widget(self, widget, *largs):
373          if not isinstance(widget, AccordionItem):
374              raise AccordionException('Accordion accept only AccordionItem')
375  
376          widget.accordion = self
377          ret = super(Accordion, self).add_widget(widget, *largs)
378          return ret
379  
380      def select(self, instance):
381          if instance not in self.children:
382              raise AccordionException(
383                  'Accordion: instance not found in children')
384          for widget in self.children:
385              widget.collapse = widget is not instance
386          self._trigger_layout()
387  
388      def _do_layout(self, dt):
389          children = self.children
390          if children:
391              all_collapsed = all(x.collapse for x in children)
392          else:
393              all_collapsed = False
394  
395          if all_collapsed:
396              children[0].collapse = False
397  
398          orientation = self.orientation
399          min_space = self.min_space
400          min_space_total = len(children) * self.min_space
401          w, h = self.size
402          x, y = self.pos
403          if orientation == 'horizontal':
404              display_space = self.width - min_space_total
405          else:
406              display_space = self.height - min_space_total
407  
408          if display_space <= 0:
409              Logger.warning('Accordion: not enough space '
410                             'for displaying all children')
411              Logger.warning('Accordion: need %dpx, got %dpx' % (
412                  min_space_total, min_space_total + display_space))
413              Logger.warning('Accordion: layout aborted.')
414              return
415  
416          if orientation == 'horizontal':
417              children = reversed(children)
418  
419          for child in children:
420              child_space = min_space
421              child_space += display_space * (1 - child.collapse_alpha)
422              child._min_space = min_space
423              child.x = x
424              child.y = y
425              child.orientation = self.orientation
426              if orientation == 'horizontal':
427                  child.content_size = display_space, h
428                  child.width = child_space
429                  child.height = h
430                  x += child_space
431              else:
432                  child.content_size = w, display_space
433                  child.width = w
434                  child.height = child_space
435                  y += child_space

439      from kivy.base import runTouchApp
440      from kivy.uix.button import Button
441      from kivy.uix.boxlayout import BoxLayout
442      from kivy.uix.label import Label
443  
444      acc = Accordion()
445      for x in range(10):
446          item = AccordionItem(title='Title %d' % x)
447          if x == 0:
448              item.add_widget(Button(text='Content %d' % x))
449          elif x == 1:
450              z = BoxLayout(orientation='vertical')
451              z.add_widget(Button(text=str(x), size_hint_y=None, height=35))
452              z.add_widget(Label(text='Content %d' % x))
453              item.add_widget(z)
454          else:
455              item.add_widget(Label(text='This is a big content\n' * 20))
456          acc.add_widget(item)
457  
458      def toggle_layout(*l):
459          o = acc.orientation
460          acc.orientation = 'vertical' if o == 'horizontal' else 'horizontal'
461      btn = Button(text='Toggle layout')
462      btn.bind(on_release=toggle_layout)
463  
464      def select_2nd_item(*l):
465          acc.select(acc.children[-2])
466      btn2 = Button(text='Select 2nd item')
467      btn2.bind(on_release=select_2nd_item)
468  
469      from kivy.uix.slider import Slider
470      slider = Slider()
471  
472      def update_min_space(instance, value):
473          acc.min_space = value
474  
475      slider.bind(value=update_min_space)
476  
477      root = BoxLayout(spacing=20, padding=20)
478      controls = BoxLayout(orientation='vertical', size_hint_x=.3)
479      controls.add_widget(btn)
480      controls.add_widget(btn2)
481      controls.add_widget(slider)
482      root.add_widget(controls)
483      root.add_widget(acc)
484      runTouchApp(root)
