1    from win32com.server.util import wrap
2    import pythoncom, sys, os, time, win32api, win32event, tempfile
3    from win32com.bits import bits
4    
5    TIMEOUT = 200  # ms
6    StopEvent = win32event.CreateEvent(None, 0, 0, None)
7    
8    job_name = "bits-pywin32-test"
9    states = dict(
10       [
11           (val, (name[13:]))
12           for name, val in vars(bits).items()
13           if name.startswith("BG_JOB_STATE_")
14       ]
15   )
16   
17   bcm = pythoncom.CoCreateInstance(
18       bits.CLSID_BackgroundCopyManager,
19       None,
20       pythoncom.CLSCTX_LOCAL_SERVER,
21       bits.IID_IBackgroundCopyManager,
22   )
23   
24   
25   class BackgroundJobCallback:
26       _com_interfaces_ = [bits.IID_IBackgroundCopyCallback]
27       _public_methods_ = ["JobTransferred", "JobError", "JobModification"]
28   
29       def JobTransferred(self, job):
30           print("Job Transferred", job)
31           job.Complete()
32           win32event.SetEvent(StopEvent)  # exit msg pump
33   
34       def JobError(self, job, error):
35           print("Job Error", job, error)
36           f = error.GetFile()
37           print("While downloading", f.GetRemoteName())
38           print("To", f.GetLocalName())
39           print("The following error happened:")
40           self._print_error(error)
41           if f.GetRemoteName().endswith("missing-favicon.ico"):
42               print("Changing to point to correct file")
43               f2 = f.QueryInterface(bits.IID_IBackgroundCopyFile2)
44               favicon = "http://www.python.org/favicon.ico"
45               print("Changing RemoteName from", f2.GetRemoteName(), "to", favicon)
46               f2.SetRemoteName(favicon)
47               job.Resume()
48           else:
49               job.Cancel()
50   
51       def _print_error(self, err):
52           ctx, hresult = err.GetError()
53           try:
54               hresult_msg = win32api.FormatMessage(hresult)
55           except win32api.error:
56               hresult_msg = ""
57           print("Context=0x%x, hresult=0x%x (%s)" % (ctx, hresult, hresult_msg))
58           print(err.GetErrorDescription())
59   
60       def JobModification(self, job, reserved):
61           state = job.GetState()
62           print("Job Modification", job.GetDisplayName(), states.get(state))
63           # Need to catch TRANSIENT_ERROR here, as JobError doesn't get
64           # called (apparently) when the error is transient.
65           if state == bits.BG_JOB_STATE_TRANSIENT_ERROR:
66               print("Error details:")
67               err = job.GetError()
68               self._print_error(err)
69   
70   
71   job = bcm.CreateJob(job_name, bits.BG_JOB_TYPE_DOWNLOAD)
72   
73   job.SetNotifyInterface(wrap(BackgroundJobCallback()))
74   job.SetNotifyFlags(
75       bits.BG_NOTIFY_JOB_TRANSFERRED
76       | bits.BG_NOTIFY_JOB_ERROR
77       | bits.BG_NOTIFY_JOB_MODIFICATION
78   )
79   
80   
81   # The idea here is to intentionally make one of the files fail to be
82   # downloaded. Then the JobError notification will be triggered, where
83   # we do fix the failing file by calling SetRemoteName to a valid URL
84   # and call Resume() on the job, making the job finish successfully.
85   #
86   # Note to self: A domain that cannot be resolved will cause
87   # TRANSIENT_ERROR instead of ERROR, and the JobError notification will
88   # not be triggered! This can bite you during testing depending on how
89   # your DNS is configured. For example, if you use OpenDNS.org's DNS
90   # servers, an invalid hostname will *always* be resolved (they
91   # redirect you to a search page), so be careful when testing.
92   job.AddFile(
93       "http://www.python.org/favicon.ico",
94       os.path.join(tempfile.gettempdir(), "bits-favicon.ico"),
95   )
96   job.AddFile(
97       "http://www.python.org/missing-favicon.ico",
98       os.path.join(tempfile.gettempdir(), "bits-missing-favicon.ico"),
99   )
100  
101  for f in job.EnumFiles():
102      print("Downloading", f.GetRemoteName())
103      print("To", f.GetLocalName())
104  
105  job.Resume()
106  
107  while True:
108      rc = win32event.MsgWaitForMultipleObjects(
109          (StopEvent,), 0, TIMEOUT, win32event.QS_ALLEVENTS
110      )
111  
112      if rc == win32event.WAIT_OBJECT_0:
113          break
114      elif rc == win32event.WAIT_OBJECT_0 + 1:
115          if pythoncom.PumpWaitingMessages():
116              break  # wm_quit
