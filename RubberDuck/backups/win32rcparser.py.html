79           t = None
80           self.template = [
81               [

84                   self.style,

141          ct = self.controlType

152              self.styleEx,
153          ]
154          # print t

168  class RCParser:
169      next_id = 1001
170      dialogs = {}
171      _dialogs = {}
172      debugEnabled = False
173      token = ""
174  
175      def __init__(self):
176          self.ungot = False
177          self.ids = {"IDC_STATIC": -1}
178          self.names = {-1: "IDC_STATIC"}
179          self.bitmaps = {}
180          self.stringTable = {}
181          self.icons = {}
182  
183      def debug(self, *args):
184          if self.debugEnabled:
185              print(args)

187      def getToken(self):
188          if self.ungot:
189              self.ungot = False
190              self.debug("getToken returns (ungot):", self.token)
191              return self.token
192          self.token = self.lex.get_token()
193          self.debug("getToken returns:", self.token)
194          if self.token == "":
195              self.token = None
196          return self.token
197  
198      def ungetToken(self):
199          self.ungot = True
200  
201      def getCheckToken(self, expected):
202          tok = self.getToken()
203          assert tok == expected, "Expected token '%s', but got token '%s'!" % (
204              expected,
205              tok,
206          )
207          return tok
208  
209      def getCommaToken(self):
210          return self.getCheckToken(",")
211  
212      # Return the *current* token as a number, only consuming a token
213      # if it is the negative-sign.
214      def currentNumberToken(self):
215          mult = 1
216          if self.token == "-":
217              mult = -1
218              self.getToken()
219          return int(self.token) * mult
220  
221      # Return the *current* token as a string literal (ie, self.token will be a
222      # quote.  consumes all tokens until the end of the string
223      def currentQuotedString(self):
224          # Handle quoted strings - pity shlex doesn't handle it.
225          assert self.token.startswith('"'), self.token
226          bits = [self.token]
227          while 1:
228              tok = self.getToken()
229              if not tok.startswith('"'):
230                  self.ungetToken()
231                  break
232              bits.append(tok)
233          sval = "".join(bits)[1:-1]  # Remove end quotes.
234          # Fixup quotes in the body, and all (some?) quoted characters back
235          # to their raw value.
236          for i, o in ('""', '"'), ("\\r", "\r"), ("\\n", "\n"), ("\\t", "\t"):
237              sval = sval.replace(i, o)
238          return sval

240      def load(self, rcstream):
241          """ 
242          RCParser.loadDialogs(rcFileName) -> None 
243          Load the dialog information into the parser. Dialog Definations can then be accessed 
244          using the "dialogs" dictionary member (name->DialogDef). The "ids" member contains the dictionary of id->name. 
245          The "names" member contains the dictionary of name->id 
246          """
247          self.open(rcstream)
248          self.getToken()
249          while self.token != None:
250              self.parse()
251              self.getToken()
252  
253      def open(self, rcstream):
254          self.lex = shlex.shlex(rcstream)
255          self.lex.commenters = "//#"
256  
257      def parseH(self, file):
258          lex = shlex.shlex(file)
259          lex.commenters = "//"
260          token = " "
261          while token is not None:
262              token = lex.get_token()
263              if token == "" or token is None:
264                  token = None
265              else:
266                  if token == "define":
267                      n = lex.get_token()
268                      i = int(lex.get_token())
269                      self.ids[n] = i
270                      if i in self.names:
271                          # Dupe ID really isn't a problem - most consumers
272                          # want to go from name->id, and this is OK.
273                          # It means you can't go from id->name though.
274                          pass
275                          # ignore AppStudio special ones
276                          # if not n.startswith("_APS_"):
277                          #    print "Duplicate id",i,"for",n,"is", self.names[i]
278                      else:
279                          self.names[i] = n
280                      if self.next_id <= i:
281                          self.next_id = i + 1
282  
283      def parse(self):
284          noid_parsers = {
285              "STRINGTABLE": self.parse_stringtable,
286          }
287  
288          id_parsers = {
289              "DIALOG": self.parse_dialog,
290              "DIALOGEX": self.parse_dialog,
291              #            "TEXTINCLUDE":      self.parse_textinclude,
292              "BITMAP": self.parse_bitmap,
293              "ICON": self.parse_icon,
294          }
295          deep = 0
296          base_token = self.token
297          rp = noid_parsers.get(base_token)
298          if rp is not None:
299              rp()
300          else:
301              # Not something we parse that isn't prefixed by an ID
302              # See if it is an ID prefixed item - if it is, our token
303              # is the resource ID.
304              resource_id = self.token
305              self.getToken()
306              if self.token is None:
307                  return
308  
309              if "BEGIN" == self.token:
310                  # A 'BEGIN' for a structure we don't understand - skip to the
311                  # matching 'END'
312                  deep = 1
313                  while deep != 0 and self.token is not None:
314                      self.getToken()
315                      self.debug("Zooming over", self.token)
316                      if "BEGIN" == self.token:
317                          deep += 1
318                      elif "END" == self.token:
319                          deep -= 1
320              else:
321                  rp = id_parsers.get(self.token)
322                  if rp is not None:
323                      self.debug("Dispatching '%s'" % (self.token,))
324                      rp(resource_id)
325                  else:
326                      # We don't know what the resource type is, but we
327                      # have already consumed the next, which can cause problems,
328                      # so push it back.
329                      self.debug("Skipping top-level '%s'" % base_token)
330                      self.ungetToken()
331  
332      def addId(self, id_name):
333          if id_name in self.ids:
334              id = self.ids[id_name]
335          else:
336              # IDOK, IDCANCEL etc are special - if a real resource has this value
337              for n in ["IDOK", "IDCANCEL", "IDYES", "IDNO", "IDABORT"]:
338                  if id_name == n:
339                      v = getattr(win32con, n)
340                      self.ids[n] = v
341                      self.names[v] = n
342                      return v
343              id = self.next_id
344              self.next_id += 1
345              self.ids[id_name] = id
346              self.names[id] = id_name
347          return id

349      def lang(self):
350          while (
351              self.token[0:4] == "LANG"
352              or self.token[0:7] == "SUBLANG"
353              or self.token == ","
354          ):
355              self.getToken()

357      def parse_textinclude(self, res_id):
358          while self.getToken() != "BEGIN":
359              pass
360          while 1:
361              if self.token == "END":
362                  break
363              s = self.getToken()

365      def parse_stringtable(self):
366          while self.getToken() != "BEGIN":
367              pass
368          while 1:
369              self.getToken()
370              if self.token == "END":
371                  break
372              sid = self.token
373              self.getToken()
374              sd = StringDef(sid, self.addId(sid), self.currentQuotedString())
375              self.stringTable[sid] = sd

377      def parse_bitmap(self, name):
378          return self.parse_bitmap_or_icon(name, self.bitmaps)
379  
380      def parse_icon(self, name):
381          return self.parse_bitmap_or_icon(name, self.icons)
382  
383      def parse_bitmap_or_icon(self, name, dic):
384          self.getToken()
385          while not self.token.startswith('"'):
386              self.getToken()
387          bmf = self.token[1:-1]  # quotes
388          dic[name] = bmf

390      def parse_dialog(self, name):
391          dlg = DialogDef(name, self.addId(name))
392          assert len(dlg.controls) == 0
393          self._dialogs[name] = dlg
394          extras = []
395          self.getToken()
396          while not self.token.isdigit():
397              self.debug("extra", self.token)
398              extras.append(self.token)
399              self.getToken()
400          dlg.x = int(self.token)
401          self.getCommaToken()
402          self.getToken()  # number
403          dlg.y = int(self.token)
404          self.getCommaToken()
405          self.getToken()  # number
406          dlg.w = int(self.token)
407          self.getCommaToken()
408          self.getToken()  # number
409          dlg.h = int(self.token)
410          self.getToken()
411          while not (self.token == None or self.token == "" or self.token == "END"):
412              if self.token == "STYLE":
413                  self.dialogStyle(dlg)
414              elif self.token == "EXSTYLE":
415                  self.dialogExStyle(dlg)
416              elif self.token == "CAPTION":
417                  self.dialogCaption(dlg)
418              elif self.token == "FONT":
419                  self.dialogFont(dlg)
420              elif self.token == "BEGIN":
421                  self.controls(dlg)
422              else:
423                  break
424          self.dialogs[name] = dlg.createDialogTemplate()
425  
426      def dialogStyle(self, dlg):
427          dlg.style, dlg.styles = self.styles([], win32con.DS_SETFONT)
428  
429      def dialogExStyle(self, dlg):
430          self.getToken()
431          dlg.styleEx, dlg.stylesEx = self.styles([], 0)
432  
433      def styles(self, defaults, defaultStyle):
434          list = defaults
435          style = defaultStyle
436  
437          if "STYLE" == self.token:
438              self.getToken()
439          i = 0
440          Not = False
441          while (
442              (i % 2 == 1 and ("|" == self.token or "NOT" == self.token)) or (i % 2 == 0)
443          ) and not self.token == None:
444              Not = False
445              if "NOT" == self.token:
446                  Not = True
447                  self.getToken()
448              i += 1
449              if self.token != "|":
450                  if self.token in win32con.__dict__:
451                      value = getattr(win32con, self.token)
452                  else:
453                      if self.token in commctrl.__dict__:
454                          value = getattr(commctrl, self.token)
455                      else:
456                          value = 0
457                  if Not:
458                      list.append("NOT " + self.token)
459                      self.debug("styles add Not", self.token, value)
460                      style &= ~value
461                  else:
462                      list.append(self.token)
463                      self.debug("styles add", self.token, value)
464                      style |= value
465              self.getToken()
466          self.debug("style is ", style)
467  
468          return style, list

470      def dialogCaption(self, dlg):
471          if "CAPTION" == self.token:
472              self.getToken()
473          self.token = self.token[1:-1]
474          self.debug("Caption is:", self.token)
475          dlg.caption = self.token
476          self.getToken()

478      def dialogFont(self, dlg):
479          if "FONT" == self.token:
480              self.getToken()
481          dlg.fontSize = int(self.token)
482          self.getCommaToken()
483          self.getToken()  # Font name
484          dlg.font = self.token[1:-1]  # it's quoted
485          self.getToken()
486          while "BEGIN" != self.token:
487              self.getToken()

489      def controls(self, dlg):
490          if self.token == "BEGIN":
491              self.getToken()
492          # All controls look vaguely like:
493          # TYPE [text, ] Control_id, l, t, r, b [, style]
494          # .rc parser documents all control types as:
495          # CHECKBOX, COMBOBOX, CONTROL, CTEXT, DEFPUSHBUTTON, EDITTEXT, GROUPBOX,
496          # ICON, LISTBOX, LTEXT, PUSHBUTTON, RADIOBUTTON, RTEXT, SCROLLBAR
497          without_text = ["EDITTEXT", "COMBOBOX", "LISTBOX", "SCROLLBAR"]
498          while self.token != "END":
499              control = ControlDef()
500              control.controlType = self.token
501              self.getToken()
502              if control.controlType not in without_text:
503                  if self.token[0:1] == '"':
504                      control.label = self.currentQuotedString()
505                  # Some funny controls, like icons and picture controls use
506                  # the "window text" as extra resource ID (ie, the ID of the
507                  # icon itself).  This may be either a literal, or an ID string.
508                  elif self.token == "-" or self.token.isdigit():
509                      control.label = str(self.currentNumberToken())
510                  else:
511                      # An ID - use the numeric equiv.
512                      control.label = str(self.addId(self.token))
513                  self.getCommaToken()
514                  self.getToken()
515              # Control IDs may be "names" or literal ints
516              if self.token == "-" or self.token.isdigit():
517                  control.id = self.currentNumberToken()
518                  control.idNum = control.id
519              else:
520                  # name of an ID
521                  control.id = self.token
522                  control.idNum = self.addId(control.id)
523              self.getCommaToken()
524  
525              if control.controlType == "CONTROL":
526                  self.getToken()
527                  control.subType = self.token[1:-1]
528                  thisDefaultStyle = defaultControlStyle | _addDefaults.get(
529                      control.subType, 0
530                  )
531                  # Styles
532                  self.getCommaToken()
533                  self.getToken()
534                  control.style, control.styles = self.styles([], thisDefaultStyle)
535              else:
536                  thisDefaultStyle = defaultControlStyle | _addDefaults.get(
537                      control.controlType, 0
538                  )
539                  # incase no style is specified.
540                  control.style = thisDefaultStyle
541              # Rect
542              control.x = int(self.getToken())
543              self.getCommaToken()
544              control.y = int(self.getToken())
545              self.getCommaToken()
546              control.w = int(self.getToken())
547              self.getCommaToken()
548              self.getToken()
549              control.h = int(self.token)
550              self.getToken()
551              if self.token == ",":
552                  self.getToken()
553                  control.style, control.styles = self.styles([], thisDefaultStyle)
554              if self.token == ",":
555                  self.getToken()
556                  control.styleEx, control.stylesEx = self.styles(
557                      [], defaultControlStyleEx
558                  )
559              # print control.toString()
560              dlg.controls.append(control)

563  def ParseStreams(rc_file, h_file):
564      rcp = RCParser()
565      if h_file:
566          rcp.parseH(h_file)
567      try:
568          rcp.load(rc_file)
569      except:
570          lex = getattr(rcp, "lex", None)
571          if lex:
572              print("ERROR parsing dialogs at line", lex.lineno)
573              print("Next 10 tokens are:")
574              for i in range(10):
575                  print(lex.get_token(), end=" ")
576              print()
577          raise
578      return rcp

581  def Parse(rc_name, h_name=None):
582      if h_name:
583          h_file = open(h_name, "r")
584      else:
585          # See if same basename as the .rc
586          h_name = rc_name[:-2] + "h"
587          try:
588              h_file = open(h_name, "r")
589          except IOError:
590              # See if MSVC default of 'resource.h' in the same dir.
591              h_name = os.path.join(os.path.dirname(rc_name), "resource.h")
592              try:
593                  h_file = open(h_name, "r")
594              except IOError:
595                  # .h files are optional anyway
596                  h_file = None
597      rc_file = open(rc_name, "r")
598      try:
599          return ParseStreams(rc_file, h_file)
600      finally:
601          if h_file is not None:
602              h_file.close()
603          rc_file.close()
604      return rcp

607  def GenerateFrozenResource(rc_name, output_name, h_name=None):
608      """Converts an .rc windows resource source file into a python source file 
609      with the same basic public interface as the rest of this module. 
610      Particularly useful for py2exe or other 'freeze' type solutions, 
611      where a frozen .py file can be used inplace of a real .rc file. 
612      """
613      rcp = Parse(rc_name, h_name)
614      in_stat = os.stat(rc_name)
615  
616      out = open(output_name, "wt")
617      out.write("#%s\n" % output_name)
618      out.write("#This is a generated file. Please edit %s instead.\n" % rc_name)
619      out.write("__version__=%r\n" % __version__)
620      out.write(
621          "_rc_size_=%d\n_rc_mtime_=%d\n"
622          % (in_stat[stat.ST_SIZE], in_stat[stat.ST_MTIME])
623      )
624  
625      out.write("class StringDef:\n")
626      out.write("\tdef __init__(self, id, idNum, value):\n")
627      out.write("\t\tself.id = id\n")
628      out.write("\t\tself.idNum = idNum\n")
629      out.write("\t\tself.value = value\n")
630      out.write("\tdef __repr__(self):\n")
631      out.write(
632          '\t\treturn "StringDef(%r, %r, %r)" % (self.id, self.idNum, self.value)\n'
633      )
634  
635      out.write("class FakeParser:\n")
636  
637      for name in "dialogs", "ids", "names", "bitmaps", "icons", "stringTable":
638          out.write("\t%s = \\\n" % (name,))
639          pprint.pprint(getattr(rcp, name), out)
640          out.write("\n")
641  
642      out.write("def Parse(s):\n")
643      out.write("\treturn FakeParser()\n")
644      out.close()
