24   class LuaLexer(RegexLexer):
25       """ 
26       For `Lua <http://www.lua.org>`_ source code. 
27    
28       Additional options accepted: 
29    
30       `func_name_highlighting` 
31           If given and ``True``, highlight builtin function names 
32           (default: ``True``). 
33       `disabled_modules` 
34           If given, must be a list of module names whose function names 
35           should not be highlighted. By default all modules are highlighted. 
36    
37           To get a list of allowed modules have a look into the 
38           `_lua_builtins` module: 
39    
40           .. sourcecode:: pycon 
41    
42               >>> from pygments.lexers._lua_builtins import MODULES 
43               >>> MODULES.keys() 
44               ['string', 'coroutine', 'modules', 'io', 'basic', ...] 
45       """
46   
47       name = 'Lua'
48       aliases = ['lua']
49       filenames = ['*.lua', '*.wlua']
50       mimetypes = ['text/x-lua', 'application/x-lua']
51   
52       _comment_multiline = r'(?:--\[(?P<level>=*)\[[\w\W]*?\](?P=level)\])'
53       _comment_single = r'(?:--.*$)'
54       _space = r'(?:\s+)'
55       _s = r'(?:%s|%s|%s)' % (_comment_multiline, _comment_single, _space)
56       _name = r'(?:[^\W\d]\w*)'
57   
58       tokens = {
59           'root': [
60               # Lua allows a file to start with a shebang.
61               (r'#!.*', Comment.Preproc),
62               default('base'),
63           ],
64           'ws': [
65               (_comment_multiline, Comment.Multiline),
66               (_comment_single, Comment.Single),
67               (_space, Text),
68           ],
69           'base': [
70               include('ws'),
71   
72               (r'(?i)0x[\da-f]*(\.[\da-f]*)?(p[+-]?\d+)?', Number.Hex),
73               (r'(?i)(\d*\.\d+|\d+\.\d*)(e[+-]?\d+)?', Number.Float),
74               (r'(?i)\d+e[+-]?\d+', Number.Float),
75               (r'\d+', Number.Integer),
76   
77               # multiline strings
78               (r'(?s)\[(=*)\[.*?\]\1\]', String),
79   
80               (r'::', Punctuation, 'label'),
81               (r'\.{3}', Punctuation),
82               (r'[=<>|~&+\-*/%#^]+|\.\.', Operator),
83               (r'[\[\]{}().,:;]', Punctuation),
84               (r'(and|or|not)\b', Operator.Word),
85   
86               ('(break|do|else|elseif|end|for|if|in|repeat|return|then|until|'
87                r'while)\b', Keyword.Reserved),
88               (r'goto\b', Keyword.Reserved, 'goto'),
89               (r'(local)\b', Keyword.Declaration),
90               (r'(true|false|nil)\b', Keyword.Constant),
91   
92               (r'(function)\b', Keyword.Reserved, 'funcname'),
93   
94               (r'[A-Za-z_]\w*(\.[A-Za-z_]\w*)?', Name),
95   
96               ("'", String.Single, combined('stringescape', 'sqs')),
97               ('"', String.Double, combined('stringescape', 'dqs'))
98           ],
99   
100          'funcname': [
101              include('ws'),
102              (r'[.:]', Punctuation),
103              (r'%s(?=%s*[.:])' % (_name, _s), Name.Class),
104              (_name, Name.Function, '#pop'),
105              # inline function
106              (r'\(', Punctuation, '#pop'),
107          ],
108  
109          'goto': [
110              include('ws'),
111              (_name, Name.Label, '#pop'),
112          ],
113  
114          'label': [
115              include('ws'),
116              (r'::', Punctuation, '#pop'),
117              (_name, Name.Label),
118          ],
119  
120          'stringescape': [
121              (r'\\([abfnrtv\\"\']|[\r\n]{1,2}|z\s*|x[0-9a-fA-F]{2}|\d{1,3}|'
122               r'u\{[0-9a-fA-F]+\})', String.Escape),
123          ],
124  
125          'sqs': [
126              (r"'", String.Single, '#pop'),
127              (r"[^\\']+", String.Single),
128          ],
129  
130          'dqs': [
131              (r'"', String.Double, '#pop'),
132              (r'[^\\"]+', String.Double),
133          ]
134      }
135  
136      def __init__(self, **options):
137          self.func_name_highlighting = get_bool_opt(
138              options, 'func_name_highlighting', True)
139          self.disabled_modules = get_list_opt(options, 'disabled_modules', [])
140  
141          self._functions = set()
142          if self.func_name_highlighting:
143              from pygments.lexers._lua_builtins import MODULES
144              for mod, func in MODULES.items():
145                  if mod not in self.disabled_modules:
146                      self._functions.update(func)
147          RegexLexer.__init__(self, **options)
148  
149      def get_tokens_unprocessed(self, text):
150          for index, token, value in \
151                  RegexLexer.get_tokens_unprocessed(self, text):
152              if token is Name:
153                  if value in self._functions:
154                      yield index, Name.Builtin, value
155                      continue
156                  elif '.' in value:
157                      a, b = value.split('.')
158                      yield index, Name, a
159                      yield index + len(a), Punctuation, '.'
160                      yield index + len(a) + 1, Name, b
161                      continue
162              yield index, token, value

164  class MoonScriptLexer(LuaLexer):
165      """ 
166      For `MoonScript <http://moonscript.org>`_ source code. 
167   
168      .. versionadded:: 1.5 
169      """
170  
171      name = 'MoonScript'
172      aliases = ['moonscript', 'moon']
173      filenames = ['*.moon']
174      mimetypes = ['text/x-moonscript', 'application/x-moonscript']
175  
176      tokens = {
177          'root': [
178              (r'#!(.*?)$', Comment.Preproc),
179              default('base'),
180          ],
181          'base': [
182              ('--.*$', Comment.Single),
183              (r'(?i)(\d*\.\d+|\d+\.\d*)(e[+-]?\d+)?', Number.Float),
184              (r'(?i)\d+e[+-]?\d+', Number.Float),
185              (r'(?i)0x[0-9a-f]*', Number.Hex),
186              (r'\d+', Number.Integer),
187              (r'\n', Text),
188              (r'[^\S\n]+', Text),
189              (r'(?s)\[(=*)\[.*?\]\1\]', String),
190              (r'(->|=>)', Name.Function),
191              (r':[a-zA-Z_]\w*', Name.Variable),
192              (r'(==|!=|~=|<=|>=|\.\.\.|\.\.|[=+\-*/%^<>#!.\\:])', Operator),
193              (r'[;,]', Punctuation),
194              (r'[\[\]{}()]', Keyword.Type),
195              (r'[a-zA-Z_]\w*:', Name.Variable),
196              (words((
197                  'class', 'extends', 'if', 'then', 'super', 'do', 'with',
198                  'import', 'export', 'while', 'elseif', 'return', 'for', 'in',
199                  'from', 'when', 'using', 'else', 'and', 'or', 'not', 'switch',
200                  'break'), suffix=r'\b'),
201               Keyword),
202              (r'(true|false|nil)\b', Keyword.Constant),
203              (r'(and|or|not)\b', Operator.Word),
204              (r'(self)\b', Name.Builtin.Pseudo),
205              (r'@@?([a-zA-Z_]\w*)?', Name.Variable.Class),
206              (r'[A-Z]\w*', Name.Class),  # proper name
207              (r'[A-Za-z_]\w*(\.[A-Za-z_]\w*)?', Name),
208              ("'", String.Single, combined('stringescape', 'sqs')),
209              ('"', String.Double, combined('stringescape', 'dqs'))
210          ],
211          'stringescape': [
212              (r'''\\([abfnrtv\\"']|\d{1,3})''', String.Escape)
213          ],
214          'sqs': [
215              ("'", String.Single, '#pop'),
216              (".", String)
217          ],
218          'dqs': [
219              ('"', String.Double, '#pop'),
220              (".", String)
221          ]
222      }
223  
224      def get_tokens_unprocessed(self, text):
225          # set . as Operator instead of Punctuation
226          for index, token, value in LuaLexer.get_tokens_unprocessed(self, text):
227              if token == Punctuation and value == ".":
228                  token = Operator
229              yield index, token, value

1046             (r'\s+', Whitespace),
1047             # Everything else just belongs to a name
1048             (_NON_DELIMITER_OR_COMMENT_PATTERN + r'+', Name),
1049          ],
1050         'after_declaration': [
1051             (_NON_DELIMITER_OR_COMMENT_PATTERN + r'+', Name.Function),
1052             default('#pop'),
1053         ],
1054         'after_macro_argument': [
1055             (r'\*.*\n', Comment.Single, '#pop'),
1056             (r'\s+', Whitespace, '#pop'),
1057             (_OPERATORS_PATTERN, Operator, '#pop'),
1058             (r"'(''|[^'])*'", String, '#pop'),
1059             # Everything else just belongs to a name
1060             (_NON_DELIMITER_OR_COMMENT_PATTERN + r'+', Name),
1061         ],
1062     }
1063     _COMMENT_LINE_REGEX = re.compile(r'^\s*\*')
1064     _MACRO_HEADER_REGEX = re.compile(r'^\s*MACRO')
1065 
1066     def analyse_text(text):
1067         """ 
1068         Perform a structural analysis for basic Easytrieve constructs. 
1069         """
1070         result = 0.0
1071         lines = text.split('\n')
1072         hasEndProc = False
1073         hasHeaderComment = False
1074         hasFile = False
1075         hasJob = False
1076         hasProc = False
1077         hasParm = False
1078         hasReport = False
1079 
1080         def isCommentLine(line):
1081             return EasytrieveLexer._COMMENT_LINE_REGEX.match(lines[0]) is not None
1082 
1083         def isEmptyLine(line):
1084             return not bool(line.strip())
1085 
1086         # Remove possible empty lines and header comments.
1087         while lines and (isEmptyLine(lines[0]) or isCommentLine(lines[0])):
1088             if not isEmptyLine(lines[0]):
1089                 hasHeaderComment = True
1090             del lines[0]
1091 
1092         if EasytrieveLexer._MACRO_HEADER_REGEX.match(lines[0]):
1093             # Looks like an Easytrieve macro.
1094             result = 0.4
1095             if hasHeaderComment:
1096                 result += 0.4
1097         else:
1098             # Scan the source for lines starting with indicators.
1099             for line in lines:
1100                 words = line.split()
1101                 if (len(words) >= 2):
1102                     firstWord = words[0]
1103                     if not hasReport:
1104                         if not hasJob:
1105                             if not hasFile:
1106                                 if not hasParm:
1107                                     if firstWord == 'PARM':
1108                                         hasParm = True
1109                                 if firstWord == 'FILE':
1110                                     hasFile = True
1111                             if firstWord == 'JOB':
1112                                 hasJob = True
1113                         elif firstWord == 'PROC':
1114                             hasProc = True
1115                         elif firstWord == 'END-PROC':
1116                             hasEndProc = True
1117                         elif firstWord == 'REPORT':
1118                             hasReport = True
1119 
1120             # Weight the findings.
1121             if hasJob and (hasProc == hasEndProc):
1122                 if hasHeaderComment:
1123                     result += 0.1
1124                 if hasParm:
1125                     if hasProc:
1126                         # Found PARM, JOB and PROC/END-PROC:
1127                         # pretty sure this is Easytrieve.
1128                         result += 0.8
1129                     else:
1130                         # Found PARAM and  JOB: probably this is Easytrieve
1131                         result += 0.5
1132                 else:
1133                     # Found JOB and possibly other keywords: might be Easytrieve
1134                     result += 0.11
1135                     if hasParm:
1136                         # Note: PARAM is not a proper English word, so this is
1137                         # regarded a much better indicator for Easytrieve than
1138                         # the other words.
1139                         result += 0.2
1140                     if hasFile:
1141                         result += 0.01
1142                     if hasReport:
1143                         result += 0.01
1144         assert 0.0 <= result <= 1.0
1145         return result
