48   def makeadopackage(testfolder):
49       adoName = os.path.normpath(os.getcwd() + "/../adodbapi.py")
50       adoPath = os.path.dirname(adoName)
51       if os.path.exists(adoName):
52           newpackage = os.path.join(testfolder, "adodbapi")
53           try:
54               os.mkdir(newpackage)
55           except OSErrors:
56               print(
57                   "*Note: temporary adodbapi package already exists: may be two versions running?"
58               )
59           for f in os.listdir(adoPath):
60               if f.endswith(".py"):
61                   shutil.copy(os.path.join(adoPath, f), newpackage)
62           if sys.version_info >= (3, 0):  # only when running Py3.n
63               save = sys.stdout
64               sys.stdout = None
65               from lib2to3.main import main  # use 2to3 to make test package
66   
67               main("lib2to3.fixes", args=["-n", "-w", newpackage])
68               sys.stdout = save
69           return testfolder
70       else:
71           raise EnvironmentError("Connot find source of adodbapi to test.")

74   def makemdb(testfolder, mdb_name):
75       # following setup code borrowed from pywin32 odbc test suite
76       # kindly contributed by Frank Millman.
77       import os
78   
79       _accessdatasource = os.path.join(testfolder, mdb_name)
80       if os.path.isfile(_accessdatasource):
81           print("using JET database=", _accessdatasource)
82       else:
83           try:
84               from win32com.client.gencache import EnsureDispatch
85               from win32com.client import constants
86   
87               win32 = True
88           except ImportError:  # perhaps we are running IronPython
89               win32 = False  # iron Python
90               try:
91                   from System import Activator, Type
92               except:
93                   pass
94   
95           # Create a brand-new database - what is the story with these?
96           dbe = None
97           for suffix in (".36", ".35", ".30"):
98               try:
99                   if win32:
100                      dbe = EnsureDispatch("DAO.DBEngine" + suffix)
101                  else:
102                      type = Type.GetTypeFromProgID("DAO.DBEngine" + suffix)
103                      dbe = Activator.CreateInstance(type)
104                  break
105              except:
106                  pass
107          if dbe:
108              print("    ...Creating ACCESS db at " + _accessdatasource)
109              if win32:
110                  workspace = dbe.Workspaces(0)
111                  newdb = workspace.CreateDatabase(
112                      _accessdatasource, constants.dbLangGeneral, constants.dbVersion40
113                  )
114              else:
115                  newdb = dbe.CreateDatabase(
116                      _accessdatasource, ";LANGID=0x0409;CP=1252;COUNTRY=0"
117                  )
118              newdb.Close()
119          else:
120              print("    ...copying test ACCESS db to " + _accessdatasource)
121              mdbName = os.path.abspath(
122                  os.path.join(os.path.dirname(__file__), "..", "examples", "test.mdb")
123              )
124              import shutil
125  
126              shutil.copy(mdbName, _accessdatasource)
127  
128      return _accessdatasource
