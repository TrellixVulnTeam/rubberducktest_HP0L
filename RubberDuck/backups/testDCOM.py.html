2    usage = """\ 
3    testDCOM.py - Simple DCOM test 
4    Usage: testDCOM.py serverName 
5     
6    Attempts to start the Python.Interpreter object on the named machine, 
7    and checks that the object is indeed running remotely. 
8     
9    Requires the named server be configured to run DCOM (using dcomcnfg.exe), 
10   and the Python.Interpreter object installed and registered on that machine. 
11    
12   The Python.Interpreter object must be installed on the local machine, 
13   but no special DCOM configuration should be necessary. 
14   """
15   # NOTE: If you configured the object locally using dcomcnfg, you could
16   # simple use Dispatch rather than DispatchEx.
17   import pythoncom, win32com.client, win32api, string, sys
18   
19   
20   def test(serverName):
21       if string.lower(serverName) == string.lower(win32api.GetComputerName()):
22           print("You must specify a remote server name, not the local machine!")
23           return
24   
25       # Hack to overcome a DCOM limitation.  As the Python.Interpreter object
26       # is probably installed locally as an InProc object, DCOM seems to ignore
27       # all settings, and use the local object.
28       clsctx = pythoncom.CLSCTX_SERVER & ~pythoncom.CLSCTX_INPROC_SERVER
29       ob = win32com.client.DispatchEx("Python.Interpreter", serverName, clsctx=clsctx)
30       ob.Exec("import win32api")
31       actualName = ob.Eval("win32api.GetComputerName()")
32       if string.lower(serverName) != string.lower(actualName):
33           print(
34               "Error: The object created on server '%s' reported its name as '%s'"
35               % (serverName, actualName)
36           )
37       else:
38           print("Object created and tested OK on server '%s'" % serverName)
39   
40   
41   if __name__ == "__main__":
42       if len(sys.argv) == 2:
43           test(sys.argv[1])
44       else:
45           print(usage)
