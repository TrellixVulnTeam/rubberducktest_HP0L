1    ''' 
2    Clipboard xsel: an implementation of the Clipboard using xsel command line 
3                    tool. 
4    '''
5    
6    __all__ = ('ClipboardXsel', )
7    
8    from kivy.utils import platform
9    from kivy.core.clipboard._clipboard_ext import ClipboardExternalBase
10   
11   if platform != 'linux':
12       raise SystemError('unsupported platform for xsel clipboard')
13   
14   try:
15       import subprocess
16       p = subprocess.Popen(['xsel'], stdout=subprocess.PIPE)
17       p.communicate()
18   except:
19       raise
20   
21   
22   class ClipboardXsel(ClipboardExternalBase):
23       @staticmethod
24       def _clip(inout, selection):
25           pipe = {'std' + inout: subprocess.PIPE}
26           sel = 'b' if selection == 'clipboard' else selection[0]
27           io = inout[0]
28           return subprocess.Popen(
29               ['xsel', '-' + sel + io], **pipe)
