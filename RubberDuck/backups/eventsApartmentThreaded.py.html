eventsApartmentThreaded.py 
1    # A sample originally provided by Richard Bell, and modified by Mark Hammond.
2    
3    # This sample demonstrates how to use COM events in an aparment-threaded
4    # world.  In this world, COM itself ensures that all calls to and events
5    # from an object happen on the same thread that created the object, even
6    # if they originated from different threads.  For this cross-thread
7    # marshalling to work, this main thread *must* run a "message-loop" (ie,
8    # a loop fetching and dispatching Windows messages).  Without such message
9    # processing, dead-locks can occur.
10   
11   # See also eventsFreeThreaded.py for how to do this in a free-threaded
12   # world where these marshalling considerations do not exist.
13   
14   # NOTE: This example uses Internet Explorer, but it should not be considerd
15   # a "best-practices" for writing against IE events, but for working with
16   # events in general. For example:
17   # * The first OnDocumentComplete event is not a reliable indicator that the
18   #   URL has completed loading
19   # * As we are demonstrating the most efficient way of handling events, when
20   #   running this sample you will see an IE Windows briefly appear, but
21   #   vanish without ever being repainted.
22   
23   import sys
24   import os
25   import win32com.client
26   import win32api
27   import win32event
28   
29   # sys.coinit_flags not set, so pythoncom initializes apartment-threaded.
30   import pythoncom
31   import time
32   
33   
34   class ExplorerEvents:
35       def __init__(self):
36           self.event = win32event.CreateEvent(None, 0, 0, None)
37   
38       def OnDocumentComplete(self, pDisp=pythoncom.Empty, URL=pythoncom.Empty):
39           thread = win32api.GetCurrentThreadId()
40           print("OnDocumentComplete event processed on thread %d" % thread)
41           # Set the event our main thread is waiting on.
42           win32event.SetEvent(self.event)
43   
44       def OnQuit(self):
45           thread = win32api.GetCurrentThreadId()
46           print("OnQuit event processed on thread %d" % thread)
47           win32event.SetEvent(self.event)
48   
49   
50   def WaitWhileProcessingMessages(event, timeout=2):
51       start = time.clock()
52       while True:
53           # Wake 4 times a second - we can't just specify the
54           # full timeout here, as then it would reset for every
55           # message we process.
56           rc = win32event.MsgWaitForMultipleObjects(
57               (event,), 0, 250, win32event.QS_ALLEVENTS
58           )
59           if rc == win32event.WAIT_OBJECT_0:
60               # event signalled - stop now!
61               return True
62           if (time.clock() - start) > timeout:
63               # Timeout expired.
64               return False
65           # must be a message.
66           pythoncom.PumpWaitingMessages()
67   
68   
69   def TestExplorerEvents():
70       iexplore = win32com.client.DispatchWithEvents(
71           "InternetExplorer.Application", ExplorerEvents
72       )
73   
74       thread = win32api.GetCurrentThreadId()
75       print("TestExplorerEvents created IE object on thread %d" % thread)
76   
77       iexplore.Visible = 1
78       try:
79           iexplore.Navigate(win32api.GetFullPathName("..\\readme.html"))
80       except pythoncom.com_error as details:
81           print("Warning - could not open the test HTML file", details)
82   
83       # Wait for the event to be signalled while pumping messages.
84       if not WaitWhileProcessingMessages(iexplore.event):
85           print("Document load event FAILED to fire!!!")
86   
87       iexplore.Quit()
88       #
89       # Give IE a chance to shutdown, else it can get upset on fast machines.
90       # Note, Quit generates events.  Although this test does NOT catch them
91       # it is NECESSARY to pump messages here instead of a sleep so that the Quit
92       # happens properly!
93       if not WaitWhileProcessingMessages(iexplore.event):
94           print("OnQuit event FAILED to fire!!!")
95   
96       iexplore = None
97   
98   
99   if __name__ == "__main__":
100      TestExplorerEvents()
