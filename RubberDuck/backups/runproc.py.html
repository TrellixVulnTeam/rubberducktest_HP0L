1    """runproc.py 
2     
3    start a process with three inherited pipes. 
4    Try to write to and read from those. 
5    """
6    
7    import win32api
8    import win32pipe
9    import win32file
10   import win32process
11   import win32security
12   import win32con
13   import msvcrt
14   import os
15   
16   
17   class Process:
18       def run(self, cmdline):
19           # security attributes for pipes
20           sAttrs = win32security.SECURITY_ATTRIBUTES()
21           sAttrs.bInheritHandle = 1
22   
23           # create pipes
24           hStdin_r, self.hStdin_w = win32pipe.CreatePipe(sAttrs, 0)
25           self.hStdout_r, hStdout_w = win32pipe.CreatePipe(sAttrs, 0)
26           self.hStderr_r, hStderr_w = win32pipe.CreatePipe(sAttrs, 0)
27   
28           # set the info structure for the new process.
29           StartupInfo = win32process.STARTUPINFO()
30           StartupInfo.hStdInput = hStdin_r
31           StartupInfo.hStdOutput = hStdout_w
32           StartupInfo.hStdError = hStderr_w
33           StartupInfo.dwFlags = win32process.STARTF_USESTDHANDLES
34           # Mark doesn't support wShowWindow yet.
35           # StartupInfo.dwFlags = StartupInfo.dwFlags | win32process.STARTF_USESHOWWINDOW
36           # StartupInfo.wShowWindow = win32con.SW_HIDE
37   
38           # Create new output read handles and the input write handle. Set
39           # the inheritance properties to FALSE. Otherwise, the child inherits
40           # the these handles; resulting in non-closeable handles to the pipes
41           # being created.
42           pid = win32api.GetCurrentProcess()
43   
44           tmp = win32api.DuplicateHandle(
45               pid,
46               self.hStdin_w,
47               pid,
48               0,
49               0,  # non-inheritable!!
50               win32con.DUPLICATE_SAME_ACCESS,
51           )
52           # Close the inhertible version of the handle
53           win32file.CloseHandle(self.hStdin_w)
54           self.hStdin_w = tmp
55           tmp = win32api.DuplicateHandle(
56               pid,
57               self.hStdout_r,
58               pid,
59               0,
60               0,  # non-inheritable!
61               win32con.DUPLICATE_SAME_ACCESS,
62           )
63           # Close the inhertible version of the handle
64           win32file.CloseHandle(self.hStdout_r)
65           self.hStdout_r = tmp
66   
67           # start the process.
68           hProcess, hThread, dwPid, dwTid = win32process.CreateProcess(
69               None,  # program
70               cmdline,  # command line
71               None,  # process security attributes
72               None,  # thread attributes
73               1,  # inherit handles, or USESTDHANDLES won't work.
74               # creation flags. Don't access the console.
75               0,  # Don't need anything here.
76               # If you're in a GUI app, you should use
77               # CREATE_NEW_CONSOLE here, or any subprocesses
78               # might fall victim to the problem described in:
79               # KB article: Q156755, cmd.exe requires
80               # an NT console in order to perform redirection..
81               None,  # no new environment
82               None,  # current directory (stay where we are)
83               StartupInfo,
84           )
85           # normally, we would save the pid etc. here...
86   
87           # Child is launched. Close the parents copy of those pipe handles
88           # that only the child should have open.
89           # You need to make sure that no handles to the write end of the
90           # output pipe are maintained in this process or else the pipe will
91           # not close when the child process exits and the ReadFile will hang.
92           win32file.CloseHandle(hStderr_w)
93           win32file.CloseHandle(hStdout_w)
94           win32file.CloseHandle(hStdin_r)
95   
96           self.stdin = os.fdopen(msvcrt.open_osfhandle(self.hStdin_w, 0), "wb")
97           self.stdin.write("hmmmmm\r\n")
98           self.stdin.flush()
99           self.stdin.close()
100  
101          self.stdout = os.fdopen(msvcrt.open_osfhandle(self.hStdout_r, 0), "rb")
102          print("Read on stdout: ", repr(self.stdout.read()))
103  
104          self.stderr = os.fdopen(msvcrt.open_osfhandle(self.hStderr_r, 0), "rb")
105          print("Read on stderr: ", repr(self.stderr.read()))
106  
107  
108  if __name__ == "__main__":
109      p = Process()
110      exe = win32api.GetModuleFileName(0)
111      p.run(exe + " cat.py")
112
