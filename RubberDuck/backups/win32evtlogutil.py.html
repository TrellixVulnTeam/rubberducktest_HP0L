11   def AddSourceToRegistry(
12       appName, msgDLL=None, eventLogType="Application", eventLogFlags=None
13   ):
14       """Add a source of messages to the event log. 
15    
16       Allows Python program to register a custom source of messages in the 
17       registry.  You must also provide the DLL name that has the message table, so the 
18       full message text appears in the event log. 
19    
20       Note that the win32evtlog.pyd file has a number of string entries with just "%1" 
21       built in, so many Python programs can simply use this DLL.  Disadvantages are that 
22       you do not get language translation, and the full text is stored in the event log, 
23       blowing the size of the log up. 
24       """
25   
26       # When an application uses the RegisterEventSource or OpenEventLog
27       # function to get a handle of an event log, the event loggging service
28       # searches for the specified source name in the registry. You can add a
29       # new source name to the registry by opening a new registry subkey
30       # under the Application key and adding registry values to the new
31       # subkey.
32   
33       if msgDLL is None:
34           msgDLL = win32evtlog.__file__
35   
36       # Create a new key for our application
37       hkey = win32api.RegCreateKey(
38           win32con.HKEY_LOCAL_MACHINE,
39           "SYSTEM\\CurrentControlSet\\Services\\EventLog\\%s\\%s"
40           % (eventLogType, appName),
41       )
42   
43       # Add the Event-ID message-file name to the subkey.
44       win32api.RegSetValueEx(
45           hkey,
46           "EventMessageFile",  # value name \
47           0,  # reserved \
48           win32con.REG_EXPAND_SZ,  # value type \
49           msgDLL,
50       )
51   
52       # Set the supported types flags and add it to the subkey.
53       if eventLogFlags is None:
54           eventLogFlags = (
55               win32evtlog.EVENTLOG_ERROR_TYPE
56               | win32evtlog.EVENTLOG_WARNING_TYPE
57               | win32evtlog.EVENTLOG_INFORMATION_TYPE
58           )
59       win32api.RegSetValueEx(
60           hkey,  # subkey handle \
61           "TypesSupported",  # value name \
62           0,  # reserved \
63           win32con.REG_DWORD,  # value type \
64           eventLogFlags,
65       )
66       win32api.RegCloseKey(hkey)

69   def RemoveSourceFromRegistry(appName, eventLogType="Application"):
70       """Removes a source of messages from the event log."""
71   
72       # Delete our key
73       try:
74           win32api.RegDeleteKey(
75               win32con.HKEY_LOCAL_MACHINE,
76               "SYSTEM\\CurrentControlSet\\Services\\EventLog\\%s\\%s"
77               % (eventLogType, appName),
78           )
79       except win32api.error as exc:
80           if exc.winerror != winerror.ERROR_FILE_NOT_FOUND:
81               raise

84   def ReportEvent(
85       appName,
86       eventID,
87       eventCategory=0,
88       eventType=win32evtlog.EVENTLOG_ERROR_TYPE,
89       strings=None,
90       data=None,
91       sid=None,
92   ):
93       """Report an event for a previously added event source."""
94       # Get a handle to the Application event log
95       hAppLog = win32evtlog.RegisterEventSource(None, appName)
96   
97       # Now report the event, which will add this event to the event log */
98       win32evtlog.ReportEvent(
99           hAppLog,  # event-log handle \
100          eventType,
101          eventCategory,
102          eventID,
103          sid,
104          strings,
105          data,
106      )
107  
108      win32evtlog.DeregisterEventSource(hAppLog)

111  def FormatMessage(eventLogRecord, logType="Application"):
112      """Given a tuple from ReadEventLog, and optionally where the event 
113      record came from, load the message, and process message inserts. 
114   
115      Note that this function may raise win32api.error.  See also the 
116      function SafeFormatMessage which will return None if the message can 
117      not be processed. 
118      """
119  
120      # From the event log source name, we know the name of the registry
121      # key to look under for the name of the message DLL that contains
122      # the messages we need to extract with FormatMessage. So first get
123      # the event log source name...
124      keyName = "SYSTEM\\CurrentControlSet\\Services\\EventLog\\%s\\%s" % (
125          logType,
126          eventLogRecord.SourceName,
127      )
128  
129      # Now open this key and get the EventMessageFile value, which is
130      # the name of the message DLL.
131      handle = win32api.RegOpenKey(win32con.HKEY_LOCAL_MACHINE, keyName)
132      try:
133          dllNames = win32api.RegQueryValueEx(handle, "EventMessageFile")[0].split(";")
134          # Win2k etc appear to allow multiple DLL names
135          data = None
136          for dllName in dllNames:
137              try:
138                  # Expand environment variable strings in the message DLL path name,
139                  # in case any are there.
140                  dllName = win32api.ExpandEnvironmentStrings(dllName)
141  
142                  dllHandle = win32api.LoadLibraryEx(
143                      dllName, 0, win32con.LOAD_LIBRARY_AS_DATAFILE
144                  )
145                  try:
146                      data = win32api.FormatMessageW(
147                          win32con.FORMAT_MESSAGE_FROM_HMODULE,
148                          dllHandle,
149                          eventLogRecord.EventID,
150                          langid,
151                          eventLogRecord.StringInserts,
152                      )
153                  finally:
154                      win32api.FreeLibrary(dllHandle)
155              except win32api.error:
156                  pass  # Not in this DLL - try the next
157              if data is not None:
158                  break
159      finally:
160          win32api.RegCloseKey(handle)
161      return data or ""  # Don't want "None" ever being returned.

164  def SafeFormatMessage(eventLogRecord, logType=None):
165      """As for FormatMessage, except returns an error message if 
166      the message can not be processed. 
167      """
168      if logType is None:
169          logType = "Application"
170      try:
171          return FormatMessage(eventLogRecord, logType)
172      except win32api.error:
173          if eventLogRecord.StringInserts is None:
174              desc = ""
175          else:
176              desc = ", ".join(eventLogRecord.StringInserts)
177          return "<The description for Event ID ( %d ) in Source ( %r ) could not be found. It contains the following insertion string(s):%r.>" % (
178              winerror.HRESULT_CODE(eventLogRecord.EventID),
179              eventLogRecord.SourceName,
180              desc,
181          )

184  def FeedEventLogRecords(
185      feeder, machineName=None, logName="Application", readFlags=None
186  ):
187      if readFlags is None:
188          readFlags = (
189              win32evtlog.EVENTLOG_BACKWARDS_READ | win32evtlog.EVENTLOG_SEQUENTIAL_READ
190          )
191  
192      h = win32evtlog.OpenEventLog(machineName, logName)
193      try:
194          while 1:
195              objects = win32evtlog.ReadEventLog(h, readFlags, 0)
196              if not objects:
197                  break
198              map(lambda item, feeder=feeder: feeder(*(item,)), objects)
199      finally:
200          win32evtlog.CloseEventLog(h)
