redisstore.py 
1    ''' 
2    Redis Store 
3    =========== 
4     
5    Store implementation using Redis. You must have redis-py installed. 
6     
7    Usage example:: 
8     
9        from kivy.storage.redisstore import RedisStore 
10    
11       params = dict(host='localhost', port=6379, db=14) 
12       store = RedisStore(params) 
13    
14   All the key-value pairs will be stored with a prefix 'store' by default. 
15   You can instantiate the storage with another prefix like this:: 
16    
17    
18       from kivy.storage.redisstore import RedisStore 
19    
20       params = dict(host='localhost', port=6379, db=14) 
21       store = RedisStore(params, prefix='mystore2') 
22    
23   The params dictionary will be passed to the redis.StrictRedis class. 
24    
25   See `redis-py <https://github.com/andymccurdy/redis-py>`_. 
26   '''
27   
28   __all__ = ('RedisStore', )
29   
30   import os
31   from json import loads, dumps
32   from kivy.compat import iteritems
33   from kivy.properties import StringProperty
34   from kivy.storage import AbstractStore
35   
36   # don't import redis during the documentation generation
37   if 'KIVY_DOC' not in os.environ:
38       import redis
39   
40   
41   class RedisStore(AbstractStore):
42       '''Store implementation using a Redis database. 
43       See the :mod:`kivy.storage` module documentation for more informations. 
44       '''
45   
46       prefix = StringProperty('store')
47   
48       def __init__(self, redis_params, **kwargs):
49           self.redis_params = redis_params
50           self.r = None
51           super(RedisStore, self).__init__(**kwargs)
52   
53       def store_load(self):
54           self.r = redis.StrictRedis(**self.redis_params)
55   
56       def store_sync(self):
57           pass
58   
59       def store_exists(self, key):
60           key = self.prefix + '.d.' + key
61           value = self.r.exists(key)
62           return value
63   
64       def store_get(self, key):
65           key = self.prefix + '.d.' + key
66           if not self.r.exists(key):
67               raise KeyError(key)
68           result = self.r.hgetall(key)
69           for k in result.keys():
70               result[k] = loads(result[k])
71           return result
72   
73       def store_put(self, key, values):
74           key = self.prefix + '.d.' + key
75           pipe = self.r.pipeline()
76           pipe.delete(key)
77           for k, v in iteritems(values):
78               pipe.hset(key, k, dumps(v))
79           pipe.execute()
80           return True
81   
82       def store_delete(self, key):
83           key = self.prefix + '.d.' + key
84           if not self.r.exists(key):
85               raise KeyError(key)
86           return self.r.delete(key)
87   
88       def store_keys(self):
89           z = len(self.prefix + '.d.')
90           return [x[z:] for x in self.r.keys(self.prefix + '.d.*')]
91   
92       def store_find(self, filters):
93           fkeys = filters.keys()
94           fvalues = filters.values()
95           for key in self.store_keys():
96               skey = self.prefix + '.d.' + key
97               svalues = self.r.hmget(skey, fkeys)
98               if None in svalues:
99                   continue
100              svalues = [loads(x) for x in svalues]
101              if fvalues != svalues:
102                  continue
103              yield key, self.r.hgetall(skey)
