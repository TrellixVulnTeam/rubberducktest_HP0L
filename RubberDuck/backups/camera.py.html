1    ''' 
2    Camera 
3    ====== 
4     
5    The :class:`Camera` widget is used to capture and display video from a camera. 
6    Once the widget is created, the texture inside the widget will be automatically 
7    updated. Our :class:`~kivy.core.camera.CameraBase` implementation is used under 
8    the hood:: 
9     
10       cam = Camera() 
11    
12   By default, the first camera found on your system is used. To use a different 
13   camera, set the index property:: 
14    
15       cam = Camera(index=1) 
16    
17   You can also select the camera resolution:: 
18    
19       cam = Camera(resolution=(320, 240)) 
20    
21   .. warning:: 
22    
23       The camera texture is not updated as soon as you have created the object. 
24       The camera initialization is asynchronous, so there may be a delay before 
25       the requested texture is created. 
26   '''
27   
28   __all__ = ('Camera', )
29   
30   from kivy.uix.image import Image
31   from kivy.core.camera import Camera as CoreCamera
32   from kivy.properties import NumericProperty, ListProperty, \
33       BooleanProperty
34   
35   
36   class Camera(Image):
37       '''Camera class. See module documentation for more information. 
38       '''
39   
40       play = BooleanProperty(True)
41       '''Boolean indicating whether the camera is playing or not. 
42       You can start/stop the camera by setting this property:: 
43    
44           # start the camera playing at creation (default) 
45           cam = Camera(play=True) 
46    
47           # create the camera, and start later 
48           cam = Camera(play=False) 
49           # and later 
50           cam.play = True 
51    
52       :attr:`play` is a :class:`~kivy.properties.BooleanProperty` and defaults to 
53       True. 
54       '''
55   
56       index = NumericProperty(-1)
57       '''Index of the used camera, starting from 0. 
58    
59       :attr:`index` is a :class:`~kivy.properties.NumericProperty` and defaults 
60       to -1 to allow auto selection. 
61       '''
62   
63       resolution = ListProperty([-1, -1])
64       '''Preferred resolution to use when invoking the camera. If you are using 
65       [-1, -1], the resolution will be the default one:: 
66    
67           # create a camera object with the best image available 
68           cam = Camera() 
69    
70           # create a camera object with an image of 320x240 if possible 
71           cam = Camera(resolution=(320, 240)) 
72    
73       .. warning:: 
74    
75           Depending on the implementation, the camera may not respect this 
76           property. 
77    
78       :attr:`resolution` is a :class:`~kivy.properties.ListProperty` and defaults 
79       to [-1, -1]. 
80       '''
81   
82       def __init__(self, **kwargs):
83           self._camera = None
84           super(Camera, self).__init__(**kwargs)
85           if self.index == -1:
86               self.index = 0
87           on_index = self._on_index
88           fbind = self.fbind
89           fbind('index', on_index)
90           fbind('resolution', on_index)
91           on_index()
92   
93       def on_tex(self, *l):
94           self.canvas.ask_update()
95   
96       def _on_index(self, *largs):
97           self._camera = None
98           if self.index < 0:
99               return
100          if self.resolution[0] < 0 or self.resolution[1] < 0:
101              self._camera = CoreCamera(index=self.index, stopped=True)
102          else:
103              self._camera = CoreCamera(index=self.index,
104                                    resolution=self.resolution, stopped=True)
105          self._camera.bind(on_load=self._camera_loaded)
106          if self.play:
107              self._camera.start()
108              self._camera.bind(on_texture=self.on_tex)
109  
110      def _camera_loaded(self, *largs):
111          if self._camera.texture is not None:
112              self.texture = self._camera.texture
113              self.texture_size = list(self.texture.size)
114  
115      def on_play(self, instance, value):
116          if not self._camera:
117              return
118          if value:
119              self._camera.start()
120          else:
121              self._camera.stop()
