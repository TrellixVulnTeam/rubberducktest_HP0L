31   class CameraAndroid(CameraBase):
32       """ 
33       Implementation of CameraBase using Android API 
34       """
35   
36       _update_ev = None
37   
38       def __init__(self, **kwargs):
39           self._android_camera = None
40           self._preview_cb = PreviewCallback(self._on_preview_frame)
41           self._buflock = threading.Lock()
42           super(CameraAndroid, self).__init__(**kwargs)
43   
44       def __del__(self):
45           self._release_camera()
46   
47       def init_camera(self):
48           self._release_camera()
49           self._android_camera = Camera.open(self._index)
50           params = self._android_camera.getParameters()
51           width, height = self._resolution
52           params.setPreviewSize(width, height)
53           params.setFocusMode('continuous-picture')
54           self._android_camera.setParameters(params)
55           # self._android_camera.setDisplayOrientation()
56           self.fps = 30.
57   
58           pf = params.getPreviewFormat()
59           assert(pf == ImageFormat.NV21)  # default format is NV21
60           self._bufsize = int(ImageFormat.getBitsPerPixel(pf) / 8. *
61                               width * height)
62   
63           self._camera_texture = Texture(width=width, height=height,
64                                          target=GL_TEXTURE_EXTERNAL_OES,
65                                          colorfmt='rgba')
66           self._surface_texture = SurfaceTexture(int(self._camera_texture.id))
67           self._android_camera.setPreviewTexture(self._surface_texture)
68   
69           self._fbo = Fbo(size=self._resolution)
70           self._fbo['resolution'] = (float(width), float(height))
71           self._fbo.shader.fs = ''' 
72               #extension GL_OES_EGL_image_external : require 
73               #ifdef GL_ES 
74                   precision highp float; 
75               #endif 
76    
77               /* Outputs from the vertex shader */ 
78               varying vec4 frag_color; 
79               varying vec2 tex_coord0; 
80    
81               /* uniform texture samplers */ 
82               uniform sampler2D texture0; 
83               uniform samplerExternalOES texture1; 
84               uniform vec2 resolution; 
85    
86               void main() 
87               { 
88                   vec2 coord = vec2(tex_coord0.y * ( 
89                       resolution.y / resolution.x), 1. -tex_coord0.x); 
90                   gl_FragColor = texture2D(texture1, tex_coord0); 
91               } 
92           '''
93           with self._fbo:
94               self._texture_cb = Callback(lambda instr:
95                                           self._camera_texture.bind)
96               Rectangle(size=self._resolution)
97   
98       def _release_camera(self):
99           if self._android_camera is None:
100              return
101  
102          self.stop()
103          self._android_camera.release()
104          self._android_camera = None
105  
106          # clear texture and it'll be reset in `_update` pointing to new FBO
107          self._texture = None
108          del self._fbo, self._surface_texture, self._camera_texture
109  
110      def _on_preview_frame(self, data, camera):
111          with self._buflock:
112              if self._buffer is not None:
113                  # add buffer back for reuse
114                  self._android_camera.addCallbackBuffer(self._buffer)
115              self._buffer = data
116          # check if frame grabbing works
117          # print self._buffer, len(self.frame_data)
118  
119      def _refresh_fbo(self):
120          self._texture_cb.ask_update()
121          self._fbo.draw()
122  
123      def start(self):
124          super(CameraAndroid, self).start()
125  
126          with self._buflock:
127              self._buffer = None
128          for k in range(2):  # double buffer
129              buf = b'\x00' * self._bufsize
130              self._android_camera.addCallbackBuffer(buf)
131          self._android_camera.setPreviewCallbackWithBuffer(self._preview_cb)
132  
133          self._android_camera.startPreview()
134          if self._update_ev is not None:
135              self._update_ev.cancel()
136          self._update_ev = Clock.schedule_interval(self._update, 1 / self.fps)
137  
138      def stop(self):
139          super(CameraAndroid, self).stop()
140          if self._update_ev is not None:
141              self._update_ev.cancel()
142              self._update_ev = None
143          self._android_camera.stopPreview()
144  
145          self._android_camera.setPreviewCallbackWithBuffer(None)
146          # buffer queue cleared as well, to be recreated on next start
147          with self._buflock:
148              self._buffer = None
149  
150      def _update(self, dt):
151          self._surface_texture.updateTexImage()
152          self._refresh_fbo()
153          if self._texture is None:
154              self._texture = self._fbo.texture
155              self.dispatch('on_load')
156          self._copy_to_gpu()
157  
158      def _copy_to_gpu(self):
159          """ 
160          A dummy placeholder (the image is already in GPU) to be consistent 
161          with other providers. 
162          """
163          self.dispatch('on_texture')
164  
165      def grab_frame(self):
166          """ 
167          Grab current frame (thread-safe, minimal overhead) 
168          """
169          with self._buflock:
170              if self._buffer is None:
171                  return None
172              buf = self._buffer.tostring()
173              return buf
174  
175      def decode_frame(self, buf):
176          """ 
177          Decode image data from grabbed frame. 
178   
179          This method depends on OpenCV and NumPy - however it is only used for 
180          fetching the current frame as a NumPy array, and not required when 
181          this :class:`CameraAndroid` provider is simply used by a 
182          :class:`~kivy.uix.camera.Camera` widget. 
183          """
184          import numpy as np
185          from cv2 import cvtColor
186  
187          w, h = self._resolution
188          arr = np.fromstring(buf, 'uint8').reshape((h + h / 2, w))
189          arr = cvtColor(arr, 93)  # NV21 -> BGR
190          return arr
191  
192      def read_frame(self):
193          """ 
194          Grab and decode frame in one call 
195          """
196          return self.decode_frame(self.grab_frame())
