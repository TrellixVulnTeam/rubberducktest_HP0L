14   def DumpPythonRegistry():
15       try:
16           h = wincerapi.CeRegOpenKeyEx(
17               win32con.HKEY_LOCAL_MACHINE,
18               "Software\\Python\\PythonCore\\%s\\PythonPath" % sys.winver,
19           )
20       except win32api.error:
21           print("The remote device does not appear to have Python installed")
22           return 0
23       path, typ = wincerapi.CeRegQueryValueEx(h, None)
24       print("The remote PythonPath is '%s'" % (str(path),))
25       h.Close()
26       return 1

29   def DumpRegistry(root, level=0):
30       # A recursive dump of the remote registry to test most functions.
31       h = wincerapi.CeRegOpenKeyEx(win32con.HKEY_LOCAL_MACHINE, None)
32       level_prefix = " " * level
33       index = 0
34       # Enumerate values.
35       while 1:
36           try:
37               name, data, typ = wincerapi.CeRegEnumValue(root, index)
38           except win32api.error:
39               break
40           print("%s%s=%s" % (level_prefix, name, repr(str(data))))
41           index = index + 1
42       # Now enumerate all keys.
43       index = 0
44       while 1:
45           try:
46               name, klass = wincerapi.CeRegEnumKeyEx(root, index)
47           except win32api.error:
48               break
49           print("%s%s\\" % (level_prefix, name))
50           subkey = wincerapi.CeRegOpenKeyEx(root, name)
51           DumpRegistry(subkey, level + 1)
52           index = index + 1

55   def DemoCopyFile():
56       # Create a file on the device, and write a string.
57       cefile = wincerapi.CeCreateFile(
58           "TestPython", win32con.GENERIC_WRITE, 0, None, win32con.OPEN_ALWAYS, 0, None
59       )
60       wincerapi.CeWriteFile(cefile, "Hello from Python")
61       cefile.Close()
62       # reopen the file and check the data.
63       cefile = wincerapi.CeCreateFile(
64           "TestPython", win32con.GENERIC_READ, 0, None, win32con.OPEN_EXISTING, 0, None
65       )
66       if wincerapi.CeReadFile(cefile, 100) != "Hello from Python":
67           print("Couldnt read the data from the device!")
68       cefile.Close()
69       # Delete the test file
70       wincerapi.CeDeleteFile("TestPython")
71       print("Created, wrote to, read from and deleted a test file!")

138  def DumpRemoteFolders():
139      # Dump all special folders possible.
140      for name, val in list(wincerapi.__dict__.items()):
141          if name[:6] == "CSIDL_":
142              try:
143                  loc = str(wincerapi.CeGetSpecialFolderPath(val))
144                  print("Folder %s is at %s" % (name, loc))
145              except win32api.error as details:
146                  pass
147  
148      # Get the shortcut targets for the "Start Menu"
149      print("Dumping start menu shortcuts...")
150      try:
151          startMenu = str(wincerapi.CeGetSpecialFolderPath(wincerapi.CSIDL_STARTMENU))
152      except win32api.error as details:
153          print("This device has no start menu!", details)
154          startMenu = None
155  
156      if startMenu:
157          for fileAttr in wincerapi.CeFindFiles(os.path.join(startMenu, "*")):
158              fileName = fileAttr[8]
159              fullPath = os.path.join(startMenu, str(fileName))
160              try:
161                  resolved = wincerapi.CeSHGetShortcutTarget(fullPath)
162              except win32api.error as xxx_todo_changeme:
163                  (rc, fn, msg) = xxx_todo_changeme.args
164                  resolved = "#Error - %s" % msg
165              print("%s->%s" % (fileName, resolved))
166  
167      #       print "The start menu is at",
168      #       print wincerapi.CeSHGetShortcutTarget("\\Windows\\Start Menu\\Shortcut to Python.exe.lnk")

171  def usage():
172      print("Options:")
173      print("-a - Execute all demos")
174      print("-p - Execute Python process on remote device")
175      print("-r - Dump the remote registry")
176      print("-f - Dump all remote special folder locations")
177      print("-s - Dont dump machine status")
178      print("-y - Perform asynch init of CE connection")

181  def main():
182      async_init = bStartPython = bDumpRegistry = bDumpFolders = 0
183      bDumpStatus = 1
184      try:
185          opts, args = getopt.getopt(sys.argv[1:], "apr")
186      except getopt.error as why:
187          print("Invalid usage:", why)
188          usage()
189          return
190  
191      for o, v in opts:
192          if o == "-a":
193              bStartPython = bDumpRegistry = bDumpStatus = bDumpFolders = asynch_init = 1
194          if o == "-p":
195              bStartPython = 1
196          if o == "-r":
197              bDumpRegistry = 1
198          if o == "-s":
199              bDumpStatus = 0
200          if o == "-f":
201              bDumpFolders = 1
202          if o == "-y":
203              print("Doing asynch init of CE connection")
204              async_init = 1
205  
206      if async_init:
207          event, rc = wincerapi.CeRapiInitEx()
208          while 1:
209              rc = win32event.WaitForSingleObject(event, 500)
210              if rc == win32event.WAIT_OBJECT_0:
211                  # We connected.
212                  break
213              else:
214                  print(
215                      "Waiting for Initialize to complete (picture a Cancel button here :)"
216                  )
217      else:
218          wincerapi.CeRapiInit()
219      print("Connected to remote CE device.")
220      try:
221          verinfo = wincerapi.CeGetVersionEx()
222          print(
223              "The device is running windows CE version %d.%d - %s"
224              % (verinfo[0], verinfo[1], verinfo[4])
225          )
226  
227          if bDumpStatus:
228              print("Dumping remote machine status")
229              DumpRemoteMachineStatus()
230  
231          if bDumpRegistry:
232              print("Dumping remote registry...")
233              DumpRegistry(win32con.HKEY_LOCAL_MACHINE)
234  
235          if bDumpFolders:
236              print("Dumping remote folder information")
237              DumpRemoteFolders()
238  
239          DemoCopyFile()
240          if bStartPython:
241              print("Starting remote Python process")
242              if DumpPythonRegistry():
243                  DemoCreateProcess()
244              else:
245                  print("Not trying to start Python, as it's not installed")
246  
247      finally:
248          wincerapi.CeRapiUninit()
249          print("Disconnected")
250  
251  
252  if __name__ == "__main__":
253      main()
