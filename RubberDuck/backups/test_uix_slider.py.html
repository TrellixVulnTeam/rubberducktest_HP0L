50       def test_slider_move(self):
51           EventLoop.ensure_window()
52           win = EventLoop.window
53           layout = BoxLayout(orientation='vertical')
54   
55           s_handle = _TestSliderHandle()
56           s_all = _TestSliderAll()
57           layout.add_widget(s_handle)
58           layout.add_widget(s_all)
59           win.add_widget(layout)
60   
61           # get widgets ready
62           EventLoop.idle()
63   
64           cur1 = s_handle.children[0]
65           cur2 = s_all.children[0]
66   
67           h1 = cur1.to_window(*cur1.center)[1]
68           h2 = h1 - s_handle.cursor_height
69           h3 = cur2.to_window(*cur2.center)[1]
70           h4 = h3 - s_all.cursor_height
71   
72           w1 = cur1.to_window(*cur1.center)[0]
73           w2 = cur2.to_window(*cur2.center)[0]
74           wh = win.width / 2.0
75           dt = 2
76   
77           # default pos, new pos, slider ID
78           points = [
79               [w1, h1, wh, h1, 'handle'],
80               [w1, h2, wh, h2, 'handle'],
81               [w2, h3, wh, h3, 'all'],
82               [w2, h4, wh, h4, 'all'],
83           ]
84   
85           for point in points:
86               x, y, nx, ny, id = point
87   
88               # custom touch
89               touch = UTMotionEvent("unittest", 1, {
90                   "x": x / float(win.width),
91                   "y": y / float(win.height),
92               })
93   
94               # touch down
95               EventLoop.post_dispatch_input("begin", touch)
96   
97               if id == 'handle':
98                   # touch on handle
99                   if x == w1 and y == h1:
100                      self.assertAlmostEqual(
101                          s_handle.value, 0.0,
102                          delta=dt
103                      )
104                  # touch in widget area (ignored, previous value)
105                  elif x == w1 and y == h2:
106                      self.assertAlmostEqual(
107                          s_handle.value, 50.0,
108                          delta=dt
109                      )
110              elif id == 'all':
111                  # touch on handle:
112                  if x == w1 and y == h3:
113                      self.assertAlmostEqual(
114                          s_all.value, 0.0,
115                          delta=dt
116                      )
117                  # touch in widget area
118                  elif x == w1 and y == h4:
119                      self.assertAlmostEqual(
120                          s_all.value, 0.0,
121                          delta=dt
122                      )
123  
124              # move from default to new pos
125              touch.move({
126                  "x": nx / float(win.width),
127                  "y": ny / float(win.height)
128              })
129              EventLoop.post_dispatch_input("update", touch)
130  
131              if id == 'handle':
132                  # move from handle to center
133                  if nx == wh and ny == h1:
134                      self.assertAlmostEqual(
135                          s_handle.value, 50.0,
136                          delta=dt
137                      )
138                  # move to center (ignored, previous value)
139                  elif nx == wh and ny == h2:
140                      self.assertAlmostEqual(
141                          s_handle.value, 50.0,
142                          delta=dt
143                      )
144              elif id == 'all':
145                  # touch on handle:
146                  if nx == wh and ny == h3:
147                      self.assertAlmostEqual(
148                          s_all.value, 50.0,
149                          delta=dt
150                      )
151                  # touch in widget area
152                  elif nx == wh and ny == h4:
153                      self.assertAlmostEqual(
154                          s_all.value, 50.0,
155                          delta=dt
156                      )
157  
158              # touch up
159              EventLoop.post_dispatch_input("end", touch)
160  
161          self.render(layout)
