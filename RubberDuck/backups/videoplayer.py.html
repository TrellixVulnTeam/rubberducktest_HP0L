videoplayer.py 
1    ''' 
2    Video player 
3    ============ 
4     
5    .. versionadded:: 1.2.0 
6     
7    The video player widget can be used to play video and let the user control the 
8    play/pausing, volume and position. The widget cannot be customized much because 
9    of the complex assembly of numerous base widgets. 
10    
11   .. image:: images/videoplayer.jpg 
12       :align: center 
13    
14   Annotations 
15   ----------- 
16    
17   If you want to display text at a specific time and for a certain duration, 
18   consider annotations. An annotation file has a ".jsa" extension. The player 
19   will automatically load the associated annotation file if it exists. 
20    
21   An annotation file is JSON-based, providing a list of label dictionary items. 
22   The key and value must match one of the :class:`VideoPlayerAnnotation` items. 
23   For example, here is a short version of a jsa file that you can find in 
24   `examples/widgets/cityCC0.jsa`:: 
25    
26    
27       [ 
28           {"start": 0, "duration": 2, 
29           "text": "This is an example of annotation"}, 
30           {"start": 2, "duration": 2, 
31           "bgcolor": [0.5, 0.2, 0.4, 0.5], 
32           "text": "You can change the background color"} 
33       ] 
34    
35   For our cityCC0.mpg example, the result will be: 
36    
37   .. image:: images/videoplayer-annotation.jpg 
38       :align: center 
39    
40   If you want to experiment with annotation files, test with:: 
41    
42       python -m kivy.uix.videoplayer examples/widgets/cityCC0.mpg 
43    
44   Fullscreen 
45   ---------- 
46    
47   The video player can play the video in fullscreen, if 
48   :attr:`VideoPlayer.allow_fullscreen` is activated by a double-tap on 
49   the video. By default, if the video is smaller than the Window, it will be not 
50   stretched. 
51    
52   You can allow stretching by passing custom options to a 
53   :class:`VideoPlayer` instance:: 
54    
55       player = VideoPlayer(source='myvideo.avi', state='play', 
56           options={'allow_stretch': True}) 
57    
58   End-of-stream behavior 
59   ---------------------- 
60    
61   You can specify what happens when the video has finished playing by passing an 
62   `eos` (end of stream) directive to the underlying 
63   :class:`~kivy.core.video.VideoBase` class. `eos` can be one of 'stop', 'pause' 
64   or 'loop' and defaults to 'stop'. For example, in order to loop the video:: 
65    
66       player = VideoPlayer(source='myvideo.avi', state='play', 
67           options={'eos': 'loop'}) 
68    
69   .. note:: 
70    
71       The `eos` property of the VideoBase class is a string specifying the 
72       end-of-stream behavior. This property differs from the `eos` 
73       properties of the :class:`VideoPlayer` and 
74       :class:`~kivy.uix.video.Video` classes, whose `eos` 
75       property is simply a boolean indicating that the end of the file has 
76       been reached. 
77    
78   '''

98   class VideoPlayerVolume(Image):
99       video = ObjectProperty(None)
100  
101      def on_touch_down(self, touch):
102          if not self.collide_point(*touch.pos):
103              return False
104          touch.grab(self)
105          # save the current volume and delta to it
106          touch.ud[self.uid] = [self.video.volume, 0]
107          return True
108  
109      def on_touch_move(self, touch):
110          if touch.grab_current is not self:
111              return
112          # calculate delta
113          dy = abs(touch.y - touch.oy)
114          if dy > 10:
115              dy = min(dy - 10, 100)
116              touch.ud[self.uid][1] = dy
117              self.video.volume = dy / 100.
118          return True
119  
120      def on_touch_up(self, touch):
121          if touch.grab_current is not self:
122              return
123          touch.ungrab(self)
124          dy = abs(touch.y - touch.oy)
125          if dy < 10:
126              if self.video.volume > 0:
127                  self.video.volume = 0
128              else:
129                  self.video.volume = 1.

155  class VideoPlayerProgressBar(ProgressBar):
156      video = ObjectProperty(None)
157      seek = NumericProperty(None, allownone=True)
158      alpha = NumericProperty(1.)
159  
160      def __init__(self, **kwargs):
161          super(VideoPlayerProgressBar, self).__init__(**kwargs)
162          self.bubble = Factory.Bubble(size=(50, 44))
163          self.bubble_label = Factory.Label(text='0:00')
164          self.bubble.add_widget(self.bubble_label)
165          self.add_widget(self.bubble)
166  
167          update = self._update_bubble
168          fbind = self.fbind
169          fbind('pos', update)
170          fbind('size', update)
171          fbind('seek', update)
172  
173      def on_video(self, instance, value):
174          self.video.bind(position=self._update_bubble,
175                          state=self._showhide_bubble)
176  
177      def on_touch_down(self, touch):
178          if not self.collide_point(*touch.pos):
179              return
180          self._show_bubble()
181          touch.grab(self)
182          self._update_seek(touch.x)
183          return True
184  
185      def on_touch_move(self, touch):
186          if touch.grab_current is not self:
187              return
188          self._update_seek(touch.x)
189          return True
190  
191      def on_touch_up(self, touch):
192          if touch.grab_current is not self:
193              return
194          touch.ungrab(self)
195          if self.seek:
196              self.video.seek(self.seek)
197          self.seek = None
198          self._hide_bubble()
199          return True
200  
201      def _update_seek(self, x):
202          if self.width == 0:
203              return
204          x = max(self.x, min(self.right, x)) - self.x
205          self.seek = x / float(self.width)
206  
207      def _show_bubble(self):
208          self.alpha = 1
209          Animation.stop_all(self, 'alpha')
210  
211      def _hide_bubble(self):
212          self.alpha = 1.
213          Animation(alpha=0, d=4, t='in_out_expo').start(self)
214  
215      def on_alpha(self, instance, value):
216          self.bubble.background_color = (1, 1, 1, value)
217          self.bubble_label.color = (1, 1, 1, value)
218  
219      def _update_bubble(self, *l):
220          seek = self.seek
221          if self.seek is None:
222              if self.video.duration == 0:
223                  seek = 0
224              else:
225                  seek = self.video.position / self.video.duration
226          # convert to minutes:seconds
227          d = self.video.duration * seek
228          minutes = int(d / 60)
229          seconds = int(d - (minutes * 60))
230          # fix bubble label & position
231          self.bubble_label.text = '%d:%02d' % (minutes, seconds)
232          self.bubble.center_x = self.x + seek * self.width
233          self.bubble.y = self.top
234  
235      def _showhide_bubble(self, instance, value):
236          if value == 'play':
237              self._hide_bubble()
238          else:
239              self._show_bubble()

480      # internals
481      container = ObjectProperty(None)
482  
483      _video_load_ev = None
484  
485      def __init__(self, **kwargs):
486          self._video = None
487          self._image = None
488          self._annotations = ''
489          self._annotations_labels = []
490          super(VideoPlayer, self).__init__(**kwargs)
491          self._load_thumbnail()
492          self._load_annotations()
493  
494          if self.source:
495              self._trigger_video_load()
496  
497      def _trigger_video_load(self, *largs):
498          ev = self._video_load_ev
499          if ev is None:
500              ev = self._video_load_ev = Clock.schedule_once(self._do_video_load,
501                                                             -1)
502          ev()
503  
504      def on_source(self, instance, value):
505          # we got a value, try to see if we have an image for it
506          self._load_thumbnail()
507          self._load_annotations()
508          if self._video is not None:
509              self._video.unload()
510              self._video = None
511          if value:
512              self._trigger_video_load()
513  
514      def on_image_overlay_play(self, instance, value):
515          self._image.image_overlay_play = value
516  
517      def on_image_loading(self, instance, value):
518          self._image.image_loading = value
519  
520      def _load_thumbnail(self):
521          if not self.container:
522              return
523          self.container.clear_widgets()
524          # get the source, remove extension, and use png
525          thumbnail = self.thumbnail
526          if not thumbnail:
527              filename = self.source.rsplit('.', 1)
528              thumbnail = filename[0] + '.png'
529              if not exists(thumbnail):
530                  thumbnail = ''
531          self._image = VideoPlayerPreview(source=thumbnail, video=self)
532          self.container.add_widget(self._image)
533  
534      def _load_annotations(self):
535          if not self.container:
536              return
537          self._annotations_labels = []
538          annotations = self.annotations
539          if not annotations:
540              filename = self.source.rsplit('.', 1)
541              annotations = filename[0] + '.jsa'
542          if exists(annotations):
543              with open(annotations, 'r') as fd:
544                  self._annotations = load(fd)
545          if self._annotations:
546              for ann in self._annotations:
547                  self._annotations_labels.append(
548                      VideoPlayerAnnotation(annotation=ann))
549  
550      def on_state(self, instance, value):
551          if self._video is not None:
552              self._video.state = value
553  
554      def _set_state(self, instance, value):
555          self.state = value
556  
557      def _do_video_load(self, *largs):
558          self._video = Video(source=self.source, state=self.state,
559                              volume=self.volume, pos_hint={'x': 0, 'y': 0},
560                              **self.options)
561          self._video.bind(texture=self._play_started,
562                           duration=self.setter('duration'),
563                           position=self.setter('position'),
564                           volume=self.setter('volume'),
565                           state=self._set_state)
566  
567      def on_play(self, instance, value):
568          value = 'play' if value else 'stop'
569          return self.on_state(instance, value)
570  
571      def on_volume(self, instance, value):
572          if not self._video:
573              return
574          self._video.volume = value
575  
576      def on_position(self, instance, value):
577          labels = self._annotations_labels
578          if not labels:
579              return
580          for label in labels:
581              start = label.start
582              duration = label.duration
583              if start > value or (start + duration) < value:
584                  if label.parent:
585                      label.parent.remove_widget(label)
586              elif label.parent is None:
587                  self.container.add_widget(label)
588  
589      def seek(self, percent, precise=True):
590          '''Change the position to a percentage (strictly, a proportion) 
591             of duration. 
592   
593          :Parameters: 
594              `percent`: float or int 
595                  Position to seek as a proportion of total duration, must 
596                  be between 0-1. 
597              `precise`: bool, defaults to True 
598                  Precise seeking is slower, but seeks to exact requested 
599                  percent. 
600   
601          .. warning:: 
602              Calling seek() before the video is loaded has no effect. 
603   
604          .. versionadded:: 1.2.0 
605   
606          .. versionchanged:: 1.10.1 
607              The `precise` keyword argument has been added. 
608          '''
609          if not self._video:
610              return
611          self._video.seek(percent, precise=precise)
612  
613      def _play_started(self, instance, value):
614          self.container.clear_widgets()
615          self.container.add_widget(self._video)
616  
617      def on_touch_down(self, touch):
618          if not self.collide_point(*touch.pos):
619              return False
620          if touch.is_double_tap and self.allow_fullscreen:
621              self.fullscreen = not self.fullscreen
622              return True
623          return super(VideoPlayer, self).on_touch_down(touch)
624  
625      def on_fullscreen(self, instance, value):
626          window = self.get_parent_window()
627          if not window:
628              Logger.warning('VideoPlayer: Cannot switch to fullscreen, '
629                             'window not found.')
630              if value:
631                  self.fullscreen = False
632              return
633          if not self.parent:
634              Logger.warning('VideoPlayer: Cannot switch to fullscreen, '
635                             'no parent.')
636              if value:
637                  self.fullscreen = False
638              return
639  
640          if value:
641              self._fullscreen_state = state = {
642                  'parent': self.parent,
643                  'pos': self.pos,
644                  'size': self.size,
645                  'pos_hint': self.pos_hint,
646                  'size_hint': self.size_hint,
647                  'window_children': window.children[:]}
648  
649              # remove all window children
650              for child in window.children[:]:
651                  window.remove_widget(child)
652  
653              # put the video in fullscreen
654              if state['parent'] is not window:
655                  state['parent'].remove_widget(self)
656              window.add_widget(self)
657  
658              # ensure the video widget is in 0, 0, and the size will be
659              # readjusted
660              self.pos = (0, 0)
661              self.size = (100, 100)
662              self.pos_hint = {}
663              self.size_hint = (1, 1)
664          else:
665              state = self._fullscreen_state
666              window.remove_widget(self)
667              for child in state['window_children']:
668                  window.add_widget(child)
669              self.pos_hint = state['pos_hint']
670              self.size_hint = state['size_hint']
671              self.pos = state['pos']
672              self.size = state['size']
673              if state['parent'] is not window:
674                  state['parent'].add_widget(self)
