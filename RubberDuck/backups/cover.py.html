1    ''' 
2    Cover Behavior 
3    ============== 
4     
5    The :class:`~kivy.uix.behaviors.cover.CoverBehavior` 
6    `mixin <https://en.wikipedia.org/wiki/Mixin>`_ is intended for rendering 
7    textures to full widget size keeping the aspect ratio of the original texture. 
8     
9    Use cases are i.e. rendering full size background images or video content in 
10   a dynamic layout. 
11    
12   For an overview of behaviors, please refer to the :mod:`~kivy.uix.behaviors` 
13   documentation. 
14    
15   Example 
16   ------- 
17    
18   The following examples add cover behavior to an image: 
19    
20   In python: 
21    
22   .. code-block:: python 
23    
24       from kivy.app import App 
25       from kivy.uix.behaviors import CoverBehavior 
26       from kivy.uix.image import Image 
27    
28    
29       class CoverImage(CoverBehavior, Image): 
30    
31           def __init__(self, **kwargs): 
32               super(CoverImage, self).__init__(**kwargs) 
33               texture = self._coreimage.texture 
34               self.reference_size = texture.size 
35               self.texture = texture 
36    
37    
38       class MainApp(App): 
39    
40           def build(self): 
41               return CoverImage(source='image.jpg') 
42    
43       MainApp().run() 
44    
45   In Kivy Language: 
46    
47   .. code-block:: kv 
48    
49       CoverImage: 
50           source: 'image.png' 
51    
52       <CoverImage@CoverBehavior+Image>: 
53           reference_size: self.texture_size 
54    
55   See :class:`~kivy.uix.behaviors.cover.CoverBehavior` for details. 
56   '''
57   
58   __all__ = ('CoverBehavior', )
59   
60   from decimal import Decimal
61   from kivy.lang import Builder
62   from kivy.properties import ListProperty
63   
64   
65   Builder.load_string(""" 
66   <-CoverBehavior>: 
67       canvas.before: 
68           StencilPush 
69           Rectangle: 
70               pos: self.pos 
71               size: self.size 
72           StencilUse 
73       canvas: 
74           Rectangle: 
75               texture: self.texture 
76               size: self.cover_size 
77               pos: self.cover_pos 
78       canvas.after: 
79           StencilUnUse 
80           Rectangle: 
81               pos: self.pos 
82               size: self.size 
83           StencilPop 
84   """)
85   
86   
87   class CoverBehavior(object):
88       '''The CoverBehavior `mixin <https://en.wikipedia.org/wiki/Mixin>`_ 
89       provides rendering a texture covering full widget size keeping aspect ratio 
90       of the original texture. 
91    
92       .. versionadded:: 1.10.0 
93       '''
94   
95       reference_size = ListProperty([])
96       '''Reference size used for aspect ratio approximation calculation. 
97    
98       :attr:`reference_size` is a :class:`~kivy.properties.ListProperty` and 
99       defaults to `[]`. 
100      '''
101  
102      cover_size = ListProperty([0, 0])
103      '''Size of the aspect ratio aware texture. Gets calculated in 
104      ``CoverBehavior.calculate_cover``. 
105   
106      :attr:`cover_size` is a :class:`~kivy.properties.ListProperty` and 
107      defaults to `[0, 0]`. 
108      '''
109  
110      cover_pos = ListProperty([0, 0])
111      '''Position of the aspect ratio aware texture. Gets calculated in 
112      ``CoverBehavior.calculate_cover``. 
113   
114      :attr:`cover_pos` is a :class:`~kivy.properties.ListProperty` and 
115      defaults to `[0, 0]`. 
116      '''
117  
118      def __init__(self, **kwargs):
119          super(CoverBehavior, self).__init__(**kwargs)
120          # bind covering
121          self.bind(
122              size=self.calculate_cover,
123              pos=self.calculate_cover
124          )
125  
126      def _aspect_ratio_approximate(self, size):
127          # return a decimal approximation of an aspect ratio.
128          return Decimal('%.2f' % (float(size[0]) / size[1]))
129  
130      def _scale_size(self, size, sizer):
131          # return scaled size based on sizer, where sizer (n, None) scales x
132          # to n and (None, n) scales y to n
133          size_new = list(sizer)
134          i = size_new.index(None)
135          j = i * -1 + 1
136          size_new[i] = (size_new[j] * size[i]) / size[j]
137          return tuple(size_new)
138  
139      def calculate_cover(self, *args):
140          # return if no reference size yet
141          if not self.reference_size:
142              return
143          size = self.size
144          origin_appr = self._aspect_ratio_approximate(self.reference_size)
145          crop_appr = self._aspect_ratio_approximate(size)
146          # same aspect ratio
147          if origin_appr == crop_appr:
148              crop_size = self.size
149              offset = (0, 0)
150          # scale x
151          elif origin_appr < crop_appr:
152              crop_size = self._scale_size(self.reference_size, (size[0], None))
153              offset = (0, ((crop_size[1] - size[1]) / 2) * -1)
154          # scale y
155          else:
156              crop_size = self._scale_size(self.reference_size, (None, size[1]))
157              offset = (((crop_size[0] - size[0]) / 2) * -1, 0)
158          # set background size and position
159          self.cover_size = crop_size
160          self.cover_pos = offset
