5    ## A nice, small 'hello world' Pythonwin application.
6    ## NOT an MDI application - just a single, normal, top-level window.
7    ##
8    ## MUST be run with the command line "pythonwin.exe /app helloapp.py"
9    ## (or if you are really keen, rename "pythonwin.exe" to something else, then
10   ## using MSVC or similar, edit the string section in the .EXE to name this file)
11   ##
12   ## Originally by Willy Heineman <wheineman@uconect.net>
13   
14   
15   import win32con
16   import win32ui
17   from pywin.mfc import window, dialog, afxres
18   from pywin.mfc.thread import WinApp
19   
20   # The main frame.
21   # Does almost nothing at all - doesnt even create a child window!
22   class HelloWindow(window.Wnd):
23       def __init__(self):
24           # The window.Wnd ctor creates a Window object, and places it in
25           # self._obj_.  Note the window object exists, but the window itself
26           # does not!
27           window.Wnd.__init__(self, win32ui.CreateWnd())
28   
29           # Now we ask the window object to create the window itself.
30           self._obj_.CreateWindowEx(
31               win32con.WS_EX_CLIENTEDGE,
32               win32ui.RegisterWndClass(0, 0, win32con.COLOR_WINDOW + 1),
33               "Hello World!",
34               win32con.WS_OVERLAPPEDWINDOW,
35               (100, 100, 400, 300),
36               None,
37               0,
38               None,
39           )
40   
41   
42   # The application object itself.
43   class HelloApp(WinApp):
44       def InitInstance(self):
45           self.frame = HelloWindow()
46           self.frame.ShowWindow(win32con.SW_SHOWNORMAL)
47           # We need to tell MFC what our main frame is.
48           self.SetMainFrame(self.frame)
49   
50   
51   # Now create the application object itself!
52   app = HelloApp()
