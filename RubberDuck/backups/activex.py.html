14   class Control(window.Wnd):
15       """An ActiveX control base class.  A new class must be derived from both 
16       this class and the Events class.  See the demos for more details. 
17       """
18   
19       def __init__(self):
20           self.__dict__["_dispobj_"] = None
21           window.Wnd.__init__(self)
22   
23       def _GetControlCLSID(self):
24           return self.CLSID
25   
26       def _GetDispatchClass(self):
27           return self.default_interface
28   
29       def _GetEventMap(self):
30           return self.default_source._dispid_to_func_
31   
32       def CreateControl(self, windowTitle, style, rect, parent, id, lic_string=None):
33           clsid = str(self._GetControlCLSID())
34           self.__dict__["_obj_"] = win32ui.CreateControl(
35               clsid, windowTitle, style, rect, parent, id, None, False, lic_string
36           )
37           klass = self._GetDispatchClass()
38           dispobj = klass(win32uiole.GetIDispatchForWindow(self._obj_))
39           self.HookOleEvents()
40           self.__dict__["_dispobj_"] = dispobj
41   
42       def HookOleEvents(self):
43           dict = self._GetEventMap()
44           for dispid, methodName in dict.items():
45               if hasattr(self, methodName):
46                   self._obj_.HookOleEvent(getattr(self, methodName), dispid)
47   
48       def __getattr__(self, attr):
49           # Delegate attributes to the windows and the Dispatch object for this class
50           try:
51               return window.Wnd.__getattr__(self, attr)
52           except AttributeError:
53               pass
54           return getattr(self._dispobj_, attr)
55   
56       def __setattr__(self, attr, value):
57           if hasattr(self.__dict__, attr):
58               self.__dict__[attr] = value
59               return
60           try:
61               if self._dispobj_:
62                   self._dispobj_.__setattr__(attr, value)
63                   return
64           except AttributeError:
65               pass
66           self.__dict__[attr] = value
67

69   def MakeControlClass(controlClass, name=None):
70       """Given a CoClass in a generated .py file, this function will return a Class 
71       object which can be used as an OCX control. 
72    
73       This function is used when you do not want to handle any events from the OCX 
74       control.  If you need events, then you should derive a class from both the 
75       activex.Control class and the CoClass 
76       """
77       if name is None:
78           name = controlClass.__name__
79       return new_type("OCX" + name, (Control, controlClass), {})
80   
81   
82   def MakeControlInstance(controlClass, name=None):
83       """As for MakeControlClass(), but returns an instance of the class."""
84       return MakeControlClass(controlClass, name)()
