touchring.py 
1    ''' 
2    Touchring 
3    ========= 
4     
5    Shows rings around every touch on the surface / screen. You can use this module 
6    to check that you don't have any calibration issues with touches. 
7     
8    Configuration 
9    ------------- 
10    
11   :Parameters: 
12       `image`: str, defaults to '<kivy>/data/images/ring.png' 
13           Filename of the image to use. 
14       `scale`: float, defaults to 1. 
15           Scale of the image. 
16       `alpha`: float, defaults to 1. 
17           Opacity of the image. 
18    
19   Example 
20   ------- 
21    
22   In your configuration (`~/.kivy/config.ini`), you can add something like 
23   this:: 
24    
25       [modules] 
26       touchring = image=mypointer.png,scale=.3,alpha=.7 
27    
28   '''
29   
30   __all__ = ('start', 'stop')
31   
32   from kivy.core.image import Image
33   from kivy.graphics import Color, Rectangle
34   from kivy import kivy_data_dir
35   from os.path import join
36   
37   pointer_image = None
38   pointer_scale = 1.0
39   pointer_alpha = 0.7
40   
41   
42   def _touch_down(win, touch):
43       ud = touch.ud
44       with win.canvas.after:
45           ud['tr.color'] = Color(1, 1, 1, pointer_alpha)
46           iw, ih = pointer_image.size
47           ud['tr.rect'] = Rectangle(
48               pos=(
49                   touch.x - (pointer_image.width / 2. * pointer_scale),
50                   touch.y - (pointer_image.height / 2. * pointer_scale)),
51               size=(iw * pointer_scale, ih * pointer_scale),
52               texture=pointer_image.texture)
53   
54       if not ud.get('tr.grab', False):
55           ud['tr.grab'] = True
56           touch.grab(win)
57   
58   
59   def _touch_move(win, touch):
60       ud = touch.ud
61       if not ud.get('tr.rect', False):
62           _touch_down(win, touch)
63       ud['tr.rect'].pos = (
64           touch.x - (pointer_image.width / 2. * pointer_scale),
65           touch.y - (pointer_image.height / 2. * pointer_scale))
66   
67   
68   def _touch_up(win, touch):
69       if touch.grab_current is win:
70           ud = touch.ud
71           win.canvas.after.remove(ud['tr.color'])
72           win.canvas.after.remove(ud['tr.rect'])
73   
74           if ud.get('tr.grab') is True:
75               touch.ungrab(win)
76               ud['tr.grab'] = False
77   
78   
79   def start(win, ctx):
80       # XXX use ctx !
81       global pointer_image, pointer_scale, pointer_alpha
82   
83       pointer_fn = ctx.config.get('image',
84                                   'atlas://data/images/defaulttheme/ring')
85       pointer_scale = float(ctx.config.get('scale', 1.0))
86       pointer_alpha = float(ctx.config.get('alpha', 1.0))
87       pointer_image = Image(pointer_fn)
88   
89       win.bind(on_touch_down=_touch_down,
90                on_touch_move=_touch_move,
91                on_touch_up=_touch_up)
92   
93   
94   def stop(win, ctx):
95       win.unbind(on_touch_down=_touch_down,
96                  on_touch_move=_touch_move,
97                  on_touch_up=_touch_up)
