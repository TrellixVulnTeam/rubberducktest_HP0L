1    ''' 
2    TUIO Input Provider 
3    =================== 
4     
5    TUIO is the de facto standard network protocol for the transmission of 
6    touch and fiducial information between a server and a client. To learn 
7    more about TUIO (which is itself based on the OSC protocol), please 
8    refer to http://tuio.org -- The specification should be of special 
9    interest. 
10    
11   Configure a TUIO provider in the config.ini 
12   ------------------------------------------- 
13    
14   The TUIO provider can be configured in the configuration file in the 
15   ``[input]`` section:: 
16    
17       [input] 
18       # name = tuio,<ip>:<port> 
19       multitouchtable = tuio,192.168.0.1:3333 
20    
21   Configure a TUIO provider in the App 
22   ------------------------------------ 
23    
24   You must add the provider before your application is run, like this:: 
25    
26       from kivy.app import App 
27       from kivy.config import Config 
28    
29       class TestApp(App): 
30           def build(self): 
31               Config.set('input', 'multitouchscreen1', 'tuio,0.0.0.0:3333') 
32               # You can also add a second TUIO listener 
33               # Config.set('input', 'source2', 'tuio,0.0.0.0:3334') 
34               # Then do the usual things 
35               # ... 
36               return 
37   '''

52   class TuioMotionEventProvider(MotionEventProvider):
53       '''The TUIO provider listens to a socket and handles some of the incoming 
54       OSC messages: 
55    
56           * /tuio/2Dcur 
57           * /tuio/2Dobj 
58    
59       You can easily extend the provider to handle new TUIO paths like so:: 
60    
61           # Create a class to handle the new TUIO type/path 
62           # Replace NEWPATH with the pathname you want to handle 
63           class TuioNEWPATHMotionEvent(MotionEvent): 
64               def __init__(self, id, args): 
65                   super(TuioNEWPATHMotionEvent, self).__init__(id, args) 
66    
67               def depack(self, args): 
68                   # In this method, implement 'unpacking' for the received 
69                   # arguments. you basically translate from TUIO args to Kivy 
70                   # MotionEvent variables. If all you receive are x and y 
71                   # values, you can do it like this: 
72                   if len(args) == 2: 
73                       self.sx, self.sy = args 
74                       self.profile = ('pos', ) 
75                   self.sy = 1 - self.sy 
76                   super(TuioNEWPATHMotionEvent, self).depack(args) 
77    
78           # Register it with the TUIO MotionEvent provider. 
79           # You obviously need to replace the PATH placeholders appropriately. 
80           TuioMotionEventProvider.register('/tuio/PATH', TuioNEWPATHMotionEvent) 
81    
82       .. note:: 
83    
84           The class name is of no technical importance. Your class will be 
85           associated with the path that you pass to the ``register()`` 
86           function. To keep things simple, you should name your class after the 
87           path that it handles, though. 
88       '''
89   
90       __handlers__ = {}
91   
92       def __init__(self, device, args):
93           super(TuioMotionEventProvider, self).__init__(device, args)
94           args = args.split(',')
95           if len(args) <= 0:
96               Logger.error('Tuio: Invalid configuration for TUIO provider')
97               Logger.error('Tuio: Format must be ip:port (eg. 127.0.0.1:3333)')
98               err = 'Tuio: Current configuration is <%s>' % (str(','.join(args)))
99               Logger.error(err)
100              return
101          ipport = args[0].split(':')
102          if len(ipport) != 2:
103              Logger.error('Tuio: Invalid configuration for TUIO provider')
104              Logger.error('Tuio: Format must be ip:port (eg. 127.0.0.1:3333)')
105              err = 'Tuio: Current configuration is <%s>' % (str(','.join(args)))
106              Logger.error(err)
107              return
108          self.ip, self.port = args[0].split(':')
109          self.port = int(self.port)
110          self.handlers = {}
111          self.oscid = None
112          self.tuio_event_q = deque()
113          self.touches = {}
114  
115      @staticmethod
116      def register(oscpath, classname):
117          '''Register a new path to handle in TUIO provider'''
118          TuioMotionEventProvider.__handlers__[oscpath] = classname
119  
120      @staticmethod
121      def unregister(oscpath, classname):
122          '''Unregister a path to stop handling it in the TUIO provider'''
123          if oscpath in TuioMotionEventProvider.__handlers__:
124              del TuioMotionEventProvider.__handlers__[oscpath]
125  
126      @staticmethod
127      def create(oscpath, **kwargs):
128          '''Create a touch event from a TUIO path'''
129          if oscpath not in TuioMotionEventProvider.__handlers__:
130              raise Exception('Unknown %s touch path' % oscpath)
131          return TuioMotionEventProvider.__handlers__[oscpath](**kwargs)
132  
133      def start(self):
134          '''Start the TUIO provider'''
135          try:
136              from oscpy.server import OSCThreadServer
137          except ImportError:
138              Logger.info(
139                  'Please install the oscpy python module to use the TUIO '
140                  'provider.'
141              )
142              raise
143          self.oscid = osc = OSCThreadServer()
144          osc.listen(self.ip, self.port, default=True)
145          for oscpath in TuioMotionEventProvider.__handlers__:
146              self.touches[oscpath] = {}
147              osc.bind(oscpath, partial(self._osc_tuio_cb, oscpath))
148  
149      def stop(self):
150          '''Stop the TUIO provider'''
151          self.oscid.stop_all()
152  
153      def update(self, dispatch_fn):
154          '''Update the TUIO provider (pop events from the queue)'''
155  
156          # read the Queue with event
157          while True:
158              try:
159                  value = self.tuio_event_q.pop()
160              except IndexError:
161                  # queue is empty, we're done for now
162                  return
163              self._update(dispatch_fn, value)
164  
165      def _osc_tuio_cb(self, oscpath, address, *args):
166          self.tuio_event_q.appendleft([oscpath, address, args])
167  
168      def _update(self, dispatch_fn, value):
169          oscpath, command, args = value
170  
171          # verify commands
172          if command not in [b'alive', b'set']:
173              return
174  
175          # move or create a new touch
176          if command == b'set':
177              id = args[0]
178              if id not in self.touches[oscpath]:
179                  # new touch
180                  touch = TuioMotionEventProvider.__handlers__[oscpath](
181                      self.device, id, args[1:])
182                  self.touches[oscpath][id] = touch
183                  dispatch_fn('begin', touch)
184              else:
185                  # update a current touch
186                  touch = self.touches[oscpath][id]
187                  touch.move(args[1:])
188                  dispatch_fn('update', touch)
189  
190          # alive event, check for deleted touch
191          if command == b'alive':
192              alives = args
193              to_delete = []
194              for id in self.touches[oscpath]:
195                  if id not in alives:
196                      # touch up
197                      touch = self.touches[oscpath][id]
198                      if touch not in to_delete:
199                          to_delete.append(touch)
200  
201              for touch in to_delete:
202                  dispatch_fn('end', touch)
203                  del self.touches[oscpath][touch.id]

206  class TuioMotionEvent(MotionEvent):
207      '''Abstraction for TUIO touches/fiducials. 
208   
209      Depending on the tracking software you use (e.g. Movid, CCV, etc.) and its 
210      TUIO implementation, the TuioMotionEvent object can support multiple 
211      profiles such as: 
212   
213          * Fiducial ID: profile name 'markerid', attribute ``.fid`` 
214          * Position: profile name 'pos', attributes ``.x``, ``.y`` 
215          * Angle: profile name 'angle', attribute ``.a`` 
216          * Velocity vector: profile name 'mov', attributes ``.X``, ``.Y`` 
217          * Rotation velocity: profile name 'rot', attribute ``.A`` 
218          * Motion acceleration: profile name 'motacc', attribute ``.m`` 
219          * Rotation acceleration: profile name 'rotacc', attribute ``.r`` 
220      '''
221      __attrs__ = ('a', 'b', 'c', 'X', 'Y', 'Z', 'A', 'B', 'C', 'm', 'r')
222  
223      def __init__(self, device, id, args):
224          super(TuioMotionEvent, self).__init__(device, id, args)
225          # Default argument for TUIO touches
226          self.a = 0.0
227          self.b = 0.0
228          self.c = 0.0
229          self.X = 0.0
230          self.Y = 0.0
231          self.Z = 0.0
232          self.A = 0.0
233          self.B = 0.0
234          self.C = 0.0
235          self.m = 0.0
236          self.r = 0.0
237  
238      angle = property(lambda self: self.a)
239      mot_accel = property(lambda self: self.m)
240      rot_accel = property(lambda self: self.r)
241      xmot = property(lambda self: self.X)
242      ymot = property(lambda self: self.Y)
243      zmot = property(lambda self: self.Z)

246  class Tuio2dCurMotionEvent(TuioMotionEvent):
247      '''A 2dCur TUIO touch.'''
248  
249      def __init__(self, device, id, args):
250          super(Tuio2dCurMotionEvent, self).__init__(device, id, args)
251  
252      def depack(self, args):
253          self.is_touch = True
254          if len(args) < 5:
255              self.sx, self.sy = list(map(float, args[0:2]))
256              self.profile = ('pos', )
257          elif len(args) == 5:
258              self.sx, self.sy, self.X, self.Y, self.m = list(map(float,
259                                                                  args[0:5]))
260              self.Y = -self.Y
261              self.profile = ('pos', 'mov', 'motacc')
262          else:
263              self.sx, self.sy, self.X, self.Y = list(map(float, args[0:4]))
264              self.m, width, height = list(map(float, args[4:7]))
265              self.Y = -self.Y
266              self.profile = ('pos', 'mov', 'motacc', 'shape')
267              if self.shape is None:
268                  self.shape = ShapeRect()
269              self.shape.width = width
270              self.shape.height = height
271          self.sy = 1 - self.sy
272          super(Tuio2dCurMotionEvent, self).depack(args)

275  class Tuio2dObjMotionEvent(TuioMotionEvent):
276      '''A 2dObj TUIO object. 
277      '''
278  
279      def __init__(self, device, id, args):
280          super(Tuio2dObjMotionEvent, self).__init__(device, id, args)
281  
282      def depack(self, args):
283          self.is_touch = True
284          if len(args) < 5:
285              self.sx, self.sy = args[0:2]
286              self.profile = ('pos', )
287          elif len(args) == 9:
288              self.fid, self.sx, self.sy, self.a, self.X, self.Y = args[:6]
289              self.A, self.m, self.r = args[6:9]
290              self.Y = -self.Y
291              self.profile = ('markerid', 'pos', 'angle', 'mov', 'rot',
292                              'motacc', 'rotacc')
293          else:
294              self.fid, self.sx, self.sy, self.a, self.X, self.Y = args[:6]
295              self.A, self.m, self.r, width, height = args[6:11]
296              self.Y = -self.Y
297              self.profile = ('markerid', 'pos', 'angle', 'mov', 'rot', 'rotacc',
298                              'acc', 'shape')
299              if self.shape is None:
300                  self.shape = ShapeRect()
301                  self.shape.width = width
302                  self.shape.height = height
303          self.sy = 1 - self.sy
304          super(Tuio2dObjMotionEvent, self).depack(args)
