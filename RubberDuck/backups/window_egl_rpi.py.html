24   
25   class WindowEglRpi(WindowBase):
26   
27       _rpi_dispmanx_id = int(environ.get("KIVY_BCM_DISPMANX_ID", "0"))
28       _rpi_dispmanx_layer = int(environ.get("KIVY_BCM_DISPMANX_LAYER", "0"))
29   
30       gl_backends_ignored = ['sdl2']
31   
32       def create_window(self):
33           bcm.host_init()
34   
35           w, h = bcm.graphics_get_display_size(self._rpi_dispmanx_id)
36           Logger.debug('Window: Actual display size: {}x{}'.format(
37               w, h))
38           self._size = w, h
39           self._create_window(w, h)
40           self._create_egl_context(self.win, 0)
41           super(WindowEglRpi, self).create_window()
42   
43       def _create_window(self, w, h):
44           dst = bcm.Rect(0, 0, w, h)
45           src = bcm.Rect(0, 0, w << 16, h << 16)
46           display = egl.bcm_display_open(self._rpi_dispmanx_id)
47           update = egl.bcm_update_start(0)
48           element = egl.bcm_element_add(
49               update, display, self._rpi_dispmanx_layer, dst, src)
50           self.win = egl.NativeWindow(element, w, h)
51           egl.bcm_update_submit_sync(update)
52   
53       def _create_egl_context(self, win, flags):
54           api = egl._constants.EGL_OPENGL_ES_API
55           c = egl._constants
56   
57           attribs = [
58               c.EGL_RED_SIZE, 8,
59               c.EGL_GREEN_SIZE, 8,
60               c.EGL_BLUE_SIZE, 8,
61               c.EGL_ALPHA_SIZE, 8,
62               c.EGL_DEPTH_SIZE, 16,
63               c.EGL_STENCIL_SIZE, 8,
64               c.EGL_SURFACE_TYPE, c.EGL_WINDOW_BIT,
65               c.EGL_NONE]
66   
67           attribs_context = [c.EGL_CONTEXT_CLIENT_VERSION, 2, c.EGL_NONE]
68   
69           display = egl.GetDisplay(c.EGL_DEFAULT_DISPLAY)
70           egl.Initialise(display)
71           egl.BindAPI(c.EGL_OPENGL_ES_API)
72           egl.GetConfigs(display)
73           config = egl.ChooseConfig(display, attribs, 1)[0]
74           surface = egl.CreateWindowSurface(display, config, win)
75           context = egl.CreateContext(display, config, None, attribs_context)
76           egl.MakeCurrent(display, surface, surface, context)
77   
78           self.egl_info = (display, surface, context)
79           egl.MakeCurrent(display, surface, surface, context)
80   
81       def close(self):
82           egl.Terminate(self.egl_info[0])
83   
84       def flip(self):
85           if not EventLoop.quit:
86               egl.SwapBuffers(self.egl_info[0], self.egl_info[1])
