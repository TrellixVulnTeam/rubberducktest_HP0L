1    ''' 
2    Retain Touch 
3    ============ 
4     
5    Reuse touch to counter lost finger behavior 
6    '''
7    
8    __all__ = ('InputPostprocRetainTouch', )
9    
10   from kivy.config import Config
11   from kivy.vector import Vector
12   import time
13   
14   
15   class InputPostprocRetainTouch(object):
16       ''' 
17       InputPostprocRetainTouch is a post-processor to delay the 'up' event of a 
18       touch, to reuse it under certains conditions. This module is designed to 
19       prevent lost finger touches on some hardware/setups. 
20    
21       Retain touch can be configured in the Kivy config file:: 
22    
23           [postproc] 
24               retain_time = 100 
25               retain_distance = 50 
26    
27       The distance parameter is in the range 0-1000 and time is in milliseconds. 
28       '''
29   
30       def __init__(self):
31           self.timeout = Config.getint('postproc', 'retain_time') / 1000.0
32           self.distance = Config.getint('postproc', 'retain_distance') / 1000.0
33           self._available = []
34           self._links = {}
35   
36       def process(self, events):
37           # check if module is disabled
38           if self.timeout == 0:
39               return events
40   
41           d = time.time()
42           for etype, touch in events[:]:
43               if not touch.is_touch:
44                   continue
45               if etype == 'end':
46                   events.remove((etype, touch))
47                   if touch.uid in self._links:
48                       selection = self._links[touch.uid]
49                       selection.ud.__pp_retain_time__ = d
50                       self._available.append(selection)
51                       del self._links[touch.uid]
52                   else:
53                       touch.ud.__pp_retain_time__ = d
54                       self._available.append(touch)
55               elif etype == 'update':
56                   if touch.uid in self._links:
57                       selection = self._links[touch.uid]
58                       selection.x = touch.x
59                       selection.y = touch.y
60                       selection.sx = touch.sx
61                       selection.sy = touch.sy
62                       events.remove((etype, touch))
63                       events.append((etype, selection))
64                   else:
65                       pass
66               elif etype == 'begin':
67                   # new touch, found the nearest one
68                   selection = None
69                   selection_distance = 99999
70                   for touch2 in self._available:
71                       touch_distance = Vector(touch2.spos).distance(touch.spos)
72                       if touch_distance > self.distance:
73                           continue
74                       if touch2.__class__ != touch.__class__:
75                           continue
76                       if touch_distance < selection_distance:
77                           # eligible for continuation
78                           selection_distance = touch_distance
79                           selection = touch2
80                   if selection is None:
81                       continue
82   
83                   self._links[touch.uid] = selection
84                   self._available.remove(selection)
85                   events.remove((etype, touch))
86   
87           for touch in self._available[:]:
88               t = touch.ud.__pp_retain_time__
89               if d - t > self.timeout:
90                   self._available.remove(touch)
91                   events.append(('end', touch))
92   
93           return events
