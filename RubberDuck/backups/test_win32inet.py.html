11   class CookieTests(unittest.TestCase):
12       def testCookies(self):
13           data = "TestData=Test"
14           InternetSetCookie("http://www.python.org", None, data)
15           got = InternetGetCookie("http://www.python.org", None)
16           # handle that there might already be cookies for the domain.
17           bits = map(lambda x: x.strip(), got.split(";"))
18           self.assertTrue(data in bits)

23               self.fail("expected win32 exception")

47       def testPythonDotOrg(self):
48           hdl = InternetOpenUrl(
49               self.hi, "http://www.python.org", None, INTERNET_FLAG_EXISTING_CONNECT
50           )
51           chunks = []
52           while 1:
53               chunk = InternetReadFile(hdl, 1024)
54               if not chunk:
55                   break
56               chunks.append(chunk)
57           data = str2bytes("").join(chunks)
58           assert data.find(str2bytes("Python")) > 0, repr(
59               data
60           )  # This must appear somewhere on the main page!
