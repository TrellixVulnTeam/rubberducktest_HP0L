101          def _event_type(self):
102              if self.flags & TOUCHEVENTF_MOVE:
103                  return 'update'
104              if self.flags & TOUCHEVENTF_DOWN:
105                  return 'begin'
106              if self.flags & TOUCHEVENTF_UP:
107                  return 'end'
108          event_type = property(_event_type)

110      def SetWindowLong_WndProc_wrapper_generator(func):
111          def _closure(hWnd, wndProc):
112              oldAddr = func(hWnd, GWL_WNDPROC, cast(wndProc, c_void_p).value)
113              return cast(c_void_p(oldAddr), WNDPROC)
114  
115          return _closure
