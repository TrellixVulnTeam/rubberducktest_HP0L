1    """Python.Dictionary COM Server. 
2     
3    This module implements a simple COM server that acts much like a Python 
4    dictionary or as a standard string-keyed VB Collection.  The keys of 
5    the dictionary are strings and are case-insensitive. 
6     
7    It uses a highly customized policy to fine-tune the behavior exposed to 
8    the COM client. 
9     
10   The object exposes the following properties: 
11    
12       int Count                       (readonly) 
13       VARIANT Item(BSTR key)          (propget for Item) 
14       Item(BSTR key, VARIANT value)   (propput for Item) 
15    
16       Note that 'Item' is the default property, so the following forms of 
17       VB code are acceptable: 
18    
19           set ob = CreateObject("Python.Dictionary") 
20           ob("hello") = "there" 
21           ob.Item("hi") = ob("HELLO") 
22    
23   All keys are defined, returning VT_NULL (None) if a value has not been 
24   stored.  To delete a key, simply assign VT_NULL to the key. 
25    
26   The object responds to the _NewEnum method by returning an enumerator over 
27   the dictionary's keys. This allows for the following type of VB code: 
28    
29       for each name in ob 
30           debug.print name, ob(name) 
31       next 
32   """
33   
34   import pythoncom
35   from win32com.server import util, policy
36   from win32com.server.exception import COMException
37   import winerror
38   import types
39   import pywintypes
40   
41   from pythoncom import DISPATCH_METHOD, DISPATCH_PROPERTYGET
42   from winerror import S_OK
43   
44   
45   class DictionaryPolicy(policy.BasicWrapPolicy):
46       ### BasicWrapPolicy looks for this
47       _com_interfaces_ = []
48   
49       ### BasicWrapPolicy looks for this
50       _name_to_dispid_ = {
51           "item": pythoncom.DISPID_VALUE,
52           "_newenum": pythoncom.DISPID_NEWENUM,
53           "count": 1,
54       }
55   
56       ### Auto-Registration process looks for these...
57       _reg_desc_ = "Python Dictionary"
58       _reg_clsid_ = "{39b61048-c755-11d0-86fa-00c04fc2e03e}"
59       _reg_progid_ = "Python.Dictionary"
60       _reg_verprogid_ = "Python.Dictionary.1"
61       _reg_policy_spec_ = "win32com.servers.dictionary.DictionaryPolicy"
62   
63       def _CreateInstance_(self, clsid, reqIID):
64           self._wrap_({})
65           return pythoncom.WrapObject(self, reqIID)
66   
67       def _wrap_(self, ob):
68           self._obj_ = ob  # ob should be a dictionary
69   
70       def _invokeex_(self, dispid, lcid, wFlags, args, kwargs, serviceProvider):
71           if dispid == 0:  # item
72               l = len(args)
73               if l < 1:
74                   raise COMException(
75                       desc="not enough parameters", scode=winerror.DISP_E_BADPARAMCOUNT
76                   )
77   
78               key = args[0]
79               if type(key) not in [str, str]:
80                   ### the nArgErr thing should be 0-based, not reversed... sigh
81                   raise COMException(
82                       desc="Key must be a string", scode=winerror.DISP_E_TYPEMISMATCH
83                   )
84   
85               key = key.lower()
86   
87               if wFlags & (DISPATCH_METHOD | DISPATCH_PROPERTYGET):
88                   if l > 1:
89                       raise COMException(scode=winerror.DISP_E_BADPARAMCOUNT)
90                   try:
91                       return self._obj_[key]
92                   except KeyError:
93                       return None  # unknown keys return None (VT_NULL)
94   
95               if l != 2:
96                   raise COMException(scode=winerror.DISP_E_BADPARAMCOUNT)
97               if args[1] is None:
98                   # delete a key when None is assigned to it
99                   try:
100                      del self._obj_[key]
101                  except KeyError:
102                      pass
103              else:
104                  self._obj_[key] = args[1]
105              return S_OK
106  
107          if dispid == 1:  # count
108              if not wFlags & DISPATCH_PROPERTYGET:
109                  raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)  # not found
110              if len(args) != 0:
111                  raise COMException(scode=winerror.DISP_E_BADPARAMCOUNT)
112              return len(self._obj_)
113  
114          if dispid == pythoncom.DISPID_NEWENUM:
115              return util.NewEnum(list(self._obj_.keys()))
116  
117          raise COMException(scode=winerror.DISP_E_MEMBERNOTFOUND)
118  
119      def _getidsofnames_(self, names, lcid):
120          ### this is a copy of MappedWrapPolicy._getidsofnames_ ...
121  
122          name = names[0].lower()
123          try:
124              return (self._name_to_dispid_[name],)
125          except KeyError:
126              raise COMException(
127                  scode=winerror.DISP_E_MEMBERNOTFOUND, desc="Member not found"
128              )
129  
130  
131  def Register():
132      from win32com.server.register import UseCommandLine
133  
134      return UseCommandLine(DictionaryPolicy)
135  
136  
137  if __name__ == "__main__":
138      Register()
