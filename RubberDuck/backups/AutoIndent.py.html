72       # usetabs true  -> literal tab characters are used by indent and
73       #                  dedent cmds, possibly mixed with spaces if
74       #                  indentwidth is not a multiple of tabwidth
75       #         false -> tab characters are converted to spaces by indent
76       #                  and dedent cmds, and ditto TAB keystrokes
77       # indentwidth is the number of characters per logical indent level.
78       # tabwidth is the display width of a literal tab character.

92       # first start num_context_lines[0] lines back, then

132      def smart_backspace_event(self, event):
133          text = self.text
134          first, last = self.editwin.get_selection_indices()
135          if first and last:
136              text.delete(first, last)
137              text.mark_set("insert", first)
138              return "break"
139          # Delete whitespace left, until hitting a real char or closest
140          # preceding virtual tab stop.
141          chars = text.get("insert linestart", "insert")
142          if chars == "":
143              if text.compare("insert", ">", "1.0"):
144                  # easy: delete preceding newline
145                  text.delete("insert-1c")
146              else:
147                  text.bell()  # at start of buffer
148              return "break"
149          if chars[-1] not in " \t":
150              # easy: delete preceding real char
151              text.delete("insert-1c")
152              return "break"
153          # Ick.  It may require *inserting* spaces if we back up over a
154          # tab character!  This is written to be clear, not fast.
155          have = len(chars.expandtabs(self.tabwidth))
156          assert have > 0
157          want = int((have - 1) / self.indentwidth) * self.indentwidth
158          ncharsdeleted = 0
159          while 1:
160              chars = chars[:-1]
161              ncharsdeleted = ncharsdeleted + 1
162              have = len(chars.expandtabs(self.tabwidth))
163              if have <= want or chars[-1] not in " \t":
164                  break
165          text.undo_block_start()
166          text.delete("insert-%dc" % ncharsdeleted, "insert")
167          if have < want:
168              text.insert("insert", " " * (want - have))
169          text.undo_block_stop()
170          return "break"

172      def smart_indent_event(self, event):
173          # if intraline selection:
174          #     delete it
175          # elif multiline selection:
176          #     do indent-region & return
177          # indent one level
178          text = self.text
179          first, last = self.editwin.get_selection_indices()
180          text.undo_block_start()
181          try:
182              if first and last:
183                  if index2line(first) != index2line(last):
184                      return self.indent_region_event(event)
185                  text.delete(first, last)
186                  text.mark_set("insert", first)
187              prefix = text.get("insert linestart", "insert")
188              raw, effective = classifyws(prefix, self.tabwidth)
189              if raw == len(prefix):
190                  # only whitespace to the left
191                  self.reindent_to(effective + self.indentwidth)
192              else:
193                  if self.usetabs:
194                      pad = "\t"
195                  else:
196                      effective = len(prefix.expandtabs(self.tabwidth))
197                      n = self.indentwidth
198                      pad = " " * (n - effective % n)
199                  text.insert("insert", pad)
200              text.see("insert")
201              return "break"
202          finally:
203              text.undo_block_stop()

205      def newline_and_indent_event(self, event):
206          text = self.text
207          first, last = self.editwin.get_selection_indices()
208          text.undo_block_start()
209          try:
210              if first and last:
211                  text.delete(first, last)
212                  text.mark_set("insert", first)
213              line = text.get("insert linestart", "insert")
214              i, n = 0, len(line)
215              while i < n and line[i] in " \t":
216                  i = i + 1
217              if i == n:
218                  # the cursor is in or at leading indentation; just inject
219                  # an empty line at the start and strip space from current line
220                  text.delete("insert - %d chars" % i, "insert")
221                  text.insert("insert linestart", "\n")
222                  return "break"
223              indent = line[:i]
224              # strip whitespace before insert point
225              i = 0
226              while line and line[-1] in " \t":
227                  line = line[:-1]
228                  i = i + 1
229              if i:
230                  text.delete("insert - %d chars" % i, "insert")
231              # strip whitespace after insert point
232              while text.get("insert") in " \t":
233                  text.delete("insert")
234              # start new line
235              text.insert("insert", "\n")
236  
237              # adjust indentation for continuations and block
238              # open/close first need to find the last stmt
239              lno = index2line(text.index("insert"))
240              y = PyParse.Parser(self.indentwidth, self.tabwidth)
241              for context in self.num_context_lines:
242                  startat = max(lno - context, 1)
243                  startatindex = repr(startat) + ".0"
244                  rawtext = text.get(startatindex, "insert")
245                  y.set_str(rawtext)
246                  bod = y.find_good_parse_start(
247                      self.context_use_ps1, self._build_char_in_string_func(startatindex)
248                  )
249                  if bod is not None or startat == 1:
250                      break
251              y.set_lo(bod or 0)
252              c = y.get_continuation_type()
253              if c != PyParse.C_NONE:
254                  # The current stmt hasn't ended yet.
255                  if c == PyParse.C_STRING:
256                      # inside a string; just mimic the current indent
257                      text.insert("insert", indent)
258                  elif c == PyParse.C_BRACKET:
259                      # line up with the first (if any) element of the
260                      # last open bracket structure; else indent one
261                      # level beyond the indent of the line with the
262                      # last open bracket
263                      self.reindent_to(y.compute_bracket_indent())
264                  elif c == PyParse.C_BACKSLASH:
265                      # if more than one line in this stmt already, just
266                      # mimic the current indent; else if initial line
267                      # has a start on an assignment stmt, indent to
268                      # beyond leftmost =; else to beyond first chunk of
269                      # non-whitespace on initial line
270                      if y.get_num_lines_in_stmt() > 1:
271                          text.insert("insert", indent)
272                      else:
273                          self.reindent_to(y.compute_backslash_indent())
274                  else:
275                      assert 0, "bogus continuation type " + repr(c)
276                  return "break"
277  
278              # This line starts a brand new stmt; indent relative to
279              # indentation of initial line of closest preceding
280              # interesting stmt.
281              indent = y.get_base_indent_string()
282              text.insert("insert", indent)
283              if y.is_block_opener():
284                  self.smart_indent_event(event)
285              elif indent and y.is_block_closer():
286                  self.smart_backspace_event(event)
287              return "break"
288          finally:
289              text.see("insert")
290              text.undo_block_stop()

305      def indent_region_event(self, event):
306          head, tail, chars, lines = self.get_region()
307          for pos in range(len(lines)):
308              line = lines[pos]
309              if line:
310                  raw, effective = classifyws(line, self.tabwidth)
311                  effective = effective + self.indentwidth
312                  lines[pos] = self._make_blanks(effective) + line[raw:]
313          self.set_region(head, tail, chars, lines)
314          return "break"

335          head, tail, chars, lines = self.get_region()
336          for pos in range(len(lines)):
337              line = lines[pos]
338              if not line:
339                  continue
340              if line[:2] == "##":
341                  line = line[2:]
342              elif line[:1] == "#":
343                  line = line[1:]
344              lines[pos] = line
345          self.set_region(head, tail, chars, lines)

347      def tabify_region_event(self, event):
348          head, tail, chars, lines = self.get_region()
349          tabwidth = self._asktabwidth()
350          for pos in range(len(lines)):
351              line = lines[pos]
352              if line:
353                  raw, effective = classifyws(line, tabwidth)
354                  ntabs, nspaces = divmod(effective, tabwidth)
355                  lines[pos] = "\t" * ntabs + " " * nspaces + line[raw:]
356          self.set_region(head, tail, chars, lines)

359          head, tail, chars, lines = self.get_region()
360          tabwidth = self._asktabwidth()

395      def get_region(self):
396          text = self.text
397          first, last = self.editwin.get_selection_indices()
398          if first and last:
399              head = text.index(first + " linestart")
400              tail = text.index(last + "-1c lineend +1c")
401          else:
402              head = text.index("insert linestart")
403              tail = text.index("insert lineend +1c")
404          chars = text.get(head, tail)
405          lines = chars.split("\n")
406          return head, tail, chars, lines

408      def set_region(self, head, tail, chars, lines):
409          text = self.text
410          newchars = "\n".join(lines)
411          if newchars == chars:
412              text.bell()
413              return
414          text.tag_remove("sel", "1.0", "end")
415          text.mark_set("insert", head)
416          text.undo_block_start()
417          text.delete(head, tail)
418          text.insert(head, newchars)
419          text.undo_block_stop()
420          text.tag_add("sel", head, "insert")

422      # Make string that displays as n leading blanks.
423  
424      def _make_blanks(self, n):
425          if self.usetabs:
426              ntabs, nspaces = divmod(n, self.tabwidth)
427              return "\t" * ntabs + " " * nspaces
428          else:
429              return " " * n

431      # Delete from beginning of line to insert point, then reinsert
432      # column logical (meaning use tabs if appropriate) spaces.
433  
434      def reindent_to(self, column):
435          text = self.text
436          text.undo_block_start()
437          if text.compare("insert linestart", "!=", "insert"):
438              text.delete("insert linestart", "insert")
439          if column:
440              text.insert("insert", self._make_blanks(column))
441          text.undo_block_stop()

444          return (

448                  parent=self.text,

456      # Guess indentwidth from text content.
457      # Return guessed indentwidth.  This should not be believed unless
458      # it's in a reasonable range (e.g., it will be 0 if no indented
459      # blocks are found).
460  
461      def guess_indent(self):
462          opener, indented = IndentSearcher(self.text, self.tabwidth).run()
463          if opener and indented:
464              raw, indentsmall = classifyws(opener, self.tabwidth)
465              raw, indentlarge = classifyws(indented, self.tabwidth)
466          else:
467              indentsmall = indentlarge = 0
468          return indentlarge - indentsmall

476  # Look at the leading whitespace in s.
477  # Return pair (# of leading ws characters,
478  #              effective # of leading blanks after expanding
479  #              tabs to width tabwidth)
480  
481  
482  def classifyws(s, tabwidth):
483      raw = effective = 0
484      for ch in s:
485          if ch == " ":
486              raw = raw + 1
487              effective = effective + 1
488          elif ch == "\t":
489              raw = raw + 1
490              effective = (effective // tabwidth + 1) * tabwidth
491          else:
492              break
493      return raw, effective

496  class IndentSearcher:
497  
498      # .run() chews over the Text widget, looking for a block opener
499      # and the stmt following it.  Returns a pair,
500      #     (line containing block opener, line containing stmt)
501      # Either or both may be None.
502  
503      def __init__(self, text, tabwidth):
504          self.text = text
505          self.tabwidth = tabwidth
506          self.i = self.finished = 0
507          self.blkopenline = self.indentedline = None
508  
509      def readline(self):
510          if self.finished:
511              val = ""
512          else:
513              i = self.i = self.i + 1
514              mark = repr(i) + ".0"
515              if self.text.compare(mark, ">=", "end"):
516                  val = ""
517              else:
518                  val = self.text.get(mark, mark + " lineend+1c")
519          # hrm - not sure this is correct in py3k - the source code may have
520          # an encoding declared, but the data will *always* be in
521          # default_scintilla_encoding - so if anyone looks at the encoding decl
522          # in the source they will be wrong.  I think.  Maybe.  Or something...
523          return val.encode(default_scintilla_encoding)
524  
525      def run(self):
526          OPENERS = ("class", "def", "for", "if", "try", "while")
527          INDENT = tokenize.INDENT
528          NAME = tokenize.NAME
529  
530          save_tabsize = tokenize.tabsize
531          tokenize.tabsize = self.tabwidth
532          try:
533              try:
534                  for (typ, token, start, end, line) in token_generator(self.readline):
535                      if typ == NAME and token in OPENERS:
536                          self.blkopenline = line
537                      elif typ == INDENT and self.blkopenline:
538                          self.indentedline = line
539                          break
540  
541              except (tokenize.TokenError, IndentationError):
542                  # since we cut off the tokenizer early, we can trigger
543                  # spurious errors
544                  pass
545          finally:
546              tokenize.tabsize = save_tabsize
547          return self.blkopenline, self.indentedline
