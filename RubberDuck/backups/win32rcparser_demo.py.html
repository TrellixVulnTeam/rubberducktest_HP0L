19   class DemoWindow:
20       def __init__(self, dlg_template):
21           self.dlg_template = dlg_template
22   
23       def CreateWindow(self):
24           self._DoCreate(win32gui.CreateDialogIndirect)
25   
26       def DoModal(self):
27           return self._DoCreate(win32gui.DialogBoxIndirect)
28   
29       def _DoCreate(self, fn):
30           message_map = {
31               win32con.WM_INITDIALOG: self.OnInitDialog,
32               win32con.WM_CLOSE: self.OnClose,
33               win32con.WM_DESTROY: self.OnDestroy,
34               win32con.WM_COMMAND: self.OnCommand,
35           }
36           return fn(0, self.dlg_template, 0, message_map)
37   
38       def OnInitDialog(self, hwnd, msg, wparam, lparam):
39           self.hwnd = hwnd
40           # centre the dialog
41           desktop = win32gui.GetDesktopWindow()
42           l, t, r, b = win32gui.GetWindowRect(self.hwnd)
43           dt_l, dt_t, dt_r, dt_b = win32gui.GetWindowRect(desktop)
44           centre_x, centre_y = win32gui.ClientToScreen(
45               desktop, ((dt_r - dt_l) // 2, (dt_b - dt_t) // 2)
46           )
47           win32gui.MoveWindow(
48               hwnd, centre_x - (r // 2), centre_y - (b // 2), r - l, b - t, 0
49           )
50   
51       def OnCommand(self, hwnd, msg, wparam, lparam):
52           # Needed to make OK/Cancel work - no other controls are handled.
53           id = win32api.LOWORD(wparam)
54           if id in [win32con.IDOK, win32con.IDCANCEL]:
55               win32gui.EndDialog(hwnd, id)
56   
57       def OnClose(self, hwnd, msg, wparam, lparam):
58           win32gui.EndDialog(hwnd, 0)
59   
60       def OnDestroy(self, hwnd, msg, wparam, lparam):
61           pass
62

64   def DemoModal():
65       # Load the .rc file.
66       resources = win32rcparser.Parse(g_rcname)
67       for id, ddef in resources.dialogs.items():
68           print("Displaying dialog", id)
69           w = DemoWindow(ddef)
70           w.DoModal()
