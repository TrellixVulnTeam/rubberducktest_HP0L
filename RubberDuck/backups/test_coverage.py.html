14       cov = coverage.Coverage(source=[os.path.dirname(__file__)])
15       cov.start()
16   
17       fname = os.path.join(os.path.dirname(__file__), 'coverage_lang.kv')
18       try:
19           widget = Builder.load_file(fname)
20       finally:
21           cov.stop()
