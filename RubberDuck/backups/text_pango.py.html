text_pango.py 
1    ''' 
2    Pango text provider 
3    =================== 
4     
5    .. versionadded:: 1.11.0 
6     
7    .. warning:: 
8        The low-level Pango API is experimental, and subject to change without 
9        notice for as long as this warning is present. 
10    
11    
12   Installation 
13   ------------ 
14    
15   1. Install pangoft2 (`apt install libfreetype6-dev libpango1.0-dev 
16      libpangoft2-1.0-0`) or ensure it is available in pkg-config 
17   2. Recompile kivy. Check that pangoft2 is found `use_pangoft2 = 1` 
18   3. Test it! Enforce the text core renderer to pango using environment variable: 
19      `export KIVY_TEXT=pango` 
20    
21   This has been tested on OSX and Linux, Python 3.6. 
22    
23    
24   Font context types for FontConfig+FreeType2 backend 
25   --------------------------------------------------- 
26    
27   * `system://` - `FcInitLoadConfigAndFonts()` 
28   * `systemconfig://` - `FcInitLoadConfig()` 
29   * `directory://<PATH>` - `FcInitLoadConfig()` + `FcAppFontAddDir()` 
30   * `fontconfig://<PATH>` - `FcConfigCreate()` + `FcConfigParseAndLoad()` 
31   * Any other context name - `FcConfigCreate()` 
32    
33    
34   Low-level Pango access 
35   ---------------------- 
36    
37   Since Kivy currently does its own text layout, the Label and TextInput widgets 
38   do not take full advantage of Pango. For example, line breaks do not take 
39   language/script into account, and switching alignment per paragraph (for bi- 
40   directional text) is not supported. For advanced i18n requirements, we provide 
41   a simple wrapper around PangoLayout that you can use to render text. 
42    
43   * https://developer.gnome.org/pango/1.40/pango-Layout-Objects.html 
44   * https://developer.gnome.org/pango/1.40/PangoMarkupFormat.html 
45   * See the `kivy/core/text/_text_pango.pyx` file @ `cdef class KivyPangoLayout` 
46     for more information. Not all features of PangoLayout are implemented. 
47    
48   .. python:: 
49       from kivy.core.window import Window  # OpenGL must be initialized 
50       from kivy.core.text._text_pango import KivyPangoLayout 
51       layout = KivyPangoLayout('system://') 
52       layout.set_markup('<span font="20">Hello <b>World!</b></span>') 
53       tex = layout.render_as_Texture() 
54    
55    
56   Known limitations 
57   ----------------- 
58    
59   * Pango versions older than v1.38 has not been tested. It may work on 
60     some systems with older pango and newer FontConfig/FreeType2 versions. 
61   * Kivy's text layout is used, not Pango. This means we do not use Pango's 
62     line-breaking feature (which is superior to Kivy's), and we can't use 
63     Pango's bidirectional cursor helpers in TextInput. 
64   * Font family collissions can happen. For example, if you use a `system://` 
65     context and add a custom `Arial.ttf`, using `arial` as the `font_family` 
66     may or may not draw with your custom font (depending on whether or not 
67     there is already a system-wide "arial" font installed) 
68   * Rendering is inefficient; the normal way to integrate Pango would be 
69     using a dedicated PangoLayout per widget. This is not currently practical 
70     due to missing abstractions in Kivy core (in the current implementation, 
71     we have a dedicated PangoLayout *per font context,* which is rendered 
72     once for each LayoutWord) 
73   '''

96   class LabelPango(LabelBase):
97   
98       _font_family_support = True
99   
100      def __init__(self, *largs, **kwargs):
101          self.get_extents = MethodType(kpango_get_extents, self)
102          self.get_ascent = MethodType(kpango_get_ascent, self)
103          self.get_descent = MethodType(kpango_get_descent, self)
104          super(LabelPango, self).__init__(*largs, **kwargs)
105  
106      find_base_direction = staticmethod(kpango_find_base_dir)
107  
108      def _render_begin(self):
109          self._rdr = KivyPangoRenderer(*self._size)
110  
111      def _render_text(self, text, x, y):
112          self._rdr.render(self, text, x, y)
113  
114      def _render_end(self):
115          imgdata = self._rdr.get_ImageData()
116          del self._rdr
117          return imgdata

120  class PangoFontContextManager(FontContextManagerBase):
121      create = staticmethod(kpango_font_context_create)
122      exists = staticmethod(kpango_font_context_exists)
123      destroy = staticmethod(kpango_font_context_destroy)
124      list = staticmethod(kpango_font_context_list)
125      list_families = staticmethod(kpango_font_context_list_families)
126      list_custom = staticmethod(kpango_font_context_list_custom)
127  
128      @staticmethod
129      def add_font(font_context, filename, autocreate=True, family=None):
130          if not autocreate and not PangoFontContextManager.exists(font_context):
131              raise Exception("FontContextManager: Attempt to add font file "
132                              "'{}' to non-existing context '{}' without "
133                              "autocreate.".format(filename, font_context))
134          if not filename:
135              raise Exception("FontContextManager: Cannot add empty font file")
136          if not isfile(filename):
137              filename = resource_find(filename)
138          if not isfile(filename):
139              if not filename.endswith('.ttf'):
140                  filename = resource_find('{}.ttf'.format(filename))
141          if filename and isfile(filename):
142              return kpango_font_context_add_font(font_context, filename)
143          raise Exception("FontContextManager: Attempt to add non-existant "
144                          "font file: '{}' to context '{}'"
145                          .format(filename, font_context))
