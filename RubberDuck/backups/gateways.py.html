415  class DebugDocumentTextConnectServer:
416      _public_methods_ = (
417          win32com.server.connect.IConnectionPointContainer_methods
418          + win32com.server.connect.IConnectionPoint_methods
419      )
420      _com_interfaces_ = [
421          pythoncom.IID_IConnectionPoint,
422          pythoncom.IID_IConnectionPointContainer,
423      ]
424      # IConnectionPoint interfaces
425      def __init__(self):
426          self.cookieNo = -1
427          self.connections = {}
428  
429      def EnumConnections(self):
430          RaiseNotImpl("EnumConnections")
431  
432      def GetConnectionInterface(self):
433          RaiseNotImpl("GetConnectionInterface")
434  
435      def GetConnectionPointContainer(self):
436          return _wrap(self)
437  
438      def Advise(self, pUnk):
439          # Creates a connection to the client.  Simply allocate a new cookie,
440          # find the clients interface, and store it in a dictionary.
441          interface = pUnk.QueryInterface(axdebug.IID_IDebugDocumentTextEvents, 1)
442          self.cookieNo = self.cookieNo + 1
443          self.connections[self.cookieNo] = interface
444          return self.cookieNo
445  
446      def Unadvise(self, cookie):
447          # Destroy a connection - simply delete interface from the map.
448          try:
449              del self.connections[cookie]
450          except KeyError:
451              return Exception(scode=winerror.E_UNEXPECTED)
452  
453      # IConnectionPointContainer interfaces
454      def EnumConnectionPoints(self):
455          RaiseNotImpl("EnumConnectionPoints")
456  
457      def FindConnectionPoint(self, iid):
458          # Find a connection we support.  Only support the single event interface.
459          if iid == axdebug.IID_IDebugDocumentTextEvents:
460              return _wrap(self)
461          raise Exception(scode=winerror.E_NOINTERFACE)  # ??
