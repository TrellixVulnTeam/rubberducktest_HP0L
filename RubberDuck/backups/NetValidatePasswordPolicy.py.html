NetValidatePasswordPolicy.py 
1    """A demo of using win32net.NetValidatePasswordPolicy. 
2     
3    Example usage: 
4     
5    % NetValidatePasswordPolicy.py --password=foo change 
6    which might return: 
7     
8    > Result of 'change' validation is 0: The operation completed successfully. 
9     
10   or depending on the policy: 
11    
12   > Result of 'change' validation is 2245: The password does not meet the 
13   > password policy requirements. Check the minimum password length, 
14   > password complexity and password history requirements. 
15    
16   Adding --user doesn't seem to change the output (even the PasswordLastSet seen 
17   when '-f' is used doesn't depend on the username), but theoretically it will 
18   also check the password history for the specified user. 
19    
20   % NetValidatePasswordPolicy.py auth 
21    
22   which always (with and without '-m') seems to return: 
23    
24   > Result of 'auth' validation is 2701: Password must change at next logon 
25   """
26   
27   import sys
28   import win32api
29   import win32net, win32netcon
30   
31   import optparse
32   from pprint import pprint
33   
34   
35   def main():
36       parser = optparse.OptionParser(
37           "%prog [options] auth|change ...",
38           description="A win32net.NetValidatePasswordPolicy demo.",
39       )
40   
41       parser.add_option(
42           "-u",
43           "--username",
44           action="store",
45           help="The username to pass to the function (only for the " "change command",
46       )
47   
48       parser.add_option(
49           "-p",
50           "--password",
51           action="store",
52           help="The clear-text password to pass to the function "
53           "(only for the 'change' command)",
54       )
55   
56       parser.add_option(
57           "-m",
58           "--password-matched",
59           action="store_false",
60           default=True,
61           help="Used to specify the password does NOT match (ie, "
62           "uses False for the PasswordMatch/PasswordMatched "
63           "arg, both 'auth' and 'change' commands)",
64       )
65   
66       parser.add_option(
67           "-s",
68           "--server",
69           action="store",
70           help="The name of the server to execute the command on",
71       )
72   
73       parser.add_option(
74           "-f",
75           "--show_fields",
76           action="store_true",
77           default=False,
78           help="Print the NET_VALIDATE_PERSISTED_FIELDS returned",
79       )
80   
81       options, args = parser.parse_args()
82   
83       if not args:
84           args = ["auth"]
85   
86       for arg in args:
87           if arg == "auth":
88               input = {
89                   "PasswordMatched": options.password_matched,
90               }
91               val_type = win32netcon.NetValidateAuthentication
92           elif arg == "change":
93               input = {
94                   "ClearPassword": options.password,
95                   "PasswordMatch": options.password_matched,
96                   "UserAccountName": options.username,
97               }
98               val_type = win32netcon.NetValidatePasswordChange
99           else:
100              parser.error("Invalid arg - must be 'auth' or 'change'")
101  
102          try:
103              fields, status = win32net.NetValidatePasswordPolicy(
104                  options.server, None, val_type, input
105              )
106          except NotImplementedError:
107              print("NetValidatePasswordPolicy not implemented on this platform.")
108              return 1
109          except win32net.error as exc:
110              print("NetValidatePasswordPolicy failed: ", exc)
111              return 1
112  
113          if options.show_fields:
114              print("NET_VALIDATE_PERSISTED_FIELDS fields:")
115              pprint(fields)
116  
117          print(
118              "Result of %r validation is %d: %s"
119              % (arg, status, win32api.FormatMessage(status).strip())
120          )
121  
122      return 0
123  
124  
125  if __name__ == "__main__":
126      sys.exit(main())
