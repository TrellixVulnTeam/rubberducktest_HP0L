35       '.'.join(map(str, get_gst_version()))))

38   def _on_gstplayer_buffer(video, width, height, data):
39       video = video()
40       # if we still receive the video but no more player, remove it.
41       if not video:
42           return
43       with video._buffer_lock:
44           video._buffer = (width, height, data)
45   
46   
47   def _on_gstplayer_message(mtype, message):
48       if mtype == 'error':
49           Logger.error('VideoGstplayer: {}'.format(message))
50       elif mtype == 'warning':
51           Logger.warning('VideoGstplayer: {}'.format(message))
52       elif mtype == 'info':
53           Logger.info('VideoGstplayer: {}'.format(message))
54   
55   
56   class VideoGstplayer(VideoBase):
57   
58       def __init__(self, **kwargs):
59           self.player = None
60           self._buffer = None
61           self._buffer_lock = Lock()
62           super(VideoGstplayer, self).__init__(**kwargs)
63   
64       def _on_gst_eos_sync(self):
65           Clock.schedule_once(self._do_eos, 0)
66   
67       def load(self):
68           Logger.debug('VideoGstplayer: Load <{}>'.format(self._filename))
69           uri = self._get_uri()
70           wk_self = ref(self)
71           self.player_callback = partial(_on_gstplayer_buffer, wk_self)
72           self.player = GstPlayer(uri, self.player_callback,
73                                   self._on_gst_eos_sync, _on_gstplayer_message)
74           self.player.load()
75   
76       def unload(self):
77           if self.player:
78               self.player.unload()
79               self.player = None
80           with self._buffer_lock:
81               self._buffer = None
82           self._texture = None
83   
84       def stop(self):
85           super(VideoGstplayer, self).stop()
86           self.player.stop()
87   
88       def pause(self):
89           super(VideoGstplayer, self).pause()
90           self.player.pause()
91   
92       def play(self):
93           super(VideoGstplayer, self).play()
94           self.player.set_volume(self.volume)
95           self.player.play()
96   
97       def seek(self, percent, precise=True):
98           self.player.seek(percent)
99   
100      def _get_position(self):
101          return self.player.get_position()
102  
103      def _get_duration(self):
104          return self.player.get_duration()
105  
106      def _set_volume(self, value):
107          self._volume = value
108          if self.player:
109              self.player.set_volume(self._volume)
110  
111      def _update(self, dt):
112          buf = None
113          with self._buffer_lock:
114              buf = self._buffer
115              self._buffer = None
116          if buf is not None:
117              self._update_texture(buf)
118              self.dispatch('on_frame')
119  
120      def _update_texture(self, buf):
121          width, height, data = buf
122  
123          # texture is not allocated yet, create it first
124          if not self._texture:
125              self._texture = Texture.create(size=(width, height),
126                                             colorfmt='rgb')
127              self._texture.flip_vertical()
128              self.dispatch('on_load')
129  
130          if self._texture:
131              self._texture.blit_buffer(
132                  data, size=(width, height), colorfmt='rgb')
133  
134      def _get_uri(self):
135          uri = self.filename
136          if not uri:
137              return
138          if '://' not in uri:
139              uri = 'file:' + pathname2url(realpath(uri))
140          return uri
