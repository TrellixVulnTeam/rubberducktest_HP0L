1    # Magic utility that "redirects" to pywintypesxx.dll
2    import importlib.util, importlib.machinery, sys, os
3    
4    
5    def __import_pywin32_system_module__(modname, globs):
6        # This has been through a number of iterations.  The problem: how to
7        # locate pywintypesXX.dll when it may be in a number of places, and how
8        # to avoid ever loading it twice.  This problem is compounded by the
9        # fact that the "right" way to do this requires win32api, but this
10       # itself requires pywintypesXX.
11       # And the killer problem is that someone may have done 'import win32api'
12       # before this code is called.  In that case Windows will have already
13       # loaded pywintypesXX as part of loading win32api - but by the time
14       # we get here, we may locate a different one.  This appears to work, but
15       # then starts raising bizarre TypeErrors complaining that something
16       # is not a pywintypes type when it clearly is!
17   
18       # So in what we hope is the last major iteration of this, we now
19       # rely on a _win32sysloader module, implemented in C but not relying
20       # on pywintypesXX.dll.  It then can check if the DLL we are looking for
21       # lib is already loaded.
22       # See if this is a debug build.
23       suffix = "_d" if "_d.pyd" in importlib.machinery.EXTENSION_SUFFIXES else ""
24       filename = "%s%d%d%s.dll" % (
25           modname,
26           sys.version_info[0],
27           sys.version_info[1],
28           suffix,
29       )
30       if hasattr(sys, "frozen"):
31           # If we are running from a frozen program (py2exe, McMillan, freeze)
32           # then we try and load the DLL from our sys.path
33           # XXX - This path may also benefit from _win32sysloader?  However,
34           # MarkH has never seen the DLL load problem with py2exe programs...
35           for look in sys.path:
36               # If the sys.path entry is a (presumably) .zip file, use the
37               # directory
38               if os.path.isfile(look):
39                   look = os.path.dirname(look)
40               found = os.path.join(look, filename)
41               if os.path.isfile(found):
42                   break
43           else:
44               raise ImportError(
45                   "Module '%s' isn't in frozen sys.path %s" % (modname, sys.path)
46               )
47       else:
48           # First see if it already in our process - if so, we must use that.
49           import _win32sysloader
50   
51           found = _win32sysloader.GetModuleFilename(filename)
52           if found is None:
53               # We ask Windows to load it next.  This is in an attempt to
54               # get the exact same module loaded should pywintypes be imported
55               # first (which is how we are here) or if, eg, win32api was imported
56               # first thereby implicitly loading the DLL.
57   
58               # Sadly though, it doesn't quite work - if pywintypesxx.dll
59               # is in system32 *and* the executable's directory, on XP SP2, an
60               # import of win32api will cause Windows to load pywintypes
61               # from system32, where LoadLibrary for that name will
62               # load the one in the exe's dir.
63               # That shouldn't really matter though, so long as we only ever
64               # get one loaded.
65               found = _win32sysloader.LoadModule(filename)
66           if found is None:
67               # Windows can't find it - which although isn't relevent here,
68               # means that we *must* be the first win32 import, as an attempt
69               # to import win32api etc would fail when Windows attempts to
70               # locate the DLL.
71               # This is most likely to happen for "non-admin" installs, where
72               # we can't put the files anywhere else on the global path.
73   
74               # If there is a version in our Python directory, use that
75               if os.path.isfile(os.path.join(sys.prefix, filename)):
76                   found = os.path.join(sys.prefix, filename)
77           if found is None:
78               # Not in the Python directory?  Maybe we were installed via
79               # easy_install...
80               if os.path.isfile(os.path.join(os.path.dirname(__file__), filename)):
81                   found = os.path.join(os.path.dirname(__file__), filename)
82           if found is None:
83               # We might have been installed via PIP and without the post-install
84               # script having been run, so they might be in the
85               # lib/site-packages/pywin32_system32 directory.
86               # This isn't ideal as it means, say 'python -c "import win32api"'
87               # will not work but 'python -c "import pywintypes, win32api"' will,
88               # but it's better than nothing...
89               import sysconfig
90   
91               maybe = os.path.join(
92                   sysconfig.get_paths()["platlib"], "pywin32_system32", filename
93               )
94               if os.path.isfile(maybe):
95                   found = maybe
96           if found is None:
97               # give up in disgust.
98               raise ImportError("No system module '%s' (%s)" % (modname, filename))
99       # After importing the module, sys.modules is updated to the DLL we just
100      # loaded - which isn't what we want. So we update sys.modules to refer to
101      # this module, and update our globals from it.
102      old_mod = sys.modules[modname]
103      # Load the DLL.
104      loader = importlib.machinery.ExtensionFileLoader(modname, found)
105      spec = importlib.machinery.ModuleSpec(name=modname, loader=loader, origin=found)
106      mod = importlib.util.module_from_spec(spec)
107      spec.loader.exec_module(mod)
108  
109      # Check the sys.modules[] behaviour we describe above is true...
110      assert sys.modules[modname] is not old_mod
111      assert sys.modules[modname] is mod
112      # as above - re-reset to the *old* module object then update globs.
113      sys.modules[modname] = old_mod
114      globs.update(mod.__dict__)
115  
116  
117  __import_pywin32_system_module__("pywintypes", globals())
