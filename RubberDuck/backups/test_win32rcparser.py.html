8    class TestParser(unittest.TestCase):

13       def testStrings(self):
14           for sid, expected in [
15               ("IDS_TEST_STRING4", "Test 'single quoted' string"),
16               ("IDS_TEST_STRING1", 'Test "quoted" string'),
17               ("IDS_TEST_STRING3", 'String with single " quote'),
18               ("IDS_TEST_STRING2", "Test string"),
19           ]:

21               self.assertEqual(got, expected)

23       def testStandardIds(self):
24           for idc in "IDOK IDCANCEL".split():
25               correct = getattr(win32con, idc)
26               self.assertEqual(self.resources.names[correct], idc)
27               self.assertEqual(self.resources.ids[idc], correct)

29       def testTabStop(self):
30           d = self.resources.dialogs["IDD_TEST_DIALOG2"]
31           tabstop_names = ["IDC_EDIT1", "IDOK"]  # should have WS_TABSTOP
32           tabstop_ids = [self.resources.ids[name] for name in tabstop_names]
33           notabstop_names = ["IDC_EDIT2"]  # should have WS_TABSTOP
34           notabstop_ids = [self.resources.ids[name] for name in notabstop_names]
35           num_ok = 0
36           for cdef in d[1:]:  # skip dlgdef
37               # print cdef
38               cid = cdef[2]
39               style = cdef[-2]
40               styleex = cdef[-1]
41               if cid in tabstop_ids:
42                   self.failUnlessEqual(style & win32con.WS_TABSTOP, win32con.WS_TABSTOP)
43                   num_ok += 1
44               elif cid in notabstop_ids:
45                   self.failUnlessEqual(style & win32con.WS_TABSTOP, 0)
46                   num_ok += 1
47           self.failUnlessEqual(num_ok, len(tabstop_ids) + len(notabstop_ids))
