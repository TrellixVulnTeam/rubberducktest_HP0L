21   Gst.init(None)
22   version = Gst.version()
23   if version < (1, 0, 0, 0):
24       raise Exception('Cannot use camera_gi, Gstreamer < 1.0 is not supported.')
25   Logger.info('CameraGi: Using Gstreamer {}'.format(
26       '.'.join(['{}'.format(x) for x in Gst.version()])))
27   install_gobject_iteration()

30   class _MapInfo(Structure):
31       _fields_ = [
32           ('memory', c_void_p),
33           ('flags', c_int),
34           ('data', c_void_p)]
35       # we don't care about the rest

38   def _on_cameragi_unref(obj):
39       if obj in CameraGi._instances:
40           CameraGi._instances.remove(obj)
41   
42   
43   class CameraGi(CameraBase):
44       '''Implementation of CameraBase using GStreamer 
45    
46       :Parameters: 
47           `video_src`: str, default is 'v4l2src' 
48               Other tested options are: 'dc1394src' for firewire 
49               dc camera (e.g. firefly MV). Any gstreamer video source 
50               should potentially work. 
51               Theoretically a longer string using "!" can be used 
52               describing the first part of a gstreamer pipeline. 
53       '''
54   
55       _instances = []
56   
57       def __init__(self, **kwargs):
58           self._pipeline = None
59           self._camerasink = None
60           self._decodebin = None
61           self._texturesize = None
62           self._video_src = kwargs.get('video_src', 'v4l2src')
63           wk = ref(self, _on_cameragi_unref)
64           CameraGi._instances.append(wk)
65           super(CameraGi, self).__init__(**kwargs)
66   
67       def init_camera(self):
68           # TODO: This doesn't work when camera resolution is resized at runtime.
69           # There must be some other way to release the camera?
70           if self._pipeline:
71               self._pipeline = None
72   
73           video_src = self._video_src
74           if video_src == 'v4l2src':
75               video_src += ' device=/dev/video%d' % self._index
76           elif video_src == 'dc1394src':
77               video_src += ' camera-number=%d' % self._index
78   
79           if Gst.version() < (1, 0, 0, 0):
80               caps = ('video/x-raw-rgb,red_mask=(int)0xff0000,'
81                       'green_mask=(int)0x00ff00,blue_mask=(int)0x0000ff')
82               pl = ('{} ! decodebin name=decoder ! ffmpegcolorspace ! '
83                     'appsink name=camerasink emit-signals=True caps={}')
84           else:
85               caps = 'video/x-raw,format=RGB'
86               pl = '{} ! decodebin name=decoder ! videoconvert ! appsink ' + \
87                    'name=camerasink emit-signals=True caps={}'
88   
89           self._pipeline = Gst.parse_launch(pl.format(video_src, caps))
90           self._camerasink = self._pipeline.get_by_name('camerasink')
91           self._camerasink.connect('new-sample', self._gst_new_sample)
92           self._decodebin = self._pipeline.get_by_name('decoder')
93   
94           if self._camerasink and not self.stopped:
95               self.start()
96   
97       def _gst_new_sample(self, *largs):
98           sample = self._camerasink.emit('pull-sample')
99           if sample is None:
100              return False
101  
102          self._sample = sample
103  
104          if self._texturesize is None:
105              # try to get the camera image size
106              for pad in self._decodebin.srcpads:
107                  s = pad.get_current_caps().get_structure(0)
108                  self._texturesize = (
109                      s.get_value('width'),
110                      s.get_value('height'))
111                  Clock.schedule_once(self._update)
112                  return False
113  
114          Clock.schedule_once(self._update)
115          return False
116  
117      def start(self):
118          super(CameraGi, self).start()
119          self._pipeline.set_state(Gst.State.PLAYING)
120  
121      def stop(self):
122          super(CameraGi, self).stop()
123          self._pipeline.set_state(Gst.State.PAUSED)
124  
125      def unload(self):
126          self._pipeline.set_state(Gst.State.NULL)
127  
128      def _update(self, dt):
129          sample, self._sample = self._sample, None
130          if sample is None:
131              return
132  
133          if self._texture is None and self._texturesize is not None:
134              self._texture = Texture.create(
135                  size=self._texturesize, colorfmt='rgb')
136              self._texture.flip_vertical()
137              self.dispatch('on_load')
138  
139          # decode sample
140          # read the data from the buffer memory
141          try:
142              buf = sample.get_buffer()
143              result, mapinfo = buf.map(Gst.MapFlags.READ)
144  
145              # We cannot get the data out of mapinfo, using Gst 1.0.6 + Gi 3.8.0
146              # related bug report:
147              # https://bugzilla.gnome.org/show_bug.cgi?id=6t8663
148              # ie: mapinfo.data is normally a char*, but here, we have an int
149              # So right now, we use ctypes instead to read the mapinfo ourself.
150              addr = mapinfo.__hash__()
151              c_mapinfo = _MapInfo.from_address(addr)
152  
153              # now get the memory
154              self._buffer = string_at(c_mapinfo.data, mapinfo.size)
155              self._copy_to_gpu()
156          finally:
157              if mapinfo is not None:
158                  buf.unmap(mapinfo)
159

161  @atexit.register
162  def camera_gi_clean():
163      # if we leave the python process with some video running, we can hit a
164      # segfault. This is forcing the stop/unload of all remaining videos before
165      # exiting the python process.
166      for weakcamera in CameraGi._instances:
167          camera = weakcamera()
168          if isinstance(camera, CameraGi):
169              camera.stop()
170              camera.unload()
