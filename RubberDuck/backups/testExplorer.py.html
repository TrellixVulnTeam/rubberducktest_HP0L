24   class ExplorerEvents:
25       def OnVisible(self, visible):
26           global bVisibleEventFired
27           bVisibleEventFired = 1

30   def TestExplorerEvents():
31       global bVisibleEventFired
32       try:
33           iexplore = win32com.client.DispatchWithEvents(
34               "InternetExplorer.Application", ExplorerEvents
35           )
36       except pythoncom.com_error as exc:
37           # In automation we see this error trying to connect to events
38           # It's a little surprising that the non-event tests seem to work, but
39           # whatever...
40           if exc.hresult not in HRESULTS_IN_AUTOMATION:
41               raise
42           print("IE events appear to not be available, so skipping this test")
43           return
44   
45       iexplore.Visible = 1
46       if not bVisibleEventFired:
47           raise RuntimeError("The IE event did not appear to fire!")
48       iexplore.Quit()
49       iexplore = None
50   
51       bVisibleEventFired = 0
52       ie = win32com.client.Dispatch("InternetExplorer.Application")
53       ie_events = win32com.client.DispatchWithEvents(ie, ExplorerEvents)
54       ie.Visible = 1
55       if not bVisibleEventFired:
56           raise RuntimeError("The IE event did not appear to fire!")
57       ie.Quit()
58       ie = None
59       print("IE Event tests worked.")

62   def TestObjectFromWindow():
63       # Check we can use ObjectFromLresult to get the COM object from the
64       # HWND - see KB Q249232
65       # Locating the HWND is different than the KB says...
66       hwnd = win32gui.FindWindow("IEFrame", None)
67       for child_class in [
68           "TabWindowClass",
69           "Shell DocObject View",
70           "Internet Explorer_Server",
71       ]:
72           hwnd = win32gui.FindWindowEx(hwnd, 0, child_class, None)
73           # ack - not working for markh on vista with IE8 (or maybe it is the
74           # lack of the 'accessibility' components mentioned in Q249232)
75           # either way - not working!
76           return
77       # But here is the point - once you have an 'Internet Explorer_Server',
78       # you can send a message and use ObjectFromLresult to get it back.
79       msg = win32gui.RegisterWindowMessage("WM_HTML_GETOBJECT")
80       rc, result = win32gui.SendMessageTimeout(
81           hwnd, msg, 0, 0, win32con.SMTO_ABORTIFHUNG, 1000
82       )
83       ob = pythoncom.ObjectFromLresult(result, pythoncom.IID_IDispatch, 0)
84       doc = Dispatch(ob)
85       # just to prove it works, set the background color of the document.
86       for color in "red green blue orange white".split():
87           doc.bgColor = color
88           time.sleep(0.2)

91   def TestExplorer(iexplore):
92       if not iexplore.Visible:
93           iexplore.Visible = -1
94       filename = os.path.join(os.path.dirname(__file__), "..\\readme.html")
95       iexplore.Navigate(win32api.GetFullPathName(filename))
96       win32api.Sleep(1000)
97       TestObjectFromWindow()
98       win32api.Sleep(3000)
99       try:
100          iexplore.Quit()
101      except (AttributeError, pythoncom.com_error):
102          # User got sick of waiting :)
103          pass
