6    # A new script engine, derived from the standard scripting engine,
7    # which dumps information.
8    
9    # This generally can be used to grab all sorts of useful details about
10   # an engine - expose bugs in it or Python, dump the object model, etc.
11   
12   # As it is derived from the standard engine, it fully supports Python
13   # as a scripting language - meaning the dumps produced can be quite dynamic,
14   # and based on the script code you execute.

36   def Register():
37       import sys
38   
39       if "-d" in sys.argv:
40           dispatcher = "DispatcherWin32trace"
41           debug_desc = " (" + dispatcher + ")"
42           debug_option = "Yes"
43       else:
44           dispatcher = None
45           debug_desc = ""
46           debug_option = ""
47   
48       categories = [axscript.CATID_ActiveScript, axscript.CATID_ActiveScriptParse]
49       clsid = PyDump_CLSID
50       lcid = 0x0409  # // english
51       policy = None  # "win32com.axscript.client.axspolicy.AXScriptPolicy"
52   
53       print("Registering COM server%s..." % debug_desc)
54       from win32com.server.register import RegisterServer
55   
56       languageName = "PyDump"
57       verProgId = "Python.Dumper.1"
58       RegisterServer(
59           clsid=clsid,
60           pythonInstString="win32com.axscript.client.pyscript.PyDumper",
61           className="Python Debugging/Dumping ActiveX Scripting Engine",
62           progID=languageName,
63           verProgID=verProgId,
64           catids=categories,
65           policy=policy,
66           dispatcher=dispatcher,
67       )
68   
69       CreateRegKey(languageName + "\\OLEScript")
70       # Basic Registration for wsh.
71       win32com.server.register._set_string(".pysDump", "pysDumpFile")
72       win32com.server.register._set_string("pysDumpFile\\ScriptEngine", languageName)
73       print("Dumping Server registered.")
74   
75   
76   if __name__ == "__main__":
77       Register()
