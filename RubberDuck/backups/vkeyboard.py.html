vkeyboard.py 
1    ''' 
2    VKeyboard 
3    ========= 
4     
5    .. image:: images/vkeyboard.jpg 
6        :align: right 
7     
8    .. versionadded:: 1.0.8 
9     
10    
11   VKeyboard is an onscreen keyboard for Kivy. Its operation is intended to be 
12   transparent to the user. Using the widget directly is NOT recommended. Read the 
13   section `Request keyboard`_ first. 
14    
15   Modes 
16   ----- 
17    
18   This virtual keyboard has a docked and free mode: 
19    
20   * docked mode (:attr:`VKeyboard.docked` = True) 
21     Generally used when only one person is using the computer, like a tablet or 
22     personal computer etc. 
23   * free mode: (:attr:`VKeyboard.docked` = False) 
24     Mostly for multitouch surfaces. This mode allows multiple virtual 
25     keyboards to be used on the screen. 
26    
27   If the docked mode changes, you need to manually call 
28   :meth:`VKeyboard.setup_mode` otherwise the change will have no impact. 
29   During that call, the VKeyboard, implemented on top of a 
30   :class:`~kivy.uix.scatter.Scatter`, will change the 
31   behavior of the scatter and position the keyboard near the target (if target 
32   and docked mode is set). 
33    
34    
35   Layouts 
36   ------- 
37    
38   The virtual keyboard is able to load a custom layout. If you create a new 
39   layout and put the JSON in :file:`<kivy_data_dir>/keyboards/<layoutid>.json`, 
40   you can load it by setting :attr:`VKeyboard.layout` to your layoutid. 
41    
42   The JSON must be structured like this:: 
43    
44       { 
45           "title": "Title of your layout", 
46           "description": "Description of your layout", 
47           "cols": 15, 
48           "rows": 5, 
49    
50           ... 
51       } 
52    
53   Then, you need to describe the keys in each row, for either a "normal", 
54   "shift" or a "special" (added in version 1.9.0) mode. Keys for this row 
55   data must be named `normal_<row>`, `shift_<row>` and `special_<row>`. 
56   Replace `row` with the row number. 
57   Inside each row, you will describe the key. A key is a 4 element list in 
58   the format:: 
59    
60       [ <text displayed on the keyboard>, <text to put when the key is pressed>, 
61         <text that represents the keycode>, <size of cols> ] 
62    
63   Here are example keys:: 
64    
65       # f key 
66       ["f", "f", "f", 1] 
67       # capslock 
68       ["\u21B9", "\t", "tab", 1.5] 
69    
70   Finally, complete the JSON:: 
71    
72       { 
73           ... 
74           "normal_1": [ 
75               ["`", "`", "`", 1],    ["1", "1", "1", 1],    ["2", "2", "2", 1], 
76               ["3", "3", "3", 1],    ["4", "4", "4", 1],    ["5", "5", "5", 1], 
77               ["6", "6", "6", 1],    ["7", "7", "7", 1],    ["8", "8", "8", 1], 
78               ["9", "9", "9", 1],    ["0", "0", "0", 1],    ["+", "+", "+", 1], 
79               ["=", "=", "=", 1],    ["\u232b", null, "backspace", 2] 
80           ], 
81    
82           "shift_1": [ ... ], 
83           "normal_2": [ ... ], 
84           "special_2": [ ... ], 
85           ... 
86       } 
87    
88    
89   Request Keyboard 
90   ---------------- 
91    
92   The instantiation of the virtual keyboard is controlled by the configuration. 
93   Check `keyboard_mode` and `keyboard_layout` in the :doc:`api-kivy.config`. 
94    
95   If you intend to create a widget that requires a keyboard, do not use the 
96   virtual keyboard directly, but prefer to use the best method available on 
97   the platform. Check the :meth:`~kivy.core.window.WindowBase.request_keyboard` 
98   method in the :doc:`api-kivy.core.window`. 
99    
100  If you want a specific layout when you request the keyboard, you should write 
101  something like this (from 1.8.0, numeric.json can be in the same directory as 
102  your main.py):: 
103   
104      keyboard = Window.request_keyboard( 
105          self._keyboard_close, self) 
106      if keyboard.widget: 
107          vkeyboard = self._keyboard.widget 
108          vkeyboard.layout = 'numeric.json'

411      def _update_layout_mode(self, *l):
412          # update mode according to capslock and shift key
413          mode = self.have_capslock != self.have_shift
414          mode = 'shift' if mode else 'normal'
415          if self.have_special:
416              mode = "special"
417          if mode != self.layout_mode:
418              self.layout_mode = mode
419              self.refresh(False)
420  
421      def _load_layout(self, *largs):
422          # ensure new layouts are loaded first
423          if self._trigger_load_layouts.is_triggered:
424              self._load_layouts()
425              self._trigger_load_layouts.cancel()
426  
427          value = self.layout
428          available_layouts = self.available_layouts
429  
430          # it's a filename, try to load it directly
431          if self.layout[-5:] == '.json':
432              if value not in available_layouts:
433                  fn = resource_find(self.layout)
434                  self._load_layout_fn(fn, self.layout)
435  
436          if not available_layouts:
437              return
438          if value not in available_layouts and value != 'qwerty':
439              Logger.error(
440                  'Vkeyboard: <%s> keyboard layout mentioned in '
441                  'conf file was not found, fallback on qwerty' %
442                  value)
443              self.layout = 'qwerty'
444          self.refresh(True)
445  
446      def _load_layouts(self, *largs):
447          # first load available layouts from json files
448          # XXX fix to be able to reload layout when path is changing
449          value = self.layout_path
450          for fn in listdir(value):
451              self._load_layout_fn(join(value, fn),
452                                   basename(splitext(fn)[0]))
453  
454      def _load_layout_fn(self, fn, name):
455          available_layouts = self.available_layouts
456          if fn[-5:] != '.json':
457              return
458          with open(fn, 'r', encoding='utf-8') as fd:
459              json_content = fd.read()
460              layout = loads(json_content)
461          available_layouts[name] = layout

463      def setup_mode(self, *largs):
464          '''Call this method when you want to readjust the keyboard according to 
465          options: :attr:`docked` or not, with attached :attr:`target` or not: 
466   
467          * If :attr:`docked` is True, it will call :meth:`setup_mode_dock` 
468          * If :attr:`docked` is False, it will call :meth:`setup_mode_free` 
469   
470          Feel free to overload these methods to create new 
471          positioning behavior. 
472          '''
473          if self.docked:
474              self.setup_mode_dock()
475          else:
476              self.setup_mode_free()
477  
478      def setup_mode_dock(self, *largs):
479          '''Setup the keyboard in docked mode. 
480   
481          Dock mode will reset the rotation, disable translation, rotation and 
482          scale. Scale and position will be automatically adjusted to attach the 
483          keyboard to the bottom of the screen. 
484   
485          .. note:: 
486              Don't call this method directly, use :meth:`setup_mode` instead. 
487          '''
488          self.do_translation = False
489          self.do_rotation = False
490          self.do_scale = False
491          self.rotation = 0
492          win = self.get_parent_window()
493          scale = win.width / float(self.width)
494          self.scale = scale
495          self.pos = 0, 0
496          win.bind(on_resize=self._update_dock_mode)
497  
498      def _update_dock_mode(self, win, *largs):
499          scale = win.width / float(self.width)
500          self.scale = scale
501          self.pos = 0, 0
502  
503      def setup_mode_free(self):
504          '''Setup the keyboard in free mode. 
505   
506          Free mode is designed to let the user control the position and 
507          orientation of the keyboard. The only real usage is for a multiuser 
508          environment, but you might found other ways to use it. 
509          If a :attr:`target` is set, it will place the vkeyboard under the 
510          target. 
511   
512          .. note:: 
513              Don't call this method directly, use :meth:`setup_mode` instead. 
514          '''
515          self.do_translation = True
516          self.do_rotation = True
517          self.do_scale = True
518          target = self.target
519          if not target:
520              return
521  
522          # NOTE all math will be done in window point of view
523          # determine rotation of the target
524          a = Vector(1, 0)
525          b = Vector(target.to_window(0, 0))
526          c = Vector(target.to_window(1, 0)) - b
527          self.rotation = -a.angle(c)
528  
529          # determine the position of center/top of the keyboard
530          dpos = Vector(self.to_window(self.width / 2., self.height))
531  
532          # determine the position of center/bottom of the target
533          cpos = Vector(target.to_window(target.center_x, target.y))
534  
535          # the goal now is to map both point, calculate the diff between them
536          diff = dpos - cpos
537  
538          # we still have an issue, self.pos represent the bounding box,
539          # not the 0,0 coordinate of the scatter. we need to apply also
540          # the diff between them (inside and outside coordinate matrix).
541          # It's hard to explain, but do a scheme on a paper, write all
542          # the vector i'm calculating, and you'll understand. :)
543          diff2 = Vector(self.x + self.width / 2., self.y + self.height) - \
544              Vector(self.to_parent(self.width / 2., self.height))
545          diff -= diff2
546  
547          # now we have a good "diff", set it as a pos.
548          self.pos = -diff

554      def refresh(self, force=False):
555          '''(internal) Recreate the entire widget and graphics according to the 
556          selected layout. 
557          '''
558          self.clear_widgets()
559          if force:
560              self.refresh_keys_hint()
561          self.refresh_keys()
562          self.refresh_active_keys_layer()
563  
564      def refresh_active_keys_layer(self):
565          self.active_keys_layer.clear()
566  
567          active_keys = self.active_keys
568          layout_geometry = self.layout_geometry
569          background = resource_find(self.key_background_down)
570          texture = Image(background, mipmap=True).texture
571  
572          with self.active_keys_layer:
573              Color(*self.key_background_color)
574              for line_nb, index in active_keys.values():
575                  pos, size = layout_geometry['LINE_%d' % line_nb][index]
576                  BorderImage(texture=texture, pos=pos, size=size,
577                              border=self.key_border)
578  
579      def refresh_keys_hint(self):
580          layout = self.available_layouts[self.layout]
581          layout_cols = layout['cols']
582          layout_rows = layout['rows']
583          layout_geometry = self.layout_geometry
584          mtop, mright, mbottom, mleft = self.margin_hint
585  
586          # get relative EFFICIENT surface of the layout without external margins
587          el_hint = 1. - mleft - mright
588          eh_hint = 1. - mtop - mbottom
589          ex_hint = 0 + mleft
590          ey_hint = 0 + mbottom
591  
592          # get relative unit surface
593          uw_hint = (1. / layout_cols) * el_hint
594          uh_hint = (1. / layout_rows) * eh_hint
595          layout_geometry['U_HINT'] = (uw_hint, uh_hint)
596  
597          # calculate individual key RELATIVE surface and pos (without key
598          # margin)
599          current_y_hint = ey_hint + eh_hint
600          for line_nb in range(1, layout_rows + 1):
601              current_y_hint -= uh_hint
602              # get line_name
603              line_name = '%s_%d' % (self.layout_mode, line_nb)
604              line_hint = 'LINE_HINT_%d' % line_nb
605              layout_geometry[line_hint] = []
606              current_x_hint = ex_hint
607              # go through the list of keys (tuples of 4)
608              for key in layout[line_name]:
609                  # calculate relative pos, size
610                  layout_geometry[line_hint].append([
611                      (current_x_hint, current_y_hint),
612                      (key[3] * uw_hint, uh_hint)])
613                  current_x_hint += key[3] * uw_hint
614  
615          self.layout_geometry = layout_geometry
616  
617      def refresh_keys(self):
618          layout = self.available_layouts[self.layout]
619          layout_rows = layout['rows']
620          layout_geometry = self.layout_geometry
621          w, h = self.size
622          kmtop, kmright, kmbottom, kmleft = self.key_margin
623          uw_hint, uh_hint = layout_geometry['U_HINT']
624  
625          for line_nb in range(1, layout_rows + 1):
626              llg = layout_geometry['LINE_%d' % line_nb] = []
627              llg_append = llg.append
628              for key in layout_geometry['LINE_HINT_%d' % line_nb]:
629                  x_hint, y_hint = key[0]
630                  w_hint, h_hint = key[1]
631                  kx = x_hint * w
632                  ky = y_hint * h
633                  kw = w_hint * w
634                  kh = h_hint * h
635  
636                  # now adjust, considering the key margin
637                  kx = int(kx + kmleft)
638                  ky = int(ky + kmbottom)
639                  kw = int(kw - kmleft - kmright)
640                  kh = int(kh - kmbottom - kmtop)
641  
642                  pos = (kx, ky)
643                  size = (kw, kh)
644                  llg_append((pos, size))
645  
646          self.layout_geometry = layout_geometry
647          self.draw_keys()
648  
649      def draw_keys(self):
650          layout = self.available_layouts[self.layout]
651          layout_rows = layout['rows']
652          layout_geometry = self.layout_geometry
653          layout_mode = self.layout_mode
654  
655          # draw background
656          background = resource_find(self.background_disabled
657                                     if self.disabled else
658                                     self.background)
659          texture = Image(background, mipmap=True).texture
660          self.background_key_layer.clear()
661          with self.background_key_layer:
662              Color(*self.background_color)
663              BorderImage(texture=texture, size=self.size,
664                          border=self.background_border)
665  
666          # XXX separate drawing the keys and the fonts to avoid
667          # XXX reloading the texture each time
668  
669          # first draw keys without the font
670          key_normal = resource_find(self.key_background_disabled_normal
671                                     if self.disabled else
672                                     self.key_background_normal)
673          texture = Image(key_normal, mipmap=True).texture
674          with self.background_key_layer:
675              Color(*self.key_background_color)
676              for line_nb in range(1, layout_rows + 1):
677                  for pos, size in layout_geometry['LINE_%d' % line_nb]:
678                      BorderImage(texture=texture, pos=pos, size=size,
679                                  border=self.key_border)
680  
681          # then draw the text
682          for line_nb in range(1, layout_rows + 1):
683              key_nb = 0
684              for pos, size in layout_geometry['LINE_%d' % line_nb]:
685                  # retrieve the relative text
686                  text = layout[layout_mode + '_' + str(line_nb)][key_nb][0]
687                  z = Label(text=text, font_size=self.font_size, pos=pos,
688                             size=size, font_name=self.font_name)
689                  self.add_widget(z)
690                  key_nb += 1

701      def get_key_at_pos(self, x, y):
702          w, h = self.size
703          x_hint = x / w
704          # focus on the surface without margins
705          layout_geometry = self.layout_geometry
706          layout = self.available_layouts[self.layout]
707          layout_rows = layout['rows']
708          mtop, mright, mbottom, mleft = self.margin_hint
709  
710          # get the line of the layout
711          e_height = h - (mbottom + mtop) * h  # efficient height in pixels
712          line_height = e_height / layout_rows  # line height in px
713          y = y - mbottom * h
714          line_nb = layout_rows - int(y / line_height)
715  
716          if line_nb > layout_rows:
717              line_nb = layout_rows
718          if line_nb < 1:
719              line_nb = 1
720  
721          # get the key within the line
722          key_index = ''
723          current_key_index = 0
724          for key in layout_geometry['LINE_HINT_%d' % line_nb]:
725              if x_hint >= key[0][0] and x_hint < key[0][0] + key[1][0]:
726                  key_index = current_key_index
727                  break
728              else:
729                  current_key_index += 1
730          if key_index == '':
731              return None
732  
733          # get the full character
734          key = layout['%s_%d' % (self.layout_mode, line_nb)][key_index]
735  
736          return [key, (line_nb, key_index)]

738      def collide_margin(self, x, y):
739          '''Do a collision test, and return True if the (x, y) is inside the 
740          vkeyboard margin. 
741          '''
742          mtop, mright, mbottom, mleft = self.margin_hint
743          x_hint = x / self.width
744          y_hint = y / self.height
745          if x_hint > mleft and x_hint < 1. - mright \
746                  and y_hint > mbottom and y_hint < 1. - mtop:
747              return False
748          return True

750      def process_key_on(self, touch):
751          if not touch:
752              return
753          x, y = self.to_local(*touch.pos)
754          key = self.get_key_at_pos(x, y)
755          if not key:
756              return
757  
758          key_data = key[0]
759          displayed_char, internal, special_char, size = key_data
760          line_nb, key_index = key[1]
761  
762          # save pressed key on the touch
763          ud = touch.ud[self.uid] = {}
764          ud['key'] = key
765  
766          # for caps lock or shift only:
767          uid = touch.uid
768          if special_char is not None:
769              # Do not repeat special keys
770              if special_char in ('capslock', 'shift', 'layout', 'special'):
771                  if self._start_repeat_key_ev is not None:
772                      self._start_repeat_key_ev.cancel()
773                      self._start_repeat_key_ev = None
774                  self.repeat_touch = None
775              if special_char == 'capslock':
776                  self.have_capslock = not self.have_capslock
777                  uid = -1
778              elif special_char == 'shift':
779                  self.have_shift = True
780              elif special_char == 'special':
781                  self.have_special = True
782              elif special_char == 'layout':
783                  self.change_layout()
784  
785          # send info to the bus
786          b_keycode = special_char
787          b_modifiers = self._get_modifiers()
788          if self.get_parent_window().__class__.__module__ == \
789              'kivy.core.window.window_sdl2' and internal:
790              self.dispatch('on_textinput', internal)
791          else:
792              self.dispatch('on_key_down', b_keycode, internal, b_modifiers)
793  
794          # save key as an active key for drawing
795          self.active_keys[uid] = key[1]
796          self.refresh_active_keys_layer()
797  
798      def process_key_up(self, touch):
799          uid = touch.uid
800          if self.uid not in touch.ud:
801              return
802  
803          # save pressed key on the touch
804          key_data, key = touch.ud[self.uid]['key']
805          displayed_char, internal, special_char, size = key_data
806  
807          # send info to the bus
808          b_keycode = special_char
809          b_modifiers = self._get_modifiers()
810          self.dispatch('on_key_up', b_keycode, internal, b_modifiers)
811  
812          if special_char == 'capslock':
813              uid = -1
814  
815          if uid in self.active_keys:
816              self.active_keys.pop(uid, None)
817              if special_char == 'shift':
818                  self.have_shift = False
819              elif special_char == 'special':
820                  self.have_special = False
821              if special_char == 'capslock' and self.have_capslock:
822                  self.active_keys[-1] = key
823              self.refresh_active_keys_layer()
824  
825      def _get_modifiers(self):
826          ret = []
827          if self.have_shift:
828              ret.append('shift')
829          if self.have_capslock:
830              ret.append('capslock')
831          return ret

833      def _start_repeat_key(self, *kwargs):
834          self._repeat_key_ev = Clock.schedule_interval(self._repeat_key, 0.05)
835  
836      def _repeat_key(self, *kwargs):
837          self.process_key_on(self.repeat_touch)

839      def on_touch_down(self, touch):
840          x, y = touch.pos
841          if not self.collide_point(x, y):
842              return
843          if self.disabled:
844              return True
845  
846          x, y = self.to_local(x, y)
847          if not self.collide_margin(x, y):
848              if self.repeat_touch is None:
849                  self._start_repeat_key_ev = Clock.schedule_once(
850                      self._start_repeat_key, 0.5)
851              self.repeat_touch = touch
852  
853              self.process_key_on(touch)
854              touch.grab(self, exclusive=True)
855  
856          else:
857              super(VKeyboard, self).on_touch_down(touch)
858          return True
859  
860      def on_touch_up(self, touch):
861          if touch.grab_current is self:
862              self.process_key_up(touch)
863  
864              if self._start_repeat_key_ev is not None:
865                  self._start_repeat_key_ev.cancel()
866                  self._start_repeat_key_ev = None
867              if touch == self.repeat_touch:
868                  if self._repeat_key_ev is not None:
869                      self._repeat_key_ev.cancel()
870                      self._repeat_key_ev = None
871                  self.repeat_touch = None
872  
873          return super(VKeyboard, self).on_touch_up(touch)
