62       def GetPositionOfLine(self, cLineNumber):
63           self.GetText()  # Prime us.
64           try:
65               return self.lineOffsets[cLineNumber]
66           except IndexError:
67               raise Exception(scode=winerror.S_FALSE)
68   
69       def GetLineOfPosition(self, charPos):
70           self.GetText()  # Prime us.
71           lastOffset = 0
72           lineNo = 0
73           for lineOffset in self.lineOffsets[1:]:
74               if lineOffset > charPos:
75                   break
76               lastOffset = lineOffset
77               lineNo = lineNo + 1
78           else:  # for not broken.
79               #                       print "Cant find", charPos, "in", self.lineOffsets
80               raise Exception(scode=winerror.S_FALSE)
81           #               print "GLOP ret=",lineNo,       (charPos-lastOffset)
82           return lineNo, (charPos - lastOffset)
83   
84       def GetNextLine(self):
85           if self.nextLineNo >= len(self.lines):
86               self.nextLineNo = 0  # auto-reset.
87               return ""
88           rc = self.lines[self.nextLineNo]
89           self.nextLineNo = self.nextLineNo + 1
90           return rc
91   
92       def GetLine(self, num):
93           self.GetText()  # Prime us.
94           return self.lines[num]
95   
96       def GetNumChars(self):
97           return len(self.GetText())
98   
99       def GetNumLines(self):
100          self.GetText()  # Prime us.
101          return len(self.lines)

103      def _buildline(self, pos):
104          i = self.text.find("\n", pos)
105          if i < 0:
106              newpos = len(self.text)
107          else:
108              newpos = i + 1
109          r = self.text[pos:newpos]
110          return r, newpos
111  
112      def _buildlines(self):
113          self.lines = []
114          self.lineOffsets = [0]
115          line, pos = self._buildline(0)
116          while line:
117              self.lines.append(line)
118              self.lineOffsets.append(pos)
119              line, pos = self._buildline(pos)
120  
121      def _ProcessToken(self, type, token, spos, epos, line):
122          srow, scol = spos
123          erow, ecol = epos
124          self.GetText()  # Prime us.
125          linenum = srow - 1  # Lines zero based for us too.
126          realCharPos = self.lineOffsets[linenum] + scol
127          numskipped = realCharPos - self.lastPos
128          if numskipped == 0:
129              pass
130          elif numskipped == 1:
131              self.attrs.append(axdebug.SOURCETEXT_ATTR_COMMENT)
132          else:
133              self.attrs.append((axdebug.SOURCETEXT_ATTR_COMMENT, numskipped))
134          kwSize = len(token)
135          self.lastPos = realCharPos + kwSize
136          attr = 0
137  
138          if type == tokenize.NAME:
139              if token in _keywords:
140                  attr = axdebug.SOURCETEXT_ATTR_KEYWORD
141          elif type == tokenize.STRING:
142              attr = axdebug.SOURCETEXT_ATTR_STRING
143          elif type == tokenize.NUMBER:
144              attr = axdebug.SOURCETEXT_ATTR_NUMBER
145          elif type == tokenize.OP:
146              attr = axdebug.SOURCETEXT_ATTR_OPERATOR
147          elif type == tokenize.COMMENT:
148              attr = axdebug.SOURCETEXT_ATTR_COMMENT
149          # else attr remains zero...
150          if kwSize == 0:
151              pass
152          elif kwSize == 1:
153              self.attrs.append(attr)
154          else:
155              self.attrs.append((attr, kwSize))
156  
157      def GetSyntaxColorAttributes(self):
158          self.lastPos = 0
159          self.attrs = []
160          try:
161              tokenize.tokenize(self.GetNextLine, self._ProcessToken)
162          except tokenize.TokenError:
163              pass  # Ignore - will cause all subsequent text to be commented.
164          numAtEnd = len(self.GetText()) - self.lastPos
165          if numAtEnd:
166              self.attrs.append((axdebug.SOURCETEXT_ATTR_COMMENT, numAtEnd))
167          return self.attrs

171          return _wrap(
172              contexts.DebugCodeContext(lineNo, charPos, len, self, self.site),
173              axdebug.IID_IDebugCodeContext,

176      # Make a context at the given position.  It should take up the entire context.
177      def _MakeContextAtPosition(self, charPos):
178          lineNo, offset = self.GetLineOfPosition(charPos)
179          try:
180              endPos = self.GetPositionOfLine(lineNo + 1)
181          except:
182              endPos = charPos
183          codecontext = self._MakeDebugCodeContext(lineNo, charPos, endPos - charPos)
184          return codecontext

186      # Returns a DebugCodeContext.  debugDocument can be None for smart hosts.
187      def GetCodeContextAtPosition(self, charPos):
188          #               trace("GetContextOfPos", charPos, maxChars)
189          # Convert to line number.
190          lineNo, offset = self.GetLineOfPosition(charPos)
191          charPos = self.GetPositionOfLine(lineNo)
192          try:
193              cc = self.codeContexts[charPos]
194          #                       trace(" GetContextOfPos using existing")
195          except KeyError:
196              cc = self._MakeContextAtPosition(charPos)
197              self.codeContexts[charPos] = cc
198          return cc

201  class SourceModuleContainer(SourceCodeContainer):
202      def __init__(self, module):
203          self.module = module
204          if hasattr(module, "__file__"):
205              fname = self.module.__file__
206              # Check for .pyc or .pyo or even .pys!
207              if fname[-1] in ["O", "o", "C", "c", "S", "s"]:
208                  fname = fname[:-1]
209              try:
210                  fname = win32api.GetFullPathName(fname)
211              except win32api.error:
212                  pass
213          else:
214              if module.__name__ == "__main__" and len(sys.argv) > 0:
215                  fname = sys.argv[0]
216              else:
217                  fname = "<Unknown!>"
218          SourceCodeContainer.__init__(self, None, fname)
219  
220      def GetText(self):
221          if self.text is None:
222              fname = self.GetFileName()
223              if fname:
224                  try:
225                      self.text = open(fname, "r").read()
226                  except IOError as details:
227                      self.text = "# Exception opening file\n# %s" % (repr(details))
228              else:
229                  self.text = "# No file available for module '%s'" % (self.module)
230              self._buildlines()
231          return self.text
232  
233      def GetName(self, dnt):
234          name = self.module.__name__
235          try:
236              fname = win32api.GetFullPathName(self.module.__file__)
237          except win32api.error:
238              fname = self.module.__file__
239          except AttributeError:
240              fname = name
241          if dnt == axdebug.DOCUMENTNAMETYPE_APPNODE:
242              return name.split(".")[-1]
243          elif dnt == axdebug.DOCUMENTNAMETYPE_TITLE:
244              return fname
245          elif dnt == axdebug.DOCUMENTNAMETYPE_FILE_TAIL:
246              return os.path.split(fname)[1]
247          elif dnt == axdebug.DOCUMENTNAMETYPE_URL:
248              return "file:%s" % fname
249          else:
250              raise Exception(scode=winerror.E_UNEXPECTED)
