1    ''' 
2    Bubble 
3    ====== 
4     
5    .. versionadded:: 1.1.0 
6     
7    .. image:: images/bubble.jpg 
8        :align: right 
9     
10   The Bubble widget is a form of menu or a small popup where the menu options 
11   are stacked either vertically or horizontally. 
12    
13   The :class:`Bubble` contains an arrow pointing in the direction you 
14   choose. 
15    
16   Simple example 
17   -------------- 
18    
19   .. include:: ../../examples/widgets/bubble_test.py 
20       :literal: 
21    
22   Customize the Bubble 
23   -------------------- 
24    
25   You can choose the direction in which the arrow points:: 
26    
27       Bubble(arrow_pos='top_mid') 
28    
29   The widgets added to the Bubble are ordered horizontally by default, like a 
30   Boxlayout. You can change that by:: 
31    
32       orientation = 'vertical' 
33    
34   To add items to the bubble:: 
35    
36       bubble = Bubble(orientation = 'vertical') 
37       bubble.add_widget(your_widget_instance) 
38    
39   To remove items:: 
40    
41       bubble.remove_widget(widget) 
42       or 
43       bubble.clear_widgets() 
44    
45   To access the list of children, use content.children:: 
46    
47       bubble.content.children 
48    
49   .. warning:: 
50     This is important! Do not use bubble.children 
51    
52   To change the appearance of the bubble:: 
53    
54       bubble.background_color = (1, 0, 0, .5) #50% translucent red 
55       bubble.border = [0, 0, 0, 0] 
56       background_image = 'path/to/background/image' 
57       arrow_image = 'path/to/arrow/image' 
58   '''

90   class Bubble(GridLayout):
91       '''Bubble class. See module documentation for more information. 
92       '''

94       background_color = ColorProperty([1, 1, 1, 1])
95       '''Background color, in the format (r, g, b, a). To use it you have to set 
96       either :attr:`background_image` or :attr:`arrow_image` first. 
97    
98       :attr:`background_color` is a :class:`~kivy.properties.ColorProperty` and 
99       defaults to [1, 1, 1, 1]. 
100   
101      .. versionchanged:: 2.0.0 
102          Changed from :class:`~kivy.properties.ListProperty` to 
103          :class:`~kivy.properties.ColorProperty`. 
104      '''

106      border = ListProperty([16, 16, 16, 16])
107      '''Border used for :class:`~kivy.graphics.vertex_instructions.BorderImage` 
108      graphics instruction. Used with the :attr:`background_image`. 
109      It should be used when using custom backgrounds. 
110   
111      It must be a list of 4 values: (bottom, right, top, left). Read the 
112      BorderImage instructions for more information about how to use it. 
113   
114      :attr:`border` is a :class:`~kivy.properties.ListProperty` and defaults to 
115      (16, 16, 16, 16) 
116      '''

118      background_image = StringProperty(
119          'atlas://data/images/defaulttheme/bubble')
120      '''Background image of the bubble.

123      defaults to 'atlas://data/images/defaulttheme/bubble'. 
124      '''

127          'atlas://data/images/defaulttheme/bubble_arrow')
128      ''' Image of the arrow pointing to the bubble.

139      :attr:`show_arrow` is a :class:`~kivy.properties.BooleanProperty` and

143      arrow_pos = OptionProperty('bottom_mid', options=(
144          'left_top', 'left_mid', 'left_bottom', 'top_left', 'top_mid',
145          'top_right', 'right_top', 'right_mid', 'right_bottom',
146          'bottom_left', 'bottom_mid', 'bottom_right'))
147      '''Specifies the position of the arrow relative to the bubble. 
148      Can be one of: left_top, left_mid, left_bottom top_left, top_mid, top_right 
149      right_top, right_mid, right_bottom bottom_left, bottom_mid, bottom_right.

197      def __init__(self, **kwargs):
198          self._prev_arrow_pos = None
199          self._arrow_layout = BoxLayout()
200          self._bk_img = Image(
201              source=self.background_image, allow_stretch=True,
202              keep_ratio=False, color=self.background_color)
203          self.background_texture = self._bk_img.texture
204          self._arrow_img = Image(source=self.arrow_image,
205                                  allow_stretch=True,
206                                  color=self.background_color)
207          self.content = content = BubbleContent(parent=self)
208          super(Bubble, self).__init__(**kwargs)
209          content.parent = None
210          self.add_widget(content)
211          self.on_arrow_pos()
212  
213      def add_widget(self, *l):
214          content = self.content
215          if content is None:
216              return
217          if l[0] == content or l[0] == self._arrow_img\
218                  or l[0] == self._arrow_layout:
219              super(Bubble, self).add_widget(*l)
220          else:
221              content.add_widget(*l)
222  
223      def remove_widget(self, *l):
224          content = self.content
225          if not content:
226              return
227          if l[0] == content or l[0] == self._arrow_img\
228                  or l[0] == self._arrow_layout:
229              super(Bubble, self).remove_widget(*l)
230          else:
231              content.remove_widget(l[0])
232  
233      def clear_widgets(self, **kwargs):
234          content = self.content
235          if not content:
236              return
237          if kwargs.get('do_super', False):
238              super(Bubble, self).clear_widgets()
239          else:
240              content.clear_widgets()
241  
242      def on_show_arrow(self, instance, value):
243          self._arrow_img.opacity = int(value)
244  
245      def on_parent(self, instance, value):
246          Clock.schedule_once(self._update_arrow)
247  
248      def on_pos(self, instance, pos):
249          lt = self.limit_to
250  
251          if lt:
252              self.limit_to = None
253              if lt is EventLoop.window:
254                  x = y = 0
255                  top = lt.height
256                  right = lt.width
257              else:
258                  x, y = lt.x, lt.y
259                  top, right = lt.top, lt.right
260  
261              self.x = max(self.x, x)
262              self.right = min(self.right, right)
263              self.top = min(self.top, top)
264              self.y = max(self.y, y)
265              self.limit_to = lt
266  
267      def on_background_image(self, *l):
268          self._bk_img.source = self.background_image
269  
270      def on_background_color(self, *l):
271          if self.content is None:
272              return
273          self._arrow_img.color = self._bk_img.color = self.background_color
274  
275      def on_orientation(self, *l):
276          content = self.content
277          if not content:
278              return
279          if self.orientation[0] == 'v':
280              content.cols = 1
281              content.rows = 99
282          else:
283              content.cols = 99
284              content.rows = 1
285  
286      def on_arrow_image(self, *l):
287          self._arrow_img.source = self.arrow_image
288  
289      def on_arrow_pos(self, *l):
290          self_content = self.content
291          if not self_content:
292              Clock.schedule_once(self.on_arrow_pos)
293              return
294          if self_content not in self.children:
295              Clock.schedule_once(self.on_arrow_pos)
296              return
297          self_arrow_pos = self.arrow_pos
298          if self._prev_arrow_pos == self_arrow_pos:
299              return
300          self._prev_arrow_pos = self_arrow_pos
301  
302          self_arrow_layout = self._arrow_layout
303          self_arrow_layout.clear_widgets()
304          self_arrow_img = self._arrow_img
305          self._sctr = self._arrow_img
306          self.clear_widgets(do_super=True)
307          self_content.parent = None
308  
309          self_arrow_img.size_hint = (1, None)
310          self_arrow_img.height = dp(self_arrow_img.texture_size[1])
311          self_arrow_img.pos = 0, 0
312          widget_list = []
313          arrow_list = []
314          parent = self_arrow_img.parent
315          if parent:
316              parent.remove_widget(self_arrow_img)
317  
318          if self_arrow_pos[0] == 'b' or self_arrow_pos[0] == 't':
319              self.cols = 1
320              self.rows = 3
321              self_arrow_layout.orientation = 'horizontal'
322              self_arrow_img.width = self.width / 3
323              self_arrow_layout.size_hint = (1, None)
324              self_arrow_layout.height = self_arrow_img.height
325              if self_arrow_pos[0] == 'b':
326                  if self_arrow_pos == 'bottom_mid':
327                      widget_list = (self_content, self_arrow_img)
328                  else:
329                      if self_arrow_pos == 'bottom_left':
330                          arrow_list = (self_arrow_img, Widget(), Widget())
331                      elif self_arrow_pos == 'bottom_right':
332                          # add two dummy widgets
333                          arrow_list = (Widget(), Widget(), self_arrow_img)
334                      widget_list = (self_content, self_arrow_layout)
335              else:
336                  sctr = Scatter(do_translation=False,
337                                 rotation=180,
338                                 do_rotation=False,
339                                 do_scale=False,
340                                 size_hint=(None, None),
341                                 size=self_arrow_img.size)
342                  sctr.add_widget(self_arrow_img)
343                  if self_arrow_pos == 'top_mid':
344                      # add two dummy widgets
345                      arrow_list = (Widget(), sctr, Widget())
346                  elif self_arrow_pos == 'top_left':
347                      arrow_list = (sctr, Widget(), Widget())
348                  elif self_arrow_pos == 'top_right':
349                      arrow_list = (Widget(), Widget(), sctr)
350                  widget_list = (self_arrow_layout, self_content)
351          elif self_arrow_pos[0] == 'l' or self_arrow_pos[0] == 'r':
352              self.cols = 3
353              self.rows = 1
354              self_arrow_img.width = self.height / 3
355              self_arrow_layout.orientation = 'vertical'
356              self_arrow_layout.cols = 1
357              self_arrow_layout.size_hint = (None, 1)
358              self_arrow_layout.width = self_arrow_img.height
359  
360              rotation = -90 if self_arrow_pos[0] == 'l' else 90
361              self._sctr = sctr = Scatter(do_translation=False,
362                                          rotation=rotation,
363                                          do_rotation=False,
364                                          do_scale=False,
365                                          size_hint=(None, None),
366                                          size=(self_arrow_img.size))
367              sctr.add_widget(self_arrow_img)
368  
369              if self_arrow_pos[-4:] == '_top':
370                  arrow_list = (Widget(size_hint=(1, .07)),
371                                sctr, Widget(size_hint=(1, .3)))
372              elif self_arrow_pos[-4:] == '_mid':
373                  arrow_list = (Widget(), sctr, Widget())
374                  Clock.schedule_once(self._update_arrow)
375              elif self_arrow_pos[-7:] == '_bottom':
376                  arrow_list = (Widget(), Widget(), sctr)
377  
378              if self_arrow_pos[0] == 'l':
379                  widget_list = (self_arrow_layout, self_content)
380              else:
381                  widget_list = (self_content, self_arrow_layout)
382  
383          # add widgets to arrow_layout
384          add = self_arrow_layout.add_widget
385          for widg in arrow_list:
386              add(widg)
387  
388          # add widgets to self
389          add = self.add_widget
390          for widg in widget_list:
391              add(widg)
392  
393      def _update_arrow(self, *dt):
394          if self.arrow_pos in ('left_mid', 'right_mid'):
395              self._sctr.center_y = self._arrow_layout.center_y
