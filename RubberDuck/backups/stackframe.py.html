17   class EnumDebugStackFrames(gateways.EnumDebugStackFrames):
18       """A class that given a debugger object, can return an enumerator 
19       of DebugStackFrame objects. 
20       """
21   
22       def __init__(self, debugger):
23           infos = []
24           frame = debugger.currentframe
25           #               print "Stack check"
26           while frame:
27               #                       print " Checking frame", frame.f_code.co_filename, frame.f_lineno-1, frame.f_trace,
28               # Get a DebugCodeContext for the stack frame.  If we fail, then it
29               # is not debuggable, and therefore not worth displaying.
30               cc = debugger.codeContainerProvider.FromFileName(frame.f_code.co_filename)
31               if cc is not None:
32                   try:
33                       address = frame.f_locals["__axstack_address__"]
34                   except KeyError:
35                       #                                       print "Couldnt find stack address for",frame.f_code.co_filename, frame.f_lineno-1
36                       # Use this one, even tho it is wrong :-(
37                       address = axdebug.GetStackAddress()
38                   frameInfo = (
39                       DebugStackFrame(frame, frame.f_lineno - 1, cc),
40                       address,
41                       address + 1,
42                       0,
43                       None,
44                   )
45                   infos.append(frameInfo)
46               #                               print "- Kept!"
47               #                       else:
48               #                               print "- rejected"
49               frame = frame.f_back
50   
51           gateways.EnumDebugStackFrames.__init__(self, infos, 0)
52

63       def _wrap(self, obj):
64           # This enum returns a tuple, with 2 com objects in it.
65           obFrame, min, lim, fFinal, obFinal = obj
66           obFrame = _wrap(obFrame, axdebug.IID_IDebugStackFrame)
67           if obFinal:
68               obFinal = _wrap(obFinal, pythoncom.IID_IUnknown)
69           return obFrame, min, lim, fFinal, obFinal

72   class DebugStackFrame(gateways.DebugStackFrame):
73       def __init__(self, frame, lineno, codeContainer):
74           self.frame = frame
75           self.lineno = lineno
76           self.codeContainer = codeContainer
77           self.expressionContext = None
78   
79       #       def __del__(self):
80       #               print "DSF dieing"
81       def _query_interface_(self, iid):
82           if iid == axdebug.IID_IDebugExpressionContext:
83               if self.expressionContext is None:
84                   self.expressionContext = _wrap(
85                       expressions.ExpressionContext(self.frame),
86                       axdebug.IID_IDebugExpressionContext,
87                   )
88               return self.expressionContext
89           #               from win32com.util import IIDToInterfaceName
90           #               print "DebugStackFrame QI with %s (%s)" % (IIDToInterfaceName(iid), str(iid))
91           return 0
