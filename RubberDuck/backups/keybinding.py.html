keybinding.py 
1    '''Keybinding 
2    ========== 
3     
4    This module forces the mapping of some keys to functions: 
5     
6    * F11: Rotate the Window through 0, 90, 180 and 270 degrees 
7    * Shift + F11: Switches between portrait and landscape on desktops 
8    * F12: Take a screenshot 
9     
10   Note: this does't work if the application requests the keyboard beforehand. 
11    
12   Usage 
13   ----- 
14    
15   For normal module usage, please see the :mod:`~kivy.modules` documentation. 
16    
17   The Keybinding module, however, can also be imported and used just 
18   like a normal python module. This has the added advantage of being 
19   able to activate and deactivate the module programmatically:: 
20    
21       from kivy.app import App 
22       from kivy.uix.button import Button 
23       from kivy.modules import keybinding 
24       from kivy.core.window import Window 
25    
26       class Demo(App): 
27    
28           def build(self): 
29               button = Button(text="Hello") 
30               keybinding.start(Window, button) 
31               return button 
32    
33       Demo().run() 
34    
35   To remove the Keybinding, you can do the following:: 
36    
37       Keybinding.stop(Window, button) 
38    
39   '''
40   
41   from kivy.utils import platform
42   
43   __all__ = ('start', 'stop')
44   
45   
46   def _on_keyboard_handler(instance, key, scancode, codepoint, modifiers):
47       if key == 293 and modifiers == []:  # F12
48           instance.screenshot()
49       elif key == 292 and modifiers == []:  # F11
50           instance.rotation += 90
51       elif key == 292 and modifiers == ['shift']:  # Shift + F11
52           if platform in ('win', 'linux', 'macosx'):
53               instance.rotation = 0
54               w, h = instance.size
55               w, h = h, w
56               instance.size = (w, h)
57   
58   
59   def start(win, ctx):
60       win.bind(on_keyboard=_on_keyboard_handler)
61   
62   
63   def stop(win, ctx):
64       win.unbind(on_keyboard=_on_keyboard_handler)
