atlas.py 
1    ''' 
2    Atlas 
3    ===== 
4     
5    .. versionadded:: 1.1.0 
6     
7    Atlas manages texture atlases: packing multiple textures into 
8    one. With it, you reduce the number of images loaded and speedup the 
9    application loading. This module contains both the Atlas class and command line 
10   processing for creating an atlas from a set of individual PNG files. The 
11   command line section requires the Pillow library, or the defunct Python Imaging 
12   Library (PIL), to be installed. 
13    
14   An Atlas is composed of 2 or more files: 
15       - a json file (.atlas) that contains the image file names and texture 
16         locations of the atlas. 
17       - one or multiple image files containing textures referenced by the .atlas 
18         file. 
19    
20   Definition of .atlas files 
21   -------------------------- 
22    
23   A file with ``<basename>.atlas`` is a json file formatted like this:: 
24    
25       { 
26           "<basename>-<index>.png": { 
27               "id1": [ <x>, <y>, <width>, <height> ], 
28               "id2": [ <x>, <y>, <width>, <height> ], 
29               # ... 
30           }, 
31           # ... 
32       } 
33    
34   Example from the Kivy ``data/images/defaulttheme.atlas``:: 
35    
36       { 
37           "defaulttheme-0.png": { 
38               "progressbar_background": [431, 224, 59, 24], 
39               "image-missing": [253, 344, 48, 48], 
40               "filechooser_selected": [1, 207, 118, 118], 
41               "bubble_btn": [83, 174, 32, 32], 
42               # ... and more ... 
43           } 
44       } 
45    
46   In this example, "defaulttheme-0.png" is a large image, with the pixels in the 
47   rectangle from (431, 224) to (431 + 59, 224 + 24) usable as 
48   ``atlas://data/images/defaulttheme/progressbar_background`` in 
49   any image parameter. 
50    
51   How to create an Atlas 
52   ---------------------- 
53    
54   .. warning:: 
55    
56       The atlas creation requires the Pillow library (or the defunct Imaging/PIL 
57       library). This requirement will be removed in the future when the Kivy core 
58       Image is able to support loading, blitting, and saving operations. 
59    
60   You can directly use this module to create atlas files with this command:: 
61    
62       $ python -m kivy.atlas <basename> <size> <list of images...> 
63    
64    
65   Let's say you have a list of images that you want to put into an Atlas. The 
66   directory is named ``images`` with lots of 64x64 png files inside:: 
67    
68       $ ls 
69       images 
70       $ cd images 
71       $ ls 
72       bubble.png bubble-red.png button.png button-down.png 
73    
74   You can combine all the png's into one and generate the atlas file with:: 
75    
76       $ python -m kivy.atlas myatlas 256x256 *.png 
77       Atlas created at myatlas.atlas 
78       1 image has been created 
79       $ ls 
80       bubble.png bubble-red.png button.png button-down.png myatlas.atlas 
81       myatlas-0.png 
82    
83   As you can see, we get 2 new files: ``myatlas.atlas`` and ``myatlas-0.png``. 
84   ``myatlas-0.png`` is a new 256x256 .png composed of all your images. If the 
85   size you specify is not large enough to fit all of the source images, more 
86   atlas images will be created as required e.g. ``myatlas-1.png``, 
87   ``myatlas-2.png`` etc. 
88    
89   .. note:: 
90    
91       When using this script, the ids referenced in the atlas are the base names 
92       of the images without the extension. So, if you are going to name a file 
93       ``../images/button.png``, the id for this image will be ``button``. 
94    
95       If you need path information included, you should include ``use_path`` as 
96       follows:: 
97    
98           $ python -m kivy.atlas -- --use_path myatlas 256 *.png 
99    
100      In which case the id for ``../images/button.png`` will be ``images_button`` 
101   
102   
103  How to use an Atlas 
104  ------------------- 
105   
106  Usually, you would specify the images by supplying the path:: 
107   
108      a = Button(background_normal='images/button.png', 
109                 background_down='images/button_down.png') 
110   
111  In our previous example, we have created the atlas containing both images and 
112  put them in ``images/myatlas.atlas``. You can use url notation to reference 
113  them:: 
114   
115      a = Button(background_normal='atlas://images/myatlas/button', 
116                 background_down='atlas://images/myatlas/button_down') 
117   
118  In other words, the path to the images is replaced by:: 
119   
120      atlas://path/to/myatlas/id 
121      # will search for the ``path/to/myatlas.atlas`` and get the image ``id`` 
122   
123  .. note:: 
124   
125      In the atlas url, there is no need to add the ``.atlas`` extension. It will 
126      be automatically append to the filename. 
127   
128  Manual usage of the Atlas 
129  ------------------------- 
130   
131  :: 
132   
133      >>> from kivy.atlas import Atlas 
134      >>> atlas = Atlas('path/to/myatlas.atlas') 
135      >>> print(atlas.textures.keys()) 
136      ['bubble', 'bubble-red', 'button', 'button-down'] 
137      >>> print(atlas['button']) 
138      <kivy.graphics.texture.TextureRegion object at 0x2404d10> 
139  '''

143  import json
144  from os.path import basename, dirname, join, splitext
145  from kivy.event import EventDispatcher
146  from kivy.logger import Logger
147  from kivy.properties import AliasProperty, DictProperty, ListProperty
148  import os
149  
150  
151  # late import to prevent recursion
152  CoreImage = None
153  
154  
155  class Atlas(EventDispatcher):
156      '''Manage texture atlas. See module documentation for more information. 
157      '''
158  
159      original_textures = ListProperty([])
160      '''List of original atlas textures (which contain the :attr:`textures`). 
161   
162      :attr:`original_textures` is a :class:`~kivy.properties.ListProperty` and 
163      defaults to []. 
164   
165      .. versionadded:: 1.9.1 
166      '''
167  
168      textures = DictProperty({})
169      '''List of available textures within the atlas. 
170   
171      :attr:`textures` is a :class:`~kivy.properties.DictProperty` and defaults 
172      to {}. 
173      '''
174  
175      def _get_filename(self):
176          return self._filename
177  
178      filename = AliasProperty(_get_filename, None)
179      '''Filename of the current Atlas. 
180   
181      :attr:`filename` is an :class:`~kivy.properties.AliasProperty` and defaults 
182      to None. 
183      '''
184  
185      def __init__(self, filename):
186          self._filename = filename
187          super(Atlas, self).__init__()
188          self._load()
189  
190      def __getitem__(self, key):
191          return self.textures[key]
192  
193      def _load(self):
194          # late import to prevent recursive import.
195          global CoreImage
196          if CoreImage is None:
197              from kivy.core.image import Image as CoreImage
198  
199          # must be a name finished by .atlas ?
200          filename = self._filename
201          assert(filename.endswith('.atlas'))
202          filename = filename.replace('/', os.sep)
203  
204          Logger.debug('Atlas: Load <%s>' % filename)
205          with open(filename, 'r') as fd:
206              meta = json.load(fd)
207  
208          Logger.debug('Atlas: Need to load %d images' % len(meta))
209          d = dirname(filename)
210          textures = {}
211          for subfilename, ids in meta.items():
212              subfilename = join(d, subfilename)
213              Logger.debug('Atlas: Load <%s>' % subfilename)
214  
215              # load the image
216              ci = CoreImage(subfilename)
217              atlas_texture = ci.texture
218              self.original_textures.append(atlas_texture)
219  
220              # for all the uid, load the image, get the region, and put
221              # it in our dict.
222              for meta_id, meta_coords in ids.items():
223                  x, y, w, h = meta_coords
224                  textures[meta_id] = atlas_texture.get_region(*meta_coords)
225  
226          self.textures = textures
227  
228      @staticmethod
229      def create(outname, filenames, size, padding=2, use_path=False):
230          '''This method can be used to create an atlas manually from a set of 
231          images. 
232   
233          :Parameters: 
234              `outname`: str 
235                  Basename to use for ``.atlas`` creation and ``-<idx>.png`` 
236                  associated images. 
237              `filenames`: list 
238                  List of filenames to put in the atlas. 
239              `size`: int or list (width, height) 
240                  Size of the atlas image. If the size is not large enough to 
241                  fit all of the source images, more atlas images will created 
242                  as required. 
243              `padding`: int, defaults to 2 
244                  Padding to put around each image. 
245   
246                  Be careful. If you're using a padding < 2, you might have 
247                  issues with the borders of the images. Because of the OpenGL 
248                  linearization, it might use the pixels of the adjacent image. 
249   
250                  If you're using a padding >= 2, we'll automatically generate a 
251                  "border" of 1px around your image. If you look at 
252                  the result, don't be scared if the image inside is not 
253                  exactly the same as yours :). 
254   
255              `use_path`: bool, defaults to False 
256                  If True, the relative path of the source png 
257                  file names will be included in the atlas ids rather 
258                  that just in the file names. Leading dots and slashes will be 
259                  excluded and all other slashes in the path will be replaced 
260                  with underscores. For example, if `use_path` is False 
261                  (the default) and the file name is 
262                  ``../data/tiles/green_grass.png``, the id will be 
263                  ``green_grass``. If `use_path` is True, it will be 
264                  ``data_tiles_green_grass``. 
265   
266              .. versionchanged:: 1.8.0 
267                  Parameter use_path added 
268          '''
269          # Thanks to
270          # omnisaurusgames.com/2011/06/texture-atlas-generation-using-python/
271          # for its initial implementation.
272          try:
273              from PIL import Image
274          except ImportError:
275              Logger.critical('Atlas: Imaging/PIL are missing')
276              raise
277  
278          if isinstance(size, (tuple, list)):
279              size_w, size_h = list(map(int, size))
280          else:
281              size_w = size_h = int(size)
282  
283          # open all of the images
284          ims = list()
285          for f in filenames:
286              fp = open(f, 'rb')
287              im = Image.open(fp)
288              im.load()
289              fp.close()
290              ims.append((f, im))
291  
292          # sort by image area
293          ims = sorted(ims, key=lambda im: im[1].size[0] * im[1].size[1],
294                       reverse=True)
295  
296          # free boxes are empty space in our output image set
297          # the freebox tuple format is: outidx, x, y, w, h
298          freeboxes = [(0, 0, 0, size_w, size_h)]
299          numoutimages = 1
300  
301          # full boxes are areas where we have placed images in the atlas
302          # the full box tuple format is: image, outidx, x, y, w, h, filename
303          fullboxes = []
304  
305          # do the actual atlasing by sticking the largest images we can
306          # have into the smallest valid free boxes
307          for imageinfo in ims:
308              im = imageinfo[1]
309              imw, imh = im.size
310              imw += padding
311              imh += padding
312              if imw > size_w or imh > size_h:
313                  Logger.error(
314                      'Atlas: image %s (%d by %d) is larger than the atlas size!'
315                      % (imageinfo[0], imw, imh))
316                  return
317  
318              inserted = False
319              while not inserted:
320                  for idx, fb in enumerate(freeboxes):
321                      # find the smallest free box that will contain this image
322                      if fb[3] >= imw and fb[4] >= imh:
323                          # we found a valid spot! Remove the current
324                          # freebox, and split the leftover space into (up to)
325                          # two new freeboxes
326                          del freeboxes[idx]
327                          if fb[3] > imw:
328                              freeboxes.append((
329                                  fb[0], fb[1] + imw, fb[2],
330                                  fb[3] - imw, imh))
331  
332                          if fb[4] > imh:
333                              freeboxes.append((
334                                  fb[0], fb[1], fb[2] + imh,
335                                  fb[3], fb[4] - imh))
336  
337                          # keep this sorted!
338                          freeboxes = sorted(freeboxes,
339                                             key=lambda fb: fb[3] * fb[4])
340                          fullboxes.append((im,
341                                            fb[0], fb[1] + padding,
342                                            fb[2] + padding, imw - padding,
343                                            imh - padding, imageinfo[0]))
344                          inserted = True
345                          break
346  
347                  if not inserted:
348                      # oh crap - there isn't room in any of our free
349                      # boxes, so we have to add a new output image
350                      freeboxes.append((numoutimages, 0, 0, size_w, size_h))
351                      numoutimages += 1
352  
353          # now that we've figured out where everything goes, make the output
354          # images and blit the source images to the appropriate locations
355          Logger.info('Atlas: create an {0}x{1} rgba image'.format(size_w,
356                                                                   size_h))
357          outimages = [Image.new('RGBA', (size_w, size_h))
358                       for i in range(0, int(numoutimages))]
359          for fb in fullboxes:
360              x, y = fb[2], fb[3]
361              out = outimages[fb[1]]
362              out.paste(fb[0], (fb[2], fb[3]))
363              w, h = fb[0].size
364              if padding > 1:
365                  out.paste(fb[0].crop((0, 0, w, 1)), (x, y - 1))
366                  out.paste(fb[0].crop((0, h - 1, w, h)), (x, y + h))
367                  out.paste(fb[0].crop((0, 0, 1, h)), (x - 1, y))
368                  out.paste(fb[0].crop((w - 1, 0, w, h)), (x + w, y))
369  
370          # save the output images
371          for idx, outimage in enumerate(outimages):
372              outimage.save('%s-%d.png' % (outname, idx))
373  
374          # write out an json file that says where everything ended up
375          meta = {}
376          for fb in fullboxes:
377              fn = '%s-%d.png' % (basename(outname), fb[1])
378              if fn not in meta:
379                  d = meta[fn] = {}
380              else:
381                  d = meta[fn]
382  
383              # fb[6] contain the filename
384              if use_path:
385                  # use the path with separators replaced by _
386                  # example '../data/tiles/green_grass.png' becomes
387                  # 'data_tiles_green_grass'
388                  uid = splitext(fb[6])[0]
389                  # remove leading dots and slashes
390                  uid = uid.lstrip('./\\')
391                  # replace remaining slashes with _
392                  uid = uid.replace('/', '_').replace('\\', '_')
393              else:
394                  # for example, '../data/tiles/green_grass.png'
395                  # just get only 'green_grass' as the uniq id.
396                  uid = splitext(basename(fb[6]))[0]
397  
398              x, y, w, h = fb[2:6]
399              d[uid] = x, size_h - y - h, w, h
400  
401          outfn = '%s.atlas' % outname
402          with open(outfn, 'w') as fd:
403              json.dump(meta, fd)
404  
405          return outfn, meta
406  
407  
408  if __name__ == '__main__':
409      """ Main line program. Process command line arguments 
410      to make a new atlas. """
411  
412      import sys
413      from glob import glob
414      argv = sys.argv[1:]
415      # earlier import of kivy has already called getopt to remove kivy system
416      # arguments from this line. That is all arguments up to the first '--'
417      if len(argv) < 3:
418          print('Usage: python -m kivy.atlas [-- [--use-path] '
419                '[--padding=2]] <outname> '
420                '<size|512x256> <img1.png> [<img2.png>, ...]')
421          sys.exit(1)
422  
423      options = {'use_path': False}
424      while True:
425          option = argv[0]
426          if option == '--use-path':
427              options['use_path'] = True
428          elif option.startswith('--padding='):
429              options['padding'] = int(option.split('=', 1)[-1])
430          elif option[:2] == '--':
431              print('Unknown option {}'.format(option))
432              sys.exit(1)
433          else:
434              break
435          argv = argv[1:]
436  
437      outname = argv[0]
438      try:
439          if 'x' in argv[1]:
440              size = list(map(int, argv[1].split('x', 1)))
441          else:
442              size = int(argv[1])
443      except ValueError:
444          print('Error: size must be an integer or <integer>x<integer>')
445          sys.exit(1)
446  
447      filenames = [fname for fnames in argv[2:] for fname in glob(fnames)]
448      ret = Atlas.create(outname, filenames, size, **options)
449      if not ret:
450          print('Error while creating atlas!')
451          sys.exit(1)
452  
453      fn, meta = ret
454      print('Atlas created at', fn)
455      print('%d image%s been created' % (len(meta),
456            's have' if len(meta) > 1 else ' has'))
