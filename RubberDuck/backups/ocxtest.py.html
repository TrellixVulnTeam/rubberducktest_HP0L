115  def GetTestVideoModule():
116      global videoControlModule, videoControlFileName
117      win32ui.DoWaitCursor(1)
118      videoControlModule = gencache.EnsureModule(
119          "{05589FA0-C356-11CE-BF01-00AA0055595A}", 0, 2, 0
120      )
121      win32ui.DoWaitCursor(0)
122      if videoControlModule is None:
123          return None
124      fnames = glob.glob(os.path.join(win32api.GetWindowsDirectory(), "*.avi"))
125      if not fnames:
126          print("No AVI files available in system directory")
127          return None
128      videoControlFileName = fnames[0]
129      return videoControlModule

170  class OCXFrame(window.MDIChildWnd):
171      def __init__(self):
172          pass  # Dont call base class doc/view version...
173  
174      def Create(self, controlClass, title, rect=None, parent=None):
175          style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.WS_OVERLAPPEDWINDOW
176          self._obj_ = win32ui.CreateMDIChild()
177          self._obj_.AttachObject(self)
178          self._obj_.CreateWindow(None, title, style, rect, parent)
179  
180          rect = self.GetClientRect()
181          rect = (0, 0, rect[2] - rect[0], rect[3] - rect[1])
182          self.ocx = controlClass()
183          self.ocx.CreateControl(
184              "", win32con.WS_VISIBLE | win32con.WS_CHILD, rect, self, 1000
185          )
