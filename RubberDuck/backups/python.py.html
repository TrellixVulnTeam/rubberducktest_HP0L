67       def innerstring_rules(ttype):
68           return [
69               # the old style '%s' % (...) string formatting (still valid in Py3)
70               (r'%(\(\w+\))?[-#0 +]*([0-9]+|[*])?(\.([0-9]+|[*]))?'
71                '[hlL]?[E-GXc-giorsaux%]', String.Interpol),
72               # the new style '{}'.format(...) string formatting
73               (r'\{'
74                r'((\w+)((\.\w+)|(\[[^\]]+\]))*)?'  # field name
75                r'(\![sra])?'                       # conversion
76                r'(\:(.?[<>=\^])?[-+ ]?#?0?(\d+)?,?(\.\d+)?[E-GXb-gnosx%]?)?'
77                r'\}', String.Interpol),
78   
79               # backslashes, quotes and formatting signs must be parsed one at a time
80               (r'[^\\\'"%{\n]+', ttype),
81               (r'[\'"\\]', ttype),
82               # unhandled string formatting sign
83               (r'%|(\{{1,2})', ttype)
84               # newlines are an error (use "nl" state)
85           ]

87       def fstring_rules(ttype):
88           return [
89               # Assuming that a '}' is the closing brace after format specifier.
90               # Sadly, this means that we won't detect syntax error. But it's
91               # more important to parse correct syntax correctly, than to
92               # highlight invalid syntax.
93               (r'\}', String.Interpol),
94               (r'\{', String.Interpol, 'expr-inside-fstring'),
95               # backslashes, quotes and formatting signs must be parsed one at a time
96               (r'[^\\\'"{}\n]+', ttype),
97               (r'[\'"\\]', ttype),
98               # newlines are an error (use "nl" state)
99           ]

609  class PythonConsoleLexer(Lexer):
610      """ 
611      For Python console output or doctests, such as: 
612   
613      .. sourcecode:: pycon 
614   
615          >>> a = 'foo' 
616          >>> print a 
617          foo 
618          >>> 1 / 0 
619          Traceback (most recent call last): 
620            File "<stdin>", line 1, in <module> 
621          ZeroDivisionError: integer division or modulo by zero 
622   
623      Additional options: 
624   
625      `python3` 
626          Use Python 3 lexer for code.  Default is ``True``. 
627   
628          .. versionadded:: 1.0 
629          .. versionchanged:: 2.5 
630             Now defaults to ``True``. 
631      """
632      name = 'Python console session'
633      aliases = ['pycon']
634      mimetypes = ['text/x-python-doctest']
635  
636      def __init__(self, **options):
637          self.python3 = get_bool_opt(options, 'python3', True)
638          Lexer.__init__(self, **options)
639  
640      def get_tokens_unprocessed(self, text):
641          if self.python3:
642              pylexer = PythonLexer(**self.options)
643              tblexer = PythonTracebackLexer(**self.options)
644          else:
645              pylexer = Python2Lexer(**self.options)
646              tblexer = Python2TracebackLexer(**self.options)
647  
648          curcode = ''
649          insertions = []
650          curtb = ''
651          tbindex = 0
652          tb = 0
653          for match in line_re.finditer(text):
654              line = match.group()
655              if line.startswith('>>> ') or line.startswith('... '):
656                  tb = 0
657                  insertions.append((len(curcode),
658                                     [(0, Generic.Prompt, line[:4])]))
659                  curcode += line[4:]
660              elif line.rstrip() == '...' and not tb:
661                  # only a new >>> prompt can end an exception block
662                  # otherwise an ellipsis in place of the traceback frames
663                  # will be mishandled
664                  insertions.append((len(curcode),
665                                     [(0, Generic.Prompt, '...')]))
666                  curcode += line[3:]
667              else:
668                  if curcode:
669                      yield from do_insertions(
670                          insertions, pylexer.get_tokens_unprocessed(curcode))
671                      curcode = ''
672                      insertions = []
673                  if (line.startswith('Traceback (most recent call last):') or
674                          re.match('  File "[^"]+", line \\d+\\n$', line)):
675                      tb = 1
676                      curtb = line
677                      tbindex = match.start()
678                  elif line == 'KeyboardInterrupt\n':
679                      yield match.start(), Name.Class, line
680                  elif tb:
681                      curtb += line
682                      if not (line.startswith(' ') or line.strip() == '...'):
683                          tb = 0
684                          for i, t, v in tblexer.get_tokens_unprocessed(curtb):
685                              yield tbindex+i, t, v
686                          curtb = ''
687                  else:
688                      yield match.start(), Generic.Output, line
689          if curcode:
690              yield from do_insertions(insertions,
691                                       pylexer.get_tokens_unprocessed(curcode))
692          if curtb:
693              for i, t, v in tblexer.get_tokens_unprocessed(curtb):
694                  yield tbindex+i, t, v
