19               return

27           basepathb = join(basepathb, b'filechooser_files')

64       def test_filechooserlistview_unicode(self):
65           if self.skip_test:
66               return
67           from kivy.uix.filechooser import FileChooserListView
68           from kivy.clock import Clock
69           from os.path import join
70   
71           wid = FileChooserListView(path=self.basepathu)
72           for i in range(1):
73               Clock.tick()
74           files = [join(self.basepathu, f) for f in wid.files]
75           for f in self.ufiles:
76               self.assertIn(f, files)
77           # we cannot test the bfiles because we'd have to know the system
78           # unicode encoding to be able to compare to returned unicode
79           for f in self.exitsfiles:
80               self.assertIn(f, files)
