13   class VeryPermissive:
14       def _dynamic_(self, name, lcid, wFlags, args):
15           if wFlags & pythoncom.DISPATCH_METHOD:
16               return getattr(self, name)(*args)
17   
18           if wFlags & pythoncom.DISPATCH_PROPERTYGET:
19               try:
20                   # to avoid problems with byref param handling, tuple results are converted to lists.
21                   ret = self.__dict__[name]
22                   if type(ret) == type(()):
23                       ret = list(ret)
24                   return ret
25               except KeyError:  # Probably a method request.
26                   raise Exception(scode=winerror.DISP_E_MEMBERNOTFOUND)
27   
28           if wFlags & (
29               pythoncom.DISPATCH_PROPERTYPUT | pythoncom.DISPATCH_PROPERTYPUTREF
30           ):
31               setattr(self, name, args[0])
32               return
33   
34           raise Exception(scode=winerror.E_INVALIDARG, desc="invalid wFlags")
35   
36       def write(self, *args):
37           if len(args) == 0:
38               raise Exception(
39                   scode=winerror.DISP_E_BADPARAMCOUNT
40               )  # Probably call as PROPGET.
41   
42           for arg in args[:-1]:
43               print(str(arg), end=" ")
44           print(str(args[-1]))
