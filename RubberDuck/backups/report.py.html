10   import os
11   import sys
12   import platform as plf
13   from time import ctime
14   from configparser import ConfigParser
15   from io import StringIO
16   
17   import kivy
18   
19   report = []
20   report_dict = {}  # One key value pair for each title.
21   
22   
23   def title(t):
24       report.append('')
25       report.append('=' * 80)
26       report.append(t)
27       report.append('=' * 80)
28       report.append('')
29   
30   # This method sends report to gist(Different file in a single gist) and
31   # returns the URL
32   
33   
34   def send_report(dict_report):
35       import requests
36       import json
37   
38       gist_report = {
39           "description": "Report",
40           "public": "true",
41           "files": {
42               "Global.txt": {
43                   "content": "\n".join(dict_report['Global']),
44                   "type": 'text'
45               },
46               "OpenGL.txt": {
47                   "content": "\n".join(dict_report['OpenGL']),
48                   "type": 'text'
49   
50               },
51               "Core selection.txt": {
52                   "content": "\n".join(dict_report['Core']),
53                   "type": 'text'
54               },
55               "Libraries.txt": {
56                   "content": "\n".join(dict_report['Libraries']),
57                   "type": 'text'
58               },
59               "Configuration.txt": {
60                   "content": "\n".join(dict_report['Configuration']),
61                   "type": 'text'
62               },
63               "Input Availablity.txt": {
64                   "content": "\n".join(dict_report['InputAvailablity']),
65                   "type": 'text'
66               },
67               "Environ.txt": {
68                   "content": "\n".join(dict_report['Environ']),
69                   "type": 'text'
70               },
71               "Options.txt": {
72                   "content": "\n".join(dict_report['Options']),
73                   "type": 'text'
74               },
75           }
76       }
77       report_json = json.dumps(gist_report)
78       response = requests.post("https://api.github.com/gists", report_json)
79       return json.loads(response.text)['html_url']

86   title('Global')
87   report.append('OS platform     : %s | %s' % (plf.platform(), plf.machine()))
88   report.append('Python EXE      : %s' % sys.executable)
89   report.append('Python Version  : %s' % sys.version)
90   report.append('Python API      : %s' % sys.api_version)
91   report.append('Kivy Version    : %s' % kivy.__version__)
92   report.append('Install path    : %s' % os.path.dirname(kivy.__file__))
93   report.append('Install date    : %s' % ctime(os.path.getctime(kivy.__file__)))
94   report_dict['Global'] = report
95   report = []
96   
97   title('OpenGL')
98   from kivy.core import gl
99   from kivy.core.window import Window
100  report.append('GL Vendor: %s' % gl.glGetString(gl.GL_VENDOR))
101  report.append('GL Renderer: %s' % gl.glGetString(gl.GL_RENDERER))
102  report.append('GL Version: %s' % gl.glGetString(gl.GL_VERSION))
103  ext = None
104  try:
105      gl.glGetString(gl.GL_EXTENSIONS)
106  except AttributeError:
107      pass
108  if ext is None:
109      report.append('GL Extensions: %s' % ext)
110  else:
111      report.append('GL Extensions:')
112      for x in ext.split():
113          report.append('\t%s' % x)
114  Window.close()
115  report_dict['OpenGL'] = report
116  report = []
117  
118  title('Core selection')
119  from kivy.core.audio import SoundLoader
120  report.append('Audio  = %s' % SoundLoader._classes)
121  from kivy.core.camera import Camera
122  report.append('Camera = %s' % Camera)
123  from kivy.core.image import ImageLoader
124  report.append('Image  = %s' % ImageLoader.loaders)
125  from kivy.core.text import Label
126  report.append('Text   = %s' % Label)
127  from kivy.core.video import Video
128  report.append('Video  = %s' % Video)
129  report.append('Window = %s' % Window)
130  report_dict['Core'] = report
131  report = []
132  
133  title('Libraries')
134  
135  
136  def testimport(libname):
137      try:
138          lib = __import__(libname)
139          report.append('%-20s exist at %s' % (libname, lib.__file__))
140      except ImportError:
141          report.append('%-20s is missing' % libname)
142  
143  
144  for x in ('gst',
145            'pygame',
146            'pygame.midi',
147            'squirtle',
148            'PIL',
149            'sdl2',
150            'glew',
151            'opencv',
152            'opencv.cv',
153            'opencv.highgui',
154            'cython'):
155      testimport(x)
156  report_dict['Libraries'] = report
157  report = []
158  
159  title('Configuration')
160  s = StringIO()
161  from kivy.config import Config
162  ConfigParser.write(Config, s)
163  report.extend(s.getvalue().split('\n'))
164  report_dict['Configuration'] = report
165  report = []
166  
167  title('Input availability')
168  from kivy.input.factory import MotionEventFactory
169  for x in MotionEventFactory.list():
170      report.append(x)
171  report_dict['InputAvailablity'] = report
172  report = []
173  
174  ''' 
175  title('Log') 
176  for x in pymt_logger_history.history: 
177      report.append(x.message) 
178  '''
179  
180  title('Environ')
181  for k, v in os.environ.items():
182      report.append('%s = %s' % (k, v))
183  report_dict['Environ'] = report
184  report = []
185  
186  title('Options')
187  for k, v in kivy.kivy_options.items():
188      report.append('%s = %s' % (k, v))
189  report_dict['Options'] = report
190  report = []
191  
192  # Prints the entire Output
193  print('\n'.join(report_dict['Global'] + report_dict['OpenGL'] +
194                  report_dict['Core'] + report_dict['Libraries'] +
195                  report_dict['Configuration'] +
196                  report_dict['InputAvailablity'] +
197                  report_dict['Environ'] + report_dict['Options']))
198  print('\n')
199  print('\n')
200  
201  try:
202      print('The report will be sent as an anonymous gist.')
203      reply = input(
204          'Do you accept to send report to https://gist.github.com/ (Y/n) : ')
205  except EOFError:
206      sys.exit(0)
207  
208  if reply.lower().strip() in ('', 'y'):
209      print('Please wait while sending the report...')
210  
211      paste_url = send_report(report_dict)
212  
213      print('\n')
214      print('\n')
215      print('REPORT posted at %s' % paste_url)
216      print('\n')
217      print('\n')
218  else:
219      print('No report posted.')
220

222  # of the dump. That's not cool if we want get report url
