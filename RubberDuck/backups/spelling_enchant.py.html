1    ''' 
2    Enchant Spelling 
3    ================ 
4     
5    Implementation spelling backend based on enchant. 
6     
7    .. warning:: pyenchant doesn't have dedicated build anymore for Windows/x64. 
8                 See https://github.com/kivy/kivy/issues/5816 for more informations 
9    '''
10   
11   
12   import enchant
13   
14   from kivy.core.spelling import SpellingBase, NoSuchLangError
15   from kivy.compat import PY2
16   
17   
18   class SpellingEnchant(SpellingBase):
19       ''' 
20       Spelling backend based on the enchant library. 
21       '''
22   
23       def __init__(self, language=None):
24           self._language = None
25           super(SpellingEnchant, self).__init__(language)
26   
27       def select_language(self, language):
28           try:
29               self._language = enchant.Dict(language)
30           except enchant.DictNotFoundError:
31               err = 'Enchant Backend: No language for "%s"' % (language, )
32               raise NoSuchLangError(err)
33   
34       def list_languages(self):
35           # Note: We do NOT return enchant.list_dicts because that also returns
36           #       the enchant dict objects and not only the language identifiers.
37           return enchant.list_languages()
38   
39       def check(self, word):
40           if not word:
41               return None
42           return self._language.check(word)
43   
44       def suggest(self, fragment):
45           suggestions = self._language.suggest(fragment)
46           # Don't show suggestions that are invalid
47           suggestions = [s for s in suggestions if self.check(s)]
48           if PY2:
49               suggestions = [s.decode('utf-8') for s in suggestions]
50           return suggestions
