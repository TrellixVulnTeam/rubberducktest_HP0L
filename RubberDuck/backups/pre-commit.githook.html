pre-commit.githook 
1    #!/usr/bin/env python
2    ''' 
3        Kivy Git Pre-Commit Hook to Enforce Styleguide 
4        ============================================== 
5     
6        This script is not supposed to be run directly. 
7        Instead, copy it to your kivy/.git/hooks/ directory, call it 'pre-commit' 
8        and make it executable. 
9     
10       If you attempt to commit, git will run this script, which in turn will run 
11       the styleguide checker over your code and abort the commit if there are any 
12       errors. If that happens, please fix & retry. 
13    
14       To install:: 
15    
16           cp kivy/tools/pep8checker/pre-commit.githook .git/hooks/pre-commit 
17           chmod +x .git/hooks/pre-commit 
18   '''
19   
20   import os
21   import sys
22   from os.path import dirname, abspath, sep, join
23   from subprocess import call, Popen, PIPE
24   
25   curdir = dirname(abspath(__file__))
26   kivydir = sep.join(curdir.split(sep)[:-2])
27   srcdir = join(kivydir, 'kivy')
28   script = join(srcdir, 'tools', 'pep8checker', 'pep8kivy.py')
29   try:
30       with open(script):
31           pass
32   except IOError:
33       # if this not the kivy project, find the script file in the kivy project
34       os.environ['KIVY_NO_CONSOLELOG'] = '1'
35       import kivy
36       script = join(dirname(kivy.__file__), 'tools', 'pep8checker', 'pep8kivy.py')
37       srcdir = ''
38   
39   # Only check the files that were staged
40   # proc = Popen(['git', 'diff', '--cached', '--name-only', 'HEAD'], stdout=PIPE)
41   # targets = [join(kivydir, target) for target in proc.stdout]
42   
43   # Correction: only check the files that were staged, but do not include
44   # deleted files.
45   proc = Popen(['git', 'diff', '--cached', '--name-status', 'HEAD'], stdout=PIPE)
46   proc.wait()
47   
48   # This gives output like the following:
49   #
50   #    A       examples/widgets/lists/list_simple_in_kv.py
51   #    A       examples/widgets/lists/list_simple_in_kv_2.py
52   #    D       kivy/uix/observerview.py
53   #
54   # So check for D entries and remove them from targets.
55   #
56   targets = []
57   for target in proc.stdout:
58       parts = [p.strip() for p in target.split()]
59       if parts[0] != 'D':
60           targets.append(join(kivydir, target.decode(encoding='UTF-8')))
61   
62   # Untested possibility: After making the changes above for removing deleted
63   # files from targets, saw also where the git diff call could be:
64   #
65   #    git diff --cached --name-only --diff-filter=ACM
66   #                                                    (leaving off D)
67   #
68   # and we could then remove the special handling in python for targets above.
69   
70   call(['git', 'stash', 'save', '--keep-index', '--quiet'])
71   retval = call([sys.executable, script, srcdir] + targets)
72   call(['git', 'stash', 'pop', '--quiet'])
73   
74   if retval:
75       # There are style guide violations
76       print("Your commit has been aborted. Please fix the violations and retry.")
77       sys.exit(1)
