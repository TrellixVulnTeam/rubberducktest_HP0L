24   class ShellTester(win32com.test.util.TestCase):
25       def testShellLink(self):
26           desktop = str(shell.SHGetSpecialFolderPath(0, CSIDL_DESKTOP))
27           num = 0
28           shellLink = pythoncom.CoCreateInstance(
29               shell.CLSID_ShellLink,
30               None,
31               pythoncom.CLSCTX_INPROC_SERVER,
32               shell.IID_IShellLink,
33           )
34           persistFile = shellLink.QueryInterface(pythoncom.IID_IPersistFile)
35           names = [os.path.join(desktop, n) for n in os.listdir(desktop)]
36           programs = str(shell.SHGetSpecialFolderPath(0, CSIDL_PROGRAMS))
37           names.extend([os.path.join(programs, n) for n in os.listdir(programs)])
38           for name in names:
39               try:
40                   persistFile.Load(name, STGM_READ)
41               except pythoncom.com_error:
42                   continue
43               # Resolve is slow - avoid it for our tests.
44               # shellLink.Resolve(0, shell.SLR_ANY_MATCH | shell.SLR_NO_UI)
45               fname, findData = shellLink.GetPath(0)
46               unc = shellLink.GetPath(shell.SLGP_UNCPRIORITY)[0]
47               num += 1
48           if num == 0:
49               # This isn't a fatal error, but is unlikely.
50               print(
51                   "Could not find any links on your desktop or programs dir, which is unusual"
52               )

54       def testShellFolder(self):
55           sf = shell.SHGetDesktopFolder()
56           names_1 = []
57           for i in sf:  # Magically calls EnumObjects
58               name = sf.GetDisplayNameOf(i, SHGDN_NORMAL)
59               names_1.append(name)
60   
61           # And get the enumerator manually
62           enum = sf.EnumObjects(
63               0, SHCONTF_FOLDERS | SHCONTF_NONFOLDERS | SHCONTF_INCLUDEHIDDEN
64           )
65           names_2 = []
66           for i in enum:
67               name = sf.GetDisplayNameOf(i, SHGDN_NORMAL)
68               names_2.append(name)
69           names_1.sort()
70           names_2.sort()
71           self.assertEqual(names_1, names_2)

75       def _rtPIDL(self, pidl):
76           pidl_str = shell.PIDLAsString(pidl)
77           pidl_rt = shell.StringAsPIDL(pidl_str)
78           self.assertEqual(pidl_rt, pidl)
79           pidl_str_rt = shell.PIDLAsString(pidl_rt)
80           self.assertEqual(pidl_str_rt, pidl_str)

84           cida_str = shell.CIDAAsString(cida)
85           cida_rt = shell.StringAsCIDA(cida_str)
86           self.assertEqual(cida, cida_rt)
87           cida_str_rt = shell.CIDAAsString(cida_rt)

90       def testPIDL(self):
91           # A PIDL of "\1" is:   cb    pidl   cb
92           expect = str2bytes("\03\00" "\1" "\0\0")
93           self.assertEqual(shell.PIDLAsString([str2bytes("\1")]), expect)
94           self._rtPIDL([str2bytes("\0")])
95           self._rtPIDL([str2bytes("\1"), str2bytes("\2"), str2bytes("\3")])
96           self._rtPIDL([str2bytes("\0") * 2048] * 2048)
97           # PIDL must be a list
98           self.assertRaises(TypeError, shell.PIDLAsString, "foo")

116  class FILEGROUPDESCRIPTORTester(win32com.test.util.TestCase):
117      def _getTestTimes(self):
118          if issubclass(pywintypes.TimeType, datetime.datetime):
119              ctime = win32timezone.now()
120              # FILETIME only has ms precision...
121              ctime = ctime.replace(microsecond=ctime.microsecond // 1000 * 1000)
122              atime = ctime + datetime.timedelta(seconds=1)
123              wtime = atime + datetime.timedelta(seconds=1)
124          else:
125              ctime = pywintypes.Time(11)
126              atime = pywintypes.Time(12)
127              wtime = pywintypes.Time(13)
128          return ctime, atime, wtime
129  
130      def _testRT(self, fd):
131          fgd_string = shell.FILEGROUPDESCRIPTORAsString([fd])
132          fd2 = shell.StringAsFILEGROUPDESCRIPTOR(fgd_string)[0]
133  
134          fd = fd.copy()
135          fd2 = fd2.copy()
136  
137          # The returned objects *always* have dwFlags and cFileName.
138          if "dwFlags" not in fd:
139              del fd2["dwFlags"]
140          if "cFileName" not in fd:
141              self.assertEqual(fd2["cFileName"], "")
142              del fd2["cFileName"]
143  
144          self.assertEqual(fd, fd2)

146      def _testSimple(self, make_unicode):
147          fgd = shell.FILEGROUPDESCRIPTORAsString([], make_unicode)
148          header = struct.pack("i", 0)
149          self.assertEqual(header, fgd[: len(header)])
150          self._testRT(dict())
151          d = dict()
152          fgd = shell.FILEGROUPDESCRIPTORAsString([d], make_unicode)
153          header = struct.pack("i", 1)
154          self.assertEqual(header, fgd[: len(header)])
155          self._testRT(d)

222  class FileOperationTester(win32com.test.util.TestCase):
223      def setUp(self):
224          import tempfile
225  
226          self.src_name = os.path.join(tempfile.gettempdir(), "pywin32_testshell")
227          self.dest_name = os.path.join(tempfile.gettempdir(), "pywin32_testshell_dest")
228          self.test_data = str2bytes("Hello from\0Python")
229          f = open(self.src_name, "wb")
230          f.write(self.test_data)
231          f.close()
232          try:
233              os.unlink(self.dest_name)
234          except os.error:
235              pass
236  
237      def tearDown(self):
238          for fname in (self.src_name, self.dest_name):
239              if os.path.isfile(fname):
240                  os.unlink(fname)
241  
242      def testCopy(self):
243          s = (0, FO_COPY, self.src_name, self.dest_name)  # hwnd,  # operation
244  
245          rc, aborted = shell.SHFileOperation(s)
246          self.failUnless(not aborted)
247          self.failUnlessEqual(0, rc)
248          self.failUnless(os.path.isfile(self.src_name))
249          self.failUnless(os.path.isfile(self.dest_name))
