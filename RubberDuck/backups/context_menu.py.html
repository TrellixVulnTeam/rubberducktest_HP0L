1    # A sample context menu handler.
2    # Adds a 'Hello from Python' menu entry to .py files.  When clicked, a
3    # simple message box is displayed.
4    #
5    # To demostrate:
6    # * Execute this script to register the context menu.
7    # * Open Windows Explorer, and browse to a directory with a .py file.
8    # * Right-Click on a .py file - locate and click on 'Hello from Python' on
9    #   the context menu.
10   
11   import pythoncom
12   from win32com.shell import shell, shellcon
13   import win32gui
14   import win32con
15   
16   
17   class ShellExtension:
18       _reg_progid_ = "Python.ShellExtension.ContextMenu"
19       _reg_desc_ = "Python Sample Shell Extension (context menu)"
20       _reg_clsid_ = "{CED0336C-C9EE-4a7f-8D7F-C660393C381F}"
21       _com_interfaces_ = [shell.IID_IShellExtInit, shell.IID_IContextMenu]
22       _public_methods_ = shellcon.IContextMenu_Methods + shellcon.IShellExtInit_Methods
23   
24       def Initialize(self, folder, dataobj, hkey):
25           print("Init", folder, dataobj, hkey)
26           self.dataobj = dataobj
27   
28       def QueryContextMenu(self, hMenu, indexMenu, idCmdFirst, idCmdLast, uFlags):
29           print("QCM", hMenu, indexMenu, idCmdFirst, idCmdLast, uFlags)
30           # Query the items clicked on
31           format_etc = win32con.CF_HDROP, None, 1, -1, pythoncom.TYMED_HGLOBAL
32           sm = self.dataobj.GetData(format_etc)
33           num_files = shell.DragQueryFile(sm.data_handle, -1)
34           if num_files > 1:
35               msg = "&Hello from Python (with %d files selected)" % num_files
36           else:
37               fname = shell.DragQueryFile(sm.data_handle, 0)
38               msg = "&Hello from Python (with '%s' selected)" % fname
39           idCmd = idCmdFirst
40           items = ["First Python content menu item"]
41           if (
42               uFlags & 0x000F
43           ) == shellcon.CMF_NORMAL:  # Check == here, since CMF_NORMAL=0
44               print("CMF_NORMAL...")
45               items.append(msg)
46           elif uFlags & shellcon.CMF_VERBSONLY:
47               print("CMF_VERBSONLY...")
48               items.append(msg + " - shortcut")
49           elif uFlags & shellcon.CMF_EXPLORE:
50               print("CMF_EXPLORE...")
51               items.append(msg + " - normal file, right-click in Explorer")
52           elif uFlags & CMF_DEFAULTONLY:
53               print("CMF_DEFAULTONLY...\r\n")
54           else:
55               print("** unknown flags", uFlags)
56           win32gui.InsertMenu(
57               hMenu, indexMenu, win32con.MF_SEPARATOR | win32con.MF_BYPOSITION, 0, None
58           )
59           indexMenu += 1
60           for item in items:
61               win32gui.InsertMenu(
62                   hMenu,
63                   indexMenu,
64                   win32con.MF_STRING | win32con.MF_BYPOSITION,
65                   idCmd,
66                   item,
67               )
68               indexMenu += 1
69               idCmd += 1
70   
71           win32gui.InsertMenu(
72               hMenu, indexMenu, win32con.MF_SEPARATOR | win32con.MF_BYPOSITION, 0, None
73           )
74           indexMenu += 1
75           return idCmd - idCmdFirst  # Must return number of menu items we added.
76   
77       def InvokeCommand(self, ci):
78           mask, hwnd, verb, params, dir, nShow, hotkey, hicon = ci
79           win32gui.MessageBox(hwnd, "Hello", "Wow", win32con.MB_OK)
80   
81       def GetCommandString(self, cmd, typ):
82           # If GetCommandString returns the same string for all items then
83           # the shell seems to ignore all but one.  This is even true in
84           # Win7 etc where there is no status bar (and hence this string seems
85           # ignored)
86           return "Hello from Python (cmd=%d)!!" % (cmd,)

89   def DllRegisterServer():
90       import winreg
91   
92       key = winreg.CreateKey(winreg.HKEY_CLASSES_ROOT, "Python.File\\shellex")
93       subkey = winreg.CreateKey(key, "ContextMenuHandlers")
94       subkey2 = winreg.CreateKey(subkey, "PythonSample")
95       winreg.SetValueEx(subkey2, None, 0, winreg.REG_SZ, ShellExtension._reg_clsid_)
96       print(ShellExtension._reg_desc_, "registration complete.")
97   
98   
99   def DllUnregisterServer():
100      import winreg
101  
102      try:
103          key = winreg.DeleteKey(
104              winreg.HKEY_CLASSES_ROOT,
105              "Python.File\\shellex\\ContextMenuHandlers\\PythonSample",
106          )
107      except WindowsError as details:
108          import errno
109  
110          if details.errno != errno.ENOENT:
111              raise
112      print(ShellExtension._reg_desc_, "unregistration complete.")
113  
114  
115  if __name__ == "__main__":
116      from win32com.server import register
117  
118      register.UseCommandLine(
119          ShellExtension,
120          finalize_register=DllRegisterServer,
121          finalize_unregister=DllUnregisterServer,
122      )
