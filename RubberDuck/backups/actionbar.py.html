1    ''' 
2    Action Bar 
3    ========== 
4     
5    .. versionadded:: 1.8.0 
6     
7    .. image:: images/actionbar.png 
8        :align: right 
9     
10   The ActionBar widget is like Android's `ActionBar 
11   <http://developer.android.com/guide/topics/ui/actionbar.html>`_, where items 
12   are stacked horizontally. When the area becomes to small, widgets are moved 
13   into the :class:`ActionOverflow` area. 
14    
15   An :class:`ActionBar` contains an :class:`ActionView` with various 
16   :class:`ContextualActionViews <kivy.uix.actionbar.ContextualActionView>`. 
17   An :class:`ActionView` will contain an :class:`ActionPrevious` having title, 
18   app_icon and previous_icon properties. An :class:`ActionView` will contain 
19   subclasses of :class:`ActionItems <ActionItem>`. Some predefined ones include 
20   an :class:`ActionButton`, an :class:`ActionToggleButton`, an 
21   :class:`ActionCheck`, an :class:`ActionSeparator` and an :class:`ActionGroup`. 
22    
23   An :class:`ActionGroup` is used to display :class:`ActionItems <ActionItem>` 
24   in a group. An :class:`ActionView` will always display an :class:`ActionGroup` 
25   after other :class:`ActionItems <ActionItem>`. An :class:`ActionView` contains 
26   an :class:`ActionOverflow`, but this is only made visible when required i.e. 
27   the available area is too small to fit all the widgets. A 
28   :class:`ContextualActionView` is a subclass of an:class:`ActionView`. 
29    
30   .. versionchanged:: 1.10.1 
31       :class:`ActionGroup` core rewritten from :class:`Spinner` to pure 
32       :class:`DropDown` 
33   '''

69   class ActionItem(object):
70       ''' 
71       ActionItem class, an abstract class for all ActionBar widgets. To create a 
72       custom widget for an ActionBar, inherit from this class. See module 
73       documentation for more information. 
74       '''
75   
76       minimum_width = NumericProperty('90sp')
77       ''' 
78       Minimum Width required by an ActionItem. 
79    
80       :attr:`minimum_width` is a :class:`~kivy.properties.NumericProperty` and 
81       defaults to '90sp'. 
82       '''
83   
84       def get_pack_width(self):
85           return max(self.minimum_width, self.width)
86   
87       pack_width = AliasProperty(get_pack_width,
88                                  bind=('minimum_width', 'width'),
89                                  cache=True)
90       ''' 
91       (read-only) The actual width to use when packing the items. Equal to the 
92       greater of minimum_width and width. 
93    
94       :attr:`pack_width` is an :class:`~kivy.properties.AliasProperty`. 
95       '''
96   
97       important = BooleanProperty(False)
98       ''' 
99       Determines if an ActionItem is important or not. If an item is important 
100      and space is limited, this item will be displayed in preference to others. 
101   
102      :attr:`important` is a :class:`~kivy.properties.BooleanProperty` and 
103      defaults to False. 
104      '''
105  
106      inside_group = BooleanProperty(False)
107      ''' 
108      (internal) Determines if an ActionItem is displayed inside an 
109      ActionGroup or not. 
110   
111      :attr:`inside_group` is a :class:`~kivy.properties.BooleanProperty` and 
112      defaults to False. 
113      '''

116          'atlas://data/images/defaulttheme/action_item')

146      ''' 
147      ActionButton class, see module documentation for more information. 
148   
149      The text color, width and size_hint_x are set manually via the Kv language 
150      file. It covers a lot of cases: with/without an icon, with/without a group 
151      and takes care of the padding between elements.

153      You don't have much control over these properties, so if you want to 
154      customize its appearance, we suggest you create you own button

156      existing widget and an :class:`ActionItem`::

170      to None.

331  class ActionGroup(ActionItem, Button):
332      ''' 
333      ActionGroup class, see module documentation for more information. 
334      '''
335  
336      use_separator = BooleanProperty(False)
337      ''' 
338      Specifies whether to use a separator after/before this group or not. 
339   
340      :attr:`use_separator` is a :class:`~kivy.properties.BooleanProperty` and 
341      defaults to False. 
342      '''
343  
344      separator_image = StringProperty(
345          'atlas://data/images/defaulttheme/separator')
346      ''' 
347      Background Image for an ActionSeparator in an ActionView. 
348   
349      :attr:`separator_image` is a :class:`~kivy.properties.StringProperty` 
350      and defaults to 'atlas://data/images/defaulttheme/separator'. 
351      '''
352  
353      separator_width = NumericProperty(0)
354      ''' 
355      Width of the ActionSeparator in an ActionView. 
356   
357      :attr:`separator_width` is a :class:`~kivy.properties.NumericProperty` 
358      and defaults to 0. 
359      '''
360  
361      mode = OptionProperty('normal', options=('normal', 'spinner'))
362      ''' 
363      Sets the current mode of an ActionGroup. If mode is 'normal', the 
364      ActionGroups children will be displayed normally if there is enough 
365      space, otherwise they will be displayed in a spinner. If mode is 
366      'spinner', then the children will always be displayed in a spinner. 
367   
368      :attr:`mode` is an :class:`~kivy.properties.OptionProperty` and defaults 
369      to 'normal'. 
370      '''
371  
372      dropdown_width = NumericProperty(0)
373      ''' 
374      If non zero, provides the width for the associated DropDown. This is 
375      useful when some items in the ActionGroup's DropDown are wider than usual 
376      and you don't want to make the ActionGroup widget itself wider. 
377   
378      :attr:`dropdown_width` is a :class:`~kivy.properties.NumericProperty` 
379      and defaults to 0. 
380   
381      .. versionadded:: 1.10.0 
382      '''
383  
384      is_open = BooleanProperty(False)
385      '''By default, the DropDown is not open. Set to True to open it. 
386   
387      :attr:`is_open` is a :class:`~kivy.properties.BooleanProperty` and 
388      defaults to False. 
389      '''
390  
391      def __init__(self, **kwargs):
392          self.list_action_item = []
393          self._list_overflow_items = []
394          super(ActionGroup, self).__init__(**kwargs)
395  
396          # real is_open independent on public event
397          self._is_open = False
398  
399          # create DropDown for the group and save its state to _is_open
400          self._dropdown = ActionDropDown()
401          self._dropdown.bind(attach_to=lambda ins, value: setattr(
402              self, '_is_open', True if value else False
403          ))
404  
405          # put open/close responsibility to the event
406          # - trigger dropdown opening when clicked
407          self.bind(on_release=lambda *args: setattr(
408              self, 'is_open', True
409          ))
410  
411          # - trigger dropdown closing when an item
412          #   in the dropdown is clicked
413          self._dropdown.bind(on_dismiss=lambda *args: setattr(
414              self, 'is_open', False
415          ))
416  
417      def on_is_open(self, instance, value):
418          # opening only if the DropDown is closed
419          if value and not self._is_open:
420              self._toggle_dropdown()
421              self._dropdown.open(self)
422              return
423  
424          # closing is_open manually, dismiss manually
425          if not value and self._is_open:
426              self._dropdown.dismiss()
427  
428      def _toggle_dropdown(self, *largs):
429          ddn = self._dropdown
430          ddn.size_hint_x = None
431  
432          # if container was set incorrectly and/or is missing
433          if not ddn.container:
434              return
435          children = ddn.container.children
436  
437          # set DropDown width manually or if not set, then widen
438          # the ActionGroup + DropDown until the widest child fits
439          if children:
440              ddn.width = self.dropdown_width or max(
441                  self.width, max(c.pack_width for c in children)
442              )
443          else:
444              ddn.width = self.width
445  
446          # set the DropDown children's height
447          for item in children:
448              item.size_hint_y = None
449              item.height = max([self.height, sp(48)])
450  
451              # dismiss DropDown manually
452              # auto_dismiss applies to touching outside of the DropDown
453              item.bind(on_release=ddn.dismiss)
454  
455      def add_widget(self, item):
456          # if adding ActionSeparator ('normal' mode,
457          # everything visible), add it to the parent
458          if isinstance(item, ActionSeparator):
459              super(ActionGroup, self).add_widget(item)
460              return
461  
462          if not isinstance(item, ActionItem):
463              raise ActionBarException('ActionGroup only accepts ActionItem')
464  
465          self.list_action_item.append(item)
466  
467      def show_group(self):
468          # 'normal' mode, items can fit to the view
469          self.clear_widgets()
470          for item in self._list_overflow_items + self.list_action_item:
471              item.inside_group = True
472              self._dropdown.add_widget(item)
473  
474      def clear_widgets(self):
475          self._dropdown.clear_widgets()

479      ''' 
480      ActionOverflow class, see module documentation for more information. 
481      '''
482  
483      overflow_image = StringProperty(
484          'atlas://data/images/defaulttheme/overflow')
485      ''' 
486      Image to be used as an Overflow Image. 
487   
488      :attr:`overflow_image` is a :class:`~kivy.properties.StringProperty` 
489      and defaults to 'atlas://data/images/defaulttheme/overflow'. 
490      '''

492      def add_widget(self, action_item, index=0):
493          if action_item is None:
494              return
495  
496          if isinstance(action_item, ActionSeparator):
497              return

499          if not isinstance(action_item, ActionItem):
500              raise ActionBarException('ActionView only accepts ActionItem'
501                                       ' (got {!r}'.format(action_item))
502  
503          else:
504              if index == 0:
505                  index = len(self._list_overflow_items)
506              self._list_overflow_items.insert(index, action_item)
507  
508      def show_default_items(self, parent):
509          # display overflow and its items if widget's directly added to it
510          if self._list_overflow_items == []:
511              return
512          self.show_group()
513          super(ActionView, parent).add_widget(self)

516  class ActionView(BoxLayout):
517      ''' 
518      ActionView class, see module documentation for more information. 
519      '''
520  
521      action_previous = ObjectProperty(None)
522      ''' 
523      Previous button for an ActionView. 
524   
525      :attr:`action_previous` is an :class:`~kivy.properties.ObjectProperty` 
526      and defaults to None. 
527      '''
528  
529      background_color = ColorProperty([1, 1, 1, 1])
530      ''' 
531      Background color in the format (r, g, b, a). 
532   
533      :attr:`background_color` is a :class:`~kivy.properties.ColorProperty` and 
534      defaults to [1, 1, 1, 1]. 
535   
536      .. versionchanged:: 2.0.0 
537          Changed from :class:`~kivy.properties.ListProperty` to 
538          :class:`~kivy.properties.ColorProperty`. 
539      '''
540  
541      background_image = StringProperty(
542          'atlas://data/images/defaulttheme/action_view')
543      ''' 
544      Background image of an ActionViews default graphical representation. 
545   
546      :attr:`background_image` is a :class:`~kivy.properties.StringProperty` 
547      and defaults to 'atlas://data/images/defaulttheme/action_view'. 
548      '''
549  
550      use_separator = BooleanProperty(False)
551      ''' 
552      Specify whether to use a separator before every ActionGroup or not. 
553   
554      :attr:`use_separator` is a :class:`~kivy.properties.BooleanProperty` and 
555      defaults to False. 
556      '''
557  
558      overflow_group = ObjectProperty(None)
559      ''' 
560      Widget to be used for the overflow. 
561   
562      :attr:`overflow_group` is an :class:`~kivy.properties.ObjectProperty` and 
563      defaults to an instance of :class:`ActionOverflow`. 
564      '''
565  
566      def __init__(self, **kwargs):
567          self._list_action_items = []
568          self._list_action_group = []
569          super(ActionView, self).__init__(**kwargs)
570          self._state = ''
571          if not self.overflow_group:
572              self.overflow_group = ActionOverflow(
573                  use_separator=self.use_separator)
574  
575      def on_action_previous(self, instance, value):
576          self._list_action_items.insert(0, value)
577  
578      def add_widget(self, action_item, index=0):
579          if action_item is None:
580              return
581  
582          if not isinstance(action_item, ActionItem):
583              raise ActionBarException('ActionView only accepts ActionItem'
584                                       ' (got {!r}'.format(action_item))
585  
586          elif isinstance(action_item, ActionOverflow):
587              self.overflow_group = action_item
588              action_item.use_separator = self.use_separator
589  
590          elif isinstance(action_item, ActionGroup):
591              self._list_action_group.append(action_item)
592              action_item.use_separator = self.use_separator
593  
594          elif isinstance(action_item, ActionPrevious):
595              self.action_previous = action_item
596  
597          else:
598              super(ActionView, self).add_widget(action_item, index)
599              if index == 0:
600                  index = len(self._list_action_items)
601              self._list_action_items.insert(index, action_item)
602  
603      def on_use_separator(self, instance, value):
604          for group in self._list_action_group:
605              group.use_separator = value
606          if self.overflow_group:
607              self.overflow_group.use_separator = value
608  
609      def remove_widget(self, widget):
610          super(ActionView, self).remove_widget(widget)
611          if isinstance(widget, ActionOverflow):
612              for item in widget.list_action_item:
613                  if item in self._list_action_items:
614                      self._list_action_items.remove(item)
615  
616          if widget in self._list_action_items:
617              self._list_action_items.remove(widget)
618  
619      def _clear_all(self):
620          lst = self._list_action_items[:]
621          self.clear_widgets()
622          for group in self._list_action_group:
623              group.clear_widgets()
624  
625          self.overflow_group.clear_widgets()
626          self.overflow_group.list_action_item = []
627          self._list_action_items = lst
628  
629      def _layout_all(self):
630          # all the items can fit to the view, so expand everything
631          super_add = super(ActionView, self).add_widget
632          self._state = 'all'
633          self._clear_all()
634          if not self.action_previous.parent:
635              super_add(self.action_previous)
636          if len(self._list_action_items) > 1:
637              for child in self._list_action_items[1:]:
638                  child.inside_group = False
639                  super_add(child)
640  
641          for group in self._list_action_group:
642              if group.mode == 'spinner':
643                  super_add(group)
644                  group.show_group()
645              else:
646                  if group.list_action_item != []:
647                      super_add(ActionSeparator())
648                  for child in group.list_action_item:
649                      child.inside_group = False
650                      super_add(child)
651  
652          self.overflow_group.show_default_items(self)
653  
654      def _layout_group(self):
655          # layout all the items in order to pack them per group
656          super_add = super(ActionView, self).add_widget
657          self._state = 'group'
658          self._clear_all()
659          if not self.action_previous.parent:
660              super_add(self.action_previous)
661          if len(self._list_action_items) > 1:
662              for child in self._list_action_items[1:]:
663                  super_add(child)
664                  child.inside_group = False
665  
666          for group in self._list_action_group:
667              super_add(group)
668              group.show_group()
669  
670          self.overflow_group.show_default_items(self)
671  
672      def _layout_random(self):
673          # layout the items in order to pack all of them grouped, and display
674          # only the action items having 'important'
675          super_add = super(ActionView, self).add_widget
676          self._state = 'random'
677          self._clear_all()
678          hidden_items = []
679          hidden_groups = []
680          total_width = 0
681          if not self.action_previous.parent:
682              super_add(self.action_previous)
683  
684          width = (self.width - self.overflow_group.pack_width -
685                   self.action_previous.minimum_width)
686  
687          if len(self._list_action_items):
688              for child in self._list_action_items[1:]:
689                  if child.important:
690                      if child.pack_width + total_width < width:
691                          super_add(child)
692                          child.inside_group = False
693                          total_width += child.pack_width
694                      else:
695                          hidden_items.append(child)
696                  else:
697                      hidden_items.append(child)
698  
699          # if space is left then display ActionItem inside their
700          # ActionGroup
701          if total_width < self.width:
702              for group in self._list_action_group:
703                  if group.pack_width + total_width +\
704                          group.separator_width < width:
705                      super_add(group)
706                      group.show_group()
707                      total_width += (group.pack_width +
708                                      group.separator_width)
709  
710                  else:
711                      hidden_groups.append(group)
712          group_index = len(self.children) - 1
713          # if space is left then display other ActionItems
714          if total_width < self.width:
715              for child in hidden_items[:]:
716                  if child.pack_width + total_width < width:
717                      super_add(child, group_index)
718                      total_width += child.pack_width
719                      child.inside_group = False
720                      hidden_items.remove(child)
721  
722          # for all the remaining ActionItems and ActionItems with in
723          # ActionGroups, Display them inside overflow_group
724          extend_hidden = hidden_items.extend
725          for group in hidden_groups:
726              extend_hidden(group.list_action_item)
727  
728          overflow_group = self.overflow_group
729  
730          if hidden_items != []:
731              over_add = super(overflow_group.__class__,
732                               overflow_group).add_widget
733              for child in hidden_items:
734                  over_add(child)
735  
736              overflow_group.show_group()
737              if not self.overflow_group.parent:
738                  super_add(overflow_group)
739  
740      def on_width(self, width, *args):
741          # determine the layout to use
742  
743          # can we display all of them?
744          total_width = 0
745          for child in self._list_action_items:
746              total_width += child.pack_width
747          for group in self._list_action_group:
748              for child in group.list_action_item:
749                  total_width += child.pack_width
750          if total_width <= self.width:
751              if self._state != 'all':
752                  self._layout_all()
753              return
754  
755          # can we display them per group?
756          total_width = 0
757          for child in self._list_action_items:
758              total_width += child.pack_width
759          for group in self._list_action_group:
760              total_width += group.pack_width
761          if total_width < self.width:
762              # ok, we can display all the items grouped
763              if self._state != 'group':
764                  self._layout_group()
765              return
766  
767          # none of the solutions worked, display them in pack mode
768          self._layout_random()

779  class ActionBar(BoxLayout):
780      ''' 
781      ActionBar class, which acts as the main container for an 
782      :class:`ActionView` instance. The ActionBar determines the overall 
783      styling aspects of the bar. :class:`ActionItem`\\s are not added to 
784      this class directly, but to the contained :class:`ActionView` instance. 
785   
786      :Events: 
787          `on_previous` 
788              Fired when action_previous of action_view is pressed. 
789   
790      Please see the module documentation for more information. 
791      '''
792  
793      action_view = ObjectProperty(None)
794      ''' 
795      action_view of the ActionBar. 
796   
797      :attr:`action_view` is an :class:`~kivy.properties.ObjectProperty` and 
798      defaults to None or the last ActionView instance added to the ActionBar. 
799      '''
800  
801      background_color = ColorProperty([1, 1, 1, 1])
802      ''' 
803      Background color, in the format (r, g, b, a). 
804   
805      :attr:`background_color` is a :class:`~kivy.properties.ColorProperty` and 
806      defaults to [1, 1, 1, 1]. 
807   
808      .. versionchanged:: 2.0.0 
809          Changed from :class:`~kivy.properties.ListProperty` to 
810          :class:`~kivy.properties.ColorProperty`. 
811      '''
812  
813      background_image = StringProperty(
814          'atlas://data/images/defaulttheme/action_bar')
815  
816      ''' 
817      Background image of the ActionBars default graphical representation. 
818   
819      :attr:`background_image` is a :class:`~kivy.properties.StringProperty` 
820      and defaults to 'atlas://data/images/defaulttheme/action_bar'. 
821      '''
822  
823      border = ListProperty([2, 2, 2, 2])
824      ''' 
825      The border to be applied to the :attr:`background_image`. 
826   
827      :attr:`border` is a :class:`~kivy.properties.ListProperty` and defaults to 
828      [2, 2, 2, 2] 
829      '''
830  
831      __events__ = ('on_previous',)
832  
833      def __init__(self, **kwargs):
834          super(ActionBar, self).__init__(**kwargs)
835          self._stack_cont_action_view = []
836          self._emit_previous = partial(self.dispatch, 'on_previous')
837  
838      def add_widget(self, view):
839          if isinstance(view, ContextualActionView):
840              self._stack_cont_action_view.append(view)
841              if view.action_previous is not None:
842                  view.action_previous.unbind(on_release=self._emit_previous)
843                  view.action_previous.bind(on_release=self._emit_previous)
844              self.clear_widgets()
845              super(ActionBar, self).add_widget(view)
846  
847          elif isinstance(view, ActionView):
848              self.action_view = view
849              super(ActionBar, self).add_widget(view)
850  
851          else:
852              raise ActionBarException(
853                  'ActionBar can only add ContextualActionView or ActionView')
854  
855      def on_previous(self, *args):
856          self._pop_contextual_action_view()
857  
858      def _pop_contextual_action_view(self):
859          '''Remove the current ContextualActionView and display either the 
860             previous one or the ActionView. 
861          '''
862          self._stack_cont_action_view.pop()
863          self.clear_widgets()
864          if self._stack_cont_action_view == []:
865              super(ActionBar, self).add_widget(self.action_view)
866          else:
867              super(ActionBar, self).add_widget(self._stack_cont_action_view[-1])
