9    def SafeApply(fn, args, err_desc=""):
10       try:
11           fn(*args)
12           return 1
13       except win32api.error as exc:
14           msg = "Error " + err_desc + "\r\n\r\n" + exc.strerror
15           win32ui.MessageBox(msg)
16           return 0

19   class SplitterFrame(window.MDIChildWnd):
20       def __init__(self):
21           # call base CreateFrame
22           self.images = None
23           window.MDIChildWnd.__init__(self)

25       def OnCreateClient(self, cp, context):
26           splitter = win32ui.CreateSplitter()
27           doc = context.doc
28           frame_rect = self.GetWindowRect()
29           size = ((frame_rect[2] - frame_rect[0]), (frame_rect[3] - frame_rect[1]) // 2)
30           sub_size = (size[0] // 3, size[1])
31           splitter.CreateStatic(self, 1, 2)
32           # CTreeControl view
33           self.keysview = RegistryTreeView(doc)
34           # CListControl view
35           self.valuesview = RegistryValueView(doc)
36   
37           splitter.CreatePane(self.keysview, 0, 0, (sub_size))
38           splitter.CreatePane(self.valuesview, 0, 1, (0, 0))  # size ignored.
39           splitter.SetRowInfo(0, size[1], 0)
40           # Setup items in the imagelist
41   
42           return 1

44       def OnItemDoubleClick(self, info, extra):
45           (hwndFrom, idFrom, code) = info
46           if idFrom == win32ui.AFX_IDW_PANE_FIRST:
47               # Tree control
48               return None
49           elif idFrom == win32ui.AFX_IDW_PANE_FIRST + 1:
50               item = self.keysview.SelectedItem()
51               self.valuesview.EditValue(item)
52               return 0
53               # List control
54           else:
55               return None  # Pass it on

109      def OnDeleteKey(self, command, code):
110          hitem = self.hierList.GetSelectedItem()
111          item = self.hierList.ItemFromHandle(hitem)
112          msg = "Are you sure you wish to delete the key '%s'?" % (item.keyName,)
113          id = win32ui.MessageBox(msg, None, win32con.MB_YESNO)
114          if id != win32con.IDYES:
115              return
116          if SafeApply(
117              win32api.RegDeleteKey, (item.keyRoot, item.keyName), "deleting registry key"
118          ):
119              # Get the items parent.
120              try:
121                  hparent = self.GetParentItem(hitem)
122              except win32ui.error:
123                  hparent = None
124              self.hierList.Refresh(hparent)

126      def OnAddKey(self, command, code):
127          from pywin.mfc import dialog
128  
129          val = dialog.GetSimpleInput("New key name", "", "Add new key")
130          if val is None:
131              return  # cancelled.
132          hitem = self.hierList.GetSelectedItem()
133          item = self.hierList.ItemFromHandle(hitem)
134          if SafeApply(win32api.RegCreateKey, (item.keyRoot, item.keyName + "\\" + val)):
135              self.hierList.Refresh(hitem)

137      def OnAddValue(self, command, code):
138          from pywin.mfc import dialog
139  
140          val = dialog.GetSimpleInput("New value", "", "Add new value")
141          if val is None:
142              return  # cancelled.
143          hitem = self.hierList.GetSelectedItem()
144          item = self.hierList.ItemFromHandle(hitem)
145          if SafeApply(
146              win32api.RegSetValue, (item.keyRoot, item.keyName, win32con.REG_SZ, val)
147          ):
148              # Simply re-select the current item to refresh the right spitter.
149              self.PerformItemSelected(item)

169  class RegistryValueView(docview.ListView):
170      def OnInitialUpdate(self):
171          hwnd = self._obj_.GetSafeHwnd()
172          style = win32api.GetWindowLong(hwnd, win32con.GWL_STYLE)
173          win32api.SetWindowLong(
174              hwnd,
175              win32con.GWL_STYLE,
176              (style & ~commctrl.LVS_TYPEMASK) | commctrl.LVS_REPORT,
177          )
178  
179          itemDetails = (commctrl.LVCFMT_LEFT, 100, "Name", 0)
180          self.InsertColumn(0, itemDetails)
181          itemDetails = (commctrl.LVCFMT_LEFT, 500, "Data", 0)
182          self.InsertColumn(1, itemDetails)

184      def UpdateForRegItem(self, item):
185          self.DeleteAllItems()
186          hkey = win32api.RegOpenKey(item.keyRoot, item.keyName)
187          try:
188              valNum = 0
189              ret = []
190              while 1:
191                  try:
192                      res = win32api.RegEnumValue(hkey, valNum)
193                  except win32api.error:
194                      break
195                  name = res[0]
196                  if not name:
197                      name = "(Default)"
198                  self.InsertItem(valNum, name)
199                  self.SetItemText(valNum, 1, str(res[1]))
200                  valNum = valNum + 1
201          finally:
202              win32api.RegCloseKey(hkey)

204      def EditValue(self, item):
205          # Edit the current value
206          class EditDialog(dialog.Dialog):
207              def __init__(self, item):
208                  self.item = item
209                  dialog.Dialog.__init__(self, win32ui.IDD_LARGE_EDIT)
210  
211              def OnInitDialog(self):
212                  self.SetWindowText("Enter new value")
213                  self.GetDlgItem(win32con.IDCANCEL).ShowWindow(win32con.SW_SHOW)
214                  self.edit = self.GetDlgItem(win32ui.IDC_EDIT1)
215                  # Modify the edit windows style
216                  style = win32api.GetWindowLong(
217                      self.edit.GetSafeHwnd(), win32con.GWL_STYLE
218                  )
219                  style = style & (~win32con.ES_WANTRETURN)
220                  win32api.SetWindowLong(
221                      self.edit.GetSafeHwnd(), win32con.GWL_STYLE, style
222                  )
223                  self.edit.SetWindowText(str(self.item))
224                  self.edit.SetSel(-1)
225                  return dialog.Dialog.OnInitDialog(self)
226  
227              def OnDestroy(self, msg):
228                  self.newvalue = self.edit.GetWindowText()
229  
230          try:
231              index = self.GetNextItem(-1, commctrl.LVNI_SELECTED)
232          except win32ui.error:
233              return  # No item selected.
234  
235          if index == 0:
236              keyVal = ""
237          else:
238              keyVal = self.GetItemText(index, 0)
239          # Query for a new value.
240          try:
241              newVal = self.GetItemsCurrentValue(item, keyVal)
242          except TypeError as details:
243              win32ui.MessageBox(details)
244              return
245  
246          d = EditDialog(newVal)
247          if d.DoModal() == win32con.IDOK:
248              try:
249                  self.SetItemsCurrentValue(item, keyVal, d.newvalue)
250              except win32api.error as exc:
251                  win32ui.MessageBox("Error setting value\r\n\n%s" % exc.strerror)
252              self.UpdateForRegItem(item)
253  
254      def GetItemsCurrentValue(self, item, valueName):
255          hkey = win32api.RegOpenKey(item.keyRoot, item.keyName)
256          try:
257              val, type = win32api.RegQueryValueEx(hkey, valueName)
258              if type != win32con.REG_SZ:
259                  raise TypeError("Only strings can be edited")
260              return val
261          finally:
262              win32api.RegCloseKey(hkey)
263  
264      def SetItemsCurrentValue(self, item, valueName, value):
265          # ** Assumes already checked is a string.
266          hkey = win32api.RegOpenKey(
267              item.keyRoot, item.keyName, 0, win32con.KEY_SET_VALUE
268          )
269          try:
270              win32api.RegSetValueEx(hkey, valueName, 0, win32con.REG_SZ, value)
271          finally:
272              win32api.RegCloseKey(hkey)

285      def OpenRegistryKey(
286          self, root=None, subkey=None
287      ):  # Use this instead of OpenDocumentFile.
288          # Look for existing open document
289          if root is None:
290              root = regutil.GetRootKey()
291          if subkey is None:
292              subkey = regutil.BuildDefaultPythonKey()
293          for doc in self.GetDocumentList():
294              if doc.root == root and doc.subkey == subkey:
295                  doc.GetFirstView().ActivateFrame()
296                  return doc
297          # not found - new one.
298          doc = RegDocument(self, root, subkey)
299          frame = self.CreateNewFrame(doc)
300          doc.OnNewDocument()
301          self.InitialUpdateFrame(frame, doc, 1)
302          return doc

355      def GetSubList(self):
356          hkey = win32api.RegOpenKey(self.keyRoot, self.keyName)
357          win32ui.DoWaitCursor(1)
358          try:
359              keyNum = 0
360              ret = []
361              while 1:
362                  try:
363                      key = win32api.RegEnumKey(hkey, keyNum)
364                  except win32api.error:
365                      break
366                  ret.append(HLIRegistryKey(self.keyRoot, self.keyName + "\\" + key, key))
367                  keyNum = keyNum + 1
368          finally:
369              win32api.RegCloseKey(hkey)
370              win32ui.DoWaitCursor(0)
371          return ret
