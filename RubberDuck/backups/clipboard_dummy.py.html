1    ''' 
2    Clipboard Dummy: an internal implementation that does not use the system

5    
6    __all__ = ('ClipboardDummy', )
7    
8    from kivy.core.clipboard import ClipboardBase
9    
10   
11   class ClipboardDummy(ClipboardBase):
12   
13       def __init__(self):
14           super(ClipboardDummy, self).__init__()
15           self._data = dict()
16           self._data['text/plain'] = None
17           self._data['application/data'] = None
18   
19       def get(self, mimetype='text/plain'):
20           return self._data.get(mimetype, None)
21   
22       def put(self, data, mimetype='text/plain'):
23           self._data[mimetype] = data
24   
25       def get_types(self):
26           return list(self._data.keys())
