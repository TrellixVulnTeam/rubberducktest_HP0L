25   # The ISAPI extension - handles requests in our virtual dir, and sends the
26   # response to the client.
27   class Extension(threaded_extension.ThreadPoolExtension):
28       "Python ISAPI Tester"
29   
30       def Dispatch(self, ecb):
31           print('Tester dispatching "%s"' % (ecb.GetServerVariable("URL"),))
32           url = ecb.GetServerVariable("URL")
33           test_name = url.split("/")[-1]
34           meth = getattr(self, test_name, None)
35           if meth is None:
36               raise AttributeError("No test named '%s'" % (test_name,))
37           result = meth(ecb)
38           if result is None:
39               # This means the test finalized everything
40               return
41           ecb.SendResponseHeaders("200 OK", "Content-type: text/html\r\n\r\n", False)
42           print("<HTML><BODY>Finished running test <i>", test_name, "</i>", file=ecb)
43           print("<pre>", file=ecb)
44           print(result, file=ecb)
45           print("</pre>", file=ecb)
46           print("</BODY></HTML>", file=ecb)
47           ecb.DoneWithSession()

49       def test1(self, ecb):
50           try:
51               ecb.GetServerVariable("foo bar")
52               raise RuntimeError("should have failed!")
53           except ExtensionError as err:
54               assert err.errno == winerror.ERROR_INVALID_INDEX, err
55           return "worked!"
56   
57       def test_long_vars(self, ecb):
58           qs = ecb.GetServerVariable("QUERY_STRING")
59           # Our implementation has a default buffer size of 8k - so we test
60           # the code that handles an overflow by ensuring there are more
61           # than 8k worth of chars in the URL.
62           expected_query = "x" * 8500
63           if len(qs) == 0:
64               # Just the URL with no query part - redirect to myself, but with
65               # a huge query portion.
66               me = ecb.GetServerVariable("URL")
67               headers = "Location: " + me + "?" + expected_query + "\r\n\r\n"
68               ecb.SendResponseHeaders("301 Moved", headers)
69               ecb.DoneWithSession()
70               return None
71           if qs == expected_query:
72               return "Total length of variable is %d - test worked!" % (len(qs),)
73           else:
74               return "Unexpected query portion!  Got %d chars, expected %d" % (
75                   len(qs),
76                   len(expected_query),
77               )
78   
79       def test_unicode_vars(self, ecb):
80           # We need to check that we are running IIS6!  This seems the only
81           # effective way from an extension.
82           ver = float(ecb.GetServerVariable("SERVER_SOFTWARE").split("/")[1])
83           if ver < 6.0:
84               return "This is IIS version %g - unicode only works in IIS6 and later" % ver
85   
86           us = ecb.GetServerVariable("UNICODE_SERVER_NAME")
87           if not isinstance(us, str):
88               raise RuntimeError("unexpected type!")
89           if us != str(ecb.GetServerVariable("SERVER_NAME")):
90               raise RuntimeError("Unicode and non-unicode values were not the same")
91           return "worked!"
