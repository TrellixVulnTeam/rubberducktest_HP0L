8    class TestEnum(win32com.test.util.TestCase):
9        def testit(self):
10           fname, tmp = win32api.GetTempFileName(win32api.GetTempPath(), "stg")
11           m = storagecon.STGM_READWRITE | storagecon.STGM_SHARE_EXCLUSIVE
12           ##  file, mode, format, attrs (always 0), IID (IStorage or IPropertySetStorage, storage options(only used with STGFMT_DOCFILE)
13           pss = pythoncom.StgOpenStorageEx(
14               fname, m, storagecon.STGFMT_FILE, 0, pythoncom.IID_IPropertySetStorage
15           )
16           ###                               {"Version":2,"reserved":0,"SectorSize":512,"TemplateFile":u'somefilename'})
17   
18           ## FMTID_SummaryInformation FMTID_DocSummaryInformation FMTID_UserDefinedProperties
19           psuser = pss.Create(
20               pythoncom.FMTID_UserDefinedProperties,
21               pythoncom.IID_IPropertySetStorage,
22               storagecon.PROPSETFLAG_DEFAULT,
23               storagecon.STGM_READWRITE
24               | storagecon.STGM_CREATE
25               | storagecon.STGM_SHARE_EXCLUSIVE,
26           )  ## its very picky about flag combinations!
27           psuser.WriteMultiple((3, 4), ("hey", "bubba"))
28           psuser.WritePropertyNames((3, 4), ("property3", "property4"))
29           expected_summaries = []
30           expected_summaries.append(("property3", 3, pythoncom.VT_BSTR))
31           expected_summaries.append(("property4", 4, pythoncom.VT_BSTR))
32           psuser = None
33   
34           pssum = pss.Create(
35               pythoncom.FMTID_SummaryInformation,
36               pythoncom.IID_IPropertySetStorage,
37               storagecon.PROPSETFLAG_DEFAULT,
38               storagecon.STGM_READWRITE
39               | storagecon.STGM_CREATE
40               | storagecon.STGM_SHARE_EXCLUSIVE,
41           )
42           pssum.WriteMultiple(
43               (storagecon.PIDSI_AUTHOR, storagecon.PIDSI_COMMENTS), ("me", "comment")
44           )
45   
46           pssum = None
47           pss = None  ## doesn't seem to be a close or release method, and you can't even reopen it from the same process until previous object is gone
48   
49           pssread = pythoncom.StgOpenStorageEx(
50               fname,
51               storagecon.STGM_READ | storagecon.STGM_SHARE_EXCLUSIVE,
52               storagecon.STGFMT_FILE,
53               0,
54               pythoncom.IID_IPropertySetStorage,
55           )
56           found_summaries = []
57           for psstat in pssread:
58               ps = pssread.Open(
59                   psstat[0], storagecon.STGM_READ | storagecon.STGM_SHARE_EXCLUSIVE
60               )
61               for p in ps:
62                   p_val = ps.ReadMultiple((p[1],))[0]
63                   if (p[1] == storagecon.PIDSI_AUTHOR and p_val == "me") or (
64                       p[1] == storagecon.PIDSI_COMMENTS and p_val == "comment"
65                   ):
66                       pass
67                   else:
68                       self.fail("Uxexpected property %s/%s" % (p, p_val))
69               ps = None
70               ## FMTID_UserDefinedProperties can't exist without FMTID_DocSummaryInformation, and isn't returned independently from Enum
71               ## also can't be open at same time
72               if psstat[0] == pythoncom.FMTID_DocSummaryInformation:
73                   ps = pssread.Open(
74                       pythoncom.FMTID_UserDefinedProperties,
75                       storagecon.STGM_READ | storagecon.STGM_SHARE_EXCLUSIVE,
76                   )
77                   for p in ps:
78                       found_summaries.append(p)
79                   ps = None
80           psread = None
81           expected_summaries.sort()
82           found_summaries.sort()
83           self.assertEqual(expected_summaries, found_summaries)
