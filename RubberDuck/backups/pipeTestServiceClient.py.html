34   def CallPipe(fn, args):
35       ret = None
36       retryCount = 0
37       while retryCount < 8:  # Keep looping until user cancels.
38           retryCount = retryCount + 1
39           try:
40               return fn(*args)
41           except win32api.error as exc:
42               if exc.winerror == winerror.ERROR_PIPE_BUSY:
43                   win32api.Sleep(5000)
44                   continue
45               else:
46                   raise
47   
48       raise RuntimeError("Could not make a connection to the server")

95   def stressTestClient(server, numThreads, numMessages):
96       import _thread
97   
98       thread_waits = []
99       for t_num in range(numThreads):
100          # Note I could just wait on thread handles (after calling DuplicateHandle)
101          # See the service itself for an example of waiting for the clients...
102          wait = CreateEvent(None, 0, 0, None)
103          thread_waits.append(wait)
104          _thread.start_new_thread(stressThread, (server, numMessages, wait))
105      # Wait for all threads to finish.
106      WaitForMultipleObjects(thread_waits, 1, INFINITE)
