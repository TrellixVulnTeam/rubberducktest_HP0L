11   import re
12   
13   from pygments.lexer import RegexLexer, words
14   from pygments.token import String, Comment, Keyword, Name, Number, Text, \

19   line_re = re.compile('.*?\n')

22   class BoaLexer(RegexLexer):
23       """ 
24       Lexer for the `Boa <http://boa.cs.iastate.edu/docs/>`_ language. 
25    
26       .. versionadded:: 2.4 
27       """
28       name = 'Boa'
29       aliases = ['boa']
30       filenames = ['*.boa']
31   
32       reserved = words(
33           ('input', 'output', 'of', 'weight', 'before', 'after', 'stop',
34            'ifall', 'foreach', 'exists', 'function', 'break', 'switch', 'case',
35            'visitor', 'default', 'return', 'visit', 'while', 'if', 'else'),
36           suffix=r'\b', prefix=r'\b')
37       keywords = words(
38           ('bottom', 'collection', 'maximum', 'mean', 'minimum', 'set', 'sum',
39            'top', 'string', 'int', 'bool', 'float', 'time', 'false', 'true',
40            'array', 'map', 'stack', 'enum', 'type'), suffix=r'\b', prefix=r'\b')
41       classes = words(
42           ('Project', 'ForgeKind', 'CodeRepository', 'Revision', 'RepositoryKind',
43            'ChangedFile', 'FileKind', 'ASTRoot', 'Namespace', 'Declaration', 'Type',
44            'Method', 'Variable', 'Statement', 'Expression', 'Modifier',
45            'StatementKind', 'ExpressionKind', 'ModifierKind', 'Visibility',
46            'TypeKind', 'Person', 'ChangeKind'),
47           suffix=r'\b', prefix=r'\b')
48       operators = ('->', ':=', ':', '=', '<<', '!', '++', '||',
49                    '&&', '+', '-', '*', ">", "<")
50       string_sep = ('`', '\"')
51       built_in_functions = words(
52           (
53               # Array functions
54               'new', 'sort',
55               # Date & Time functions
56               'yearof', 'dayofyear', 'hourof', 'minuteof', 'secondof', 'now',
57               'addday', 'addmonth', 'addweek', 'addyear', 'dayofmonth', 'dayofweek',
58               'dayofyear', 'formattime', 'trunctoday', 'trunctohour', 'trunctominute',
59               'trunctomonth', 'trunctosecond', 'trunctoyear',
60               # Map functions
61               'clear', 'haskey', 'keys', 'lookup', 'remove', 'values',
62               # Math functions
63               'abs', 'acos', 'acosh', 'asin', 'asinh', 'atan', 'atan2', 'atanh',
64               'ceil', 'cos', 'cosh', 'exp', 'floor', 'highbit', 'isfinite', 'isinf',
65               'isnan', 'isnormal', 'log', 'log10', 'max', 'min', 'nrand', 'pow',
66               'rand', 'round', 'sin', 'sinh', 'sqrt', 'tan', 'tanh', 'trunc',
67               # Other functions
68               'def', 'hash', 'len',
69               # Set functions
70               'add', 'contains', 'remove',
71               # String functions
72               'format', 'lowercase', 'match', 'matchposns', 'matchstrs', 'regex',
73               'split', 'splitall', 'splitn', 'strfind', 'strreplace', 'strrfind',
74               'substring', 'trim', 'uppercase',
75               # Type Conversion functions
76               'bool', 'float', 'int', 'string', 'time',
77               # Domain-Specific functions
78               'getast', 'getsnapshot', 'hasfiletype', 'isfixingrevision', 'iskind',
79               'isliteral',
80           ),
81           prefix=r'\b',
82           suffix=r'\(')
83   
84       tokens = {
85           'root': [
86               (r'#.*?$', Comment.Single),
87               (r'/\*.*?\*/', Comment.Multiline),
88               (reserved, Keyword.Reserved),
89               (built_in_functions, Name.Function),
90               (keywords, Keyword.Type),
91               (classes, Name.Classes),
92               (words(operators), Operator),
93               (r'[][(),;{}\\.]', Punctuation),
94               (r'"(\\\\|\\[^\\]|[^"\\])*"', String.Double),
95               (r"`(\\\\|\\[^\\]|[^`\\])*`", String.Backtick),
96               (words(string_sep), String.Delimiter),
97               (r'[a-zA-Z_]+', Name.Variable),
98               (r'[0-9]+', Number.Integer),
99               (r'\s+?', Text),  # Whitespace
100          ]
101      }
