271      def analyse_text(text):
272          """endelse seems to be unique to IDL, endswitch is rare at least."""
273          result = 0
274  
275          if 'endelse' in text:
276              result += 0.2
277          if 'endswitch' in text:
278              result += 0.01
279  
280          return result
