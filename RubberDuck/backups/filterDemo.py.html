10   class FileParser:
11       # Property IDs for the Storage Property Set
12       PIDS_BODY = 0x00000013
13   
14       # property IDs for HTML Storage Property Set
15       PIDH_DESCRIPTION = "DESCRIPTION"
16       PIDH_HREF = "A.HREF"
17       PIDH_IMGSRC = "IMG.SRC"
18   
19       # conversion map to convert ifilter properties to more user friendly names
20       propertyToName = {
21           PSGUID_STORAGE: {PIDS_BODY: "body"},
22           PSGUID_SUMMARYINFORMATION: {
23               PIDSI_TITLE: "title",
24               PIDSI_SUBJECT: "description",
25               PIDSI_AUTHOR: "author",
26               PIDSI_KEYWORDS: "keywords",
27               PIDSI_COMMENTS: "comments",
28           },
29           PSGUID_HTMLINFORMATION: {PIDH_DESCRIPTION: "description"},
30           PSGUID_HTML2_INFORMATION: {PIDH_HREF: "href", PIDH_IMGSRC: "img"},
31       }

33       def __init__(self, verbose=False):
34           self.f = None
35           self.stg = None
36           self.verbose = verbose
37   
38       def Close(self):
39           self.f = None
40           self.stg = None

42       def Parse(self, fileName, maxErrors=10):
43           properties = {}
44   
45           try:
46               self._bind_to_filter(fileName)
47               try:
48                   flags = self.f.Init(
49                       IFILTER_INIT_APPLY_INDEX_ATTRIBUTES
50                       | IFILTER_INIT_APPLY_OTHER_ATTRIBUTES
51                   )
52                   if flags == IFILTER_FLAGS_OLE_PROPERTIES and self.stg is not None:
53                       self._trace("filter requires to get properities via ole")
54                       self._get_properties(properties)
55   
56                   errCnt = 0
57                   while True:
58                       try:
59                           # each chunk returns a tuple with the following:-
60                           # idChunk       = The chunk identifier. each chunk has a unique identifier
61                           # breakType     = The type of break that separates the previous chunk from the current chunk. Values are:-
62                           #                 CHUNK_NO_BREAK=0,CHUNK_EOW=1,CHUNK_EOS= 2,CHUNK_EOP= 3,CHUNK_EOC= 4
63                           # flags         = Flags indicate whether this chunk contains a text-type or a value-type property
64                           #                 locale = The language and sublanguage associated with a chunk of text
65                           # attr          = A tuple containing the property to be applied to the chunk. Tuple is (propertyset GUID, property ID)
66                           #                 Property ID can be a number or string
67                           # idChunkSource = The ID of the source of a chunk. The value of the idChunkSource member depends on the nature of the chunk
68                           # startSource   = The offset from which the source text for a derived chunk starts in the source chunk
69                           # lenSource     = The length in characters of the source text from which the current chunk was derived.
70                           #                 A zero value signifies character-by-character correspondence between the source text and the derived text.
71   
72                           (
73                               idChunk,
74                               breakType,
75                               flags,
76                               locale,
77                               attr,
78                               idChunkSource,
79                               startSource,
80                               lenSource,
81                           ) = self.f.GetChunk()
82                           self._trace(
83                               "Chunk details:",
84                               idChunk,
85                               breakType,
86                               flags,
87                               locale,
88                               attr,
89                               idChunkSource,
90                               startSource,
91                               lenSource,
92                           )
93   
94                           # attempt to map each property to a more user friendly name. If we don't know what it is just return
95                           # the set guid and property id. (note: the id can be a number or a string.
96                           propSet = self.propertyToName.get(attr[0])
97                           if propSet:
98                               propName = propSet.get(attr[1], "%s:%s" % attr)
99                           else:
100                              propName = "%s:%s" % attr
101  
102                      except pythoncom.com_error as e:
103                          if e[0] == FILTER_E_END_OF_CHUNKS:
104                              # we have read all the chunks
105                              break
106                          elif e[0] in [
107                              FILTER_E_EMBEDDING_UNAVAILABLE,
108                              FILTER_E_LINK_UNAVAILABLE,
109                          ]:
110                              # the next chunk can't be read. Also keep track of the number of times we
111                              # fail as some filters (ie. the Msoft office ones can get stuck here)
112                              errCnt += 1
113                              if errCnt > maxErrors:
114                                  raise
115                              else:
116                                  continue
117                          elif e[0] == FILTER_E_ACCESS:
118                              self._trace("Access denied")
119                              raise
120                          elif e[0] == FILTER_E_PASSWORD:
121                              self._trace("Password required")
122                              raise
123                          else:
124                              # any other type of error really can't be recovered from
125                              raise
126  
127                      # reset consecutive errors (some filters may get stuck in a lopp if embedding or link failures occurs
128                      errCnt = 0
129  
130                      if flags == CHUNK_TEXT:
131                          # its a text segment - get all available text for this chunk.
132                          body_chunks = properties.setdefault(propName, [])
133                          self._get_text(body_chunks)
134                      elif flags == CHUNK_VALUE:
135                          # its a data segment - get the value
136                          properties[propName] = self.f.GetValue()
137                      else:
138                          self._trace("Unknown flag returned by GetChunk:", flags)
139              finally:
140                  self.Close()
141  
142          except pythoncom.com_error as e:
143              self._trace("ERROR processing file", e)
144              raise
145  
146          return properties

148      def _bind_to_filter(self, fileName):
149          """ 
150          See if the file is a structured storage file or a normal file 
151          and then return an ifilter interface by calling the appropriate bind/load function 
152          """
153          if pythoncom.StgIsStorageFile(fileName):
154              self.stg = pythoncom.StgOpenStorage(
155                  fileName, None, storagecon.STGM_READ | storagecon.STGM_SHARE_DENY_WRITE
156              )
157              try:
158                  self.f = ifilter.BindIFilterFromStorage(self.stg)
159              except pythoncom.com_error as e:
160                  if (
161                      e[0] == -2147467262
162                  ):  # 0x80004002: # no interface, try the load interface (this happens for some MSoft files)
163                      self.f = ifilter.LoadIFilter(fileName)
164                  else:
165                      raise
166          else:
167              self.f = ifilter.LoadIFilter(fileName)
168              self.stg = None

170      def _get_text(self, body_chunks):
171          """ 
172          Gets all the text for a particular chunk. We need to keep calling get text till all the 
173          segments for this chunk are retrieved 
174          """
175          while True:
176              try:
177                  body_chunks.append(self.f.GetText())
178              except pythoncom.com_error as e:
179                  if e[0] in [
180                      FILTER_E_NO_MORE_TEXT,
181                      FILTER_E_NO_MORE_TEXT,
182                      FILTER_E_NO_TEXT,
183                  ]:
184                      break
185                  else:
186                      raise  # not one of the values we were expecting

188      def _get_properties(self, properties):
189          """ 
190          Use OLE property sets to get base properties 
191          """
192          try:
193              pss = self.stg.QueryInterface(pythoncom.IID_IPropertySetStorage)
194          except pythoncom.com_error as e:
195              self._trace("No Property information could be retrieved", e)
196              return
197  
198          ps = pss.Open(PSGUID_SUMMARYINFORMATION)
199  
200          props = (
201              PIDSI_TITLE,
202              PIDSI_SUBJECT,
203              PIDSI_AUTHOR,
204              PIDSI_KEYWORDS,
205              PIDSI_COMMENTS,
206          )
207  
208          title, subject, author, keywords, comments = ps.ReadMultiple(props)
209          if title is not None:
210              properties["title"] = title
211          if subject is not None:
212              properties["description"] = subject
213          if author is not None:
214              properties["author"] = author
215          if keywords is not None:
216              properties["keywords"] = keywords
217          if comments is not None:
218              properties["comments"] = comments

220      def _trace(self, *args):
221  
222          if self.verbose:
223              ret = " ".join([str(arg) for arg in args])
224              try:
225                  print(ret)
226              except IOError:
227                  pass
