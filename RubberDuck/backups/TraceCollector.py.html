9    def CollectorThread(stopEvent, file):
10       win32trace.InitRead()
11       handle = win32trace.GetHandle()
12       # Run this thread at a lower priority to the main message-loop (and printing output)
13       # thread can keep up
14       import win32process
15   
16       win32process.SetThreadPriority(
17           win32api.GetCurrentThread(), win32process.THREAD_PRIORITY_BELOW_NORMAL
18       )
19   
20       try:
21           while 1:
22               rc = win32event.WaitForMultipleObjects(
23                   (handle, stopEvent), 0, win32event.INFINITE
24               )
25               if rc == win32event.WAIT_OBJECT_0:
26                   # About the only char we can't live with is \0!
27                   file.write(win32trace.read().replace("\0", "<null>"))
28               else:
29                   # Stop event
30                   break
31       finally:
32           win32trace.TermRead()
33           print("Thread dieing")

36   class WindowOutput(winout.WindowOutput):
37       def __init__(self, *args):
38           winout.WindowOutput.__init__(*(self,) + args)
39           self.hStopThread = win32event.CreateEvent(None, 0, 0, None)
40           _thread.start_new(CollectorThread, (self.hStopThread, self))
41   
42       def _StopThread(self):
43           win32event.SetEvent(self.hStopThread)
44           self.hStopThread = None
45   
46       def Close(self):
47           self._StopThread()
48           winout.WindowOutput.Close(self)

56   def MakeOutputWindow():
57       # Note that it will not show until the first string written or
58       # you pass bShow = 1
59       global outputWindow
60       if outputWindow is None:
61           title = "Python Trace Collector"
62           # queueingFlag doesnt matter, as all output will come from new thread
63           outputWindow = WindowOutput(title, title)
64           # Let people know what this does!
65           msg = """\ 
66   # This window will display output from any programs that import win32traceutil 
67   # win32com servers registered with '--debug' are in this category. 
68   """
69           outputWindow.write(msg)
70       # force existing window open
71       outputWindow.write("")
72       return outputWindow
