tabbedpanel.py 
1    ''' 
2    TabbedPanel 
3    =========== 
4     
5    .. image:: images/tabbed_panel.jpg 
6        :align: right 
7     
8    .. versionadded:: 1.3.0 
9     
10    
11   The `TabbedPanel` widget manages different widgets in tabs, with a header area 
12   for the actual tab buttons and a content area for showing the current tab 
13   content. 
14    
15   The :class:`TabbedPanel` provides one default tab. 
16    
17   Simple example 
18   -------------- 
19    
20   .. include:: ../../examples/widgets/tabbedpanel.py 
21       :literal: 
22    
23   .. note:: 
24    
25       A new class :class:`TabbedPanelItem` has been introduced in 1.5.0 for 
26       convenience. So now one can simply add a :class:`TabbedPanelItem` to a 
27       :class:`TabbedPanel` and `content` to the :class:`TabbedPanelItem` 
28       as in the example provided above. 
29    
30   Customize the Tabbed Panel 
31   -------------------------- 
32    
33   You can choose the position in which the tabs are displayed:: 
34    
35       tab_pos = 'top_mid' 
36    
37   An individual tab is called a TabbedPanelHeader. It is a special button 
38   containing a `content` property. You add the TabbedPanelHeader first, and set 
39   its `content` property separately:: 
40    
41       tp = TabbedPanel() 
42       th = TabbedPanelHeader(text='Tab2') 
43       tp.add_widget(th) 
44    
45   An individual tab, represented by a TabbedPanelHeader, needs its content set. 
46   This content can be any widget. It could be a layout with a deep 
47   hierarchy of widgets, or it could be an individual widget, such as a label or a 
48   button:: 
49    
50       th.content = your_content_instance 
51    
52   There is one "shared" main content area active at any given time, for all 
53   the tabs. Your app is responsible for adding the content of individual tabs 
54   and for managing them, but it's not responsible for content switching. The 
55   tabbed panel handles switching of the main content object as per user action. 
56    
57   There is a default tab added when the tabbed panel is instantiated. 
58   Tabs that you add individually as above, are added in addition to the default 
59   tab. Thus, depending on your needs and design, you will want to customize the 
60   default tab:: 
61    
62       tp.default_tab_text = 'Something Specific To Your Use' 
63    
64    
65   The default tab machinery requires special consideration and management. 
66   Accordingly, an `on_default_tab` event is provided for associating a callback:: 
67    
68       tp.bind(default_tab = my_default_tab_callback) 
69    
70   It's important to note that by default, :attr:`default_tab_cls` is of type 
71   :class:`TabbedPanelHeader` and thus has the same properties as other tabs. 
72    
73   Since 1.5.0, it is now possible to disable the creation of the 
74   :attr:`default_tab` by setting :attr:`do_default_tab` to False. 
75    
76   Tabs and content can be removed in several ways:: 
77    
78       tp.remove_widget(widget/tabbed_panel_header) 
79       or 
80       tp.clear_widgets() # to clear all the widgets in the content area 
81       or 
82       tp.clear_tabs() # to remove the TabbedPanelHeaders 
83    
84   To access the children of the tabbed panel, use content.children:: 
85    
86       tp.content.children 
87    
88   To access the list of tabs:: 
89    
90       tp.tab_list 
91    
92   To change the appearance of the main tabbed panel content:: 
93    
94       background_color = (1, 0, 0, .5) #50% translucent red 
95       border = [0, 0, 0, 0] 
96       background_image = 'path/to/background/image' 
97    
98   To change the background of a individual tab, use these two properties:: 
99    
100      tab_header_instance.background_normal = 'path/to/tab_head/img' 
101      tab_header_instance.background_down = 'path/to/tab_head/img_pressed' 
102   
103  A TabbedPanelStrip contains the individual tab headers. To change the 
104  appearance of this tab strip, override the canvas of TabbedPanelStrip. 
105  For example, in the kv language: 
106   
107  .. code-block:: kv 
108   
109      <TabbedPanelStrip> 
110          canvas: 
111              Color: 
112                  rgba: (0, 1, 0, 1) # green 
113              Rectangle: 
114                  size: self.size 
115                  pos: self.pos 
116   
117  By default the tabbed panel strip takes its background image and color from the 
118  tabbed panel's background_image and background_color. 
119   
120  '''

149  class TabbedPanelHeader(ToggleButton):
150      '''A Base for implementing a Tabbed Panel Head. A button intended to be 
151      used as a Heading/Tab for a TabbedPanel widget. 
152   
153      You can use this TabbedPanelHeader widget to add a new tab to a 
154      TabbedPanel. 
155      '''
156  
157      content = ObjectProperty(None, allownone=True)
158      '''Content to be loaded when this tab header is selected. 
159   
160      :attr:`content` is an :class:`~kivy.properties.ObjectProperty` and defaults 
161      to None. 
162      '''
163  
164      # only allow selecting the tab if not already selected
165      def on_touch_down(self, touch):
166          if self.state == 'down':
167              # dispatch to children, not to self
168              for child in self.children:
169                  child.dispatch('on_touch_down', touch)
170              return
171          else:
172              super(TabbedPanelHeader, self).on_touch_down(touch)
173  
174      def on_release(self, *largs):
175          # Tabbed panel header is a child of tab_strib which has a
176          # `tabbed_panel` property
177          if self.parent:
178              self.parent.tabbed_panel.switch_to(self)
179          else:
180              # tab removed before we could switch to it. Switch back to
181              # previous tab
182              self.panel.switch_to(self.panel.current_tab)

185  class TabbedPanelItem(TabbedPanelHeader):
186      '''This is a convenience class that provides a header of type 
187      TabbedPanelHeader and links it with the content automatically. Thus 
188      facilitating you to simply do the following in kv language: 
189   
190      .. code-block:: kv 
191   
192          <TabbedPanel>: 
193              # ...other settings 
194              TabbedPanelItem: 
195                  BoxLayout: 
196                      Label: 
197                          text: 'Second tab content area' 
198                      Button: 
199                          text: 'Button that does nothing' 
200   
201      .. versionadded:: 1.5.0 
202      '''
203  
204      def add_widget(self, widget, index=0):
205          self.content = widget
206          if not self.parent:
207              return
208          panel = self.parent.tabbed_panel
209          if panel.current_tab == self:
210              panel.switch_to(self)
211  
212      def remove_widget(self, widget):
213          self.content = None
214          if not self.parent:
215              return
216          panel = self.parent.tabbed_panel
217          if panel.current_tab == self:
218              panel.remove_widget(widget)

471      def __init__(self, **kwargs):
472          # these variables need to be initialized before the kv lang is
473          # processed setup the base layout for the tabbed panel
474          self._childrens = []
475          self._tab_layout = StripLayout(rows=1)
476          self.rows = 1
477          self._tab_strip = TabbedPanelStrip(
478              tabbed_panel=self,
479              rows=1, size_hint=(None, None),
480              height=self.tab_height, width=self.tab_width)

482          self._partial_update_scrollview = None
483          self.content = TabbedPanelContent()
484          self._current_tab = self._original_tab \
485              = self._default_tab = TabbedPanelHeader()
486  
487          super(TabbedPanel, self).__init__(**kwargs)
488  
489          self.fbind('size', self._reposition_tabs)
490          if not self.do_default_tab:
491              Clock.schedule_once(self._switch_to_first_tab)
492              return
493          self._setup_default_tab()
494          self.switch_to(self.default_tab)

496      def switch_to(self, header, do_scroll=False):
497          '''Switch to a specific panel header. 
498   
499          .. versionchanged:: 1.10.0 
500   
501          If used with `do_scroll=True`, it scrolls 
502          to the header's tab too. 
503          '''
504          header_content = header.content
505          self._current_tab.state = 'normal'
506          header.state = 'down'
507          self._current_tab = header
508          self.clear_widgets()
509          if header_content is None:
510              return
511          # if content has a previous parent remove it from that parent
512          parent = header_content.parent
513          if parent:
514              parent.remove_widget(header_content)
515          self.add_widget(header_content)
516  
517          if do_scroll:
518              tabs = self._tab_strip
519              tabs.parent.scroll_to(header)
520  
521      def clear_tabs(self, *l):
522          self_tabs = self._tab_strip
523          self_tabs.clear_widgets()
524          if self.do_default_tab:
525              self_default_tab = self._default_tab
526              self_tabs.add_widget(self_default_tab)
527              self_tabs.width = self_default_tab.width
528          self._reposition_tabs()

530      def add_widget(self, widget, index=0):
531          content = self.content
532          if content is None:
533              return
534          parent = widget.parent
535          if parent:
536              parent.remove_widget(widget)
537          if widget in (content, self._tab_layout):
538              super(TabbedPanel, self).add_widget(widget, index)
539          elif isinstance(widget, TabbedPanelHeader):
540              self_tabs = self._tab_strip
541              self_tabs.add_widget(widget, index)
542              widget.group = '__tab%r__' % self_tabs.uid
543              self.on_tab_width()
544          else:
545              widget.pos_hint = {'x': 0, 'top': 1}
546              self._childrens.append(widget)
547              content.disabled = self.current_tab.disabled
548              content.add_widget(widget, index)
549  
550      def remove_widget(self, widget):
551          content = self.content
552          if content is None:
553              return
554          if widget in (content, self._tab_layout):
555              super(TabbedPanel, self).remove_widget(widget)
556          elif isinstance(widget, TabbedPanelHeader):
557              if not (self.do_default_tab and widget is self._default_tab):
558                  self_tabs = self._tab_strip
559                  self_tabs.width -= widget.width
560                  self_tabs.remove_widget(widget)
561                  if widget.state == 'down' and self.do_default_tab:
562                      self._default_tab.on_release()
563                  self._reposition_tabs()
564              else:
565                  Logger.info('TabbedPanel: default tab! can\'t be removed.\n' +
566                              'Change `default_tab` to a different tab.')
567          else:
568              if widget in self._childrens:
569                  self._childrens.remove(widget)
570              if widget in content.children:
571                  content.remove_widget(widget)
572  
573      def clear_widgets(self, **kwargs):
574          content = self.content
575          if content is None:
576              return
577          if kwargs.get('do_super', False):
578              super(TabbedPanel, self).clear_widgets()
579          else:
580              content.clear_widgets()

621      def _setup_default_tab(self):
622          if self._default_tab in self.tab_list:
623              return
624          content = self._default_tab.content
625          _tabs = self._tab_strip
626          cls = self.default_tab_cls
627  
628          if isinstance(cls, string_types):
629              cls = Factory.get(cls)
630  
631          if not issubclass(cls, TabbedPanelHeader):
632              raise TabbedPanelException('`default_tab_class` should be\ 
633                  subclassed from `TabbedPanelHeader`')
634  
635          # no need to instantiate if class is TabbedPanelHeader
636          if cls != TabbedPanelHeader:
637              self._current_tab = self._original_tab = self._default_tab = cls()
638  
639          default_tab = self.default_tab
640          if self._original_tab == self.default_tab:
641              default_tab.text = self.default_tab_text
642  
643          default_tab.height = self.tab_height
644          default_tab.group = '__tab%r__' % _tabs.uid
645          default_tab.state = 'down'
646          default_tab.width = self.tab_width if self.tab_width else 100
647          default_tab.content = content
648  
649          tl = self.tab_list
650          if default_tab not in tl:
651              _tabs.add_widget(default_tab, len(tl))
652  
653          if default_tab.content:
654              self.clear_widgets()
655              self.add_widget(self.default_tab.content)
656          else:
657              Clock.schedule_once(self._load_default_tab_content)
658          self._current_tab = default_tab
659  
660      def _switch_to_first_tab(self, *l):
661          ltl = len(self.tab_list) - 1
662          if ltl > -1:
663              self._current_tab = dt = self._original_tab \
664                  = self.tab_list[ltl]
665              self.switch_to(dt)
666  
667      def _load_default_tab_content(self, dt):
668          if self.default_tab:
669              self.switch_to(self.default_tab)
670  
671      def _reposition_tabs(self, *l):
672          ev = self._update_tabs_ev
673          if ev is None:
674              ev = self._update_tabs_ev = Clock.create_trigger(
675                  self._update_tabs, 0)
676          ev()
677  
678      def _update_tabs(self, *l):
679          self_content = self.content
680          if not self_content:
681              return
682          # cache variables for faster access
683          tab_pos = self.tab_pos
684          tab_layout = self._tab_layout
685          tab_layout.clear_widgets()
686          scrl_v = ScrollView(size_hint=(None, 1), always_overscroll=False)
687          tabs = self._tab_strip
688          parent = tabs.parent
689          if parent:
690              parent.remove_widget(tabs)
691          scrl_v.add_widget(tabs)
692          scrl_v.pos = (0, 0)
693          self_update_scrollview = self._update_scrollview
694  
695          # update scrlv width when tab width changes depends on tab_pos
696          if self._partial_update_scrollview is not None:
697              tabs.unbind(width=self._partial_update_scrollview)
698          self._partial_update_scrollview = partial(
699              self_update_scrollview, scrl_v)
700          tabs.bind(width=self._partial_update_scrollview)
701  
702          # remove all widgets from the tab_strip
703          self.clear_widgets(do_super=True)
704          tab_height = self.tab_height
705  
706          widget_list = []
707          tab_list = []
708          pos_letter = tab_pos[0]
709          if pos_letter == 'b' or pos_letter == 't':
710              # bottom or top positions
711              # one col containing the tab_strip and the content
712              self.cols = 1
713              self.rows = 2
714              # tab_layout contains the scrollview containing tabs and two blank
715              # dummy widgets for spacing
716              tab_layout.rows = 1
717              tab_layout.cols = 3
718              tab_layout.size_hint = (1, None)
719              tab_layout.height = (tab_height + tab_layout.padding[1] +
720                                   tab_layout.padding[3] + dp(2))
721              self_update_scrollview(scrl_v)
722  
723              if pos_letter == 'b':
724                  # bottom
725                  if tab_pos == 'bottom_mid':
726                      tab_list = (Widget(), scrl_v, Widget())
727                      widget_list = (self_content, tab_layout)
728                  else:
729                      if tab_pos == 'bottom_left':
730                          tab_list = (scrl_v, Widget(), Widget())
731                      elif tab_pos == 'bottom_right':
732                          # add two dummy widgets
733                          tab_list = (Widget(), Widget(), scrl_v)
734                      widget_list = (self_content, tab_layout)
735              else:
736                  # top
737                  if tab_pos == 'top_mid':
738                      tab_list = (Widget(), scrl_v, Widget())
739                  elif tab_pos == 'top_left':
740                      tab_list = (scrl_v, Widget(), Widget())
741                  elif tab_pos == 'top_right':
742                      tab_list = (Widget(), Widget(), scrl_v)
743                  widget_list = (tab_layout, self_content)
744          elif pos_letter == 'l' or pos_letter == 'r':
745              # left ot right positions
746              # one row containing the tab_strip and the content
747              self.cols = 2
748              self.rows = 1
749              # tab_layout contains two blank dummy widgets for spacing
750              # "vertically" and the scatter containing scrollview
751              # containing tabs
752              tab_layout.rows = 3
753              tab_layout.cols = 1
754              tab_layout.size_hint = (None, 1)
755              tab_layout.width = tab_height
756              scrl_v.height = tab_height
757              self_update_scrollview(scrl_v)
758  
759              # rotate the scatter for vertical positions
760              rotation = 90 if tab_pos[0] == 'l' else -90
761              sctr = Scatter(do_translation=False,
762                             rotation=rotation,
763                             do_rotation=False,
764                             do_scale=False,
765                             size_hint=(None, None),
766                             auto_bring_to_front=False,
767                             size=scrl_v.size)
768              sctr.add_widget(scrl_v)
769  
770              lentab_pos = len(tab_pos)
771  
772              # Update scatter's top when its pos changes.
773              # Needed for repositioning scatter to the correct place after its
774              # added to the parent. Use clock_schedule_once to ensure top is
775              # calculated after the parent's pos on canvas has been calculated.
776              # This is needed for when tab_pos changes to correctly position
777              # scatter. Without clock.schedule_once the positions would look
778              # fine but touch won't translate to the correct position
779  
780              if tab_pos[lentab_pos - 4:] == '_top':
781                  # on positions 'left_top' and 'right_top'
782                  sctr.bind(pos=partial(self._update_top, sctr, 'top', None))
783                  tab_list = (sctr, )
784              elif tab_pos[lentab_pos - 4:] == '_mid':
785                  # calculate top of scatter
786                  sctr.bind(pos=partial(self._update_top, sctr, 'mid',
787                                        scrl_v.width))
788                  tab_list = (Widget(), sctr, Widget())
789              elif tab_pos[lentab_pos - 7:] == '_bottom':
790                  tab_list = (Widget(), Widget(), sctr)
791  
792              if pos_letter == 'l':
793                  widget_list = (tab_layout, self_content)
794              else:
795                  widget_list = (self_content, tab_layout)
796  
797          # add widgets to tab_layout
798          add = tab_layout.add_widget
799          for widg in tab_list:
800              add(widg)
801  
802          # add widgets to self
803          add = self.add_widget
804          for widg in widget_list:
805              add(widg)

807      def _update_tab_width(self, *l):
808          if self.tab_width:
809              for tab in self.tab_list:
810                  tab.size_hint_x = 1
811              tsw = self.tab_width * len(self._tab_strip.children)
812          else:
813              # tab_width = None
814              tsw = 0
815              for tab in self.tab_list:
816                  if tab.size_hint_x:
817                      # size_hint_x: x/.xyz
818                      tab.size_hint_x = 1
819                      # drop to default tab_width
820                      tsw += 100
821                  else:
822                      # size_hint_x: None
823                      tsw += tab.width
824          self._tab_strip.width = tsw
825          self._reposition_tabs()
826  
827      def _update_top(self, *args):
828          sctr, top, scrl_v_width, x, y = args
829          ev = self._update_top_ev
830          if ev is not None:
831              ev.cancel()
832  
833          ev = self._update_top_ev = Clock.schedule_once(
834              partial(self._updt_top, sctr, top, scrl_v_width), 0)
835  
836      def _updt_top(self, sctr, top, scrl_v_width, *args):
837          if top[0] == 't':
838              sctr.top = self.top
839          else:
840              sctr.top = self.top - (self.height - scrl_v_width) / 2
841  
842      def _update_scrollview(self, scrl_v, *l):
843          self_tab_pos = self.tab_pos
844          self_tabs = self._tab_strip
845          if self_tab_pos[0] == 'b' or self_tab_pos[0] == 't':
846              # bottom or top
847              scrl_v.width = min(self.width, self_tabs.width)
848              # required for situations when scrl_v's pos is calculated
849              # when it has no parent
850              scrl_v.top += 1
851              scrl_v.top -= 1
852          else:
853              # left or right
854              scrl_v.width = min(self.height, self_tabs.width)
855              self_tabs.pos = (0, 0)
