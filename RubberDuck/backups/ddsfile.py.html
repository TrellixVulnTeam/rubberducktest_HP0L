1    ''' 
2    DDS File library 
3    ================ 
4     
5    This library can be used to parse and save DDS 
6    (`DirectDraw Surface <https://en.wikipedia.org/wiki/DirectDraw_Surface>`) 
7    files. 
8     
9    The initial version was written by:: 
10    
11       Alexey Borzenkov (snaury@gmail.com) 
12    
13   All the initial work credits go to him! Thank you :) 
14    
15   This version uses structs instead of ctypes. 
16    
17    
18   DDS Format 
19   ---------- 
20    
21   :: 
22    
23       [DDS ][SurfaceDesc][Data] 
24    
25       [SurfaceDesc]:: (everything is uint32) 
26           Size 
27           Flags 
28           Height 
29           Width 
30           PitchOrLinearSize 
31           Depth 
32           MipmapCount 
33           Reserved1 * 11 
34           [PixelFormat]:: 
35               Size 
36               Flags 
37               FourCC 
38               RGBBitCount 
39               RBitMask 
40               GBitMask 
41               BBitMask 
42               ABitMask 
43           [Caps]:: 
44               Caps1 
45               Caps2 
46               Reserved1 * 2 
47           Reserverd2 
48    
49   .. warning:: 
50    
51       This is an external library and Kivy does not provide any support for it. 
52       It might change in the future and we advise you don't rely on it in your 
53       code. 
54    
55   '''

58   from struct import pack, unpack, calcsize

60   # DDSURFACEDESC2 dwFlags
61   DDSD_CAPS                  = 0x00000001
62   DDSD_HEIGHT                = 0x00000002
63   DDSD_WIDTH                 = 0x00000004
64   DDSD_PITCH                 = 0x00000008
65   DDSD_PIXELFORMAT           = 0x00001000
66   DDSD_MIPMAPCOUNT           = 0x00020000
67   DDSD_LINEARSIZE            = 0x00080000
68   DDSD_DEPTH                 = 0x00800000
69   
70   # DDPIXELFORMAT dwFlags
71   DDPF_ALPHAPIXELS           = 0x00000001
72   DDPF_FOURCC                = 0x00000004
73   DDPF_RGB                   = 0x00000040
74   DDPF_LUMINANCE             = 0x00020000
75   
76   # DDSCAPS2 dwCaps1
77   DDSCAPS_COMPLEX            = 0x00000008
78   DDSCAPS_TEXTURE            = 0x00001000
79   DDSCAPS_MIPMAP             = 0x00400000
80   
81   # DDSCAPS2 dwCaps2
82   DDSCAPS2_CUBEMAP           = 0x00000200
83   DDSCAPS2_CUBEMAP_POSITIVEX = 0x00000400
84   DDSCAPS2_CUBEMAP_NEGATIVEX = 0x00000800
85   DDSCAPS2_CUBEMAP_POSITIVEY = 0x00001000
86   DDSCAPS2_CUBEMAP_NEGATIVEY = 0x00002000
87   DDSCAPS2_CUBEMAP_POSITIVEZ = 0x00004000
88   DDSCAPS2_CUBEMAP_NEGATIVEZ = 0x00008000
89   DDSCAPS2_VOLUME            = 0x00200000
90   
91   # Common FOURCC codes
92   DDS_DXTN = 0x00545844
93   DDS_DXT1 = 0x31545844
94   DDS_DXT2 = 0x32545844
95   DDS_DXT3 = 0x33545844
96   DDS_DXT4 = 0x34545844
97   DDS_DXT5 = 0x35545844

99   def dxt_to_str(dxt):
100      if dxt == DDS_DXT1:
101          return 's3tc_dxt1'
102      elif dxt == DDS_DXT2:
103          return 's3tc_dxt2'
104      elif dxt == DDS_DXT3:
105          return 's3tc_dxt3'
106      elif dxt == DDS_DXT4:
107          return 's3tc_dxt4'
108      elif dxt == DDS_DXT5:
109          return 's3tc_dxt5'
110      elif dxt == 0:
111          return 'rgba'
112      elif dxt == 1:
113          return 'alpha'
114      elif dxt == 2:
115          return 'luminance'
116      elif dxt == 3:
117          return 'luminance_alpha'

119  def str_to_dxt(dxt):
120      if dxt == 's3tc_dxt1':
121          return DDS_DXT1
122      if dxt == 's3tc_dxt2':
123          return DDS_DXT2
124      if dxt == 's3tc_dxt3':
125          return DDS_DXT3
126      if dxt == 's3tc_dxt4':
127          return DDS_DXT4
128      if dxt == 's3tc_dxt5':
129          return DDS_DXT5
130      if dxt == 'rgba':
131          return 0
132      if dxt == 'alpha':
133          return 1
134      if dxt == 'luminance':
135          return 2
136      if dxt == 'luminance_alpha':
137          return 3

139  def align_value(val, b):
140      return val + (-val % b)
141  
142  def check_flags(val, fl):
143      return (val & fl) == fl
144  
145  def dxt_size(w, h, dxt):
146      w = max(1, w // 4)
147      h = max(1, h // 4)
148      if dxt == DDS_DXT1:
149          return w * h * 8
150      elif dxt in (DDS_DXT2, DDS_DXT3, DDS_DXT4, DDS_DXT5):
151          return w * h * 16
152      return -1

154  class QueryDict(dict):
155      def __getattr__(self, attr):
156          try:
157              return self.__getitem__(attr)
158          except KeyError:
159              try:
160                  return super(QueryDict, self).__getattr__(attr)
161              except AttributeError:
162                  raise KeyError(attr)
163  
164      def __setattr__(self, attr, value):
165          self.__setitem__(attr, value)

167  class DDSException(Exception):
168      pass
169

171  class DDSFile(object):
172      fields = (
173          ('size', 0), ('flags', 1), ('height', 2),
174          ('width', 3), ('pitchOrLinearSize', 4), ('depth', 5),
175          ('mipmapCount', 6), ('pf_size', 18), ('pf_flags', 19),
176          ('pf_fourcc', 20), ('pf_rgbBitCount', 21), ('pf_rBitMask', 22),
177          ('pf_gBitMask', 23), ('pf_bBitMask', 24), ('pf_aBitMask', 25),
178          ('caps1', 26), ('caps2', 27))
179  
180      def __init__(self, filename=None):
181          super(DDSFile, self).__init__()
182          self._dxt = 0
183          self._fmt = None
184          self.meta = meta = QueryDict()
185          self.count = 0
186          self.images = []
187          self.images_size = []
188          for field, index in DDSFile.fields:
189              meta[field] = 0
190          if filename:
191              self.load(filename)
192  
193      def load(self, filename):
194          self.filename = filename
195          with open(filename, 'rb') as fd:
196              data = fd.read()
197  
198          # ensure magic
199          magic_header = data[:4]
200          if isinstance(magic_header, bytes):
201              magic_header = magic_header.decode()
202          if data[:4] != 'DDS ':
203              raise DDSException('Invalid magic header')
204  
205          # read header
206          fmt = 'I' * 31
207          fmt_size = calcsize(fmt)
208          pf_size = calcsize('I' * 8)
209          header, data = data[4:4+fmt_size], data[4+fmt_size:]
210          if len(header) != fmt_size:
211              raise DDSException('Truncated header in')
212  
213          # depack
214          header = unpack(fmt, header)
215          meta = self.meta
216          for name, index in DDSFile.fields:
217              meta[name] = header[index]
218  
219          # check header validity
220          if meta.size != fmt_size:
221              raise DDSException('Invalid header size (%d instead of %d)' %
222                      (meta.size, fmt_size))
223          if meta.pf_size != pf_size:
224              raise DDSException('Invalid pixelformat size (%d instead of %d)' %
225                      (meta.pf_size, pf_size))
226          if not check_flags(meta.flags,
227                  DDSD_CAPS | DDSD_PIXELFORMAT | DDSD_WIDTH | DDSD_HEIGHT):
228              raise DDSException('Not enough flags')
229          if not check_flags(meta.caps1, DDSCAPS_TEXTURE):
230              raise DDSException('Not a DDS texture')
231  
232          self.count = 1
233          if check_flags(meta.flags, DDSD_MIPMAPCOUNT):
234              if not check_flags(meta.caps1, DDSCAPS_COMPLEX | DDSCAPS_MIPMAP):
235                  raise DDSException('Invalid mipmap without flags')
236              self.count = meta.mipmapCount
237  
238          hasrgb = check_flags(meta.pf_flags, DDPF_RGB)
239          hasalpha = check_flags(meta.pf_flags, DDPF_ALPHAPIXELS)
240          hasluminance = check_flags(meta.pf_flags, DDPF_LUMINANCE)
241          bpp = None
242          dxt = block = pitch = 0
243          if hasrgb or hasalpha or hasluminance:
244              bpp = meta.pf_rgbBitCount
245  
246          if hasrgb and hasluminance:
247              raise DDSException('File have RGB and Luminance')
248  
249          if hasrgb:
250              dxt = 0
251          elif hasalpha and not hasluminance:
252              dxt = 1
253          elif hasluminance and not hasalpha:
254              dxt = 2
255          elif hasalpha and hasluminance:
256              dxt = 3
257          elif check_flags(meta.pf_flags, DDPF_FOURCC):
258              dxt = meta.pf_fourcc
259              if dxt not in (DDS_DXT1, DDS_DXT2, DDS_DXT3, DDS_DXT4, DDS_DXT5):
260                  raise DDSException('Unsupported FOURCC')
261          else:
262              raise DDSException('Unsupported format specified')
263  
264          if bpp:
265              block = align_value(bpp, 8) // 8
266              pitch = align_value(block * meta.width, 4)
267  
268          if check_flags(meta.flags, DDSD_LINEARSIZE):
269              if dxt in (0, 1, 2, 3):
270                  size = pitch * meta.height
271              else:
272                  size = dxt_size(meta.width, meta.height, dxt)
273  
274          w = meta.width
275          h = meta.height
276          images = self.images
277          images_size = self.images_size
278          for i in range(self.count):
279              if dxt in (0, 1, 2, 3):
280                  size = align_value(block * w, 4) * h
281              else:
282                  size = dxt_size(w, h, dxt)
283              image, data = data[:size], data[size:]
284              if len(image) < size:
285                  raise DDSException('Truncated image for mipmap %d' % i)
286              images_size.append((w, h))
287              images.append(image)
288              if w == 1 and h == 1:
289                  break
290              w = max(1, w // 2)
291              h = max(1, h // 2)
292  
293          if len(images) == 0:
294              raise DDSException('No images available')
295          if len(images) < self.count:
296              raise DDSException('Not enough images')
297  
298          self._dxt = dxt
299  
300      def save(self, filename):
301          if len(self.images) == 0:
302              raise DDSException('No images to save')
303  
304          fields = dict(DDSFile.fields)
305          fields_keys = list(fields.keys())
306          fields_index = list(fields.values())
307          mget = self.meta.get
308          header = []
309          for idx in range(31):
310              if idx in fields_index:
311                  value = mget(fields_keys[fields_index.index(idx)], 0)
312              else:
313                  value = 0
314              header.append(value)
315  
316          with open(filename, 'wb') as fd:
317              fd.write('DDS ')
318              fd.write(pack('I' * 31, *header))
319              for image in self.images:
320                  fd.write(image)

322      def add_image(self, level, bpp, fmt, width, height, data):
323          assert(bpp == 32)
324          assert(fmt in ('rgb', 'rgba', 'dxt1', 'dxt2', 'dxt3', 'dxt4', 'dxt5'))
325          assert(width > 0)
326          assert(height > 0)
327          assert(level >= 0)
328  
329          meta = self.meta
330          images = self.images
331          if len(images) == 0:
332              assert(level == 0)
333  
334              # first image, set defaults !
335              for k in meta.keys():
336                  meta[k] = 0
337  
338              self._fmt = fmt
339              meta.size = calcsize('I' * 31)
340              meta.pf_size = calcsize('I' * 8)
341              meta.pf_flags = 0
342              meta.flags = DDSD_CAPS | DDSD_PIXELFORMAT | DDSD_WIDTH | DDSD_HEIGHT
343              meta.width = width
344              meta.height = height
345              meta.caps1 = DDSCAPS_TEXTURE
346  
347              meta.flags |= DDSD_LINEARSIZE
348              meta.pitchOrLinearSize = len(data)
349  
350              meta.pf_rgbBitCount = 32
351              meta.pf_rBitMask = 0x00ff0000
352              meta.pf_gBitMask = 0x0000ff00
353              meta.pf_bBitMask = 0x000000ff
354              meta.pf_aBitMask = 0xff000000
355  
356              if fmt in ('rgb', 'rgba'):
357                  assert(True)
358                  assert(bpp == 32)
359                  meta.pf_flags |= DDPF_RGB
360                  meta.pf_rgbBitCount = 32
361                  meta.pf_rBitMask = 0x00ff0000
362                  meta.pf_gBitMask = 0x0000ff00
363                  meta.pf_bBitMask = 0x000000ff
364                  meta.pf_aBitMask = 0x00000000
365                  if fmt == 'rgba':
366                      meta.pf_flags |= DDPF_ALPHAPIXELS
367                      meta.pf_aBitMask = 0xff000000
368              else:
369                  meta.pf_flags |= DDPF_FOURCC
370                  if fmt == 'dxt1':
371                      meta.pf_fourcc = DDS_DXT1
372                  elif fmt == 'dxt2':
373                      meta.pf_fourcc = DDS_DXT2
374                  elif fmt == 'dxt3':
375                      meta.pf_fourcc = DDS_DXT3
376                  elif fmt == 'dxt4':
377                      meta.pf_fourcc = DDS_DXT4
378                  elif fmt == 'dxt5':
379                      meta.pf_fourcc = DDS_DXT5
380  
381              images.append(data)
382          else:
383              assert(level == len(images))
384              assert(fmt == self._fmt)
385  
386              images.append(data)
387  
388              meta.flags |= DDSD_MIPMAPCOUNT
389              meta.caps1 |= DDSCAPS_COMPLEX | DDSCAPS_MIPMAP
390              meta.mipmapCount = len(images)
391  
392      def __repr__(self):
393          return '<DDSFile filename=%r size=%r dxt=%r len(images)=%r>' % (
394                  self.filename, self.size, self.dxt, len(self.images))
395  
396      def _get_size(self):
397          meta = self.meta
398          return meta.width, meta.height
399      def _set_size(self, size):
400          self.meta.update({'width': size[0], 'height': size[1]})
401      size = property(_get_size, _set_size)
402  
403      def _get_dxt(self):
404          return dxt_to_str(self._dxt)
405      def _set_dxt(self, dxt):
406          self._dxt = str_to_dxt(dxt)
407      dxt = property(_get_dxt, _set_dxt)
