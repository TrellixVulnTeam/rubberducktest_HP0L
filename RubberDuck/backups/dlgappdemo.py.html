1    # dlgappdemo - a demo of a dialog application.
2    # This is a demonstration of both a custom "application" module,
3    # and a Python program in a dialog box.
4    #
5    # NOTE:  You CAN NOT import this module from either PythonWin or Python.
6    # This module must be specified on the commandline to PythonWin only.
7    # eg, PythonWin /app dlgappdemo.py
8    
9    from pywin.framework import dlgappcore, app
10   import win32ui
11   import sys
12   
13   
14   class TestDialogApp(dlgappcore.DialogApp):
15       def CreateDialog(self):
16           return TestAppDialog()
17   
18   
19   class TestAppDialog(dlgappcore.AppDialog):
20       def __init__(self):
21           self.edit = None
22           dlgappcore.AppDialog.__init__(self, win32ui.IDD_LARGE_EDIT)
23   
24       def OnInitDialog(self):
25           self.SetWindowText("Test dialog application")
26           self.edit = self.GetDlgItem(win32ui.IDC_EDIT1)
27           print("Hello from Python")
28           print("args are:", end=" ")
29           for arg in sys.argv:
30               print(arg)
31           return 1
32   
33       def PreDoModal(self):
34           sys.stdout = sys.stderr = self
35   
36       def write(self, str):
37           if self.edit:
38               self.edit.SetSel(-2)
39               # translate \n to \n\r
40               self.edit.ReplaceSel(str.replace("\n", "\r\n"))
41           else:
42               win32ui.OutputDebug("dlgapp - no edit control! >>\n%s\n<<\n" % str)
43   
44   
45   app.AppBuilder = TestDialogApp
