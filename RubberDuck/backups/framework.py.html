213      def GetSourceTypeInfo(self, typeinfo):
214          """Gets the typeinfo for the Source Events for the passed typeinfo"""
215          attr = typeinfo.GetTypeAttr()
216          cFuncs = attr[6]
217          typeKind = attr[5]
218          if typeKind not in [pythoncom.TKIND_COCLASS, pythoncom.TKIND_INTERFACE]:
219              RaiseAssert(
220                  winerror.E_UNEXPECTED, "The typeKind of the object is unexpected"
221              )
222          cImplType = attr[8]
223          for i in range(cImplType):
224              # Look for the [source, default] interface on the coclass
225              # that isn't marked as restricted.
226              flags = typeinfo.GetImplTypeFlags(i)
227              flagsNeeded = (
228                  pythoncom.IMPLTYPEFLAG_FDEFAULT | pythoncom.IMPLTYPEFLAG_FSOURCE
229              )
230              if (flags & (flagsNeeded | pythoncom.IMPLTYPEFLAG_FRESTRICTED)) == (
231                  flagsNeeded
232              ):
233                  # Get the handle to the implemented interface.
234                  href = typeinfo.GetRefTypeOfImplType(i)
235                  return typeinfo.GetRefTypeInfo(href)

237      def BuildEvents(self):
238          # See if it is an extender object.
239          try:
240              mainTypeInfo = self.coDispatch.QueryInterface(
241                  axscript.IID_IProvideMultipleClassInfo
242              )
243              isMulti = 1
244              numTypeInfos = mainTypeInfo.GetMultiTypeInfoCount()
245          except pythoncom.com_error:
246              isMulti = 0
247              numTypeInfos = 1
248              try:
249                  mainTypeInfo = self.coDispatch.QueryInterface(
250                      pythoncom.IID_IProvideClassInfo
251                  )
252              except pythoncom.com_error:
253                  numTypeInfos = 0
254          # Create an event handler for the item.
255          for item in range(numTypeInfos):
256              if isMulti:
257                  typeinfo, flags = mainTypeInfo.GetInfoOfIndex(
258                      item, axscript.MULTICLASSINFO_GETTYPEINFO
259                  )
260              else:
261                  typeinfo = mainTypeInfo.GetClassInfo()
262              sourceType = self.GetSourceTypeInfo(typeinfo)
263              cFuncs = 0
264              if sourceType:
265                  attr = sourceType.GetTypeAttr()
266                  self.iid = attr[0]
267                  cFuncs = attr[6]
268                  for i in range(cFuncs):
269                      funcdesc = sourceType.GetFuncDesc(i)
270                      event = Event()
271                      event.Build(sourceType, funcdesc)
272                      self.events[event.dispid] = event

275          if self.connection is not None or self.iid is None:
276              return
277          #       trace("Connect for sink item", self.myScriptItem.name, "with IID",str(self.iid))
278          self.connection = win32com.client.connect.SimpleConnection(
279              self.coDispatch, self, self.iid
280          )

284              try:
285                  self.connection.Disconnect()
286              except pythoncom.com_error:
287                  pass  # Ignore disconnection errors.
288              self.connection = None

400      def GetCreateSubItem(self, parentItem, name, dispatch, flags):
401          keyName = name.lower()
402          try:
403              rc = self.subItems[keyName]
404              # No changes allowed to existing flags.
405              if not rc.flags is None and not flags is None and rc.flags != flags:
406                  raise Exception(scode=winerror.E_INVALIDARG)
407              # Existing item must not have a dispatch.
408              if not rc.dispatch is None and not dispatch is None:
409                  raise Exception(scode=winerror.E_INVALIDARG)
410              rc.flags = flags  # Setup the real flags.
411              rc.dispatch = dispatch
412          except KeyError:
413              rc = self.subItems[keyName] = self.GetSubItemClass()(
414                  parentItem, name, dispatch, flags
415              )
416          return rc

421      def CreateConnections(self):
422          # Create (but do not connect to) the connection points.
423          if self.createdConnections:
424              return

427          # This flags means self, _and_ children, are connectable.

457      def FindBuildSubItemEvents(self):
458          # Called during connection to event source.  Seeks out and connects to
459          # all children.  As per the AX spec, this is not recursive
460          # (ie, children sub-items are not seeked)
461          try:
462              multiTypeInfo = self.dispatch.QueryInterface(
463                  axscript.IID_IProvideMultipleClassInfo
464              )
465              numTypeInfos = multiTypeInfo.GetMultiTypeInfoCount()
466          except pythoncom.com_error:
467              return
468          for item in range(numTypeInfos):
469              typeinfo, flags = multiTypeInfo.GetInfoOfIndex(
470                  item, axscript.MULTICLASSINFO_GETTYPEINFO
471              )
472              defaultType = self.GetDefaultSourceTypeInfo(typeinfo)
473              index = 0
474              while 1:
475                  try:
476                      fdesc = defaultType.GetFuncDesc(index)
477                  except pythoncom.com_error:
478                      break  # No more funcs
479                  index = index + 1
480                  dispid = fdesc[0]
481                  funckind = fdesc[3]
482                  invkind = fdesc[4]
483                  elemdesc = fdesc[8]
484                  funcflags = fdesc[9]
485                  try:
486                      isSubObject = (
487                          not (funcflags & pythoncom.FUNCFLAG_FRESTRICTED)
488                          and funckind == pythoncom.FUNC_DISPATCH
489                          and invkind == pythoncom.INVOKE_PROPERTYGET
490                          and elemdesc[0][0] == pythoncom.VT_PTR
491                          and elemdesc[0][1][0] == pythoncom.VT_USERDEFINED
492                      )
493                  except:
494                      isSubObject = 0
495                  if isSubObject:
496                      try:
497                          # We found a sub-object.
498                          names = typeinfo.GetNames(dispid)
499                          result = self.dispatch.Invoke(
500                              dispid, 0x0, pythoncom.DISPATCH_PROPERTYGET, 1
501                          )
502                          # IE has an interesting problem - there are lots of synonyms for the same object.  Eg
503                          # in a simple form, "window.top", "window.window", "window.parent", "window.self"
504                          # all refer to the same object.  Our event implementation code does not differentiate
505                          # eg, "window_onload" will fire for *all* objects named "window".  Thus,
506                          # "window" and "window.window" will fire the same event handler :(
507                          # One option would be to check if the sub-object is indeed the
508                          # parent object - however, this would stop "top_onload" from firing,
509                          # as no event handler for "top" would work.
510                          # I think we simply need to connect to a *single* event handler.
511                          # As use in IE is deprecated, I am not solving this now.
512                          if type(result) == pythoncom.TypeIIDs[pythoncom.IID_IDispatch]:
513                              name = names[0]
514                              subObj = self.GetCreateSubItem(
515                                  self, name, result, axscript.SCRIPTITEM_ISVISIBLE
516                              )
517                              # print "subobj", name, "flags are", subObj.flags, "mydisp=", self.dispatch, "result disp=", result, "compare=", self.dispatch==result
518                              subObj.BuildEvents()
519                              subObj.Register()
520                      except pythoncom.com_error:
521                          pass

523      def GetDefaultSourceTypeInfo(self, typeinfo):
524          """Gets the typeinfo for the Default Dispatch for the passed typeinfo"""
525          attr = typeinfo.GetTypeAttr()
526          cFuncs = attr[6]
527          typeKind = attr[5]
528          if typeKind not in [pythoncom.TKIND_COCLASS, pythoncom.TKIND_INTERFACE]:
529              RaiseAssert(
530                  winerror.E_UNEXPECTED, "The typeKind of the object is unexpected"
531              )
532          cImplType = attr[8]
533          for i in range(cImplType):
534              # Look for the [source, default] interface on the coclass
535              # that isn't marked as restricted.
536              flags = typeinfo.GetImplTypeFlags(i)
537              if (
538                  flags
539                  & (
540                      pythoncom.IMPLTYPEFLAG_FDEFAULT
541                      | pythoncom.IMPLTYPEFLAG_FSOURCE
542                      | pythoncom.IMPLTYPEFLAG_FRESTRICTED
543                  )
544              ) == pythoncom.IMPLTYPEFLAG_FDEFAULT:
545                  # Get the handle to the implemented interface.
546                  href = typeinfo.GetRefTypeOfImplType(i)
547                  defTypeInfo = typeinfo.GetRefTypeInfo(href)
548                  attr = defTypeInfo.GetTypeAttr()
549                  typeKind = attr[5]
550                  typeFlags = attr[11]
551                  if (
552                      typeKind == pythoncom.TKIND_INTERFACE
553                      and typeFlags & pythoncom.TYPEFLAG_FDUAL
554                  ):
555                      # Get corresponding Disp interface
556                      # -1 is a special value which does this for us.
557                      href = typeinfo.GetRefTypeOfImplType(-1)
558                      return defTypeInfo.GetRefTypeInfo(href)
559                  else:
560                      return defTypeInfo

651      def ParseScriptText(
652          self,
653          code,
654          itemName,
655          context,
656          delimiter,
657          sourceContextCookie,
658          startLineNumber,
659          flags,
660          bWantResult,
661      ):
662          #       trace ("ParseScriptText", code[:20],"...", itemName, context, delimiter, sourceContextCookie, startLineNumber, flags, bWantResult)
663          if (
664              bWantResult
665              or self.scriptState == axscript.SCRIPTSTATE_STARTED
666              or self.scriptState == axscript.SCRIPTSTATE_CONNECTED
667              or self.scriptState == axscript.SCRIPTSTATE_DISCONNECTED
668          ):
669              flags = flags | SCRIPTTEXT_FORCEEXECUTION
670          else:
671              flags = flags & (~SCRIPTTEXT_FORCEEXECUTION)
672  
673          if flags & SCRIPTTEXT_FORCEEXECUTION:
674              # About to execute the code.
675              self.RegisterNewNamedItems()
676          return self.DoParseScriptText(
677              code, sourceContextCookie, startLineNumber, bWantResult, flags
678          )

760      def SetScriptState(self, state):
761          # print "SetScriptState with %s - currentstate = %s" % (state_map.get(state),state_map.get(self.scriptState))
762          if state == self.scriptState:
763              return
764          # If closed, allow no other state transitions
765          if self.scriptState == axscript.SCRIPTSTATE_CLOSED:
766              raise Exception(scode=winerror.E_INVALIDARG)
767  
768          if state == axscript.SCRIPTSTATE_INITIALIZED:
769              # Re-initialize - shutdown then reset.
770              if self.scriptState in [
771                  axscript.SCRIPTSTATE_CONNECTED,
772                  axscript.SCRIPTSTATE_STARTED,
773              ]:
774                  self.Stop()
775          elif state == axscript.SCRIPTSTATE_STARTED:
776              if self.scriptState == axscript.SCRIPTSTATE_CONNECTED:
777                  self.Disconnect()
778              if self.scriptState == axscript.SCRIPTSTATE_DISCONNECTED:
779                  self.Reset()
780              self.Run()
781              self.ChangeScriptState(axscript.SCRIPTSTATE_STARTED)
782          elif state == axscript.SCRIPTSTATE_CONNECTED:
783              if self.scriptState in [
784                  axscript.SCRIPTSTATE_UNINITIALIZED,
785                  axscript.SCRIPTSTATE_INITIALIZED,
786              ]:
787                  self.ChangeScriptState(
788                      axscript.SCRIPTSTATE_STARTED
789                  )  # report transition through started
790                  self.Run()
791              if self.scriptState == axscript.SCRIPTSTATE_STARTED:
792                  self.Connect()
793                  self.ChangeScriptState(state)
794          elif state == axscript.SCRIPTSTATE_DISCONNECTED:
795              if self.scriptState == axscript.SCRIPTSTATE_CONNECTED:
796                  self.Disconnect()
797          elif state == axscript.SCRIPTSTATE_CLOSED:
798              self.Close()
799          elif state == axscript.SCRIPTSTATE_UNINITIALIZED:
800              if self.scriptState == axscript.SCRIPTSTATE_STARTED:
801                  self.Stop()
802              if self.scriptState == axscript.SCRIPTSTATE_CONNECTED:
803                  self.Disconnect()
804              if self.scriptState == axscript.SCRIPTSTATE_DISCONNECTED:
805                  self.Reset()
806              self.ChangeScriptState(state)
807          else:
808              raise Exception(scode=winerror.E_INVALIDARG)

1171     def HandleException(self, codeBlock):
1172         # NOTE - Never returns - raises a ComException
1173         exc_type, exc_value, exc_traceback = sys.exc_info()
1174         # If a SERVER exception, re-raise it.  If a client side COM error, it is
1175         # likely to have originated from the script code itself, and therefore
1176         # needs to be reported like any other exception.
1177         if IsCOMServerException(exc_type):
1178             # Ensure the traceback doesnt cause a cycle.
1179             exc_traceback = None
1180             raise
1181         # It could be an error by another script.
1182         if (
1183             issubclass(pythoncom.com_error, exc_type)
1184             and exc_value.hresult == axscript.SCRIPT_E_REPORTED
1185         ):
1186             # Ensure the traceback doesnt cause a cycle.
1187             exc_traceback = None
1188             raise Exception(scode=exc_value.hresult)
1189 
1190         exception = error.AXScriptException(
1191             self, codeBlock, exc_type, exc_value, exc_traceback
1192         )
1193 
1194         # Ensure the traceback doesnt cause a cycle.
1195         exc_traceback = None
1196         result_exception = error.ProcessAXScriptException(
1197             self.scriptSite, self.debugManager, exception
1198         )
1199         if result_exception is not None:
1200             try:
1201                 self.scriptSite.OnScriptTerminate(None, result_exception)
1202             except pythoncom.com_error:
1203                 pass  # Ignore errors telling engine we stopped.
1204             # reset ourselves to 'connected' so further events continue to fire.
1205             self.SetScriptState(axscript.SCRIPTSTATE_CONNECTED)
1206             raise result_exception
1207         # I think that in some cases this should just return - but the code
1208         # that could return None above is disabled, so it never happens.
1209         RaiseAssert(
1210             winerror.E_UNEXPECTED, "Don't have an exception to raise to the caller!"
1211         )
