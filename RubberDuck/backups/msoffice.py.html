16   class OleClientItem(object.CmdTarget):
17       def __init__(self, doc):
18           object.CmdTarget.__init__(self, win32uiole.CreateOleClientItem(doc))
19   
20       def OnGetItemPosition(self):
21           # For now return a hard-coded rect.
22           return (10, 10, 210, 210)
23   
24       def OnActivate(self):
25           # Allow only one inplace activate item per frame
26           view = self.GetActiveView()
27           item = self.GetDocument().GetInPlaceActiveItem(view)
28           if item is not None and item._obj_ != self._obj_:
29               item.Close()
30           self._obj_.OnActivate()
31   
32       def OnChange(self, oleNotification, dwParam):
33           self._obj_.OnChange(oleNotification, dwParam)
34           self.GetDocument().UpdateAllViews(None)
35   
36       def OnChangeItemPosition(self, rect):
37           # During in-place activation CEmbed_ExcelCntrItem::OnChangeItemPosition
38           #  is called by the server to change the position of the in-place
39           #  window.  Usually, this is a result of the data in the server
40           #  document changing such that the extent has changed or as a result
41           #  of in-place resizing.
42           #
43           # The default here is to call the base class, which will call
44           #  COleClientItem::SetItemRects to move the item
45           #  to the new position.
46           if not self._obj_.OnChangeItemPosition(self, rect):
47               return 0
48   
49           # TODO: update any cache you may have of the item's rectangle/extent
50           return 1

59   class ExcelView(docview.ScrollView):
60       def OnInitialUpdate(self):
61           self.HookMessage(self.OnSetFocus, win32con.WM_SETFOCUS)
62           self.HookMessage(self.OnSize, win32con.WM_SIZE)
63   
64           self.SetScrollSizes(win32con.MM_TEXT, (100, 100))
65           rc = self._obj_.OnInitialUpdate()
66           self.EmbedExcel()
67           return rc
68   
69       def EmbedExcel(self):
70           doc = self.GetDocument()
71           self.clientItem = OleClientItem(doc)
72           self.clientItem.CreateNewItem("Excel.Sheet")
73           self.clientItem.DoVerb(-1, self)
74           doc.UpdateAllViews(None)
75   
76       def OnDraw(self, dc):
77           doc = self.GetDocument()
78           pos = doc.GetStartPosition()
79           clientItem, pos = doc.GetNextItem(pos)
80           clientItem.Draw(dc, (10, 10, 210, 210))
81   
82       # Special handling of OnSetFocus and OnSize are required for a container
83       #  when an object is being edited in-place.
84       def OnSetFocus(self, msg):
85           item = self.GetDocument().GetInPlaceActiveItem(self)
86           if (
87               item is not None
88               and item.GetItemState() == win32uiole.COleClientItem_activeUIState
89           ):
90               wnd = item.GetInPlaceWindow()
91               if wnd is not None:
92                   wnd.SetFocus()
93               return 0  # Dont get the base version called.
94           return 1  # Call the base version.
95   
96       def OnSize(self, params):
97           item = self.GetDocument().GetInPlaceActiveItem(self)
98           if item is not None:
99               item.SetItemRects()
100          return 1  # do call the base!
