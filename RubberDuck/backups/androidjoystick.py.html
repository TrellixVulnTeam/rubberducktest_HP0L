2    ''' 
3    Android Joystick Input Provider 
4    =============================== 
5     
6    This module is based on the PyGame JoyStick Input Provider. For more 
7    information, please refer to 
8    `<http://www.pygame.org/docs/ref/joystick.html>`_ 
9     
10    
11   '''
12   __all__ = ('AndroidMotionEventProvider', )
13   
14   import os
15   
16   try:
17       import android  # NOQA
18   except ImportError:
19       if 'KIVY_DOC' not in os.environ:
20           raise Exception('android lib not found.')
21   
22   from kivy.logger import Logger
23   from kivy.input.provider import MotionEventProvider
24   from kivy.input.factory import MotionEventFactory
25   from kivy.input.shape import ShapeRect
26   from kivy.input.motionevent import MotionEvent
27   if 'KIVY_DOC' not in os.environ:
28       import pygame.joystick
29   
30   
31   class AndroidMotionEvent(MotionEvent):
32   
33       def depack(self, args):
34           self.is_touch = True
35           self.profile = ['pos', 'pressure', 'shape']
36           self.sx, self.sy, self.pressure, radius = args
37           self.shape = ShapeRect()
38           self.shape.width = radius
39           self.shape.height = radius
40           super(AndroidMotionEvent, self).depack(args)
41   
42   
43   class AndroidMotionEventProvider(MotionEventProvider):
44   
45       def __init__(self, device, args):
46           super(AndroidMotionEventProvider, self).__init__(device, args)
47           self.joysticks = []
48           self.touches = {}
49           self.uid = 0
50           self.window = None
51   
52       def create_joystick(self, index):
53           Logger.info('Android: create joystick <%d>' % index)
54           js = pygame.joystick.Joystick(index)
55           js.init()
56           if js.get_numbuttons() == 0:
57               Logger.info('Android: discard joystick <%d> cause no button' %
58                           index)
59               return
60           self.joysticks.append(js)
61   
62       def start(self):
63           pygame.joystick.init()
64           Logger.info('Android: found %d joystick' % pygame.joystick.get_count())
65           for i in range(pygame.joystick.get_count()):
66               self.create_joystick(i)
67   
68       def stop(self):
69           self.joysticks = []
70   
71       def update(self, dispatch_fn):
72           if not self.window:
73               from kivy.core.window import Window
74               self.window = Window
75           w, h = self.window.system_size
76           touches = self.touches
77           for joy in self.joysticks:
78               jid = joy.get_id()
79               pressed = joy.get_button(0)
80               if pressed or jid in touches:
81                   x = joy.get_axis(0) * 32768. / w
82                   y = 1. - (joy.get_axis(1) * 32768. / h)
83   
84                   # python for android do * 1000.
85                   pressure = joy.get_axis(2) / 1000.
86                   radius = joy.get_axis(3) / 1000.
87   
88                   # new touch ?
89                   if pressed and jid not in touches:
90                       self.uid += 1
91                       touch = AndroidMotionEvent(self.device, self.uid,
92                                               [x, y, pressure, radius])
93                       touches[jid] = touch
94                       dispatch_fn('begin', touch)
95                   # update touch
96                   elif pressed:
97                       touch = touches[jid]
98                       # avoid same touch position
99                       if (touch.sx == x and touch.sy == y and
100                              touch.pressure == pressure):
101                          continue
102                      touch.move([x, y, pressure, radius])
103                      dispatch_fn('update', touch)
104                  # disappear
105                  elif not pressed and jid in touches:
106                      touch = touches[jid]
107                      touch.move([x, y, pressure, radius])
108                      touch.update_time_end()
109                      dispatch_fn('end', touch)
110                      touches.pop(jid)
111  
112  
113  MotionEventFactory.register('android', AndroidMotionEventProvider)
