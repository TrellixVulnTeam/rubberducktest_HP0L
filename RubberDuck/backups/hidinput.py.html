2    ''' 
3    Native support for HID input from the linux kernel 
4    ================================================== 
5     
6    Support starts from 2.6.32-ubuntu, or 2.6.34. 
7     
8    To configure HIDInput, add this to your configuration:: 
9     
10       [input] 
11       # devicename = hidinput,/dev/input/eventXX 
12       # example with Stantum MTP4.3" screen 
13       stantum = hidinput,/dev/input/event2 
14    
15   .. note:: 
16       You must have read access to the input event. 
17    
18   You can use a custom range for the X, Y and pressure values. 
19   For some drivers, the range reported is invalid. 
20   To fix that, you can add these options to the argument line: 
21    
22   * invert_x : 1 to invert X axis 
23   * invert_y : 1 to invert Y axis 
24   * min_position_x : X relative minimum 
25   * max_position_x : X relative maximum 
26   * min_position_y : Y relative minimum 
27   * max_position_y : Y relative maximum 
28   * min_abs_x : X absolute minimum 
29   * min_abs_y : Y absolute minimum 
30   * max_abs_x : X absolute maximum 
31   * max_abs_y : Y absolute maximum 
32   * min_pressure : pressure minimum 
33   * max_pressure : pressure maximum 
34   * rotation : rotate the input coordinate (0, 90, 180, 270) 
35    
36   For example, on the Asus T101M, the touchscreen reports a range from 0-4095 for 
37   the X and Y values, but the real values are in a range from 0-32768. To correct 
38   this, you can add the following to the configuration:: 
39    
40       [input] 
41       t101m = hidinput,/dev/input/event7,max_position_x=32768,\ 
42   max_position_y=32768 
43    
44   .. versionadded:: 1.9.1 
45    
46       `rotation` configuration token added. 
47    
48   '''

62       def depack(self, args):
63           self.is_touch = True
64           self.sx = args['x']
65           self.sy = args['y']
66           self.profile = ['pos']
67           if 'size_w' in args and 'size_h' in args:
68               self.shape = ShapeRect()
69               self.shape.width = args['size_w']
70               self.shape.height = args['size_h']
71               self.profile.append('shape')
72           if 'pressure' in args:
73               self.pressure = args['pressure']
74               self.profile.append('pressure')
75           if 'button' in args:
76               self.button = args['button']
77               self.profile.append('button')
78   
79           super(HIDMotionEvent, self).depack(args)
80   
81       def __str__(self):
82           return '<HIDMotionEvent id=%d pos=(%f, %f) device=%s>' \
83               % (self.id, self.sx, self.sy, self.device)

320      class HIDInputMotionEventProvider(MotionEventProvider):
321  
322          options = ('min_position_x', 'max_position_x',
323                     'min_position_y', 'max_position_y',
324                     'min_pressure', 'max_pressure',
325                     'min_abs_x', 'max_abs_x',
326                     'min_abs_y', 'max_abs_y',
327                     'invert_x', 'invert_y', 'rotation')
328  
329          def __init__(self, device, args):
330              super(HIDInputMotionEventProvider, self).__init__(device, args)
331              global Window, Keyboard
332  
333              if Window is None:
334                  from kivy.core.window import Window
335              if Keyboard is None:
336                  from kivy.core.window import Keyboard
337  
338              self.input_fn = None
339              self.default_ranges = dict()
340  
341              # split arguments
342              args = args.split(',')
343              if not args:
344                  Logger.error('HIDInput: Filename missing in configuration')
345                  Logger.error('HIDInput: Use /dev/input/event0 for example')
346                  return None
347  
348              # read filename
349              self.input_fn = args[0]
350              Logger.info('HIDInput: Read event from <%s>' % self.input_fn)
351  
352              # read parameters
353              for arg in args[1:]:
354                  if arg == '':
355                      continue
356                  arg = arg.split('=')
357  
358                  # ensure it's a key = value
359                  if len(arg) != 2:
360                      Logger.error('HIDInput: invalid parameter '
361                                   '%s, not in key=value format.' % arg)
362                      continue
363  
364                  # ensure the key exist
365                  key, value = arg
366                  if key not in HIDInputMotionEventProvider.options:
367                      Logger.error('HIDInput: unknown %s option' % key)
368                      continue
369  
370                  # ensure the value
371                  try:
372                      self.default_ranges[key] = int(value)
373                  except ValueError:
374                      err = 'HIDInput: invalid value "%s" for "%s"' % (
375                          key, value)
376                      Logger.error(err)
377                      continue
378  
379                  # all good!
380                  Logger.info('HIDInput: Set custom %s to %d' % (
381                      key, int(value)))
382  
383              if 'rotation' not in self.default_ranges:
384                  self.default_ranges['rotation'] = 0
385              elif self.default_ranges['rotation'] not in (0, 90, 180, 270):
386                  Logger.error('HIDInput: invalid rotation value ({})'.format(
387                      self.default_ranges['rotation']))
388                  self.default_ranges['rotation'] = 0
389  
390          def start(self):
391              if self.input_fn is None:
392                  return
393              self.uid = 0
394              self.queue = collections.deque()
395              self.dispatch_queue = []
396              self.thread = threading.Thread(
397                  name=self.__class__.__name__,
398                  target=self._thread_run,
399                  kwargs=dict(
400                      queue=self.queue,
401                      input_fn=self.input_fn,
402                      device=self.device,
403                      default_ranges=self.default_ranges))
404              self.thread.daemon = True
405              self.thread.start()
406  
407          def _thread_run(self, **kwargs):
408              input_fn = kwargs.get('input_fn')
409              queue = self.queue
410              dispatch_queue = self.dispatch_queue
411              device = kwargs.get('device')
412              drs = kwargs.get('default_ranges').get
413              touches = {}
414              touches_sent = []
415              point = {}
416              l_points = []
417  
418              # prepare some vars to get limit of some component
419              range_min_position_x = 0
420              range_max_position_x = 2048
421              range_min_position_y = 0
422              range_max_position_y = 2048
423              range_min_pressure = 0
424              range_max_pressure = 255
425              range_min_abs_x = 0
426              range_max_abs_x = 255
427              range_min_abs_y = 0
428              range_max_abs_y = 255
429              range_min_abs_pressure = 0
430              range_max_abs_pressure = 255
431              invert_x = int(bool(drs('invert_x', 0)))
432              invert_y = int(bool(drs('invert_y', 1)))
433              rotation = drs('rotation', 0)
434  
435              def assign_coord(point, value, invert, coords):
436                  cx, cy = coords
437                  if invert:
438                      value = 1. - value
439                  if rotation == 0:
440                      point[cx] = value
441                  elif rotation == 90:
442                      point[cy] = value
443                  elif rotation == 180:
444                      point[cx] = 1. - value
445                  elif rotation == 270:
446                      point[cy] = 1. - value
447  
448              def assign_rel_coord(point, value, invert, coords):
449                  cx, cy = coords
450                  if invert:
451                      value = -1 * value
452                  if rotation == 0:
453                      point[cx] += value
454                  elif rotation == 90:
455                      point[cy] += value
456                  elif rotation == 180:
457                      point[cx] += -value
458                  elif rotation == 270:
459                      point[cy] += -value
460  
461                  # limit it to the screen area 0-1
462                  point['x'] = min(1., max(0., point['x']))
463                  point['y'] = min(1., max(0., point['y']))
464  
465              def process_as_multitouch(tv_sec, tv_usec, ev_type,
466                                        ev_code, ev_value):
467                  # sync event
468                  if ev_type == EV_SYN:
469                      if ev_code == SYN_MT_REPORT:
470                          if 'id' not in point:
471                              return
472                          l_points.append(point.copy())
473                      elif ev_code == SYN_REPORT:
474                          process(l_points)
475                          del l_points[:]
476  
477                  elif ev_type == EV_MSC and ev_code in (MSC_RAW, MSC_SCAN):
478                      pass
479  
480                  else:
481                      # compute multitouch track
482                      if ev_code == ABS_MT_TRACKING_ID:
483                          point.clear()
484                          point['id'] = ev_value
485                      elif ev_code == ABS_MT_POSITION_X:
486                          val = normalize(ev_value,
487                                          range_min_position_x,
488                                          range_max_position_x)
489                          assign_coord(point, val, invert_x, 'xy')
490                      elif ev_code == ABS_MT_POSITION_Y:
491                          val = 1. - normalize(ev_value,
492                                               range_min_position_y,
493                                               range_max_position_y)
494                          assign_coord(point, val, invert_y, 'yx')
495                      elif ev_code == ABS_MT_ORIENTATION:
496                          point['orientation'] = ev_value
497                      elif ev_code == ABS_MT_BLOB_ID:
498                          point['blobid'] = ev_value
499                      elif ev_code == ABS_MT_PRESSURE:
500                          point['pressure'] = normalize(ev_value,
501                                                        range_min_pressure,
502                                                        range_max_pressure)
503                      elif ev_code == ABS_MT_TOUCH_MAJOR:
504                          point['size_w'] = ev_value
505                      elif ev_code == ABS_MT_TOUCH_MINOR:
506                          point['size_h'] = ev_value
507  
508              def process_as_mouse_or_keyboard(
509                      tv_sec, tv_usec, ev_type, ev_code, ev_value):
510                  if ev_type == EV_SYN:
511                      if ev_code == SYN_REPORT:
512                          process([point])
513                          if ('button' in point and
514                                  point['button'].startswith('scroll')):
515                              # for scrolls we need to remove it as there is
516                              # no up key
517                              del point['button']
518                              point['id'] += 1
519                              point['_avoid'] = True
520                              process([point])
521  
522                  elif ev_type == EV_REL:
523                      if ev_code == 0:
524                          assign_rel_coord(point,
525                                           min(1., max(-1., ev_value / 1000.)),
526                                           invert_x, 'xy')
527                      elif ev_code == 1:
528                          assign_rel_coord(point,
529                                           min(1., max(-1., ev_value / 1000.)),
530                                           invert_y, 'yx')
531                      elif ev_code == 8:  # Wheel
532                          # translates the wheel move to a button
533                          b = "scrollup" if ev_value < 0 else "scrolldown"
534                          if 'button' not in point:
535                              point['button'] = b
536                              point['id'] += 1
537                              if '_avoid' in point:
538                                  del point['_avoid']
539  
540                  elif ev_type != EV_KEY:
541                      if ev_code == ABS_X:
542                          val = normalize(ev_value,
543                                          range_min_abs_x,
544                                          range_max_abs_x)
545                          assign_coord(point, val, invert_x, 'xy')
546                      elif ev_code == ABS_Y:
547                          val = 1. - normalize(ev_value,
548                                               range_min_abs_y,
549                                               range_max_abs_y)
550                          assign_coord(point, val, invert_y, 'yx')
551                      elif ev_code == ABS_PRESSURE:
552                          point['pressure'] = normalize(ev_value,
553                                                        range_min_abs_pressure,
554                                                        range_max_abs_pressure)
555                  else:
556                      buttons = {
557                          272: 'left',
558                          273: 'right',
559                          274: 'middle',
560                          275: 'side',
561                          276: 'extra',
562                          277: 'forward',
563                          278: 'back',
564                          279: 'task',
565                          330: 'touch',
566                          320: 'pen'}
567  
568                      if ev_code in buttons.keys():
569                          if ev_value:
570                              if 'button' not in point:
571                                  point['button'] = buttons[ev_code]
572                                  point['id'] += 1
573                                  if '_avoid' in point:
574                                      del point['_avoid']
575                          elif 'button' in point:
576                              if point['button'] == buttons[ev_code]:
577                                  del point['button']
578                                  point['id'] += 1
579                                  point['_avoid'] = True
580                      else:
581                          if not 0 <= ev_value <= 1:
582                              return
583  
584                          if ev_code not in keyboard_keys:
585                              Logger.warn('HIDInput: unhandled HID code: {}'.
586                                          format(ev_code))
587                              return
588  
589                          z = keyboard_keys[ev_code][-1 if 'shift' in
590                                                     Window._modifiers else 0]
591                          if z.lower() not in Keyboard.keycodes:
592                              # or if it is not in this LUT
593                              Logger.warn('HIDInput: unhandled character: {}'.
594                                          format(z))
595                              return
596  
597                          keycode = Keyboard.keycodes[z.lower()]
598  
599                          if ev_value == 1:
600                              if z == 'shift' or z == 'alt':
601                                  Window._modifiers.append(z)
602                              elif z.endswith('ctrl'):
603                                  Window._modifiers.append('ctrl')
604  
605                              dispatch_queue.append(('key_down', (
606                                  keycode, ev_code,
607                                  keys_str.get(z, z), Window._modifiers)))
608                          elif ev_value == 0:
609                              dispatch_queue.append(('key_up', (
610                                  keycode, ev_code,
611                                  keys_str.get(z, z), Window._modifiers)))
612                              if ((z == 'shift' or z == 'alt') and
613                                      (z in Window._modifiers)):
614                                  Window._modifiers.remove(z)
615                              elif (z.endswith('ctrl') and
616                                    'ctrl' in Window._modifiers):
617                                  Window._modifiers.remove('ctrl')
618  
619              def process(points):
620                  if not is_multitouch:
621                      dispatch_queue.append(('mouse_pos', (
622                          points[0]['x'] * Window.width,
623                          points[0]['y'] * Window.height)))
624  
625                  actives = [args['id']
626                             for args in points
627                             if 'id' in args and '_avoid' not in args]
628                  for args in points:
629                      tid = args['id']
630                      try:
631                          touch = touches[tid]
632                          if touch.sx == args['x'] and touch.sy == args['y']:
633                              continue
634                          touch.move(args)
635                          if tid not in touches_sent:
636                              queue.append(('begin', touch))
637                              touches_sent.append(tid)
638                          queue.append(('update', touch))
639                      except KeyError:
640                          if '_avoid' not in args:
641                              touch = HIDMotionEvent(device, tid, args)
642                              touches[touch.id] = touch
643                              if tid not in touches_sent:
644                                  queue.append(('begin', touch))
645                                  touches_sent.append(tid)
646  
647                  for tid in list(touches.keys())[:]:
648                      if tid not in actives:
649                          touch = touches[tid]
650                          if tid in touches_sent:
651                              touch.update_time_end()
652                              queue.append(('end', touch))
653                              touches_sent.remove(tid)
654                          del touches[tid]
655  
656              def normalize(value, vmin, vmax):
657                  return (value - vmin) / float(vmax - vmin)
658  
659              # open the input
660              fd = open(input_fn, 'rb')
661  
662              # get the controler name (EVIOCGNAME)
663              device_name = fcntl.ioctl(fd, EVIOCGNAME + (256 << 16),
664                                        " " * 256).decode().strip()
665              Logger.info('HIDMotionEvent: using <%s>' % device_name)
666  
667              # get abs infos
668              bit = fcntl.ioctl(fd, EVIOCGBIT + (EV_MAX << 16), ' ' * sz_l)
669              bit, = struct.unpack('Q', bit)
670              is_multitouch = False
671              for x in range(EV_MAX):
672                  # preserve this, we may want other things than EV_ABS
673                  if x != EV_ABS:
674                      continue
675                  # EV_ABS available for this device ?
676                  if (bit & (1 << x)) == 0:
677                      continue
678                  # ask abs info keys to the devices
679                  sbit = fcntl.ioctl(fd, EVIOCGBIT + x + (KEY_MAX << 16),
680                                     ' ' * sz_l)
681                  sbit, = struct.unpack('Q', sbit)
682                  for y in range(KEY_MAX):
683                      if (sbit & (1 << y)) == 0:
684                          continue
685                      absinfo = fcntl.ioctl(fd, EVIOCGABS + y +
686                                            (struct_input_absinfo_sz << 16),
687                                            ' ' * struct_input_absinfo_sz)
688                      abs_value, abs_min, abs_max, abs_fuzz, \
689                          abs_flat, abs_res = struct.unpack('iiiiii', absinfo)
690                      if y == ABS_MT_POSITION_X:
691                          is_multitouch = True
692                          range_min_position_x = drs('min_position_x', abs_min)
693                          range_max_position_x = drs('max_position_x', abs_max)
694                          Logger.info('HIDMotionEvent: ' +
695                                      '<%s> range position X is %d - %d' % (
696                                          device_name, abs_min, abs_max))
697                      elif y == ABS_MT_POSITION_Y:
698                          is_multitouch = True
699                          range_min_position_y = drs('min_position_y', abs_min)
700                          range_max_position_y = drs('max_position_y', abs_max)
701                          Logger.info('HIDMotionEvent: ' +
702                                      '<%s> range position Y is %d - %d' % (
703                                          device_name, abs_min, abs_max))
704                      elif y == ABS_MT_PRESSURE:
705                          range_min_pressure = drs('min_pressure', abs_min)
706                          range_max_pressure = drs('max_pressure', abs_max)
707                          Logger.info('HIDMotionEvent: ' +
708                                      '<%s> range pressure is %d - %d' % (
709                                          device_name, abs_min, abs_max))
710                      elif y == ABS_X:
711                          range_min_abs_x = drs('min_abs_x', abs_min)
712                          range_max_abs_x = drs('max_abs_x', abs_max)
713                          Logger.info('HIDMotionEvent: ' +
714                                      '<%s> range ABS X position is %d - %d' % (
715                                          device_name, abs_min, abs_max))
716                      elif y == ABS_Y:
717                          range_min_abs_y = drs('min_abs_y', abs_min)
718                          range_max_abs_y = drs('max_abs_y', abs_max)
719                          Logger.info('HIDMotionEvent: ' +
720                                      '<%s> range ABS Y position is %d - %d' % (
721                                          device_name, abs_min, abs_max))
722                      elif y == ABS_PRESSURE:
723                          range_min_abs_pressure = drs(
724                              'min_abs_pressure', abs_min)
725                          range_max_abs_pressure = drs(
726                              'max_abs_pressure', abs_max)
727                          Logger.info('HIDMotionEvent: ' +
728                                      '<%s> range ABS pressure is %d - %d' % (
729                                          device_name, abs_min, abs_max))
730  
731              # init the point
732              if not is_multitouch:
733                  point = {'x': .5, 'y': .5, 'id': 0, '_avoid': True}
734  
735              # read until the end
736              while fd:
737  
738                  data = fd.read(struct_input_event_sz)
739                  if len(data) < struct_input_event_sz:
740                      break
741  
742                  # extract each event
743                  for i in range(int(len(data) / struct_input_event_sz)):
744                      ev = data[i * struct_input_event_sz:]
745  
746                      # extract timeval + event infos
747                      infos = struct.unpack('LLHHi', ev[:struct_input_event_sz])
748  
749                      if is_multitouch:
750                          process_as_multitouch(*infos)
751                      else:
752                          process_as_mouse_or_keyboard(*infos)
753  
754          def update(self, dispatch_fn):
755              # dispatch all events from threads
756              dispatch_queue = self.dispatch_queue
757              n = len(dispatch_queue)
758              for name, args in dispatch_queue[:n]:
759                  if name == 'mouse_pos':
760                      Window.mouse_pos = args
761                  elif name == 'key_down':
762                      if not Window.dispatch('on_key_down', *args):
763                          Window.dispatch('on_keyboard', *args)
764                  elif name == 'key_up':
765                      Window.dispatch('on_key_up', *args)
766              del dispatch_queue[:n]
767  
768              try:
769                  while True:
770                      event_type, touch = self.queue.popleft()
771                      dispatch_fn(event_type, touch)
772              except:
773                  pass
