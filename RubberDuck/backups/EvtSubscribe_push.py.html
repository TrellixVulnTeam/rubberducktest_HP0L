1    ## Demonstrates a "push" subscription with a callback function
2    import win32evtlog
3    
4    query_text = '*[System[Provider[@Name="Microsoft-Windows-Winlogon"]]]'
5    
6    
7    def c(reason, context, evt):
8        if reason == win32evtlog.EvtSubscribeActionError:
9            print("EvtSubscribeActionError")
10       elif reason == win32evtlog.EvtSubscribeActionDeliver:
11           print("EvtSubscribeActionDeliver")
12       else:
13           print("??? Unknown action ???", reason)
14       context.append(win32evtlog.EvtRender(evt, win32evtlog.EvtRenderEventXml))
15       return 0
16   
17   
18   evttext = []
19   s = win32evtlog.EvtSubscribe(
20       "System",
21       win32evtlog.EvtSubscribeStartAtOldestRecord,
22       Query="*",
23       Callback=c,
24       Context=evttext,
25   )
