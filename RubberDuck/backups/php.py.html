89   class PsyshConsoleLexer(Lexer):
90       """ 
91       For `PsySH`_ console output, such as: 
92    
93       .. sourcecode:: psysh 
94    
95           >>> $greeting = function($name): string { 
96           ...     return "Hello, {$name}"; 
97           ... }; 
98           => Closure($name): string {#2371 …3} 
99           >>> $greeting('World') 
100          => "Hello, World" 
101   
102      .. _PsySH: https://psysh.org/ 
103      .. versionadded:: 2.7 
104      """
105      name = 'PsySH console session for PHP'
106      aliases = ['psysh']
107  
108      def __init__(self, **options):
109          options['startinline'] = True
110          Lexer.__init__(self, **options)
111  
112      def get_tokens_unprocessed(self, text):
113          phplexer = PhpLexer(**self.options)
114          curcode = ''
115          insertions = []
116          for match in line_re.finditer(text):
117              line = match.group()
118              if line.startswith('>>> ') or line.startswith('... '):
119                  insertions.append((len(curcode),
120                                     [(0, Generic.Prompt, line[:4])]))
121                  curcode += line[4:]
122              elif line.rstrip() == '...':
123                  insertions.append((len(curcode),
124                                     [(0, Generic.Prompt, '...')]))
125                  curcode += line[3:]
126              else:
127                  if curcode:
128                      yield from do_insertions(
129                          insertions, phplexer.get_tokens_unprocessed(curcode))
130                      curcode = ''
131                      insertions = []
132                  yield match.start(), Generic.Output, line
133          if curcode:
134              yield from do_insertions(insertions,
135                                       phplexer.get_tokens_unprocessed(curcode))

138  class PhpLexer(RegexLexer):
139      """ 
140      For `PHP <http://www.php.net/>`_ source code. 
141      For PHP embedded in HTML, use the `HtmlPhpLexer`. 
142   
143      Additional options accepted: 
144   
145      `startinline` 
146          If given and ``True`` the lexer starts highlighting with 
147          php code (i.e.: no starting ``<?php`` required).  The default 
148          is ``False``. 
149      `funcnamehighlighting` 
150          If given and ``True``, highlight builtin function names 
151          (default: ``True``). 
152      `disabledmodules` 
153          If given, must be a list of module names whose function names 
154          should not be highlighted. By default all modules are highlighted 
155          except the special ``'unknown'`` module that includes functions 
156          that are known to php but are undocumented. 
157   
158          To get a list of allowed modules have a look into the 
159          `_php_builtins` module: 
160   
161          .. sourcecode:: pycon 
162   
163              >>> from pygments.lexers._php_builtins import MODULES 
164              >>> MODULES.keys() 
165              ['PHP Options/Info', 'Zip', 'dba', ...] 
166   
167          In fact the names of those modules match the module names from 
168          the php documentation. 
169      """
170  
171      name = 'PHP'
172      aliases = ['php', 'php3', 'php4', 'php5']
173      filenames = ['*.php', '*.php[345]', '*.inc']
174      mimetypes = ['text/x-php']
175  
176      # Note that a backslash is included in the following two patterns
177      # PHP uses a backslash as a namespace separator
178      _ident_char = r'[\\\w]|[^\x00-\x7f]'
179      _ident_begin = r'(?:[\\_a-z]|[^\x00-\x7f])'
180      _ident_end = r'(?:' + _ident_char + ')*'
181      _ident_inner = _ident_begin + _ident_end
182  
183      flags = re.IGNORECASE | re.DOTALL | re.MULTILINE
184      tokens = {
185          'root': [
186              (r'<\?(php)?', Comment.Preproc, 'php'),
187              (r'[^<]+', Other),
188              (r'<', Other)
189          ],
190          'php': [
191              (r'\?>', Comment.Preproc, '#pop'),
192              (r'(<<<)([\'"]?)(' + _ident_inner + r')(\2\n.*?\n\s*)(\3)(;?)(\n)',
193               bygroups(String, String, String.Delimiter, String, String.Delimiter,
194                        Punctuation, Text)),
195              (r'\s+', Text),
196              (r'#.*?\n', Comment.Single),
197              (r'//.*?\n', Comment.Single),
198              # put the empty comment here, it is otherwise seen as
199              # the start of a docstring
200              (r'/\*\*/', Comment.Multiline),
201              (r'/\*\*.*?\*/', String.Doc),
202              (r'/\*.*?\*/', Comment.Multiline),
203              (r'(->|::)(\s*)(' + _ident_inner + ')',
204               bygroups(Operator, Text, Name.Attribute)),
205              (r'[~!%^&*+=|:.<>/@-]+', Operator),
206              (r'\?', Operator),  # don't add to the charclass above!
207              (r'[\[\]{}();,]+', Punctuation),
208              (r'(class)(\s+)', bygroups(Keyword, Text), 'classname'),
209              (r'(function)(\s*)(?=\()', bygroups(Keyword, Text)),
210              (r'(function)(\s+)(&?)(\s*)',
211               bygroups(Keyword, Text, Operator, Text), 'functionname'),
212              (r'(const)(\s+)(' + _ident_inner + ')',
213               bygroups(Keyword, Text, Name.Constant)),
214              (r'(and|E_PARSE|old_function|E_ERROR|or|as|E_WARNING|parent|'
215               r'eval|PHP_OS|break|exit|case|extends|PHP_VERSION|cfunction|'
216               r'FALSE|print|for|require|continue|foreach|require_once|'
217               r'declare|return|default|static|do|switch|die|stdClass|'
218               r'echo|else|TRUE|elseif|var|empty|if|xor|enddeclare|include|'
219               r'virtual|endfor|include_once|while|endforeach|global|'
220               r'endif|list|endswitch|new|endwhile|not|'
221               r'array|E_ALL|NULL|final|php_user_filter|interface|'
222               r'implements|public|private|protected|abstract|clone|try|'
223               r'catch|throw|this|use|namespace|trait|yield|'
224               r'finally)\b', Keyword),
225              (r'(true|false|null)\b', Keyword.Constant),
226              include('magicconstants'),
227              (r'\$\{\$+' + _ident_inner + r'\}', Name.Variable),
228              (r'\$+' + _ident_inner, Name.Variable),
229              (_ident_inner, Name.Other),
230              (r'(\d+\.\d*|\d*\.\d+)(e[+-]?[0-9]+)?', Number.Float),
231              (r'\d+e[+-]?[0-9]+', Number.Float),
232              (r'0[0-7]+', Number.Oct),
233              (r'0x[a-f0-9]+', Number.Hex),
234              (r'\d+', Number.Integer),
235              (r'0b[01]+', Number.Bin),
236              (r"'([^'\\]*(?:\\.[^'\\]*)*)'", String.Single),
237              (r'`([^`\\]*(?:\\.[^`\\]*)*)`', String.Backtick),
238              (r'"', String.Double, 'string'),
239          ],
240          'magicfuncs': [
241              # source: http://php.net/manual/en/language.oop5.magic.php
242              (words((
243                  '__construct', '__destruct', '__call', '__callStatic', '__get', '__set',
244                  '__isset', '__unset', '__sleep', '__wakeup', '__toString', '__invoke',
245                  '__set_state', '__clone', '__debugInfo',), suffix=r'\b'),
246               Name.Function.Magic),
247          ],
248          'magicconstants': [
249              # source: http://php.net/manual/en/language.constants.predefined.php
250              (words((
251                  '__LINE__', '__FILE__', '__DIR__', '__FUNCTION__', '__CLASS__',
252                  '__TRAIT__', '__METHOD__', '__NAMESPACE__',),
253                  suffix=r'\b'),
254               Name.Constant),
255          ],
256          'classname': [
257              (_ident_inner, Name.Class, '#pop')
258          ],
259          'functionname': [
260              include('magicfuncs'),
261              (_ident_inner, Name.Function, '#pop'),
262              default('#pop')
263          ],
264          'string': [
265              (r'"', String.Double, '#pop'),
266              (r'[^{$"\\]+', String.Double),
267              (r'\\([nrt"$\\]|[0-7]{1,3}|x[0-9a-f]{1,2})', String.Escape),
268              (r'\$' + _ident_inner + r'(\[\S+?\]|->' + _ident_inner + ')?',
269               String.Interpol),
270              (r'(\{\$\{)(.*?)(\}\})',
271               bygroups(String.Interpol, using(this, _startinline=True),
272                        String.Interpol)),
273              (r'(\{)(\$.*?)(\})',
274               bygroups(String.Interpol, using(this, _startinline=True),
275                        String.Interpol)),
276              (r'(\$\{)(\S+)(\})',
277               bygroups(String.Interpol, Name.Variable, String.Interpol)),
278              (r'[${\\]', String.Double)
279          ],
280      }
281  
282      def __init__(self, **options):
283          self.funcnamehighlighting = get_bool_opt(
284              options, 'funcnamehighlighting', True)
285          self.disabledmodules = get_list_opt(
286              options, 'disabledmodules', ['unknown'])
287          self.startinline = get_bool_opt(options, 'startinline', False)
288  
289          # private option argument for the lexer itself
290          if '_startinline' in options:
291              self.startinline = options.pop('_startinline')
292  
293          # collect activated functions in a set
294          self._functions = set()
295          if self.funcnamehighlighting:
296              from pygments.lexers._php_builtins import MODULES
297              for key, value in MODULES.items():
298                  if key not in self.disabledmodules:
299                      self._functions.update(value)
300          RegexLexer.__init__(self, **options)
301  
302      def get_tokens_unprocessed(self, text):
303          stack = ['root']
304          if self.startinline:
305              stack.append('php')
306          for index, token, value in \
307                  RegexLexer.get_tokens_unprocessed(self, text, stack):
308              if token is Name.Other:
309                  if value in self._functions:
310                      yield index, Name.Builtin, value
311                      continue
312              yield index, token, value
313  
314      def analyse_text(text):
315          if shebang_matches(text, r'php'):
316              return True
317          rv = 0.0
318          if re.search(r'<\?(?!xml)', text):
319              rv += 0.3
320          return rv
