26   def OnPaint_1(hwnd, msg, wp, lp):
27       dc, ps = win32gui.BeginPaint(hwnd)
28       win32gui.SetGraphicsMode(dc, win32con.GM_ADVANCED)
29       br = win32gui.CreateSolidBrush(win32api.RGB(255, 0, 0))
30       win32gui.SelectObject(dc, br)
31       angle = win32gui.GetWindowLong(hwnd, win32con.GWL_USERDATA)
32       win32gui.SetWindowLong(hwnd, win32con.GWL_USERDATA, angle + 2)
33       r_angle = angle * (math.pi / 180)
34       win32gui.SetWorldTransform(
35           dc,
36           {
37               "M11": math.cos(r_angle),
38               "M12": math.sin(r_angle),
39               "M21": math.sin(r_angle) * -1,
40               "M22": math.cos(r_angle),
41               "Dx": 250,
42               "Dy": 250,
43           },
44       )
45       win32gui.MoveToEx(dc, 250, 250)
46       win32gui.BeginPath(dc)
47       win32gui.Pie(dc, 10, 70, 200, 200, 350, 350, 75, 10)
48       win32gui.Chord(dc, 200, 200, 850, 0, 350, 350, 75, 10)
49       win32gui.LineTo(dc, 300, 300)
50       win32gui.LineTo(dc, 100, 20)
51       win32gui.LineTo(dc, 20, 100)
52       win32gui.LineTo(dc, 400, 0)
53       win32gui.LineTo(dc, 0, 400)
54       win32gui.EndPath(dc)
55       win32gui.StrokeAndFillPath(dc)
56       win32gui.EndPaint(hwnd, ps)
57       return 0

104  def TestSetWorldTransform():
105      wc = win32gui.WNDCLASS()
106      wc.lpszClassName = "test_win32gui_1"
107      wc.style = win32con.CS_GLOBALCLASS | win32con.CS_VREDRAW | win32con.CS_HREDRAW
108      wc.hbrBackground = win32con.COLOR_WINDOW + 1
109      wc.lpfnWndProc = wndproc_1
110      class_atom = win32gui.RegisterClass(wc)
111      hwnd = win32gui.CreateWindow(
112          wc.lpszClassName,
113          "Spin the Lobster!",
114          win32con.WS_CAPTION | win32con.WS_VISIBLE,
115          100,
116          100,
117          900,
118          900,
119          0,
120          0,
121          0,
122          None,
123      )
124      for x in range(500):
125          win32gui.InvalidateRect(hwnd, None, True)
126          win32gui.PumpWaitingMessages()
127          time.sleep(0.01)
128      win32gui.DestroyWindow(hwnd)
129      win32gui.UnregisterClass(wc.lpszClassName, None)

135      wc.style = win32con.CS_GLOBALCLASS | win32con.CS_VREDRAW | win32con.CS_HREDRAW

137      wc.lpfnWndProc = wndproc_2
138      class_atom = win32gui.RegisterClass(wc)
139      hwnd = win32gui.CreateWindowEx(
140          0,
141          class_atom,
142          "Kaleidoscope",
143          win32con.WS_CAPTION
144          | win32con.WS_VISIBLE
145          | win32con.WS_THICKFRAME
146          | win32con.WS_SYSMENU,
147          100,
148          100,
149          900,
150          900,
151          0,
152          0,
153          0,
154          None,
155      )
156      s = win32gui.GetWindowLong(hwnd, win32con.GWL_EXSTYLE)
157      win32gui.SetWindowLong(hwnd, win32con.GWL_EXSTYLE, s | win32con.WS_EX_LAYERED)
158      win32gui.SetLayeredWindowAttributes(hwnd, 0, 175, win32con.LWA_ALPHA)
159      for x in range(30):
160          win32gui.InvalidateRect(hwnd, None, True)
161          win32gui.PumpWaitingMessages()
162          time.sleep(0.3)
163      win32gui.DestroyWindow(hwnd)
164      win32gui.UnregisterClass(class_atom, None)
