71       def testValues(self):
72           key_name = r"PythonTestHarness\win32api"
73           ## tuples containing value name, value type, data
74           values = (
75               (None, win32con.REG_SZ, "This is default unnamed value"),
76               ("REG_SZ", win32con.REG_SZ, "REG_SZ text data"),
77               ("REG_EXPAND_SZ", win32con.REG_EXPAND_SZ, "%systemdir%"),
78               ## REG_MULTI_SZ value needs to be a list since strings are returned as a list
79               (
80                   "REG_MULTI_SZ",
81                   win32con.REG_MULTI_SZ,
82                   ["string 1", "string 2", "string 3", "string 4"],
83               ),
84               ("REG_MULTI_SZ_empty", win32con.REG_MULTI_SZ, []),
85               ("REG_DWORD", win32con.REG_DWORD, 666),
86               ("REG_QWORD_INT", win32con.REG_QWORD, 99),
87               ("REG_QWORD", win32con.REG_QWORD, 2 ** 33),
88               (
89                   "REG_BINARY",
90                   win32con.REG_BINARY,
91                   str2bytes("\x00\x01\x02\x03\x04\x05\x06\x07\x08\x01\x00"),
92               ),
93           )
94   
95           hkey = win32api.RegCreateKey(win32con.HKEY_CURRENT_USER, key_name)
96           for value_name, reg_type, data in values:
97               win32api.RegSetValueEx(hkey, value_name, None, reg_type, data)
98   
99           for value_name, orig_type, orig_data in values:
100              data, typ = win32api.RegQueryValueEx(hkey, value_name)
101              self.assertEqual(typ, orig_type)
102              self.assertEqual(data, orig_data)

104      def testNotifyChange(self):
105          def change():
106              hkey = win32api.RegCreateKey(win32con.HKEY_CURRENT_USER, self.key_name)
107              try:
108                  win32api.RegSetValue(hkey, None, win32con.REG_SZ, "foo")
109              finally:
110                  win32api.RegDeleteKey(win32con.HKEY_CURRENT_USER, self.key_name)
111  
112          evt = win32event.CreateEvent(None, 0, 0, None)
113          ## REG_NOTIFY_CHANGE_LAST_SET - values
114          ## REG_CHANGE_NOTIFY_NAME - keys
115          ## REG_NOTIFY_CHANGE_SECURITY - security descriptor
116          ## REG_NOTIFY_CHANGE_ATTRIBUTES
117          win32api.RegNotifyChangeKeyValue(
118              win32con.HKEY_CURRENT_USER,
119              1,
120              win32api.REG_NOTIFY_CHANGE_LAST_SET,
121              evt,
122              True,
123          )
124          ret_code = win32event.WaitForSingleObject(evt, 0)
125          # Should be no change.
126          self.failUnless(ret_code == win32con.WAIT_TIMEOUT)
127          change()
128          # Our event should now be in a signalled state.
129          ret_code = win32event.WaitForSingleObject(evt, 0)
130          self.failUnless(ret_code == win32con.WAIT_OBJECT_0)

182      def testLongLongPathNames(self):
183          # We need filename where the FQN is > 256 - simplest way is to create a
184          # 250 character directory in the cwd (except - cwd may be on a drive
185          # not supporting \\\\?\\ (eg, network share) - so use temp.
186          import win32file
187  
188          basename = "a" * 250
189          # but we need to ensure we use the 'long' version of the
190          # temp dir for later comparison.
191          long_temp_dir = win32api.GetLongPathNameW(tempfile.gettempdir())
192          fname = "\\\\?\\" + os.path.join(long_temp_dir, basename)
193          try:
194              win32file.CreateDirectoryW(fname, None)
195          except win32api.error as details:
196              if details.winerror != winerror.ERROR_ALREADY_EXISTS:
197                  raise
198          try:
199              # GetFileAttributes automatically calls GetFileAttributesW when
200              # passed unicode
201              try:
202                  attr = win32api.GetFileAttributes(fname)
203              except win32api.error as details:
204                  if details.winerror != winerror.ERROR_FILENAME_EXCED_RANGE:
205                      raise
206  
207              attr = win32api.GetFileAttributes(str(fname))
208              self.failUnless(attr & win32con.FILE_ATTRIBUTE_DIRECTORY, attr)
209  
210              long_name = win32api.GetLongPathNameW(fname)
211              self.failUnlessEqual(long_name.lower(), fname.lower())
212          finally:
213              win32file.RemoveDirectory(fname)
