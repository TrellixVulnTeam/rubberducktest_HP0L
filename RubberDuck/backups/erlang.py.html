154  class ErlangShellLexer(Lexer):
155      """ 
156      Shell sessions in erl (for Erlang code). 
157   
158      .. versionadded:: 1.1 
159      """
160      name = 'Erlang erl session'
161      aliases = ['erl']
162      filenames = ['*.erl-sh']
163      mimetypes = ['text/x-erl-shellsession']
164  
165      _prompt_re = re.compile(r'(?:\([\w@_.]+\))?\d+>(?=\s|\Z)')
166  
167      def get_tokens_unprocessed(self, text):
168          erlexer = ErlangLexer(**self.options)
169  
170          curcode = ''
171          insertions = []
172          for match in line_re.finditer(text):
173              line = match.group()
174              m = self._prompt_re.match(line)
175              if m is not None:
176                  end = m.end()
177                  insertions.append((len(curcode),
178                                     [(0, Generic.Prompt, line[:end])]))
179                  curcode += line[end:]
180              else:
181                  if curcode:
182                      yield from do_insertions(insertions,
183                                               erlexer.get_tokens_unprocessed(curcode))
184                      curcode = ''
185                      insertions = []
186                  if line.startswith('*'):
187                      yield match.start(), Generic.Traceback, line
188                  else:
189                      yield match.start(), Generic.Output, line
190          if curcode:
191              yield from do_insertions(insertions,
192                                       erlexer.get_tokens_unprocessed(curcode))

195  def gen_elixir_string_rules(name, symbol, token):
196      states = {}
197      states['string_' + name] = [
198          (r'[^#%s\\]+' % (symbol,), token),
199          include('escapes'),
200          (r'\\.', token),
201          (r'(%s)' % (symbol,), bygroups(token), "#pop"),
202          include('interpol')
203      ]
204      return states
205  
206  
207  def gen_elixir_sigstr_rules(term, term_class, token, interpol=True):
208      if interpol:
209          return [
210              (r'[^#%s\\]+' % (term_class,), token),
211              include('escapes'),
212              (r'\\.', token),
213              (r'%s[a-zA-Z]*' % (term,), token, '#pop'),
214              include('interpol')
215          ]
216      else:
217          return [
218              (r'[^%s\\]+' % (term_class,), token),
219              (r'\\.', token),
220              (r'%s[a-zA-Z]*' % (term,), token, '#pop'),
221          ]

224  class ElixirLexer(RegexLexer):
225      """ 
226      For the `Elixir language <http://elixir-lang.org>`_. 
227   
228      .. versionadded:: 1.5 
229      """
230  
231      name = 'Elixir'
232      aliases = ['elixir', 'ex', 'exs']
233      filenames = ['*.ex', '*.eex', '*.exs', '*.leex']
234      mimetypes = ['text/x-elixir']
235  
236      KEYWORD = ('fn', 'do', 'end', 'after', 'else', 'rescue', 'catch')
237      KEYWORD_OPERATOR = ('not', 'and', 'or', 'when', 'in')
238      BUILTIN = (
239          'case', 'cond', 'for', 'if', 'unless', 'try', 'receive', 'raise',
240          'quote', 'unquote', 'unquote_splicing', 'throw', 'super',
241      )
242      BUILTIN_DECLARATION = (
243          'def', 'defp', 'defmodule', 'defprotocol', 'defmacro', 'defmacrop',
244          'defdelegate', 'defexception', 'defstruct', 'defimpl', 'defcallback',
245      )
246  
247      BUILTIN_NAMESPACE = ('import', 'require', 'use', 'alias')
248      CONSTANT = ('nil', 'true', 'false')
249  
250      PSEUDO_VAR = ('_', '__MODULE__', '__DIR__', '__ENV__', '__CALLER__')
251  
252      OPERATORS3 = (
253          '<<<', '>>>', '|||', '&&&', '^^^', '~~~', '===', '!==',
254          '~>>', '<~>', '|~>', '<|>',
255      )
256      OPERATORS2 = (
257          '==', '!=', '<=', '>=', '&&', '||', '<>', '++', '--', '|>', '=~',
258          '->', '<-', '|', '.', '=', '~>', '<~',
259      )
260      OPERATORS1 = ('<', '>', '+', '-', '*', '/', '!', '^', '&')
261  
262      PUNCTUATION = (
263          '\\\\', '<<', '>>', '=>', '(', ')', ':', ';', ',', '[', ']',
264      )
265  
266      def get_tokens_unprocessed(self, text):
267          for index, token, value in RegexLexer.get_tokens_unprocessed(self, text):
268              if token is Name:
269                  if value in self.KEYWORD:
270                      yield index, Keyword, value
271                  elif value in self.KEYWORD_OPERATOR:
272                      yield index, Operator.Word, value
273                  elif value in self.BUILTIN:
274                      yield index, Keyword, value
275                  elif value in self.BUILTIN_DECLARATION:
276                      yield index, Keyword.Declaration, value
277                  elif value in self.BUILTIN_NAMESPACE:
278                      yield index, Keyword.Namespace, value
279                  elif value in self.CONSTANT:
280                      yield index, Name.Constant, value
281                  elif value in self.PSEUDO_VAR:
282                      yield index, Name.Builtin.Pseudo, value
283                  else:
284                      yield index, token, value
285              else:
286                  yield index, token, value

288      def gen_elixir_sigil_rules():
289          # all valid sigil terminators (excluding heredocs)
290          terminators = [
291              (r'\{', r'\}', '}',   'cb'),
292              (r'\[', r'\]', r'\]', 'sb'),
293              (r'\(', r'\)', ')',   'pa'),
294              ('<',   '>',   '>',   'ab'),
295              ('/',   '/',   '/',   'slas'),
296              (r'\|', r'\|', '|',   'pipe'),
297              ('"',   '"',   '"',   'quot'),
298              ("'",   "'",   "'",   'apos'),
299          ]
300  
301          # heredocs have slightly different rules
302          triquotes = [(r'"""', 'triquot'), (r"'''", 'triapos')]
303  
304          token = String.Other
305          states = {'sigils': []}
306  
307          for term, name in triquotes:
308              states['sigils'] += [
309                  (r'(~[a-z])(%s)' % (term,), bygroups(token, String.Heredoc),
310                      (name + '-end', name + '-intp')),
311                  (r'(~[A-Z])(%s)' % (term,), bygroups(token, String.Heredoc),
312                      (name + '-end', name + '-no-intp')),
313              ]
314  
315              states[name + '-end'] = [
316                  (r'[a-zA-Z]+', token, '#pop'),
317                  default('#pop'),
318              ]
319              states[name + '-intp'] = [
320                  (r'^\s*' + term, String.Heredoc, '#pop'),
321                  include('heredoc_interpol'),
322              ]
323              states[name + '-no-intp'] = [
324                  (r'^\s*' + term, String.Heredoc, '#pop'),
325                  include('heredoc_no_interpol'),
326              ]
327  
328          for lterm, rterm, rterm_class, name in terminators:
329              states['sigils'] += [
330                  (r'~[a-z]' + lterm, token, name + '-intp'),
331                  (r'~[A-Z]' + lterm, token, name + '-no-intp'),
332              ]
333              states[name + '-intp'] = \
334                  gen_elixir_sigstr_rules(rterm, rterm_class, token)
335              states[name + '-no-intp'] = \
336                  gen_elixir_sigstr_rules(rterm, rterm_class, token, interpol=False)
337  
338          return states
339  
340      op3_re = "|".join(re.escape(s) for s in OPERATORS3)
341      op2_re = "|".join(re.escape(s) for s in OPERATORS2)
342      op1_re = "|".join(re.escape(s) for s in OPERATORS1)
343      ops_re = r'(?:%s|%s|%s)' % (op3_re, op2_re, op1_re)
344      punctuation_re = "|".join(re.escape(s) for s in PUNCTUATION)
345      alnum = r'\w'
346      name_re = r'(?:\.\.\.|[a-z_]%s*[!?]?)' % alnum
347      modname_re = r'[A-Z]%(alnum)s*(?:\.[A-Z]%(alnum)s*)*' % {'alnum': alnum}
348      complex_name_re = r'(?:%s|%s|%s)' % (name_re, modname_re, ops_re)
349      special_atom_re = r'(?:\.\.\.|<<>>|%\{\}|%|\{\})'
350  
351      long_hex_char_re = r'(\\x\{)([\da-fA-F]+)(\})'
352      hex_char_re = r'(\\x[\da-fA-F]{1,2})'
353      escape_char_re = r'(\\[abdefnrstv])'
354  
355      tokens = {
356          'root': [
357              (r'\s+', Text),
358              (r'#.*$', Comment.Single),
359  
360              # Various kinds of characters
361              (r'(\?)' + long_hex_char_re,
362                  bygroups(String.Char,
363                           String.Escape, Number.Hex, String.Escape)),
364              (r'(\?)' + hex_char_re,
365                  bygroups(String.Char, String.Escape)),
366              (r'(\?)' + escape_char_re,
367                  bygroups(String.Char, String.Escape)),
368              (r'\?\\?.', String.Char),
369  
370              # '::' has to go before atoms
371              (r':::', String.Symbol),
372              (r'::', Operator),
373  
374              # atoms
375              (r':' + special_atom_re, String.Symbol),
376              (r':' + complex_name_re, String.Symbol),
377              (r':"', String.Symbol, 'string_double_atom'),
378              (r":'", String.Symbol, 'string_single_atom'),
379  
380              # [keywords: ...]
381              (r'(%s|%s)(:)(?=\s|\n)' % (special_atom_re, complex_name_re),
382                  bygroups(String.Symbol, Punctuation)),
383  
384              # @attributes
385              (r'@' + name_re, Name.Attribute),
386  
387              # identifiers
388              (name_re, Name),
389              (r'(%%?)(%s)' % (modname_re,), bygroups(Punctuation, Name.Class)),
390  
391              # operators and punctuation
392              (op3_re, Operator),
393              (op2_re, Operator),
394              (punctuation_re, Punctuation),
395              (r'&\d', Name.Entity),   # anon func arguments
396              (op1_re, Operator),
397  
398              # numbers
399              (r'0b[01]+', Number.Bin),
400              (r'0o[0-7]+', Number.Oct),
401              (r'0x[\da-fA-F]+', Number.Hex),
402              (r'\d(_?\d)*\.\d(_?\d)*([eE][-+]?\d(_?\d)*)?', Number.Float),
403              (r'\d(_?\d)*', Number.Integer),
404  
405              # strings and heredocs
406              (r'"""\s*', String.Heredoc, 'heredoc_double'),
407              (r"'''\s*$", String.Heredoc, 'heredoc_single'),
408              (r'"', String.Double, 'string_double'),
409              (r"'", String.Single, 'string_single'),
410  
411              include('sigils'),
412  
413              (r'%\{', Punctuation, 'map_key'),
414              (r'\{', Punctuation, 'tuple'),
415          ],
416          'heredoc_double': [
417              (r'^\s*"""', String.Heredoc, '#pop'),
418              include('heredoc_interpol'),
419          ],
420          'heredoc_single': [
421              (r"^\s*'''", String.Heredoc, '#pop'),
422              include('heredoc_interpol'),
423          ],
424          'heredoc_interpol': [
425              (r'[^#\\\n]+', String.Heredoc),
426              include('escapes'),
427              (r'\\.', String.Heredoc),
428              (r'\n+', String.Heredoc),
429              include('interpol'),
430          ],
431          'heredoc_no_interpol': [
432              (r'[^\\\n]+', String.Heredoc),
433              (r'\\.', String.Heredoc),
434              (r'\n+', String.Heredoc),
435          ],
436          'escapes': [
437              (long_hex_char_re,
438                  bygroups(String.Escape, Number.Hex, String.Escape)),
439              (hex_char_re, String.Escape),
440              (escape_char_re, String.Escape),
441          ],
442          'interpol': [
443              (r'#\{', String.Interpol, 'interpol_string'),
444          ],
445          'interpol_string': [
446              (r'\}', String.Interpol, "#pop"),
447              include('root')
448          ],
449          'map_key': [
450              include('root'),
451              (r':', Punctuation, 'map_val'),
452              (r'=>', Punctuation, 'map_val'),
453              (r'\}', Punctuation, '#pop'),
454          ],
455          'map_val': [
456              include('root'),
457              (r',', Punctuation, '#pop'),
458              (r'(?=\})', Punctuation, '#pop'),
459          ],
460          'tuple': [
461              include('root'),
462              (r'\}', Punctuation, '#pop'),
463          ],
464      }
465      tokens.update(gen_elixir_string_rules('double', '"', String.Double))
466      tokens.update(gen_elixir_string_rules('single', "'", String.Single))
467      tokens.update(gen_elixir_string_rules('double_atom', '"', String.Symbol))
468      tokens.update(gen_elixir_string_rules('single_atom', "'", String.Symbol))
469      tokens.update(gen_elixir_sigil_rules())

472  class ElixirConsoleLexer(Lexer):
473      """ 
474      For Elixir interactive console (iex) output like: 
475   
476      .. sourcecode:: iex 
477   
478          iex> [head | tail] = [1,2,3] 
479          [1,2,3] 
480          iex> head 
481          1 
482          iex> tail 
483          [2,3] 
484          iex> [head | tail] 
485          [1,2,3] 
486          iex> length [head | tail] 
487          3 
488   
489      .. versionadded:: 1.5 
490      """
491  
492      name = 'Elixir iex session'
493      aliases = ['iex']
494      mimetypes = ['text/x-elixir-shellsession']
495  
496      _prompt_re = re.compile(r'(iex|\.{3})((?:\([\w@_.]+\))?\d+|\(\d+\))?> ')
497  
498      def get_tokens_unprocessed(self, text):
499          exlexer = ElixirLexer(**self.options)
500  
501          curcode = ''
502          in_error = False
503          insertions = []
504          for match in line_re.finditer(text):
505              line = match.group()
506              if line.startswith('** '):
507                  in_error = True
508                  insertions.append((len(curcode),
509                                     [(0, Generic.Error, line[:-1])]))
510                  curcode += line[-1:]
511              else:
512                  m = self._prompt_re.match(line)
513                  if m is not None:
514                      in_error = False
515                      end = m.end()
516                      insertions.append((len(curcode),
517                                         [(0, Generic.Prompt, line[:end])]))
518                      curcode += line[end:]
519                  else:
520                      if curcode:
521                          yield from do_insertions(
522                              insertions, exlexer.get_tokens_unprocessed(curcode))
523                          curcode = ''
524                          insertions = []
525                      token = Generic.Error if in_error else Generic.Output
526                      yield match.start(), token, line
527          if curcode:
528              yield from do_insertions(
529                  insertions, exlexer.get_tokens_unprocessed(curcode))
530
