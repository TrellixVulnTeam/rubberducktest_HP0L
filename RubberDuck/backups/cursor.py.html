1    ''' 
2    Cursor 
3    ====== 
4     
5    Shows a cursor following mouse motion events, useful on systems with no 
6    visible native mouse cursor. 
7     
8    Configuration 
9    ------------- 
10    
11   :Parameters: 
12       `texture`: str, defaults to 
13           'data/images/cursor.png' Image used to represent the cursor if 
14           displayed 
15       `size`: tuple, defaults to (40, 40) 
16           Apparent size of the mouse cursor, if displayed, (None,None) value 
17           will keep its real size. 
18       `offset`: tuple, defaults to (None, None) 
19           Offset of the texture image. The default value will align the 
20           top-left corner of the image to the mouse pos. 
21    
22   Example 
23   ------- 
24    
25   In your configuration (`~/.kivy/config.ini`), you can add something like 
26   this:: 
27    
28       [modules] 
29       cursor = texture=mypointer.png,size=20x20,offset=20x20 
30    
31   .. versionadded:: 1.10.0 
32   '''
33   
34   __all__ = ('start', 'stop')
35   
36   from kivy.core.image import Image
37   from kivy.graphics import Color, Rectangle
38   from kivy import kivy_data_dir
39   from kivy.compat import string_types
40   from os.path import join
41   from functools import partial
42   
43   
44   def _mouse_move(texture, size, offset, win, pos, *args):
45       if hasattr(win, '_cursor'):
46           c = win._cursor
47       else:
48           with win.canvas.after:
49               Color(1, 1, 1, 1, mode='rgba')
50               win._cursor = c = Rectangle(texture=texture, size=size)
51   
52       c.pos = pos[0] + offset[0], pos[1] - size[1] + offset[1]
53   
54   
55   def start(win, ctx):
56       cursor_texture = Image(
57           ctx.config.get('texture', join(kivy_data_dir, 'images', 'cursor.png'))
58       ).texture
59       cursor_size = ctx.config.get('size')
60       if isinstance(cursor_size, string_types):
61           cursor_size = [int(x) for x in cursor_size.split('x')]
62       elif not cursor_size:
63           cursor_size = cursor_texture.size
64   
65       cursor_offset = ctx.config.get('offset', (0, 0))
66       if isinstance(cursor_offset, string_types):
67           cursor_offset = [int(x) for x in cursor_offset.split('x')]
68   
69       win.bind(
70           mouse_pos=partial(
71               _mouse_move, cursor_texture, cursor_size, cursor_offset))
72   
73   
74   def stop(win, ctx):
75       win.unbind(mouse_pos=_mouse_move)
