13   class GenericFrame(window.MDIChildWnd):
14       def OnCreateClient(self, cp, context):
15           # handlers for toolbar buttons
16           self.HookCommand(self.OnPrevious, 401)
17           self.HookCommand(self.OnNext, 402)
18           # Its not necessary for us to hook both of these - the
19           # common controls should fall-back all by themselves.
20           # Indeed, given we hook TTN_NEEDTEXTW, commctrl.TTN_NEEDTEXTA
21           # will not be called.
22           self.HookNotify(self.GetTTText, commctrl.TTN_NEEDTEXT)
23           self.HookNotify(self.GetTTText, commctrl.TTN_NEEDTEXTW)
24   
25           #       parent = win32ui.GetMainFrame()
26           parent = self
27           style = (
28               win32con.WS_CHILD
29               | win32con.WS_VISIBLE
30               | afxres.CBRS_SIZE_DYNAMIC
31               | afxres.CBRS_TOP
32               | afxres.CBRS_TOOLTIPS
33               | afxres.CBRS_FLYBY
34           )
35   
36           buttons = (win32ui.ID_APP_ABOUT, win32ui.ID_VIEW_INTERACTIVE)
37           bitmap = win32ui.IDB_BROWSER_HIER
38           tbid = 0xE840
39           self.toolbar = tb = win32ui.CreateToolBar(parent, style, tbid)
40           tb.LoadBitmap(bitmap)
41           tb.SetButtons(buttons)
42   
43           tb.EnableDocking(afxres.CBRS_ALIGN_ANY)
44           tb.SetWindowText("Test")
45           parent.EnableDocking(afxres.CBRS_ALIGN_ANY)
46           parent.DockControlBar(tb)
47           parent.LoadBarState("ToolbarTest")
48           window.MDIChildWnd.OnCreateClient(self, cp, context)
49           return 1
50   
51       def OnDestroy(self, msg):
52           self.SaveBarState("ToolbarTest")
53   
54       def GetTTText(self, std, extra):
55           (hwndFrom, idFrom, code) = std
56           text, hinst, flags = extra
57           if flags & commctrl.TTF_IDISHWND:
58               return  # Not handled
59           if idFrom == win32ui.ID_APP_ABOUT:
60               # our 'extra' return value needs to be the following
61               # entries from a NMTTDISPINFO[W] struct:
62               # (szText, hinst, uFlags).  None means 'don't change
63               # the value'
64               return 0, ("It works!", None, None)
65           return None  # not handled.
66   
67       def GetMessageString(self, id):
68           if id == win32ui.ID_APP_ABOUT:
69               return "Dialog Test\nTest"
70           else:
71               return self._obj_.GetMessageString(id)
72   
73       def OnSize(self, params):
74           print("OnSize called with ", params)
75   
76       def OnNext(self, id, cmd):
77           print("OnNext called")
78   
79       def OnPrevious(self, id, cmd):
80           print("OnPrevious called")
