24   class PrintDemoView(docview.ScrollView):
25       def OnInitialUpdate(self):
26           ret = self._obj_.OnInitialUpdate()
27           self.colors = {
28               "Black": (0x00 << 0) + (0x00 << 8) + (0x00 << 16),
29               "Red": (0xFF << 0) + (0x00 << 8) + (0x00 << 16),
30               "Green": (0x00 << 0) + (0xFF << 8) + (0x00 << 16),
31               "Blue": (0x00 << 0) + (0x00 << 8) + (0xFF << 16),
32               "Cyan": (0x00 << 0) + (0xFF << 8) + (0xFF << 16),
33               "Magenta": (0xFF << 0) + (0x00 << 8) + (0xFF << 16),
34               "Yellow": (0xFF << 0) + (0xFF << 8) + (0x00 << 16),
35           }
36           self.pens = {}
37           for name, color in self.colors.items():
38               self.pens[name] = win32ui.CreatePen(win32con.PS_SOLID, 5, color)
39           self.pen = None
40           self.size = (128, 128)
41           self.SetScaleToFitSize(self.size)
42           self.HookCommand(self.OnFilePrint, afxres.ID_FILE_PRINT)
43           self.HookCommand(self.OnFilePrintPreview, win32ui.ID_FILE_PRINT_PREVIEW)
44           return ret
45   
46       def OnDraw(self, dc):
47           oldPen = None
48           x, y = self.size
49           delta = 2
50           colors = list(self.colors.keys())
51           colors.sort()
52           colors = colors * 2
53           for color in colors:
54               if oldPen is None:
55                   oldPen = dc.SelectObject(self.pens[color])
56               else:
57                   dc.SelectObject(self.pens[color])
58               dc.MoveTo((delta, delta))
59               dc.LineTo((x - delta, delta))
60               dc.LineTo((x - delta, y - delta))
61               dc.LineTo((delta, y - delta))
62               dc.LineTo((delta, delta))
63               delta = delta + 4
64               if x - delta <= 0 or y - delta <= 0:
65                   break
66           dc.SelectObject(oldPen)
67   
68       def OnPrepareDC(self, dc, pInfo):
69           if dc.IsPrinting():
70               mag = self.prtDlg["mag"]
71               dc.SetMapMode(win32con.MM_ANISOTROPIC)
72               dc.SetWindowOrg((0, 0))
73               dc.SetWindowExt((1, 1))
74               dc.SetViewportOrg((0, 0))
75               dc.SetViewportExt((mag, mag))
76   
77       def OnPreparePrinting(self, pInfo):
78           flags = (
79               win32ui.PD_USEDEVMODECOPIES
80               | win32ui.PD_PAGENUMS
81               | win32ui.PD_NOPAGENUMS
82               | win32ui.PD_NOSELECTION
83           )
84           self.prtDlg = ImagePrintDialog(pInfo, PRINTDLGORD, flags)
85           pInfo.SetPrintDialog(self.prtDlg)
86           pInfo.SetMinPage(1)
87           pInfo.SetMaxPage(1)
88           pInfo.SetFromPage(1)
89           pInfo.SetToPage(1)
90           ret = self.DoPreparePrinting(pInfo)
91           return ret
92   
93       def OnBeginPrinting(self, dc, pInfo):
94           return self._obj_.OnBeginPrinting(dc, pInfo)
95   
96       def OnEndPrinting(self, dc, pInfo):
97           del self.prtDlg
98           return self._obj_.OnEndPrinting(dc, pInfo)
99   
100      def OnFilePrintPreview(self, *arg):
101          self._obj_.OnFilePrintPreview()
102  
103      def OnFilePrint(self, *arg):
104          self._obj_.OnFilePrint()

106      def OnPrint(self, dc, pInfo):
107          doc = self.GetDocument()
108          metrics = dc.GetTextMetrics()
109          cxChar = metrics["tmAveCharWidth"]
110          cyChar = metrics["tmHeight"]
111          left, top, right, bottom = pInfo.GetDraw()
112          dc.TextOut(0, 2 * cyChar, doc.GetTitle())
113          top = top + (7 * cyChar) / 2
114          dc.MoveTo(left, top)
115          dc.LineTo(right, top)
116          top = top + cyChar
117          # this seems to have not effect...
118          # get what I want with the dc.SetWindowOrg calls
119          pInfo.SetDraw((left, top, right, bottom))
120          dc.SetWindowOrg((0, -top))
121  
122          self.OnDraw(dc)
123          dc.SetTextAlign(win32con.TA_LEFT | win32con.TA_BOTTOM)
124  
125          rect = self.GetWindowRect()
126          rect = self.ScreenToClient(rect)
127          height = rect[3] - rect[1]
128          dc.SetWindowOrg((0, -(top + height + cyChar)))
129          dc.MoveTo(left, 0)
130          dc.LineTo(right, 0)
131  
132          x = 0
133          y = (3 * cyChar) / 2
134  
135          dc.TextOut(x, y, doc.GetTitle())
136          y = y + cyChar

139  class PrintDemoApp(app.CApp):
140      def __init__(self):
141          app.CApp.__init__(self)
142  
143      def InitInstance(self):
144          template = PrintDemoTemplate(None, None, None, PrintDemoView)
145          self.AddDocTemplate(template)
146          self._obj_.InitMDIInstance()
147          self.LoadMainFrame()
148          doc = template.OpenDocumentFile(None)
149          doc.SetTitle("Custom Print Document")

152  class ImagePrintDialog(dialog.PrintDialog):
153  
154      sectionPos = "Image Print Demo"
155  
156      def __init__(self, pInfo, dlgID, flags=win32ui.PD_USEDEVMODECOPIES):
157          dialog.PrintDialog.__init__(self, pInfo, dlgID, flags=flags)
158          mag = win32ui.GetProfileVal(self.sectionPos, "Document Magnification", 0)
159          if mag <= 0:
160              mag = 2
161              win32ui.WriteProfileVal(self.sectionPos, "Document Magnification", mag)
162  
163          self["mag"] = mag
164  
165      def OnInitDialog(self):
166          self.magCtl = self.GetDlgItem(IDC_PRINT_MAG_EDIT)
167          self.magCtl.SetWindowText(repr(self["mag"]))
168          return dialog.PrintDialog.OnInitDialog(self)
169  
170      def OnOK(self):
171          dialog.PrintDialog.OnOK(self)
172          strMag = self.magCtl.GetWindowText()
173          try:
174              self["mag"] = int(strMag)
175          except:
176              pass
177          win32ui.WriteProfileVal(self.sectionPos, "Document Magnification", self["mag"])
