BackupSeek_streamheaders.py 
1    ## demonstrates using BackupSeek to enumerate data streams for a file
2    import win32file, win32api, win32con
3    from win32com import storagecon
4    import pythoncom, pywintypes
5    import struct, traceback
6    
7    stream_types = {
8        win32con.BACKUP_DATA: "Standard data",
9        win32con.BACKUP_EA_DATA: "Extended attribute data",
10       win32con.BACKUP_SECURITY_DATA: "Security descriptor data",
11       win32con.BACKUP_ALTERNATE_DATA: "Alternative data streams",
12       win32con.BACKUP_LINK: "Hard link information",
13       win32con.BACKUP_PROPERTY_DATA: "Property data",
14       win32con.BACKUP_OBJECT_ID: "Objects identifiers",
15       win32con.BACKUP_REPARSE_DATA: "Reparse points",
16       win32con.BACKUP_SPARSE_BLOCK: "Sparse file",
17   }
18   
19   tempdir = win32api.GetTempPath()
20   tempfile = win32api.GetTempFileName(tempdir, "bkr")[0]
21   print("Filename:", tempfile)
22   
23   f = open(tempfile, "w")
24   f.write("some random junk" + "x" * 100)
25   f.close()
26   
27   f = open(tempfile + ":streamdata", "w")
28   f.write("data written to alternate stream" + "y" * 100)
29   f.close()
30   
31   f = open(tempfile + ":anotherstream", "w")
32   f.write("z" * 200)
33   f.close()
34   
35   ## add Summary Information, which is stored as a separate stream
36   m = storagecon.STGM_READWRITE | storagecon.STGM_SHARE_EXCLUSIVE | storagecon.STGM_DIRECT
37   pss = pythoncom.StgOpenStorageEx(
38       tempfile, m, storagecon.STGFMT_FILE, 0, pythoncom.IID_IPropertySetStorage, None
39   )
40   ps = pss.Create(
41       pythoncom.FMTID_SummaryInformation,
42       pythoncom.IID_IPropertyStorage,
43       0,
44       storagecon.STGM_READWRITE | storagecon.STGM_SHARE_EXCLUSIVE,
45   )
46   ps.WriteMultiple(
47       (storagecon.PIDSI_KEYWORDS, storagecon.PIDSI_COMMENTS), ("keywords", "comments")
48   )
49   ps = None
50   pss = None
51   
52   sa = pywintypes.SECURITY_ATTRIBUTES()
53   sa.bInheritHandle = False
54   h = win32file.CreateFile(
55       tempfile,
56       win32con.GENERIC_ALL,
57       win32con.FILE_SHARE_READ,
58       sa,
59       win32con.OPEN_EXISTING,
60       win32file.FILE_FLAG_BACKUP_SEMANTICS,
61       None,
62   )
63   
64   
65   """ stream header: 
66   typedef struct _WIN32_STREAM_ID { 
67       DWORD dwStreamId;  DWORD dwStreamAttributes;  LARGE_INTEGER Size; 
68       DWORD dwStreamNameSize;  WCHAR cStreamName[ANYSIZE_ARRAY]; 
69   } 
70   """
71   
72   win32_stream_id_format = "LLQL"
73   win32_stream_id_size = struct.calcsize(win32_stream_id_format)
74   
75   
76   def parse_stream_header(h, ctxt, data):
77       stream_type, stream_attributes, stream_size, stream_name_size = struct.unpack(
78           win32_stream_id_format, data
79       )
80       print(
81           "\nType:",
82           stream_type,
83           stream_types[stream_type],
84           "Attributes:",
85           stream_attributes,
86           "Size:",
87           stream_size,
88           "Name len:",
89           stream_name_size,
90       )
91       if stream_name_size > 0:
92           ## ??? sdk says this size is in characters, but it appears to be number of bytes ???
93           bytes_read, stream_name_buf, ctxt = win32file.BackupRead(
94               h, stream_name_size, None, False, True, ctxt
95           )
96           stream_name = pywintypes.UnicodeFromRaw(stream_name_buf[:])
97       else:
98           stream_name = "Unnamed"
99       print("Name:" + stream_name)
100      return (
101          ctxt,
102          stream_type,
103          stream_attributes,
104          stream_size,
105          stream_name_size,
106          stream_name,
107      )
108  
109  
110  ctxt = 0
111  win32_stream_id_buf = (
112      None  ## gets rebound to a writable buffer on first call and reused
113  )
114  while 1:
115      bytes_read, win32_stream_id_buf, ctxt = win32file.BackupRead(
116          h, win32_stream_id_size, win32_stream_id_buf, False, True, ctxt
117      )
118      if bytes_read == 0:
119          break
120      (
121          ctxt,
122          stream_type,
123          stream_attributes,
124          stream_size,
125          stream_name_size,
126          stream_name,
127      ) = parse_stream_header(h, ctxt, win32_stream_id_buf[:])
128      if stream_size > 0:
129          bytes_moved = win32file.BackupSeek(h, stream_size, ctxt)
130          print("Moved: ", bytes_moved)
131  
132  win32file.BackupRead(h, win32_stream_id_size, win32_stream_id_buf, True, True, ctxt)
133  win32file.CloseHandle(h)
