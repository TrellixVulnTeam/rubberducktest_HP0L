filechooser.py 
1    ''' 
2    FileChooser 
3    =========== 
4     
5    The FileChooser module provides various classes for describing, displaying and 
6    browsing file systems. 
7     
8    Simple widgets 
9    -------------- 
10    
11   There are two ready-to-use widgets that provide views of the file system. Each 
12   of these present the files and folders in a different style. 
13    
14   The :class:`FileChooserListView` displays file entries as text items in a 
15   vertical list, where folders can be collapsed and expanded. 
16    
17   .. image:: images/filechooser_list.png 
18    
19   The :class:`FileChooserIconView` presents icons and text from left to right, 
20   wrapping them as required. 
21    
22   .. image:: images/filechooser_icon.png 
23    
24   They both provide for scrolling, selection and basic user interaction. 
25   Please refer to the :class:`FileChooserController` for details on supported 
26   events and properties. 
27    
28   Widget composition 
29   ------------------ 
30    
31   FileChooser classes adopt a 
32   `MVC <https://en.wikipedia.org/wiki/Model%E2%80%93view%E2%80%93controller>`_ 
33   design. They are exposed so that you to extend and customize your file chooser 
34   according to your needs. 
35    
36   The FileChooser classes can be categorized as follows: 
37    
38   * Models are represented by concrete implementations of the 
39     :class:`FileSystemAbstract` class, such as the :class:`FileSystemLocal`. 
40    
41   * Views are represented by the :class:`FileChooserListLayout` and 
42     :class:`FileChooserIconLayout` classes. These are used by the 
43     :class:`FileChooserListView` and :class:`FileChooserIconView` widgets 
44     respectively. 
45    
46   * Controllers are represented by concrete implementations of the 
47     :class:`FileChooserController`, namely the :class:`FileChooser`, 
48     :class:`FileChooserIconView` and :class:`FileChooserListView` classes. 
49    
50   This means you can define your own views or provide :class:`FileSystemAbstract` 
51   implementations for alternative file systems for use with these widgets. 
52   The :class:`FileChooser` can be used as a controller for handling multiple, 
53   synchronized views of the same path. By combining these elements, you can add 
54   your own views and file systems and have them easily interact with the existing 
55   components. 
56    
57   Usage example 
58   ------------- 
59    
60   main.py 
61    
62   .. include:: ../../examples/RST_Editor/main.py 
63       :literal: 
64    
65   editor.kv 
66    
67   .. highlight:: kv 
68    
69   .. include:: ../../examples/RST_Editor/editor.kv 
70       :literal: 
71    
72   .. versionadded:: 1.0.5 
73    
74   .. versionchanged:: 1.2.0 
75    
76       In the chooser template, the `controller` is no longer a direct reference 
77       but a weak-reference. If you are upgrading, you should change the notation 
78       `root.controller.xxx` to `root.controller().xxx`. 
79

130  def alphanumeric_folders_first(files, filesystem):
131      return (sorted(f for f in files if filesystem.is_dir(f)) +
132              sorted(f for f in files if not filesystem.is_dir(f)))

175      def is_hidden(self, fn):
176          if platform == 'win':
177              if not _have_win32file:
178                  return False
179              try:
180                  return GetFileAttributesExW(fn)[0] & FILE_ATTRIBUTE_HIDDEN
181              except error:
182                  # This error can occurred when a file is already accessed by
183                  # someone else. So don't return to True, because we have lot
184                  # of chances to not being able to do anything with it.
185                  Logger.exception('unable to access to <%s>' % fn)
186                  return True
187  
188          return basename(fn).startswith('.')
189  
190      def is_dir(self, fn):
191          return isdir(fn)

312          `on_entry_added`: entry, parent

314              return True from this event, the entry is not added to FileChooser. 
315          `on_entries_cleared` 
316              Fired when the the entries list is cleared, usually when the 
317              root is refreshed. 
318          `on_subentry_to_entry`: entry, parent 
319              Fired when a sub-entry is added to an existing entry or

336      path = StringProperty(u'/')
337      ''' 
338      path is a :class:`~kivy.properties.StringProperty` and defaults to the 
339      current working directory as a unicode string. It specifies the path on the 
340      filesystem that this controller should refer to. 
341   
342      .. warning:: 
343   
344          If a unicode path is specified, all the files returned will be in 
345          unicode, allowing the display of unicode files and paths. If a bytes 
346          path is specified, only files and paths with ascii names will be 
347          displayed properly: non-ascii filenames will be displayed and listed

351      filters = ListProperty([])
352      ''' 
353      filters specifies the filters to be applied to the files in the directory. 
354      filters is a :class:`~kivy.properties.ListProperty` and defaults to []. 
355      This is equivalent to '\\*' i.e. nothing is filtered. 
356   
357      The filters are not reset when the path changes. You need to do that 
358      yourself if desired. 
359   
360      There are two kinds of filters: patterns and callbacks. 
361   
362      #. Patterns 
363   
364          e.g. ['\\*.png']. 
365          You can use the following patterns: 
366   
367              ========== ================================= 
368              Pattern     Meaning 
369              ========== ================================= 
370              \\*         matches everything 
371              ?          matches any single character 
372              [seq]      matches any character in seq 
373              [!seq]     matches any character not in seq 
374              ========== ================================= 
375   
376      #. Callbacks 
377   
378          You can specify a function that will be called for each file. The 
379          callback will be passed the folder and file name as the first 
380          and second parameters respectively. It should return True to 
381          indicate a match and False otherwise.

383      .. versionchanged:: 1.4.0

600      def entry_touched(self, entry, touch):
601          '''(internal) This method must be called by the template when an entry 
602          is touched by the user. 
603          '''
604          if (
605              'button' in touch.profile and touch.button in (
606                  'scrollup', 'scrolldown', 'scrollleft', 'scrollright')):
607              return False
608  
609          _dir = self.file_system.is_dir(entry.path)
610          dirselect = self.dirselect
611  
612          if _dir and dirselect and touch.is_double_tap:
613              self.open_entry(entry)
614              return
615  
616          if self.multiselect:
617              if entry.path in self.selection:
618                  self.selection.remove(entry.path)
619              else:
620                  if _dir and not self.dirselect:
621                      self.open_entry(entry)
622                      return
623                  self.selection.append(entry.path)
624          else:
625              if _dir and not self.dirselect:
626                  return
627              self.selection = [abspath(join(self.path, entry.path)), ]
628  
629      def entry_released(self, entry, touch):
630          '''(internal) This method must be called by the template when an entry 
631          is touched by the user. 
632   
633          .. versionadded:: 1.1.0 
634          '''
635          if (
636              'button' in touch.profile and touch.button in (
637                  'scrollup', 'scrolldown', 'scrollleft', 'scrollright')):
638              return False
639          if not self.multiselect:
640              if self.file_system.is_dir(entry.path) and not self.dirselect:
641                  self.open_entry(entry)
642              elif touch.is_double_tap:
643                  if self.dirselect and self.file_system.is_dir(entry.path):
644                      return
645                  else:
646                      self.dispatch('on_submit', self.selection, touch)

648      def open_entry(self, entry):
649          try:
650              # Just check if we can list the directory. This is also what
651              # _add_file does, so if it fails here, it would also fail later
652              # on. Do the check here to prevent setting path to an invalid
653              # directory that we cannot list.
654              self.file_system.listdir(entry.path)
655          except OSError:
656              entry.locked = True
657          else:
658              # If entry.path is to jump to previous directory, update path with
659              # parent directory
660              self.path = abspath(join(self.path, entry.path))
661              self.selection = [self.path, ] if self.dirselect else []

663      def _apply_filters(self, files):
664          if not self.filters:
665              return files
666          filtered = []
667          for filt in self.filters:
668              if isinstance(filt, collections.Callable):
669                  filtered.extend([fn for fn in files if filt(self.path, fn)])
670              else:
671                  filtered.extend([fn for fn in files if fnmatch(fn, filt)])
672          if not self.filter_dirs:
673              dirs = [fn for fn in files if self.file_system.is_dir(fn)]
674              filtered.extend(dirs)
675          return list(set(filtered))

677      def get_nice_size(self, fn):
678          '''Pass the filepath. Returns the size in the best human readable 
679          format or '' if it is a directory (Don't recursively calculate size). 
680          '''
681          if self.file_system.is_dir(fn):
682              return ''
683          try:
684              size = self.file_system.getsize(fn)
685          except OSError:
686              return '--'
687  
688          for unit in filesize_units:
689              if size < 1024.0:
690                  return '%1.0f %s' % (size, unit)
691              size /= 1024.0

693      def _update_files(self, *args, **kwargs):
694          # trigger to start gathering the files in the new directory
695          # we'll start a timer that will do the job, 10 times per frames
696          # (default)
697          self._gitems = []
698          self._gitems_parent = kwargs.get('parent', None)
699          self._gitems_gen = self._generate_file_entries(
700              path=kwargs.get('path', self.path),
701              parent=self._gitems_parent)
702          self.path = abspath(self.path)
703  
704          # cancel any previous clock if exist
705          ev = self._create_files_entries_ev
706          if ev is not None:
707              ev.cancel()
708  
709          # show the progression screen
710          self._hide_progress()
711          if self._create_files_entries():
712              # not enough for creating all the entries, all a clock to continue
713              # start a timer for the next 100 ms
714              if ev is None:
715                  ev = self._create_files_entries_ev = Clock.schedule_interval(
716                      self._create_files_entries, .1)
717              ev()

719      def _get_file_paths(self, items):
720          return [file.path for file in items]

722      def _create_files_entries(self, *args):
723          # create maximum entries during 50ms max, or 10 minimum (slow system)
724          # (on a "fast system" (core i7 2700K), we can create up to 40 entries
725          # in 50 ms. So 10 is fine for low system.
726          start = time()
727          finished = False
728          index = total = count = 1
729          while time() - start < 0.05 or count < 10:
730              try:
731                  index, total, item = next(self._gitems_gen)
732                  self._gitems.append(item)
733                  count += 1
734              except StopIteration:
735                  finished = True
736                  break
737              except TypeError:  # in case _gitems_gen is None
738                  finished = True
739                  break
740  
741          # if this wasn't enough for creating all the entries, show a progress
742          # bar, and report the activity to the user.
743          if not finished:
744              self._show_progress()
745              self._progress.total = total
746              self._progress.index = index
747              return True
748  
749          # we created all the files, now push them on the view
750          self._items = items = self._gitems
751          parent = self._gitems_parent
752          if parent is None:
753              self.dispatch('on_entries_cleared')
754              for entry in items:
755                  self.dispatch('on_entry_added', entry, parent)
756          else:
757              parent.entries[:] = items
758              for entry in items:
759                  self.dispatch('on_subentry_to_entry', entry, parent)
760          self.files[:] = self._get_file_paths(items)
761  
762          # stop the progression / creation
763          self._hide_progress()
764          self._gitems = None
765          self._gitems_gen = None
766          ev = self._create_files_entries_ev
767          if ev is not None:
768              ev.cancel()
769          return False

771      def cancel(self, *largs):
772          '''Cancel any background action started by filechooser, such as loading 
773          a new directory. 
774   
775          .. versionadded:: 1.2.0 
776          '''
777          ev = self._create_files_entries_ev
778          if ev is not None:
779              ev.cancel()
780  
781          self._hide_progress()
782          if len(self._previous_path) > 1:
783              # if we cancel any action, the path will be set same as the
784              # previous one, so we can safely cancel the update of the previous
785              # path.
786              self.path = self._previous_path[-2]
787  
788              ev = self._update_files_ev
789              if ev is not None:
790                  ev.cancel()

792      def _show_progress(self):
793          if self._progress:
794              return
795          cls = self.progress_cls
796          if isinstance(cls, string_types):
797              cls = Factory.get(cls)
798          self._progress = cls(path=self.path)
799          self._progress.value = 0
800          self.add_widget(self._progress)

802      def _hide_progress(self):
803          if self._progress:
804              self.remove_widget(self._progress)
805              self._progress = None

807      def _generate_file_entries(self, *args, **kwargs):
808          # Generator that will create all the files entries.
809          # the generator is used via _update_files() and _create_files_entries()
810          # don't use it directly.
811          is_root = False
812          path = kwargs.get('path', self.path)
813          have_parent = kwargs.get('parent', None) is not None
814  
815          # Add the components that are always needed
816          if self.rootpath:
817              rootpath = realpath(self.rootpath)
818              path = realpath(path)
819              if not path.startswith(rootpath):
820                  self.path = rootpath
821                  return
822              elif path == rootpath:
823                  is_root = True
824          else:
825              if platform == 'win':
826                  is_root = splitdrive(path)[1] in (sep, altsep)
827              elif platform in ('macosx', 'linux', 'android', 'ios'):
828                  is_root = normpath(expanduser(path)) == sep
829              else:
830                  # Unknown fs, just always add the .. entry but also log
831                  Logger.warning('Filechooser: Unsupported OS: %r' % platform)
832          # generate an entries to go back to previous
833          if not is_root and not have_parent:
834              back = '..' + sep
835              if platform == 'win':
836                  new_path = path[:path.rfind(sep)]
837                  if sep not in new_path:
838                      new_path += sep
839                  pardir = self._create_entry_widget(dict(
840                      name=back, size='', path=new_path, controller=ref(self),
841                      isdir=True, parent=None, sep=sep,
842                      get_nice_size=lambda: ''))
843              else:
844                  pardir = self._create_entry_widget(dict(
845                      name=back, size='', path=back, controller=ref(self),
846                      isdir=True, parent=None, sep=sep,
847                      get_nice_size=lambda: ''))
848              yield 0, 1, pardir
849  
850          # generate all the entries for files
851          try:
852              for index, total, item in self._add_files(path):
853                  yield index, total, item
854          except OSError:
855              Logger.exception('Unable to open directory <%s>' % self.path)
856              self.files[:] = []
857  
858      def _create_entry_widget(self, ctx):
859          template = self.layout._ENTRY_TEMPLATE\
860              if self.layout else self._ENTRY_TEMPLATE
861          return Builder.template(template, **ctx)

863      def _add_files(self, path, parent=None):
864          path = expanduser(path)
865          if isfile(path):
866              path = dirname(path)
867  
868          files = []
869          fappend = files.append
870          for f in self.file_system.listdir(path):
871              try:
872                  # In the following, use fully qualified filenames
873                  fappend(normpath(join(path, f)))
874              except UnicodeDecodeError:
875                  Logger.exception('unable to decode <{}>'.format(f))
876              except UnicodeEncodeError:
877                  Logger.exception('unable to encode <{}>'.format(f))
878          # Apply filename filters
879          files = self._apply_filters(files)
880          # Sort the list of files
881          files = self.sort_func(files, self.file_system)
882          is_hidden = self.file_system.is_hidden
883          if not self.show_hidden:
884              files = [x for x in files if not is_hidden(x)]
885          self.files[:] = files
886          total = len(files)
887          wself = ref(self)
888          for index, fn in enumerate(files):
889  
890              def get_nice_size():
891                  # Use a closure for lazy-loading here
892                  return self.get_nice_size(fn)
893  
894              ctx = {'name': basename(fn),
895                     'get_nice_size': get_nice_size,
896                     'path': fn,
897                     'controller': wself,
898                     'isdir': self.file_system.is_dir(fn),
899                     'parent': parent,
900                     'sep': sep}
901              entry = self._create_entry_widget(ctx)
902              yield index, total, entry

1030         views = [view.VIEWNAME for view in self._views]
1031         if views != self._view_list:
1032             self._view_list = views
1033             if self._view_mode not in self._view_list:
1034                 self._view_mode = self._view_list[0]

1037     def update_view(self, *args):
1038         self.rebuild_views()
1039 
1040         sm = self.manager
1041         viewlist = self._view_list
1042         view = self.view_mode
1043         current = sm.current[:-4]
1044 
1045         viewindex = viewlist.index(view) if view in viewlist else 0
1046         currentindex = viewlist.index(current) if current in viewlist else 0
1047 
1048         direction = 'left' if currentindex < viewindex else 'right'
1049 
1050         sm.transition.direction = direction
1051         sm.current = view + 'view'
