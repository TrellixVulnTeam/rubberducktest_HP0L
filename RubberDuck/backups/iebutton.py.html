77   class IEButton:
78       """ 
79       The actual COM server class 
80       """
81   
82       _com_interfaces_ = _iebutton_com_interfaces_
83       _public_methods_ = _iebutton_methods_
84       _reg_clsctx_ = pythoncom.CLSCTX_INPROC_SERVER
85       _button_text_ = "IE Button"
86       _tool_tip_ = "An example implementation for an IE Button."
87       _icon_ = ""
88       _hot_icon_ = ""
89   
90       def __init__(self):
91           # put stubs for non-implemented methods
92           for method in self._public_methods_:
93               if not hasattr(self, method):
94                   print("providing default stub for %s" % method)
95                   setattr(self, method, Stub(method))
96   
97       def QueryStatus(self, pguidCmdGroup, prgCmds, cmdtextf):
98           # 'cmdtextf' is the 'cmdtextf' element from the OLECMDTEXT structure,
99           # or None if a NULL pointer was passed.
100          result = []
101          for id, flags in prgCmds:
102              flags |= axcontrol.OLECMDF_SUPPORTED | axcontrol.OLECMDF_ENABLED
103              result.append((id, flags))
104          if cmdtextf is None:
105              cmdtext = None  # must return None if nothing requested.
106          # IE never seems to want any text - this code is here for
107          # demo purposes only
108          elif cmdtextf == axcontrol.OLECMDTEXTF_NAME:
109              cmdtext = "IEButton Name"
110          else:
111              cmdtext = "IEButton State"
112          return result, cmdtext
113  
114      def Exec(self, pguidCmdGroup, nCmdID, nCmdExecOpt, pvaIn):
115          print(pguidCmdGroup, nCmdID, nCmdExecOpt, pvaIn)
116          print("IOleCommandTarget::Exec called.")
117          # self.webbrowser.ShowBrowserBar(GUID_IETOOLBAR, not is_ietoolbar_visible())
118  
119      def SetSite(self, unknown):
120          if unknown:
121              # first get a command target
122              cmdtarget = unknown.QueryInterface(axcontrol.IID_IOleCommandTarget)
123              # then travel over to a service provider
124              serviceprovider = cmdtarget.QueryInterface(pythoncom.IID_IServiceProvider)
125              # finally ask for the internet explorer application, returned as a dispatch object
126              self.webbrowser = win32com.client.Dispatch(
127                  serviceprovider.QueryService(
128                      "{0002DF05-0000-0000-C000-000000000046}", pythoncom.IID_IDispatch
129                  )
130              )
131          else:
132              # lose all references
133              self.webbrowser = None

139  def register(classobj):
140      import winreg
141  
142      subKeyCLSID = (
143          "SOFTWARE\\Microsoft\\Internet Explorer\\Extensions\\%38s"
144          % classobj._reg_clsid_
145      )
146      try:
147          hKey = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, subKeyCLSID)
148          subKey = winreg.SetValueEx(
149              hKey, "ButtonText", 0, winreg.REG_SZ, classobj._button_text_
150          )
151          winreg.SetValueEx(
152              hKey, "ClsidExtension", 0, winreg.REG_SZ, classobj._reg_clsid_
153          )  # reg value for calling COM object
154          winreg.SetValueEx(
155              hKey, "CLSID", 0, winreg.REG_SZ, "{1FBA04EE-3024-11D2-8F1F-0000F87ABD16}"
156          )  # CLSID for button that sends command to COM object
157          winreg.SetValueEx(hKey, "Default Visible", 0, winreg.REG_SZ, "Yes")
158          winreg.SetValueEx(hKey, "ToolTip", 0, winreg.REG_SZ, classobj._tool_tip_)
159          winreg.SetValueEx(hKey, "Icon", 0, winreg.REG_SZ, classobj._icon_)
160          winreg.SetValueEx(hKey, "HotIcon", 0, winreg.REG_SZ, classobj._hot_icon_)
161      except WindowsError:
162          print("Couldn't set standard toolbar reg keys.")
163      else:
164          print("Set standard toolbar reg keys.")
165

167  def unregister(classobj):
168      import winreg
169  
170      subKeyCLSID = (
171          "SOFTWARE\\Microsoft\\Internet Explorer\\Extensions\\%38s"
172          % classobj._reg_clsid_
173      )
174      try:
175          hKey = winreg.CreateKey(winreg.HKEY_LOCAL_MACHINE, subKeyCLSID)
176          subKey = winreg.DeleteValue(hKey, "ButtonText")
177          winreg.DeleteValue(hKey, "ClsidExtension")  # for calling COM object
178          winreg.DeleteValue(hKey, "CLSID")
179          winreg.DeleteValue(hKey, "Default Visible")
180          winreg.DeleteValue(hKey, "ToolTip")
181          winreg.DeleteValue(hKey, "Icon")
182          winreg.DeleteValue(hKey, "HotIcon")
183          winreg.DeleteKey(winreg.HKEY_LOCAL_MACHINE, subKeyCLSID)
184      except WindowsError:
185          print("Couldn't delete Standard toolbar regkey.")
186      else:
187          print("Deleted Standard toolbar regkey.")
