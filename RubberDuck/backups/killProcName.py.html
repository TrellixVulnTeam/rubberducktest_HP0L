killProcName.py 
1    # Kills a process by process name
2    #
3    # Uses the Performance Data Helper to locate the PID, then kills it.
4    # Will only kill the process if there is only one process of that name
5    # (eg, attempting to kill "Python.exe" will only work if there is only
6    # one Python.exe running.  (Note that the current process does not
7    # count - ie, if Python.exe is hosting this script, you can still kill
8    # another Python.exe (as long as there is only one other Python.exe)
9    
10   # Really just a demo for the win32pdh(util) module, which allows you
11   # to get all sorts of information about a running process and many
12   # other aspects of your system.
13   
14   import win32api, win32pdhutil, win32con, sys
15   
16   
17   def killProcName(procname):
18       # Change suggested by Dan Knierim, who found that this performed a
19       # "refresh", allowing us to kill processes created since this was run
20       # for the first time.
21       try:
22           win32pdhutil.GetPerformanceAttributes("Process", "ID Process", procname)
23       except:
24           pass
25   
26       pids = win32pdhutil.FindPerformanceAttributesByName(procname)
27   
28       # If _my_ pid in there, remove it!
29       try:
30           pids.remove(win32api.GetCurrentProcessId())
31       except ValueError:
32           pass
33   
34       if len(pids) == 0:
35           result = "Can't find %s" % procname
36       elif len(pids) > 1:
37           result = "Found too many %s's - pids=`%s`" % (procname, pids)
38       else:
39           handle = win32api.OpenProcess(win32con.PROCESS_TERMINATE, 0, pids[0])
40           win32api.TerminateProcess(handle, 0)
41           win32api.CloseHandle(handle)
42           result = ""
43   
44       return result
45   
46   
47   if __name__ == "__main__":
48       if len(sys.argv) > 1:
49           for procname in sys.argv[1:]:
50               result = killProcName(procname)
51               if result:
52                   print(result)
53                   print("Dumping all processes...")
54                   win32pdhutil.ShowAllProcesses()
55               else:
56                   print("Killed %s" % procname)
57       else:
58           print("Usage: killProcName.py procname ...")
