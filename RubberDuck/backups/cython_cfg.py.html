8    def get_cython_versions(setup_cfg=''):
9        _cython_config = configparser.ConfigParser()
10       if setup_cfg:
11           _cython_config.read(setup_cfg)
12       else:
13           _cython_config.read(
14               join(dirname(__file__), '..', '..', '..', 'setup.cfg'))
15   
16       cython_min = _cython_config['kivy']['cython_min']
17       cython_max = _cython_config['kivy']['cython_max']
18       cython_unsupported = _cython_config['kivy']['cython_exclude'].split(',')
19       # ref https://github.com/cython/cython/issues/1968
20   
21       cython_requires = (
22           'cython>={min_version},<={max_version},{exclusion}'.format(
23               min_version=cython_min,
24               max_version=cython_max,
25               exclusion=','.join('!=%s' % excl for excl in cython_unsupported),
26           )
27       )
28   
29       return cython_requires, cython_min, cython_max, cython_unsupported

32   def get_cython_msg():
33       cython_requires, cython_min, cython_max, cython_unsupported = \
34           get_cython_versions()
35       cython_unsupported_append = ''' 
36    
37       Please note that the following versions of Cython are not supported 
38       at all: {}'''.format(', '.join(map(str, cython_unsupported)))
39   
40       cython_min_msg = textwrap.dedent(''' 
41       This version of Cython is not compatible with Kivy. Please upgrade to 
42       at least version {0}, preferably the newest supported version {1}. 
43    
44       If your platform provides a Cython package, make sure you have upgraded 
45       to the newest version. If the newest version available is still too low, 
46       please remove it and install the newest supported Cython via pip: 
47    
48           pip install -I "{3}"{2} 
49       '''.format(cython_min, cython_max,
50                  cython_unsupported_append if cython_unsupported else '',
51                  cython_requires))
52   
53       cython_max_msg = textwrap.dedent(''' 
54       This version of Cython is untested with Kivy. While this version may 
55       work perfectly fine, it is possible that you may experience issues. 
56       Please downgrade to a supported version, or update cython_max in 
57       setup.cfg to your version of Cython. It is best to use the newest 
58       supported version, {1}, but the minimum supported version is {0}. 
59    
60       If your platform provides a Cython package, check if you can downgrade 
61       to a supported version. Otherwise, uninstall the platform package and 
62       install Cython via pip: 
63    
64           pip install -I "{3}"{2} 
65       '''.format(cython_min, cython_max,
66                  cython_unsupported_append if cython_unsupported else '',
67                  cython_requires))
68   
69       cython_unsupported_msg = textwrap.dedent(''' 
70       This version of Cython suffers from known bugs and is unsupported. 
71       Please install the newest supported version, {1}, if possible, but 
72       the minimum supported version is {0}. 
73    
74       If your platform provides a Cython package, check if you can install 
75       a supported version. Otherwise, uninstall the platform package and 
76       install Cython via pip: 
77    
78           pip install -I "{3}"{2} 
79       '''.format(cython_min, cython_max, cython_unsupported_append,
80                  cython_requires))
81   
82       return cython_min_msg, cython_max_msg, cython_unsupported_msg
