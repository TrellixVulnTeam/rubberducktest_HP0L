1212         """Parse keywords in lex.h."""
1213 
1214         results = set()
1215         for m in re.finditer(r'{SYM(?:_HK)?\("(?P<keyword>[a-z0-9_]+)",', f, flags=re.I):
1216             results.add(m.group('keyword').lower())
1217 
1218         if not results:
1219             raise ValueError('No keywords found')
