11   import re
12   
13   from pygments.lexer import RegexLexer, include, default
14   from pygments.token import Text, Comment, Operator, Keyword, Name, String, \
15       Number, Punctuation, Error
16   
17   __all__ = ['NimrodLexer']
18   
19   
20   class NimrodLexer(RegexLexer):
21       """ 
22       For `Nim <http://nim-lang.org/>`_ source code. 
23    
24       .. versionadded:: 1.5 
25       """
26   
27       name = 'Nimrod'
28       aliases = ['nimrod', 'nim']
29       filenames = ['*.nim', '*.nimrod']
30       mimetypes = ['text/x-nim']
31   
32       flags = re.MULTILINE | re.IGNORECASE | re.UNICODE
33   
34       def underscorize(words):
35           newWords = []
36           new = ""
37           for word in words:
38               for ch in word:
39                   new += (ch + "_?")
40               newWords.append(new)
41               new = ""
42           return "|".join(newWords)
43   
44       keywords = [
45           'addr', 'and', 'as', 'asm', 'bind', 'block', 'break', 'case',
46           'cast', 'concept', 'const', 'continue', 'converter', 'defer', 'discard',
47           'distinct', 'div', 'do', 'elif', 'else', 'end', 'enum', 'except',
48           'export', 'finally', 'for', 'func', 'if', 'in', 'yield', 'interface',
49           'is', 'isnot', 'iterator', 'let', 'macro', 'method', 'mixin', 'mod',
50           'not', 'notin', 'object', 'of', 'or', 'out', 'proc', 'ptr', 'raise',
51           'ref', 'return', 'shl', 'shr', 'static', 'template', 'try',
52           'tuple', 'type', 'using', 'when', 'while', 'xor'
53       ]
54   
55       keywordsPseudo = [
56           'nil', 'true', 'false'
57       ]
58   
59       opWords = [
60           'and', 'or', 'not', 'xor', 'shl', 'shr', 'div', 'mod', 'in',
61           'notin', 'is', 'isnot'
62       ]
63   
64       types = [
65           'int', 'int8', 'int16', 'int32', 'int64', 'float', 'float32', 'float64',
66           'bool', 'char', 'range', 'array', 'seq', 'set', 'string'
67       ]
68   
69       tokens = {
70           'root': [
71               (r'##.*$', String.Doc),
72               (r'#.*$', Comment),
73               (r'[*=><+\-/@$~&%!?|\\\[\]]', Operator),
74               (r'\.\.|\.|,|\[\.|\.\]|\{\.|\.\}|\(\.|\.\)|\{|\}|\(|\)|:|\^|`|;',
75                Punctuation),
76   
77               # Strings
78               (r'(?:[\w]+)"', String, 'rdqs'),
79               (r'"""', String, 'tdqs'),
80               ('"', String, 'dqs'),
81   
82               # Char
83               ("'", String.Char, 'chars'),
84   
85               # Keywords
86               (r'(%s)\b' % underscorize(opWords), Operator.Word),
87               (r'(p_?r_?o_?c_?\s)(?![(\[\]])', Keyword, 'funcname'),
88               (r'(%s)\b' % underscorize(keywords), Keyword),
89               (r'(%s)\b' % underscorize(['from', 'import', 'include']),
90                Keyword.Namespace),
91               (r'(v_?a_?r)\b', Keyword.Declaration),
92               (r'(%s)\b' % underscorize(types), Keyword.Type),
93               (r'(%s)\b' % underscorize(keywordsPseudo), Keyword.Pseudo),
94               # Identifiers
95               (r'\b((?![_\d])\w)(((?!_)\w)|(_(?!_)\w))*', Name),
96               # Numbers
97               (r'[0-9][0-9_]*(?=([e.]|\'f(32|64)))',
98                Number.Float, ('float-suffix', 'float-number')),
99               (r'0x[a-f0-9][a-f0-9_]*', Number.Hex, 'int-suffix'),
100              (r'0b[01][01_]*', Number.Bin, 'int-suffix'),
101              (r'0o[0-7][0-7_]*', Number.Oct, 'int-suffix'),
102              (r'[0-9][0-9_]*', Number.Integer, 'int-suffix'),
103              # Whitespace
104              (r'\s+', Text),
105              (r'.+$', Error),
106          ],
107          'chars': [
108              (r'\\([\\abcefnrtvl"\']|x[a-f0-9]{2}|[0-9]{1,3})', String.Escape),
109              (r"'", String.Char, '#pop'),
110              (r".", String.Char)
111          ],
112          'strings': [
113              (r'(?<!\$)\$(\d+|#|\w+)+', String.Interpol),
114              (r'[^\\\'"$\n]+', String),
115              # quotes, dollars and backslashes must be parsed one at a time
116              (r'[\'"\\]', String),
117              # unhandled string formatting sign
118              (r'\$', String)
119              # newlines are an error (use "nl" state)
120          ],
121          'dqs': [
122              (r'\\([\\abcefnrtvl"\']|\n|x[a-f0-9]{2}|[0-9]{1,3})',
123               String.Escape),
124              (r'"', String, '#pop'),
125              include('strings')
126          ],
127          'rdqs': [
128              (r'"(?!")', String, '#pop'),
129              (r'""', String.Escape),
130              include('strings')
131          ],
132          'tdqs': [
133              (r'"""(?!")', String, '#pop'),
134              include('strings'),
135              include('nl')
136          ],
137          'funcname': [
138              (r'((?![\d_])\w)(((?!_)\w)|(_(?!_)\w))*', Name.Function, '#pop'),
139              (r'`.+`', Name.Function, '#pop')
140          ],
141          'nl': [
142              (r'\n', String)
143          ],
144          'float-number': [
145              (r'\.(?!\.)[0-9_]*', Number.Float),
146              (r'e[+-]?[0-9][0-9_]*', Number.Float),
147              default('#pop')
148          ],
149          'float-suffix': [
150              (r'\'f(32|64)', Number.Float),
151              default('#pop')
152          ],
153          'int-suffix': [
154              (r'\'i(32|64)', Number.Integer.Long),
155              (r'\'i(8|16)', Number.Integer),
156              default('#pop')
157          ],
158      }
