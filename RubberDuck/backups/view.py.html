213  class MarkerMapLayer(MapLayer):
214      """A map layer for :class:`MapMarker` 
215      """
216  
217      order_marker_by_latitude = BooleanProperty(True)
218  
219      def __init__(self, **kwargs):
220          self.markers = []
221          super().__init__(**kwargs)
222  
223      def insert_marker(self, marker, **kwargs):
224          if self.order_marker_by_latitude:
225              before = list(
226                  takewhile(lambda i_m: i_m[1].lat < marker.lat, enumerate(self.children))
227              )
228              if before:
229                  kwargs['index'] = before[-1][0] + 1
230  
231          super().add_widget(marker, **kwargs)
232  
233      def add_widget(self, marker):
234          marker._layer = self
235          self.markers.append(marker)
236          self.insert_marker(marker)
237  
238      def remove_widget(self, marker):
239          marker._layer = None
240          if marker in self.markers:
241              self.markers.remove(marker)
242          super().remove_widget(marker)
243  
244      def reposition(self):
245          if not self.markers:
246              return
247          mapview = self.parent
248          set_marker_position = self.set_marker_position
249          bbox = None
250          # reposition the markers depending the latitude
251          markers = sorted(self.markers, key=lambda x: -x.lat)
252          margin = max((max(marker.size) for marker in markers))
253          bbox = mapview.get_bbox(margin)
254          for marker in markers:
255              if bbox.collide(marker.lat, marker.lon):
256                  set_marker_position(mapview, marker)
257                  if not marker.parent:
258                      self.insert_marker(marker)
259              else:
260                  super().remove_widget(marker)
261  
262      def set_marker_position(self, mapview, marker):
263          x, y = mapview.get_window_xy_from(marker.lat, marker.lon, mapview.zoom)
264          marker.x = int(x - marker.width * marker.anchor_x)
265          marker.y = int(y - marker.height * marker.anchor_y)
266  
267      def unload(self):
268          self.clear_widgets()
269          del self.markers[:]

282  class MapView(Widget):
283      """MapView is the widget that control the map displaying, navigation, and 
284      layers management. 
285      """
286  
287      lon = NumericProperty()
288      """Longitude at the center of the widget 
289      """
290  
291      lat = NumericProperty()
292      """Latitude at the center of the widget 
293      """
294  
295      zoom = NumericProperty(0)
296      """Zoom of the widget. Must be between :meth:`MapSource.get_min_zoom` and 
297      :meth:`MapSource.get_max_zoom`. Default to 0. 
298      """
299  
300      map_source = ObjectProperty(MapSource())
301      """Provider of the map, default to a empty :class:`MapSource`. 
302      """
303  
304      double_tap_zoom = BooleanProperty(False)
305      """If True, this will activate the double-tap to zoom. 
306      """
307  
308      pause_on_action = BooleanProperty(True)
309      """Pause any map loading / tiles loading when an action is done. 
310      This allow better performance on mobile, but can be safely deactivated on 
311      desktop. 
312      """
313  
314      snap_to_zoom = BooleanProperty(True)
315      """When the user initiate a zoom, it will snap to the closest zoom for 
316      better graphics. The map can be blur if the map is scaled between 2 zoom. 
317      Default to True, even if it doesn't fully working yet. 
318      """
319  
320      animation_duration = NumericProperty(100)
321      """Duration to animate Tiles alpha from 0 to 1 when it's ready to show. 
322      Default to 100 as 100ms. Use 0 to deactivate. 
323      """
324  
325      delta_x = NumericProperty(0)
326      delta_y = NumericProperty(0)
327      background_color = ListProperty([181 / 255.0, 208 / 255.0, 208 / 255.0, 1])
328      cache_dir = StringProperty(CACHE_DIR)
329      _zoom = NumericProperty(0)
330      _pause = BooleanProperty(False)
331      _scale = 1.0
332      _disabled_count = 0
333  
334      __events__ = ["on_map_relocated"]
335  
336      # Public API
337  
338      @property
339      def viewport_pos(self):
340          vx, vy = self._scatter.to_local(self.x, self.y)
341          return vx - self.delta_x, vy - self.delta_y
342  
343      @property
344      def scale(self):
345          if self._invalid_scale:
346              self._invalid_scale = False
347              self._scale = self._scatter.scale
348          return self._scale
349  
350      def get_bbox(self, margin=0):
351          """Returns the bounding box from the bottom/left (lat1, lon1) to 
352          top/right (lat2, lon2). 
353          """
354          x1, y1 = self.to_local(0 - margin, 0 - margin)
355          x2, y2 = self.to_local((self.width + margin), (self.height + margin))
356          c1 = self.get_latlon_at(x1, y1)
357          c2 = self.get_latlon_at(x2, y2)
358          return Bbox((c1.lat, c1.lon, c2.lat, c2.lon))
359  
360      bbox = AliasProperty(get_bbox, None, bind=["lat", "lon", "_zoom"])
361  
362      def unload(self):
363          """Unload the view and all the layers. 
364          It also cancel all the remaining downloads. 
365          """
366          self.remove_all_tiles()
367  
368      def get_window_xy_from(self, lat, lon, zoom):
369          """Returns the x/y position in the widget absolute coordinates 
370          from a lat/lon"""
371          scale = self.scale
372          vx, vy = self.viewport_pos
373          ms = self.map_source
374          x = ms.get_x(zoom, lon) - vx
375          y = ms.get_y(zoom, lat) - vy
376          x *= scale
377          y *= scale
378          x = x + self.pos[0]
379          y = y + self.pos[1]
380          return x, y
381  
382      def center_on(self, *args):
383          """Center the map on the coordinate :class:`Coordinate`, or a (lat, lon) 
384          """
385          map_source = self.map_source
386          zoom = self._zoom
387  
388          if len(args) == 1 and isinstance(args[0], Coordinate):
389              coord = args[0]
390              lat = coord.lat
391              lon = coord.lon
392          elif len(args) == 2:
393              lat, lon = args
394          else:
395              raise Exception("Invalid argument for center_on")
396          lon = clamp(lon, MIN_LONGITUDE, MAX_LONGITUDE)
397          lat = clamp(lat, MIN_LATITUDE, MAX_LATITUDE)
398          scale = self._scatter.scale
399          x = map_source.get_x(zoom, lon) - self.center_x / scale
400          y = map_source.get_y(zoom, lat) - self.center_y / scale
401          self.delta_x = -x
402          self.delta_y = -y
403          self.lon = lon
404          self.lat = lat
405          self._scatter.pos = 0, 0
406          self.trigger_update(True)
407  
408      def set_zoom_at(self, zoom, x, y, scale=None):
409          """Sets the zoom level, leaving the (x, y) at the exact same point 
410          in the view. 
411          """
412          zoom = clamp(
413              zoom, self.map_source.get_min_zoom(), self.map_source.get_max_zoom()
414          )
415          if int(zoom) == int(self._zoom):
416              if scale is None:
417                  return
418              elif scale == self.scale:
419                  return
420          scale = scale or 1.0
421  
422          # first, rescale the scatter
423          scatter = self._scatter
424          scale = clamp(scale, scatter.scale_min, scatter.scale_max)
425          rescale = scale * 1.0 / scatter.scale
426          scatter.apply_transform(
427              Matrix().scale(rescale, rescale, rescale),
428              post_multiply=True,
429              anchor=scatter.to_local(x, y),
430          )
431  
432          # adjust position if the zoom changed
433          c1 = self.map_source.get_col_count(self._zoom)
434          c2 = self.map_source.get_col_count(zoom)
435          if c1 != c2:
436              f = float(c2) / float(c1)
437              self.delta_x = scatter.x + self.delta_x * f
438              self.delta_y = scatter.y + self.delta_y * f
439              # back to 0 every time
440              scatter.apply_transform(
441                  Matrix().translate(-scatter.x, -scatter.y, 0), post_multiply=True
442              )
443  
444          # avoid triggering zoom changes.
445          self._zoom = zoom
446          self.zoom = self._zoom
447  
448      def on_zoom(self, instance, zoom):
449          if zoom == self._zoom:
450              return
451          x = self.map_source.get_x(zoom, self.lon) - self.delta_x
452          y = self.map_source.get_y(zoom, self.lat) - self.delta_y
453          self.set_zoom_at(zoom, x, y)
454          self.center_on(self.lat, self.lon)
455  
456      def get_latlon_at(self, x, y, zoom=None):
457          """Return the current :class:`Coordinate` within the (x, y) widget 
458          coordinate. 
459          """
460          if zoom is None:
461              zoom = self._zoom
462          vx, vy = self.viewport_pos
463          scale = self._scale
464          return Coordinate(
465              lat=self.map_source.get_lat(zoom, y / scale + vy),
466              lon=self.map_source.get_lon(zoom, x / scale + vx),
467          )

469      def add_marker(self, marker, layer=None):
470          """Add a marker into the layer. If layer is None, it will be added in 
471          the default marker layer. If there is no default marker layer, a new 
472          one will be automatically created 
473          """
474          if layer is None:
475              if not self._default_marker_layer:
476                  layer = MarkerMapLayer()
477                  self.add_layer(layer)
478              else:
479                  layer = self._default_marker_layer
480          layer.add_widget(marker)
481          layer.set_marker_position(self, marker)
482  
483      def remove_marker(self, marker):
484          """Remove a marker from its layer 
485          """
486          marker.detach()

488      def add_layer(self, layer, mode="window"):
489          """Add a new layer to update at the same time the base tile layer. 
490          mode can be either "scatter" or "window". If "scatter", it means the 
491          layer will be within the scatter transformation. It's perfect if you 
492          want to display path / shape, but not for text. 
493          If "window", it will have no transformation. You need to position the 
494          widget yourself: think as Z-sprite / billboard. 
495          Defaults to "window". 
496          """
497          assert mode in ("scatter", "window")
498          if self._default_marker_layer is None and isinstance(layer, MarkerMapLayer):
499              self._default_marker_layer = layer
500          self._layers.append(layer)
501          c = self.canvas
502          if mode == "scatter":
503              self.canvas = self.canvas_layers
504          else:
505              self.canvas = self.canvas_layers_out
506          layer.canvas_parent = self.canvas
507          super().add_widget(layer)
508          self.canvas = c
509  
510      def remove_layer(self, layer):
511          """Remove the layer 
512          """
513          c = self.canvas
514          self._layers.remove(layer)
515          self.canvas = layer.canvas_parent
516          super().remove_widget(layer)
517          self.canvas = c

519      def sync_to(self, other):
520          """Reflect the lat/lon/zoom of the other MapView to the current one. 
521          """
522          if self._zoom != other._zoom:
523              self.set_zoom_at(other._zoom, *self.center)
524          self.center_on(other.get_latlon_at(*self.center))

558      def _animate_color(self, dt):
559          # fast path
560          d = self.animation_duration
561          if d == 0:
562              for tile in self._tiles:
563                  if tile.state == "need-animation":
564                      tile.g_color.a = 1.0
565                      tile.state = "animated"
566              for tile in self._tiles_bg:
567                  if tile.state == "need-animation":
568                      tile.g_color.a = 1.0
569                      tile.state = "animated"
570          else:
571              d = d / 1000.0
572              for tile in self._tiles:
573                  if tile.state != "need-animation":
574                      continue
575                  tile.g_color.a += dt / d
576                  if tile.g_color.a >= 1:
577                      tile.state = "animated"
578              for tile in self._tiles_bg:
579                  if tile.state != "need-animation":
580                      continue
581                  tile.g_color.a += dt / d
582                  if tile.g_color.a >= 1:
583                      tile.state = "animated"

604      def animated_diff_scale_at(self, d, x, y):
605          self._scale_target_time = 1.0
606          self._scale_target_pos = x, y
607          if self._scale_target_anim is False:
608              self._scale_target_anim = True
609              self._scale_target = d
610          else:
611              self._scale_target += d
612          Clock.unschedule(self._animate_scale)
613          Clock.schedule_interval(self._animate_scale, 1 / 60.0)
614  
615      def _animate_scale(self, dt):
616          diff = self._scale_target / 3.0
617          if abs(diff) < 0.01:
618              diff = self._scale_target
619              self._scale_target = 0
620          else:
621              self._scale_target -= diff
622          self._scale_target_time -= dt
623          self.diff_scale_at(diff, *self._scale_target_pos)
624          ret = self._scale_target != 0
625          if not ret:
626              self._pause = False
627          return ret
628  
629      def diff_scale_at(self, d, x, y):
630          scatter = self._scatter
631          scale = scatter.scale * (2 ** d)
632          self.scale_at(scale, x, y)
633  
634      def scale_at(self, scale, x, y):
635          scatter = self._scatter
636          scale = clamp(scale, scatter.scale_min, scatter.scale_max)
637          rescale = scale * 1.0 / scatter.scale
638          scatter.apply_transform(
639              Matrix().scale(rescale, rescale, rescale),
640              post_multiply=True,
641              anchor=scatter.to_local(x, y),
642          )

644      def on_touch_down(self, touch):
645          if not self.collide_point(*touch.pos):
646              return
647          if self.pause_on_action:
648              self._pause = True
649          if "button" in touch.profile and touch.button in ("scrolldown", "scrollup"):
650              d = 1 if touch.button == "scrollup" else -1
651              self.animated_diff_scale_at(d, *touch.pos)
652              return True
653          elif touch.is_double_tap and self.double_tap_zoom:
654              self.animated_diff_scale_at(1, *touch.pos)
655              return True
656          touch.grab(self)
657          self._touch_count += 1
658          if self._touch_count == 1:
659              self._touch_zoom = (self.zoom, self._scale)
660          return super().on_touch_down(touch)
661  
662      def on_touch_up(self, touch):
663          if touch.grab_current == self:
664              touch.ungrab(self)
665              self._touch_count -= 1
666              if self._touch_count == 0:
667                  # animate to the closest zoom
668                  zoom, scale = self._touch_zoom
669                  cur_zoom = self.zoom
670                  cur_scale = self._scale
671                  if cur_zoom < zoom or cur_scale < scale:
672                      self.animated_diff_scale_at(1.0 - cur_scale, *touch.pos)
673                  elif cur_zoom > zoom or cur_scale > scale:
674                      self.animated_diff_scale_at(2.0 - cur_scale, *touch.pos)
675                  self._pause = False
676              return True
677          return super().on_touch_up(touch)
678  
679      def on_transform(self, *args):
680          self._invalid_scale = True
681          if self._transform_lock:
682              return
683          self._transform_lock = True
684          # recalculate viewport
685          map_source = self.map_source
686          zoom = self._zoom
687          scatter = self._scatter
688          scale = scatter.scale
689          if scale >= 2.0:
690              zoom += 1
691              scale /= 2.0
692          elif scale < 1:
693              zoom -= 1
694              scale *= 2.0
695          zoom = clamp(zoom, map_source.min_zoom, map_source.max_zoom)
696          if zoom != self._zoom:
697              self.set_zoom_at(zoom, scatter.x, scatter.y, scale=scale)
698              self.trigger_update(True)
699          else:
700              if zoom == map_source.min_zoom and scatter.scale < 1.0:
701                  scatter.scale = 1.0
702                  self.trigger_update(True)
703              else:
704                  self.trigger_update(False)
705  
706          if map_source.bounds:
707              self._apply_bounds()
708          self._transform_lock = False
709          self._scale = self._scatter.scale
710  
711      def _apply_bounds(self):
712          # if the map_source have any constraints, apply them here.
713          map_source = self.map_source
714          zoom = self._zoom
715          min_lon, min_lat, max_lon, max_lat = map_source.bounds
716          xmin = map_source.get_x(zoom, min_lon)
717          xmax = map_source.get_x(zoom, max_lon)
718          ymin = map_source.get_y(zoom, min_lat)
719          ymax = map_source.get_y(zoom, max_lat)
720  
721          dx = self.delta_x
722          dy = self.delta_y
723          oxmin, oymin = self._scatter.to_local(self.x, self.y)
724          oxmax, oymax = self._scatter.to_local(self.right, self.top)
725          s = self._scale
726          cxmin = oxmin - dx
727          if cxmin < xmin:
728              self._scatter.x += (cxmin - xmin) * s
729          cymin = oymin - dy
730          if cymin < ymin:
731              self._scatter.y += (cymin - ymin) * s
732          cxmax = oxmax - dx
733          if cxmax > xmax:
734              self._scatter.x -= (xmax - cxmax) * s
735          cymax = oymax - dy
736          if cymax > ymax:
737              self._scatter.y -= (ymax - cymax) * s
738  
739      def on__pause(self, instance, value):
740          if not value:
741              self.trigger_update(True)
742  
743      def trigger_update(self, full):
744          self._need_redraw_full = full or self._need_redraw_full
745          Clock.unschedule(self.do_update)
746          Clock.schedule_once(self.do_update, -1)

748      def do_update(self, dt):
749          zoom = self._zoom
750          scale = self._scale
751          self.lon = self.map_source.get_lon(
752              zoom, (self.center_x - self._scatter.x) / scale - self.delta_x
753          )
754          self.lat = self.map_source.get_lat(
755              zoom, (self.center_y - self._scatter.y) / scale - self.delta_y
756          )
757          self.dispatch("on_map_relocated", zoom, Coordinate(self.lon, self.lat))
758          for layer in self._layers:
759              layer.reposition()
760  
761          if self._need_redraw_full:
762              self._need_redraw_full = False
763              self.move_tiles_to_background()
764              self.load_visible_tiles()
765          else:
766              self.load_visible_tiles()
767  
768      def bbox_for_zoom(self, vx, vy, w, h, zoom):
769          # return a tile-bbox for the zoom
770          map_source = self.map_source
771          size = map_source.dp_tile_size
772          scale = self._scale
773  
774          max_x_end = map_source.get_col_count(zoom)
775          max_y_end = map_source.get_row_count(zoom)
776  
777          x_count = int(ceil(w / scale / float(size))) + 1
778          y_count = int(ceil(h / scale / float(size))) + 1
779  
780          tile_x_first = int(clamp(vx / float(size), 0, max_x_end))
781          tile_y_first = int(clamp(vy / float(size), 0, max_y_end))
782          tile_x_last = tile_x_first + x_count
783          tile_y_last = tile_y_first + y_count
784          tile_x_last = int(clamp(tile_x_last, tile_x_first, max_x_end))
785          tile_y_last = int(clamp(tile_y_last, tile_y_first, max_y_end))
786  
787          x_count = tile_x_last - tile_x_first
788          y_count = tile_y_last - tile_y_first
789          return (tile_x_first, tile_y_first, tile_x_last, tile_y_last, x_count, y_count)

791      def load_visible_tiles(self):
792          map_source = self.map_source
793          vx, vy = self.viewport_pos
794          zoom = self._zoom
795          dirs = [0, 1, 0, -1, 0]
796          bbox_for_zoom = self.bbox_for_zoom
797          size = map_source.dp_tile_size
798  
799          (
800              tile_x_first,
801              tile_y_first,
802              tile_x_last,
803              tile_y_last,
804              x_count,
805              y_count,
806          ) = bbox_for_zoom(vx, vy, self.width, self.height, zoom)
807  
808          # Adjust tiles behind us
809          for tile in self._tiles_bg[:]:
810              tile_x = tile.tile_x
811              tile_y = tile.tile_y
812  
813              f = 2 ** (zoom - tile.zoom)
814              w = self.width / f
815              h = self.height / f
816              (
817                  btile_x_first,
818                  btile_y_first,
819                  btile_x_last,
820                  btile_y_last,
821                  _,
822                  _,
823              ) = bbox_for_zoom(vx / f, vy / f, w, h, tile.zoom)
824  
825              if (
826                  tile_x < btile_x_first
827                  or tile_x >= btile_x_last
828                  or tile_y < btile_y_first
829                  or tile_y >= btile_y_last
830              ):
831                  tile.state = "done"
832                  self._tiles_bg.remove(tile)
833                  self.canvas_map.before.remove(tile.g_color)
834                  self.canvas_map.before.remove(tile)
835                  continue
836  
837              tsize = size * f
838              tile.size = tsize, tsize
839              tile.pos = (tile_x * tsize + self.delta_x, tile_y * tsize + self.delta_y)
840  
841          # Get rid of old tiles first
842          for tile in self._tiles[:]:
843              tile_x = tile.tile_x
844              tile_y = tile.tile_y
845  
846              if (
847                  tile_x < tile_x_first
848                  or tile_x >= tile_x_last
849                  or tile_y < tile_y_first
850                  or tile_y >= tile_y_last
851              ):
852                  tile.state = "done"
853                  self.tile_map_set(tile_x, tile_y, False)
854                  self._tiles.remove(tile)
855                  self.canvas_map.remove(tile)
856                  self.canvas_map.remove(tile.g_color)
857              else:
858                  tile.size = (size, size)
859                  tile.pos = (tile_x * size + self.delta_x, tile_y * size + self.delta_y)
860  
861          # Load new tiles if needed
862          x = tile_x_first + x_count // 2 - 1
863          y = tile_y_first + y_count // 2 - 1
864          arm_max = max(x_count, y_count) + 2
865          arm_size = 1
866          turn = 0
867          while arm_size < arm_max:
868              for i in range(arm_size):
869                  if (
870                      not self.tile_in_tile_map(x, y)
871                      and y >= tile_y_first
872                      and y < tile_y_last
873                      and x >= tile_x_first
874                      and x < tile_x_last
875                  ):
876                      self.load_tile(x, y, size, zoom)
877  
878                  x += dirs[turn % 4 + 1]
879                  y += dirs[turn % 4]
880  
881              if turn % 2 == 1:
882                  arm_size += 1
883  
884              turn += 1
885  
886      def load_tile(self, x, y, size, zoom):
887          if self.tile_in_tile_map(x, y) or zoom != self._zoom:
888              return
889          self.load_tile_for_source(self.map_source, 1.0, size, x, y, zoom)
890          # XXX do overlay support
891          self.tile_map_set(x, y, True)
892  
893      def load_tile_for_source(self, map_source, opacity, size, x, y, zoom):
894          tile = Tile(size=(size, size), cache_dir=self.cache_dir)
895          tile.g_color = Color(1, 1, 1, 0)
896          tile.tile_x = x
897          tile.tile_y = y
898          tile.zoom = zoom
899          tile.pos = (x * size + self.delta_x, y * size + self.delta_y)
900          tile.map_source = map_source
901          tile.state = "loading"
902          if not self._pause:
903              map_source.fill_tile(tile)
904          self.canvas_map.add(tile.g_color)
905          self.canvas_map.add(tile)
906          self._tiles.append(tile)

908      def move_tiles_to_background(self):
909          # remove all the tiles of the main map to the background map
910          # retain only the one who are on the current zoom level
911          # for all the tile in the background, stop the download if not yet started.
912          zoom = self._zoom
913          tiles = self._tiles
914          btiles = self._tiles_bg
915          canvas_map = self.canvas_map
916          tile_size = self.map_source.tile_size
917  
918          # move all tiles to background
919          while tiles:
920              tile = tiles.pop()
921              if tile.state == "loading":
922                  tile.state = "done"
923                  continue
924              btiles.append(tile)
925  
926          # clear the canvas
927          canvas_map.clear()
928          canvas_map.before.clear()
929          self._tilemap = {}
930  
931          # unsure if it's really needed, i personnally didn't get issues right now
932          # btiles.sort(key=lambda z: -z.zoom)
933  
934          # add all the btiles into the back canvas.
935          # except for the tiles that are owned by the current zoom level
936          for tile in btiles[:]:
937              if tile.zoom == zoom:
938                  btiles.remove(tile)
939                  tiles.append(tile)
940                  tile.size = tile_size, tile_size
941                  canvas_map.add(tile.g_color)
942                  canvas_map.add(tile)
943                  self.tile_map_set(tile.tile_x, tile.tile_y, True)
944                  continue
945              canvas_map.before.add(tile.g_color)
946              canvas_map.before.add(tile)
947  
948      def remove_all_tiles(self):
949          # clear the map of all tiles.
950          self.canvas_map.clear()
951          self.canvas_map.before.clear()
952          for tile in self._tiles:
953              tile.state = "done"
954          del self._tiles[:]
955          del self._tiles_bg[:]
956          self._tilemap = {}

958      def tile_map_set(self, tile_x, tile_y, value):
959          key = tile_y * self.map_source.get_col_count(self._zoom) + tile_x
960          if value:
961              self._tilemap[key] = value
962          else:
963              self._tilemap.pop(key, None)

965      def tile_in_tile_map(self, tile_x, tile_y):
966          key = tile_y * self.map_source.get_col_count(self._zoom) + tile_x
967          return key in self._tilemap

979      def on_map_source(self, instance, source):
980          if isinstance(source, string_types):
981              self.map_source = MapSource.from_provider(source)
982          elif isinstance(source, (tuple, list)):
983              cache_key, min_zoom, max_zoom, url, attribution, options = source
984              self.map_source = MapSource(
985                  url=url,
986                  cache_key=cache_key,
987                  min_zoom=min_zoom,
988                  max_zoom=max_zoom,
989                  attribution=attribution,
990                  cache_dir=self.cache_dir,
991                  **options
992              )
993          elif isinstance(source, MapSource):
994              self.map_source = source
995          else:
996              raise Exception("Invalid map source provider")
997          self.zoom = clamp(self.zoom, self.map_source.min_zoom, self.map_source.max_zoom)
998          self.remove_all_tiles()
999          self.trigger_update(True)
