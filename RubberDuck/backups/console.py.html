18   class VCTreeStatusLexer(RegexLexer):
19       """ 
20       For colorizing output of version control status commands, like "hg 
21       status" or "svn status". 
22    
23       .. versionadded:: 2.0 
24       """
25       name = 'VCTreeStatus'
26       aliases = ['vctreestatus']
27       filenames = []
28       mimetypes = []
29   
30       tokens = {
31           'root': [
32               (r'^A  \+  C\s+', Generic.Error),
33               (r'^A\s+\+?\s+', String),
34               (r'^M\s+', Generic.Inserted),
35               (r'^C\s+', Generic.Error),
36               (r'^D\s+', Generic.Deleted),
37               (r'^[?!]\s+', Comment.Preproc),
38               (r'      >\s+.*\n', Comment.Preproc),
39               (r'.*\n', Text)
40           ]
41       }

44   class PyPyLogLexer(RegexLexer):
45       """ 
46       Lexer for PyPy log files. 
47    
48       .. versionadded:: 1.5 
49       """
50       name = "PyPy Log"
51       aliases = ["pypylog", "pypy"]
52       filenames = ["*.pypylog"]
53       mimetypes = ['application/x-pypylog']

55       tokens = {
56           "root": [

58               (r"\[\w+\] \{jit-backend-counts$", Keyword, "jit-backend-counts"),
59               include("extra-stuff"),
60           ],
61           "jit-log": [
62               (r"\[\w+\] jit-log-.*?}$", Keyword, "#pop"),

65               (r"[ifp]\d+", Name),
66               (r"ptr\d+", Name),

68                bygroups(Punctuation, Name.Builtin, Punctuation)),
69               (r"[\[\]=,()]", Punctuation),
70               (r"(\d+\.\d+|inf|-inf)", Number.Float),
71               (r"-?\d+", Number.Integer),
72               (r"'.*'", String),
73               (r"(None|descr|ConstClass|ConstPtr|TargetToken)", Name),
74               (r"<.*?>+", Name.Builtin),
75               (r"(label|debug_merge_point|jump|finish)", Name.Class),
76               (r"(int_add_ovf|int_add|int_sub_ovf|int_sub|int_mul_ovf|int_mul|"
77                r"int_floordiv|int_mod|int_lshift|int_rshift|int_and|int_or|"
78                r"int_xor|int_eq|int_ne|int_ge|int_gt|int_le|int_lt|int_is_zero|"

84                r"cast_int_to_float|cast_float_to_int|"
85                r"force_token|quasiimmut_field|same_as|virtual_ref_finish|"
86                r"virtual_ref|mark_opaque_ptr|"
87                r"call_may_force|call_assembler|call_loopinvariant|"
88                r"call_release_gil|call_pure|call|"
89                r"new_with_vtable|new_array|newstr|newunicode|new|"
90                r"arraylen_gc|"
91                r"getarrayitem_gc_pure|getarrayitem_gc|setarrayitem_gc|"
92                r"getarrayitem_raw|setarrayitem_raw|getfield_gc_pure|"

97                r"guard_true|guard_false|guard_value|guard_isnull|"
98                r"guard_nonnull_class|guard_nonnull|guard_class|guard_no_overflow|"
99                r"guard_not_forced|guard_no_exception|guard_not_invalidated)",
100               Name.Builtin),
101              include("extra-stuff"),
102          ],
103          "jit-backend-counts": [
104              (r"\[\w+\] jit-backend-counts}$", Keyword, "#pop"),
105              (r":", Punctuation),

107              include("extra-stuff"),
108          ],
