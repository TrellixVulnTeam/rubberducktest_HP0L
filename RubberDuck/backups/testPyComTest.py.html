50   def ensure_long(int_val):
51       if sys.version_info > (3,):
52           # py3k - no such thing as a 'long'
53           return int_val
54       # on py2x, we just use an expression that results in a long
55       return 0x100000000 - 0x100000000 + int_val

106  # Simple handler class.  This demo only fires one event.
107  class RandomEventHandler:
108      def _Init(self):
109          self.fireds = {}
110  
111      def OnFire(self, no):
112          try:
113              self.fireds[no] = self.fireds[no] + 1
114          except KeyError:
115              self.fireds[no] = 0
116  
117      def OnFireWithNamedParams(self, no, a_bool, out1, out2):
118          # This test exists mainly to help with an old bug, where named
119          # params would come in reverse.
120          Missing = pythoncom.Missing
121          if no is not Missing:
122              # We know our impl called 'OnFire' with the same ID
123              assert no in self.fireds
124              assert no + 1 == out1, "expecting 'out1' param to be ID+1"
125              assert no + 2 == out2, "expecting 'out2' param to be ID+2"
126          # The middle must be a boolean.
127          assert a_bool is Missing or type(a_bool) == bool, "middle param not a bool"
128          return out1 + 2, out2 + 2
129  
130      def _DumpFireds(self):
131          if not self.fireds:
132              print("ERROR: Nothing was received!")
133          for firedId, no in self.fireds.items():
134              progress("ID %d fired %d times" % (firedId, no))

556      try:
557          for i in range(3):
558              session = o.Start()
559              sessions.append(session)

562          # Stop the servers

569  def _TestPyVariant(o, is_generated, val, checker=None):
570      if is_generated:
571          vt, got = o.GetVariantAndType(val)
572      else:
573          # Gotta supply all 3 args with the last 2 being explicit variants to
574          # get the byref behaviour.
575          var_vt = VARIANT(pythoncom.VT_UI2 | pythoncom.VT_BYREF, 0)
576          var_result = VARIANT(pythoncom.VT_VARIANT | pythoncom.VT_BYREF, 0)
577          o.GetVariantAndType(val, var_vt, var_result)
578          vt = var_vt.value
579          got = var_result.value
580      if checker is not None:
581          checker(got)
582          return
583      # default checking.
584      assert vt == val.varianttype, (vt, val.varianttype)
585      # Handle our safe-array test - if the passed value is a list of variants,
586      # compare against the actual values.
587      if type(val.value) in (tuple, list):
588          check = [v.value if isinstance(v, VARIANT) else v for v in val.value]
589          # pythoncom always returns arrays as tuples.
590          got = list(got)
591      else:
592          check = val.value
593      assert type(check) == type(got), (type(check), type(got))
594      assert check == got, (check, got)
