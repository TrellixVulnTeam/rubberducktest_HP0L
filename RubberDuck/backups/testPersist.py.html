30           "ReadAt",

32           "Flush",
33           "SetSize",
34           "LockRegion",
35           "UnlockRegion",
36           "Stat",

41           self.data = str2bytes(data)
42           self.ctime = now
43           self.mtime = now
44           self.atime = now

46       def ReadAt(self, offset, cb):
47           print("ReadAt")
48           result = self.data[offset : offset + cb]
49           return result

51       def WriteAt(self, offset, data):
52           print("WriteAt " + str(offset))
53           print("len " + str(len(data)))
54           print("data:")
55           # print data
56           if len(self.data) >= offset:
57               newdata = self.data[0:offset] + data
58           print(len(newdata))
59           if len(self.data) >= offset + len(data):
60               newdata = newdata + self.data[offset + len(data) :]
61           print(len(newdata))
62           self.data = newdata
63           return len(data)

65       def Flush(self, whatsthis=0):
66           print("Flush" + str(whatsthis))
67           fname = os.path.join(win32api.GetTempPath(), "persist.doc")
68           open(fname, "wb").write(self.data)
69           return S_OK
70   
71       def SetSize(self, size):
72           print("Set Size" + str(size))
73           if size > len(self.data):
74               self.data = self.data + str2bytes("\000" * (size - len(self.data)))
75           else:
76               self.data = self.data[0:size]
77           return S_OK
