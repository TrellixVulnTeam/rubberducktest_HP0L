10   
11   from pygments.lexer import RegexLexer, include, words
12   from pygments.token import Text, Comment, Operator, Keyword, Name, String, \
13       Number, Punctuation
14   
15   __all__ = ['EiffelLexer']
16   
17   
18   class EiffelLexer(RegexLexer):
19       """ 
20       For `Eiffel <http://www.eiffel.com>`_ source code. 
21    
22       .. versionadded:: 2.0 
23       """
24       name = 'Eiffel'
25       aliases = ['eiffel']
26       filenames = ['*.e']
27       mimetypes = ['text/x-eiffel']
28   
29       tokens = {
30           'root': [
31               (r'[^\S\n]+', Text),
32               (r'--.*?\n', Comment.Single),
33               (r'[^\S\n]+', Text),
34               # Please note that keyword and operator are case insensitive.
35               (r'(?i)(true|false|void|current|result|precursor)\b', Keyword.Constant),
36               (r'(?i)(and(\s+then)?|not|xor|implies|or(\s+else)?)\b', Operator.Word),
37               (words((
38                   'across', 'agent', 'alias', 'all', 'as', 'assign', 'attached',
39                   'attribute', 'check', 'class', 'convert', 'create', 'debug',
40                   'deferred', 'detachable', 'do', 'else', 'elseif', 'end', 'ensure',
41                   'expanded', 'export', 'external', 'feature', 'from', 'frozen', 'if',
42                   'inherit', 'inspect', 'invariant', 'like', 'local', 'loop', 'none',
43                   'note', 'obsolete', 'old', 'once', 'only', 'redefine', 'rename',
44                   'require', 'rescue', 'retry', 'select', 'separate', 'then',
45                   'undefine', 'until', 'variant', 'when'), prefix=r'(?i)\b', suffix=r'\b'),
46                Keyword.Reserved),
47               (r'"\[([^\]%]|%(.|\n)|\][^"])*?\]"', String),
48               (r'"([^"%\n]|%.)*?"', String),
49               include('numbers'),
50               (r"'([^'%]|%'|%%)'", String.Char),
51               (r"(//|\\\\|>=|<=|:=|/=|~|/~|[\\?!#%&@|+/\-=>*$<^\[\]])", Operator),
52               (r"([{}():;,.])", Punctuation),
53               (r'([a-z]\w*)|([A-Z][A-Z0-9_]*[a-z]\w*)', Name),
54               (r'([A-Z][A-Z0-9_]*)', Name.Class),
55               (r'\n+', Text),
56           ],
57           'numbers': [
58               (r'0[xX][a-fA-F0-9]+', Number.Hex),
59               (r'0[bB][01]+', Number.Bin),
60               (r'0[cC][0-7]+', Number.Oct),
61               (r'([0-9]+\.[0-9]*)|([0-9]*\.[0-9]+)', Number.Float),
62               (r'[0-9]+', Number.Integer),
63           ],
64       }
