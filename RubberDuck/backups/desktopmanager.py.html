11   def desktop_name_dlgproc(hwnd, msg, wparam, lparam):
12       """Handles messages from the desktop name dialog box"""
13       if msg in (win32con.WM_CLOSE, win32con.WM_DESTROY):
14           win32gui.DestroyWindow(hwnd)
15       elif msg == win32con.WM_COMMAND:
16           if wparam == win32con.IDOK:
17               desktop_name = win32gui.GetDlgItemText(hwnd, 72)
18               print("new desktop name: ", desktop_name)
19               win32gui.DestroyWindow(hwnd)
20               create_desktop(desktop_name)
21   
22           elif wparam == win32con.IDCANCEL:
23               win32gui.DestroyWindow(hwnd)

74   def new_icon(hdesk, desktop_name):
75       """Runs as a thread on each desktop to create a new tray icon and handle its messages"""
76       global id
77       id = id + 1
78       hdesk.SetThreadDesktop()
79       ## apparently the threads can't use same hinst, so each needs its own window class
80       windowclassname = "PythonDesktopManager" + desktop_name
81       wc = win32gui.WNDCLASS()
82       wc.hInstance = win32api.GetModuleHandle(None)
83       wc.lpszClassName = windowclassname
84       wc.style = win32con.CS_VREDRAW | win32con.CS_HREDRAW | win32con.CS_GLOBALCLASS
85       wc.hCursor = win32gui.LoadCursor(0, win32con.IDC_ARROW)
86       wc.hbrBackground = win32con.COLOR_WINDOW
87       wc.lpfnWndProc = icon_wndproc
88       windowclass = win32gui.RegisterClass(wc)
89       style = win32con.WS_OVERLAPPED | win32con.WS_SYSMENU
90       hwnd = win32gui.CreateWindow(
91           windowclass,
92           "dm_" + desktop_name,
93           win32con.WS_SYSMENU,
94           0,
95           0,
96           win32con.CW_USEDEFAULT,
97           win32con.CW_USEDEFAULT,
98           0,
99           0,
100          wc.hInstance,
101          None,
102      )
103      win32gui.UpdateWindow(hwnd)
104      flags = win32gui.NIF_ICON | win32gui.NIF_MESSAGE | win32gui.NIF_TIP
105      notify_info = (
106          hwnd,
107          id,
108          flags,
109          win32con.WM_USER + 20,
110          hicon,
111          "Desktop Manager (%s)" % desktop_name,
112      )
113      window_info[hwnd] = notify_info
114      ## wait for explorer to initialize system tray for new desktop
115      tray_found = 0
116      while not tray_found:
117          try:
118              tray_found = win32gui.FindWindow("Shell_TrayWnd", None)
119          except win32gui.error:
120              traceback.print_exc
121              time.sleep(0.5)
122      win32gui.Shell_NotifyIcon(win32gui.NIM_ADD, notify_info)
123      win32gui.PumpMessages()

126  def create_desktop(desktop_name, start_explorer=1):
127      """Creates a new desktop and spawns a thread running on it 
128      Will also start a new icon thread on an existing desktop 
129      """
130      sa = pywintypes.SECURITY_ATTRIBUTES()
131      sa.bInheritHandle = 1
132  
133      try:
134          hdesk = win32service.CreateDesktop(
135              desktop_name, 0, win32con.MAXIMUM_ALLOWED, sa
136          )
137      except win32service.error:
138          traceback.print_exc()
139          errbuf = io.StringIO()
140          traceback.print_exc(None, errbuf)
141          win32api.MessageBox(0, errbuf.getvalue(), "Desktop creation failed")
142          return
143      if start_explorer:
144          s = win32process.STARTUPINFO()
145          s.lpDesktop = desktop_name
146          prc_info = win32process.CreateProcess(
147              None,
148              "Explorer.exe",
149              None,
150              None,
151              True,
152              win32con.CREATE_NEW_CONSOLE,
153              None,
154              "c:\\",
155              s,
156          )
157  
158      th = _thread.start_new_thread(new_icon, (hdesk, desktop_name))
159      hdesk.SwitchDesktop()

162  def icon_wndproc(hwnd, msg, wp, lp):
163      """Window proc for the tray icons"""
164      if lp == win32con.WM_LBUTTONDOWN:
165          ## popup menu won't disappear if you don't do this
166          win32gui.SetForegroundWindow(hwnd)
167  
168          curr_desktop = win32service.OpenInputDesktop(0, True, win32con.MAXIMUM_ALLOWED)
169          curr_desktop_name = win32service.GetUserObjectInformation(
170              curr_desktop, win32con.UOI_NAME
171          )
172          winsta = win32service.GetProcessWindowStation()
173          desktops = winsta.EnumDesktops()
174          m = win32gui.CreatePopupMenu()
175          desktop_cnt = len(desktops)
176          ## *don't* create an item 0
177          for d in range(1, desktop_cnt + 1):
178              mf_flags = win32con.MF_STRING
179              ## if you switch to winlogon yourself, there's nothing there and you're stuck
180              if desktops[d - 1].lower() in ("winlogon", "disconnect"):
181                  mf_flags = mf_flags | win32con.MF_GRAYED | win32con.MF_DISABLED
182              if desktops[d - 1] == curr_desktop_name:
183                  mf_flags = mf_flags | win32con.MF_CHECKED
184              win32gui.AppendMenu(m, mf_flags, d, desktops[d - 1])
185          win32gui.AppendMenu(m, win32con.MF_STRING, desktop_cnt + 1, "Create new ...")
186          win32gui.AppendMenu(m, win32con.MF_STRING, desktop_cnt + 2, "Exit")
187  
188          x, y = win32gui.GetCursorPos()
189          d = win32gui.TrackPopupMenu(
190              m,
191              win32con.TPM_LEFTBUTTON | win32con.TPM_RETURNCMD | win32con.TPM_NONOTIFY,
192              x,
193              y,
194              0,
195              hwnd,
196              None,
197          )
198          win32gui.PumpWaitingMessages()
199          win32gui.DestroyMenu(m)
200          if d == desktop_cnt + 1:  ## Create new
201              get_new_desktop_name(hwnd)
202          elif d == desktop_cnt + 2:  ## Exit
203              win32gui.PostQuitMessage(0)
204              win32gui.Shell_NotifyIcon(win32gui.NIM_DELETE, window_info[hwnd])
205              del window_info[hwnd]
206              origin_desktop.SwitchDesktop()
207          elif d > 0:
208              hdesk = win32service.OpenDesktop(
209                  desktops[d - 1], 0, 0, win32con.MAXIMUM_ALLOWED
210              )
211              hdesk.SwitchDesktop()
212          return 0
213      else:
214          return win32gui.DefWindowProc(hwnd, msg, wp, lp)
