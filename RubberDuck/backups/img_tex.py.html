19       def load(self, filename):
20           try:
21               fd = open(filename, 'rb')
22               if fd.read(4) != 'KTEX':
23                   raise Exception('Invalid tex identifier')
24   
25               headersize = unpack('I', fd.read(4))[0]
26               header = fd.read(headersize)
27               if len(header) != headersize:
28                   raise Exception('Truncated tex header')
29   
30               info = json.loads(header)
31               data = fd.read()
32               if len(data) != info['datalen']:
33                   raise Exception('Truncated tex data')
34   
35           except:
36               Logger.warning('Image: Image <%s> is corrupted' % filename)
37               raise
38   
39           width, height = info['image_size']
40           tw, th = info['texture_size']
41   
42           images = [data]
43           im = ImageData(width, height, str(info['format']), images[0],
44                          source=filename)
