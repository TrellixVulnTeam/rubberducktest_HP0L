1    """Simple base-classes for extensions and filters. 
2     
3    None of the filter and extension functions are considered 'optional' by the 
4    framework.  These base-classes provide simple implementations for the 
5    Initialize and Terminate functions, allowing you to omit them, 
6     
7    It is not necessary to use these base-classes - but if you don't, you 
8    must ensure each of the required methods are implemented. 
9    """
10   
11   
12   class SimpleExtension:
13       "Base class for a simple ISAPI extension"
14   
15       def __init__(self):
16           pass
17   
18       def GetExtensionVersion(self, vi):
19           """Called by the ISAPI framework to get the extension version 
20    
21           The default implementation uses the classes docstring to 
22           set the extension description."""
23           # nod to our reload capability - vi is None when we are reloaded.
24           if vi is not None:
25               vi.ExtensionDesc = self.__doc__
26   
27       def HttpExtensionProc(self, control_block):
28           """Called by the ISAPI framework for each extension request. 
29    
30           sub-classes must provide an implementation for this method. 
31           """
32           raise NotImplementedError("sub-classes should override HttpExtensionProc")
33   
34       def TerminateExtension(self, status):
35           """Called by the ISAPI framework as the extension terminates."""
36           pass
37   
38   
39   class SimpleFilter:
40       "Base class for a a simple ISAPI filter"
41       filter_flags = None
42   
43       def __init__(self):
44           pass
45   
46       def GetFilterVersion(self, fv):
47           """Called by the ISAPI framework to get the extension version 
48    
49           The default implementation uses the classes docstring to 
50           set the extension description, and uses the classes 
51           filter_flags attribute to set the ISAPI filter flags - you 
52           must specify filter_flags in your class. 
53           """
54           if self.filter_flags is None:
55               raise RuntimeError("You must specify the filter flags")
56           # nod to our reload capability - fv is None when we are reloaded.
57           if fv is not None:
58               fv.Flags = self.filter_flags
59               fv.FilterDesc = self.__doc__
60   
61       def HttpFilterProc(self, fc):
62           """Called by the ISAPI framework for each filter request. 
63    
64           sub-classes must provide an implementation for this method. 
65           """
66           raise NotImplementedError("sub-classes should override HttpExtensionProc")
67   
68       def TerminateFilter(self, status):
69           """Called by the ISAPI framework as the filter terminates."""
70           pass
