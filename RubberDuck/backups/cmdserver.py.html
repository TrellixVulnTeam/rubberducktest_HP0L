14   class ThreadWriter:
15       "Assign an instance to sys.stdout for per-thread printing objects - Courtesy Guido!"
16   
17       def __init__(self):
18           "Constructor -- initialize the table of writers"
19           self.writers = {}
20           self.origStdOut = None
21   
22       def register(self, writer):
23           "Register the writer for the current thread"
24           self.writers[_thread.get_ident()] = writer
25           if self.origStdOut is None:
26               self.origStdOut = sys.stdout
27               sys.stdout = self
28   
29       def unregister(self):
30           "Remove the writer for the current thread, if any"
31           try:
32               del self.writers[_thread.get_ident()]
33           except KeyError:
34               pass
35           if len(self.writers) == 0:
36               sys.stdout = self.origStdOut
37               self.origStdOut = None
38   
39       def getwriter(self):
40           "Return the current thread's writer, default sys.stdout"
41           try:
42               return self.writers[_thread.get_ident()]
43           except KeyError:
44               return self.origStdOut
45   
46       def write(self, str):
47           "Write to the current thread's writer, default sys.stdout"
48           self.getwriter().write(str)
