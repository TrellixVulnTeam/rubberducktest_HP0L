19   def _fillvkmap():
20       # Pull the VK_names from win32con
21       names = [entry for entry in win32con.__dict__ if entry.startswith("VK_")]
22       for name in names:
23           code = getattr(win32con, name)
24           n = name[3:].lower()
25           key_name_to_vk[n] = code
26           if n in _better_names:
27               n = _better_names[n]
28               key_name_to_vk[n] = code
29           key_code_to_name[code] = n

35   def get_vk(chardesc):
36       if len(chardesc) == 1:
37           # it is a character.
38           info = win32api.VkKeyScan(chardesc)
39           if info == -1:
40               # Note: returning None, None causes an error when keyboard layout is non-English, see the report below
41               # https://stackoverflow.com/questions/45138084/pythonwin-occasionally-gives-an-error-on-opening
42               return 0, 0
43           vk = win32api.LOBYTE(info)
44           state = win32api.HIBYTE(info)
45           modifiers = 0
46           if state & 0x1:
47               modifiers |= win32con.SHIFT_PRESSED
48           if state & 0x2:
49               modifiers |= win32con.LEFT_CTRL_PRESSED | win32con.RIGHT_CTRL_PRESSED
50           if state & 0x4:
51               modifiers |= win32con.LEFT_ALT_PRESSED | win32con.RIGHT_ALT_PRESSED
52           return vk, modifiers
53       # must be a 'key name'
54       return key_name_to_vk.get(chardesc.lower()), 0

73   def parse_key_name(name):
74       name = name + "-"  # Add a sentinal
75       start = pos = 0
76       max = len(name)
77       toks = []
78       while pos < max:
79           if name[pos] in "+-":
80               tok = name[start:pos]
81               # use the ascii lower() version of tok, so ascii chars require
82               # an explicit shift modifier - ie 'Ctrl+G' should be treated as
83               # 'ctrl+g' - 'ctrl+shift+g' would be needed if desired.
84               # This is mainly to avoid changing all the old keystroke defs
85               toks.append(tok.lower())
86               pos += 1  # skip the sep
87               start = pos
88           pos += 1
89       flags = 0
90       # do the modifiers
91       for tok in toks[:-1]:
92           mod = modifiers.get(tok.lower())
93           if mod is not None:
94               flags |= mod
95       # the key name
96       vk, this_flags = get_vk(toks[-1])
97       return vk, flags | this_flags

117  def make_key_name(vk, flags):
118      # Check alt keys.
119      flags_done = 0
120      parts = []
121      for moddata in _checks:
122          for name, checkflag in moddata:
123              if flags & checkflag:
124                  parts.append(name)
125                  flags_done = flags_done & checkflag
126                  break
127      if flags_done & flags:
128          parts.append(hex(flags & ~flags_done))
129      # Now the key name.
130      if vk is None:
131          parts.append("<Unknown scan code>")
132      else:
133          try:
134              parts.append(key_code_to_name[vk])
135          except KeyError:
136              # Not in our virtual key map - ask Windows what character this
137              # key corresponds to.
138              scancode = win32api.MapVirtualKey(vk, MAPVK_VK_TO_CHAR)
139              parts.append(chr(scancode))
140      sep = "+"
141      if sep in parts:
142          sep = "-"
143      return sep.join([p.capitalize() for p in parts])
