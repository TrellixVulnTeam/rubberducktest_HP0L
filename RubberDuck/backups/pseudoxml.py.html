5    """ 
6    Simple internal document tree Writer, writes indented pseudo-XML. 
7    """
8    
9    __docformat__ = 'reStructuredText'
10   
11   
12   from docutils import writers, frontend
13   
14   
15   class Writer(writers.Writer):
16   
17       supported = ('pprint', 'pformat', 'pseudoxml')
18       """Formats this writer supports."""
19       
20       settings_spec = (
21           '"Docutils pseudo-XML" Writer Options',
22           None,
23           (('Pretty-print <#text> nodes.',
24             ['--detailed'],
25             {'action': 'store_true', 'validator': frontend.validate_boolean}),
26           ))
27   
28       config_section = 'pseudoxml writer'
29       config_section_dependencies = ('writers',)
30   
31       output = None
32       """Final translated form of `document`."""
33   
34       def translate(self):
35           self.output = self.document.pformat()
36   
37       def supports(self, format):
38           """This writer supports all format-specific elements."""
39           return True
