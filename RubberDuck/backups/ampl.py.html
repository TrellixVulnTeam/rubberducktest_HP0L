18   class AmplLexer(RegexLexer):
19       """ 
20       For `AMPL <http://ampl.com/>`_ source code.

22       .. versionadded:: 2.2 
23       """
24       name = 'Ampl'
25       aliases = ['ampl']
26       filenames = ['*.run']

28       tokens = {
29           'root': [
30               (r'\n', Text),
31               (r'\s+', Text.Whitespace),
32               (r'#.*?\n', Comment.Single),
33               (r'/[*](.|\n)*?[*]/', Comment.Multiline),
34               (words((
35                   'call', 'cd', 'close', 'commands', 'data', 'delete', 'display',
36                   'drop', 'end', 'environ', 'exit', 'expand', 'include', 'load',
37                   'model', 'objective', 'option', 'problem', 'purge', 'quit',
38                   'redeclare', 'reload', 'remove', 'reset', 'restore', 'shell',
39                   'show', 'solexpand', 'solution', 'solve', 'update', 'unload',
40                   'xref', 'coeff', 'coef', 'cover', 'obj', 'interval', 'default',
41                   'from', 'to', 'to_come', 'net_in', 'net_out', 'dimen',
42                   'dimension', 'check', 'complements', 'write', 'function',
43                   'pipe', 'format', 'if', 'then', 'else', 'in', 'while', 'repeat',
44                   'for'), suffix=r'\b'), Keyword.Reserved),
45               (r'(integer|binary|symbolic|ordered|circular|reversed|INOUT|IN|OUT|LOCAL)',
46                Keyword.Type),
47               (r'\".*?\"', String.Double),
48               (r'\'.*?\'', String.Single),
49               (r'[()\[\]{},;:]+', Punctuation),
50               (r'\b(\w+)(\.)(astatus|init0|init|lb0|lb1|lb2|lb|lrc|'
51                r'lslack|rc|relax|slack|sstatus|status|ub0|ub1|ub2|'
52                r'ub|urc|uslack|val)',
53                bygroups(Name.Variable, Punctuation, Keyword.Reserved)),
54               (r'(set|param|var|arc|minimize|maximize|subject to|s\.t\.|subj to|'
55                r'node|table|suffix|read table|write table)(\s+)(\w+)',
56                bygroups(Keyword.Declaration, Text, Name.Variable)),
57               (r'(param)(\s*)(:)(\s*)(\w+)(\s*)(:)(\s*)((\w|\s)+)',
58                bygroups(Keyword.Declaration, Text, Punctuation, Text,
59                         Name.Variable, Text, Punctuation, Text, Name.Variable)),
60               (r'(let|fix|unfix)(\s*)((?:\{.*\})?)(\s*)(\w+)',
61                bygroups(Keyword.Declaration, Text, using(this), Text, Name.Variable)),
62               (words((
63                   'abs', 'acos', 'acosh', 'alias', 'asin', 'asinh', 'atan', 'atan2',
64                   'atanh', 'ceil', 'ctime', 'cos', 'exp', 'floor', 'log', 'log10',
65                   'max', 'min', 'precision', 'round', 'sin', 'sinh', 'sqrt', 'tan',
66                   'tanh', 'time', 'trunc', 'Beta', 'Cauchy', 'Exponential', 'Gamma',
67                   'Irand224', 'Normal', 'Normal01', 'Poisson', 'Uniform', 'Uniform01',
68                   'num', 'num0', 'ichar', 'char', 'length', 'substr', 'sprintf',
69                   'match', 'sub', 'gsub', 'print', 'printf', 'next', 'nextw', 'prev',
70                   'prevw', 'first', 'last', 'ord', 'ord0', 'card', 'arity',
71                   'indexarity'), prefix=r'\b', suffix=r'\b'), Name.Builtin),
72               (r'(\+|\-|\*|/|\*\*|=|<=|>=|==|\||\^|<|>|\!|\.\.|:=|\&|\!=|<<|>>)',
73                Operator),
74               (words((
75                   'or', 'exists', 'forall', 'and', 'in', 'not', 'within', 'union',
76                   'diff', 'difference', 'symdiff', 'inter', 'intersect',
77                   'intersection', 'cross', 'setof', 'by', 'less', 'sum', 'prod',
78                   'product', 'div', 'mod'), suffix=r'\b'),
79                Keyword.Reserved),  # Operator.Name but not enough emphasized with that
80               (r'(\d+\.(?!\.)\d*|\.(?!.)\d+)([eE][+-]?\d+)?', Number.Float),
81               (r'\d+([eE][+-]?\d+)?', Number.Integer),
82               (r'[+-]?Infinity', Number.Integer),
83               (r'(\w+|(\.(?!\.)))', Text)
84           ]

86       }
