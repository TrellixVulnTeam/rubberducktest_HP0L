69   class AXSite:
70       """An Active Scripting site.  A Site can have exactly one engine."""
71   
72       _public_methods_ = IActiveScriptSite_methods
73       _com_interfaces_ = [axscript.IID_IActiveScriptSite]
74   
75       def __init__(self, objModel={}, engine=None, lcid=0):
76           self.lcid = lcid
77           self.objModel = {}
78           for name, object in objModel.items():
79               # Gregs code did string.lower this - I think that is callers job if he wants!
80               self.objModel[name] = object
81   
82           self.engine = None
83           if engine:
84               self._AddEngine(engine)
85   
86       def AddEngine(self, engine):
87           """Adds a new engine to the site. 
88           engine can be a string, or a fully wrapped engine object. 
89           """
90           if type(engine) == type(""):
91               newEngine = AXEngine(util.wrap(self), engine)
92           else:
93               newEngine = engine
94           self.engine = newEngine
95           flags = (
96               axscript.SCRIPTITEM_ISVISIBLE
97               | axscript.SCRIPTITEM_NOCODE
98               | axscript.SCRIPTITEM_GLOBALMEMBERS
99               | axscript.SCRIPTITEM_ISPERSISTENT
100          )
101          for name in self.objModel.keys():
102              newEngine.AddNamedItem(name, flags)
103              newEngine.SetScriptState(axscript.SCRIPTSTATE_INITIALIZED)
104          return newEngine
105  
106      # B/W compat
107      _AddEngine = AddEngine
108

120              )

122          ### for now, we don't have any type information
123  
124          if returnMask & axscript.SCRIPTINFO_IUNKNOWN:
125              return (self.objModel[name], None)
