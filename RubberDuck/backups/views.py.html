1    ''' 
2    RecycleView Views 
3    ================= 
4     
5    .. versionadded:: 1.10.0 
6     
7    The adapter part of the RecycleView which together with the layout is the 
8    view part of the model-view-controller pattern. 
9     
10   The view module handles converting the data to a view using the adapter class 
11   which is then displayed by the layout. A view can be any Widget based class. 
12   However, inheriting from RecycleDataViewBehavior adds methods for converting 
13   the data to a view. 
14    
15   TODO: 
16       * Make view caches specific to each view class type. 
17    
18   '''

20   from kivy.properties import ObjectProperty
21   from kivy.event import EventDispatcher
22   from collections import defaultdict

24   __all__ = (
25       'RecycleDataViewBehavior', 'RecycleKVIDsDataViewBehavior',

28   _view_base_cache = {}
29   '''Cache whose keys are classes and values is a boolean indicating whether the 
30   class inherits from :class:`RecycleDataViewBehavior`. 
31   '''
32   
33   _cached_views = defaultdict(list)
34   '''A size limited cache that contains old views (instances) that are not used. 
35   Each key is a class whose value is the list of the instances of that class. 
36   '''
37   # current number of unused classes in the class cache
38   _cache_count = 0
39   # maximum number of items in the class cache
40   _max_cache_size = 1000
41   
42   
43   def _clean_cache():
44       '''Trims _cached_views cache to half the size of `_max_cache_size`. 
45       '''
46       # all keys will be reduced to max_size.
47       max_size = (_max_cache_size // 2) // len(_cached_views)
48       global _cache_count
49       for cls, instances in _cached_views.items():
50           _cache_count -= max(0, len(instances) - max_size)
51           del instances[max_size:]

54   class RecycleDataViewBehavior(object):
55       '''A optional base class for data views (:attr:`RecycleView`.viewclass). 
56       If a view inherits from this class, the class's functions will be called 
57       when the view needs to be updated due to a data change or layout update. 
58       '''
59   
60       def refresh_view_attrs(self, rv, index, data):
61           '''Called by the :class:`RecycleAdapter` when the view is initially 
62           populated with the values from the `data` dictionary for this item. 
63    
64           Any pos or size info should be removed because they are set 
65           subsequently with :attr:`refresh_view_layout`. 
66    
67           :Parameters: 
68    
69               `rv`: :class:`RecycleView` instance 
70                   The :class:`RecycleView` that caused the update. 
71               `data`: dict 
72                   The data dict used to populate this view. 
73           '''
74           sizing_attrs = RecycleDataAdapter._sizing_attrs
75           for key, value in data.items():
76               if key not in sizing_attrs:
77                   setattr(self, key, value)
78   
79       def refresh_view_layout(self, rv, index, layout, viewport):
80           '''Called when the view's size is updated by the layout manager, 
81           :class:`RecycleLayoutManagerBehavior`. 
82    
83           :Parameters: 
84    
85               `rv`: :class:`RecycleView` instance 
86                   The :class:`RecycleView` that caused the update. 
87               `viewport`: 4-tuple 
88                   The coordinates of the bottom left and width height in layout 
89                   manager coordinates. This may be larger than this view item. 
90    
91           :raises: 
92               `LayoutChangeException`: If the sizing or data changed during a 
93               call to this method, raising a `LayoutChangeException` exception 
94               will force a refresh. Useful when data changed and we don't want 
95               to layout further since it'll be overwritten again soon. 
96           '''
97           w, h = layout.pop('size')
98           if w is None:
99               if h is not None:
100                  self.height = h
101          else:
102              if h is None:
103                  self.width = w
104              else:
105                  self.size = w, h
106  
107          for name, value in layout.items():
108              setattr(self, name, value)
109  
110      def apply_selection(self, rv, index, is_selected):
111          pass

114  class RecycleKVIDsDataViewBehavior(RecycleDataViewBehavior):
115      """Similar to :class:`RecycleDataViewBehavior`, except that the data keys 
116      can signify properties of an object named with an id in the root KV rule. 
117   
118      E.g. given a KV rule:: 
119   
120          <MyRule@RecycleKVIDsDataViewBehavior+BoxLayout>: 
121              Label: 
122                  id: name 
123              Label: 
124                  id: value 
125   
126      Then setting the data list with 
127      ``rv.data = [{'name.text': 'Kivy user', 'value.text': '12'}]`` would 
128      automatically set the corresponding labels. 
129   
130      So, if the key doesn't have a period, the named property of the root widget 
131      will be set to the corresponding value. If there is a period, the named 
132      property of the widget with the id listed before the period will be set to 
133      the corresponding value. 
134   
135      .. versionadded:: 2.0.0 
136      """
137  
138      def refresh_view_attrs(self, rv, index, data):
139          sizing_attrs = RecycleDataAdapter._sizing_attrs
140          for key, value in data.items():
141              if key not in sizing_attrs:
142                  name, *ids = key.split('.')
143                  if ids:
144                      if len(ids) != 1:
145                          raise ValueError(
146                              f'Data key "{key}" has more than one period')
147                      setattr(self.ids[name], ids[0], value)
148                  else:
149                      setattr(self, name, value)

152  class RecycleDataAdapter(EventDispatcher):
153      '''The class that converts data to a view. 
154   
155      --- Internal details --- 
156      A view can have 3 states. 
157   
158          * It can be completely in sync with the data, which 
159            occurs when the view is displayed. These are stored in :attr:`views`. 
160          * It can be dirty, which occurs when the view is in sync with the data, 
161            except for the size/pos parameters which is controlled by the layout. 
162            This occurs when the view is not currently displayed but the data has 
163            not changed. These views are stored in :attr:`dirty_views`. 
164          * Finally the view can be dead which occurs when the data changes and 
165            the view was not updated or when a view is just created. Such views 
166            are typically added to the internal cache. 
167   
168      Typically what happens is that the layout manager lays out the data 
169      and then asks for views, using :meth:`set_visible_views,` for some specific 
170      data items that it displays. 
171   
172      These views are gotten from the current views, dirty or global cache. Then 
173      depending on the view state :meth:`refresh_view_attrs` is called to bring 
174      the view up to date with the data (except for sizing parameters). Finally, 
175      the layout manager gets these views, updates their size and displays them. 
176      '''
177  
178      recycleview = ObjectProperty(None, allownone=True)
179      '''The :class:`~kivy.uix.recycleview.RecycleViewBehavior` associated 
180      with this instance. 
181      '''
182  
183      # internals
184      views = {}  # current displayed items
185      # items whose attrs, except for pos/size is still accurate
186      dirty_views = defaultdict(dict)
187  
188      _sizing_attrs = {
189          'size', 'width', 'height', 'size_hint', 'size_hint_x', 'size_hint_y',
190          'pos', 'x', 'y', 'center', 'center_x', 'center_y', 'pos_hint',
191          'size_hint_min', 'size_hint_min_x', 'size_hint_min_y', 'size_hint_max',
192          'size_hint_max_x', 'size_hint_max_y'}
193  
194      def __init__(self, **kwargs):
195          """ 
196          Fix for issue https://github.com/kivy/kivy/issues/5913: 
197          Scrolling RV A, then Scrolling RV B, content of A and B seemed 
198          to be getting mixed up 
199          """
200          self.views = {}
201          self.dirty_views = defaultdict(dict)
202          super(RecycleDataAdapter, self).__init__(**kwargs)
203  
204      def attach_recycleview(self, rv):
205          '''Associates a :class:`~kivy.uix.recycleview.RecycleViewBehavior` 
206          with this instance. It is stored in :attr:`recycleview`. 
207          '''
208          self.recycleview = rv
209  
210      def detach_recycleview(self):
211          '''Removes the :class:`~kivy.uix.recycleview.RecycleViewBehavior` 
212          associated with this instance and clears :attr:`recycleview`. 
213          '''
214          self.recycleview = None
215  
216      def create_view(self, index, data_item, viewclass):
217          '''(internal) Creates and initializes the view for the data at `index`. 
218   
219          The returned view is synced with the data, except for the pos/size 
220          information. 
221          '''
222          if viewclass is None:
223              return
224  
225          view = viewclass()
226          self.refresh_view_attrs(index, data_item, view)
227          return view
228  
229      def get_view(self, index, data_item, viewclass):
230          '''(internal) Returns a view instance for the data at `index` 
231   
232          It looks through the various caches and finally creates a view if it 
233          doesn't exist. The returned view is synced with the data, except for 
234          the pos/size information. 
235   
236          If found in the cache it's removed from the source 
237          before returning. It doesn't check the current views. 
238          '''
239          # is it in the dirtied views?
240          dirty_views = self.dirty_views
241          if viewclass is None:
242              return
243          stale = False
244          view = None
245  
246          if viewclass in dirty_views:  # get it first from dirty list
247              dirty_class = dirty_views[viewclass]
248              if index in dirty_class:
249                  # we found ourself in the dirty list, no need to update data!
250                  view = dirty_class.pop(index)
251              elif _cached_views[viewclass]:
252                  # global cache has this class, update data
253                  view, stale = _cached_views[viewclass].pop(), True
254              elif dirty_class:
255                  # random any dirty view element - update data
256                  view, stale = dirty_class.popitem()[1], True
257          elif _cached_views[viewclass]:  # otherwise go directly to cache
258              # global cache has this class, update data
259              view, stale = _cached_views[viewclass].pop(), True
260  
261          if view is None:
262              view = self.create_view(index, data_item, viewclass)
263              if view is None:
264                  return
265  
266          if stale:
267              self.refresh_view_attrs(index, data_item, view)
268          return view
269  
270      def refresh_view_attrs(self, index, data_item, view):
271          '''(internal) Syncs the view and brings it up to date with the data. 
272   
273          This method calls :meth:`RecycleDataViewBehavior.refresh_view_attrs` 
274          if the view inherits from :class:`RecycleDataViewBehavior`. See that 
275          method for more details. 
276   
277          .. note:: 
278              Any sizing and position info is skipped when syncing with the data. 
279          '''
280          viewclass = view.__class__
281          if viewclass not in _view_base_cache:
282              _view_base_cache[viewclass] = isinstance(view,
283                                                       RecycleDataViewBehavior)
284  
285          if _view_base_cache[viewclass]:
286              view.refresh_view_attrs(self.recycleview, index, data_item)
287          else:
288              sizing_attrs = RecycleDataAdapter._sizing_attrs
289              for key, value in data_item.items():
290                  if key not in sizing_attrs:
291                      setattr(view, key, value)
292  
293      def refresh_view_layout(self, index, layout, view, viewport):
294          '''Updates the sizing information of the view. 
295   
296          viewport is in coordinates of the layout manager. 
297   
298          This method calls :meth:`RecycleDataViewBehavior.refresh_view_attrs` 
299          if the view inherits from :class:`RecycleDataViewBehavior`. See that 
300          method for more details. 
301   
302          .. note:: 
303              Any sizing and position info is skipped when syncing with the data. 
304          '''
305          if view.__class__ not in _view_base_cache:
306              _view_base_cache[view.__class__] = isinstance(
307                  view, RecycleDataViewBehavior)
308  
309          if _view_base_cache[view.__class__]:
310              view.refresh_view_layout(
311                  self.recycleview, index, layout, viewport)
312          else:
313              w, h = layout.pop('size')
314              if w is None:
315                  if h is not None:
316                      view.height = h
317              else:
318                  if h is None:
319                      view.width = w
320                  else:
321                      view.size = w, h
322  
323              for name, value in layout.items():
324                  setattr(view, name, value)
325  
326      def make_view_dirty(self, view, index):
327          '''(internal) Used to flag this view as dirty, ready to be used for 
328          others. See :meth:`make_views_dirty`. 
329          '''
330          del self.views[index]
331          self.dirty_views[view.__class__][index] = view
332  
333      def make_views_dirty(self):
334          '''Makes all the current views dirty. 
335   
336          Dirty views are still in sync with the corresponding data. However, the 
337          size information may go out of sync. Therefore a dirty view can be 
338          reused by the same index by just updating the sizing information. 
339   
340          Once the underlying data of this index changes, the view should be 
341          removed from the dirty views and moved to the global cache with 
342          :meth:`invalidate`. 
343   
344          This is typically called when the layout manager needs to re-layout all 
345          the data. 
346          '''
347          views = self.views
348          if not views:
349              return
350  
351          dirty_views = self.dirty_views
352          for index, view in views.items():
353              dirty_views[view.__class__][index] = view
354          self.views = {}
355  
356      def invalidate(self):
357          '''Moves all the current views into the global cache. 
358   
359          As opposed to making a view dirty where the view is in sync with the 
360          data except for sizing information, this will completely disconnect the 
361          view from the data, as it is assumed the data has gone out of sync with 
362          the view. 
363   
364          This is typically called when the data changes. 
365          '''
366          global _cache_count
367          for view in self.views.values():
368              _cached_views[view.__class__].append(view)
369              _cache_count += 1
370  
371          for cls, views in self.dirty_views.items():
372              _cached_views[cls].extend(views.values())
373              _cache_count += len(views)
374  
375          if _cache_count >= _max_cache_size:
376              _clean_cache()
377          self.views = {}
378          self.dirty_views.clear()
379  
380      def set_visible_views(self, indices, data, viewclasses):
381          '''Gets a 3-tuple of the new, remaining, and old views for the current 
382          viewport. 
383   
384          The new views are synced to the data except for the size/pos 
385          properties. 
386          The old views need to be removed from the layout, and the new views 
387          added. 
388   
389          The new views are not necessarily *new*, but are all the currently 
390          visible views. 
391          '''
392          visible_views = {}
393          previous_views = self.views
394          ret_new = []
395          ret_remain = []
396          get_view = self.get_view
397  
398          # iterate though the visible view
399          # add them into the container if not already done
400          for index in indices:
401              view = previous_views.pop(index, None)
402              if view is not None:  # was current view
403                  visible_views[index] = view
404                  ret_remain.append((index, view))
405              else:
406                  view = get_view(index, data[index],
407                                  viewclasses[index]['viewclass'])
408                  if view is None:
409                      continue
410                  visible_views[index] = view
411                  ret_new.append((index, view))
412  
413          old_views = previous_views.items()
414          self.make_views_dirty()
415          self.views = visible_views
416          return ret_new, ret_remain, old_views
417  
418      def get_visible_view(self, index):
419          '''Returns the currently visible view associated with ``index``. 
420   
421          If no view is currently displayed for ``index`` it returns ``None``. 
422          '''
423          return self.views.get(index)
