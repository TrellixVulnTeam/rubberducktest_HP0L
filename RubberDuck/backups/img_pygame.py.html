12   from kivy.logger import Logger
13   from kivy.core.image import ImageLoaderBase, ImageData, ImageLoader
14   from os.path import isfile
15   from kivy.utils import deprecated
16   
17   try:
18       import pygame
19   except:
20       raise
21   
22   
23   class ImageLoaderPygame(ImageLoaderBase):
24       '''Image loader based on the PIL library'''
25   
26       @deprecated(
27           msg='Pygame has been deprecated and will be removed after 1.11.0')
28       def __init__(self, *largs, **kwargs):
29           super(ImageLoaderPygame, self).__init__(*largs, **kwargs)
30   
31       @staticmethod
32       def extensions():
33           '''Return accepted extensions for this loader'''
34           # under OS X, i got with "pygame.error: File is not a Windows BMP
35           # file". documentation said: The image module is a required dependency
36           # of Pygame, but it only optionally supports any extended file formats.
37           # By default it can only load uncompressed BMP image
38           if pygame.image.get_extended() == 0:
39               return ('bmp', )
40           return ('jpg', 'jpeg', 'jpe', 'png', 'bmp', 'pcx', 'tga', 'tiff',
41                   'tif', 'lbm', 'pbm', 'ppm', 'xpm')
42   
43       @staticmethod
44       def can_save(fmt, is_bytesio):
45           if is_bytesio:
46               return False
47           return fmt in ('png', 'jpg')
48   
49       @staticmethod
50       def can_load_memory():
51           return True
52   
53       def load(self, filename):
54           if not filename:
55               import traceback
56               traceback.print_stack()
57               return
58           try:
59               im = None
60               if self._inline:
61                   im = pygame.image.load(filename, 'x.{}'.format(self._ext))
62               elif isfile(filename):
63                   with open(filename, 'rb') as fd:
64                       im = pygame.image.load(fd)
65               elif isinstance(filename, bytes):
66                   try:
67                       fname = filename.decode()
68                       if isfile(fname):
69                           with open(fname, 'rb') as fd:
70                               im = pygame.image.load(fd)
71                   except UnicodeDecodeError:
72                       pass
73               if im is None:
74                   im = pygame.image.load(filename)
75           except:
76               # Logger.warning(type(filename)('Image: Unable to load image <%s>')
77               #               % filename)
78               raise
79   
80           fmt = ''
81           if im.get_bytesize() == 3 and not im.get_colorkey():
82               fmt = 'rgb'
83           elif im.get_bytesize() == 4:
84               fmt = 'rgba'
85   
86           # image loader work only with rgb/rgba image
87           if fmt not in ('rgb', 'rgba'):
88               try:
89                   imc = im.convert(32)
90                   fmt = 'rgba'
91               except:
92                   try:
93                       imc = im.convert_alpha()
94                       fmt = 'rgba'
95                   except:
96                       Logger.warning(
97                           'Image: Unable to convert image %r to rgba (was %r)' %
98                           (filename, im.fmt))
99                       raise
100              im = imc
101  
102          # update internals
103          if not self._inline:
104              self.filename = filename
105          data = pygame.image.tostring(im, fmt.upper())
106          return [ImageData(im.get_width(), im.get_height(),
107                  fmt, data, source=filename)]
108  
109      @staticmethod
110      def save(filename, width, height, pixelfmt, pixels, flipped,
111               imagefmt=None):
112          surface = pygame.image.fromstring(
113              pixels, (width, height), pixelfmt.upper(), flipped)
114          pygame.image.save(surface, filename)
115          return True

117  
118  # register
119  ImageLoader.register(ImageLoaderPygame)
