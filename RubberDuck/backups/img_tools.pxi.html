8    @cython.boundscheck(False)
9    @cython.wraparound(False)
10   cdef inline convert_to_gl_format(data, fmt, width, height):
11       ''' Takes data as a bytes object or an instance that implements the python
12       buffer interface. If the data format is supported by opengl, the data
13       is returned unchanged. Otherwise, the data is converted to a supported
14       format, when possible, and returned as a python array object.
15   
16       Note that conversion is currently only supported for bytes data.
17       '''
18       cdef array ret_array
19       cdef char *src_buffer
20       cdef char *dst_buffer
21       cdef char [::1] view
22       cdef int datasize
23       cdef str ret_format
24       cdef int i, k
25       cdef char c, c2
26       cdef int pitch, rowlen
27       cdef int pitchalign = 0
28   
29       # if native support of this format is available, use it
30       if gl_has_texture_native_format(fmt):
31           return data, fmt
32   
33       # no native support, can we at least convert it ?
34       if not gl_has_texture_conversion(fmt):
35           raise Exception('Unimplemented texture conversion for {}'.format(fmt))
36   
37       # do appropriate conversion, since we accepted it
38       if isinstance(data, bytes):
39           datasize = <int>len(data)
40           ret_array = clone(array('b'), datasize, False)
41           src_buffer = <char *>data
42       else:
43           view = data
44           datasize = view.nbytes
45           ret_array = clone(array('b'), datasize, False)
46           src_buffer = &view[0]
47       dst_buffer = ret_array.data.as_chars
48   
49       # BGR -> RGB
50       if fmt == 'bgr':
51           ret_format = 'rgb'
52           memcpy(dst_buffer, src_buffer, datasize)
53   
54           i = 0
55           rowlen = width * 3
56           pitch = (rowlen + 3) & ~3
57           if rowlen * height < datasize:
58               # FIXME: warn/fail if pitch * height != datasize:
59               pitchalign = pitch - rowlen
60               rowlen -= 1 # to match 0-based k below
61   
62           # note, this is the fastest copying method. copying element by element
63           # from a memoryview is slower then copying the whole buffer and then
64           # properly modifying the elements
65           with nogil:
66               while i < datasize:
67                   c = dst_buffer[i]
68                   k = i + 2
69                   dst_buffer[i] = dst_buffer[k]
70                   dst_buffer[k] = c
71                   if pitchalign and k % rowlen == 0:
72                       i += pitchalign
73                   i += 3
74   
75       # BGRA -> RGBA
76       elif fmt == 'bgra':
77           ret_format = 'rgba'
78           memcpy(dst_buffer, src_buffer, datasize)
79           with nogil:
80               for i in range(0, datasize, 4):
81                   c = dst_buffer[i]
82                   dst_buffer[i] = dst_buffer[i + 2]
83                   dst_buffer[i + 2] = c
84   
85       # ARGB -> RGBA
86       elif fmt == 'argb':
87           ret_format = 'rgba'
88           memcpy(dst_buffer, &src_buffer[1], datasize-1)
89           c2 = src_buffer[0]
90           with nogil:
91               for i in range(0, datasize, 4):
92                   c = dst_buffer[i+3]
93                   dst_buffer[i+3] = c2
94                   c2 = c
95   
96       # ABGR -> RGBA
97       elif fmt == 'abgr':
98           ret_format = 'rgba'
99           memcpy(dst_buffer, &src_buffer[1], datasize-1)
100          c2 = src_buffer[0]
101          with nogil:
102              for i in range(0, datasize, 4):
103                  c = dst_buffer[i+3]
104                  dst_buffer[i+3] = c2
105                  c2 = c
106  
107                  c = dst_buffer[i]
108                  dst_buffer[i] = dst_buffer[i + 2]
109                  dst_buffer[i + 2] = c
110  
111      else:
112          assert False, 'Non implemented texture conversion {}'.format(fmt)
113  
114      return ret_array, ret_format
