13           path = sys.argv[1]
14   
15       query = win32evtlog.EvtQuery(path, win32evtlog.EvtQueryForwardDirection)
16       events = win32evtlog.EvtNext(query, num_events)
17       context = win32evtlog.EvtCreateRenderContext(win32evtlog.EvtRenderContextSystem)
18   
19       for i, event in enumerate(events, 1):
20           result = win32evtlog.EvtRender(
21               event, win32evtlog.EvtRenderEventValues, Context=context
22           )
23   
24           print("Event {}".format(i))
25   
26           level_value, level_variant = result[win32evtlog.EvtSystemLevel]
27           if level_variant != win32evtlog.EvtVarTypeNull:
28               if level_value == 1:
29                   print("    Level: CRITICAL")
30               elif level_value == 2:
31                   print("    Level: ERROR")
32               elif level_value == 3:
33                   print("    Level: WARNING")
34               elif level_value == 4:
35                   print("    Level: INFO")
36               elif level_value == 5:
37                   print("    Level: VERBOSE")
38               else:
39                   print("    Level: UNKNOWN")
40   
41           time_created_value, time_created_variant = result[
42               win32evtlog.EvtSystemTimeCreated
43           ]
44           if time_created_variant != win32evtlog.EvtVarTypeNull:
45               print("    Timestamp: {}".format(time_created_value.isoformat()))
46   
47           computer_value, computer_variant = result[win32evtlog.EvtSystemComputer]
48           if computer_variant != win32evtlog.EvtVarTypeNull:
49               print("    FQDN: {}".format(computer_value))
50   
51           provider_name_value, provider_name_variant = result[
52               win32evtlog.EvtSystemProviderName
53           ]
54           if provider_name_variant != win32evtlog.EvtVarTypeNull:
55               print("    Provider: {}".format(provider_name_value))
56   
57               try:
58                   metadata = win32evtlog.EvtOpenPublisherMetadata(provider_name_value)
59               # pywintypes.error: (2, 'EvtOpenPublisherMetadata', 'The system cannot find the file specified.')
60               except Exception:
61                   pass
62               else:
63                   try:
64                       message = win32evtlog.EvtFormatMessage(
65                           metadata, event, win32evtlog.EvtFormatMessageEvent
66                       )
67                   # pywintypes.error: (15027, 'EvtFormatMessage: allocated 0, need buffer of size 0', 'The message resource is present but the message was not found in the message table.')
68                   except Exception:
69                       pass

71                       try:
72                           print("    Message: {}".format(message))

74                           # Obscure error when run under subprocess.Popen(), presumably due to
75                           # not knowing the correct encoding for the console.

78                           # than ours:
