7    class TypelibSpec:
8        def __init__(self, clsid, lcid, major, minor, flags=0):
9            self.clsid = str(clsid)
10           self.lcid = int(lcid)
11           # We avoid assuming 'major' or 'minor' are integers - when
12           # read from the registry there is some confusion about if
13           # they are base 10 or base 16 (they *should* be base 16, but
14           # how they are written is beyond our control.)
15           self.major = major
16           self.minor = minor
17           self.dll = None
18           self.desc = None
19           self.ver_desc = None
20           self.flags = flags
21   
22       # For the SelectList
23       def __getitem__(self, item):
24           if item == 0:
25               return self.ver_desc
26           raise IndexError("Cant index me!")
27   
28       def __lt__(self, other):  # rich-cmp/py3k-friendly version
29           me = (
30               (self.ver_desc or "").lower(),
31               (self.desc or "").lower(),
32               self.major,
33               self.minor,
34           )
35           them = (
36               (other.ver_desc or "").lower(),
37               (other.desc or "").lower(),
38               other.major,
39               other.minor,
40           )
41           return me < them
42   
43       def __eq__(self, other):  # rich-cmp/py3k-friendly version
44           return (
45               (self.ver_desc or "").lower() == (other.ver_desc or "").lower()
46               and (self.desc or "").lower() == (other.desc or "").lower()
47               and self.major == other.major
48               and self.minor == other.minor
49           )
50   
51       def Resolve(self):
52           if self.dll is None:
53               return 0
54           tlb = pythoncom.LoadTypeLib(self.dll)
55           self.FromTypelib(tlb, None)
56           return 1
57   
58       def FromTypelib(self, typelib, dllName=None):
59           la = typelib.GetLibAttr()
60           self.clsid = str(la[0])
61           self.lcid = la[1]
62           self.major = la[3]
63           self.minor = la[4]
64           if dllName:
65               self.dll = dllName

68   def EnumKeys(root):
69       index = 0
70       ret = []
71       while 1:
72           try:
73               item = win32api.RegEnumKey(root, index)
74           except win32api.error:
75               break
76           try:
77               # Note this doesn't handle REG_EXPAND_SZ, but the implementation
78               # here doesn't need to - that is handled as the data is read.
79               val = win32api.RegQueryValue(root, item)
80           except win32api.error:
81               val = ""  # code using this assumes a string.
82   
83           ret.append((item, val))
84           index = index + 1
85       return ret

93   def EnumTlbs(excludeFlags=0):
94       """Return a list of TypelibSpec objects, one for each registered library."""
95       key = win32api.RegOpenKey(win32con.HKEY_CLASSES_ROOT, "Typelib")
96       iids = EnumKeys(key)
97       results = []
98       for iid, crap in iids:
99           try:
100              key2 = win32api.RegOpenKey(key, str(iid))
101          except win32api.error:
102              # A few good reasons for this, including "access denied".
103              continue
104          for version, tlbdesc in EnumKeys(key2):
105              major_minor = version.split(".", 1)
106              if len(major_minor) < 2:
107                  major_minor.append("0")
108              # For some reason, this code used to assume the values were hex.
109              # This seems to not be true - particularly for CDO 1.21
110              # *sigh* - it appears there are no rules here at all, so when we need
111              # to know the info, we must load the tlb by filename and request it.
112              # The Resolve() method on the TypelibSpec does this.
113              # For this reason, keep the version numbers as strings - that
114              # way we can't be wrong!  Let code that really needs an int to work
115              # out what to do.  FWIW, http://support.microsoft.com/kb/816970 is
116              # pretty clear that they *should* be hex.
117              major = major_minor[0]
118              minor = major_minor[1]
119              key3 = win32api.RegOpenKey(key2, str(version))
120              try:
121                  # The "FLAGS" are at this point
122                  flags = int(win32api.RegQueryValue(key3, "FLAGS"))
123              except (win32api.error, ValueError):
124                  flags = 0
125              if flags & excludeFlags == 0:
126                  for lcid, crap in EnumKeys(key3):
127                      try:
128                          lcid = int(lcid)
129                      except ValueError:  # not an LCID entry
130                          continue
131                      # Check for both "{lcid}\win32" and "{lcid}\win64" keys.
132                      try:
133                          key4 = win32api.RegOpenKey(key3, "%s\\win32" % (lcid,))
134                      except win32api.error:
135                          try:
136                              key4 = win32api.RegOpenKey(key3, "%s\\win64" % (lcid,))
137                          except win32api.error:
138                              continue
139                      try:
140                          dll, typ = win32api.RegQueryValueEx(key4, None)
141                          if typ == win32con.REG_EXPAND_SZ:
142                              dll = win32api.ExpandEnvironmentStrings(dll)
143                      except win32api.error:
144                          dll = None
145                      spec = TypelibSpec(iid, lcid, major, minor, flags)
146                      spec.dll = dll
147                      spec.desc = tlbdesc
148                      spec.ver_desc = tlbdesc + " (" + version + ")"
149                      results.append(spec)
150      return results

153  def FindTlbsWithDescription(desc):
154      """Find all installed type libraries with the specified description"""
155      ret = []
156      items = EnumTlbs()
157      for item in items:
158          if item.desc == desc:
159              ret.append(item)
160      return ret

163  def SelectTlb(title="Select Library", excludeFlags=0):
164      """Display a list of all the type libraries, and select one.   Returns None if cancelled"""
165      import pywin.dialogs.list
166  
167      items = EnumTlbs(excludeFlags)
168      # fixup versions - we assume hex (see __init__ above)
169      for i in items:
170          i.major = int(i.major, 16)
171          i.minor = int(i.minor, 16)
172      items.sort()
173      rc = pywin.dialogs.list.SelectFromLists(title, items, ["Type Library"])
174      if rc is None:
175          return None
176      return items[rc]
