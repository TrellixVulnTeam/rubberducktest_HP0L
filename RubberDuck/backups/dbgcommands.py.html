1    # Command Handlers for the debugger.
2    
3    # Not in the debugger package, as I always want these interfaces to be
4    # available, even if the debugger has not yet been (or can not be)
5    # imported
6    import win32ui, win32con
7    from . import scriptutils
8    import warnings
9    from pywin.scintilla.control import CScintillaEditInterface
10   
11   IdToBarNames = {
12       win32ui.IDC_DBG_STACK: ("Stack", 0),
13       win32ui.IDC_DBG_BREAKPOINTS: ("Breakpoints", 0),
14       win32ui.IDC_DBG_WATCH: ("Watch", 1),
15   }
16   
17   
18   class DebuggerCommandHandler:
19       def HookCommands(self):
20           commands = (
21               (self.OnStep, None, win32ui.IDC_DBG_STEP),
22               (self.OnStepOut, self.OnUpdateOnlyBreak, win32ui.IDC_DBG_STEPOUT),
23               (self.OnStepOver, None, win32ui.IDC_DBG_STEPOVER),
24               (self.OnGo, None, win32ui.IDC_DBG_GO),
25               (self.OnClose, self.OnUpdateClose, win32ui.IDC_DBG_CLOSE),
26               (self.OnAdd, self.OnUpdateAddBreakpoints, win32ui.IDC_DBG_ADD),
27               (self.OnClearAll, self.OnUpdateClearAllBreakpoints, win32ui.IDC_DBG_CLEAR),
28               #                         (self.OnDebuggerToolbar, self.OnUpdateDebuggerToolbar, win32ui.ID_DEBUGGER_TOOLBAR),
29           )
30   
31           frame = win32ui.GetMainFrame()
32   
33           for methHandler, methUpdate, id in commands:
34               frame.HookCommand(methHandler, id)
35               if not methUpdate is None:
36                   frame.HookCommandUpdate(methUpdate, id)
37   
38           for id in list(IdToBarNames.keys()):
39               frame.HookCommand(self.OnDebuggerBar, id)
40               frame.HookCommandUpdate(self.OnUpdateDebuggerBar, id)
41   
42       def OnDebuggerToolbar(self, id, code):
43           if code == 0:
44               return not win32ui.GetMainFrame().OnBarCheck(id)
45   
46       def OnUpdateDebuggerToolbar(self, cmdui):
47           win32ui.GetMainFrame().OnUpdateControlBarMenu(cmdui)
48           cmdui.Enable(1)
49   
50       def _GetDebugger(self):
51           try:
52               import pywin.debugger
53   
54               return pywin.debugger.currentDebugger
55           except ImportError:
56               return None
57   
58       def _DoOrStart(self, doMethod, startFlag):
59           d = self._GetDebugger()
60           if d is not None and d.IsDebugging():
61               method = getattr(d, doMethod)
62               method()
63           else:
64               scriptutils.RunScript(
65                   defName=None, defArgs=None, bShowDialog=0, debuggingType=startFlag
66               )
67   
68       def OnStep(self, msg, code):
69           self._DoOrStart("do_set_step", scriptutils.RS_DEBUGGER_STEP)
70   
71       def OnStepOver(self, msg, code):
72           self._DoOrStart("do_set_next", scriptutils.RS_DEBUGGER_STEP)
73   
74       def OnStepOut(self, msg, code):
75           d = self._GetDebugger()
76           if d is not None and d.IsDebugging():
77               d.do_set_return()
78   
79       def OnGo(self, msg, code):
80           self._DoOrStart("do_set_continue", scriptutils.RS_DEBUGGER_GO)
81   
82       def OnClose(self, msg, code):
83           d = self._GetDebugger()
84           if d is not None:
85               if d.IsDebugging():
86                   d.set_quit()
87               else:
88                   d.close()
89   
90       def OnUpdateClose(self, cmdui):
91           d = self._GetDebugger()
92           if d is not None and d.inited:
93               cmdui.Enable(1)
94           else:
95               cmdui.Enable(0)
96   
97       def OnAdd(self, msg, code):
98           doc, view = scriptutils.GetActiveEditorDocument()
99           if doc is None:
100              ## Don't do a messagebox, as this could be triggered from the app's
101              ## idle loop whenever the debug toolbar is visible, giving a never-ending
102              ## series of dialogs.  This can happen when the OnUpdate handler
103              ## for the toolbar button IDC_DBG_ADD fails, since MFC falls back to
104              ## sending a normal command if the UI update command fails.
105              ## win32ui.MessageBox('There is no active window - no breakpoint can be added')
106              warnings.warn("There is no active window - no breakpoint can be added")
107              return None
108          pathName = doc.GetPathName()
109          lineNo = view.LineFromChar(view.GetSel()[0]) + 1
110          # If I have a debugger, then tell it, otherwise just add a marker
111          d = self._GetDebugger()
112          if d is None:
113              import pywin.framework.editor.color.coloreditor
114  
115              doc.MarkerToggle(
116                  lineNo, pywin.framework.editor.color.coloreditor.MARKER_BREAKPOINT
117              )
118          else:
119              if d.get_break(pathName, lineNo):
120                  win32ui.SetStatusText("Clearing breakpoint", 1)
121                  rc = d.clear_break(pathName, lineNo)
122              else:
123                  win32ui.SetStatusText("Setting breakpoint", 1)
124                  rc = d.set_break(pathName, lineNo)
125              if rc:
126                  win32ui.MessageBox(rc)
127              d.GUIRespondDebuggerData()
128  
129      def OnClearAll(self, msg, code):
130          win32ui.SetStatusText("Clearing all breakpoints")
131          d = self._GetDebugger()
132          if d is None:
133              import pywin.framework.editor
134              import pywin.framework.editor.color.coloreditor
135  
136              for doc in pywin.framework.editor.editorTemplate.GetDocumentList():
137                  doc.MarkerDeleteAll(
138                      pywin.framework.editor.color.coloreditor.MARKER_BREAKPOINT
139                  )
140          else:
141              d.clear_all_breaks()
142              d.UpdateAllLineStates()
143              d.GUIRespondDebuggerData()
144  
145      def OnUpdateOnlyBreak(self, cmdui):
146          d = self._GetDebugger()
147          ok = d is not None and d.IsBreak()
148          cmdui.Enable(ok)
149  
150      def OnUpdateAddBreakpoints(self, cmdui):
151          doc, view = scriptutils.GetActiveEditorDocument()
152          if doc is None or not isinstance(view, CScintillaEditInterface):
153              enabled = 0
154          else:
155              enabled = 1
156              lineNo = view.LineFromChar(view.GetSel()[0]) + 1
157              import pywin.framework.editor.color.coloreditor
158  
159              cmdui.SetCheck(
160                  doc.MarkerAtLine(
161                      lineNo, pywin.framework.editor.color.coloreditor.MARKER_BREAKPOINT
162                  )
163                  != 0
164              )
165          cmdui.Enable(enabled)
166  
167      def OnUpdateClearAllBreakpoints(self, cmdui):
168          d = self._GetDebugger()
169          cmdui.Enable(d is None or len(d.breaks) != 0)
170  
171      def OnUpdateDebuggerBar(self, cmdui):
172          name, always = IdToBarNames.get(cmdui.m_nID)
173          enabled = always
174          d = self._GetDebugger()
175          if d is not None and d.IsDebugging() and name is not None:
176              enabled = 1
177              bar = d.GetDebuggerBar(name)
178              cmdui.SetCheck(bar.IsWindowVisible())
179          cmdui.Enable(enabled)
180  
181      def OnDebuggerBar(self, id, code):
182          name = IdToBarNames.get(id)[0]
183          d = self._GetDebugger()
184          if d is not None and name is not None:
185              bar = d.GetDebuggerBar(name)
186              newState = not bar.IsWindowVisible()
187              win32ui.GetMainFrame().ShowControlBar(bar, newState, 1)
