1    # The property page to define generic IDE options for Pythonwin
2    
3    from pywin.mfc import dialog
4    from pywin.framework import interact
5    import win32ui
6    import win32con
7    
8    buttonControlMap = {
9        win32ui.IDC_BUTTON1: win32ui.IDC_EDIT1,
10       win32ui.IDC_BUTTON2: win32ui.IDC_EDIT2,
11       win32ui.IDC_BUTTON3: win32ui.IDC_EDIT3,
12   }
13   
14   
15   class OptionsPropPage(dialog.PropertyPage):
16       def __init__(self):
17           dialog.PropertyPage.__init__(self, win32ui.IDD_PP_IDE)
18           self.AddDDX(win32ui.IDC_CHECK1, "bShowAtStartup")
19           self.AddDDX(win32ui.IDC_CHECK2, "bDocking")
20           self.AddDDX(win32ui.IDC_EDIT4, "MRUSize", "i")
21   
22       def OnInitDialog(self):
23   
24           edit = self.GetDlgItem(win32ui.IDC_EDIT1)
25           format = eval(
26               win32ui.GetProfileVal(
27                   interact.sectionProfile,
28                   interact.STYLE_INTERACTIVE_PROMPT,
29                   str(interact.formatInput),
30               )
31           )
32           edit.SetDefaultCharFormat(format)
33           edit.SetWindowText("Input Text")
34   
35           edit = self.GetDlgItem(win32ui.IDC_EDIT2)
36           format = eval(
37               win32ui.GetProfileVal(
38                   interact.sectionProfile,
39                   interact.STYLE_INTERACTIVE_OUTPUT,
40                   str(interact.formatOutput),
41               )
42           )
43           edit.SetDefaultCharFormat(format)
44           edit.SetWindowText("Output Text")
45   
46           edit = self.GetDlgItem(win32ui.IDC_EDIT3)
47           format = eval(
48               win32ui.GetProfileVal(
49                   interact.sectionProfile,
50                   interact.STYLE_INTERACTIVE_ERROR,
51                   str(interact.formatOutputError),
52               )
53           )
54           edit.SetDefaultCharFormat(format)
55           edit.SetWindowText("Error Text")
56   
57           self["bShowAtStartup"] = interact.LoadPreference("Show at startup", 1)
58           self["bDocking"] = interact.LoadPreference("Docking", 0)
59           self["MRUSize"] = win32ui.GetProfileVal("Settings", "Recent File List Size", 10)
60   
61           # Hook the button clicks.
62           self.HookCommand(self.HandleCharFormatChange, win32ui.IDC_BUTTON1)
63           self.HookCommand(self.HandleCharFormatChange, win32ui.IDC_BUTTON2)
64           self.HookCommand(self.HandleCharFormatChange, win32ui.IDC_BUTTON3)
65   
66           # Ensure the spin control remains in range.
67           spinner = self.GetDlgItem(win32ui.IDC_SPIN1)
68           spinner.SetRange(1, 16)
69   
70           return dialog.PropertyPage.OnInitDialog(self)
71   
72       # Called to save away the new format tuple for the specified item.
73       def HandleCharFormatChange(self, id, code):
74           if code == win32con.BN_CLICKED:
75               editId = buttonControlMap.get(id)
76               assert editId is not None, "Format button has no associated edit control"
77               editControl = self.GetDlgItem(editId)
78               existingFormat = editControl.GetDefaultCharFormat()
79               flags = win32con.CF_SCREENFONTS
80               d = win32ui.CreateFontDialog(existingFormat, flags, None, self)
81               if d.DoModal() == win32con.IDOK:
82                   cf = d.GetCharFormat()
83                   editControl.SetDefaultCharFormat(cf)
84                   self.SetModified(1)
85               return 0  # We handled this fully!
86   
87       def OnOK(self):
88           # Handle the edit controls - get all the fonts, put them back into interact, then
89           # get interact to save its stuff!
90           controlAttrs = [
91               (win32ui.IDC_EDIT1, interact.STYLE_INTERACTIVE_PROMPT),
92               (win32ui.IDC_EDIT2, interact.STYLE_INTERACTIVE_OUTPUT),
93               (win32ui.IDC_EDIT3, interact.STYLE_INTERACTIVE_ERROR),
94           ]
95           for id, key in controlAttrs:
96               control = self.GetDlgItem(id)
97               fmt = control.GetDefaultCharFormat()
98               win32ui.WriteProfileVal(interact.sectionProfile, key, str(fmt))
99   
100          # Save the other interactive window options.
101          interact.SavePreference("Show at startup", self["bShowAtStartup"])
102          interact.SavePreference("Docking", self["bDocking"])
103  
104          # And the other options.
105          win32ui.WriteProfileVal("Settings", "Recent File List Size", self["MRUSize"])
106  
107          return 1

109      def ChangeFormat(self, fmtAttribute, fmt):
110          dlg = win32ui.CreateFontDialog(fmt)
111          if dlg.DoModal() != win32con.IDOK:
112              return None
113          return dlg.GetCharFormat()

115      def OnFormatTitle(self, command, code):
116          fmt = self.GetFormat(interact.formatTitle)
117          if fmt:
118              formatTitle = fmt
119              SaveFontPreferences()
120  
121      def OnFormatInput(self, command, code):
122          global formatInput
123          fmt = self.GetFormat(formatInput)
124          if fmt:
125              formatInput = fmt
126              SaveFontPreferences()
127  
128      def OnFormatOutput(self, command, code):
129          global formatOutput
130          fmt = self.GetFormat(formatOutput)
131          if fmt:
132              formatOutput = fmt
133              SaveFontPreferences()
134  
135      def OnFormatError(self, command, code):
136          global formatOutputError
137          fmt = self.GetFormat(formatOutputError)
138          if fmt:
139              formatOutputError = fmt
140              SaveFontPreferences()
