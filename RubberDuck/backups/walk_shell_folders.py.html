6    def walk(folder, depth=2, indent=""):
7        try:
8            pidls = folder.EnumObjects(0, shellcon.SHCONTF_FOLDERS)
9        except shell.error:
10           # no items
11           return
12       for pidl in pidls:
13           dn = folder.GetDisplayNameOf(pidl, shellcon.SHGDN_NORMAL)
14           print(indent, dn)
15           if depth:
16               try:
17                   child = folder.BindToObject(pidl, None, shell.IID_IShellFolder)
18               except shell.error:
19                   pass
20               else:
21                   walk(child, depth - 1, indent + " ")
