32   class GestureContainer(EventDispatcher):
33       '''Container object that stores information about a gesture. It has 
34       various properties that are updated by `GestureSurface` as drawing 
35       progresses. 
36    
37       :Arguments: 
38           `touch` 
39               Touch object (as received by on_touch_down) used to initialize 
40               the gesture container. Required. 
41    
42       :Properties: 
43           `active` 
44               Set to False once the gesture is complete (meets 
45               `max_stroke` setting or `GestureSurface.temporal_window`) 
46    
47               :attr:`active` is a 
48               :class:`~kivy.properties.BooleanProperty` 
49    
50           `active_strokes` 
51               Number of strokes currently active in the gesture, ie 
52               concurrent touches associated with this gesture. 
53    
54               :attr:`active_strokes` is a 
55               :class:`~kivy.properties.NumericProperty` 
56    
57           `max_strokes` 
58               Max number of strokes allowed in the gesture. This 
59               is set by `GestureSurface.max_strokes` but can 
60               be overridden for example from `on_gesture_start`. 
61    
62               :attr:`max_strokes` is a 
63               :class:`~kivy.properties.NumericProperty` 
64    
65           `was_merged` 
66               Indicates that this gesture has been merged with another 
67               gesture and should be considered discarded. 
68    
69               :attr:`was_merged` is a 
70               :class:`~kivy.properties.BooleanProperty` 
71    
72           `bbox` 
73               Dictionary with keys minx, miny, maxx, maxy. Represents the size 
74               of the gesture bounding box. 
75    
76               :attr:`bbox` is a 
77               :class:`~kivy.properties.DictProperty` 
78    
79           `width` 
80               Represents the width of the gesture. 
81    
82               :attr:`width` is a 
83               :class:`~kivy.properties.NumericProperty` 
84    
85           `height` 
86               Represents the height of the gesture. 
87    
88               :attr:`height` is a 
89               :class:`~kivy.properties.NumericProperty` 
90       '''
91       active = BooleanProperty(True)
92       active_strokes = NumericProperty(0)
93       max_strokes = NumericProperty(0)
94       was_merged = BooleanProperty(False)
95       bbox = DictProperty({'minx': float('inf'), 'miny': float('inf'),
96                            'maxx': float('-inf'), 'maxy': float('-inf')})
97       width = NumericProperty(0)
98       height = NumericProperty(0)
99   
100      def __init__(self, touch, **kwargs):
101          # The color is applied to all canvas items of this gesture
102          self.color = kwargs.pop('color', [1., 1., 1.])
103  
104          super(GestureContainer, self).__init__(**kwargs)
105  
106          # This is the touch.uid of the oldest touch represented
107          self.id = str(touch.uid)
108  
109          # Store various timestamps for decision making
110          self._create_time = Clock.get_time()
111          self._update_time = None
112          self._cleanup_time = None
113          self._cache_time = 0
114  
115          # We can cache the candidate here to save zip()/Vector instantiation
116          self._vectors = None
117  
118          # Key is touch.uid; value is a kivy.graphics.Line(); it's used even
119          # if line_width is 0 (i.e. not actually drawn anywhere)
120          self._strokes = {}
121  
122          # Make sure the bbox is up to date with the first touch position
123          self.update_bbox(touch)
124  
125      def get_vectors(self, **kwargs):
126          '''Return strokes in a format that is acceptable for 
127          `kivy.multistroke.Recognizer` as a gesture candidate or template. The 
128          result is cached automatically; the cache is invalidated at the start 
129          and end of a stroke and if `update_bbox` is called. If you are going 
130          to analyze a gesture mid-stroke, you may need to set the `no_cache` 
131          argument to True.'''
132          if self._cache_time == self._update_time and \
133                  not kwargs.get('no_cache'):
134              return self._vectors
135  
136          vecs = []
137          append = vecs.append
138          for tuid, l in self._strokes.items():
139              lpts = l.points
140              append([Vector(*pts) for pts in zip(lpts[::2], lpts[1::2])])
141  
142          self._vectors = vecs
143          self._cache_time = self._update_time
144          return vecs
145  
146      def handles(self, touch):
147          '''Returns True if this container handles the given touch'''
148          if not self.active:
149              return False
150          return str(touch.uid) in self._strokes
151  
152      def accept_stroke(self, count=1):
153          '''Returns True if this container can accept `count` new strokes'''
154          if not self.max_strokes:
155              return True
156          return len(self._strokes) + count <= self.max_strokes
157  
158      def update_bbox(self, touch):
159          '''Update gesture bbox from a touch coordinate'''
160          x, y = touch.x, touch.y
161          bb = self.bbox
162          if x < bb['minx']:
163              bb['minx'] = x
164          if y < bb['miny']:
165              bb['miny'] = y
166          if x > bb['maxx']:
167              bb['maxx'] = x
168          if y > bb['maxy']:
169              bb['maxy'] = y
170          self.width = bb['maxx'] - bb['minx']
171          self.height = bb['maxy'] - bb['miny']
172          self._update_time = Clock.get_time()
173  
174      def add_stroke(self, touch, line):
175          '''Associate a list of points with a touch.uid; the line itself is 
176          created by the caller, but subsequent move/up events look it 
177          up via us. This is done to avoid problems during merge.'''
178          self._update_time = Clock.get_time()
179          self._strokes[str(touch.uid)] = line
180          self.active_strokes += 1
181  
182      def complete_stroke(self):
183          '''Called on touch up events to keep track of how many strokes 
184          are active in the gesture (we only want to dispatch event when 
185          the *last* stroke in the gesture is released)'''
186          self._update_time = Clock.get_time()
187          self.active_strokes -= 1
188  
189      def single_points_test(self):
190          '''Returns True if the gesture consists only of single-point strokes, 
191          we must discard it in this case, or an exception will be raised'''
192          for tuid, l in self._strokes.items():
193              if len(l.points) > 2:
194                  return False
195          return True

198  class GestureSurface(FloatLayout):
199      '''Simple gesture surface to track/draw touch movements. Typically used 
200      to gather user input suitable for :class:`kivy.multistroke.Recognizer`. 
201   
202      :Properties: 
203          `temporal_window` 
204              Time to wait from the last touch_up event before attempting 
205              to recognize the gesture. If you set this to 0, the 
206              `on_gesture_complete` event is not fired unless the 
207              :attr:`max_strokes` condition is met. 
208   
209              :attr:`temporal_window` is a 
210              :class:`~kivy.properties.NumericProperty` and defaults to 2.0 
211   
212          `max_strokes` 
213              Max number of strokes in a single gesture; if this is reached, 
214              recognition will start immediately on the final touch_up event. 
215              If this is set to 0, the `on_gesture_complete` event is not 
216              fired unless the :attr:`temporal_window` expires. 
217   
218              :attr:`max_strokes` is a 
219              :class:`~kivy.properties.NumericProperty` and defaults to 2.0 
220   
221          `bbox_margin` 
222              Bounding box margin for detecting gesture collisions, in 
223              pixels. 
224   
225              :attr:`bbox_margin` is a 
226              :class:`~kivy.properties.NumericProperty` and defaults to 30 
227   
228          `draw_timeout` 
229              Number of seconds to keep lines/bbox on canvas after the 
230              `on_gesture_complete` event is fired. If this is set to 0, 
231              gestures are immediately removed from the surface when 
232              complete. 
233   
234              :attr:`draw_timeout` is a 
235              :class:`~kivy.properties.NumericProperty` and defaults to 3.0 
236   
237          `color` 
238              Color used to draw the gesture, in RGB. This option does not 
239              have an effect if :attr:`use_random_color` is True. 
240   
241              :attr:`color` is a 
242              :class:`~kivy.properties.ColorProperty` and defaults to 
243              [1, 1, 1, 1] (white) 
244   
245              .. versionchanged:: 2.0.0 
246                  Changed from :class:`~kivy.properties.ListProperty` to 
247                  :class:`~kivy.properties.ColorProperty`. 
248   
249          `use_random_color` 
250              Set to True to pick a random color for each gesture, if you do 
251              this then `color` is ignored. Defaults to False. 
252   
253              :attr:`use_random_color` is a 
254              :class:`~kivy.properties.BooleanProperty` and defaults to False 
255   
256          `line_width` 
257              Line width used for tracing touches on the surface. Set to 0 
258              if you only want to detect gestures without drawing anything. 
259              If you use 1.0, OpenGL GL_LINE is used for drawing; values > 1 
260              will use an internal drawing method based on triangles (less 
261              efficient), see :mod:`kivy.graphics`. 
262   
263              :attr:`line_width` is a 
264              :class:`~kivy.properties.NumericProperty` and defaults to 2 
265   
266          `draw_bbox` 
267              Set to True if you want to draw bounding box behind gestures. 
268              This only works if `line_width` >= 1. Default is False. 
269   
270              :attr:`draw_bbox` is a 
271              :class:`~kivy.properties.BooleanProperty` and defaults to True 
272   
273          `bbox_alpha` 
274              Opacity for bounding box if `draw_bbox` is True. Default 0.1 
275   
276              :attr:`bbox_alpha` is a 
277              :class:`~kivy.properties.NumericProperty` and defaults to 0.1 
278   
279      :Events: 
280          `on_gesture_start` :class:`GestureContainer` 
281              Fired when a new gesture is initiated on the surface, i.e. the 
282              first on_touch_down that does not collide with an existing 
283              gesture on the surface. 
284   
285          `on_gesture_extend` :class:`GestureContainer` 
286              Fired when a touch_down event occurs within an existing gesture. 
287   
288          `on_gesture_merge` :class:`GestureContainer`, :class:`GestureContainer` 
289              Fired when two gestures collide and get merged to one gesture. 
290              The first argument is the gesture that has been merged (no longer 
291              valid); the second is the combined (resulting) gesture. 
292   
293          `on_gesture_complete` :class:`GestureContainer` 
294              Fired when a set of strokes is considered a complete gesture, 
295              this happens when `temporal_window` expires or `max_strokes` 
296              is reached. Typically you will bind to this event and use 
297              the provided `GestureContainer` get_vectors() method to 
298              match against your gesture database. 
299   
300          `on_gesture_cleanup` :class:`GestureContainer` 
301              Fired `draw_timeout` seconds after `on_gesture_complete`, 
302              The gesture will be removed from the canvas (if line_width > 0 or 
303              draw_bbox is True) and the internal gesture list before this. 
304   
305          `on_gesture_discard` :class:`GestureContainer` 
306              Fired when a gesture does not meet the minimum size requirements 
307              for recognition (width/height < 5, or consists only of single- 
308              point strokes). 
309      '''
310  
311      temporal_window = NumericProperty(2.0)
312      draw_timeout = NumericProperty(3.0)
313      max_strokes = NumericProperty(4)
314      bbox_margin = NumericProperty(30)
315  
316      line_width = NumericProperty(2)
317      color = ColorProperty([1., 1., 1., 1.])
318      use_random_color = BooleanProperty(False)
319      draw_bbox = BooleanProperty(False)
320      bbox_alpha = NumericProperty(0.1)
321  
322      def __init__(self, **kwargs):
323          super(GestureSurface, self).__init__(**kwargs)
324          # A list of GestureContainer objects (all gestures on the surface)
325          self._gestures = []
326          self.register_event_type('on_gesture_start')
327          self.register_event_type('on_gesture_extend')
328          self.register_event_type('on_gesture_merge')
329          self.register_event_type('on_gesture_complete')
330          self.register_event_type('on_gesture_cleanup')
331          self.register_event_type('on_gesture_discard')

334  # Touch Events

336      def on_touch_down(self, touch):
337          '''When a new touch is registered, the first thing we do is to test if 
338          it collides with the bounding box of another known gesture. If so, it 
339          is assumed to be part of that gesture. 
340          '''
341          # If the touch originates outside the surface, ignore it.
342          if not self.collide_point(touch.x, touch.y):
343              return
344  
345          touch.grab(self)
346  
347          # Add the stroke to existing gesture, or make a new one
348          g = self.find_colliding_gesture(touch)
349          new = False
350          if g is None:
351              g = self.init_gesture(touch)
352              new = True
353  
354          # We now belong to a gesture (new or old); start a new stroke.
355          self.init_stroke(g, touch)
356  
357          if new:
358              self.dispatch('on_gesture_start', g, touch)
359          else:
360              self.dispatch('on_gesture_extend', g, touch)
361  
362          return True
363  
364      def on_touch_move(self, touch):
365          '''When a touch moves, we add a point to the line on the canvas so the 
366          path is updated. We must also check if the new point collides with the 
367          bounding box of another gesture - if so, they should be merged.'''
368          if touch.grab_current is not self:
369              return
370          if not self.collide_point(touch.x, touch.y):
371              return
372  
373          # Retrieve the GestureContainer object that handles this touch, and
374          # test for colliding gestures. If found, merge them to one.
375          g = self.get_gesture(touch)
376          collision = self.find_colliding_gesture(touch)
377          if collision is not None and g.accept_stroke(len(collision._strokes)):
378              merge = self.merge_gestures(g, collision)
379              if g.was_merged:
380                  self.dispatch('on_gesture_merge', g, collision)
381              else:
382                  self.dispatch('on_gesture_merge', collision, g)
383              g = merge
384          else:
385              g.update_bbox(touch)
386  
387          # Add the new point to gesture stroke list and update the canvas line
388          g._strokes[str(touch.uid)].points += (touch.x, touch.y)
389  
390          # Draw the gesture bounding box; if it is a single press that
391          # does not trigger a move event, we would miss it otherwise.
392          if self.draw_bbox:
393              self._update_canvas_bbox(g)
394          return True
395  
396      def on_touch_up(self, touch):
397          if touch.grab_current is not self:
398              return
399          touch.ungrab(self)
400  
401          g = self.get_gesture(touch)
402          g.complete_stroke()
403  
404          # If this stroke hit the maximum limit, dispatch immediately
405          if not g.accept_stroke():
406              self._complete_dispatcher(0)
407  
408          # dispatch later only if we have a window
409          elif self.temporal_window > 0:
410              Clock.schedule_once(self._complete_dispatcher,
411                                      self.temporal_window)

416      def init_gesture(self, touch):
417          '''Create a new gesture from touch, i.e. it's the first on 
418          surface, or was not close enough to any existing gesture (yet)'''
419          col = self.color
420          if self.use_random_color:
421              col = hsv_to_rgb(random(), 1., 1.)
422  
423          g = GestureContainer(touch, max_strokes=self.max_strokes, color=col)
424  
425          # Create the bounding box Rectangle for the gesture
426          if self.draw_bbox:
427              bb = g.bbox
428              with self.canvas:
429                  Color(col[0], col[1], col[2], self.bbox_alpha, mode='rgba',
430                        group=g.id)
431  
432                  g._bbrect = Rectangle(
433                      group=g.id,
434                      pos=(bb['minx'], bb['miny']),
435                      size=(bb['maxx'] - bb['minx'],
436                            bb['maxy'] - bb['miny']))
437  
438          self._gestures.append(g)
439          return g
440  
441      def init_stroke(self, g, touch):
442          points = [touch.x, touch.y]
443          col = g.color
444  
445          new_line = Line(
446              points=points,
447              width=self.line_width,
448              group=g.id)
449          g._strokes[str(touch.uid)] = new_line
450  
451          if self.line_width:
452              canvas_add = self.canvas.add
453              canvas_add(Color(col[0], col[1], col[2], mode='rgb', group=g.id))
454              canvas_add(new_line)
455  
456          # Update the bbox in case; this will normally be done in on_touch_move,
457          # but we want to update it also for a single press, force that here:
458          g.update_bbox(touch)
459          if self.draw_bbox:
460              self._update_canvas_bbox(g)
461  
462          # Register the stroke in GestureContainer so we can look it up later
463          g.add_stroke(touch, new_line)
464  
465      def get_gesture(self, touch):
466          '''Returns GestureContainer associated with given touch'''
467          for g in self._gestures:
468              if g.active and g.handles(touch):
469                  return g
470          raise Exception('get_gesture() failed to identify ' + str(touch.uid))
471  
472      def find_colliding_gesture(self, touch):
473          '''Checks if a touch x/y collides with the bounding box of an existing 
474          gesture. If so, return it (otherwise returns None) 
475          '''
476          touch_x, touch_y = touch.pos
477          for g in self._gestures:
478              if g.active and not g.handles(touch) and g.accept_stroke():
479                  bb = g.bbox
480                  margin = self.bbox_margin
481                  minx = bb['minx'] - margin
482                  miny = bb['miny'] - margin
483                  maxx = bb['maxx'] + margin
484                  maxy = bb['maxy'] + margin
485                  if minx <= touch_x <= maxx and miny <= touch_y <= maxy:
486                      return g
487          return None
488  
489      def merge_gestures(self, g, other):
490          '''Merges two gestures together, the oldest one is retained and the 
491          newer one gets the `GestureContainer.was_merged` flag raised.'''
492          # Swap order depending on gesture age (the merged gesture gets
493          # the color from the oldest one of the two).
494          swap = other._create_time < g._create_time
495          a = swap and other or g
496          b = swap and g or other
497  
498          # Apply the outer limits of bbox to the merged gesture
499          abbox = a.bbox
500          bbbox = b.bbox
501          if bbbox['minx'] < abbox['minx']:
502              abbox['minx'] = bbbox['minx']
503          if bbbox['miny'] < abbox['miny']:
504              abbox['miny'] = bbbox['miny']
505          if bbbox['maxx'] > abbox['maxx']:
506              abbox['maxx'] = bbbox['maxx']
507          if bbbox['maxy'] > abbox['maxy']:
508              abbox['maxy'] = bbbox['maxy']
509  
510          # Now transfer the coordinates from old to new gesture;
511          # FIXME: This can probably be copied more efficiently?
512          astrokes = a._strokes
513          lw = self.line_width
514          a_id = a.id
515          col = a.color
516  
517          self.canvas.remove_group(b.id)
518          canv_add = self.canvas.add
519          for uid, old in b._strokes.items():
520              # FIXME: Can't figure out how to change group= for existing Line()
521              new_line = Line(
522                  points=old.points,
523                  width=old.width,
524                  group=a_id)
525              astrokes[uid] = new_line
526              if lw:
527                  canv_add(Color(col[0], col[1], col[2], mode='rgb', group=a_id))
528                  canv_add(new_line)
529  
530          b.active = False
531          b.was_merged = True
532          a.active_strokes += b.active_strokes
533          a._update_time = Clock.get_time()
534          return a
535  
536      def _update_canvas_bbox(self, g):
537          # If draw_bbox is changed while two gestures are active,
538          # we might not have a bbrect member
539          if not hasattr(g, '_bbrect'):
540              return
541  
542          bb = g.bbox
543          g._bbrect.pos = (bb['minx'], bb['miny'])
544          g._bbrect.size = (bb['maxx'] - bb['minx'],
545                            bb['maxy'] - bb['miny'])
546

548  # Timeout callbacks

550      def _complete_dispatcher(self, dt):
551          '''This method is scheduled on all touch up events. It will dispatch 
552          the `on_gesture_complete` event for all completed gestures, and remove 
553          merged gestures from the internal gesture list.'''
554          need_cleanup = False
555          gest = self._gestures
556          timeout = self.draw_timeout
557          twin = self.temporal_window
558          get_time = Clock.get_time
559  
560          for idx, g in enumerate(gest):
561              # Gesture is part of another gesture, just delete it
562              if g.was_merged:
563                  del gest[idx]
564                  continue
565  
566              # Not active == already handled, or has active strokes (it cannot
567              # possibly be complete). Proceed to next gesture on surface.
568              if not g.active or g.active_strokes != 0:
569                  continue
570  
571              t1 = g._update_time + twin
572              t2 = get_time() + UNDERSHOOT_MARGIN
573  
574              # max_strokes reached, or temporal window has expired. The gesture
575              # is complete; need to dispatch _complete or _discard event.
576              if not g.accept_stroke() or t1 <= t2:
577                  discard = False
578                  if g.width < 5 and g.height < 5:
579                      discard = True
580                  elif g.single_points_test():
581                      discard = True
582  
583                  need_cleanup = True
584                  g.active = False
585                  g._cleanup_time = get_time() + timeout
586  
587                  if discard:
588                      self.dispatch('on_gesture_discard', g)
589                  else:
590                      self.dispatch('on_gesture_complete', g)
591  
592          if need_cleanup:
593              Clock.schedule_once(self._cleanup, timeout)
594  
595      def _cleanup(self, dt):
596          '''This method is scheduled from _complete_dispatcher to clean up the 
597          canvas and internal gesture list after a gesture is completed.'''
598          m = UNDERSHOOT_MARGIN
599          rg = self.canvas.remove_group
600          gestures = self._gestures
601          for idx, g in enumerate(gestures):
602              if g._cleanup_time is None:
603                  continue
604              if g._cleanup_time <= Clock.get_time() + m:
605                  rg(g.id)
606                  del gestures[idx]
607                  self.dispatch('on_gesture_cleanup', g)
608
