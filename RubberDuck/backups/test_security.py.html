10   class SecurityTests(unittest.TestCase):
11       def setUp(self):
12           self.pwr_sid = win32security.LookupAccountName("", "Power Users")[0]
13           try:
14               self.admin_sid = win32security.LookupAccountName("", "Administrator")[0]
15           except pywintypes.error as exc:
16               # in automation we see:
17               # pywintypes.error: (1332, 'LookupAccountName', 'No mapping between account names and security IDs was done.')
18               if exc.winerror != winerror.ERROR_NONE_MAPPED:
19                   raise
20               self.admin_sid = None
21   
22       def tearDown(self):
23           pass
24   
25       def testEqual(self):
26           if self.admin_sid is None:
27               raise TestSkipped("No 'Administrator' account is available")
28           self.failUnlessEqual(
29               win32security.LookupAccountName("", "Administrator")[0],
30               win32security.LookupAccountName("", "Administrator")[0],
31           )
32   
33       def testNESID(self):
34           self.failUnless(self.pwr_sid == self.pwr_sid)
35           if self.admin_sid:
36               self.failUnless(self.pwr_sid != self.admin_sid)
37   
38       def testNEOther(self):
39           self.failUnless(self.pwr_sid != None)
40           self.failUnless(None != self.pwr_sid)
41           self.failIf(self.pwr_sid == None)
42           self.failIf(None == self.pwr_sid)
43           self.failIfEqual(None, self.pwr_sid)
44   
45       def testSIDInDict(self):
46           d = dict(foo=self.pwr_sid)
47           self.failUnlessEqual(d["foo"], self.pwr_sid)
48   
49       def testBuffer(self):
50           if self.admin_sid is None:
51               raise TestSkipped("No 'Administrator' account is available")
52           self.failUnlessEqual(
53               ob2memory(win32security.LookupAccountName("", "Administrator")[0]),
54               ob2memory(win32security.LookupAccountName("", "Administrator")[0]),
55           )
56   
57       def testMemory(self):
58           pwr_sid = self.pwr_sid
59           admin_sid = self.admin_sid
60           sd1 = win32security.SECURITY_DESCRIPTOR()
61           sd2 = win32security.SECURITY_DESCRIPTOR()
62           sd3 = win32security.SECURITY_DESCRIPTOR()
63           dacl = win32security.ACL()
64           dacl.AddAccessAllowedAce(
65               win32security.ACL_REVISION, win32con.GENERIC_READ, pwr_sid
66           )
67           if admin_sid is not None:
68               dacl.AddAccessAllowedAce(
69                   win32security.ACL_REVISION, win32con.GENERIC_ALL, admin_sid
70               )
71           sd4 = win32security.SECURITY_DESCRIPTOR()
72           sacl = win32security.ACL()
73           if admin_sid is not None:
74               sacl.AddAuditAccessAce(
75                   win32security.ACL_REVISION, win32con.DELETE, admin_sid, 1, 1
76               )
77           sacl.AddAuditAccessAce(
78               win32security.ACL_REVISION, win32con.GENERIC_ALL, pwr_sid, 1, 1
79           )
80           for x in range(0, 200000):
81               if admin_sid is not None:
82                   sd1.SetSecurityDescriptorOwner(admin_sid, 0)
83               sd2.SetSecurityDescriptorGroup(pwr_sid, 0)
84               sd3.SetSecurityDescriptorDacl(1, dacl, 0)
85               sd4.SetSecurityDescriptorSacl(1, sacl, 0)

88   class DomainTests(unittest.TestCase):
89       def setUp(self):
90           self.ds_handle = None
91           try:
92               # saving the handle means the other test itself should bind faster.
93               self.ds_handle = win32security.DsBind()
94           except win32security.error as exc:
95               if exc.winerror != winerror.ERROR_NO_SUCH_DOMAIN:
96                   raise
97               raise TestSkipped(exc)
98   
99       def tearDown(self):
100          if self.ds_handle is not None:
101              self.ds_handle.close()

119      def testDsCrackNames(self):
120          h = win32security.DsBind()
121          fmt_offered = ntsecuritycon.DS_FQDN_1779_NAME
122          name = win32api.GetUserNameEx(fmt_offered)
123          result = win32security.DsCrackNames(h, 0, fmt_offered, fmt_offered, (name,))

127          # Do a syntax check only - that allows us to avoid binding.

130          fmt_offered = ntsecuritycon.DS_FQDN_1779_NAME

134              ntsecuritycon.DS_NAME_FLAG_SYNTACTICAL_ONLY,
135              fmt_offered,
136              ntsecuritycon.DS_CANONICAL_NAME,
