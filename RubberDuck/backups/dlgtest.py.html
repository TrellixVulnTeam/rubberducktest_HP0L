22   class TestDialog(dialog.Dialog):
23       def __init__(self, modal=1):
24           dialog.Dialog.__init__(self, IDD_SET_TABSTOPS)
25           self.counter = 0
26           if modal:
27               self.DoModal()
28           else:
29               self.CreateWindow()
30   
31       def OnInitDialog(self):
32           # Set the caption of the dialog itself.
33           self.SetWindowText("Used to be Tab Stops!")
34           # Get a child control, remember it, and change its text.
35           self.edit = self.GetDlgItem(IDC_EDIT_TABS)  # the text box.
36           self.edit.SetWindowText("Test")
37           # Hook a Windows message for the dialog.
38           self.edit.HookMessage(self.KillFocus, win32con.WM_KILLFOCUS)
39           # Get the prompt control, and change its next.
40           prompt = self.GetDlgItem(IDC_PROMPT_TABS)  # the prompt box.
41           prompt.SetWindowText("Prompt")
42           # And the same for the button..
43           cancel = self.GetDlgItem(IDCANCEL)  # the cancel button
44           cancel.SetWindowText("&Kill me")
45   
46           # And just for demonstration purposes, we hook the notify message for the dialog.
47           # This allows us to be notified when the Edit Control text changes.
48           self.HookCommand(self.OnNotify, IDC_EDIT_TABS)
49   
50       def OnNotify(self, controlid, code):
51           if code == win32con.EN_CHANGE:
52               print("Edit text changed!")
53           return 1  # I handled this, so no need to call defaults!
54   
55       # kill focus for the edit box.
56       # Simply increment the value in the text box.
57       def KillFocus(self, msg):
58           self.counter = self.counter + 1
59           if self.edit != None:
60               self.edit.SetWindowText(str(self.counter))
61   
62       # Called when the dialog box is terminating...
63       def OnDestroy(self, msg):
64           del self.edit
65           del self.counter

79   # A very simply Property Page, which will be "owned" by the above
80   # Property Sheet.
81   # We create a new class, just so we can hook a control notification.
82   class TestPage(dialog.PropertyPage):
83       def OnInitDialog(self):
84           # We use the HookNotify function to allow Python to respond to
85           # Windows WM_NOTIFY messages.
86           # In this case, we are interested in BN_CLICKED messages.
87           self.HookNotify(self.OnNotify, win32con.BN_CLICKED)
88   
89       def OnNotify(self, std, extra):
90           print("OnNotify", std, extra)
