17       def load(self, filename):
18           # Get an IPersist interface
19           # which allows save/restore of object to/from files
20           self._base.QueryInterface(pythoncom.IID_IPersistFile).Load(filename)
21   
22       def save(self, filename):
23           self._base.QueryInterface(pythoncom.IID_IPersistFile).Save(filename, 0)
24   
25       def __getattr__(self, name):
26           if name != "_base":
27               return getattr(self._base, name)
