62       # alarm sound effect
63       def playSound_alarm(self):

74       # behavior for 'break' action
75       def timedBreak(self, instance):
76           self.greeting.text = "See you in 30 minutes!"
77           App.get_running_app().root_window.minimize()
78           Clock.schedule_once(self.breakOver, 5) # time set to 5 seconds for testing and demoing
79   
80       # restoring app on break finish
81       def breakOver(self, instance):
82           self.playSound_quack()
83           self.playSound_alarm()
84           App.get_running_app().root_window.restore()
85           self.greeting.text = "Welcome back!"

87       # 'dismiss' handling
88       def checkBackLater(self, instance):
