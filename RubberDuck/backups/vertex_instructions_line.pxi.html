22   cdef inline int line_intersection(double x1, double y1, double x2, double y2,
23           double x3, double y3, double x4, double y4, double *px, double *py):
24       cdef double u = (x1 * y2 - y1 * x2)
25       cdef double v = (x3 * y4 - y3 * x4)
26       cdef double denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4)
27       if denom == 0:
28           return 0
29       px[0] = (u * (x3 - x4) - (x1 - x2) * v) / denom
30       py[0] = (u * (y3 - y4) - (y1 - y2) * v) / denom
31       return 1

33   cdef class Line(VertexInstruction):
34       '''A 2d line.
35   
36       Drawing a line can be done easily::
37   
38           with self.canvas:
39               Line(points=[100, 100, 200, 100, 100, 200], width=10)
40   
41       The line has 3 internal drawing modes that you should be aware of
42       for optimal results:
43   
44       #. If the :attr:`width` is 1.0, then the standard GL_LINE drawing from
45          OpenGL will be used. :attr:`dash_length`, :attr:`dash_offset`, and :attr:`dashes` will
46          work, while properties for cap and joint have no meaning here.
47       #. If the :attr:`width` is greater than 1.0, then a custom drawing method,
48          based on triangulation, will be used. :attr:`dash_length`,
49          :attr:`dash_offset`, and :attr:`dashes` do not work in this mode.
50          Additionally, if the current color has an alpha less than 1.0, a
51          stencil will be used internally to draw the line.
52   
53       .. image:: images/line-instruction.png
54           :align: center
55   
56       :Parameters:
57           `points`: list
58               List of points in the format (x1, y1, x2, y2...)
59           `dash_length`: int
60               Length of a segment (if dashed), defaults to 1.
61           `dash_offset`: int
62               Offset between the end of a segment and the beginning of the
63               next one, defaults to 0. Changing this makes it dashed.
64           `dashes`: list of ints
65               List of [ON length, offset, ON length, offset, ...]. E.g. ``[2,4,1,6,8,2]``
66               would create a line with the first dash length 2 then an offset of 4 then
67               a dash lenght of 1 then an offset of 6 and so on. Defaults to ``[]``.
68               Changing this makes it dashed and overrides `dash_length` and `dash_offset`.
69           `width`: float
70               Width of the line, defaults to 1.0.
71           `cap`: str, defaults to 'round'
72               See :attr:`cap` for more information.
73           `joint`: str, defaults to 'round'
74               See :attr:`joint` for more information.
75           `cap_precision`: int, defaults to 10
76               See :attr:`cap_precision` for more information
77           `joint_precision`: int, defaults to 10
78               See :attr:`joint_precision` for more information
79               See :attr:`cap_precision` for more information.
80           `joint_precision`: int, defaults to 10
81               See :attr:`joint_precision` for more information.
82           `close`: bool, defaults to False
83               If True, the line will be closed.
84           `circle`: list
85               If set, the :attr:`points` will be set to build a circle. See
86               :attr:`circle` for more information.
87           `ellipse`: list
88               If set, the :attr:`points` will be set to build an ellipse. See
89               :attr:`ellipse` for more information.
90           `rectangle`: list
91               If set, the :attr:`points` will be set to build a rectangle. See
92               :attr:`rectangle` for more information.
93           `bezier`: list
94               If set, the :attr:`points` will be set to build a bezier line. See
95               :attr:`bezier` for more information.
96           `bezier_precision`: int, defaults to 180
97               Precision of the Bezier drawing.
98   
99       .. versionchanged:: 1.0.8
100          `dash_offset` and `dash_length` have been added.
101  
102      .. versionchanged:: 1.4.1

625              step = (a1 - a2) / float(self._cap_precision)
626              siv = iv
627              cx = p[0]
628              cy = p[1]
629              vertices[iv].x = <float>cx
630              vertices[iv].y = <float>cy
631              vertices[iv].s0 = 0
632              vertices[iv].t0 = 0
633              iv += 1
634              for i in xrange(0, self._cap_precision - 1):
635                  vertices[iv].x = <float>(cx + cos(a1 + step * i) * w)
636                  vertices[iv].y = <float>(cy + sin(a1 + step * i) * w)
637                  vertices[iv].s0 = 1
638                  vertices[iv].t0 = 1
639                  if i == 0:
640                      indices[ii] = siv
641                      indices[ii + 1] = 0
642                      indices[ii + 2] = iv
643                  else:
644                      indices[ii] = siv
645                      indices[ii + 1] = iv - 1
646                      indices[ii + 2] = iv
647                  iv += 1
648                  ii += 3
649              indices[ii] = siv
650              indices[ii + 1] = iv - 1
651              indices[ii + 2] = 3
652              ii += 3
653  
654              # cap end
655              a1 = angle - PI2
656              a2 = angle + PI2
657              step = (a2 - a1) / float(self._cap_precision)
658              siv = iv
659              cx = p[-2]
660              cy = p[-1]
661              vertices[iv].x = <float>cx
662              vertices[iv].y = <float>cy
663              vertices[iv].s0 = 0
664              vertices[iv].t0 = 0
665              iv += 1
666              for i in xrange(0, self._cap_precision - 1):
667                  vertices[iv].x = <float>(cx + cos(a1 + step * i) * w)
668                  vertices[iv].y = <float>(cy + sin(a1 + step * i) * w)
669                  vertices[iv].s0 = 0
670                  vertices[iv].t0 = 0
671                  if i == 0:
672                      indices[ii] = siv
673                      indices[ii + 1] = piv + 1
674                      indices[ii + 2] = iv
675                  else:
676                      indices[ii] = siv
677                      indices[ii + 1] = iv - 1
678                      indices[ii + 2] = iv
679                  iv += 1
680                  ii += 3
681              indices[ii] = siv
682              indices[ii + 1] = iv - 1
683              indices[ii + 2] = piv + 2
684              ii += 3
685  
686          while ii < indices_count:
687              # make all the remaining indices point to the last vertice
688              indices[ii] = siv
689              ii += 1
690  
691          # compute bbox
692          cdef unsigned long iul
693          for iul in xrange(vertices_count):
694              if vertices[iul].x < self._bxmin:
695                  self._bxmin = vertices[iul].x
696              if vertices[iul].x > self._bxmax:
697                  self._bxmax = vertices[iul].x
698              if vertices[iul].y < self._bymin:
699                  self._bymin = vertices[iul].y
700              if vertices[iul].y > self._bymax:
701                  self._bymax = vertices[iul].y
702  
703          self.batch.set_data(vertices, <int>vertices_count,
704                             indices, <int>indices_count)
705  
706          free(vertices)
707          free(indices)

887      property ellipse:
888          '''Use this property to build an ellipse, without calculating the
889          :attr:`points`. You can only set this property, not get it.
890  
891          The argument must be a tuple of (x, y, width, height, angle_start,
892          angle_end, segments):
893  
894          * x and y represent the bottom left of the ellipse
895          * width and height represent the size of the ellipse
896          * (optional) angle_start and angle_end are in degree. The default
897            value is 0 and 360.
898          * (optional) segments is the precision of the ellipse. The default
899            value is calculated from the range between angle.
900  
901          Note that it's up to you to :attr:`close` the ellipse or not.
902  
903          For example, for building a simple ellipse, in python::
904  
905              # simple ellipse
906              Line(ellipse=(0, 0, 150, 150))
907  
908              # only from 90 to 180 degrees
909              Line(ellipse=(0, 0, 150, 150, 90, 180))
910  
911              # only from 90 to 180 degrees, with few segments
912              Line(ellipse=(0, 0, 150, 150, 90, 180, 20))
913  
914          .. versionadded:: 1.4.1
915          '''
916  
917          def __set__(self, args):
918              if args == None:
919                  raise GraphicException(
920                          'Invalid ellipse value: {0!r}'.format(args))
921              if len(args) not in (4, 6, 7):
922                  raise GraphicException('Invalid number of arguments: '
923                          '{0} instead of 4, 6 or 7.'.format(len(args)))
924              self._mode_args = tuple(args)
925              self._mode = LINE_MODE_ELLIPSE
926              self.flag_update()

928      cdef void prebuild_ellipse(self):
929          cdef double x, y, w, h, angle_start = 0, angle_end = 360
930          cdef int angle_dir, segments = 0
931          cdef double angle_range
932          cdef tuple args = self._mode_args
933  
934          if len(args) == 4:
935              x, y, w, h = args
936          elif len(args) == 6:
937              x, y, w, h, angle_start, angle_end = args
938          elif len(args) == 7:
939              x, y, w, h, angle_start, angle_end, segments = args
940              segments += 2
941          else:
942              x = y = w = h = 0
943              assert(0)
944  
945          if angle_end > angle_start:
946              angle_dir = 1
947          else:
948              angle_dir = -1
949          if segments == 0:
950              segments = int(abs(angle_end - angle_start) / 2) + 3
951              if segments % 2 == 1:
952                  segments += 1
953          # rad = deg * (pi / 180), where pi/180 = 0.0174...
954          angle_start = angle_start * 0.017453292519943295
955          angle_end = angle_end * 0.017453292519943295
956          angle_range = abs(angle_end - angle_start) / (segments - 2)
957  
958          cdef list points = [0, ] * segments
959          cdef double angle
960          cdef double rx = w * 0.5
961          cdef double ry = h * 0.5
962          for i in xrange(0, segments, 2):
963              angle = angle_start + (angle_dir * (i - 1) * angle_range)
964              points[i] = (x + rx) + (rx * sin(angle))
965              points[i + 1] = (y + ry) + (ry * cos(angle))
966  
967          self._points = points

1192         '''Use this property to build a bezier line, without calculating the
1193         :attr:`points`. You can only set this property, not get it.
1194 
1195         The argument must be a tuple of 2n elements, n being the number of points.
1196 
1197         Usage::
1198 
1199             Line(bezier=(x1, y1, x2, y2, x3, y3)
1200 
1201         .. versionadded:: 1.4.2
1202 
1203         .. note:: Bezier lines calculations are inexpensive for a low number of
1204             points, but complexity is quadratic, so lines with a lot of points
1205             can be very expensive to build, use with care!
1206         '''
1207 
1208         def __set__(self, args):
1209             if args == None or len(args) % 2:
1210                 raise GraphicException(
1211                         'Invalid bezier value: {0!r}'.format(args))

1216     cdef void prebuild_bezier(self):
1217         cdef double x, y, l
1218         cdef int segments = self._bezier_precision
1219         cdef list T = list(self._mode_args)[:]
1220 
1221         self._points = []
1222         for x in xrange(segments):
1223             l = x / (1.0 * segments)
1224             # http://en.wikipedia.org/wiki/De_Casteljau%27s_algorithm
1225             # as the list is in the form of (x1, y1, x2, y2...) iteration is
1226             # done on each item and the current item (xn or yn) in the list is
1227             # replaced with a calculation of "xn + x(n+1) - xn" x(n+1) is
1228             # placed at n+2. Each iteration makes the list one item shorter
1229             for i in range(1, len(T)):
1230                 for j in xrange(len(T) - 2*i):
1231                     T[j] = T[j] + (T[j+2] - T[j]) * l
1232 
1233             # we got the coordinates of the point in T[0] and T[1]
1234             self._points.append(T[0])
1235             self._points.append(T[1])
1236 
1237         # add one last point to join the curve to the end
1238         self._points.append(T[-2])
1239         self._points.append(T[-1])
