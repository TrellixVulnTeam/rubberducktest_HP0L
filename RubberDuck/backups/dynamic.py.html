1    """Support for dynamic COM client support. 
2     
3    Introduction 
4     Dynamic COM client support is the ability to use a COM server without 
5     prior knowledge of the server.  This can be used to talk to almost all 
6     COM servers, including much of MS Office. 
7     
8     In general, you should not use this module directly - see below. 
9     
10   Example 
11    >>> import win32com.client 
12    >>> xl = win32com.client.Dispatch("Excel.Application") 
13    # The line above invokes the functionality of this class. 
14    # xl is now an object we can use to talk to Excel. 
15    >>> xl.Visible = 1 # The Excel window becomes visible.

80   def _GetGoodDispatch(IDispatch, clsctx=pythoncom.CLSCTX_SERVER):
81       # quick return for most common case
82       if isinstance(IDispatch, PyIDispatchType):
83           return IDispatch
84       if isinstance(IDispatch, _GoodDispatchTypes):
85           try:
86               IDispatch = pythoncom.connect(IDispatch)
87           except pythoncom.ole_error:
88               IDispatch = pythoncom.CoCreateInstance(
89                   IDispatch, None, clsctx, pythoncom.IID_IDispatch
90               )
91       else:
92           # may already be a wrapped class.
93           IDispatch = getattr(IDispatch, "_oleobj_", IDispatch)
94       return IDispatch

109  def _GetDescInvokeType(entry, invoke_type):
110      # determine the wFlags argument passed as input to IDispatch::Invoke
111      # Only ever called by __getattr__ and __setattr__ from dynamic objects!
112      # * `entry` is a MapEntry with whatever typeinfo we have about the property we are getting/setting.
113      # * `invoke_type` is either INVOKE_PROPERTYGET | INVOKE_PROPERTYSET and really just
114      #   means "called by __getattr__" or "called by __setattr__"
115      if not entry or not entry.desc:
116          return invoke_type
117  
118      if entry.desc.desckind == pythoncom.DESCKIND_VARDESC:
119          return invoke_type
120  
121      # So it's a FUNCDESC - just use what it specifies.
122      return entry.desc.invkind

125  def Dispatch(
126      IDispatch,
127      userName=None,
128      createClass=None,
129      typeinfo=None,
130      UnicodeToString=None,
131      clsctx=pythoncom.CLSCTX_SERVER,
132  ):
133      assert UnicodeToString is None, "this is deprecated and will go away"
134      IDispatch, userName = _GetGoodDispatchAndUserName(IDispatch, userName, clsctx)
135      if createClass is None:
136          createClass = CDispatch
137      lazydata = None
138      try:
139          if typeinfo is None:
140              typeinfo = IDispatch.GetTypeInfo()
141          if typeinfo is not None:
142              try:
143                  # try for a typecomp
144                  typecomp = typeinfo.GetTypeComp()
145                  lazydata = typeinfo, typecomp
146              except pythoncom.com_error:
147                  pass
148      except pythoncom.com_error:
149          typeinfo = None
150      olerepr = MakeOleRepr(IDispatch, typeinfo, lazydata)
151      return createClass(IDispatch, olerepr, userName, lazydata=lazydata)

154  def MakeOleRepr(IDispatch, typeinfo, typecomp):
155      olerepr = None
156      if typeinfo is not None:
157          try:
158              attr = typeinfo.GetTypeAttr()
159              # If the type info is a special DUAL interface, magically turn it into
160              # a DISPATCH typeinfo.
161              if (
162                  attr[5] == pythoncom.TKIND_INTERFACE
163                  and attr[11] & pythoncom.TYPEFLAG_FDUAL
164              ):
165                  # Get corresponding Disp interface;
166                  # -1 is a special value which does this for us.
167                  href = typeinfo.GetRefTypeOfImplType(-1)
168                  typeinfo = typeinfo.GetRefTypeInfo(href)
169                  attr = typeinfo.GetTypeAttr()
170              if typecomp is None:
171                  olerepr = build.DispatchItem(typeinfo, attr, None, 0)
172              else:
173                  olerepr = build.LazyDispatchItem(attr, None)
174          except pythoncom.ole_error:
175              pass
176      if olerepr is None:
177          olerepr = build.DispatchItem()
178      return olerepr

198          self, IDispatch, olerepr, userName=None, UnicodeToString=None, lazydata=None

201          if userName is None:

212      def __call__(self, *args):
213          "Provide 'default dispatch' COM functionality - allow instance to be called"
214          if self._olerepr_.defaultDispatchName:
215              invkind, dispid = self._find_dispatch_type_(
216                  self._olerepr_.defaultDispatchName
217              )
218          else:
219              invkind, dispid = (
220                  pythoncom.DISPATCH_METHOD | pythoncom.DISPATCH_PROPERTYGET,
221                  pythoncom.DISPID_VALUE,
222              )
223          if invkind is not None:
224              allArgs = (dispid, LCID, invkind, 1) + args
225              return self._get_good_object_(
226                  self._oleobj_.Invoke(*allArgs), self._olerepr_.defaultDispatchName, None
227              )
228          raise TypeError("This dispatch object does not define a default method")

239          # __str__ is used when the user does "print object", so we gracefully
240          # fall back to the __repr__ if the object has no default method.
241          try:

243          except pythoncom.com_error as details:
244              if details.hresult not in ERRORS_BAD_CONTEXT:
245                  raise
246              return self.__repr__()

248      def __dir__(self):
249          lst = list(self.__dict__.keys()) + dir(self.__class__) + self._dir_ole_()
250          try:
251              lst += [p.Name for p in self.Properties_]
252          except AttributeError:
253              pass
254          return list(set(lst))

256      def _dir_ole_(self):
257          items_dict = {}
258          for iTI in range(0, self._oleobj_.GetTypeInfoCount()):
259              typeInfo = self._oleobj_.GetTypeInfo(iTI)
260              self._UpdateWithITypeInfo_(items_dict, typeInfo)
261          return list(items_dict.keys())
262  
263      def _UpdateWithITypeInfo_(self, items_dict, typeInfo):
264          typeInfos = [typeInfo]
265          # suppress IDispatch and IUnknown methods
266          inspectedIIDs = {pythoncom.IID_IDispatch: None}
267  
268          while len(typeInfos) > 0:
269              typeInfo = typeInfos.pop()
270              typeAttr = typeInfo.GetTypeAttr()
271  
272              if typeAttr.iid not in inspectedIIDs:
273                  inspectedIIDs[typeAttr.iid] = None
274                  for iFun in range(0, typeAttr.cFuncs):
275                      funDesc = typeInfo.GetFuncDesc(iFun)
276                      funName = typeInfo.GetNames(funDesc.memid)[0]
277                      if funName not in items_dict:
278                          items_dict[funName] = None
279  
280                  # Inspect the type info of all implemented types
281                  # E.g. IShellDispatch5 implements IShellDispatch4 which implements IShellDispatch3 ...
282                  for iImplType in range(0, typeAttr.cImplTypes):
283                      iRefType = typeInfo.GetRefTypeOfImplType(iImplType)
284                      refTypeInfo = typeInfo.GetRefTypeInfo(iRefType)
285                      typeInfos.append(refTypeInfo)

317      def __getitem__(self, index):  # syver modified
318          # Improved __getitem__ courtesy Syver Enstad
319          # Must check _NewEnum before Item, to ensure b/w compat.
320          if isinstance(index, int):
321              if self.__dict__["_enum_"] is None:
322                  self.__dict__["_enum_"] = self._NewEnum()
323              if self.__dict__["_enum_"] is not None:

326          invkind, dispid = self._find_dispatch_type_("Item")
327          if invkind is not None:
328              return self._get_good_object_(
329                  self._oleobj_.Invoke(dispid, LCID, invkind, 1, index)

352      def _find_dispatch_type_(self, methodName):
353          if methodName in self._olerepr_.mapFuncs:
354              item = self._olerepr_.mapFuncs[methodName]
355              return item.desc[4], item.dispid
356  
357          if methodName in self._olerepr_.propMapGet:
358              item = self._olerepr_.propMapGet[methodName]
359              return item.desc[4], item.dispid

361          try:
362              dispid = self._oleobj_.GetIDsOfNames(0, methodName)
363          except:  ### what error?
364              return None, None
365          return pythoncom.DISPATCH_METHOD | pythoncom.DISPATCH_PROPERTYGET, dispid

380      def _get_good_single_object_(self, ob, userName=None, ReturnCLSID=None):
381          if isinstance(ob, PyIDispatchType):
382              # make a new instance of (probably this) class.
383              return self._wrap_dispatch_(ob, userName, ReturnCLSID)
384          if isinstance(ob, PyIUnknownType):
385              try:
386                  ob = ob.QueryInterface(pythoncom.IID_IDispatch)
387              except pythoncom.com_error:
388                  # It is an IUnknown, but not an IDispatch, so just let it through.
389                  return ob
390              return self._wrap_dispatch_(ob, userName, ReturnCLSID)
391          return ob

393      def _get_good_object_(self, ob, userName=None, ReturnCLSID=None):
394          """Given an object (usually the retval from a method), make it a good object to return. 
395          Basically checks if it is a COM object, and wraps it up. 
396          Also handles the fact that a retval may be a tuple of retvals"""
397          if ob is None:  # Quick exit!
398              return None
399          elif isinstance(ob, tuple):
400              return tuple(
401                  map(
402                      lambda o, s=self, oun=userName, rc=ReturnCLSID: s._get_good_single_object_(
403                          o, oun, rc
404                      ),
405                      ob,
406                  )
407              )
408          else:
409              return self._get_good_single_object_(ob)

411      def _make_method_(self, name):
412          "Make a method object - Assumes in olerepr funcmap"
413          methodName = build.MakePublicAttributeName(name)  # translate keywords etc.
414          methodCodeList = self._olerepr_.MakeFuncMethod(
415              self._olerepr_.mapFuncs[name], methodName, 0
416          )
417          methodCode = "\n".join(methodCodeList)
418          try:
419              #           print "Method code for %s is:\n" % self._username_, methodCode
420              #           self._print_details_()
421              codeObject = compile(methodCode, "<COMObject %s>" % self._username_, "exec")
422              # Exec the code object
423              tempNameSpace = {}
424              # "Dispatch" in the exec'd code is win32com.client.Dispatch, not ours.
425              globNameSpace = globals().copy()
426              globNameSpace["Dispatch"] = win32com.client.Dispatch
427              exec(
428                  codeObject, globNameSpace, tempNameSpace
429              )  # self.__dict__, self.__dict__
430              name = methodName
431              # Save the function in map.
432              fn = self._builtMethods_[name] = tempNameSpace[name]
433              newMeth = MakeMethod(fn, self, self.__class__)
434              return newMeth
435          except:
436              debug_print("Error building OLE definition for code ", methodCode)
437              traceback.print_exc()
438          return None

440      def _Release_(self):
441          """Cleanup object - like a close - to force cleanup when you dont 
442          want to rely on Python's reference counting."""
443          for childCont in self._mapCachedItems_.values():
444              childCont._Release_()
445          self._mapCachedItems_ = {}
446          if self._oleobj_:
447              self._oleobj_.Release()
448              self.__dict__["_oleobj_"] = None
449          if self._olerepr_:
450              self.__dict__["_olerepr_"] = None
451          self._enum_ = None

453      def _proc_(self, name, *args):
454          """Call the named method as a procedure, rather than function. 
455          Mainly used by Word.Basic, which whinges about such things."""
456          try:
457              item = self._olerepr_.mapFuncs[name]
458              dispId = item.dispid
459              return self._get_good_object_(
460                  self._oleobj_.Invoke(*(dispId, LCID, item.desc[4], 0) + (args))
461              )
462          except KeyError:
463              raise AttributeError(name)

494      # Using the typecomp, lazily create a new attribute definition.
495      def _LazyAddAttr_(self, attr):
496          if self._lazydata_ is None:
497              return 0
498          res = 0
499          typeinfo, typecomp = self._lazydata_
500          olerepr = self._olerepr_
501          # We need to explicitly check each invoke type individually - simply
502          # specifying '0' will bind to "any member", which may not be the one
503          # we are actually after (ie, we may be after prop_get, but returned
504          # the info for the prop_put.)
505          for i in ALL_INVOKE_TYPES:
506              try:
507                  x, t = typecomp.Bind(attr, i)
508                  # Support 'Get' and 'Set' properties - see
509                  # bug 1587023
510                  if x == 0 and attr[:3] in ("Set", "Get"):
511                      x, t = typecomp.Bind(attr[3:], i)
512                  if x == pythoncom.DESCKIND_FUNCDESC:  # it's a FUNCDESC
513                      r = olerepr._AddFunc_(typeinfo, t, 0)
514                  elif x == pythoncom.DESCKIND_VARDESC:  # it's a VARDESC
515                      r = olerepr._AddVar_(typeinfo, t, 0)
516                  else:  # not found or TYPEDESC/IMPLICITAPP
517                      r = None
518                  if not r is None:
519                      key, map = r[0], r[1]
520                      item = map[key]
521                      if map == olerepr.propMapPut:
522                          olerepr._propMapPutCheck_(key, item)
523                      elif map == olerepr.propMapGet:
524                          olerepr._propMapGetCheck_(key, item)
525                      res = 1
526              except:
527                  pass
528          return res

530      def _FlagAsMethod(self, *methodNames):
531          """Flag these attribute names as being methods. 
532          Some objects do not correctly differentiate methods and 
533          properties, leading to problems when calling these methods. 
534   
535          Specifically, trying to say: ob.SomeFunc() 
536          may yield an exception "None object is not callable" 
537          In this case, an attempt to fetch the *property* has worked 
538          and returned None, rather than indicating it is really a method. 
539          Calling: ob._FlagAsMethod("SomeFunc") 
540          should then allow this to work. 
541          """
542          for name in methodNames:
543              details = build.MapEntry(self.__AttrToID__(name), (name,))
544              self._olerepr_.mapFuncs[name] = details

546      def __AttrToID__(self, attr):
547          debug_attr_print(

549              % (attr, self._username_)
550          )
551          return self._oleobj_.GetIDsOfNames(0, attr)

553      def __getattr__(self, attr):
554          if attr == "__iter__":
555              # We can't handle this as a normal method, as if the attribute
556              # exists, then it must return an iterable object.
557              try:
558                  invkind = pythoncom.DISPATCH_METHOD | pythoncom.DISPATCH_PROPERTYGET
559                  enum = self._oleobj_.InvokeTypes(
560                      pythoncom.DISPID_NEWENUM, LCID, invkind, (13, 10), ()
561                  )
562              except pythoncom.com_error:
563                  raise AttributeError("This object can not function as an iterator")
564              # We must return a callable object.
565              class Factory:
566                  def __init__(self, ob):
567                      self.ob = ob
568  
569                  def __call__(self):
570                      import win32com.client.util
571  
572                      return win32com.client.util.Iterator(self.ob)
573  
574              return Factory(enum)
575  
576          if attr.startswith("_") and attr.endswith("_"):  # Fast-track.
577              raise AttributeError(attr)
578          # If a known method, create new instance and return.
579          try:
580              return MakeMethod(self._builtMethods_[attr], self, self.__class__)
581          except KeyError:
582              pass
583          # XXX - Note that we current are case sensitive in the method.
584          # debug_attr_print("GetAttr called for %s on DispatchContainer %s" % (attr,self._username_))
585          # First check if it is in the method map.  Note that an actual method
586          # must not yet exist, (otherwise we would not be here).  This
587          # means we create the actual method object - which also means
588          # this code will never be asked for that method name again.
589          if attr in self._olerepr_.mapFuncs:
590              return self._make_method_(attr)
591  
592          # Delegate to property maps/cached items
593          retEntry = None
594          if self._olerepr_ and self._oleobj_:
595              # first check general property map, then specific "put" map.
596              retEntry = self._olerepr_.propMap.get(attr)
597              if retEntry is None:
598                  retEntry = self._olerepr_.propMapGet.get(attr)
599              # Not found so far - See what COM says.
600              if retEntry is None:
601                  try:
602                      if self.__LazyMap__(attr):
603                          if attr in self._olerepr_.mapFuncs:
604                              return self._make_method_(attr)
605                          retEntry = self._olerepr_.propMap.get(attr)
606                          if retEntry is None:
607                              retEntry = self._olerepr_.propMapGet.get(attr)
608                      if retEntry is None:
609                          retEntry = build.MapEntry(self.__AttrToID__(attr), (attr,))
610                  except pythoncom.ole_error:
611                      pass  # No prop by that name - retEntry remains None.
612  
613          if retEntry is not None:  # see if in my cache
614              try:
615                  ret = self._mapCachedItems_[retEntry.dispid]
616                  debug_attr_print("Cached items has attribute!", ret)
617                  return ret
618              except (KeyError, AttributeError):
619                  debug_attr_print("Attribute %s not in cache" % attr)
620  
621          # If we are still here, and have a retEntry, get the OLE item
622          if retEntry is not None:
623              invoke_type = _GetDescInvokeType(retEntry, pythoncom.INVOKE_PROPERTYGET)
624              debug_attr_print(
625                  "Getting property Id 0x%x from OLE object" % retEntry.dispid
626              )
627              try:
628                  ret = self._oleobj_.Invoke(retEntry.dispid, 0, invoke_type, 1)
629              except pythoncom.com_error as details:
630                  if details.hresult in ERRORS_BAD_CONTEXT:
631                      # May be a method.
632                      self._olerepr_.mapFuncs[attr] = retEntry
633                      return self._make_method_(attr)
634                  raise
635              debug_attr_print("OLE returned ", ret)
636              return self._get_good_object_(ret)
637  
638          # no where else to look.
639          raise AttributeError("%s.%s" % (self._username_, attr))
640  
641      def __setattr__(self, attr, value):
642          if (
643              attr in self.__dict__
644          ):  # Fast-track - if already in our dict, just make the assignment.
645              # XXX - should maybe check method map - if someone assigns to a method,
646              # it could mean something special (not sure what, tho!)
647              self.__dict__[attr] = value
648              return
649          # Allow property assignment.
650          debug_attr_print(
651              "SetAttr called for %s.%s=%s on DispatchContainer"
652              % (self._username_, attr, repr(value))
653          )
654  
655          if self._olerepr_:
656              # Check the "general" property map.
657              if attr in self._olerepr_.propMap:
658                  entry = self._olerepr_.propMap[attr]
659                  invoke_type = _GetDescInvokeType(entry, pythoncom.INVOKE_PROPERTYPUT)
660                  self._oleobj_.Invoke(entry.dispid, 0, invoke_type, 0, value)
661                  return
662              # Check the specific "put" map.
663              if attr in self._olerepr_.propMapPut:
664                  entry = self._olerepr_.propMapPut[attr]
665                  invoke_type = _GetDescInvokeType(entry, pythoncom.INVOKE_PROPERTYPUT)
666                  self._oleobj_.Invoke(entry.dispid, 0, invoke_type, 0, value)
667                  return
668  
669          # Try the OLE Object
670          if self._oleobj_:
671              if self.__LazyMap__(attr):
672                  # Check the "general" property map.
673                  if attr in self._olerepr_.propMap:
674                      entry = self._olerepr_.propMap[attr]
675                      invoke_type = _GetDescInvokeType(
676                          entry, pythoncom.INVOKE_PROPERTYPUT
677                      )
678                      self._oleobj_.Invoke(entry.dispid, 0, invoke_type, 0, value)
679                      return
680                  # Check the specific "put" map.
681                  if attr in self._olerepr_.propMapPut:
682                      entry = self._olerepr_.propMapPut[attr]
683                      invoke_type = _GetDescInvokeType(
684                          entry, pythoncom.INVOKE_PROPERTYPUT
685                      )
686                      self._oleobj_.Invoke(entry.dispid, 0, invoke_type, 0, value)
687                      return
688              try:
689                  entry = build.MapEntry(self.__AttrToID__(attr), (attr,))
690              except pythoncom.com_error:
691                  # No attribute of that name
692                  entry = None
693              if entry is not None:
694                  try:
695                      invoke_type = _GetDescInvokeType(
696                          entry, pythoncom.INVOKE_PROPERTYPUT
697                      )
698                      self._oleobj_.Invoke(entry.dispid, 0, invoke_type, 0, value)
699                      self._olerepr_.propMap[attr] = entry
700                      debug_attr_print(
701                          "__setattr__ property %s (id=0x%x) in Dispatch container %s"
702                          % (attr, entry.dispid, self._username_)
703                      )
704                      return
705                  except pythoncom.com_error:
706                      pass
707          raise AttributeError(
708              "Property '%s.%s' can not be set." % (self._username_, attr)
709          )
