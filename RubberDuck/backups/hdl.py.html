133      def analyse_text(text):
134          """Verilog code will use one of reg/wire/assign for sure, and that 
135          is not common elsewhere."""
136          result = 0
137          if 'reg' in text:
138              result += 0.1
139          if 'wire' in text:
140              result += 0.1
141          if 'assign' in text:
142              result += 0.1
143  
144          return result
