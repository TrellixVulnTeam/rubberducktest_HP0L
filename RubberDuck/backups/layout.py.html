26   class LayoutSelectionBehavior(CompoundSelectionBehavior):
27       '''The :class:`LayoutSelectionBehavior` can be combined with 
28       :class:`RecycleLayoutManagerBehavior` to allow its derived classes 
29       selection behaviors similarly to how 
30       :class:`~kivy.uix.behaviors.compoundselection.CompoundSelectionBehavior` 
31       can be used to add selection behaviors to normal layout. 
32    
33       :class:`RecycleLayoutManagerBehavior` manages its children 
34       differently than normal layouts or widgets so this class adapts 
35       :class:`~kivy.uix.behaviors.compoundselection.CompoundSelectionBehavior` 
36       based selection to work with :class:`RecycleLayoutManagerBehavior` as well. 
37    
38       Similarly to 
39       :class:`~kivy.uix.behaviors.compoundselection.CompoundSelectionBehavior`, 
40       one can select using the keyboard or touch, which calls :meth:`select_node` 
41       or :meth:`deselect_node`, or one can call these methods directly. When a 
42       item is selected or deselected :meth:`apply_selection` is called. See 
43       :meth:`apply_selection`. 
44    
45    
46       '''
47   
48       key_selection = StringProperty(None, allownone=True)
49       '''The key used to check whether a view of a data item can be selected 
50       with touch or the keyboard. 
51    
52       :attr:`key_selection` is the key in data, which if present and ``True`` 
53       will enable selection for this item from the keyboard or with a touch. 
54       When None, the default, not item will be selectable. 
55    
56       :attr:`key_selection` is a :class:`StringProperty` and defaults to None. 
57    
58       .. note:: 
59           All data items can be selected directly using :meth:`select_node` or 
60           :meth:`deselect_node`, even if :attr:`key_selection` is False. 
61       '''
62   
63       _selectable_nodes = []
64       _nodes_map = {}
65   
66       def __init__(self, **kwargs):
67           self.nodes_order_reversed = False
68           super(LayoutSelectionBehavior, self).__init__(**kwargs)
69   
70       def compute_sizes_from_data(self, data, flags):
71           # overwrite this method so that when data changes we update
72           # selectable nodes.
73           key = self.key_selection
74           if key is None:
75               nodes = self._selectable_nodes = []
76           else:
77               nodes = self._selectable_nodes = [
78                   i for i, d in enumerate(data) if d.get(key)]
79   
80           self._nodes_map = {v: k for k, v in enumerate(nodes)}
81           return super(LayoutSelectionBehavior, self).compute_sizes_from_data(
82               data, flags)
83   
84       def get_selectable_nodes(self):
85           # the indices of the data is used as the nodes
86           return self._selectable_nodes
87   
88       def get_index_of_node(self, node, selectable_nodes):
89           # the indices of the data is used as the nodes, so node
90           return self._nodes_map[node]
91   
92       def goto_node(self, key, last_node, last_node_idx):
93           node, idx = super(LayoutSelectionBehavior, self).goto_node(
94               key, last_node, last_node_idx)
95           if node is not last_node:
96               self.goto_view(node)
97           return node, idx
98   
99       def select_node(self, node):
100          if super(LayoutSelectionBehavior, self).select_node(node):
101              view = self.recycleview.view_adapter.get_visible_view(node)
102              if view is not None:
103                  self.apply_selection(node, view, True)
104  
105      def deselect_node(self, node):
106          if super(LayoutSelectionBehavior, self).deselect_node(node):
107              view = self.recycleview.view_adapter.get_visible_view(node)
108              if view is not None:
109                  self.apply_selection(node, view, False)
110  
111      def apply_selection(self, index, view, is_selected):
112          '''Applies the selection to the view. This is called internally when 
113          a view is displayed and it needs to be shown as selected or as not 
114          selected. 
115   
116          It is called when :meth:`select_node` or :meth:`deselect_node` is 
117          called or when a view needs to be refreshed. Its function is purely to 
118          update the view to reflect the selection state. So the function may be 
119          called multiple times even if the selection state may not have changed. 
120   
121          If the view is a instance of 
122          :class:`~kivy.uix.recycleview.views.RecycleDataViewBehavior`, its 
123          :meth:`~kivy.uix.recycleview.views.RecycleDataViewBehavior.\ 
124  apply_selection` method will be called every time the view needs to refresh 
125          the selection state. Otherwise, the this method is responsible 
126          for applying the selection. 
127   
128          :Parameters: 
129   
130              `index`: int 
131                  The index of the data item that is associated with the view. 
132              `view`: widget 
133                  The widget that is the view of this data item. 
134              `is_selected`: bool 
135                  Whether the item is selected. 
136          '''
137          viewclass = view.__class__
138          if viewclass not in _view_base_cache:
139              _view_base_cache[viewclass] = isinstance(view,
140                                                       RecycleDataViewBehavior)
141  
142          if _view_base_cache[viewclass]:
143              view.apply_selection(self.recycleview, index, is_selected)
144  
145      def refresh_view_layout(self, index, layout, view, viewport):
146          super(LayoutSelectionBehavior, self).refresh_view_layout(
147              index, layout, view, viewport)
148          self.apply_selection(index, view, index in self.selected_nodes)

151  class RecycleLayoutManagerBehavior(object):
152      """A RecycleLayoutManagerBehavior is responsible for positioning views into 
153      the :attr:`RecycleView.data` within a :class:`RecycleView`. It adds new 
154      views into the data when it becomes visible to the user, and removes them 
155      when they leave the visible area. 
156      """
157  
158      viewclass = ObjectProperty(None)
159      '''See :attr:`RecyclerView.viewclass`. 
160      '''
161      key_viewclass = StringProperty(None)
162      '''See :attr:`RecyclerView.key_viewclass`. 
163      '''
164  
165      recycleview = ObjectProperty(None, allownone=True)
166  
167      asked_sizes = None
168  
169      def attach_recycleview(self, rv):
170          self.recycleview = rv
171          if rv:
172              fbind = self.fbind
173              # can be made more selective update than refresh_from_data which
174              # causes a full update. But this likely affects most of the data.
175              fbind('viewclass', rv.refresh_from_data)
176              fbind('key_viewclass', rv.refresh_from_data)
177              fbind('viewclass', rv._dispatch_prop_on_source, 'viewclass')
178              fbind('key_viewclass', rv._dispatch_prop_on_source,
179                    'key_viewclass')
180  
181      def detach_recycleview(self):
182          self.clear_layout()
183          rv = self.recycleview
184          if rv:
185              funbind = self.funbind
186              funbind('viewclass', rv.refresh_from_data)
187              funbind('key_viewclass', rv.refresh_from_data)
188              funbind('viewclass', rv._dispatch_prop_on_source, 'viewclass')
189              funbind('key_viewclass', rv._dispatch_prop_on_source,
190                    'key_viewclass')
191          self.recycleview = None
192  
193      def compute_sizes_from_data(self, data, flags):
194          pass
195  
196      def compute_layout(self, data, flags):
197          pass
198  
199      def compute_visible_views(self, data, viewport):
200          '''`viewport` is in coordinates of the layout manager. 
201          '''
202          pass
203  
204      def set_visible_views(self, indices, data, viewport):
205          '''`viewport` is in coordinates of the layout manager. 
206          '''
207          pass
208  
209      def refresh_view_layout(self, index, layout, view, viewport):
210          '''`See :meth:`~kivy.uix.recycleview.views.RecycleDataAdapter.\ 
211  refresh_view_layout`. 
212          '''
213          self.recycleview.view_adapter.refresh_view_layout(
214              index, layout, view, viewport)
215  
216      def get_view_index_at(self, pos):
217          """Return the view `index` on which position, `pos`, falls. 
218   
219          `pos` is in coordinates of the layout manager. 
220          """
221          pass
222  
223      def remove_views(self):
224          rv = self.recycleview
225          if rv:
226              adapter = rv.view_adapter
227              if adapter:
228                  adapter.make_views_dirty()
229  
230      def remove_view(self, view, index):
231          rv = self.recycleview
232          if rv:
233              adapter = rv.view_adapter
234              if adapter:
235                  adapter.make_view_dirty(view, index)
236  
237      def clear_layout(self):
238          rv = self.recycleview
239          if rv:
240              adapter = rv.view_adapter
241              if adapter:
242                  adapter.invalidate()
243  
244      def goto_view(self, index):
245          '''Moves the views so that the view corresponding to `index` is 
246          visible. 
247          '''
248          pass
249  
250      def on_viewclass(self, instance, value):
251          # resolve the real class if it was a string.
252          if isinstance(value, string_types):
253              self.viewclass = getattr(Factory, value)
