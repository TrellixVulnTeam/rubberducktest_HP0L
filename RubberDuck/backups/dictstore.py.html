1    ''' 
2    Dictionary store 
3    ================= 
4     
5    Use a Python dictionary as a store. 
6    '''
7    
8    __all__ = ('DictStore', )
9    
10   try:
11       import cPickle as pickle
12   except ImportError:
13       import pickle
14   
15   import errno
16   from os.path import exists, abspath, dirname
17   from kivy.compat import iteritems
18   from kivy.storage import AbstractStore
19   
20   
21   class DictStore(AbstractStore):
22       '''Store implementation using a pickled `dict`. 
23       See the :mod:`kivy.storage` module documentation for more information. 
24       '''
25       def __init__(self, filename, data=None, **kwargs):
26           if isinstance(filename, dict):
27               # backward compatibility, first argument was a dict.
28               self.filename = None
29               self._data = filename
30           else:
31               self.filename = filename
32               self._data = data or {}
33           self._is_changed = True
34           super(DictStore, self).__init__(**kwargs)
35   
36       def store_load(self):
37           if self.filename is None:
38               return
39           if not exists(self.filename):
40               folder = abspath(dirname(self.filename))
41               if not exists(folder):
42                   not_found = IOError(
43                       "The folder '{}' doesn't exist!"
44                       "".format(folder)
45                   )
46                   not_found.errno = errno.ENOENT
47                   raise not_found
48               return
49           with open(self.filename, 'rb') as fd:
50               data = fd.read()
51               if data:
52                   self._data = pickle.loads(data)
53   
54       def store_sync(self):
55           if self.filename is None:
56               return
57           if not self._is_changed:
58               return
59   
60           with open(self.filename, 'wb') as fd:
61               pickle.dump(self._data, fd)
62   
63           self._is_changed = False
64   
65       def store_exists(self, key):
66           return key in self._data
67   
68       def store_get(self, key):
69           return self._data[key]
70   
71       def store_put(self, key, value):
72           self._data[key] = value
73           self._is_changed = True
74           return True
75   
76       def store_delete(self, key):
77           del self._data[key]
78           self._is_changed = True
79           return True
80   
81       def store_find(self, filters):
82           for key, values in iteritems(self._data):
83               found = True
84               for fkey, fvalue in iteritems(filters):
85                   if fkey not in values:
86                       found = False
87                       break
88                   if values[fkey] != fvalue:
89                       found = False
90                       break
91               if found:
92                   yield key, values
93   
94       def store_count(self):
95           return len(self._data)
96   
97       def store_keys(self):
98           return list(self._data.keys())
