20   class GAPLexer(RegexLexer):
21       """ 
22       For `GAP <http://www.gap-system.org>`_ source code. 
23    
24       .. versionadded:: 2.0 
25       """
26       name = 'GAP'
27       aliases = ['gap']
28       filenames = ['*.g', '*.gd', '*.gi', '*.gap']
29   
30       tokens = {
31           'root': [
32               (r'#.*$', Comment.Single),
33               (r'"(?:[^"\\]|\\.)*"', String),
34               (r'\(|\)|\[|\]|\{|\}', Punctuation),
35               (r'''(?x)\b(?: 
36                   if|then|elif|else|fi| 
37                   for|while|do|od| 
38                   repeat|until| 
39                   break|continue| 
40                   function|local|return|end| 
41                   rec| 
42                   quit|QUIT| 
43                   IsBound|Unbind| 
44                   TryNextMethod| 
45                   Info|Assert 
46                 )\b''', Keyword),
47               (r'''(?x)\b(?: 
48                   true|false|fail|infinity 
49                 )\b''',
50                Name.Constant),
51               (r'''(?x)\b(?: 
52                   (Declare|Install)([A-Z][A-Za-z]+)| 
53                      BindGlobal|BIND_GLOBAL 
54                 )\b''',
55                Name.Builtin),
56               (r'\.|,|:=|;|=|\+|-|\*|/|\^|>|<', Operator),
57               (r'''(?x)\b(?: 
58                   and|or|not|mod|in 
59                 )\b''',
60                Operator.Word),
61               (r'''(?x) 
62                 (?:\w+|`[^`]*`) 
63                 (?:::\w+|`[^`]*`)*''', Name.Variable),
64               (r'[0-9]+(?:\.[0-9]*)?(?:e[0-9]+)?', Number),
65               (r'\.[0-9]+(?:e[0-9]+)?', Number),
66               (r'.', Text)
67           ],
68       }
69   
70       def analyse_text(text):
71           score = 0.0
72   
73           # Declaration part
74           if re.search(
75               r"(InstallTrueMethod|Declare(Attribute|Category|Filter|Operation" +
76               r"|GlobalFunction|Synonym|SynonymAttr|Property))", text
77           ):
78               score += 0.7
79   
80           # Implementation part
81           if re.search(
82               r"(DeclareRepresentation|Install(GlobalFunction|Method|" +
83               r"ImmediateMethod|OtherMethod)|New(Family|Type)|Objectify)", text
84           ):
85               score += 0.7
86   
87           return min(score, 1.0)

90   class MathematicaLexer(RegexLexer):
91       """ 
92       Lexer for `Mathematica <http://www.wolfram.com/mathematica/>`_ source code. 
93    
94       .. versionadded:: 2.0 
95       """
96       name = 'Mathematica'
97       aliases = ['mathematica', 'mma', 'nb']
98       filenames = ['*.nb', '*.cdf', '*.nbp', '*.ma']
99       mimetypes = ['application/mathematica',
100                   'application/vnd.wolfram.mathematica',
101                   'application/vnd.wolfram.mathematica.package',
102                   'application/vnd.wolfram.cdf']
103  
104      # http://reference.wolfram.com/mathematica/guide/Syntax.html
105      operators = (
106          ";;", "=", "=.", "!=" "==", ":=", "->", ":>", "/.", "+", "-", "*", "/",
107          "^", "&&", "||", "!", "<>", "|", "/;", "?", "@", "//", "/@", "@@",
108          "@@@", "~~", "===", "&", "<", ">", "<=", ">=",
109      )
110  
111      punctuation = (",", ";", "(", ")", "[", "]", "{", "}")
112  
113      def _multi_escape(entries):
114          return '(%s)' % ('|'.join(re.escape(entry) for entry in entries))
115  
116      tokens = {
117          'root': [
118              (r'(?s)\(\*.*?\*\)', Comment),
119  
120              (r'([a-zA-Z]+[A-Za-z0-9]*`)', Name.Namespace),
121              (r'([A-Za-z0-9]*_+[A-Za-z0-9]*)', Name.Variable),
122              (r'#\d*', Name.Variable),
123              (r'([a-zA-Z]+[a-zA-Z0-9]*)', Name),
124  
125              (r'-?\d+\.\d*', Number.Float),
126              (r'-?\d*\.\d+', Number.Float),
127              (r'-?\d+', Number.Integer),
128  
129              (words(operators), Operator),
130              (words(punctuation), Punctuation),
131              (r'".*?"', String),
132              (r'\s+', Text.Whitespace),
133          ],
134      }

137  class MuPADLexer(RegexLexer):
138      """ 
139      A `MuPAD <http://www.mupad.com>`_ lexer. 
140      Contributed by Christopher Creutzig <christopher@creutzig.de>. 
141   
142      .. versionadded:: 0.8 
143      """
144      name = 'MuPAD'
145      aliases = ['mupad']
146      filenames = ['*.mu']
147  
148      tokens = {
149          'root': [
150              (r'//.*?$', Comment.Single),
151              (r'/\*', Comment.Multiline, 'comment'),
152              (r'"(?:[^"\\]|\\.)*"', String),
153              (r'\(|\)|\[|\]|\{|\}', Punctuation),
154              (r'''(?x)\b(?: 
155                  next|break|end| 
156                  axiom|end_axiom|category|end_category|domain|end_domain|inherits| 
157                  if|%if|then|elif|else|end_if| 
158                  case|of|do|otherwise|end_case| 
159                  while|end_while| 
160                  repeat|until|end_repeat| 
161                  for|from|to|downto|step|end_for| 
162                  proc|local|option|save|begin|end_proc| 
163                  delete|frame 
164                )\b''', Keyword),
165              (r'''(?x)\b(?: 
166                  DOM_ARRAY|DOM_BOOL|DOM_COMPLEX|DOM_DOMAIN|DOM_EXEC|DOM_EXPR| 
167                  DOM_FAIL|DOM_FLOAT|DOM_FRAME|DOM_FUNC_ENV|DOM_HFARRAY|DOM_IDENT| 
168                  DOM_INT|DOM_INTERVAL|DOM_LIST|DOM_NIL|DOM_NULL|DOM_POLY|DOM_PROC| 
169                  DOM_PROC_ENV|DOM_RAT|DOM_SET|DOM_STRING|DOM_TABLE|DOM_VAR 
170                )\b''', Name.Class),
171              (r'''(?x)\b(?: 
172                  PI|EULER|E|CATALAN| 
173                  NIL|FAIL|undefined|infinity| 
174                  TRUE|FALSE|UNKNOWN 
175                )\b''',
176               Name.Constant),
177              (r'\b(?:dom|procname)\b', Name.Builtin.Pseudo),
178              (r'\.|,|:|;|=|\+|-|\*|/|\^|@|>|<|\$|\||!|\'|%|~=', Operator),
179              (r'''(?x)\b(?: 
180                  and|or|not|xor| 
181                  assuming| 
182                  div|mod| 
183                  union|minus|intersect|in|subset 
184                )\b''',
185               Operator.Word),
186              (r'\b(?:I|RDN_INF|RD_NINF|RD_NAN)\b', Number),
187              # (r'\b(?:adt|linalg|newDomain|hold)\b', Name.Builtin),
188              (r'''(?x) 
189                ((?:[a-zA-Z_#][\w#]*|`[^`]*`) 
190                (?:::[a-zA-Z_#][\w#]*|`[^`]*`)*)(\s*)([(])''',
191               bygroups(Name.Function, Text, Punctuation)),
192              (r'''(?x) 
193                (?:[a-zA-Z_#][\w#]*|`[^`]*`) 
194                (?:::[a-zA-Z_#][\w#]*|`[^`]*`)*''', Name.Variable),
195              (r'[0-9]+(?:\.[0-9]*)?(?:e[0-9]+)?', Number),
196              (r'\.[0-9]+(?:e[0-9]+)?', Number),
197              (r'.', Text)
198          ],
199          'comment': [
200              (r'[^*/]', Comment.Multiline),
201              (r'/\*', Comment.Multiline, '#push'),
202              (r'\*/', Comment.Multiline, '#pop'),
203              (r'[*/]', Comment.Multiline)
204          ],
205      }

208  class BCLexer(RegexLexer):
209      """ 
210      A `BC <https://www.gnu.org/software/bc/>`_ lexer. 
211   
212      .. versionadded:: 2.1 
213      """
214      name = 'BC'
215      aliases = ['bc']
216      filenames = ['*.bc']
217  
218      tokens = {
219          'root': [
220              (r'/\*', Comment.Multiline, 'comment'),
221              (r'"(?:[^"\\]|\\.)*"', String),
222              (r'[{}();,]', Punctuation),
223              (words(('if', 'else', 'while', 'for', 'break', 'continue',
224                      'halt', 'return', 'define', 'auto', 'print', 'read',
225                      'length', 'scale', 'sqrt', 'limits', 'quit',
226                      'warranty'), suffix=r'\b'), Keyword),
227              (r'\+\+|--|\|\||&&|'
228               r'([-<>+*%\^/!=])=?', Operator),
229              # bc doesn't support exponential
230              (r'[0-9]+(\.[0-9]*)?', Number),
231              (r'\.[0-9]+', Number),
232              (r'.', Text)
233          ],
234          'comment': [
235              (r'[^*/]+', Comment.Multiline),
236              (r'\*/', Comment.Multiline, '#pop'),
237              (r'[*/]', Comment.Multiline)
238          ],
239      }
