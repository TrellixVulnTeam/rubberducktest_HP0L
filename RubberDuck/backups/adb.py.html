44   def _dumpf(frame):
45       if frame is None:
46           return "<None>"
47       else:
48           addn = "(with trace!)"
49           if frame.f_trace is None:
50               addn = " **No Trace Set **"
51           return "Frame at %d, file %s, line: %d%s" % (
52               id(frame),
53               frame.f_code.co_filename,
54               frame.f_lineno,
55               addn,
56           )

102      def __xxxxx__set_break(self, filename, lineno, cond=None):
103          # As per standard one, except no linecache checking!
104          if filename not in self.breaks:
105              self.breaks[filename] = []
106          list = self.breaks[filename]
107          if lineno in list:

109          list.append(lineno)
110          if cond is not None:

124      def break_here(self, frame):
125          traceenter("break_here", self.breakFlags, _dumpf(frame))
126          self.breakReason = None
127          if self.breakFlags == axdebug.APPBREAKFLAG_DEBUGGER_HALT:
128              self.breakReason = axdebug.BREAKREASON_DEBUGGER_HALT
129          elif self.breakFlags == axdebug.APPBREAKFLAG_DEBUGGER_BLOCK:
130              self.breakReason = axdebug.BREAKREASON_DEBUGGER_BLOCK
131          elif self.breakFlags == axdebug.APPBREAKFLAG_STEP:
132              self.breakReason = axdebug.BREAKREASON_STEP
133          else:
134              print("Calling base 'break_here' with", self.breaks)
135              if bdb.Bdb.break_here(self, frame):
136                  self.breakReason = axdebug.BREAKREASON_BREAKPOINT
137          return self.breakReason is not None

159      def dispatch_line(self, frame):
160          traceenter("dispatch_line", _dumpf(frame), _dumpf(self.botframe))
161          #               trace("logbotframe is", _dumpf(self.logicalbotframe), "botframe is", self.botframe)
162          if frame is self.logicalbotframe:
163              trace("dispatch_line", _dumpf(frame), "for bottom frame returing tracer")
164              # The next code executed in the frame above may be a builtin (eg, apply())
165              # in which sys.trace needs to be set.
166              sys.settrace(self.trace_dispatch)
167              # And return the tracer incase we are about to execute Python code,
168              # in which case sys tracer is ignored!
169              return self.trace_dispatch
170  
171          if self.codeContainerProvider.FromFileName(frame.f_code.co_filename) is None:
172              trace(
173                  "dispatch_line has no document for", _dumpf(frame), "- skipping trace!"
174              )
175              return None
176          self.currentframe = (
177              frame  # So the stack sniffer knows our most recent, debuggable code.
178          )
179          return bdb.Bdb.dispatch_line(self, frame)

181      def dispatch_call(self, frame, arg):
182          traceenter("dispatch_call", _dumpf(frame))
183          frame.f_locals["__axstack_address__"] = axdebug.GetStackAddress()
184          if frame is self.botframe:
185              trace("dispatch_call is self.botframe - returning tracer")
186              return self.trace_dispatch
187          # Not our bottom frame.  If we have a document for it,
188          # then trace it, otherwise run at full speed.
189          if self.codeContainerProvider.FromFileName(frame.f_code.co_filename) is None:
190              trace(
191                  "dispatch_call has no document for", _dumpf(frame), "- skipping trace!"
192              )
193              ##                      sys.settrace(None)
194              return None
195          return self.trace_dispatch

231      def _HandleBreakPoint(self, frame, tb, reason):
232          traceenter(
233              "Calling HandleBreakPoint with reason", reason, "at frame", _dumpf(frame)
234          )
235          traceenter(" Current frame is", _dumpf(self.currentframe))
236          try:
237              resumeAction = self.debugApplication.HandleBreakPoint(reason)
238              tracev("HandleBreakPoint returned with ", resumeAction)
239          except pythoncom.com_error as details:
240              # Eeek - the debugger is dead, or something serious is happening.
241              # Assume we should continue
242              resumeAction = axdebug.BREAKRESUMEACTION_CONTINUE
243              trace("HandleBreakPoint FAILED with", details)
244  
245          self.stack = []
246          self.curindex = 0
247          if resumeAction == axdebug.BREAKRESUMEACTION_ABORT:
248              self.set_quit()
249          elif resumeAction == axdebug.BREAKRESUMEACTION_CONTINUE:
250              tracev("resume action is continue")
251              self.set_continue()
252          elif resumeAction == axdebug.BREAKRESUMEACTION_STEP_INTO:
253              tracev("resume action is step")
254              self.set_step()
255          elif resumeAction == axdebug.BREAKRESUMEACTION_STEP_OVER:
256              tracev("resume action is next")
257              self.set_next(frame)
258          elif resumeAction == axdebug.BREAKRESUMEACTION_STEP_OUT:
259              tracev("resume action is stop out")
260              self.set_return(frame)
261          else:
262              raise ValueError("unknown resume action flags")
263          self.breakReason = None

269      def CloseApp(self):
270          traceenter("ClosingApp")
271          self.reset()
272          self.logicalbotframe = None
273          if self.stackSnifferCookie is not None:
274              try:
275                  self.debugApplication.RemoveStackFrameSniffer(self.stackSnifferCookie)
276  
277              except pythoncom.com_error:
278                  trace(
279                      "*** Could not RemoveStackFrameSniffer %d"
280                      % (self.stackSnifferCookie)
281                  )
282          if self.stackSniffer:
283              _wrap_remove(self.stackSniffer)
284          self.stackSnifferCookie = self.stackSniffer = None
285  
286          if self.appEventConnection is not None:
287              self.appEventConnection.Disconnect()
288              self.appEventConnection = None
289          self.debugApplication = None
290          self.appDebugger = None
291          if self.codeContainerProvider is not None:
292              self.codeContainerProvider.Close()
293              self.codeContainerProvider = None

312      def ResetAXDebugging(self):
313          traceenter("ResetAXDebugging", self, "with refcount", len(self.recursiveData))
314          if win32api.GetCurrentThreadId() != self.debuggingThread:
315              trace("ResetAXDebugging called on other thread")
316              return
317  
318          if len(self.recursiveData) == 0:
319              #                       print "ResetAXDebugging called for final time."
320              self.logicalbotframe = None
321              self.debuggingThread = None
322              self.currentframe = None
323              self.debuggingThreadStateHandle = None
324              return
325  
326          (
327              self.logbotframe,
328              self.stopframe,
329              self.currentframe,
330              self.debuggingThreadStateHandle,
331          ) = self.recursiveData[0]
332          self.recursiveData = self.recursiveData[1:]

334      def SetupAXDebugging(self, baseFrame=None, userFrame=None):
335          """Get ready for potential debugging.  Must be called on the thread 
336          that is being debugged. 
337          """
338          # userFrame is for non AXScript debugging.  This is the first frame of the
339          # users code.
340          if userFrame is None:
341              userFrame = baseFrame
342          else:
343              # We have missed the "dispatch_call" function, so set this up now!
344              userFrame.f_locals["__axstack_address__"] = axdebug.GetStackAddress()
345  
346          traceenter("SetupAXDebugging", self)
347          self._threadprotectlock.acquire()
348          try:
349              thisThread = win32api.GetCurrentThreadId()
350              if self.debuggingThread is None:
351                  self.debuggingThread = thisThread
352              else:
353                  if self.debuggingThread != thisThread:
354                      trace("SetupAXDebugging called on other thread - ignored!")
355                      return
356                  # push our context.
357                  self.recursiveData.insert(
358                      0,
359                      (
360                          self.logicalbotframe,
361                          self.stopframe,
362                          self.currentframe,
363                          self.debuggingThreadStateHandle,
364                      ),
365                  )
366          finally:
367              self._threadprotectlock.release()
368  
369          trace("SetupAXDebugging has base frame as", _dumpf(baseFrame))
370          self.botframe = baseFrame
371          self.stopframe = userFrame
372          self.logicalbotframe = baseFrame
373          self.currentframe = None
374          self.debuggingThreadStateHandle = axdebug.GetThreadStateHandle()
375  
376          self._BreakFlagsChanged()

379      def OnConnectDebugger(self, appDebugger):
380          traceenter("OnConnectDebugger", appDebugger)
381          self.appDebugger = appDebugger
382          # Reflect output to appDebugger
383          writefunc = lambda s: appDebugger.onDebugOutput(s)
384          sys.stdout = OutputReflector(sys.stdout, writefunc)
385          sys.stderr = OutputReflector(sys.stderr, writefunc)
386  
387      def OnDisconnectDebugger(self):
388          traceenter("OnDisconnectDebugger")
389          # Stop reflecting output
390          if isinstance(sys.stdout, OutputReflector):
391              sys.stdout = sys.stdout.file
392          if isinstance(sys.stderr, OutputReflector):
393              sys.stderr = sys.stderr.file
394          self.appDebugger = None
395          self.set_quit()

423      def _BreakFlagsChanged(self):
424          traceenter(
425              "_BreakFlagsChanged to %s with our thread = %s, and debugging thread = %s"
426              % (self.breakFlags, self.debuggingThread, win32api.GetCurrentThreadId())
427          )
428          trace("_BreakFlagsChanged has breaks", self.breaks)
429          # If a request comes on our debugging thread, then do it now!
430          #               if self.debuggingThread!=win32api.GetCurrentThreadId():
431          #                       return
432  
433          if len(self.breaks) or self.breakFlags:
434  
435              if self.logicalbotframe:
436                  trace("BreakFlagsChange with bot frame", _dumpf(self.logicalbotframe))
437                  # We have frames not to be debugged (eg, Scripting engine frames
438                  # (sys.settrace will be set when out logicalbotframe is hit -
439                  #  this may not be the right thing to do, as it may not cause the
440                  #  immediate break we desire.)
441                  self.logicalbotframe.f_trace = self.trace_dispatch
442              else:
443                  trace("BreakFlagsChanged, but no bottom frame")
444                  if self.stopframe is not None:
445                      self.stopframe.f_trace = self.trace_dispatch
446              # If we have the thread-state for the thread being debugged, then
447              # we dynamically set its trace function - it is possible that the thread
448              # being debugged is in a blocked call (eg, a message box) and we
449              # want to hit the debugger the instant we return
450          if (
451              self.debuggingThreadStateHandle is not None
452              and self.breakFlags
453              and self.debuggingThread != win32api.GetCurrentThreadId()
454          ):
455              axdebug.SetThreadStateTrace(
456                  self.debuggingThreadStateHandle, self.trace_dispatch
457              )

459      def _OnSetBreakPoint(self, key, codeContext, bps, lineNo):
460          traceenter("_OnSetBreakPoint", self, key, codeContext, bps, lineNo)
461          if bps == axdebug.BREAKPOINT_ENABLED:
462              problem = self.set_break(key, lineNo)
463              if problem:
464                  print("*** set_break failed -", problem)
465              trace("_OnSetBreakPoint just set BP and has breaks", self.breaks)
466          else:
467              self.clear_break(key, lineNo)
468          self._BreakFlagsChanged()
469          trace("_OnSetBreakPoint leaving with breaks", self.breaks)
