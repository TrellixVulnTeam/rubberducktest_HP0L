7    def ParseArgs(str):
8        import string
9    
10       ret = []
11       pos = 0
12       length = len(str)
13       while pos < length:
14           try:
15               while str[pos] in string.whitespace:
16                   pos = pos + 1
17           except IndexError:
18               break
19           if pos >= length:
20               break
21           if str[pos] == '"':
22               pos = pos + 1
23               try:
24                   endPos = str.index('"', pos) - 1
25                   nextPos = endPos + 2
26               except ValueError:
27                   endPos = length
28                   nextPos = endPos + 1
29           else:
30               endPos = pos
31               while endPos < length and not str[endPos] in string.whitespace:
32                   endPos = endPos + 1
33               nextPos = endPos + 1
34           ret.append(str[pos : endPos + 1].strip())
35           pos = nextPos
36       return ret

39   def FixArgFileName(fileName):
40       """Convert a filename on the commandline to something useful. 
41       Given an automatic filename on the commandline, turn it a python module name, 
42       with the path added to sys.path."""
43       import os
44   
45       path, fname = os.path.split(fileName)
46       if len(path) == 0:
47           path = os.curdir
48       path = os.path.abspath(path)
49       # must check that the command line arg's path is in sys.path
50       for syspath in sys.path:
51           if os.path.abspath(syspath) == path:
52               break
53       else:
54           sys.path.append(path)
55       return os.path.splitext(fname)[0]
