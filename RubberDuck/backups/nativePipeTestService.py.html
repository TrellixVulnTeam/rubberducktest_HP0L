1    # This is an example of a service hosted by python.exe rather than
2    # pythonservice.exe.
3    
4    # Note that it is very rare that using python.exe is a better option
5    # than the default pythonservice.exe - the latter has better error handling
6    # so that if Python itself can't be initialized or there are very early
7    # import errors, you will get error details written to the event log.  When
8    # using python.exe instead, you are forced to wait for the interpreter startup
9    # and imports to succeed before you are able to effectively setup your own
10   # error handling.
11   
12   # So in short, please make sure you *really* want to do this, otherwise just
13   # stick with the default.
14   
15   import sys
16   import os
17   import win32serviceutil
18   import servicemanager
19   
20   from pipeTestService import TestPipeService
21   
22   
23   class NativeTestPipeService(TestPipeService):
24       _svc_name_ = "PyNativePipeTestService"
25       _svc_display_name_ = "Python Native Pipe Test Service"
26       _svc_description_ = "Tests Python.exe hosted services"
27       # tell win32serviceutil we have a custom executable and custom args
28       # so registration does the right thing.
29       _exe_name_ = sys.executable
30       _exe_args_ = '"' + os.path.abspath(sys.argv[0]) + '"'
31   
32   
33   def main():
34       if len(sys.argv) == 1:
35           # service must be starting...
36           # for the sake of debugging etc, we use win32traceutil to see
37           # any unhandled exceptions and print statements.
38           import win32traceutil
39   
40           print("service is starting...")
41           print("(execute this script with '--help' if that isn't what you want)")
42   
43           servicemanager.Initialize()
44           servicemanager.PrepareToHostSingle(NativeTestPipeService)
45           # Now ask the service manager to fire things up for us...
46           servicemanager.StartServiceCtrlDispatcher()
47           print("service done!")
48       else:
49           win32serviceutil.HandleCommandLine(NativeTestPipeService)
50   
51   
52   if __name__ == "__main__":
53       try:
54           main()
55       except (SystemExit, KeyboardInterrupt):
56           raise
57       except:
58           print("Something went bad!")
59           import traceback
60   
61           traceback.print_exc()
