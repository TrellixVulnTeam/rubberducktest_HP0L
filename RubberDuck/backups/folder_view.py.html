121  # We construct a PIDL from a pickle of a dict - turn it back into a
122  # dict (we should *never* be called with a PIDL that the last elt is not
123  # ours, so it is safe to assume we created it (assume->"ass" = "u" + "me" :)
124  def pidl_to_item(pidl):
125      # Note that only the *last* elt in the PIDL is certainly ours,
126      # but it contains everything we need encoded as a dict.
127      return pickle.loads(pidl[-1])

132  def make_item_enum(level, flags):
133      pidls = []
134      nums = """zero one two three four five size seven eight nine ten""".split()
135      for i, name in enumerate(nums):
136          size = random.randint(0, 255)
137          sides = 1
138          while sides in [1, 2]:
139              sides = random.randint(0, 5)
140          is_folder = (i % 2) != 0
141          # check the flags say to include it.
142          # (This seems strange; if you ask the same folder for, but appear
143          skip = False
144          if not (flags & shellcon.SHCONTF_STORAGE):
145              if is_folder:
146                  skip = not (flags & shellcon.SHCONTF_FOLDERS)
147              else:
148                  skip = not (flags & shellcon.SHCONTF_NONFOLDERS)
149          if not skip:
150              data = dict(
151                  name=name, size=size, sides=sides, level=level, is_folder=is_folder
152              )
153              pidls.append([pickle.dumps(data)])
154      return NewEnum(pidls, shell.IID_IEnumIDList)

671          if flags & shellcon.SHGDN_FORPARSING:

679                  parent = shell.SHGetNameFromIDList(self.pidl, sigdn)
680                  return parent + "\\" + item["name"]
681          else:

790  def DllRegisterServer():
791      import winreg
792  
793      if sys.getwindowsversion()[0] < 6:
794          print("This sample only works on Vista")
795          sys.exit(1)
796  
797      key = winreg.CreateKey(
798          winreg.HKEY_LOCAL_MACHINE,
799          "SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\"
800          "Explorer\\Desktop\\Namespace\\" + ShellFolder._reg_clsid_,
801      )
802      winreg.SetValueEx(key, None, 0, winreg.REG_SZ, ShellFolder._reg_desc_)
803      # And special shell keys under our CLSID
804      key = winreg.CreateKey(
805          winreg.HKEY_CLASSES_ROOT, "CLSID\\" + ShellFolder._reg_clsid_ + "\\ShellFolder"
806      )
807      # 'Attributes' is an int stored as a binary! use struct
808      attr = (
809          shellcon.SFGAO_FOLDER | shellcon.SFGAO_HASSUBFOLDER | shellcon.SFGAO_BROWSABLE
810      )
811      import struct
812  
813      s = struct.pack("i", attr)
814      winreg.SetValueEx(key, "Attributes", 0, winreg.REG_BINARY, s)
815      # register the context menu handler under the FolderViewSampleType type.
816      keypath = "%s\\shellex\\ContextMenuHandlers\\%s" % (
817          ContextMenu._context_menu_type_,
818          ContextMenu._reg_desc_,
819      )
820      key = winreg.CreateKey(winreg.HKEY_CLASSES_ROOT, keypath)
821      winreg.SetValueEx(key, None, 0, winreg.REG_SZ, ContextMenu._reg_clsid_)
822      propsys.PSRegisterPropertySchema(get_schema_fname())
823      print(ShellFolder._reg_desc_, "registration complete.")

831          + ShellFolder._reg_clsid_,
