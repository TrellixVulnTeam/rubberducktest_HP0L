37       def OnOpenDocument(self, filename):
38           # init data members
39           # print "Opening", filename
40           self.SetPathName(filename)  # Must set this early!
41           try:
42               # load the text as binary we can get smart
43               # about detecting any existing EOL conventions.
44               f = open(filename, "rb")
45               try:
46                   self._LoadTextFromFile(f)
47               finally:
48                   f.close()
49           except IOError:
50               rc = win32ui.MessageBox(
51                   "Could not load the file from %s\n\nDo you want to create a new file?"
52                   % filename,
53                   "Pythonwin",
54                   win32con.MB_YESNO | win32con.MB_ICONWARNING,
55               )
56               if rc == win32con.IDNO:
57                   return 0
58               assert rc == win32con.IDYES, rc
59               try:
60                   f = open(filename, "wb+")
61                   try:
62                       self._LoadTextFromFile(f)
63                   finally:
64                       f.close()
65               except IOError as e:
66                   rc = win32ui.MessageBox("Cannot create the file %s" % filename)
67           return 1

81       # Helper to transfer text from the MFC document to the control.
82       def _LoadTextFromFile(self, f):
83           # detect EOL mode - we don't support \r only - so find the
84           # first '\n' and guess based on the char before.
85           l = f.readline()
86           l2 = f.readline()
87           # If line ends with \r\n or has no line ending, use CRLF.
88           if l.endswith(crlf_bytes) or not l.endswith(lf_bytes):
89               eol_mode = scintillacon.SC_EOL_CRLF
90           else:
91               eol_mode = scintillacon.SC_EOL_LF
92   
93           # Detect the encoding - first look for a BOM, and if not found,
94           # look for a pep263 encoding declaration.
95           for bom, encoding in (
96               (codecs.BOM_UTF8, "utf8"),
97               (codecs.BOM_UTF16_LE, "utf_16_le"),
98               (codecs.BOM_UTF16_BE, "utf_16_be"),
99           ):
100              if l.startswith(bom):
101                  self.bom = bom
102                  self.source_encoding = encoding
103                  l = l[len(bom) :]  # remove it.
104                  break
105          else:
106              # no bom detected - look for pep263 encoding decl.
107              for look in (l, l2):
108                  # Note we are looking at raw bytes here: so
109                  # both the re itself uses bytes and the result
110                  # is bytes - but we need the result as a string.
111                  match = re_encoding_bytes.search(look)
112                  if match is not None:
113                      self.source_encoding = match.group(1).decode("ascii")
114                      break
115  
116          # reading by lines would be too slow?  Maybe we can use the
117          # incremental encoders? For now just stick with loading the
118          # entire file in memory.
119          text = l + l2 + f.read()
120  
121          # Translate from source encoding to UTF-8 bytes for Scintilla
122          source_encoding = self.source_encoding
123          # If we don't know an encoding, try utf-8 - if that fails we will
124          # fallback to latin-1 to treat it as bytes...
125          if source_encoding is None:
126              source_encoding = "utf-8"
127          # we could optimize this by avoiding utf8 to-ing and from-ing,
128          # but then we would lose the ability to handle invalid utf8
129          # (and even then, the use of encoding aliases makes this tricky)
130          # To create an invalid utf8 file:
131          # >>> open(filename, "wb").write(codecs.BOM_UTF8+"bad \xa9har\r\n")
132          try:
133              dec = text.decode(source_encoding)
134          except UnicodeError:
135              print(
136                  "WARNING: Failed to decode bytes from '%s' encoding - treating as latin1"
137                  % source_encoding
138              )
139              dec = text.decode("latin1")
140          except LookupError:
141              print(
142                  "WARNING: Invalid encoding '%s' specified - treating as latin1"
143                  % source_encoding
144              )
145              dec = text.decode("latin1")
146          # and put it back as utf8 - this shouldn't fail.
147          text = dec.encode(default_scintilla_encoding)
148  
149          view = self.GetFirstView()
150          if view.IsWindow():
151              # Turn off undo collection while loading
152              view.SendScintilla(scintillacon.SCI_SETUNDOCOLLECTION, 0, 0)
153              # Make sure the control isnt read-only
154              view.SetReadOnly(0)
155              view.SendScintilla(scintillacon.SCI_CLEARALL)
156              view.SendMessage(scintillacon.SCI_ADDTEXT, text)
157              view.SendScintilla(scintillacon.SCI_SETUNDOCOLLECTION, 1, 0)
158              view.SendScintilla(win32con.EM_EMPTYUNDOBUFFER, 0, 0)
159              # set EOL mode
160              view.SendScintilla(scintillacon.SCI_SETEOLMODE, eol_mode)

162      def _SaveTextToFile(self, view, filename, encoding=None):
163          s = view.GetTextRange()  # already decoded from scintilla's encoding
164          source_encoding = encoding
165          if source_encoding is None:
166              if self.bom:
167                  source_encoding = self.source_encoding
168              else:
169                  # no BOM - look for an encoding.
170                  bits = re.split("[\r\n]+", s, 3)
171                  for look in bits[:-1]:
172                      match = re_encoding_text.search(look)
173                      if match is not None:
174                          source_encoding = match.group(1)
175                          self.source_encoding = source_encoding
176                          break
177  
178              if source_encoding is None:
179                  source_encoding = "utf-8"
180  
181          ## encode data before opening file so script is not lost if encoding fails
182          file_contents = s.encode(source_encoding)
183          # Open in binary mode as scintilla itself ensures the
184          # line endings are already appropriate
185          f = open(filename, "wb")
186          try:
187              if self.bom:
188                  f.write(self.bom)
189              f.write(file_contents)
190          finally:
191              f.close()
192          self.SetModifiedFlag(0)

209          parent.HookNotify(
210              DocumentNotifyDelegate(self, "OnSavePointReached"),
211              scintillacon.SCN_SAVEPOINTREACHED,
212          )
213          parent.HookNotify(
214              DocumentNotifyDelegate(self, "OnSavePointLeft"),
215              scintillacon.SCN_SAVEPOINTLEFT,
216          )
217          parent.HookNotify(
218              DocumentNotifyDelegate(self, "OnModifyAttemptRO"),
219              scintillacon.SCN_MODIFYATTEMPTRO,

222          view.SCIAutoCStops(string.whitespace + "()[]:;+-/*=\\?'!#@$%^&,<>\"'|")

240      def MarkerCheck(self, lineNo, marker):
241          v = self.GetEditorView()
242          lineNo = lineNo - 1  # Make 0 based
243          markerState = v.SCIMarkerGet(lineNo)
244          return markerState & (1 << marker) != 0

246      def MarkerToggle(self, lineNo, marker):
247          v = self.GetEditorView()
248          if self.MarkerCheck(lineNo, marker):
249              v.SCIMarkerDelete(lineNo - 1, marker)
250          else:
251              v.SCIMarkerAdd(lineNo - 1, marker)

289  # Delegate to the correct view, based on the control that sent it.
290  class ViewNotifyDelegate:
291      def __init__(self, doc, name):
292          self.doc = doc
293          self.name = name
294  
295      def __call__(self, std, extra):
296          (hwndFrom, idFrom, code) = std
297          for v in self.doc.GetAllViews():
298              if v.GetSafeHwnd() == hwndFrom:
299                  return getattr(v, self.name)(*(std, extra))

303  class DocumentNotifyDelegate:
304      def __init__(self, doc, name):
305          self.doc = doc

308      def __call__(self, std, extra):
309          (hwndFrom, idFrom, code) = std
310          if hwndFrom == self.doc.GetEditorView().GetSafeHwnd():
311              self.delegate(*(std, extra))
