10   def test_remove_widget(loop):
11       from kivy.uix.carousel import Carousel
12       from kivy.uix.widget import Widget
13   
14       c = Carousel(loop=loop)
15       assert c.index is None
16       assert c.current_slide is None
17       assert len(c.children) == 0
18       assert len(c.slides) == 0
19   
20       N_SLIDES = 4
21       for i in range(N_SLIDES):
22           c.add_widget(Widget())
23       assert c.index == 0
24       assert c.current_slide == c.slides[0]
25       assert len(c.children) == 3 if loop else 2
26       assert len(c.slides) == N_SLIDES
27   
28       # test issue #6370
29       c.index = len(c.slides) - 1
30       c.remove_widget(c.slides[0])
31   
32       # remove a slide(smaller index than the current_slide's).
33       c.index = 1
34       old_current_slide = c.current_slide
35       c.remove_widget(c.slides[c.index - 1])
36       assert c.index == 0
37       assert c.current_slide is old_current_slide
38       assert len(c.children) == 2
39       assert len(c.slides) == 2
40   
41       # remove a slide(bigger index than the current_slide's).
42       old_current_slide = c.current_slide
43       c.remove_widget(c.slides[c.index + 1])
44       assert c.index == 0
45       assert c.current_slide is old_current_slide
46       assert len(c.children) == 1
47       assert len(c.slides) == 1
48   
49       # remove the current_slide(the last one left).
50       c.remove_widget(c.current_slide)
51       assert c.index is None
52       assert c.current_slide is None
53       assert len(c.children) == 0
54       assert len(c.slides) == 0

75   def test_previous_and_next(
76       n_slides, index, loop, index_of_previous_slide, index_of_next_slide
77   ):
78       from kivy.uix.carousel import Carousel
79       from kivy.uix.widget import Widget
80       c = Carousel(loop=loop)
81       for i in range(n_slides):
82           c.add_widget(Widget())
83       c.index = index
84       p_slide = c.previous_slide
85       assert (p_slide and c.slides.index(p_slide)) == index_of_previous_slide
86       n_slide = c.next_slide
87       assert (n_slide and c.slides.index(n_slide)) == index_of_next_slide
