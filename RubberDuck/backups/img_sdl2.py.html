6    __all__ = ('ImageLoaderSDL2', )

8    from kivy.logger import Logger
9    from kivy.core.image import ImageLoaderBase, ImageData, ImageLoader
10   try:
11       from kivy.core.image import _img_sdl2
12   except ImportError:
13       from kivy.core import handle_win_lib_import_error
14       handle_win_lib_import_error(
15           'image', 'sdl2', 'kivy.core.image._img_sdl2')
16       raise
17   
18   
19   class ImageLoaderSDL2(ImageLoaderBase):
20       '''Image loader based on SDL2_image'''
21   
22       def _ensure_ext(self):
23           _img_sdl2.init()
24   
25       @staticmethod
26       def extensions():
27           '''Return accepted extensions for this loader'''
28           return ('bmp', 'jpg', 'jpeg', 'jpe', 'lbm', 'pcx', 'png', 'pnm',
29                   'tga', 'tiff', 'webp', 'xcf', 'xpm', 'xv')
30   
31       @staticmethod
32       def can_save(fmt, is_bytesio):
33           return fmt in ('jpg', 'png')
34   
35       @staticmethod
36       def can_load_memory():
37           return True
38   
39       def load(self, filename):
40           if self._inline:
41               data = filename.read()
42               info = _img_sdl2.load_from_memory(data)
43           else:
44               info = _img_sdl2.load_from_filename(filename)
45           if not info:
46               Logger.warning('Image: Unable to load image <%s>' % filename)
47               raise Exception('SDL2: Unable to load image')
48   
49           w, h, fmt, pixels, rowlength = info
50   
51           # update internals
52           if not self._inline:
53               self.filename = filename
54           return [ImageData(
55               w, h, fmt, pixels, source=filename,
56               rowlength=rowlength)]
57   
58       @staticmethod
59       def save(filename, width, height, pixelfmt, pixels, flipped, imagefmt):
60           _img_sdl2.save(filename, width, height, pixelfmt, pixels, flipped,
61                          imagefmt)
62           return True
63   
64   
65   # register
66   ImageLoader.register(ImageLoaderSDL2)
