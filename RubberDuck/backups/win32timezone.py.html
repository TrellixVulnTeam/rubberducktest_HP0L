18       To use this time zone module with the datetime module, simply pass 
19   the TimeZoneInfo object to the datetime constructor.  For example, 
20    
21   >>> import win32timezone, datetime 
22   >>> assert 'Mountain Standard Time' in win32timezone.TimeZoneInfo.get_sorted_time_zone_names() 
23   >>> MST = win32timezone.TimeZoneInfo('Mountain Standard Time') 
24   >>> now = datetime.datetime.now(MST)

27   aware methods may be called on it. 
28    
29   >>> now.utcoffset() in (datetime.timedelta(-1, 61200), datetime.timedelta(-1, 64800)) 
30   True 
31    
32   (note that the result of utcoffset call will be different based on when now was 
33   generated, unless standard time is always used) 
34    
35   >>> now = datetime.datetime.now(TimeZoneInfo('Mountain Standard Time', True)) 
36   >>> now.utcoffset() 
37   datetime.timedelta(days=-1, seconds=61200) 
38    
39   >>> aug2 = datetime.datetime(2003, 8, 2, tzinfo = MST) 
40   >>> tuple(aug2.utctimetuple()) 
41   (2003, 8, 2, 6, 0, 0, 5, 214, 0) 
42   >>> nov2 = datetime.datetime(2003, 11, 25, tzinfo = MST) 
43   >>> tuple(nov2.utctimetuple()) 
44   (2003, 11, 25, 7, 0, 0, 1, 329, 0)

47    
48   >>> aug2.isoformat() 
49   '2003-08-02T00:00:00-06:00'

51   >>> aug2est.isoformat() 
52   '2003-08-02T02:00:00-04:00'

249  # A couple of objects for working with objects as if they were native C-type
250  # structures.
251  class _SimpleStruct(object):
252      _fields_ = None  # must be overridden by subclasses
253  
254      def __init__(self, *args, **kw):
255          for i, (name, typ) in enumerate(self._fields_):
256              def_arg = None
257              if i < len(args):
258                  def_arg = args[i]
259              if name in kw:
260                  def_arg = kw[name]
261              if def_arg is not None:
262                  if not isinstance(def_arg, tuple):
263                      def_arg = (def_arg,)
264              else:
265                  def_arg = ()
266              if len(def_arg) == 1 and isinstance(def_arg[0], typ):
267                  # already an object of this type.
268                  # XXX - should copy.copy???
269                  def_val = def_arg[0]
270              else:
271                  def_val = typ(*def_arg)
272              setattr(self, name, def_val)
273  
274      def field_names(self):
275          return [f[0] for f in self._fields_]
276  
277      def __eq__(self, other):
278          if not hasattr(other, "_fields_"):
279              return False
280          if self._fields_ != other._fields_:
281              return False
282          for name, _ in self._fields_:
283              if getattr(self, name) != getattr(other, name):
284                  return False
285          return True
286  
287      def __ne__(self, other):
288          return not self.__eq__(other)

323  class TimeZoneDefinition(DYNAMIC_TIME_ZONE_INFORMATION):
324      """ 
325      A time zone definition class based on the win32 
326      DYNAMIC_TIME_ZONE_INFORMATION structure. 
327   
328      Describes a bias against UTC (bias), and two dates at which a separate 
329      additional bias applies (standard_bias and daylight_bias). 
330      """
331  
332      def __init__(self, *args, **kwargs):
333          """ 
334          Try to construct a TimeZoneDefinition from 
335          a) [DYNAMIC_]TIME_ZONE_INFORMATION args 
336          b) another TimeZoneDefinition 
337          c) a byte structure (using _from_bytes) 
338          """
339          try:
340              super(TimeZoneDefinition, self).__init__(*args, **kwargs)
341              return
342          except (TypeError, ValueError):
343              pass
344  
345          try:
346              self.__init_from_other(*args, **kwargs)
347              return
348          except TypeError:
349              pass
350  
351          try:
352              self.__init_from_bytes(*args, **kwargs)
353              return
354          except TypeError:
355              pass
356  
357          raise TypeError("Invalid arguments for %s" % self.__class__)
358  
359      def __init_from_bytes(
360          self,
361          bytes,
362          standard_name="",
363          daylight_name="",
364          key_name="",
365          daylight_disabled=False,
366      ):
367          format = "3l8h8h"
368          components = struct.unpack(format, bytes)
369          bias, standard_bias, daylight_bias = components[:3]
370          standard_start = SYSTEMTIME(*components[3:11])
371          daylight_start = SYSTEMTIME(*components[11:19])
372          super(TimeZoneDefinition, self).__init__(
373              bias,
374              standard_name,
375              standard_start,
376              standard_bias,
377              daylight_name,
378              daylight_start,
379              daylight_bias,
380              key_name,
381              daylight_disabled,
382          )
383  
384      def __init_from_other(self, other):
385          if not isinstance(other, TIME_ZONE_INFORMATION):
386              raise TypeError("Not a TIME_ZONE_INFORMATION")
387          for name in other.field_names():
388              # explicitly get the value from the underlying structure
389              value = super(TimeZoneDefinition, other).__getattribute__(other, name)
390              setattr(self, name, value)
391          # consider instead of the loop above just copying the memory directly
392          # size = max(ctypes.sizeof(DYNAMIC_TIME_ZONE_INFO), ctypes.sizeof(other))
393          # ctypes.memmove(ctypes.addressof(self), other, size)
394  
395      def __getattribute__(self, attr):
396          value = super(TimeZoneDefinition, self).__getattribute__(attr)
397          if "bias" in attr:
398              value = datetime.timedelta(minutes=value)
399          return value

421      @staticmethod
422      def _locate_day(year, cutoff):
423          """ 
424          Takes a SYSTEMTIME object, such as retrieved from a TIME_ZONE_INFORMATION 
425          structure or call to GetTimeZoneInformation and interprets it based on the given 
426          year to identify the actual day. 
427   
428          This method is necessary because the SYSTEMTIME structure refers to a day by its 
429          day of the week and week of the month (e.g. 4th saturday in March). 
430   
431          >>> SATURDAY = 6 
432          >>> MARCH = 3 
433          >>> st = SYSTEMTIME(2000, MARCH, SATURDAY, 4, 0, 0, 0, 0) 
434   
435          # according to my calendar, the 4th Saturday in March in 2009 was the 28th 
436          >>> expected_date = datetime.datetime(2009, 3, 28) 
437          >>> TimeZoneDefinition._locate_day(2009, st) == expected_date 
438          True 
439          """
440          # MS stores Sunday as 0, Python datetime stores Monday as zero
441          target_weekday = (cutoff.day_of_week + 6) % 7
442          # For SYSTEMTIMEs relating to time zone inforamtion, cutoff.day
443          #  is the week of the month
444          week_of_month = cutoff.day
445          # so the following is the first day of that week
446          day = (week_of_month - 1) * 7 + 1
447          result = datetime.datetime(
448              year,
449              cutoff.month,
450              day,
451              cutoff.hour,
452              cutoff.minute,
453              cutoff.second,
454              cutoff.millisecond,
455          )
456          # now the result is the correct week, but not necessarily the correct day of the week
457          days_to_go = (target_weekday - result.weekday()) % 7
458          result += datetime.timedelta(days_to_go)
459          # if we selected a day in the month following the target month,
460          #  move back a week or two.
461          # This is necessary because Microsoft defines the fifth week in a month
462          #  to be the last week in a month and adding the time delta might have
463          #  pushed the result into the next month.
464          while result.month == cutoff.month + 1:
465              result -= datetime.timedelta(weeks=1)
466          return result

469  class TimeZoneInfo(datetime.tzinfo):
470      """ 
471      Main class for handling Windows time zones. 
472      Usage: 
473          TimeZoneInfo(<Time Zone Standard Name>, [<Fix Standard Time>]) 
474   
475      If <Fix Standard Time> evaluates to True, daylight savings time is 
476      calculated in the same way as standard time. 
477   
478      >>> tzi = TimeZoneInfo('Pacific Standard Time') 
479      >>> march31 = datetime.datetime(2000,3,31) 
480   
481      We know that time zone definitions haven't changed from 2007 
482      to 2012, so regardless of whether dynamic info is available, 
483      there should be consistent results for these years. 
484      >>> subsequent_years = [march31.replace(year=year) 
485      ...     for year in range(2007, 2013)] 
486      >>> offsets = set(tzi.utcoffset(year) for year in subsequent_years) 
487      >>> len(offsets) 
488      1 
489      """
490  
491      # this key works for WinNT+, but not for the Win95 line.
492      tzRegKey = r"SOFTWARE\Microsoft\Windows NT\CurrentVersion\Time Zones"
493  
494      def __init__(self, param=None, fix_standard_time=False):
495          if isinstance(param, TimeZoneDefinition):
496              self._LoadFromTZI(param)
497          if isinstance(param, str):
498              self.timeZoneName = param
499              self._LoadInfoFromKey()
500          self.fixedStandardTime = fix_standard_time
501  
502      def _FindTimeZoneKey(self):
503          """Find the registry key for the time zone name (self.timeZoneName)."""
504          # for multi-language compatability, match the time zone name in the
505          # "Std" key of the time zone key.
506          zoneNames = dict(self._get_indexed_time_zone_keys("Std"))
507          # Also match the time zone key name itself, to be compatible with
508          # English-based hard-coded time zones.
509          timeZoneName = zoneNames.get(self.timeZoneName, self.timeZoneName)
510          key = _RegKeyDict.open(winreg.HKEY_LOCAL_MACHINE, self.tzRegKey)
511          try:
512              result = key.subkey(timeZoneName)
513          except Exception:
514              raise ValueError("Timezone Name %s not found." % timeZoneName)
515          return result
516  
517      def _LoadInfoFromKey(self):
518          """Loads the information from an opened time zone registry key 
519          into relevant fields of this TZI object"""
520          key = self._FindTimeZoneKey()
521          self.displayName = key["Display"]
522          self.standardName = key["Std"]
523          self.daylightName = key["Dlt"]
524          self.staticInfo = TimeZoneDefinition(key["TZI"])
525          self._LoadDynamicInfoFromKey(key)
526  
527      def _LoadFromTZI(self, tzi):
528          self.timeZoneName = tzi.standard_name
529          self.displayName = "Unknown"
530          self.standardName = tzi.standard_name
531          self.daylightName = tzi.daylight_name
532          self.staticInfo = tzi
533  
534      def _LoadDynamicInfoFromKey(self, key):
535          """ 
536          >>> tzi = TimeZoneInfo('Central Standard Time') 
537   
538          Here's how the RangeMap is supposed to work: 
539          >>> m = RangeMap(zip([2006,2007], 'BC'), 
540          ...     sort_params = dict(reverse=True), 
541          ...     key_match_comparator=operator.ge) 
542          >>> m.get(2000, 'A') 
543          'A' 
544          >>> m[2006] 
545          'B' 
546          >>> m[2007] 
547          'C' 
548          >>> m[2008] 
549          'C' 
550   
551          >>> m[RangeMap.last_item] 
552          'B' 
553   
554          >>> m.get(2008, m[RangeMap.last_item]) 
555          'C' 
556   
557   
558          Now test the dynamic info (but fallback to our simple RangeMap 
559          on systems that don't have dynamicInfo). 
560   
561          >>> dinfo = getattr(tzi, 'dynamicInfo', m) 
562          >>> 2007 in dinfo 
563          True 
564          >>> 2008 in dinfo 
565          False 
566          >>> dinfo[2007] == dinfo[2008] == dinfo[2012] 
567          True 
568          """
569          try:
570              info = key.subkey("Dynamic DST")
571          except WindowsError:
572              return
573          del info["FirstEntry"]
574          del info["LastEntry"]
575          years = map(int, list(info.keys()))
576          values = map(TimeZoneDefinition, list(info.values()))
577          # create a range mapping that searches by descending year and matches
578          # if the target year is greater or equal.
579          self.dynamicInfo = RangeMap(
580              zip(years, values),
581              sort_params=dict(reverse=True),
582              key_match_comparator=operator.ge,
583          )
584  
585      def __repr__(self):
586          result = "%s(%s" % (self.__class__.__name__, repr(self.timeZoneName))
587          if self.fixedStandardTime:
588              result += ", True"
589          result += ")"
590          return result

595      def tzname(self, dt):
596          winInfo = self.getWinInfo(dt)
597          if self.dst(dt) == winInfo.daylight_bias:
598              result = self.daylightName
599          elif self.dst(dt) == winInfo.standard_bias:
600              result = self.standardName
601          return result

603      def getWinInfo(self, targetYear):
604          """ 
605          Return the most relevant "info" for this time zone 
606          in the target year. 
607          """
608          if not hasattr(self, "dynamicInfo") or not self.dynamicInfo:
609              return self.staticInfo
610          # Find the greatest year entry in self.dynamicInfo which is for
611          #  a year greater than or equal to our targetYear. If not found,
612          #  default to the earliest year.
613          return self.dynamicInfo.get(targetYear, self.dynamicInfo[RangeMap.last_item])
614  
615      def _getStandardBias(self, dt):
616          winInfo = self.getWinInfo(dt.year)
617          return winInfo.bias + winInfo.standard_bias
618  
619      def _getDaylightBias(self, dt):
620          winInfo = self.getWinInfo(dt.year)
621          return winInfo.bias + winInfo.daylight_bias
622  
623      def utcoffset(self, dt):
624          "Calculates the utcoffset according to the datetime.tzinfo spec"
625          if dt is None:
626              return
627          winInfo = self.getWinInfo(dt.year)
628          return -winInfo.bias + self.dst(dt)
629  
630      def dst(self, dt):
631          """ 
632          Calculate the daylight savings offset according to the 
633          datetime.tzinfo spec. 
634          """
635          if dt is None:
636              return
637          winInfo = self.getWinInfo(dt.year)
638          if not self.fixedStandardTime and self._inDaylightSavings(dt):
639              result = winInfo.daylight_bias
640          else:
641              result = winInfo.standard_bias
642          return -result
643  
644      def _inDaylightSavings(self, dt):
645          dt = dt.replace(tzinfo=None)
646          winInfo = self.getWinInfo(dt.year)
647          try:
648              dstStart = self.GetDSTStartTime(dt.year)
649              dstEnd = self.GetDSTEndTime(dt.year)
650  
651              # at the end of DST, when clocks are moved back, there's a period
652              #  of daylight_bias where it's ambiguous whether we're in DST or
653              #  not.
654              dstEndAdj = dstEnd + winInfo.daylight_bias
655  
656              # the same thing could theoretically happen at the start of DST
657              #  if there's a standard_bias (which I suspect is always 0).
658              dstStartAdj = dstStart + winInfo.standard_bias
659  
660              if dstStart < dstEnd:
661                  in_dst = dstStartAdj <= dt < dstEndAdj
662              else:
663                  # in the southern hemisphere, daylight savings time
664                  #  typically ends before it begins in a given year.
665                  in_dst = not (dstEndAdj < dt <= dstStartAdj)
666          except ValueError:
667              # there was an error parsing the time zone, which is normal when a
668              #  start and end time are not specified.
669              in_dst = False
670  
671          return in_dst
672  
673      def GetDSTStartTime(self, year):
674          "Given a year, determines the time when daylight savings time starts"
675          return self.getWinInfo(year).locate_daylight_start(year)
676  
677      def GetDSTEndTime(self, year):
678          "Given a year, determines the time when daylight savings ends."
679          return self.getWinInfo(year).locate_standard_start(year)

690      @classmethod
691      def local(class_):
692          """Returns the local time zone as defined by the operating system in the 
693          registry. 
694          >>> localTZ = TimeZoneInfo.local() 
695          >>> now_local = datetime.datetime.now(localTZ) 
696          >>> now_UTC = datetime.datetime.utcnow() 
697          >>> (now_UTC - now_local) < datetime.timedelta(seconds = 5) 
698          Traceback (most recent call last): 
699          ... 
700          TypeError: can't subtract offset-naive and offset-aware datetimes 
701   
702          >>> now_UTC = now_UTC.replace(tzinfo = TimeZoneInfo('GMT Standard Time', True)) 
703   
704          Now one can compare the results of the two offset aware values 
705          >>> (now_UTC - now_local) < datetime.timedelta(seconds = 5) 
706          True 
707          """
708          code, info = TimeZoneDefinition.current()
709          # code is 0 if daylight savings is disabled or not defined
710          #  code is 1 or 2 if daylight savings is enabled, 2 if currently active
711          fix_standard_time = not code
712          # note that although the given information is sufficient
713          # to construct a WinTZI object, it's
714          # not sufficient to represent the time zone in which
715          # the current user is operating due
716          # to dynamic time zones.
717          return class_(info, fix_standard_time)

719      @classmethod
720      def utc(class_):
721          """Returns a time-zone representing UTC. 
722   
723          Same as TimeZoneInfo('GMT Standard Time', True) but caches the result 
724          for performance. 
725   
726          >>> isinstance(TimeZoneInfo.utc(), TimeZoneInfo) 
727          True 
728          """
729          if "_tzutc" not in class_.__dict__:
730              setattr(class_, "_tzutc", class_("GMT Standard Time", True))
731          return class_._tzutc

749          """

751          ignoring any keys for which that value is empty or missing.

889  def resolveMUITimeZone(spec):
890      """Resolve a multilingual user interface resource for the time zone name 
891      >>> #some pre-amble for the doc-tests to be py2k and py3k aware) 
892      >>> try: unicode and None 
893      ... except NameError: unicode=str 
894      ... 
895      >>> import sys 
896      >>> result = resolveMUITimeZone('@tzres.dll,-110') 
897      >>> expectedResultType = [type(None),unicode][sys.getwindowsversion() >= (6,)] 
898      >>> type(result) is expectedResultType 
899      True 
900   
901      spec should be of the format @path,-stringID[;comment] 
902      see http://msdn2.microsoft.com/en-us/library/ms725481.aspx for details 
903      """
904      pattern = re.compile(r"@(?P<dllname>.*),-(?P<index>\d+)(?:;(?P<comment>.*))?")
905      matcher = pattern.match(spec)
906      assert matcher, "Could not parse MUI spec"
907  
908      try:
909          handle = DLLCache[matcher.groupdict()["dllname"]]
910          result = win32api.LoadString(handle, int(matcher.groupdict()["index"]))
911      except win32api.error:
912          result = None
913      return result

917  class RangeMap(dict):
918      """ 
919      A dictionary-like object that uses the keys as bounds for a range. 
920      Inclusion of the value for that range is determined by the 
921      key_match_comparator, which defaults to less-than-or-equal. 
922      A value is returned for a key if it is the first key that matches in 
923      the sorted list of keys. 
924   
925      One may supply keyword parameters to be passed to the sort function used 
926      to sort keys (i.e. cmp [python 2 only], keys, reverse) as sort_params. 
927   
928      Let's create a map that maps 1-3 -> 'a', 4-6 -> 'b' 
929      >>> r = RangeMap({3: 'a', 6: 'b'})  # boy, that was easy 
930      >>> r[1], r[2], r[3], r[4], r[5], r[6] 
931      ('a', 'a', 'a', 'b', 'b', 'b') 
932   
933      Even float values should work so long as the comparison operator 
934      supports it. 
935      >>> r[4.5] 
936      'b' 
937   
938      But you'll notice that the way rangemap is defined, it must be open-ended on one side. 
939      >>> r[0] 
940      'a' 
941      >>> r[-1] 
942      'a' 
943   
944      One can close the open-end of the RangeMap by using undefined_value 
945      >>> r = RangeMap({0: RangeMap.undefined_value, 3: 'a', 6: 'b'}) 
946      >>> r[0] 
947      Traceback (most recent call last): 
948      ... 
949      KeyError: 0 
950   
951      One can get the first or last elements in the range by using RangeMap.Item 
952      >>> last_item = RangeMap.Item(-1) 
953      >>> r[last_item] 
954      'b' 
955   
956      .last_item is a shortcut for Item(-1) 
957      >>> r[RangeMap.last_item] 
958      'b' 
959   
960      Sometimes it's useful to find the bounds for a RangeMap 
961      >>> r.bounds() 
962      (0, 6) 
963   
964      RangeMap supports .get(key, default) 
965      >>> r.get(0, 'not found') 
966      'not found' 
967   
968      >>> r.get(7, 'not found') 
969      'not found' 
970   
971      """
972  
973      def __init__(self, source, sort_params={}, key_match_comparator=operator.le):
974          dict.__init__(self, source)
975          self.sort_params = sort_params
976          self.match = key_match_comparator
977  
978      def __getitem__(self, item):
979          sorted_keys = sorted(list(self.keys()), **self.sort_params)
980          if isinstance(item, RangeMap.Item):
981              result = self.__getitem__(sorted_keys[item])
982          else:
983              key = self._find_first_match_(sorted_keys, item)
984              result = dict.__getitem__(self, key)
985              if result is RangeMap.undefined_value:
986                  raise KeyError(key)
987          return result
988  
989      def get(self, key, default=None):
990          """ 
991          Return the value for key if key is in the dictionary, else default. 
992          If default is not given, it defaults to None, so that this method 
993          never raises a KeyError. 
994          """
995          try:
996              return self[key]
997          except KeyError:
998              return default
999  
1000     def _find_first_match_(self, keys, item):
1001         def is_match(k):
1002             return self.match(item, k)
1003 
1004         matches = list(filter(is_match, keys))
1005         if matches:
1006             return matches[0]
1007         raise KeyError(item)
1008 
1009     def bounds(self):
1010         sorted_keys = sorted(list(self.keys()), **self.sort_params)
1011         return (
1012             sorted_keys[RangeMap.first_item],
1013             sorted_keys[RangeMap.last_item],
1014         )
1015 
1016     # some special values for the RangeMap
1017     undefined_value = type(str("RangeValueUndefined"), (object,), {})()
1018 
1019     class Item(int):
1020         pass
1021 
1022     first_item = Item(0)
1023     last_item = Item(-1)
