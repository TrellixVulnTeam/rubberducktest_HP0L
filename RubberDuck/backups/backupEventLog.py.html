7    def BackupClearLog(logType):
8        datePrefix = time.strftime("%Y%m%d", time.localtime(time.time()))
9        fileExists = 1
10       retry = 0
11       while fileExists:
12           if retry == 0:
13               index = ""
14           else:
15               index = "-%d" % retry
16           try:
17               fname = os.path.join(
18                   win32api.GetTempPath(),
19                   "%s%s-%s" % (datePrefix, index, logType) + ".evt",
20               )
21               os.stat(fname)
22           except os.error:
23               fileExists = 0
24           retry = retry + 1
25       # OK - have unique file name.
26       try:
27           hlog = win32evtlog.OpenEventLog(None, logType)
28       except win32evtlogutil.error as details:
29           print("Could not open the event log", details)
30           return
31       try:
32           if win32evtlog.GetNumberOfEventLogRecords(hlog) == 0:
33               print("No records in event log %s - not backed up" % logType)
34               return
35           win32evtlog.ClearEventLog(hlog, fname)
36           print("Backed up %s log to %s" % (logType, fname))
37       finally:
38           win32evtlog.CloseEventLog(hlog)
