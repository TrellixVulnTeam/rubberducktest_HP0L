135      def ColorizeInteractiveCode(self, cdoc, styleStart, stylePyStart):
136          lengthDoc = len(cdoc)
137          if lengthDoc == 0:
138              return
139          state = styleStart
140          # As per comments in Colorize(), we work with the raw utf8
141          # bytes. To avoid too muych py3k pain, we treat each utf8 byte
142          # as a latin-1 unicode character - we only use it to compare
143          # against ascii chars anyway...
144          chNext = cdoc[0:1].decode("latin-1")
145          startSeg = 0
146          i = 0
147          lastState = state  # debug only
148          while i < lengthDoc:
149              ch = chNext
150              chNext = cdoc[i + 1 : i + 2].decode("latin-1")
151  
152              #           trace("ch=%r, i=%d, next=%r, state=%s" % (ch, i, chNext, state))
153              if state == STYLE_INTERACTIVE_EOL:
154                  if ch not in "\r\n":
155                      self.ColorSeg(startSeg, i - 1, state)
156                      startSeg = i
157                      if ch in [sys.ps1[0], sys.ps2[0]]:
158                          state = STYLE_INTERACTIVE_PROMPT
159                      elif cdoc[i : i + len(tracebackHeader)] == tracebackHeader:
160                          state = STYLE_INTERACTIVE_ERROR
161                      else:
162                          state = STYLE_INTERACTIVE_OUTPUT
163              elif state == STYLE_INTERACTIVE_PROMPT:
164                  if ch not in sys.ps1 + sys.ps2 + " ":
165                      self.ColorSeg(startSeg, i - 1, state)
166                      startSeg = i
167                      if ch in "\r\n":
168                          state = STYLE_INTERACTIVE_EOL
169                      else:
170                          state = stylePyStart  # Start coloring Python code.
171              elif state in [STYLE_INTERACTIVE_OUTPUT]:
172                  if ch in "\r\n":
173                      self.ColorSeg(startSeg, i - 1, state)
174                      startSeg = i
175                      state = STYLE_INTERACTIVE_EOL
176              elif state == STYLE_INTERACTIVE_ERROR:
177                  if ch in "\r\n" and chNext and chNext not in string.whitespace:
178                      # Everything including me
179                      self.ColorSeg(startSeg, i, state)
180                      startSeg = i + 1
181                      state = STYLE_INTERACTIVE_ERROR_FINALLINE
182                  elif i == 0 and ch not in string.whitespace:
183                      # If we are coloring from the start of a line,
184                      # we need this better check for the last line
185                      # Color up to not including me
186                      self.ColorSeg(startSeg, i - 1, state)
187                      startSeg = i
188                      state = STYLE_INTERACTIVE_ERROR_FINALLINE
189              elif state == STYLE_INTERACTIVE_ERROR_FINALLINE:
190                  if ch in "\r\n":
191                      self.ColorSeg(startSeg, i - 1, state)
192                      startSeg = i
193                      state = STYLE_INTERACTIVE_EOL
194              elif state == STYLE_INTERACTIVE_BANNER:
195                  if ch in "\r\n" and (chNext == "" or chNext in ">["):
196                      # Everything including me
197                      self.ColorSeg(startSeg, i - 1, state)
198                      startSeg = i
199                      state = STYLE_INTERACTIVE_EOL
200              else:
201                  # It is a PythonColorizer state - seek past the end of the line
202                  # and ask the Python colorizer to color that.
203                  end = startSeg
204                  while end < lengthDoc and cdoc[end] not in "\r\n".encode("ascii"):
205                      end = end + 1
206                  self.ColorizePythonCode(cdoc[:end], startSeg, state)
207                  stylePyStart = self.GetStringStyle(end - 1)
208                  if stylePyStart is None:
209                      stylePyStart = pywin.scintilla.formatter.STYLE_DEFAULT
210                  else:
211                      stylePyStart = stylePyStart.name
212                  startSeg = end
213                  i = end - 1  # ready for increment.
214                  chNext = cdoc[end : end + 1].decode("latin-1")
215                  state = STYLE_INTERACTIVE_EOL
216              if lastState != state:
217                  lastState = state
218              i = i + 1
219          # and the rest
220          if startSeg < i:
221              self.ColorSeg(startSeg, i - 1, state)

223      def Colorize(self, start=0, end=-1):
224          # scintilla's formatting is all done in terms of utf, so
225          # we work with utf8 bytes instead of unicode.  This magically
226          # works as any extended chars found in the utf8 don't change
227          # the semantics.
228          stringVal = self.scintilla.GetTextRange(start, end, decode=False)
229          styleStart = None
230          stylePyStart = None
231          if start > 1:
232              # Likely we are being asked to color from the start of the line.
233              # We find the last formatted character on the previous line.
234              # If TQString, we continue it.  Otherwise, we reset.
235              look = start - 1
236              while look and self.scintilla.SCIGetCharAt(look) in "\n\r":
237                  look = look - 1
238              if look and look < start - 1:  # Did we find a char before the \n\r sets?
239                  strstyle = self.GetStringStyle(look)
240                  quote_char = None
241                  if strstyle is not None:
242                      if strstyle.name == pywin.scintilla.formatter.STYLE_TQSSTRING:
243                          quote_char = "'"
244                      elif strstyle.name == pywin.scintilla.formatter.STYLE_TQDSTRING:
245                          quote_char = '"'
246                      if quote_char is not None:
247                          # It is a TQS.  If the TQS is not terminated, we
248                          # carry the style through.
249                          if look > 2:
250                              look_str = (
251                                  self.scintilla.SCIGetCharAt(look - 2)
252                                  + self.scintilla.SCIGetCharAt(look - 1)
253                                  + self.scintilla.SCIGetCharAt(look)
254                              )
255                              if look_str != quote_char * 3:
256                                  stylePyStart = strstyle.name
257          if stylePyStart is None:
258              stylePyStart = pywin.scintilla.formatter.STYLE_DEFAULT
259  
260          if start > 0:
261              stylenum = self.scintilla.SCIGetStyleAt(start - 1)
262              styleStart = self.GetStyleByNum(stylenum).name
263          elif self.bannerDisplayed:
264              styleStart = STYLE_INTERACTIVE_EOL
265          else:
266              styleStart = STYLE_INTERACTIVE_BANNER
267              self.bannerDisplayed = True
268          self.scintilla.SCIStartStyling(start, 31)
269          self.style_buffer = array.array("b", (0,) * len(stringVal))
270          self.ColorizeInteractiveCode(stringVal, styleStart, stylePyStart)
271          self.scintilla.SCISetStylingEx(self.style_buffer)
272          self.style_buffer = None

380      def AppendToPrompt(self, bufLines, oldPrompt=None):
381          "Take a command and stick it at the end of the buffer (with python prompts inserted if required)."
382          self.flush()
383          lastLineNo = self.GetLineCount() - 1
384          line = self.DoGetLine(lastLineNo)
385          if oldPrompt and line == oldPrompt:
386              self.SetSel(self.GetTextLength() - len(oldPrompt), self.GetTextLength())
387              self.ReplaceSel(sys.ps1)
388          elif line != str(sys.ps1):
389              if len(line) != 0:
390                  self.write("\n")
391              self.write(sys.ps1)
392          self.flush()
393          self.idle.text.mark_set("iomark", "end-1c")
394          if not bufLines:
395              return
396          terms = (["\n" + sys.ps2] * (len(bufLines) - 1)) + [""]
397          for bufLine, term in zip(bufLines, terms):
398              if bufLine.strip():
399                  self.write(bufLine + term)
400          self.flush()

429      # GetBlockBoundary takes a line number, and will return the
430      # start and and line numbers of the block, and a flag indicating if the
431      # block is a Python code block.
432      # If the line specified has a Python prompt, then the lines are parsed
433      # backwards and forwards, and the flag is true.
434      # If the line does not start with a prompt, the block is searched forward
435      # and backward until a prompt _is_ found, and all lines in between without
436      # prompts are returned, and the flag is false.
437      def GetBlockBoundary(self, lineNo):
438          line = self.DoGetLine(lineNo)
439          maxLineNo = self.GetLineCount() - 1
440          prefix = GetPromptPrefix(line)
441          if prefix is None:  # Non code block
442              flag = 0
443              startLineNo = lineNo
444              while startLineNo > 0:
445                  if GetPromptPrefix(self.DoGetLine(startLineNo - 1)) is not None:
446                      break  # there _is_ a prompt
447                  startLineNo = startLineNo - 1
448              endLineNo = lineNo
449              while endLineNo < maxLineNo:
450                  if GetPromptPrefix(self.DoGetLine(endLineNo + 1)) is not None:
451                      break  # there _is_ a prompt
452                  endLineNo = endLineNo + 1
453          else:  # Code block
454              flag = 1
455              startLineNo = lineNo
456              while startLineNo > 0 and prefix != str(sys.ps1):
457                  prefix = GetPromptPrefix(self.DoGetLine(startLineNo - 1))
458                  if prefix is None:
459                      break
460                      # there is no prompt.
461                  startLineNo = startLineNo - 1
462              endLineNo = lineNo
463              while endLineNo < maxLineNo:
464                  prefix = GetPromptPrefix(self.DoGetLine(endLineNo + 1))
465                  if prefix is None:
466                      break  # there is no prompt
467                  if prefix == str(sys.ps1):
468                      break  # this is another command
469                  endLineNo = endLineNo + 1
470                  # continue until end of buffer, or no prompt
471          return (startLineNo, endLineNo, flag)

473      def ExtractCommand(self, lines):
474          start, end = lines
475          retList = []
476          while end >= start:
477              thisLine = self.DoGetLine(end)
478              promptLen = len(GetPromptPrefix(thisLine))
479              retList = [thisLine[promptLen:]] + retList
480              end = end - 1
481          return retList

492          # a command may have overwritten these - only restore if not.

509      def ProcessEnterEvent(self, event):
510          # If autocompletion has been triggered, complete and do not process event
511          if self.SCIAutoCActive():
512              self.SCIAutoCComplete()
513              self.SCICancel()
514              return
515  
516          self.SCICancel()
517          # First, check for an error message
518          haveGrabbedOutput = 0
519          if self.HandleSpecialLine():
520              return 0
521  
522          lineNo = self.LineFromChar()
523          start, end, isCode = self.GetBlockBoundary(lineNo)
524          # If we are not in a code block just go to the prompt (or create a new one)
525          if not isCode:
526              self.AppendToPrompt([])
527              win32ui.SetStatusText(win32ui.LoadString(afxres.AFX_IDS_IDLEMESSAGE))
528              return
529  
530          lines = self.ExtractCommand((start, end))
531  
532          # If we are in a code-block, but it isnt at the end of the buffer
533          # then copy it to the end ready for editing and subsequent execution
534          if end != self.GetLineCount() - 1:
535              win32ui.SetStatusText("Press ENTER to execute command")
536              self.AppendToPrompt(lines)
537              self.SetSel(-2)
538              return
539  
540          # If SHIFT held down, we want new code here and now!
541          bNeedIndent = (
542              win32api.GetKeyState(win32con.VK_SHIFT) < 0
543              or win32api.GetKeyState(win32con.VK_CONTROL) < 0
544          )
545          if bNeedIndent:
546              self.ReplaceSel("\n")
547          else:
548              self.SetSel(-2)
549              self.ReplaceSel("\n")
550              source = "\n".join(lines)
551              while source and source[-1] in "\t ":
552                  source = source[:-1]
553              self.OutputGrab()  # grab the output for the command exec.
554              try:
555                  if self.interp.runsource(
556                      source, "<interactive input>"
557                  ):  # Need more input!
558                      bNeedIndent = 1
559                  else:
560                      # If the last line isnt empty, append a newline
561                      if self.history is not None:
562                          self.history.history_store(source)
563                      self.AppendToPrompt([])
564                      win32ui.SetStatusText(
565                          win32ui.LoadString(afxres.AFX_IDS_IDLEMESSAGE)
566                      )
567              #                   win32ui.SetStatusText('Successfully executed statement')
568              finally:
569                  self.OutputRelease()
570          if bNeedIndent:
571              win32ui.SetStatusText("Ready to continue the command")
572              # Now attempt correct indentation (should use IDLE?)
573              curLine = self.DoGetLine(lineNo)[len(sys.ps2) :]
574              pos = 0
575              indent = ""
576              while len(curLine) > pos and curLine[pos] in string.whitespace:
577                  indent = indent + curLine[pos]
578                  pos = pos + 1
579              if _is_block_opener(curLine):
580                  indent = indent + "\t"
581              elif _is_block_closer(curLine):
582                  indent = indent[:-1]
583              # use ReplaceSel to ensure it goes at the cursor rather than end of buffer.
584              self.ReplaceSel(sys.ps2 + indent)
585          return 0

606      def OnEditCopyCode(self, command, code):
607          """Sanitizes code from interactive window, removing prompts and output, 
608          and inserts it in the clipboard."""
609          code = self.GetSelText()
610          lines = code.splitlines()
611          out_lines = []
612          for line in lines:
613              if line.startswith(sys.ps1):
614                  line = line[len(sys.ps1) :]
615                  out_lines.append(line)
616              elif line.startswith(sys.ps2):
617                  line = line[len(sys.ps2) :]
618                  out_lines.append(line)
619          out_code = os.linesep.join(out_lines)
620          win32clipboard.OpenClipboard()
621          try:
622              win32clipboard.SetClipboardData(
623                  win32clipboard.CF_UNICODETEXT, str(out_code)
624              )
625          finally:
626              win32clipboard.CloseClipboard()

628      def OnEditExecClipboard(self, command, code):
629          """Executes python code directly from the clipboard."""
630          win32clipboard.OpenClipboard()
631          try:
632              code = win32clipboard.GetClipboardData(win32clipboard.CF_UNICODETEXT)
633          finally:
634              win32clipboard.CloseClipboard()
635  
636          code = code.replace("\r\n", "\n") + "\n"
637          try:
638              o = compile(code, "<clipboard>", "exec")
639              exec(o, __main__.__dict__)

674          parent = self.GetParentFrame()
675          if parent == win32ui.GetMainFrame():
676              # It is docked.
677              try:
678                  wnd, isactive = parent.MDIGetActive()
679                  wnd.SetFocus()
680              except win32ui.error:
681                  # No MDI window active!
682                  pass
683          else:
684              # Normal Window
685              try:

687                  # If the window is invalid, reset it.
688                  if lastActive is not None and (
689                      lastActive._obj_ is None or lastActive.GetSafeHwnd() == 0

692                      win32ui.SetStatusText("The last active Window has been closed.")
693              except AttributeError:
694                  print("Can't find the last active window!")
695                  lastActive = None
696              if lastActive is not None:
697                  lastActive.MDIActivate()

765  class DockedInteractiveView(DockedInteractiveViewParent):
766      def HookHandlers(self):
767          DockedInteractiveViewParent.HookHandlers(self)
768          self.HookMessage(self.OnSetFocus, win32con.WM_SETFOCUS)
769          self.HookMessage(self.OnKillFocus, win32con.WM_KILLFOCUS)
770  
771      def OnSetFocus(self, msg):
772          self.GetParentFrame().SetActiveView(self)
773          return 1
774  
775      def OnKillFocus(self, msg):
776          # If we are losing focus to another in this app, reset the main frame's active view.
777          hwnd = wparam = msg[2]
778          try:
779              wnd = win32ui.CreateWindowFromHandle(hwnd)
780              reset = wnd.GetTopLevelFrame() == self.GetTopLevelFrame()
781          except win32ui.error:
782              reset = 0  # Not my window
783          if reset:
784              self.GetParentFrame().SetActiveView(None)
785          return 1
786  
787      def OnDestroy(self, msg):
788          newSize = self.GetWindowPlacement()[4]
789          pywin.framework.app.SaveWindowSize("Interactive Window", newSize, "docked")
790          try:
791              if self.GetParentFrame().GetActiveView == self:
792                  self.GetParentFrame().SetActiveView(None)
793          except win32ui.error:
794              pass
795          try:
796              if win32ui.GetMainFrame().GetActiveView() == self:
797                  win32ui.GetMainFrame().SetActiveView(None)
798          except win32ui.error:
799              pass
800          return DockedInteractiveViewParent.OnDestroy(self, msg)

803  class CDockedInteractivePython(CInteractivePython):
804      def __init__(self, dockbar):
805          self.bFirstCreated = 0
806          self.dockbar = dockbar
807          CInteractivePython.__init__(self)
808  
809      def NeedRecreateWindow(self):
810          if self.bCreating:
811              return 0
812          try:
813              frame = win32ui.GetMainFrame()
814              if frame.closing:
815                  return 0  # Dieing!
816          except (win32ui.error, AttributeError):
817              return 0  # The app is dieing!
818          try:
819              cb = frame.GetControlBar(ID_DOCKED_INTERACTIVE_CONTROLBAR)
820              return not cb.IsWindowVisible()
821          except win32ui.error:
822              return 1  # Control bar does not exist!
823  
824      def RecreateWindow(self):
825          try:
826              dockbar = win32ui.GetMainFrame().GetControlBar(
827                  ID_DOCKED_INTERACTIVE_CONTROLBAR
828              )
829              win32ui.GetMainFrame().ShowControlBar(dockbar, 1, 1)
830          except win32ui.error:
831              CreateDockedInteractiveWindow()
832  
833      def Create(self):
834          self.bCreating = 1
835          doc = InteractiveDocument(None, self.DoCreateDoc())
836          view = DockedInteractiveView(doc)
837          defRect = pywin.framework.app.LoadWindowSize("Interactive Window", "docked")
838          if defRect[2] - defRect[0] == 0:
839              defRect = 0, 0, 500, 200
840          style = win32con.WS_CHILD | win32con.WS_VISIBLE | win32con.WS_BORDER
841          id = 1050  # win32ui.AFX_IDW_PANE_FIRST
842          view.CreateWindow(self.dockbar, id, style, defRect)
843          view.OnInitialUpdate()
844          self.bFirstCreated = 1
845  
846          self.currentView = doc.GetFirstView()
847          self.bCreating = 0
848          if self.title:
849              doc.SetTitle(self.title)

924  def DestroyInteractiveWindow():
925      """Destroy the interactive window. 
926      This is different to Closing the window, 
927      which may automatically re-appear.  Once destroyed, it can never be recreated, 
928      and a complete new instance must be created (which the various other helper 
929      functions will then do after making this call 
930      """
931      global edit
932      if edit is not None and edit.currentView is not None:
933          if edit.currentView.GetParentFrame() == win32ui.GetMainFrame():
934              # It is docked - do nothing now (this is only called at shutdown!)
935              pass
936          else:
937              # It is a standard window - call Close on the container.
938              edit.Close()
939              edit = None
940

942  def CloseInteractiveWindow():
943      """Close the interactive window, allowing it to be re-created on demand."""
944      global edit
945      if edit is not None and edit.currentView is not None:
946          if edit.currentView.GetParentFrame() == win32ui.GetMainFrame():
947              # It is docked, just hide the dock bar.
948              frame = win32ui.GetMainFrame()
949              cb = frame.GetControlBar(ID_DOCKED_INTERACTIVE_CONTROLBAR)
950              frame.ShowControlBar(cb, 0, 1)
951          else:
952              # It is a standard window - destroy the frame/view, allowing the object itself to remain.
953              edit.currentView.GetParentFrame().DestroyWindow()

956  def ToggleInteractiveWindow():
957      """If the interactive window is visible, hide it, otherwise show it."""
958      if edit is None:
959          CreateInteractiveWindow()
960      else:
961          if edit.NeedRecreateWindow():
962              edit.RecreateWindow()
963          else:
964              # Close it, allowing a reopen.
965              CloseInteractiveWindow()

968  def ShowInteractiveWindow():
969      """Shows (or creates if necessary) an interactive window"""
970      if edit is None:
971          CreateInteractiveWindow()
972      else:
973          if edit.NeedRecreateWindow():
974              edit.RecreateWindow()
975          else:
976              parent = edit.currentView.GetParentFrame()
977              if parent == win32ui.GetMainFrame():  # It is docked.
978                  edit.currentView.SetFocus()
979              else:  # It is a "normal" window
980                  edit.currentView.GetParentFrame().AutoRestore()
981                  win32ui.GetMainFrame().MDIActivate(edit.currentView.GetParentFrame())
