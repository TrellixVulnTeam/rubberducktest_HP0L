1    # The purpose of this test is to ensure that the gateways objects
2    # do the right thing WRT COM rules about object identity etc.
3    
4    # Also includes a basic test that we support inheritance correctly in
5    # gateway interfaces.
6    
7    # For our test, we create an object of type IID_IPersistStorage
8    # This interface derives from IPersist.
9    # Therefore, QI's for IID_IDispatch, IID_IUnknown, IID_IPersist and
10   # IID_IPersistStorage should all return the same gateway object.
11   #
12   # In addition, the interface should only need to declare itself as
13   # using the IPersistStorage interface, and as the gateway derives
14   # from IPersist, it should automatically be available without declaration.
15   #
16   # We also create an object of type IID_I??, and perform a QI for it.
17   # We then jump through a number of hoops, ensuring that the objects
18   # returned by the QIs follow all the rules.
19   #
20   # Here is Gregs summary of the rules:
21   # 1) the set of supported interfaces is static and unchanging
22   # 2) symmetric: if you QI an interface for that interface, it succeeds
23   # 3) reflexive: if you QI against A for B, the new pointer must succeed
24   #   for a QI for A
25   # 4) transitive: if you QI for B, then QI that for C, then QI'ing A for C
26   #   must succeed
27   #
28   #
29   # Note that 1) Requires cooperation of the Python programmer.  The rule to keep is:
30   # "whenever you return an _object_ from _query_interface_(), you must return the
31   # same object each time for a given IID.  Note that you must return the same
32   # _wrapped_ object
33   # you
34   # The rest are tested here.
35

43   # Check that the 2 objects both have identical COM pointers.
44   def CheckSameCOMObject(ob1, ob2):
45       addr1 = repr(ob1).split()[6][:-1]
46       addr2 = repr(ob2).split()[6][:-1]
47       return addr1 == addr2
48

50   # Check that the objects conform to COM identity rules.
51   def CheckObjectIdentity(ob1, ob2):
52       u1 = ob1.QueryInterface(pythoncom.IID_IUnknown)
53       u2 = ob2.QueryInterface(pythoncom.IID_IUnknown)
54       return CheckSameCOMObject(u1, u2)
55   
56   
57   def FailObjectIdentity(ob1, ob2, when):
58       if not CheckObjectIdentity(ob1, ob2):
59           global numErrors
60           numErrors = numErrors + 1
61           print(when, "are not identical (%s, %s)" % (repr(ob1), repr(ob2)))

91   def TestGatewayInheritance():
92       # By default, wrap() creates and discards a temporary object.
93       # This is not necessary, but just the current implementation of wrap.
94       # As the object is correctly discarded, it doesnt affect this test.
95       o = wrap(Dummy(), pythoncom.IID_IPersistStorage)
96       o2 = o.QueryInterface(pythoncom.IID_IUnknown)
97       FailObjectIdentity(o, o2, "IID_IPersistStorage->IID_IUnknown")
98   
99       o3 = o2.QueryInterface(pythoncom.IID_IDispatch)
100  
101      FailObjectIdentity(o2, o3, "IID_IUnknown->IID_IDispatch")
102      FailObjectIdentity(o, o3, "IID_IPersistStorage->IID_IDispatch")
103  
104      o4 = o3.QueryInterface(pythoncom.IID_IPersistStorage)
105      FailObjectIdentity(o, o4, "IID_IPersistStorage->IID_IPersistStorage(2)")
106      FailObjectIdentity(o2, o4, "IID_IUnknown->IID_IPersistStorage(2)")
107      FailObjectIdentity(o3, o4, "IID_IDispatch->IID_IPersistStorage(2)")
108  
109      o5 = o4.QueryInterface(pythoncom.IID_IPersist)
110      FailObjectIdentity(o, o5, "IID_IPersistStorage->IID_IPersist")
111      FailObjectIdentity(o2, o5, "IID_IUnknown->IID_IPersist")
112      FailObjectIdentity(o3, o5, "IID_IDispatch->IID_IPersist")
113      FailObjectIdentity(o4, o5, "IID_IPersistStorage(2)->IID_IPersist")

116  def TestMultiInterface():
117      o = wrap(Dummy2(), pythoncom.IID_IPersistStorage)
118      o2 = o.QueryInterface(pythoncom.IID_IExternalConnection)
119  
120      FailObjectIdentity(o, o2, "IID_IPersistStorage->IID_IExternalConnection")
121  
122      # Make the same QI again, to make sure it is stable.
123      o22 = o.QueryInterface(pythoncom.IID_IExternalConnection)
124      FailObjectIdentity(o, o22, "IID_IPersistStorage->IID_IExternalConnection")
125      FailObjectIdentity(
126          o2, o22, "IID_IPersistStorage->IID_IExternalConnection (stability)"
127      )

131      FailObjectIdentity(
