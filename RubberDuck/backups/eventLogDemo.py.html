9    def ReadLog(computer, logType="Application", dumpEachRecord=0):
10       # read the entire log back.
11       h = win32evtlog.OpenEventLog(computer, logType)
12       numRecords = win32evtlog.GetNumberOfEventLogRecords(h)
13       #       print "There are %d records" % numRecords
14   
15       num = 0
16       while 1:
17           objects = win32evtlog.ReadEventLog(
18               h,
19               win32evtlog.EVENTLOG_BACKWARDS_READ | win32evtlog.EVENTLOG_SEQUENTIAL_READ,
20               0,
21           )
22           if not objects:
23               break
24           for object in objects:
25               # get it for testing purposes, but dont print it.
26               msg = win32evtlogutil.SafeFormatMessage(object, logType)
27               if object.Sid is not None:
28                   try:
29                       domain, user, typ = win32security.LookupAccountSid(
30                           computer, object.Sid
31                       )
32                       sidDesc = "%s/%s" % (domain, user)
33                   except win32security.error:
34                       sidDesc = str(object.Sid)
35                   user_desc = "Event associated with user %s" % (sidDesc,)
36               else:
37                   user_desc = None
38               if dumpEachRecord:
39                   print(
40                       "Event record from %r generated at %s"
41                       % (object.SourceName, object.TimeGenerated.Format())
42                   )
43                   if user_desc:
44                       print(user_desc)
45                   try:
46                       print(msg)
47                   except UnicodeError:
48                       print("(unicode error printing message: repr() follows...)")
49                       print(repr(msg))
50   
51           num = num + len(objects)
52   
53       if numRecords == num:
54           print("Successfully read all", numRecords, "records")
55       else:
56           print(
57               "Couldn't get all records - reported %d, but found %d" % (numRecords, num)
58           )
59           print(
60               "(Note that some other app may have written records while we were running!)"
61           )
62       win32evtlog.CloseEventLog(h)

74   def test():
75       # check if running on Windows NT, if not, display notice and terminate
76       if win32api.GetVersion() & 0x80000000:
77           print("This sample only runs on NT")
78           return
79   
80       import sys, getopt
81   
82       opts, args = getopt.getopt(sys.argv[1:], "rwh?c:t:v")
83       computer = None
84       do_read = do_write = 1
85   
86       logType = "Application"
87       verbose = 0
88   
89       if len(args) > 0:
90           print("Invalid args")
91           usage()
92           return 1
93       for opt, val in opts:
94           if opt == "-t":
95               logType = val
96           if opt == "-c":
97               computer = val
98           if opt in ["-h", "-?"]:
99               usage()
100              return
101          if opt == "-r":
102              do_read = 0
103          if opt == "-w":
104              do_write = 0
105          if opt == "-v":
106              verbose = verbose + 1
107      if do_write:
108          ph = win32api.GetCurrentProcess()
109          th = win32security.OpenProcessToken(ph, win32con.TOKEN_READ)
110          my_sid = win32security.GetTokenInformation(th, win32security.TokenUser)[0]
111  
112          win32evtlogutil.ReportEvent(
113              logType,
114              2,
115              strings=["The message text for event 2", "Another insert"],
116              data="Raw\0Data".encode("ascii"),
117              sid=my_sid,
118          )
119          win32evtlogutil.ReportEvent(
120              logType,
121              1,
122              eventType=win32evtlog.EVENTLOG_WARNING_TYPE,
123              strings=["A warning", "An even more dire warning"],
124              data="Raw\0Data".encode("ascii"),
125              sid=my_sid,
126          )
127          win32evtlogutil.ReportEvent(
128              logType,
129              1,
130              eventType=win32evtlog.EVENTLOG_INFORMATION_TYPE,
131              strings=["An info", "Too much info"],
132              data="Raw\0Data".encode("ascii"),
133              sid=my_sid,
134          )
135          print("Successfully wrote 3 records to the log")
136  
137      if do_read:
138          ReadLog(computer, logType, verbose > 0)
