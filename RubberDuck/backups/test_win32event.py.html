9    class TestWaitableTimer(unittest.TestCase):
10       def testWaitableFire(self):
11           h = win32event.CreateWaitableTimer(None, 0, None)
12           dt = -160  # 160 ns.
13           win32event.SetWaitableTimer(h, dt, 0, None, None, 0)
14           rc = win32event.WaitForSingleObject(h, 1000)
15           self.failUnlessEqual(rc, win32event.WAIT_OBJECT_0)
16   
17       def testWaitableTrigger(self):
18           h = win32event.CreateWaitableTimer(None, 0, None)
19           # for the sake of this, pass a long that doesn't fit in an int.
20           dt = -2000000000
21           win32event.SetWaitableTimer(h, dt, 0, None, None, 0)
22           rc = win32event.WaitForSingleObject(h, 10)  # 10 ms.
23           self.failUnlessEqual(rc, win32event.WAIT_TIMEOUT)
24   
25       def testWaitableError(self):
26           h = win32event.CreateWaitableTimer(None, 0, None)
27           h.close()
28           self.assertRaises(
29               pywintypes.error, win32event.SetWaitableTimer, h, -42, 0, None, None, 0
30           )

57   class TestEvent(unittest.TestCase):
58       def assertSignaled(self, event):
59           self.assertEquals(
60               win32event.WaitForSingleObject(event, 0), win32event.WAIT_OBJECT_0
61           )
62   
63       def assertNotSignaled(self, event):
64           self.assertEquals(
65               win32event.WaitForSingleObject(event, 0), win32event.WAIT_TIMEOUT
66           )
67   
68       def testCreateEvent(self):
69           event = win32event.CreateEvent(None, False, False, None)
70           self.assertNotSignaled(event)
71           event = win32event.CreateEvent(None, False, True, None)
72           self.assertSignaled(event)
73           self.assertNotSignaled(event)
74           event = win32event.CreateEvent(None, True, True, None)
75           self.assertSignaled(event)
76           self.assertSignaled(event)
77   
78       def testSetEvent(self):
79           event = win32event.CreateEvent(None, True, False, None)
80           self.assertNotSignaled(event)
81           res = win32event.SetEvent(event)
82           self.assertEquals(res, None)
83           self.assertSignaled(event)
84           event.close()
85           self.assertRaises(pywintypes.error, win32event.SetEvent, event)
86   
87       def testResetEvent(self):
88           event = win32event.CreateEvent(None, True, True, None)
89           self.assertSignaled(event)

91           self.assertEquals(res, None)
