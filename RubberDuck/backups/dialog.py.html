14   def dllFromDll(dllid):
15       "given a 'dll' (maybe a dll, filename, etc), return a DLL object"
16       if dllid == None:
17           return None
18       elif type("") == type(dllid):
19           return win32ui.LoadLibrary(dllid)
20       else:
21           try:
22               dllid.GetFileName()
23           except AttributeError:
24               raise TypeError("DLL parameter must be None, a filename or a dll object")
25           return dllid

28   class Dialog(window.Wnd):
29       "Base class for a dialog"
30   
31       def __init__(self, id, dllid=None):
32           """id is the resource ID, or a template 
33           dllid may be None, a dll object, or a string with a dll name"""
34           # must take a reference to the DLL until InitDialog.
35           self.dll = dllFromDll(dllid)
36           if type(id) == type([]):  # a template
37               dlg = win32ui.CreateDialogIndirect(id)
38           else:
39               dlg = win32ui.CreateDialog(id, self.dll)
40           window.Wnd.__init__(self, dlg)
41           self.HookCommands()
42           self.bHaveInit = None
43   
44       def HookCommands(self):
45           pass
46   
47       def OnAttachedObjectDeath(self):
48           self.data = self._obj_.data
49           window.Wnd.OnAttachedObjectDeath(self)
50   
51       # provide virtuals.
52       def OnOK(self):
53           self._obj_.OnOK()
54   
55       def OnCancel(self):
56           self._obj_.OnCancel()
57   
58       def OnInitDialog(self):
59           self.bHaveInit = 1
60           if self._obj_.data:
61               self._obj_.UpdateData(0)
62           return 1  # I did NOT set focus to a child window.
63   
64       def OnDestroy(self, msg):
65           self.dll = None  # theoretically not needed if object destructs normally.
66   
67       # DDX support
68       def AddDDX(self, *args):
69           self._obj_.datalist.append(args)
70   
71       # Make a dialog object look like a dictionary for the DDX support
72       def __bool__(self):
73           return True
74   
75       def __len__(self):
76           return len(self.data)
77   
78       def __getitem__(self, key):
79           return self.data[key]
80   
81       def __setitem__(self, key, item):
82           self._obj_.data[key] = item  # self.UpdateData(0)
83   
84       def keys(self):
85           return list(self.data.keys())
86   
87       def items(self):
88           return list(self.data.items())
89   
90       def values(self):
91           return list(self.data.values())
92   
93       # XXX - needs py3k work!
94       def has_key(self, key):
95           return key in self.data

98   class PrintDialog(Dialog):
99       "Base class for a print dialog"
100  
101      def __init__(
102          self,
103          pInfo,
104          dlgID,
105          printSetupOnly=0,
106          flags=(
107              win32ui.PD_ALLPAGES
108              | win32ui.PD_USEDEVMODECOPIES
109              | win32ui.PD_NOPAGENUMS
110              | win32ui.PD_HIDEPRINTTOFILE
111              | win32ui.PD_NOSELECTION
112          ),
113          parent=None,
114          dllid=None,
115      ):
116          self.dll = dllFromDll(dllid)
117          if type(dlgID) == type([]):  # a template
118              raise TypeError("dlgID parameter must be an integer resource ID")
119          dlg = win32ui.CreatePrintDialog(dlgID, printSetupOnly, flags, parent, self.dll)
120          window.Wnd.__init__(self, dlg)
121          self.HookCommands()
122          self.bHaveInit = None
123          self.pInfo = pInfo
124          # init values (if PrintSetup is called, values still available)
125          flags = pInfo.GetFlags()
126          self["toFile"] = flags & win32ui.PD_PRINTTOFILE != 0
127          self["direct"] = pInfo.GetDirect()
128          self["preview"] = pInfo.GetPreview()
129          self["continuePrinting"] = pInfo.GetContinuePrinting()
130          self["curPage"] = pInfo.GetCurPage()
131          self["numPreviewPages"] = pInfo.GetNumPreviewPages()
132          self["userData"] = pInfo.GetUserData()
133          self["draw"] = pInfo.GetDraw()
134          self["pageDesc"] = pInfo.GetPageDesc()
135          self["minPage"] = pInfo.GetMinPage()
136          self["maxPage"] = pInfo.GetMaxPage()
137          self["offsetPage"] = pInfo.GetOffsetPage()
138          self["fromPage"] = pInfo.GetFromPage()
139          self["toPage"] = pInfo.GetToPage()
140          # these values updated after OnOK
141          self["copies"] = 0
142          self["deviceName"] = ""
143          self["driverName"] = ""
144          self["printAll"] = 0
145          self["printCollate"] = 0
146          self["printRange"] = 0
147          self["printSelection"] = 0
148  
149      def OnInitDialog(self):
150          self.pInfo.CreatePrinterDC()  # This also sets the hDC of the pInfo structure.
151          return self._obj_.OnInitDialog()
152  
153      def OnCancel(self):
154          del self.pInfo
155  
156      def OnOK(self):
157          """DoModal has finished. Can now access the users choices"""
158          self._obj_.OnOK()
159          pInfo = self.pInfo
160          # user values
161          flags = pInfo.GetFlags()
162          self["toFile"] = flags & win32ui.PD_PRINTTOFILE != 0
163          self["direct"] = pInfo.GetDirect()
164          self["preview"] = pInfo.GetPreview()
165          self["continuePrinting"] = pInfo.GetContinuePrinting()
166          self["curPage"] = pInfo.GetCurPage()
167          self["numPreviewPages"] = pInfo.GetNumPreviewPages()
168          self["userData"] = pInfo.GetUserData()
169          self["draw"] = pInfo.GetDraw()
170          self["pageDesc"] = pInfo.GetPageDesc()
171          self["minPage"] = pInfo.GetMinPage()
172          self["maxPage"] = pInfo.GetMaxPage()
173          self["offsetPage"] = pInfo.GetOffsetPage()
174          self["fromPage"] = pInfo.GetFromPage()
175          self["toPage"] = pInfo.GetToPage()
176          self["copies"] = pInfo.GetCopies()
177          self["deviceName"] = pInfo.GetDeviceName()
178          self["driverName"] = pInfo.GetDriverName()
179          self["printAll"] = pInfo.PrintAll()
180          self["printCollate"] = pInfo.PrintCollate()
181          self["printRange"] = pInfo.PrintRange()
182          self["printSelection"] = pInfo.PrintSelection()
183          del self.pInfo

186  class PropertyPage(Dialog):
187      "Base class for a Property Page"
188  
189      def __init__(self, id, dllid=None, caption=0):
190          """id is the resource ID 
191          dllid may be None, a dll object, or a string with a dll name"""
192  
193          self.dll = dllFromDll(dllid)
194          if self.dll:
195              oldRes = win32ui.SetResource(self.dll)
196          if type(id) == type([]):
197              dlg = win32ui.CreatePropertyPageIndirect(id)
198          else:
199              dlg = win32ui.CreatePropertyPage(id, caption)
200          if self.dll:
201              win32ui.SetResource(oldRes)
202          # dont call dialog init!
203          window.Wnd.__init__(self, dlg)
204          self.HookCommands()

207  class PropertySheet(window.Wnd):
208      def __init__(self, caption, dll=None, pageList=None):  # parent=None, style,etc):
209          "Initialize a property sheet.  pageList is a list of ID's"
210          # must take a reference to the DLL until InitDialog.
211          self.dll = dllFromDll(dll)
212          self.sheet = win32ui.CreatePropertySheet(caption)
213          window.Wnd.__init__(self, self.sheet)
214          if not pageList is None:
215              self.AddPage(pageList)
216  
217      def OnInitDialog(self):
218          return self._obj_.OnInitDialog()
219  
220      def DoModal(self):
221          if self.dll:
222              oldRes = win32ui.SetResource(self.dll)
223          rc = self.sheet.DoModal()
224          if self.dll:
225              win32ui.SetResource(oldRes)
226          return rc
227  
228      def AddPage(self, pages):
229          if self.dll:
230              oldRes = win32ui.SetResource(self.dll)
231          try:  # try list style access
232              pages[0]
233              isSeq = 1
234          except (TypeError, KeyError):
235              isSeq = 0
236          if isSeq:
237              for page in pages:
238                  self.DoAddSinglePage(page)
239          else:
240              self.DoAddSinglePage(pages)
241          if self.dll:
242              win32ui.SetResource(oldRes)
243  
244      def DoAddSinglePage(self, page):
245          "Page may be page, or int ID. Assumes DLL setup"
246          if type(page) == type(0):
247              self.sheet.AddPage(win32ui.CreatePropertyPage(page))
248          else:
249              self.sheet.AddPage(page)

253  def GetSimpleInput(prompt, defValue="", title=None):
254      """displays a dialog, and returns a string, or None if cancelled. 
255      args prompt, defValue='', title=main frames title"""
256      # uses a simple dialog to return a string object.
257      if title is None:
258          title = win32ui.GetMainFrame().GetWindowText()
259      # 2to3 insists on converting 'Dialog.__init__' to 'tkinter.dialog...'
260      DlgBaseClass = Dialog
261  
262      class DlgSimpleInput(DlgBaseClass):
263          def __init__(self, prompt, defValue, title):
264              self.title = title
265              DlgBaseClass.__init__(self, win32ui.IDD_SIMPLE_INPUT)
266              self.AddDDX(win32ui.IDC_EDIT1, "result")
267              self.AddDDX(win32ui.IDC_PROMPT1, "prompt")
268              self._obj_.data["result"] = defValue
269              self._obj_.data["prompt"] = prompt
270  
271          def OnInitDialog(self):
272              self.SetWindowText(self.title)
273              return DlgBaseClass.OnInitDialog(self)
274  
275      dlg = DlgSimpleInput(prompt, defValue, title)
276      if dlg.DoModal() != win32con.IDOK:
277          return None
278      return dlg["result"]
