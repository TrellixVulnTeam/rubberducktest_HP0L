1    ''' 
2    Clipboard Gtk3: an implementation of the Clipboard using Gtk3. 
3    '''
4    
5    __all__ = ('ClipboardGtk3',)
6    
7    from kivy.utils import platform
8    from kivy.support import install_gobject_iteration
9    from kivy.core.clipboard import ClipboardBase
10   
11   if platform != 'linux':
12       raise SystemError('unsupported platform for gtk3 clipboard')
13   
14   import gi
15   gi.require_version('Gtk', '3.0')
16   from gi.repository import Gtk, Gdk
17   clipboard = Gtk.Clipboard.get(Gdk.SELECTION_CLIPBOARD)
18   
19   
20   class ClipboardGtk3(ClipboardBase):
21   
22       _is_init = False
23   
24       def init(self):
25           if self._is_init:
26               return
27           install_gobject_iteration()
28           self._is_init = True
29   
30       def get(self, mimetype='text/plain;charset=utf-8'):
31           self.init()
32           if mimetype == 'text/plain;charset=utf-8':
33               contents = clipboard.wait_for_text()
34               if contents:
35                   return contents
36           return ''
37   
38       def put(self, data, mimetype='text/plain;charset=utf-8'):
39           self.init()
40           if mimetype == 'text/plain;charset=utf-8':
41               text = data.decode(self._encoding)
42               clipboard.set_text(text, -1)
43               clipboard.store()
44   
45       def get_types(self):
46           self.init()
47           return ['text/plain;charset=utf-8']
