86   (defconst kivy-blank-line-re "^ *$"
87     "Regexp matching a line containing only (valid) whitespace.")
88   
89   (defconst kivy-comment-re "\\(?:^\\|\\s-+\\)\\(#.*\\)"
90     "Regexp matching a line containing a kivy comment or delimiter.")
91   
92   (defconst kivy-directive-re "^\\(?:#:\\)\\(\\w+ +.*\\)"
93     "Regexp matching a line containing a kivy directive.")
94   
95   (defconst kivy-tag-re "^ *id: *\\([^ \n]+\\)$"
96     "Rexexp matching a kivy tag.")
97   
98   (defconst kivy-bare-scalar-re
99     "\\(?:[^-:,#!\n{\\[ ]\\|[^#!\n{\\[ ]\\S-\\)[^#\n]*?"
100    "Rexexp matching a kivy bare scalar.")
101  
102  (defconst kivy-hash-key-re
103    (concat "^ *"
104            "\\(" kivy-bare-scalar-re "\\) *:"
105            "\\(?: +\\|$\\)")
106    "Regexp matching a single kivy hash key.")
107  
108  (defconst kivy-nested-map-re
109    (concat ".*: *$")
110    "Regexp matching a line beginning a kivy nested structure.")
111  
112  (defconst kivy-constant-scalars-re
113    (concat "\\(?:^\\|\\(?::\\|-\\|,\\|{\\|\\[\\) +\\) *"
114            (regexp-opt
115             '("True" "False" "None") t)
116            " *$")
117    "Regexp matching certain scalar constants in scalar context")

123  (defvar kivy-mode-map ()
124    "Keymap used in `kivy-mode' buffers.")
125  (if kivy-mode-map
126      nil
127    (setq kivy-mode-map (make-sparse-keymap))
128    (define-key kivy-mode-map [backspace] 'kivy-electric-backspace)
129    (define-key kivy-mode-map "\C-c<" 'kivy-indent-shift-left)
130    (define-key kivy-mode-map "\C-c>" 'kivy-indent-shift-right)
131    )
132  
133  (defvar kivy-mode-syntax-table nil
134    "Syntax table in use in kivy-mode buffers.")
135  (if kivy-mode-syntax-table
136      nil
137    (setq kivy-mode-syntax-table (make-syntax-table))
138    (modify-syntax-entry ?\' "\"" kivy-mode-syntax-table)
139    (modify-syntax-entry ?\" "\"" kivy-mode-syntax-table)
140    (modify-syntax-entry ?# "<" kivy-mode-syntax-table)
141    (modify-syntax-entry ?\n ">" kivy-mode-syntax-table)
142    (modify-syntax-entry ?\\ "\\" kivy-mode-syntax-table)
143    (modify-syntax-entry ?- "_" kivy-mode-syntax-table)
144    (modify-syntax-entry ?_ "w" kivy-mode-syntax-table)
145    (modify-syntax-entry ?< "." kivy-mode-syntax-table)
146    (modify-syntax-entry ?> "." kivy-mode-syntax-table)
147    (modify-syntax-entry ?_ "_" kivy-mode-syntax-table)
148    )

188  (defun kivy-compute-indentation ()
189    "Calculate the maximum sensible indentation for the current line."
190    (save-excursion
191      (beginning-of-line)
192      (forward-line -1)
193      (while (and (looking-at kivy-blank-line-re)
194                  (> (point) (point-min)))
195        (forward-line -1))
196      (+ (current-indentation)
197         (if (looking-at kivy-nested-map-re) kivy-indent-offset 0)
198         )))
199  
200  (defun kivy-indent-line ()
201    "Indent the current line.
202  The first time this command is used, the line will be indented to the
203  maximum sensible indentation.  Each immediately subsequent usage will
204  back-dent the line by `kivy-indent-offset' spaces.  On reaching column
205  0, it will cycle back to the maximum sensible indentation."
206    (interactive "*")
207    (let ((ci (current-indentation))
208          (cc (current-column))
209          (need (kivy-compute-indentation)))
210      (save-excursion
211        (beginning-of-line)
212        (delete-horizontal-space)
213        (if (and (equal last-command this-command) (/= ci 0))
214            (indent-to (* (/ (- ci 1) kivy-indent-offset) kivy-indent-offset))
215          (indent-to need)))
216      (if (< (current-column) (current-indentation))
217          (forward-to-indentation 0))))
218  
219  (defun kivy-electric-backspace (arg)
220    "Delete characters or back-dent the current line.
221  If invoked following only whitespace on a line, will back-dent to the
222  immediately previous multiple of `kivy-indent-offset' spaces."
223    (interactive "*p")
224    (if (or (/= (current-indentation) (current-column)) (bolp))
225        (funcall kivy-backspace-function arg)
226      (let ((ci (current-column)))
227        (beginning-of-line)
228        (delete-horizontal-space)
229        (indent-to (* (/ (- ci (* arg kivy-indent-offset))
230                         kivy-indent-offset)
231                      kivy-indent-offset)))))

252  (defun kivy-indent-shift-left (start end &optional count)
253    "Shift lines contained in region START END by COUNT columns to the left.
254  COUNT defaults to `kivy-indent-offset'.  If region isn't
255  active, the current line is shifted.  The shifted region includes
256  the lines in which START and END lie.  An error is signaled if
257  any lines in the region are indented less than COUNT columns."
258    (interactive
259     (if mark-active
260         (list (region-beginning) (region-end) current-prefix-arg)
261       (list (line-beginning-position) (line-end-position) current-prefix-arg)))
262    (if count
263        (setq count (prefix-numeric-value count))
264      (setq count kivy-indent-offset))
265    (when (> count 0)
266      (let ((deactivate-mark nil))
267        (save-excursion
268          (goto-char start)
269          (while (< (point) end)
270            (if (and (< (current-indentation) count)
271                     (not (looking-at "[ \t]*$")))
272                (error "Can't shift all lines enough"))
273            (forward-line))
274          (indent-rigidly start end (- count))))))
275  
276  (defun kivy-indent-shift-right (start end &optional count)
277    "Shift lines contained in region START END by COUNT columns to the left.
278  COUNT defaults to `kivy-indent-offset'.  If region isn't
279  active, the current line is shifted.  The shifted region includes
280  the lines in which START and END lie."
281    (interactive
282     (if mark-active
283         (list (region-beginning) (region-end) current-prefix-arg)
284       (list (line-beginning-position) (line-end-position) current-prefix-arg)))
285    (let ((deactivate-mark nil))
286      (if count
287          (setq count (prefix-numeric-value count))
288        (setq count kivy-indent-offset))
289      (indent-rigidly start end count)))
