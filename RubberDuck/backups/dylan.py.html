24       .. versionadded:: 0.7

28       aliases = ['dylan']

30       mimetypes = ['text/x-dylan']

90       valid_name = '\\\\?[\\w!&*<>|^$%@\\-+~?/=]+'
91   
92       def get_tokens_unprocessed(self, text):
93           for index, token, value in RegexLexer.get_tokens_unprocessed(self, text):
94               if token is Name:
95                   lowercase_value = value.lower()
96                   if lowercase_value in self.builtins:
97                       yield index, Name.Builtin, value
98                       continue
99                   if lowercase_value in self.keywords:
100                      yield index, Keyword, value
101                      continue
102                  if lowercase_value in self.functions:
103                      yield index, Name.Builtin, value
104                      continue
105                  if lowercase_value in self.operators:
106                      yield index, Operator, value
107                      continue
108              yield index, token, value

110      tokens = {
111          'root': [
112              # Whitespace
113              (r'\s+', Text),
114  
115              # single line comment
116              (r'//.*?\n', Comment.Single),
117  
118              # lid header
119              (r'([a-z0-9-]+)(:)([ \t]*)(.*(?:\n[ \t].+)*)',
120                  bygroups(Name.Attribute, Operator, Text, String)),
121  
122              default('code')  # no header match, switch to code
123          ],
124          'code': [
125              # Whitespace
126              (r'\s+', Text),
127  
128              # single line comment
129              (r'//.*?\n', Comment.Single),
130  
131              # multi-line comment
132              (r'/\*', Comment.Multiline, 'comment'),
133  
134              # strings and characters
135              (r'"', String, 'string'),
136              (r"'(\\.|\\[0-7]{1,3}|\\x[a-f0-9]{1,2}|[^\\\'\n])'", String.Char),
137  
138              # binary integer
139              (r'#b[01]+', Number.Bin),
140  
141              # octal integer
142              (r'#o[0-7]+', Number.Oct),
143  
144              # floating point
145              (r'[-+]?(\d*\.\d+(e[-+]?\d+)?|\d+(\.\d*)?e[-+]?\d+)', Number.Float),
146  
147              # decimal integer
148              (r'[-+]?\d+', Number.Integer),
149  
150              # hex integer
151              (r'#x[0-9a-f]+', Number.Hex),
152  
153              # Macro parameters
154              (r'(\?' + valid_name + ')(:)'
155               r'(token|name|variable|expression|body|case-body|\*)',
156                  bygroups(Name.Tag, Operator, Name.Builtin)),
157              (r'(\?)(:)(token|name|variable|expression|body|case-body|\*)',
158                  bygroups(Name.Tag, Operator, Name.Builtin)),
159              (r'\?' + valid_name, Name.Tag),
160  
161              # Punctuation
162              (r'(=>|::|#\(|#\[|##|\?\?|\?=|\?|[(){}\[\],.;])', Punctuation),
163  
164              # Most operators are picked up as names and then re-flagged.
165              # This one isn't valid in a name though, so we pick it up now.
166              (r':=', Operator),
167  
168              # Pick up #t / #f before we match other stuff with #.
169              (r'#[tf]', Literal),
170  
171              # #"foo" style keywords
172              (r'#"', String.Symbol, 'keyword'),
173  
174              # #rest, #key, #all-keys, etc.
175              (r'#[a-z0-9-]+', Keyword),
176  
177              # required-init-keyword: style keywords.
178              (valid_name + ':', Keyword),
179  
180              # class names
181              ('<' + valid_name + '>', Name.Class),
182  
183              # define variable forms.
184              (r'\*' + valid_name + r'\*', Name.Variable.Global),
185  
186              # define constant forms.
187              (r'\$' + valid_name, Name.Constant),
188  
189              # everything else. We re-flag some of these in the method above.
190              (valid_name, Name),
191          ],
192          'comment': [
193              (r'[^*/]', Comment.Multiline),
194              (r'/\*', Comment.Multiline, '#push'),
195              (r'\*/', Comment.Multiline, '#pop'),
196              (r'[*/]', Comment.Multiline)
197          ],
198          'keyword': [
199              (r'"', String.Symbol, '#pop'),
200              (r'[^\\"]+', String.Symbol),  # all other characters
201          ],
202          'string': [
203              (r'"', String, '#pop'),
204              (r'\\([\\abfnrtv"\']|x[a-f0-9]{2,4}|[0-7]{1,3})', String.Escape),
205              (r'[^\\"\n]+', String),  # all other characters
206              (r'\\\n', String),  # line continuation
207              (r'\\', String),  # stray backslash
208          ]
209      }

241  class DylanConsoleLexer(Lexer):
242      """ 
243      For Dylan interactive console output like: 
244   
245      .. sourcecode:: dylan-console 
246   
247          ? let a = 1; 
248          => 1 
249          ? a 
250          => 1 
251   
252      This is based on a copy of the RubyConsoleLexer. 
253   
254      .. versionadded:: 1.6 
255      """
256      name = 'Dylan session'
257      aliases = ['dylan-console', 'dylan-repl']
258      filenames = ['*.dylan-console']
259      mimetypes = ['text/x-dylan-console']
260  
261      _line_re = re.compile('.*?\n')
262      _prompt_re = re.compile(r'\?| ')
263  
264      def get_tokens_unprocessed(self, text):
265          dylexer = DylanLexer(**self.options)
266  
267          curcode = ''
268          insertions = []
269          for match in self._line_re.finditer(text):
270              line = match.group()
271              m = self._prompt_re.match(line)
272              if m is not None:
273                  end = m.end()
274                  insertions.append((len(curcode),
275                                     [(0, Generic.Prompt, line[:end])]))
276                  curcode += line[end:]
277              else:
278                  if curcode:
279                      yield from do_insertions(insertions,
280                                               dylexer.get_tokens_unprocessed(curcode))
281                      curcode = ''
282                      insertions = []
283                  yield match.start(), Generic.Output, line
284          if curcode:
285              yield from do_insertions(insertions,
286                                       dylexer.get_tokens_unprocessed(curcode))
