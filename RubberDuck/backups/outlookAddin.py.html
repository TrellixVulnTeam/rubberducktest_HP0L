1    # A demo plugin for Microsoft Outlook (NOT Outlook Express)
2    #
3    # This addin simply adds a new button to the main Outlook toolbar,
4    # and displays a message box when clicked.  Thus, it demonstrates
5    # how to plug in to Outlook itself, and hook outlook events.
6    #
7    # Additionally, each time a new message arrives in the Inbox, a message
8    # is printed with the subject of the message.
9    #
10   # To register the addin, simply execute:
11   #   outlookAddin.py
12   # This will install the COM server, and write the necessary
13   # AddIn key to Outlook
14   #
15   # To unregister completely:
16   #   outlookAddin.py --unregister
17   #
18   # To debug, execute:
19   #   outlookAddin.py --debug
20   #
21   # Then open Pythonwin, and select "Tools->Trace Collector Debugging Tool"
22   # Restart Outlook, and you should see some output generated.
23   #
24   # NOTE: If the AddIn fails with an error, Outlook will re-register
25   # the addin to not automatically load next time Outlook starts.  To
26   # correct this, simply re-register the addin (see above)
27

52           import win32ui  # Possible, but not necessary, to use a Pythonwin GUI

54           win32ui.MessageBox("Hello from Python")
55           return cancel

58   class FolderEvent:
59       def OnItemAdd(self, item):
60           try:
61               print("An item was added to the inbox with subject:", item.Subject)
62           except AttributeError:
63               print(
64                   "An item was added to the inbox, but it has no subject! - ", repr(item)
65               )

68   class OutlookAddin:
69       _com_interfaces_ = ["_IDTExtensibility2"]
70       _public_methods_ = []
71       _reg_clsctx_ = pythoncom.CLSCTX_INPROC_SERVER
72       _reg_clsid_ = "{0F47D9F3-598B-4d24-B7E3-92AC15ED27E2}"
73       _reg_progid_ = "Python.Test.OutlookAddin"
74       _reg_policy_spec_ = "win32com.server.policy.EventHandlerPolicy"
75   
76       def OnConnection(self, application, connectMode, addin, custom):
77           print("OnConnection", application, connectMode, addin, custom)
78           # ActiveExplorer may be none when started without a UI (eg, WinCE synchronisation)
79           activeExplorer = application.ActiveExplorer()
80           if activeExplorer is not None:
81               bars = activeExplorer.CommandBars
82               toolbar = bars.Item("Standard")
83               item = toolbar.Controls.Add(Type=constants.msoControlButton, Temporary=True)
84               # Hook events for the item
85               item = self.toolbarButton = DispatchWithEvents(item, ButtonEvent)
86               item.Caption = "Python"
87               item.TooltipText = "Click for Python"
88               item.Enabled = True
89   
90           # And now, for the sake of demonstration, setup a hook for all new messages
91           inbox = application.Session.GetDefaultFolder(constants.olFolderInbox)
92           self.inboxItems = DispatchWithEvents(inbox.Items, FolderEvent)
93   
94       def OnDisconnection(self, mode, custom):
95           print("OnDisconnection")
96   
97       def OnAddInsUpdate(self, custom):
98           print("OnAddInsUpdate", custom)
99   
100      def OnStartupComplete(self, custom):
101          print("OnStartupComplete", custom)
102  
103      def OnBeginShutdown(self, custom):
104          print("OnBeginShutdown", custom)

107  def RegisterAddin(klass):
108      import winreg
109  
110      key = winreg.CreateKey(
111          winreg.HKEY_CURRENT_USER, "Software\\Microsoft\\Office\\Outlook\\Addins"
112      )
113      subkey = winreg.CreateKey(key, klass._reg_progid_)
114      winreg.SetValueEx(subkey, "CommandLineSafe", 0, winreg.REG_DWORD, 0)
115      winreg.SetValueEx(subkey, "LoadBehavior", 0, winreg.REG_DWORD, 3)
116      winreg.SetValueEx(subkey, "Description", 0, winreg.REG_SZ, klass._reg_progid_)
117      winreg.SetValueEx(subkey, "FriendlyName", 0, winreg.REG_SZ, klass._reg_progid_)
118  
119  
120  def UnregisterAddin(klass):
121      import winreg
122  
123      try:
124          winreg.DeleteKey(
125              winreg.HKEY_CURRENT_USER,
126              "Software\\Microsoft\\Office\\Outlook\\Addins\\" + klass._reg_progid_,
127          )
128      except WindowsError:
129          pass
