2    # bulkstamp.py:
3    #    Stamp versions on all files that can be found in a given tree.
4    #
5    # USAGE: python bulkstamp.py <version> <root directory> <descriptions>
6    #
7    # Example: python bulkstamp.py 103 ..\win32\Build\ desc.txt
8    #
9    # <version> corresponds to the build number. It will be concatenated with
10   # the major and minor version numbers found in the description file.
11   #
12   # Description information is pulled from an input text file with lines of
13   # the form:
14   #
15   #    <basename> <white space> <description>
16   #
17   # For example:
18   #
19   #    PyWinTypes.dll Common types for Python on Win32
20   #    etc
21   #
22   # The product's name, major, and minor versions are specified as:
23   #
24   #    name <white space> <value>
25   #    major <white space> <value>
26   #    minor <white space> <value>
27   #
28   # The tags are case-sensitive.
29   #
30   # Any line beginning with "#" will be ignored. Empty lines are okay.
31   #
32   
33   import sys
34   import os
35   import verstamp
36   import fnmatch
37   import win32api
38   
39   numStamped = 0
40   
41   g_patterns = [
42       "*.dll",
43       "*.pyd",
44       "*.exe",
45       "*.ocx",
46   ]
47   
48   
49   def walk(arg, dirname, names):
50       global numStamped
51       vars, debug, descriptions = arg
52       for name in names:
53           for pat in g_patterns:
54               if fnmatch.fnmatch(name, pat):
55                   # Handle the "_d" thing.
56                   pathname = os.path.join(dirname, name)
57                   base, ext = os.path.splitext(name)
58                   if base[-2:] == "_d":
59                       name = base[:-2] + ext
60                   is_dll = ext.lower() != ".exe"
61                   if os.path.normcase(name) in descriptions:
62                       desc = descriptions[os.path.normcase(name)]
63                       try:
64                           verstamp.stamp(vars, pathname, desc, is_dll=is_dll)
65                           numStamped = numStamped + 1
66                       except win32api.error as exc:
67                           print(
68                               "Could not stamp",
69                               pathname,
70                               "Error",
71                               exc.winerror,
72                               "-",
73                               exc.strerror,
74                           )
75                   else:
76                       print("WARNING: description not provided for:", name)
77                       # skip branding this - assume already branded or handled elsewhere
78   
79   
80   #        print "Stamped", pathname

83   def load_descriptions(fname, vars):
84       retvars = {}
85       descriptions = {}
86   
87       lines = open(fname, "r").readlines()
88   
89       for i in range(len(lines)):
90           line = lines[i].strip()
91           if line != "" and line[0] != "#":
92               idx1 = line.find(" ")
93               idx2 = line.find("\t")
94               if idx1 == -1 or idx2 < idx1:
95                   idx1 = idx2
96               if idx1 == -1:
97                   print("ERROR: bad syntax in description file at line %d." % (i + 1))
98                   sys.exit(1)
99   
100              key = line[:idx1]
101              val = line[idx1:].strip()
102              if key in vars:
103                  retvars[key] = val
104              else:
105                  descriptions[key] = val
106  
107      if "product" not in retvars:
108          print("ERROR: description file is missing the product name.")
109          sys.exit(1)
110      if "major" not in retvars:
111          print("ERROR: description file is missing the major version number.")
112          sys.exit(1)
113      if "minor" not in retvars:
114          print("ERROR: description file is missing the minor version number.")
115          sys.exit(1)
116  
117      return retvars, descriptions

120  def scan(build, root, desc, **custom_vars):
121      global numStamped
122      numStamped = 0
123      try:
124          build = int(build)
125      except ValueError:
126          print("ERROR: build number is not a number: %s" % build)
127          sys.exit(1)
128  
129      debug = 0  ### maybe fix this one day
130  
131      varList = ["major", "minor", "sub", "company", "copyright", "trademarks", "product"]
132  
133      vars, descriptions = load_descriptions(desc, varList)
134      vars["build"] = build
135      vars.update(custom_vars)
136  
137      arg = vars, debug, descriptions
138      os.path.walk(root, walk, arg)
139  
140      print("Stamped %d files." % (numStamped))
141  
142  
143  if __name__ == "__main__":
144      if len(sys.argv) != 4:
145          print("ERROR: incorrect invocation. See script's header comments.")
146          sys.exit(1)
147  
148      scan(*tuple(sys.argv[1:]))
