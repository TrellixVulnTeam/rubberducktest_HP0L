36   class PerfApp(App, FloatLayout):
37   
38       def build(self):
39           return self
40   
41       def __init__(self, **kwargs):
42           super(PerfApp, self).__init__(**kwargs)
43           self.tests = []
44           tests = (self.load_large_text, self.stress_insert,
45               self.stress_del, self.stress_selection)
46           for test in tests:
47               but = type(self.but)(text=test.__name__)
48               self.but.parent.add_widget(but)
49               but.test = test
50               self.tests.append(but)
51           self.test_done = True

53       def load_large_text(self, *largs):
54           print('loading uix/textinput.py....')
55           self.test_done = False
56           fd = open(resource_find('uix/textinput.py'), 'r')
57           print('putting text in textinput')
58   
59           def load_text(*l):
60               self.text_input.text = fd.read()
61   
62           t = timeit.Timer(load_text)
63           ttk = t.timeit(1)
64           fd.close()
65           import resource
66           print('mem usage after test')
67           print(resource.getrusage(resource.RUSAGE_SELF).ru_maxrss / 1024, 'MB')
68           print('------------------------------------------')
69           print('Loaded', len(self.text_input._lines), 'lines', ttk, 'secs')
70           print('------------------------------------------')
71           self.test_done = True

73       def stress_del(self, *largs):
74           self.test_done = False
75           text_input = self.text_input
76           self.lt = len_text = len(text_input.text)
77           target = len_text - (210 * 9)
78           self.tot_time = 0
79           ev = None
80   
81           def dlt(*l):
82               if len(text_input.text) <= target:
83                   ev.cancel()
84                   print('Done!')
85                   m_len = len(text_input._lines)
86                   print('deleted 210 characters 9 times')
87                   import resource
88                   print('mem usage after test')
89                   print(resource.getrusage(resource.RUSAGE_SELF).ru_maxrss /
90                       1024, 'MB')
91                   print('total lines in text input:', m_len)
92                   print('--------------------------------------')
93                   print('total time elapsed:', self.tot_time)
94                   print('--------------------------------------')
95                   self.test_done = True
96                   return
97               text_input.select_text(self.lt - 220, self.lt - 10)
98               text_input.delete_selection()
99               self.lt -= 210
100              text_input.scroll_y -= 100
101              self.tot_time += l[0]
102              ev()
103          ev = Clock.create_trigger(dlt)
104          ev()

106      def stress_insert(self, *largs):
107          self.test_done = False
108          text_input = self.text_input
109          text_input.select_all()
110          text_input._copy(text_input.selection_text)
111          text_input.cursor = text_input.get_cursor_from_index(
112              text_input.selection_to)
113          len_text = len(text_input._lines)
114          self.tot_time = 0
115          ev = None
116  
117          def pste(*l):
118              if len(text_input._lines) >= (len_text) * 9:
119                  ev.cancel()
120                  print('Done!')
121                  m_len = len(text_input._lines)
122                  print('pasted', len_text, 'lines',
123                      round((m_len - len_text) / len_text), 'times')
124                  import resource
125                  print('mem usage after test')
126                  print(resource.getrusage(resource.RUSAGE_SELF).ru_maxrss /
127                      1024, 'MB')
128                  print('total lines in text input:', m_len)
129                  print('--------------------------------------')
130                  print('total time elapsed:', self.tot_time)
131                  print('--------------------------------------')
132                  self.test_done = True
133                  return
134              self.tot_time += l[0]
135              text_input._paste()
136              ev()
137          ev = Clock.create_trigger(pste)
138          ev()
139  
140      def stress_selection(self, *largs):
141          self.test_done = False
142          text_input = self.text_input
143          self.tot_time = 0
144          old_selection_from = text_input.selection_from - 210
145          ev = None
146  
147          def pste(*l):
148              if text_input.selection_from >= old_selection_from:
149                  ev.cancel()
150                  print('Done!')
151                  import resource
152                  print('mem usage after test')
153                  print(resource.getrusage(resource.RUSAGE_SELF).ru_maxrss /
154                      1024, 'MB')
155                  print('--------------------------------------')
156                  print('total time elapsed:', self.tot_time)
157                  print('--------------------------------------')
158                  self.test_done = True
159                  return
160              text_input.select_text(text_input.selection_from - 1,
161                  text_input.selection_to)
162              ev()
163          ev = Clock.create_trigger(pste)
164          ev()
165  
166      def start_test(self, *largs):
167          self.but.text = 'test started'
168          self.slider.max = len(self.tests)
169          ev = None
170  
171          def test(*l):
172              if self.test_done:
173                  try:
174                      but = self.tests[int(self.slider.value)]
175                      self.slider.value += 1
176                      but.state = 'down'
177                      print('=====================')
178                      print('Test:', but.text)
179                      print('=====================')
180                      but.test(but)
181                  except IndexError:
182                      for but in self.tests:
183                          but.state = 'normal'
184                      self.but.text = 'Start Test'
185                      self.slider.value = 0
186                      print('===================')
187                      print('All Tests Completed')
188                      print('===================')
189                      ev.cancel()
190  
191          ev = Clock.schedule_interval(test, 1)
192
