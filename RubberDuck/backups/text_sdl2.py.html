2    SDL2 text provider

4     
5    Based on SDL2 + SDL2_ttf

8    __all__ = ('LabelSDL2', )
9    
10   from kivy.compat import PY2
11   from kivy.core.text import LabelBase
12   try:
13       from kivy.core.text._text_sdl2 import (_SurfaceContainer, _get_extents,
14                                              _get_fontdescent, _get_fontascent)
15   except ImportError:
16       from kivy.core import handle_win_lib_import_error
17       handle_win_lib_import_error(
18           'text', 'sdl2', 'kivy.core.text._text_sdl2')
19       raise
20   
21   
22   class LabelSDL2(LabelBase):
23   
24       def _get_font_id(self):
25           return '|'.join([str(self.options[x]) for x
26               in ('font_size', 'font_name_r', 'bold',
27                   'italic', 'underline', 'strikethrough')])
28   
29       def get_extents(self, text):
30           try:
31               if PY2:
32                   text = text.encode('UTF-8')
33           except:
34               pass
35           return _get_extents(self, text)
36   
37       def get_descent(self):
38           return _get_fontdescent(self)
39   
40       def get_ascent(self):
41           return _get_fontascent(self)
42   
43       def _render_begin(self):
44           self._surface = _SurfaceContainer(self._size[0], self._size[1])
45   
46       def _render_text(self, text, x, y):
47           self._surface.render(self, text, x, y)
48   
49       def _render_end(self):
50           return self._surface.get_data()
