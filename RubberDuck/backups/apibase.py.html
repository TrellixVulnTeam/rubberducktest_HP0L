113  # # # # # ----- Type Objects and Constructors ----- # # # # #
114  # Many databases need to have the input in a particular format for binding to an operation's input parameters.
115  # For example, if an input is destined for a DATE column, then it must be bound to the database in a particular
116  # string format. Similar problems exist for "Row ID" columns or large binary items (e.g. blobs or RAW columns).
117  # This presents problems for Python since the parameters to the executeXXX() method are untyped.
118  # When the database module sees a Python string object, it doesn't know if it should be bound as a simple CHAR
119  # column, as a raw BINARY item, or as a DATE.
120  #
121  # To overcome this problem, a module must provide the constructors defined below to create objects that can
122  # hold special values. When passed to the cursor methods, the module can then detect the proper type of
123  # the input parameter and bind it accordingly.
124  
125  # A Cursor Object's description attribute returns information about each of the result columns of a query.
126  # The type_code must compare equal to one of Type Objects defined below. Type Objects may be equal to more than
127  # one type code (e.g. DATETIME could be equal to the type codes for date, time and timestamp columns;
128  # see the Implementation Hints below for details).
129  
130  # SQL NULL values are represented by the Python None singleton on input and output.

166  # -----     Time converters ----------------------------------------------
167  class TimeConverter(object):  # this is a generic time converter skeleton
168      def __init__(self):  # the details will be filled in by instances
169          self._ordinal_1899_12_31 = datetime.date(1899, 12, 31).toordinal() - 1
170          # Use cls.types to compare if an input parameter is a datetime
171          self.types = {
172              type(self.Date(2000, 1, 1)),
173              type(self.Time(12, 1, 1)),
174              type(self.Timestamp(2000, 1, 1, 12, 1, 1)),
175              datetime.datetime,
176              datetime.time,
177              datetime.date,
178          }
179  
180      def COMDate(self, obj):
181          """Returns a ComDate from a date-time"""
182          try:  # most likely a datetime
183              tt = obj.timetuple()
184  
185              try:
186                  ms = obj.microsecond
187              except:
188                  ms = 0
189              return self.ComDateFromTuple(tt, ms)
190          except:  # might be a tuple
191              try:
192                  return self.ComDateFromTuple(obj)
193              except:  # try an mxdate
194                  try:
195                      return obj.COMDate()
196                  except:
197                      raise ValueError('Cannot convert "%s" to COMdate.' % repr(obj))
198  
199      def ComDateFromTuple(self, t, microseconds=0):
200          d = datetime.date(t[0], t[1], t[2])
201          integerPart = d.toordinal() - self._ordinal_1899_12_31
202          ms = (t[3] * 3600 + t[4] * 60 + t[5]) * 1000000 + microseconds
203          fractPart = float(ms) / 86400000000.0
204          return integerPart + fractPart
205  
206      def DateObjectFromCOMDate(self, comDate):
207          "Returns an object of the wanted type from a ComDate"
208          raise NotImplementedError  # "Abstract class"
209  
210      def Date(self, year, month, day):
211          "This function constructs an object holding a date value."
212          raise NotImplementedError  # "Abstract class"
213  
214      def Time(self, hour, minute, second):
215          "This function constructs an object holding a time value."
216          raise NotImplementedError  # "Abstract class"
217  
218      def Timestamp(self, year, month, day, hour, minute, second):
219          "This function constructs an object holding a time stamp value."
220          raise NotImplementedError  # "Abstract class"
221          # all purpose date to ISO format converter
222  
223      def DateObjectToIsoFormatString(self, obj):
224          "This function should return a string in the format 'YYYY-MM-dd HH:MM:SS:ms' (ms optional)"
225          try:  # most likely, a datetime.datetime
226              s = obj.isoformat(" ")
227          except (TypeError, AttributeError):
228              if isinstance(obj, datetime.date):
229                  s = obj.isoformat() + " 00:00:00"  # return exact midnight
230              else:
231                  try:  # maybe it has a strftime method, like mx
232                      s = obj.strftime("%Y-%m-%d %H:%M:%S")
233                  except AttributeError:
234                      try:  # but may be time.struct_time
235                          s = time.strftime("%Y-%m-%d %H:%M:%S", obj)
236                      except:
237                          raise ValueError('Cannot convert "%s" to isoformat' % repr(obj))
238          return s

274  class pythonDateTimeConverter(TimeConverter):  # standard since Python 2.3
275      def __init__(self):
276          TimeConverter.__init__(self)

278      def DateObjectFromCOMDate(self, comDate):
279          if isinstance(comDate, datetime.datetime):
280              odn = comDate.toordinal()
281              tim = comDate.time()
282              new = datetime.datetime.combine(datetime.datetime.fromordinal(odn), tim)
283              return new
284              # return comDate.replace(tzinfo=None) # make non aware
285          elif isinstance(comDate, DateTime):
286              fComDate = comDate.ToOADate()  # ironPython clr Date/Time
287          else:
288              fComDate = float(comDate)  # ComDate is number of days since 1899-12-31
289          integerPart = int(fComDate)
290          floatpart = fComDate - integerPart
291          ##if floatpart == 0.0:
292          ##    return datetime.date.fromordinal(integerPart + self._ordinal_1899_12_31)
293          dte = datetime.datetime.fromordinal(
294              integerPart + self._ordinal_1899_12_31
295          ) + datetime.timedelta(milliseconds=floatpart * 86400000)
296          # millisecondsperday=86400000 # 24*60*60*1000
297          return dte

309  class pythonTimeConverter(TimeConverter):  # the old, ?nix type date and time
310      def __init__(self):  # caution: this Class gets confised by timezones and DST
311          TimeConverter.__init__(self)
312          self.types.add(time.struct_time)

314      def DateObjectFromCOMDate(self, comDate):
315          "Returns ticks since 1970"
316          if isinstance(comDate, datetime.datetime):
317              return comDate.timetuple()
318          elif isinstance(comDate, DateTime):  # ironPython clr date/time
319              fcomDate = comDate.ToOADate()
320          else:
321              fcomDate = float(comDate)
322          secondsperday = 86400  # 24*60*60
323          # ComDate is number of days since 1899-12-31, gmtime epoch is 1970-1-1 = 25569 days
324          t = time.gmtime(secondsperday * (fcomDate - 25569.0))
325          return t  # year,month,day,hour,minute,second,weekday,julianday,daylightsaving=t

327      def Date(self, year, month, day):
328          return self.Timestamp(year, month, day, 0, 0, 0)

330      def Time(self, hour, minute, second):
331          return time.gmtime((hour * 60 + minute) * 60 + second)
332  
333      def Timestamp(self, year, month, day, hour, minute, second):
334          return time.localtime(
335              time.mktime((year, month, day, hour, minute, second, 0, 0, -1))

438  def pyTypeToADOType(d):
439      tp = type(d)
440      try:
441          return typeMap[tp]
442      except KeyError:  #   The type was not defined in the pre-computed Type table
443          from . import dateconverter
444  
445          if (
446              tp in dateconverter.types
447          ):  # maybe it is one of our supported Date/Time types
448              return adc.adDate
449          #  otherwise, attempt to discern the type by probing the data object itself -- to handle duck typing
450          if isinstance(d, StringTypes):
451              return adc.adBSTR
452          if isinstance(d, numbers.Integral):
453              return adc.adBigInt
454          if isinstance(d, numbers.Real):
455              return adc.adDouble
456          raise DataError('cannot convert "%s" (type=%s) to ADO' % (repr(d), tp))

539  class MultiMap(dict):  # builds a dictionary from {(sequence,of,keys) : function}
540      """A dictionary of ado.type : function -- but you can set multiple items by passing a sequence of keys"""
541  
542      # useful for defining conversion functions for groups of similar data types.
543      def __init__(self, aDict):
544          for k, v in list(aDict.items()):
545              self[k] = v  # we must call __setitem__
546  
547      def __setitem__(self, adoType, cvtFn):
548          "set a single item, or a whole sequence of items"
549          try:  # user passed us a sequence, set them individually
550              for type in adoType:
551                  dict.__setitem__(self, type, cvtFn)
552          except TypeError:  # a single value fails attempt to iterate
553              dict.__setitem__(self, adoType, cvtFn)

579  class SQLrow(object):  # a single database row
580      # class to emulate a sequence, so that a column may be retrieved by either number or name
581      def __init__(self, rows, index):  # "rows" is an _SQLrows object, index is which row
582          self.rows = rows  # parent 'fetch' container object
583          self.index = index  # my row number within parent
584  
585      def __getattr__(self, name):  # used for row.columnName type of value access
586          try:
587              return self._getValue(self.rows.columnNames[name.lower()])
588          except KeyError:
589              raise AttributeError('Unknown column name "{}"'.format(name))
590  
591      def _getValue(self, key):  # key must be an integer
592          if (
593              self.rows.recordset_format == RS_ARRAY
594          ):  # retrieve from two-dimensional array
595              v = self.rows.ado_results[key, self.index]
596          elif self.rows.recordset_format == RS_REMOTE:
597              v = self.rows.ado_results[self.index][key]
598          else:  # pywin32 - retrieve from tuple of tuples
599              v = self.rows.ado_results[key][self.index]
600          if self.rows.converters is NotImplemented:
601              return v
602          return convert_to_python(v, self.rows.converters[key])
603  
604      def __len__(self):
605          return self.rows.numberOfColumns
606  
607      def __getitem__(self, key):  # used for row[key] type of value access
608          if isinstance(key, int):  # normal row[1] designation
609              try:
610                  return self._getValue(key)
611              except IndexError:
612                  raise
613          if isinstance(key, slice):
614              indices = key.indices(self.rows.numberOfColumns)
615              vl = [self._getValue(i) for i in range(*indices)]
616              return tuple(vl)
617          try:
618              return self._getValue(
619                  self.rows.columnNames[key.lower()]
620              )  # extension row[columnName] designation
621          except (KeyError, TypeError):
622              er, st, tr = sys.exc_info()
623              raise er(
624                  'No such key as "%s" in %s' % (repr(key), self.__repr__())
625              ).with_traceback(tr)
626  
627      def __iter__(self):
628          return iter(self.__next__())
629  
630      def __next__(self):
631          for n in range(self.rows.numberOfColumns):
632              yield self._getValue(n)
633  
634      def __repr__(self):  # create a human readable representation
635          taglist = sorted(list(self.rows.columnNames.items()), key=lambda x: x[1])
636          s = "<SQLrow={"
637          for name, i in taglist:
638              s += name + ":" + repr(self._getValue(i)) + ", "
639          return s[:-2] + "}>"
640  
641      def __str__(self):  # create a pretty human readable representation
642          return str(
643              tuple(str(self._getValue(i)) for i in range(self.rows.numberOfColumns))
644          )

655  class SQLrows(object):
656      # class to emulate a sequence for multiple rows using a container object
657      def __init__(self, ado_results, numberOfRows, cursor):
658          self.ado_results = ado_results  # raw result of SQL get
659          try:
660              self.recordset_format = cursor.recordset_format
661              self.numberOfColumns = cursor.numberOfColumns
662              self.converters = cursor.converters
663              self.columnNames = cursor.columnNames
664          except AttributeError:
665              self.recordset_format = RS_ARRAY
666              self.numberOfColumns = 0
667              self.converters = []
668              self.columnNames = {}
669          self.numberOfRows = numberOfRows
670  
671      def __len__(self):
672          return self.numberOfRows
673  
674      def __getitem__(self, item):  # used for row or row,column access
675          if not self.ado_results:
676              return []
677          if isinstance(item, slice):  # will return a list of row objects
678              indices = item.indices(self.numberOfRows)
679              return [SQLrow(self, k) for k in range(*indices)]
680          elif isinstance(item, tuple) and len(item) == 2:
681              # d = some_rowsObject[i,j] will return a datum from a two-dimension address
682              i, j = item
683              if not isinstance(j, int):
684                  try:
685                      j = self.columnNames[j.lower()]  # convert named column to numeric
686                  except KeyError:
687                      raise KeyError('adodbapi: no such column name as "%s"' % repr(j))
688              if self.recordset_format == RS_ARRAY:  # retrieve from two-dimensional array
689                  v = self.ado_results[j, i]
690              elif self.recordset_format == RS_REMOTE:
691                  v = self.ado_results[i][j]
692              else:  # pywin32 - retrieve from tuple of tuples
693                  v = self.ado_results[j][i]
694              if self.converters is NotImplemented:
695                  return v
696              return convert_to_python(v, self.converters[j])
697          else:
698              row = SQLrow(self, item)  # new row descriptor
699              return row

701      def __iter__(self):
702          return iter(self.__next__())
703  
704      def __next__(self):
705          for n in range(self.numberOfRows):
706              row = SQLrow(self, n)
707              yield row

713  def changeNamedToQmark(
714      op,
715  ):  # convert from 'named' paramstyle to ADO required '?'mark parameters
716      outOp = ""
717      outparms = []
718      chunks = op.split(
719          "'"
720      )  # quote all literals -- odd numbered list results are literals.
721      inQuotes = False
722      for chunk in chunks:
723          if inQuotes:  # this is inside a quote
724              if chunk == "":  # double apostrophe to quote one apostrophe
725                  outOp = outOp[:-1]  # so take one away
726              else:
727                  outOp += "'" + chunk + "'"  # else pass the quoted string as is.
728          else:  # is SQL code -- look for a :namedParameter
729              while chunk:  # some SQL string remains
730                  sp = chunk.split(":", 1)
731                  outOp += sp[0]  # concat the part up to the :
732                  s = ""
733                  try:
734                      chunk = sp[1]
735                  except IndexError:
736                      chunk = None
737                  if chunk:  # there was a parameter - parse it out
738                      i = 0
739                      c = chunk[0]
740                      while c.isalnum() or c == "_":
741                          i += 1
742                          try:
743                              c = chunk[i]
744                          except IndexError:
745                              break
746                      s = chunk[:i]
747                      chunk = chunk[i:]
748                  if s:
749                      outparms.append(s)  # list the parameters in order
750                      outOp += "?"  # put in the Qmark
751          inQuotes = not inQuotes
752      return outOp, outparms

755  def changeFormatToQmark(
756      op,
757  ):  # convert from 'format' paramstyle to ADO required '?'mark parameters
758      outOp = ""
759      outparams = []
760      chunks = op.split(
761          "'"
762      )  # quote all literals -- odd numbered list results are literals.
763      inQuotes = False
764      for chunk in chunks:
765          if inQuotes:
766              if (
767                  outOp != "" and chunk == ""
768              ):  # he used a double apostrophe to quote one apostrophe
769                  outOp = outOp[:-1]  # so take one away
770              else:
771                  outOp += "'" + chunk + "'"  # else pass the quoted string as is.
772          else:  # is SQL code -- look for a %s parameter
773              if "%(" in chunk:  # ugh! pyformat!
774                  while chunk:  # some SQL string remains
775                      sp = chunk.split("%(", 1)
776                      outOp += sp[0]  # concat the part up to the %
777                      if len(sp) > 1:
778                          try:
779                              s, chunk = sp[1].split(")s", 1)  # find the ')s'
780                          except ValueError:
781                              raise ProgrammingError(
782                                  'Pyformat SQL has incorrect format near "%s"' % chunk
783                              )
784                          outparams.append(s)
785                          outOp += "?"  # put in the Qmark
786                      else:
787                          chunk = None
788              else:  # proper '%s' format
789                  sp = chunk.split("%s")  # make each %s
790                  outOp += "?".join(sp)  # into ?
791          inQuotes = not inQuotes  # every other chunk is a quoted string
792      return outOp, outparams
