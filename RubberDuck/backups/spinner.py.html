1    ''' 
2    Spinner 
3    ======= 
4     
5    .. versionadded:: 1.4.0 
6     
7    .. image:: images/spinner.jpg 
8        :align: right 
9     
10   Spinner is a widget that provides a quick way to select one value from a set. 
11   In the default state, a spinner shows its currently selected value. 
12   Touching the spinner displays a dropdown menu with all the other available 
13   values from which the user can select a new one. 
14    
15   Example:: 
16    
17       from kivy.base import runTouchApp 
18       from kivy.uix.spinner import Spinner 
19    
20       spinner = Spinner( 
21           # default value shown 
22           text='Home', 
23           # available values 
24           values=('Home', 'Work', 'Other', 'Custom'), 
25           # just for positioning in our example 
26           size_hint=(None, None), 
27           size=(100, 44), 
28           pos_hint={'center_x': .5, 'center_y': .5}) 
29    
30       def show_selected_value(spinner, text): 
31           print('The spinner', spinner, 'has text', text) 
32    
33       spinner.bind(text=show_selected_value) 
34    
35       runTouchApp(spinner) 
36    
37    
38   Kv Example:: 
39    
40       FloatLayout: 
41           Spinner: 
42               size_hint: None, None 
43               size: 100, 44 
44               pos_hint: {'center': (.5, .5)} 
45               text: 'Home' 
46               values: 'Home', 'Work', 'Other', 'Custom' 
47               on_text: 
48                   print("The spinner {} has text {}".format(self, self.text)) 
49   '''
50   
51   __all__ = ('Spinner', 'SpinnerOption')
52   
53   from kivy.compat import string_types
54   from kivy.factory import Factory
55   from kivy.properties import ListProperty, ObjectProperty, BooleanProperty
56   from kivy.uix.button import Button
57   from kivy.uix.dropdown import DropDown
58   
59   
60   class SpinnerOption(Button):
61       '''Special button used in the :class:`Spinner` dropdown list. By default, 
62       this is just a :class:`~kivy.uix.button.Button` with a size_hint_y of None 
63       and a height of :meth:`48dp <kivy.metrics.dp>`. 
64       '''
65       pass
66   
67   
68   class Spinner(Button):
69       '''Spinner class, see module documentation for more information. 
70       '''
71   
72       values = ListProperty()
73       '''Values that can be selected by the user. It must be a list of strings. 
74    
75       :attr:`values` is a :class:`~kivy.properties.ListProperty` and defaults to 
76       []. 
77       '''
78   
79       text_autoupdate = BooleanProperty(False)
80       '''Indicates if the spinner's :attr:`text` should be automatically 
81       updated with the first value of the :attr:`values` property. 
82       Setting it to True will cause the spinner to update its :attr:`text` 
83       property every time attr:`values` are changed. 
84    
85       .. versionadded:: 1.10.0 
86    
87       :attr:`text_autoupdate` is a :class:`~kivy.properties.BooleanProperty` and 
88       defaults to False. 
89       '''
90   
91       option_cls = ObjectProperty(SpinnerOption)
92       '''Class used to display the options within the dropdown list displayed 
93       under the Spinner. The `text` property of the class will be used to 
94       represent the value. 
95    
96       The option class requires: 
97    
98       - a `text` property, used to display the value. 
99       - an `on_release` event, used to trigger the option when pressed/touched. 
100      - a :attr:`~kivy.uix.widget.Widget.size_hint_y` of None. 
101      - the :attr:`~kivy.uix.widget.Widget.height` to be set. 
102   
103      :attr:`option_cls` is an :class:`~kivy.properties.ObjectProperty` and 
104      defaults to :class:`SpinnerOption`. 
105   
106      .. versionchanged:: 1.8.0 
107          If you set a string, the :class:`~kivy.factory.Factory` will be used to 
108          resolve the class. 
109   
110      '''
111  
112      dropdown_cls = ObjectProperty(DropDown)
113      '''Class used to display the dropdown list when the Spinner is pressed. 
114   
115      :attr:`dropdown_cls` is an :class:`~kivy.properties.ObjectProperty` and 
116      defaults to :class:`~kivy.uix.dropdown.DropDown`. 
117   
118      .. versionchanged:: 1.8.0 
119          If set to a string, the :class:`~kivy.factory.Factory` will be used to 
120          resolve the class name. 
121   
122      '''
123  
124      is_open = BooleanProperty(False)
125      '''By default, the spinner is not open. Set to True to open it. 
126   
127      :attr:`is_open` is a :class:`~kivy.properties.BooleanProperty` and 
128      defaults to False. 
129   
130      .. versionadded:: 1.4.0 
131      '''
132  
133      sync_height = BooleanProperty(False)
134      '''Each element in a dropdown list uses a default/user-supplied height. 
135      Set to True to propagate the Spinner's height value to each dropdown 
136      list element. 
137   
138      .. versionadded:: 1.10.0 
139   
140      :attr:`sync_height` is a :class:`~kivy.properties.BooleanProperty` and 
141      defaults to False. 
142      '''
143  
144      def __init__(self, **kwargs):
145          self._dropdown = None
146          super(Spinner, self).__init__(**kwargs)
147          fbind = self.fbind
148          build_dropdown = self._build_dropdown
149          fbind('on_release', self._toggle_dropdown)
150          fbind('dropdown_cls', build_dropdown)
151          fbind('option_cls', build_dropdown)
152          fbind('values', self._update_dropdown)
153          fbind('size', self._update_dropdown_size)
154          fbind('text_autoupdate', self._update_dropdown)
155          build_dropdown()
156  
157      def _build_dropdown(self, *largs):
158          if self._dropdown:
159              self._dropdown.unbind(on_select=self._on_dropdown_select)
160              self._dropdown.unbind(on_dismiss=self._close_dropdown)
161              self._dropdown.dismiss()
162              self._dropdown = None
163          cls = self.dropdown_cls
164          if isinstance(cls, string_types):
165              cls = Factory.get(cls)
166          self._dropdown = cls()
167          self._dropdown.bind(on_select=self._on_dropdown_select)
168          self._dropdown.bind(on_dismiss=self._close_dropdown)
169          self._update_dropdown()
170  
171      def _update_dropdown_size(self, *largs):
172          if not self.sync_height:
173              return
174          dp = self._dropdown
175          if not dp:
176              return
177  
178          container = dp.container
179          if not container:
180              return
181          h = self.height
182          for item in container.children[:]:
183              item.height = h
184  
185      def _update_dropdown(self, *largs):
186          dp = self._dropdown
187          cls = self.option_cls
188          values = self.values
189          text_autoupdate = self.text_autoupdate
190          if isinstance(cls, string_types):
191              cls = Factory.get(cls)
192          dp.clear_widgets()
193          for value in values:
194              item = cls(text=value)
195              item.height = self.height if self.sync_height else item.height
196              item.bind(on_release=lambda option: dp.select(option.text))
197              dp.add_widget(item)
198          if text_autoupdate:
199              if values:
200                  if not self.text or self.text not in values:
201                      self.text = values[0]
202              else:
203                  self.text = ''
204  
205      def _toggle_dropdown(self, *largs):
206          if self.values:
207              self.is_open = not self.is_open
208  
209      def _close_dropdown(self, *largs):
210          self.is_open = False
211  
212      def _on_dropdown_select(self, instance, data, *largs):
213          self.text = data
214          self.is_open = False
215  
216      def on_is_open(self, instance, value):
217          if value:
218              self._dropdown.open(self)
219          else:
220              if self._dropdown.attach_to:
221                  self._dropdown.dismiss()
