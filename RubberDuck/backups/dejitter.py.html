dejitter.py 
1    ''' 
2    Dejitter 
3    ======== 
4     
5    Prevent blob jittering. 
6     
7    A problem that is often faced (esp. in optical MT setups) is that of 
8    jitterish BLOBs caused by bad camera characteristics. With this module 
9    you can get rid of that jitter. You just define a threshold 
10   `jitter_distance` in your config, and all touch movements that move 
11   the touch by less than the jitter distance are considered 'bad' 
12   movements caused by jitter and will be discarded. 
13   '''
14   
15   __all__ = ('InputPostprocDejitter', )
16   
17   from kivy.config import Config
18   
19   
20   class InputPostprocDejitter(object):
21       ''' 
22       Get rid of jitterish BLOBs. 
23       Example:: 
24    
25           [postproc] 
26           jitter_distance = 0.004 
27           jitter_ignore_devices = mouse,mactouch 
28    
29       :Configuration: 
30           `jitter_distance`: float 
31               A float in range 0-1. 
32           `jitter_ignore_devices`: string 
33               A comma-separated list of device identifiers that 
34               should not be processed by dejitter (because they're 
35               very precise already). 
36       '''
37   
38       def __init__(self):
39           self.jitterdist = Config.getfloat('postproc', 'jitter_distance')
40           ignore_devices = Config.get('postproc', 'jitter_ignore_devices')
41           self.ignore_devices = ignore_devices.split(',')
42           self.last_touches = {}
43   
44       def taxicab_distance(self, p, q):
45           # Get the taxicab/manhattan/citiblock distance for efficiency reasons
46           return abs(p[0] - q[0]) + abs(p[1] - q[1])
47   
48       def process(self, events):
49           if not self.jitterdist:
50               return events
51           processed = []
52           for etype, touch in events:
53               if not touch.is_touch:
54                   continue
55               if touch.device in self.ignore_devices:
56                   processed.append((etype, touch))
57                   continue
58               if etype == 'begin':
59                   self.last_touches[touch.id] = touch.spos
60               if etype == 'end':
61                   if touch.id in self.last_touches:
62                       del self.last_touches[touch.id]
63               if etype != 'update':
64                   processed.append((etype, touch))
65                   continue
66               # Check whether the touch moved more than the jitter distance
67               last_spos = self.last_touches[touch.id]
68               dist = self.taxicab_distance(last_spos, touch.spos)
69               if dist > self.jitterdist:
70                   # Only if the touch has moved more than the jitter dist we take
71                   # it into account and dispatch it. Otherwise suppress it.
72                   self.last_touches[touch.id] = touch.spos
73                   processed.append((etype, touch))
74           return processed
