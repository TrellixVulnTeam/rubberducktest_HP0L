43       def heredoc_callback(self, match, ctx):
44           # okay, this is the hardest part of parsing Crystal...
45           # match: 1 = <<-?, 2 = quote? 3 = name 4 = quote? 5 = rest of line
46   
47           start = match.start(1)
48           yield start, Operator, match.group(1)        # <<-?
49           yield match.start(2), String.Heredoc, match.group(2)    # quote ", ', `
50           yield match.start(3), String.Delimiter, match.group(3)  # heredoc name
51           yield match.start(4), String.Heredoc, match.group(4)    # quote again
52   
53           heredocstack = ctx.__dict__.setdefault('heredocstack', [])
54           outermost = not bool(heredocstack)
55           heredocstack.append((match.group(1) == '<<-', match.group(3)))
56   
57           ctx.pos = match.start(5)
58           ctx.end = match.end(5)
59           # this may find other heredocs, so limit the recursion depth
60           if len(heredocstack) < 100:
61               yield from self.get_tokens_unprocessed(context=ctx)
62           else:
63               yield ctx.pos, String.Heredoc, match.group(5)
64           ctx.pos = match.end()
65   
66           if outermost:
67               # this is the outer heredoc again, now we can process them all
68               for tolerant, hdname in heredocstack:
69                   lines = []
70                   for match in line_re.finditer(ctx.text, ctx.pos):
71                       if tolerant:
72                           check = match.group().strip()
73                       else:
74                           check = match.group().rstrip()
75                       if check == hdname:
76                           for amatch in lines:
77                               yield amatch.start(), String.Heredoc, amatch.group()
78                           yield match.start(), String.Delimiter, match.group()
79                           ctx.pos = match.end()
80                           break
81                       else:
82                           lines.append(match)
83                   else:
84                       # end of heredoc not found -- error!
85                       for amatch in lines:
86                           yield amatch.start(), Error, amatch.group()
87               ctx.end = len(ctx.text)
88               del heredocstack[:]

90       def gen_crystalstrings_rules():
91           states = {}
92           states['strings'] = [
93               (r'\:\w+[!?]?', String.Symbol),
94               (words(CRYSTAL_OPERATORS, prefix=r'\:'), String.Symbol),
95               (r":'(\\\\|\\[^\\]|[^'\\])*'", String.Symbol),
96               # This allows arbitrary text after '\ for simplicity
97               (r"'(\\\\|\\'|[^']|\\[^'\\]+)'", String.Char),
98               (r':"', String.Symbol, 'simple-sym'),
99               # Crystal doesn't have "symbol:"s but this simplifies function args
100              (r'([a-zA-Z_]\w*)(:)(?!:)', bygroups(String.Symbol, Punctuation)),
101              (r'"', String.Double, 'simple-string'),
102              (r'(?<!\.)`', String.Backtick, 'simple-backtick'),
103          ]
104  
105          # double-quoted string and symbol
106          for name, ttype, end in ('string', String.Double, '"'), \
107                                  ('sym', String.Symbol, '"'), \
108                                  ('backtick', String.Backtick, '`'):
109              states['simple-'+name] = [
110                  include('string-escaped' if name == 'sym' else 'string-intp-escaped'),
111                  (r'[^\\%s#]+' % end, ttype),
112                  (r'[\\#]', ttype),
113                  (end, ttype, '#pop'),
114              ]
115  
116          # https://crystal-lang.org/docs/syntax_and_semantics/literals/string.html#percent-string-literals
117          for lbrace, rbrace, bracecc, name in \
118                  ('\\{', '\\}', '{}', 'cb'), \
119                  ('\\[', '\\]', '\\[\\]', 'sb'), \
120                  ('\\(', '\\)', '()', 'pa'), \
121                  ('<', '>', '<>', 'ab'), \
122                  ('\\|', '\\|', '\\|', 'pi'):
123              states[name+'-intp-string'] = [
124                  (r'\\' + lbrace, String.Other),
125              ] + (lbrace != rbrace) * [
126                  (lbrace, String.Other, '#push'),
127              ] + [
128                  (rbrace, String.Other, '#pop'),
129                  include('string-intp-escaped'),
130                  (r'[\\#' + bracecc + ']', String.Other),
131                  (r'[^\\#' + bracecc + ']+', String.Other),
132              ]
133              states['strings'].append((r'%Q?' + lbrace, String.Other,
134                                        name+'-intp-string'))
135              states[name+'-string'] = [
136                  (r'\\[\\' + bracecc + ']', String.Other),
137              ] + (lbrace != rbrace) * [
138                  (lbrace, String.Other, '#push'),
139              ] + [
140                  (rbrace, String.Other, '#pop'),
141                  (r'[\\#' + bracecc + ']', String.Other),
142                  (r'[^\\#' + bracecc + ']+', String.Other),
143              ]
144              # https://crystal-lang.org/docs/syntax_and_semantics/literals/array.html#percent-array-literals
145              states['strings'].append((r'%[qwi]' + lbrace, String.Other,
146                                        name+'-string'))
147              states[name+'-regex'] = [
148                  (r'\\[\\' + bracecc + ']', String.Regex),
149              ] + (lbrace != rbrace) * [
150                  (lbrace, String.Regex, '#push'),
151              ] + [
152                  (rbrace + '[imsx]*', String.Regex, '#pop'),
153                  include('string-intp'),
154                  (r'[\\#' + bracecc + ']', String.Regex),
155                  (r'[^\\#' + bracecc + ']+', String.Regex),
156              ]
157              states['strings'].append((r'%r' + lbrace, String.Regex,
158                                        name+'-regex'))
159  
160          return states

162      tokens = {

165              # keywords
166              (words('''

177              (words(['true', 'false', 'nil'], suffix=r'\b'), Keyword.Constant),
178              # start of function, class and module names

184              (r'(annotation|class|struct|union|type|alias|enum)(\s+)((?:[a-zA-Z_]\w*::)*)',
