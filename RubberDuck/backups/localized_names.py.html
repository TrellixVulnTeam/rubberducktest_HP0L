1    # A Python port of the MS knowledge base article Q157234
2    # "How to deal with localized and renamed user and group names"
3    # http://support.microsoft.com/default.aspx?kbid=157234
4    
5    import sys
6    from win32net import NetUserModalsGet
7    from win32security import LookupAccountSid
8    import pywintypes
9    from ntsecuritycon import *
10   
11   
12   def LookupAliasFromRid(TargetComputer, Rid):
13       # Sid is the same regardless of machine, since the well-known
14       # BUILTIN domain is referenced.
15       sid = pywintypes.SID()
16       sid.Initialize(SECURITY_NT_AUTHORITY, 2)
17   
18       for i, r in enumerate((SECURITY_BUILTIN_DOMAIN_RID, Rid)):
19           sid.SetSubAuthority(i, r)
20   
21       name, domain, typ = LookupAccountSid(TargetComputer, sid)
22       return name
23   
24   
25   def LookupUserGroupFromRid(TargetComputer, Rid):
26       # get the account domain Sid on the target machine
27       # note: if you were looking up multiple sids based on the same
28       # account domain, only need to call this once.
29       umi2 = NetUserModalsGet(TargetComputer, 2)
30       domain_sid = umi2["domain_id"]
31   
32       SubAuthorityCount = domain_sid.GetSubAuthorityCount()
33   
34       # create and init new sid with acct domain Sid + acct Rid
35       sid = pywintypes.SID()
36       sid.Initialize(domain_sid.GetSidIdentifierAuthority(), SubAuthorityCount + 1)
37   
38       # copy existing subauthorities from account domain Sid into
39       # new Sid
40       for i in range(SubAuthorityCount):
41           sid.SetSubAuthority(i, domain_sid.GetSubAuthority(i))
42   
43       # append Rid to new Sid
44       sid.SetSubAuthority(SubAuthorityCount, Rid)
45   
46       name, domain, typ = LookupAccountSid(TargetComputer, sid)
47       return name

50   def main():
51       if len(sys.argv) == 2:
52           targetComputer = sys.argv[1]
53       else:
54           targetComputer = None
55   
56       name = LookupUserGroupFromRid(targetComputer, DOMAIN_USER_RID_ADMIN)
57       print("'Administrator' user name = %s" % (name,))
58   
59       name = LookupAliasFromRid(targetComputer, DOMAIN_ALIAS_RID_ADMINS)
60       print("'Administrators' local group/alias name = %s" % (name,))
61   
62   
63   if __name__ == "__main__":
64       main()
