54   # Given a propget function's arg desc, return the default parameters for all
55   # params bar the first.  Eg, then Python does a:
56   # object.Property = "foo"
57   # Python can only pass the "foo" value.  If the property has
58   # multiple args, and the rest have default values, this allows
59   # Python to correctly pass those defaults.
60   def MakeDefaultArgsForPropertyPut(argsDesc):
61       ret = []
62       for desc in argsDesc[1:]:
63           default = build.MakeDefaultArgRepr(desc)
64           if default is None:
65               break
66           ret.append(default)
67       return tuple(ret)

70   def MakeMapLineEntry(dispid, wFlags, retType, argTypes, user, resultCLSID):
71       # Strip the default value
72       argTypes = tuple([what[:2] for what in argTypes])
73       return '(%s, %d, %s, %s, "%s", %s)' % (
74           dispid,
75           wFlags,
76           retType[:2],
77           argTypes,
78           user,
79           resultCLSID,

107  class WritableItem:
108      # __cmp__ used for sorting in py2x...
109      def __cmp__(self, other):
110          "Compare for sorting"
111          ret = cmp(self.order, other.order)
112          if ret == 0 and self.doc:
113              ret = cmp(self.doc[0], other.doc[0])
114          return ret
115  
116      # ... but not used in py3k - __lt__ minimum needed there
117      def __lt__(self, other):  # py3k variant
118          if self.order == other.order:
119              return self.doc < other.doc
120          return self.order < other.order
121  
122      def __repr__(self):
123          return "OleItem: doc=%s, order=%d" % (repr(self.doc), self.order)

153  class AliasItem(build.OleItem, WritableItem):
154      order = 2
155      typename = "ALIAS"
156  
157      def __init__(self, typeinfo, attr, doc=None, bForUser=1):
158          build.OleItem.__init__(self, doc)
159  
160          ai = attr[14]
161          self.attr = attr
162          if type(ai) == type(()) and type(ai[1]) == type(
163              0
164          ):  # XXX - This is a hack - why tuples?  Need to resolve?
165              href = ai[1]
166              alinfo = typeinfo.GetRefTypeInfo(href)
167              self.aliasDoc = alinfo.GetDocumentation(-1)
168              self.aliasAttr = alinfo.GetTypeAttr()
169          else:
170              self.aliasDoc = None
171              self.aliasAttr = None
172  
173      def WriteAliasItem(self, aliasDict, stream):
174          # we could have been written as part of an alias dependency
175          if self.bWritten:
176              return
177  
178          if self.aliasDoc:
179              depName = self.aliasDoc[0]
180              if depName in aliasDict:
181                  aliasDict[depName].WriteAliasItem(aliasDict, stream)
182              print(self.doc[0] + " = " + depName, file=stream)
183          else:
184              ai = self.attr[14]
185              if type(ai) == type(0):
186                  try:
187                      typeStr = mapVTToTypeString[ai]
188                      print("# %s=%s" % (self.doc[0], typeStr), file=stream)
189                  except KeyError:
190                      print(
191                          self.doc[0] + " = None # Can't convert alias info " + str(ai),
192                          file=stream,
193                      )
194          print(file=stream)
195          self.bWritten = 1

198  class EnumerationItem(build.OleItem, WritableItem):
199      order = 1
200      typename = "ENUMERATION"
201  
202      def __init__(self, typeinfo, attr, doc=None, bForUser=1):
203          build.OleItem.__init__(self, doc)
204  
205          self.clsid = attr[0]
206          self.mapVars = {}
207          typeFlags = attr[11]
208          self.hidden = (
209              typeFlags & pythoncom.TYPEFLAG_FHIDDEN
210              or typeFlags & pythoncom.TYPEFLAG_FRESTRICTED
211          )
212  
213          for j in range(attr[7]):
214              vdesc = typeinfo.GetVarDesc(j)
215              name = typeinfo.GetNames(vdesc[0])[0]
216              self.mapVars[name] = build.MapEntry(vdesc)

218      ##  def WriteEnumerationHeaders(self, aliasItems, stream):

221      ##    WriteAliasesForItem(self, aliasItems)

223      def WriteEnumerationItems(self, stream):
224          num = 0
225          enumName = self.doc[0]
226          # Write in name alpha order
227          names = list(self.mapVars.keys())
228          names.sort()
229          for name in names:
230              entry = self.mapVars[name]
231              vdesc = entry.desc
232              if vdesc[4] == pythoncom.VAR_CONST:
233                  val = vdesc[1]
234  
235                  use = repr(val)
236                  # Make sure the repr of the value is valid python syntax
237                  # still could cause an error on import if it contains a module or type name
238                  # not available in the global namespace
239                  try:
240                      compile(use, "<makepy>", "eval")
241                  except SyntaxError:
242                      # At least add the repr as a string, so it can be investigated further
243                      # Sanitize it, in case the repr contains its own quotes.  (??? line breaks too ???)
244                      use = use.replace('"', "'")
245                      use = (
246                          '"'
247                          + use
248                          + '"'
249                          + " # This VARIANT type cannot be converted automatically"
250                      )
251                  print(
252                      "\t%-30s=%-10s # from enum %s"
253                      % (build.MakePublicAttributeName(name, True), use, enumName),
254                      file=stream,
255                  )
256                  num += 1
257          return num

459      def WriteClassBody(self, generator):
460          stream = generator.file
461          # Write in alpha order.
462          names = list(self.mapFuncs.keys())
463          names.sort()
464          specialItems = {
465              "count": None,
466              "item": None,
467              "value": None,
468              "_newenum": None,
469          }  # If found, will end up with (entry, invoke_tupe)
470          itemCount = None
471          for name in names:
472              entry = self.mapFuncs[name]
473              assert entry.desc.desckind == pythoncom.DESCKIND_FUNCDESC
474              # skip [restricted] methods, unless it is the
475              # enumerator (which, being part of the "system",
476              # we know about and can use)
477              dispid = entry.desc.memid
478              if (
479                  entry.desc.wFuncFlags & pythoncom.FUNCFLAG_FRESTRICTED
480                  and dispid != pythoncom.DISPID_NEWENUM
481              ):
482                  continue
483              # If not accessible via IDispatch, then we can't use it here.
484              if entry.desc.funckind != pythoncom.FUNC_DISPATCH:
485                  continue
486              if dispid == pythoncom.DISPID_VALUE:
487                  lkey = "value"
488              elif dispid == pythoncom.DISPID_NEWENUM:
489                  specialItems["_newenum"] = (entry, entry.desc.invkind, None)
490                  continue  # Dont build this one now!
491              else:
492                  lkey = name.lower()
493              if (
494                  lkey in specialItems and specialItems[lkey] is None
495              ):  # remember if a special one.
496                  specialItems[lkey] = (entry, entry.desc.invkind, None)
497              if generator.bBuildHidden or not entry.hidden:
498                  if entry.GetResultName():
499                      print("\t# Result is of type " + entry.GetResultName(), file=stream)
500                  if entry.wasProperty:
501                      print(
502                          "\t# The method %s is actually a property, but must be used as a method to correctly pass the arguments"
503                          % name,
504                          file=stream,
505                      )
506                  ret = self.MakeFuncMethod(entry, build.MakePublicAttributeName(name))
507                  for line in ret:
508                      print(line, file=stream)
509          print("\t_prop_map_get_ = {", file=stream)
510          names = list(self.propMap.keys())
511          names.sort()
512          for key in names:
513              entry = self.propMap[key]
514              if generator.bBuildHidden or not entry.hidden:
515                  resultName = entry.GetResultName()
516                  if resultName:
517                      print(
518                          "\t\t# Property '%s' is an object of type '%s'"
519                          % (key, resultName),
520                          file=stream,
521                      )
522                  lkey = key.lower()
523                  details = entry.desc
524                  resultDesc = details[2]
525                  argDesc = ()
526                  mapEntry = MakeMapLineEntry(
527                      details.memid,
528                      pythoncom.DISPATCH_PROPERTYGET,
529                      resultDesc,
530                      argDesc,
531                      key,
532                      entry.GetResultCLSIDStr(),
533                  )
534  
535                  if details.memid == pythoncom.DISPID_VALUE:
536                      lkey = "value"
537                  elif details.memid == pythoncom.DISPID_NEWENUM:
538                      lkey = "_newenum"
539                  else:
540                      lkey = key.lower()
541                  if (
542                      lkey in specialItems and specialItems[lkey] is None
543                  ):  # remember if a special one.
544                      specialItems[lkey] = (
545                          entry,
546                          pythoncom.DISPATCH_PROPERTYGET,
547                          mapEntry,
548                      )
549                      # All special methods, except _newenum, are written
550                      # "normally".  This is a mess!
551                      if details.memid == pythoncom.DISPID_NEWENUM:
552                          continue
553  
554                  print(
555                      '\t\t"%s": %s,' % (build.MakePublicAttributeName(key), mapEntry),
556                      file=stream,
557                  )
558          names = list(self.propMapGet.keys())
559          names.sort()
560          for key in names:
561              entry = self.propMapGet[key]
562              if generator.bBuildHidden or not entry.hidden:
563                  if entry.GetResultName():
564                      print(
565                          "\t\t# Method '%s' returns object of type '%s'"
566                          % (key, entry.GetResultName()),
567                          file=stream,
568                      )
569                  details = entry.desc
570                  assert details.desckind == pythoncom.DESCKIND_FUNCDESC
571                  lkey = key.lower()
572                  argDesc = details[2]
573                  resultDesc = details[8]
574                  mapEntry = MakeMapLineEntry(
575                      details[0],
576                      pythoncom.DISPATCH_PROPERTYGET,
577                      resultDesc,
578                      argDesc,
579                      key,
580                      entry.GetResultCLSIDStr(),
581                  )
582                  if details.memid == pythoncom.DISPID_VALUE:
583                      lkey = "value"
584                  elif details.memid == pythoncom.DISPID_NEWENUM:
585                      lkey = "_newenum"
586                  else:
587                      lkey = key.lower()
588                  if (
589                      lkey in specialItems and specialItems[lkey] is None
590                  ):  # remember if a special one.
591                      specialItems[lkey] = (
592                          entry,
593                          pythoncom.DISPATCH_PROPERTYGET,
594                          mapEntry,
595                      )
596                      # All special methods, except _newenum, are written
597                      # "normally".  This is a mess!
598                      if details.memid == pythoncom.DISPID_NEWENUM:
599                          continue
600                  print(
601                      '\t\t"%s": %s,' % (build.MakePublicAttributeName(key), mapEntry),
602                      file=stream,
603                  )
604  
605          print("\t}", file=stream)
606  
607          print("\t_prop_map_put_ = {", file=stream)
608          # These are "Invoke" args
609          names = list(self.propMap.keys())
610          names.sort()
611          for key in names:
612              entry = self.propMap[key]
613              if generator.bBuildHidden or not entry.hidden:
614                  lkey = key.lower()
615                  details = entry.desc
616                  # If default arg is None, write an empty tuple
617                  defArgDesc = build.MakeDefaultArgRepr(details[2])
618                  if defArgDesc is None:
619                      defArgDesc = ""
620                  else:
621                      defArgDesc = defArgDesc + ","
622                  print(
623                      '\t\t"%s" : ((%s, LCID, %d, 0),(%s)),'
624                      % (
625                          build.MakePublicAttributeName(key),
626                          details[0],
627                          pythoncom.DISPATCH_PROPERTYPUT,
628                          defArgDesc,
629                      ),
630                      file=stream,
631                  )
632  
633          names = list(self.propMapPut.keys())
634          names.sort()
635          for key in names:
636              entry = self.propMapPut[key]
637              if generator.bBuildHidden or not entry.hidden:
638                  details = entry.desc

939      def _Build_CoClassChildren(self, coclass, coclass_info, oleItems, vtableItems):
940          sources = {}
941          interfaces = {}
942          for info, info_type, refType, doc, refAttr, flags in coclass_info:
943              #          sys.stderr.write("Attr typeflags for coclass referenced object %s=%d (%d), typekind=%d\n" % (name, refAttr.wTypeFlags, refAttr.wTypeFlags & pythoncom.TYPEFLAG_FDUAL,refAttr.typekind))

945                  refAttr.typekind == pythoncom.TKIND_INTERFACE

972      def _Build_Interface(self, type_info_tuple):
973          info, infotype, doc, attr = type_info_tuple
974          oleItem = vtableItem = None
975          if infotype == pythoncom.TKIND_DISPATCH or (
976              infotype == pythoncom.TKIND_INTERFACE
977              and attr[11] & pythoncom.TYPEFLAG_FDISPATCHABLE
978          ):
979              oleItem = DispatchItem(info, attr, doc)
980              # If this DISPATCH interface dual, then build that too.
981              if attr.wTypeFlags & pythoncom.TYPEFLAG_FDUAL:
982                  # Get the vtable interface
983                  refhtype = info.GetRefTypeOfImplType(-1)
984                  info = info.GetRefTypeInfo(refhtype)
985                  attr = info.GetTypeAttr()
986                  infotype = pythoncom.TKIND_INTERFACE
987              else:
988                  infotype = None
989          assert infotype in [
990              None,
991              pythoncom.TKIND_INTERFACE,
992          ], "Must be a real interface at this point"
993          if infotype == pythoncom.TKIND_INTERFACE:
994              vtableItem = VTableItem(info, attr, doc)
995          return oleItem, vtableItem

997      def BuildOleItemsFromType(self):
998          assert (
999              self.bBuildHidden
1000         ), "This code doesnt look at the hidden flag - I thought everyone set it true!?!?!"
1001         oleItems = {}
1002         enumItems = {}
1003         recordItems = {}
1004         vtableItems = {}
1005 
1006         for type_info_tuple in self.CollectOleItemInfosFromType():
1007             info, infotype, doc, attr = type_info_tuple
1008             clsid = attr[0]
1009             if infotype == pythoncom.TKIND_ENUM or infotype == pythoncom.TKIND_MODULE:
1010                 newItem = EnumerationItem(info, attr, doc)
1011                 enumItems[newItem.doc[0]] = newItem
1012             # We never hide interfaces (MSAccess, for example, nominates interfaces as
1013             # hidden, assuming that you only ever use them via the CoClass)
1014             elif infotype in [pythoncom.TKIND_DISPATCH, pythoncom.TKIND_INTERFACE]:
1015                 if clsid not in oleItems:
1016                     oleItem, vtableItem = self._Build_Interface(type_info_tuple)
1017                     oleItems[clsid] = oleItem  # Even "None" goes in here.
1018                     if vtableItem is not None:
1019                         vtableItems[clsid] = vtableItem
1020             elif (
1021                 infotype == pythoncom.TKIND_RECORD or infotype == pythoncom.TKIND_UNION
1022             ):
1023                 newItem = RecordItem(info, attr, doc)
1024                 recordItems[newItem.clsid] = newItem
1025             elif infotype == pythoncom.TKIND_ALIAS:
1026                 # We dont care about alias' - handled intrinsicly.
1027                 continue
1028             elif infotype == pythoncom.TKIND_COCLASS:
1029                 newItem, child_infos = self._Build_CoClass(type_info_tuple)
1030                 self._Build_CoClassChildren(newItem, child_infos, oleItems, vtableItems)
1031                 oleItems[newItem.clsid] = newItem
1032             else:
1033                 self.progress.LogWarning("Unknown TKIND found: %d" % infotype)
1034

1037     def open_writer(self, filename, encoding="mbcs"):
1038         # A place to put code to open a file with the appropriate encoding.
1039         # Does *not* set self.file - just opens and returns a file.
1040         # Actually returns a handle to a temp file - finish_writer then deletes
1041         # the filename asked for and puts everything back in place.  This
1042         # is so errors don't leave a 1/2 generated file around causing bizarre
1043         # errors later, and so that multiple processes writing the same file
1044         # don't step on each others' toes.
1045         # Could be a classmethod one day...
1046         temp_filename = self.get_temp_filename(filename)
1047         return open(temp_filename, "wt", encoding=encoding)

1049     def finish_writer(self, filename, f, worked):
1050         f.close()
1051         try:
1052             os.unlink(filename)
1053         except os.error:
1054             pass
1055         temp_filename = self.get_temp_filename(filename)
1056         if worked:
1057             try:
1058                 os.rename(temp_filename, filename)
1059             except os.error:
1060                 # If we are really unlucky, another process may have written the
1061                 # file in between our calls to os.unlink and os.rename. So try
1062                 # again, but only once.
1063                 # There are still some race conditions, but they seem difficult to
1064                 # fix, and they probably occur much less frequently:
1065                 # * The os.rename failure could occur more than once if more than
1066                 #   two processes are involved.
1067                 # * In between os.unlink and os.rename, another process could try
1068                 #   to import the module, having seen that it already exists.
1069                 # * If another process starts a COM server while we are still
1070                 #   generating __init__.py, that process sees that the folder
1071                 #   already exists and assumes that __init__.py is already there
1072                 #   as well.
1073                 try:
1074                     os.unlink(filename)
1075                 except os.error:
1076                     pass
1077                 os.rename(temp_filename, filename)
1078         else:
1079             os.unlink(temp_filename)

1276     def generate_child(self, child, dir):
1277         "Generate a single child.  May force a few children to be built as we generate deps"
1278         self.generate_type = GEN_DEMAND_CHILD
1279

1281         lcid = la[1]
1282         clsid = la[0]
1283         major = la[3]
1284         minor = la[4]
1285         self.base_mod_name = (
1286             "win32com.gen_py." + str(clsid)[1:-1] + "x%sx%sx%s" % (lcid, major, minor)
1287         )
1288         try:
1289             # Process the type library's CoClass objects, looking for the
1290             # specified name, or where a child has the specified name.
1291             # This ensures that all interesting things (including event interfaces)
1292             # are generated correctly.
1293             oleItems = {}
1294             vtableItems = {}
1295             infos = self.CollectOleItemInfosFromType()
1296             found = 0
1297             for type_info_tuple in infos:
1298                 info, infotype, doc, attr = type_info_tuple
1299                 if infotype == pythoncom.TKIND_COCLASS:
1300                     coClassItem, child_infos = self._Build_CoClass(type_info_tuple)
1301                     found = build.MakePublicAttributeName(doc[0]) == child
1302                     if not found:
1303                         # OK, check the child interfaces
1304                         for (
1305                             info,
1306                             info_type,
1307                             refType,
1308                             doc,
1309                             refAttr,
1310                             flags,
1311                         ) in child_infos:
1312                             if build.MakePublicAttributeName(doc[0]) == child:
1313                                 found = 1
1314                                 break
1315                     if found:
1316                         oleItems[coClassItem.clsid] = coClassItem
1317                         self._Build_CoClassChildren(
1318                             coClassItem, child_infos, oleItems, vtableItems
1319                         )
1320                         break
1321             if not found:
1322                 # Doesn't appear in a class defn - look in the interface objects for it
1323                 for type_info_tuple in infos:
1324                     info, infotype, doc, attr = type_info_tuple
1325                     if infotype in [
1326                         pythoncom.TKIND_INTERFACE,
1327                         pythoncom.TKIND_DISPATCH,
1328                     ]:
1329                         if build.MakePublicAttributeName(doc[0]) == child:
1330                             found = 1
1331                             oleItem, vtableItem = self._Build_Interface(type_info_tuple)
1332                             oleItems[clsid] = oleItem  # Even "None" goes in here.
1333                             if vtableItem is not None:
1334                                 vtableItems[clsid] = vtableItem
1335 
1336             assert (
1337                 found
1338             ), "Cant find the '%s' interface in the CoClasses, or the interfaces" % (
1339                 child,
1340             )
1341             # Make a map of iid: dispitem, vtableitem)
1342             items = {}
1343             for key, value in oleItems.items():
1344                 items[key] = (value, None)
1345             for key, value in vtableItems.items():
1346                 existing = items.get(key, None)
1347                 if existing is not None:
1348                     new_val = existing[0], value
1349                 else:
1350                     new_val = None, value
1351                 items[key] = new_val
1352

1354             for oleitem, vtableitem in items.values():

1365                     if vtableitem is not None:
1366                         self.do_gen_child_item(vtableitem)

1368                     worked = True
1369                 finally:
1370                     self.finish_writer(out_name, self.file, worked)
