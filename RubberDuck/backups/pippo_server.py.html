1    # A little test server, complete with typelib, we can use for testing.
2    # Originally submitted with bug:
3    # [ 753154 ] memory leak wrapping object having _typelib_guid_ attribute
4    # but modified by mhammond for use as part of the test suite.
5    import sys, os
6    import pythoncom
7    import win32com
8    import winerror
9    from win32com.server.util import wrap
10   
11   
12   class CPippo:
13       #
14       # COM declarations
15       #
16       _reg_clsid_ = "{1F0F75D6-BD63-41B9-9F88-2D9D2E1AA5C3}"
17       _reg_desc_ = "Pippo Python test object"
18       _reg_progid_ = "Python.Test.Pippo"
19       # _reg_clsctx_ = pythoncom.CLSCTX_LOCAL_SERVER
20       ###
21       ### Link to typelib
22       _typelib_guid_ = "{7783054E-9A20-4584-8C62-6ED2A08F6AC6}"
23       _typelib_version_ = 1, 0
24       _com_interfaces_ = ["IPippo"]
25   
26       def __init__(self):
27           self.MyProp1 = 10
28   
29       def Method1(self):
30           return wrap(CPippo())
31   
32       def Method2(self, in1, inout1):
33           return in1, inout1 * 2
34   
35       def Method3(self, in1):
36           # in1 will be a tuple, not a list.
37           # Yet, we are not allowed to return a tuple, but need to convert it to a list first. (Bug?)
38           return list(in1)

41   def BuildTypelib():
42       from distutils.dep_util import newer
43   
44       this_dir = os.path.dirname(__file__)
45       idl = os.path.abspath(os.path.join(this_dir, "pippo.idl"))
46       tlb = os.path.splitext(idl)[0] + ".tlb"
47       if newer(idl, tlb):
48           print("Compiling %s" % (idl,))
49           rc = os.system('midl "%s"' % (idl,))
50           if rc:
51               raise RuntimeError("Compiling MIDL failed!")
52           # Can't work out how to prevent MIDL from generating the stubs.
53           # just nuke them
54           for fname in "dlldata.c pippo_i.c pippo_p.c pippo.h".split():
55               os.remove(os.path.join(this_dir, fname))
56   
57       print("Registering %s" % (tlb,))
58       tli = pythoncom.LoadTypeLib(tlb)
59       pythoncom.RegisterTypeLib(tli, tlb)
60   
61   
62   def UnregisterTypelib():
63       k = CPippo
64       try:
65           pythoncom.UnRegisterTypeLib(
66               k._typelib_guid_,
67               k._typelib_version_[0],
68               k._typelib_version_[1],
69               0,
70               pythoncom.SYS_WIN32,
71           )
72           print("Unregistered typelib")
73       except pythoncom.error as details:
74           if details[0] == winerror.TYPE_E_REGISTRYACCESS:
75               pass
76           else:
77               raise
78   
79   
80   def main(argv=None):
81       if argv is None:
82           argv = sys.argv[1:]
83       if "--unregister" in argv:
84           # Unregister the type-libraries.
85           UnregisterTypelib()
86       else:
87           # Build and register the type-libraries.
88           BuildTypelib()
89       import win32com.server.register
90   
91       win32com.server.register.UseCommandLine(CPippo)
92   
93   
94   if __name__ == "__main__":
95       main(sys.argv)
