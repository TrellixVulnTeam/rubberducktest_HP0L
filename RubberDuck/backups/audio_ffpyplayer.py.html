audio_ffpyplayer.py 
1    ''' 
2    FFmpeg based audio player 
3    ========================= 
4     
5    To use, you need to install ffpyplayer and have a compiled ffmpeg shared 
6    library. 
7     
8        https://github.com/matham/ffpyplayer 
9     
10   The docs there describe how to set this up. But briefly, first you need to 
11   compile ffmpeg using the shared flags while disabling the static flags (you'll 
12   probably have to set the fPIC flag, e.g. CFLAGS=-fPIC). Here's some 
13   instructions: https://trac.ffmpeg.org/wiki/CompilationGuide. For Windows, you 
14   can download compiled GPL binaries from http://ffmpeg.zeranoe.com/builds/. 
15   Similarly, you should download SDL. 
16    
17   Now, you should a ffmpeg and sdl directory. In each, you should have a include, 
18   bin, and lib directory, where e.g. for Windows, lib contains the .dll.a files, 
19   while bin contains the actual dlls. The include directory holds the headers. 
20   The bin directory is only needed if the shared libraries are not already on 
21   the path. In the environment define FFMPEG_ROOT and SDL_ROOT, each pointing to 
22   the ffmpeg, and SDL directories, respectively. (If you're using SDL2, 
23   the include directory will contain a directory called SDL2, which then holds 
24   the headers). 
25    
26   Once defined, download the ffpyplayer git and run 
27    
28       python setup.py build_ext --inplace 
29    
30   Finally, before running you need to ensure that ffpyplayer is in python's path. 
31    
32   ..Note:: 
33    
34       When kivy exits by closing the window while the audio is playing, 
35       it appears that the __del__method of SoundFFPy 
36       is not called. Because of this the SoundFFPy object is not 
37       properly deleted when kivy exits. The consequence is that because 
38       MediaPlayer creates internal threads which do not have their daemon 
39       flag set, when the main threads exists it'll hang and wait for the other 
40       MediaPlayer threads to exit. But since __del__ is not called to delete the 
41       MediaPlayer object, those threads will remain alive hanging kivy. What this 
42       means is that you have to be sure to delete the MediaPlayer object before 
43       kivy exits by setting it to None. 
44   '''

56   from kivy.clock import Clock
57   from kivy.logger import Logger

63       Logger.info(
64           'SoundFFPy: Using ffpyplayer {}'.format(ffpyplayer.__version__))
65   except:
66       Logger.info('SoundFFPy: Using ffpyplayer {}'.format(ffpyplayer.version))

69   logger_func = {'quiet': Logger.critical, 'panic': Logger.critical,
70                  'fatal': Logger.critical, 'error': Logger.error,
71                  'warning': Logger.warning, 'info': Logger.info,
72                  'verbose': Logger.debug, 'debug': Logger.debug}
73   
74   
75   def _log_callback(message, level):
76       message = message.strip()
77       if message:
78           logger_func[level]('ffpyplayer: {}'.format(message))
79

81   class SoundFFPy(Sound):
82   
83       @staticmethod
84       def extensions():
85           return formats_in
86   
87       def __init__(self, **kwargs):
88           self._ffplayer = None
89           self.quitted = False
90           self._log_callback_set = False
91           self._state = ''
92           self.state = 'stop'
93   
94           if not get_log_callback():
95               set_log_callback(_log_callback)
96               self._log_callback_set = True
97   
98           super(SoundFFPy, self).__init__(**kwargs)
99   
100      def __del__(self):
101          self.unload()
102          if self._log_callback_set:
103              set_log_callback(None)
104  
105      def _player_callback(self, selector, value):
106          if self._ffplayer is None:
107              return
108          if selector == 'quit':
109              def close(*args):
110                  self.quitted = True
111                  self.unload()
112              Clock.schedule_once(close, 0)
113          elif selector == 'eof':
114              Clock.schedule_once(self._do_eos, 0)
115  
116      def load(self):
117          self.unload()
118          ff_opts = {'vn': True, 'sn': True}  # only audio
119          self._ffplayer = MediaPlayer(self.source,
120                                       callback=self._player_callback,
121                                       loglevel='info', ff_opts=ff_opts)
122          player = self._ffplayer
123          player.set_volume(self.volume)
124          player.toggle_pause()
125          self._state = 'paused'
126          # wait until loaded or failed, shouldn't take long, but just to make
127          # sure metadata is available.
128          s = time.perf_counter()
129          while (player.get_metadata()['duration'] is None and
130                 not self.quitted and time.perf_counter() - s < 10.):
131              time.sleep(0.005)
132  
133      def unload(self):
134          if self._ffplayer:
135              self._ffplayer = None
136          self._state = ''
137          self.state = 'stop'
138          self.quitted = False
139  
140      def play(self):
141          if self._state == 'playing':
142              super(SoundFFPy, self).play()
143              return
144          if not self._ffplayer:
145              self.load()
146          self._ffplayer.toggle_pause()
147          self._state = 'playing'
148          self.state = 'play'
149          super(SoundFFPy, self).play()
150  
151      def stop(self):
152          if self._ffplayer and self._state == 'playing':
153              self._ffplayer.toggle_pause()
154              self._state = 'paused'
155              self.state = 'stop'
156          super(SoundFFPy, self).stop()
157  
158      def seek(self, position):
159          if self._ffplayer is None:
160              return
161          self._ffplayer.seek(position, relative=False)
162  
163      def get_pos(self):
164          if self._ffplayer is not None:
165              return self._ffplayer.get_pts()
166          return 0
167  
168      def on_volume(self, instance, volume):
169          if self._ffplayer is not None:
170              self._ffplayer.set_volume(volume)
171  
172      def _get_length(self):
173          if self._ffplayer is None:
174              return super(SoundFFPy, self)._get_length()
175          return self._ffplayer.get_metadata()['duration']
176  
177      def _do_eos(self, *args):
178          if not self.loop:
179              self.stop()
180          else:
181              self.seek(0.)
182  
183  
184  SoundLoader.register(SoundFFPy)
