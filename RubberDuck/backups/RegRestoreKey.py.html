1    import win32api, win32security
2    import win32con, ntsecuritycon, winnt
3    import os
4    
5    temp_dir = win32api.GetTempPath()
6    fname = win32api.GetTempFileName(temp_dir, "rsk")[0]
7    print(fname)
8    ## file can't exist
9    os.remove(fname)
10   
11   ## enable backup and restore privs
12   required_privs = (
13       (
14           win32security.LookupPrivilegeValue("", ntsecuritycon.SE_BACKUP_NAME),
15           win32con.SE_PRIVILEGE_ENABLED,
16       ),
17       (
18           win32security.LookupPrivilegeValue("", ntsecuritycon.SE_RESTORE_NAME),
19           win32con.SE_PRIVILEGE_ENABLED,
20       ),
21   )
22   ph = win32api.GetCurrentProcess()
23   th = win32security.OpenProcessToken(
24       ph, win32con.TOKEN_READ | win32con.TOKEN_ADJUST_PRIVILEGES
25   )
26   adjusted_privs = win32security.AdjustTokenPrivileges(th, 0, required_privs)
27   
28   try:
29       sa = win32security.SECURITY_ATTRIBUTES()
30       my_sid = win32security.GetTokenInformation(th, ntsecuritycon.TokenUser)[0]
31       sa.SECURITY_DESCRIPTOR.SetSecurityDescriptorOwner(my_sid, 0)
32   
33       k, disp = win32api.RegCreateKeyEx(
34           win32con.HKEY_CURRENT_USER,
35           "Python test key",
36           SecurityAttributes=sa,
37           samDesired=win32con.KEY_ALL_ACCESS,
38           Class="some class",
39           Options=0,
40       )
41       win32api.RegSetValue(k, None, win32con.REG_SZ, "Default value for python test key")
42   
43       subk, disp = win32api.RegCreateKeyEx(
44           k,
45           "python test subkey",
46           SecurityAttributes=sa,
47           samDesired=win32con.KEY_ALL_ACCESS,
48           Class="some other class",
49           Options=0,
50       )
51       win32api.RegSetValue(subk, None, win32con.REG_SZ, "Default value for subkey")
52   
53       win32api.RegSaveKeyEx(
54           k, fname, Flags=winnt.REG_STANDARD_FORMAT, SecurityAttributes=sa
55       )
56   
57       restored_key, disp = win32api.RegCreateKeyEx(
58           win32con.HKEY_CURRENT_USER,
59           "Python test key(restored)",
60           SecurityAttributes=sa,
61           samDesired=win32con.KEY_ALL_ACCESS,
62           Class="restored class",
63           Options=0,
64       )
65       win32api.RegRestoreKey(restored_key, fname)
66   finally:
67       win32security.AdjustTokenPrivileges(th, 0, adjusted_privs)
