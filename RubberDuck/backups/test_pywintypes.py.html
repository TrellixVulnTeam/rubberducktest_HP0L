11       def testPyTimeFormat(self):
12           struct_current = time.localtime()
13           pytime_current = pywintypes.Time(struct_current)
14           # try and test all the standard parts of the format
15           # Note we used to include '%Z' testing, but that was pretty useless as
16           # it always returned the local timezone.
17           format_strings = "%a %A %b %B %c %d %H %I %j %m %M %p %S %U %w %W %x %X %y %Y"
18           for fmt in format_strings.split():
19               v1 = pytime_current.Format(fmt)
20               v2 = time.strftime(fmt, struct_current)
21               self.assertEquals(v1, v2, "format %s failed - %r != %r" % (fmt, v1, v2))

58           now = datetime.datetime.now()  # has usec...
59           # timetuple() lost usec - pt must be <=...
60           pt = pywintypes.Time(now.timetuple())
61           # *sob* - only if we have a datetime object can we compare like this.

65       def testTimeTuplems(self):
66           now = datetime.datetime.now()  # has usec...
67           tt = now.timetuple() + (now.microsecond // 1000,)
68           pt = pywintypes.Time(tt)
69           # we can't compare if using the old type, as it loses all sub-second res.
70           if isinstance(pt, datetime.datetime):
71               # but even with datetime, we lose sub-millisecond.
72               expectedDelta = datetime.timedelta(milliseconds=1)
73               self.assertTrue(-expectedDelta < (now - pt) < expectedDelta)
