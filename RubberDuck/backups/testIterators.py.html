28       def _do_test_nonenum(self, object):
29           try:
30               for i in object:
31                   pass
32               self.fail("Could iterate over a non-iterable object")
33           except TypeError:
34               pass  # this is expected.
35           self.assertRaises(TypeError, iter, object)
36           self.assertRaises(AttributeError, getattr, object, "next")

38       def test_nonenum_wrapper(self):
39           # Check our raw PyIDispatch
40           ob = self.object._oleobj_
41           try:
42               for i in ob:
43                   pass
44               self.fail("Could iterate over a non-iterable object")
45           except TypeError:
46               pass  # this is expected.
47           self.assertRaises(TypeError, iter, ob)
48           self.assertRaises(AttributeError, getattr, ob, "next")
49   
50           # And our Dispatch wrapper
51           ob = self.object
52           try:
53               for i in ob:
54                   pass
55               self.fail("Could iterate over a non-iterable object")
56           except TypeError:
57               pass  # this is expected.
58           # Note that as our object may be dynamic, we *do* have a __getitem__
59           # method, meaning we *can* call iter() on the object.  In this case
60           # actual iteration is what fails.
61           # So either the 'iter(); will raise a type error, or an attempt to
62           # fetch it
63           try:
64               next(iter(ob))
65               self.fail("Expected a TypeError fetching this iterator")
66           except TypeError:
67               pass
68           # And it should never have a 'next' method
69           self.assertRaises(AttributeError, getattr, ob, "next")

72   class VBTestCase(_BaseTestCase):
73       def setUp(self):
74           def factory():
75               # Our VB test harness exposes a property with IEnumVariant.
76               ob = self.object.EnumerableCollectionProperty
77               for i in self.expected_data:
78                   ob.Add(i)
79               # Get the raw IEnumVARIANT.
80               invkind = pythoncom.DISPATCH_METHOD | pythoncom.DISPATCH_PROPERTYGET
81               iter = ob._oleobj_.InvokeTypes(
82                   pythoncom.DISPID_NEWENUM, 0, invkind, (13, 10), ()
83               )
84               return ob, iter.QueryInterface(pythoncom.IID_IEnumVARIANT)
85   
86           # We *need* generated dispatch semantics, so dynamic __getitem__ etc
87           # don't get in the way of our tests.
88           self.object = EnsureDispatch("PyCOMVBTest.Tester")
89           self.expected_data = [1, "Two", "3"]
90           self.iter_factory = factory
91   
92       def tearDown(self):
93           self.object = None

112              ob = self.object.GetCollection()
113              flags = pythoncom.DISPATCH_METHOD | pythoncom.DISPATCH_PROPERTYGET

126  def suite():
127      # We dont want our base class run
