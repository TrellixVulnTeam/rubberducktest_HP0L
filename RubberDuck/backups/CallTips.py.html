44       def _remove_calltip_window(self):

49       def paren_open_event(self, event):
50           self._remove_calltip_window()
51           arg_text = get_arg_text(self.get_object_at_cursor())
52           if arg_text:
53               self.calltip_start = self.text.index("insert")
54               self.calltip = self._make_calltip_window()
55               self.calltip.showtip(arg_text)
56           return ""  # so the event is handled normally.
57   
58       def paren_close_event(self, event):
59           # Now just hides, but later we should check if other
60           # paren'd expressions remain open.
61           self._remove_calltip_window()
62           return ""  # so the event is handled normally.

64       def check_calltip_cancel_event(self, event):
65           if self.calltip:
66               # If we have moved before the start of the calltip,
67               # or off the calltip line, then cancel the tip.
68               # (Later need to be smarter about multi-line, etc)
69               if self.text.compare(
70                   "insert", "<=", self.calltip_start
71               ) or self.text.compare("insert", ">", self.calltip_start + " lineend"):
72                   self._remove_calltip_window()
73           return ""  # so the event is handled normally.
74   
75       def calltip_cancel_event(self, event):
76           self._remove_calltip_window()
77           return ""  # so the event is handled normally.

79       def get_object_at_cursor(
80           self,
81           wordchars="._"
82           + string.ascii_uppercase
83           + string.ascii_lowercase
84           + string.digits,
85       ):
86           # XXX - This needs to be moved to a better place
87           # so the "." attribute lookup code can also use it.
88           text = self.text
89           chars = text.get("insert linestart", "insert")
90           i = len(chars)
91           while i and chars[i - 1] in wordchars:
92               i = i - 1
93           word = chars[i:]
94           if word:
95               # How is this for a hack!
96               import sys, __main__
97   
98               namespace = sys.modules.copy()
99               namespace.update(__main__.__dict__)
100              try:
101                  return eval(word, namespace)
102              except:
103                  pass
104          return None  # Can't find an object.

107  def _find_constructor(class_ob):
108      # Given a class object, return a function object used for the
109      # constructor (ie, __init__() ) or None if we can't find one.
110      try:
111          return class_ob.__init__
112      except AttributeError:
113          for base in class_ob.__bases__:
114              rc = _find_constructor(base)
115              if rc is not None:
116                  return rc
117      return None

120  def get_arg_text(ob):
121      # Get a string describing the arguments for the given object.
122      argText = ""
123      if ob is not None:
124          argOffset = 0
125          if inspect.isclass(ob):
126              # Look for the highest __init__ in the class chain.
127              fob = _find_constructor(ob)
128              if fob is None:
129                  fob = lambda: None
130          else:
131              fob = ob
132          if inspect.isfunction(fob) or inspect.ismethod(fob):
133              try:
134                  # py3k has a 'getfullargspec' which can handle py3k specific things.
135                  arg_getter = getattr(inspect, "getfullargspec", inspect.getargspec)
136                  argText = inspect.formatargspec(*arg_getter(fob))
137              except:
138                  print("Failed to format the args")
139                  traceback.print_exc()
140          # See if we can use the docstring
141          if hasattr(ob, "__doc__"):
142              doc = ob.__doc__
143              try:
144                  doc = doc.strip()
145                  pos = doc.find("\n")
146              except AttributeError:
147                  ## New style classes may have __doc__ slot without actually
148                  ## having a string assigned to it
149                  pass
150              else:
151                  if pos < 0 or pos > 70:
152                      pos = 70
153                  if argText:
154                      argText = argText + "\n"
155                  argText = argText + doc[:pos]
156  
157      return argText
