12       TOUCHINPUT, PEN_OR_TOUCH_MASK, PEN_OR_TOUCH_SIGNATURE, PEN_EVENT_TOUCH_MASK
13   from kivy.input.motionevent import MotionEvent
14   from kivy.input.shape import ShapeRect
15   
16   Window = None

18   
19   class WM_MotionEvent(MotionEvent):
20       '''MotionEvent representing the WM_MotionEvent event. 
21          Supports pos, shape and size profiles. 
22       '''
23       __attrs__ = ('size', )
24   
25       def depack(self, args):
26           self.is_touch = True
27           self.shape = ShapeRect()
28           self.sx, self.sy = args[0], args[1]
29           self.shape.width = args[2][0]
30           self.shape.height = args[2][1]
31           self.size = self.shape.width * self.shape.height
32           self.profile = ('pos', 'shape', 'size')
33   
34           super(WM_MotionEvent, self).depack(args)
35   
36       def __str__(self):
37           args = (self.id, self.uid, str(self.spos), self.device)
38           return '<WMMotionEvent id:%d uid:%d pos:%s device:%s>' % args

41   if 'KIVY_DOC' in os.environ:
42       # documentation hack
43       WM_MotionEventProvider = None
44   
45   else:
46       from ctypes.wintypes import HANDLE
47       from ctypes import (windll, sizeof, byref)
48       from collections import deque
49       from kivy.input.provider import MotionEventProvider
50       from kivy.input.factory import MotionEventFactory
51   
52       class WM_MotionEventProvider(MotionEventProvider):
53   
54           def start(self):
55               global Window
56               if not Window:
57                   from kivy.core.window import Window
58   
59               self.touch_events = deque()
60               self.touches = {}
61               self.uid = 0
62   
63               # get window handle, and register to receive WM_TOUCH messages
64               self.hwnd = windll.user32.GetActiveWindow()
65               windll.user32.RegisterTouchWindow(self.hwnd, 1)
66   
67               # inject our own wndProc to handle messages
68               # before window manager does
69               self.new_windProc = WNDPROC(self._touch_wndProc)
70               self.old_windProc = SetWindowLong_WndProc_wrapper(
71                   self.hwnd, self.new_windProc)
72   
73           def update(self, dispatch_fn):
74               c_rect = RECT()
75               windll.user32.GetClientRect(self.hwnd, byref(c_rect))
76               pt = POINT(x=0, y=0)
77               windll.user32.ClientToScreen(self.hwnd, byref(pt))
78               x_offset, y_offset = pt.x, pt.y
79               usable_w, usable_h = float(c_rect.w), float(c_rect.h)
80   
81               while True:
82                   try:
83                       t = self.touch_events.pop()
84                   except:
85                       break
86   
87                   # adjust x,y to window coordinates (0.0 to 1.0)
88                   x = (t.screen_x() - x_offset) / usable_w
89                   y = 1.0 - (t.screen_y() - y_offset) / usable_h
90   
91                   # actually dispatch input
92                   if t.event_type == 'begin':
93                       self.uid += 1
94                       self.touches[t.id] = WM_MotionEvent(
95                           self.device, self.uid, [x, y, t.size()])
96                       dispatch_fn('begin', self.touches[t.id])
97   
98                   if t.event_type == 'update' and t.id in self.touches:
99                       self.touches[t.id].move([x, y, t.size()])
100                      dispatch_fn('update', self.touches[t.id])
101  
102                  if t.event_type == 'end' and t.id in self.touches:
103                      touch = self.touches[t.id]
104                      touch.move([x, y, t.size()])
105                      touch.update_time_end()
106                      dispatch_fn('end', touch)
107                      del self.touches[t.id]
108  
109          def stop(self):
110              windll.user32.UnregisterTouchWindow(self.hwnd)
111              self.new_windProc = SetWindowLong_WndProc_wrapper(
112                  self.hwnd, self.old_windProc)
113  
114          # we inject this wndProc into our main window, to process
115          # WM_TOUCH and mouse messages before the window manager does
116          def _touch_wndProc(self, hwnd, msg, wParam, lParam):
117              done = False
118              if msg == WM_TABLET_QUERYSYSTEMGESTURE:
119                  return QUERYSYSTEMGESTURE_WNDPROC
120  
121              if msg == WM_TOUCH:
122                  done = self._touch_handler(msg, wParam, lParam)
123  
124              if msg >= WM_MOUSEMOVE and msg <= WM_MOUSELAST:
125                  done = self._mouse_handler(msg, wParam, lParam)
126  
127              if not done:
128                  return windll.user32.CallWindowProcW(self.old_windProc,
129                                                       hwnd, msg, wParam,
130                                                       lParam)
131              return 1
132  
133          # this on pushes WM_TOUCH messages onto our event stack
134          def _touch_handler(self, msg, wParam, lParam):
135              touches = (TOUCHINPUT * wParam)()
136              windll.user32.GetTouchInputInfo(HANDLE(lParam),
137                                              wParam,
138                                              touches,
139                                              sizeof(TOUCHINPUT))
140              for i in range(wParam):
141                  self.touch_events.appendleft(touches[i])
142              windll.user32.CloseTouchInputHandle(HANDLE(lParam))
143              return True
144  
145          # filter fake mouse events, because touch and stylus
146          # also make mouse events
147          def _mouse_handler(self, msg, wparam, lParam):
148              info = windll.user32.GetMessageExtraInfo()
149              # its a touch or a pen
150              if (info & PEN_OR_TOUCH_MASK) == PEN_OR_TOUCH_SIGNATURE:
151                  if info & PEN_EVENT_TOUCH_MASK:
152                      return True
153  
154      MotionEventFactory.register('wm_touch', WM_MotionEventProvider)
