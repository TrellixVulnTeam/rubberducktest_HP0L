5    # This module exposes an API for a Hierarchical Tree Control.
6    # Previously, a custom tree control was included in Pythonwin which
7    # has an API very similar to this.
8    
9    # The current control used is the common "Tree Control".  This module exists now
10   # to provide an API similar to the old control, but for the new Tree control.
11   
12   # If you need to use the Tree Control, you may still find this API a reasonable
13   # choice.  However, you should investigate using the tree control directly
14   # to provide maximum flexibility (but with extra work).
15

26   def GetItemText(item):
27       if type(item) == type(()) or type(item) == type([]):
28           use = item[0]
29       else:
30           use = item
31       if type(use) == type(""):
32           return use
33       else:
34           return repr(item)

60       def __init__(
61           self, root, bitmapID=win32ui.IDB_HIERFOLDERS, listBoxId=None, bitmapMask=None

63           self.listControl = None
64           self.bitmapID = bitmapID

85       def HierInit(self, parent, listControl=None):  # Used when window first exists.
86           # this also calls "Create" on the listbox.
87           # params - id of listbbox, ID of bitmap, size of bitmaps
88           if self.bitmapMask is None:
89               bitmapMask = RGB(0, 0, 255)
90           else:
91               bitmapMask = self.bitmapMask
92           self.imageList = win32ui.CreateImageList(self.bitmapID, 16, 0, bitmapMask)
93           if listControl is None:
94               if self.listBoxId is None:
95                   self.listBoxId = win32ui.IDC_LIST1
96               self.listControl = parent.GetDlgItem(self.listBoxId)
97           else:
98               self.listControl = listControl
99               lbid = listControl.GetDlgCtrlID()
100              assert self.listBoxId is None or self.listBoxId == lbid, (
101                  "An invalid listbox control ID has been specified (specified as %s, but exists as %s)"
102                  % (self.listBoxId, lbid)
103              )
104              self.listBoxId = lbid
105          self.listControl.SetImageList(self.imageList, commctrl.LVSIL_NORMAL)
106          #       self.list.AttachObject(self)
107  
108          ## ??? Need a better way to do this - either some way to detect if it's compiled with UNICODE
109          ##  defined, and/or a way to switch the constants based on UNICODE ???
110          if sys.version_info[0] < 3:
111              parent.HookNotify(self.OnTreeItemExpanding, commctrl.TVN_ITEMEXPANDINGA)
112              parent.HookNotify(self.OnTreeItemSelChanged, commctrl.TVN_SELCHANGEDA)
113          else:
114              parent.HookNotify(self.OnTreeItemExpanding, commctrl.TVN_ITEMEXPANDINGW)
115              parent.HookNotify(self.OnTreeItemSelChanged, commctrl.TVN_SELCHANGEDW)
116          parent.HookNotify(self.OnTreeItemDoubleClick, commctrl.NM_DBLCLK)
117          self.notify_parent = parent
118  
119          if self.root:
120              self.AcceptRoot(self.root)

122      def DeleteAllItems(self):
123          self.listControl.DeleteAllItems()
124          self.root = None
125          self.itemHandleMap = {}
126          self.filledItemHandlesMap = {}

128      def HierTerm(self):
129          # Dont want notifies as we kill the list.
130          parent = self.notify_parent  # GetParentFrame()
131          if sys.version_info[0] < 3:
132              parent.HookNotify(None, commctrl.TVN_ITEMEXPANDINGA)
133              parent.HookNotify(None, commctrl.TVN_SELCHANGEDA)
134          else:
135              parent.HookNotify(None, commctrl.TVN_ITEMEXPANDINGW)
136              parent.HookNotify(None, commctrl.TVN_SELCHANGEDW)
137          parent.HookNotify(None, commctrl.NM_DBLCLK)
138  
139          self.DeleteAllItems()
140          self.listControl = None
141          self.notify_parent = None  # Break a possible cycle

143      def OnTreeItemDoubleClick(self, info, extra):
144          (hwndFrom, idFrom, code) = info
145          if idFrom != self.listBoxId:
146              return None
147          item = self.itemHandleMap[self.listControl.GetSelectedItem()]
148          self.TakeDefaultAction(item)
149          return 1
150  
151      def OnTreeItemExpanding(self, info, extra):
152          (hwndFrom, idFrom, code) = info
153          if idFrom != self.listBoxId:
154              return None
155          action, itemOld, itemNew, pt = extra
156          itemHandle = itemNew[0]
157          if itemHandle not in self.filledItemHandlesMap:
158              item = self.itemHandleMap[itemHandle]
159              self.AddSubList(itemHandle, self.GetSubList(item))
160              self.filledItemHandlesMap[itemHandle] = None
161          return 0
162  
163      def OnTreeItemSelChanged(self, info, extra):
164          (hwndFrom, idFrom, code) = info
165          if idFrom != self.listBoxId:
166              return None
167          action, itemOld, itemNew, pt = extra
168          itemHandle = itemNew[0]
169          item = self.itemHandleMap[itemHandle]
170          self.PerformItemSelected(item)
171          return 1

173      def AddSubList(self, parentHandle, subItems):
174          for item in subItems:
175              self.AddItem(parentHandle, item)
176  
177      def AddItem(self, parentHandle, item, hInsertAfter=commctrl.TVI_LAST):
178          text = self.GetText(item)
179          if self.IsExpandable(item):
180              cItems = 1  # Trick it !!
181          else:
182              cItems = 0
183          bitmapCol = self.GetBitmapColumn(item)
184          bitmapSel = self.GetSelectedBitmapColumn(item)
185          if bitmapSel is None:
186              bitmapSel = bitmapCol
187          ## if type(text) is str:
188          ##  text = text.encode("mbcs")
189          hitem = self.listControl.InsertItem(
190              parentHandle,
191              hInsertAfter,
192              (None, None, None, text, bitmapCol, bitmapSel, cItems, 0),
193          )
194          self.itemHandleMap[hitem] = item
195          return hitem

197      def _GetChildHandles(self, handle):
198          ret = []
199          try:
200              handle = self.listControl.GetChildItem(handle)
201              while 1:
202                  ret.append(handle)
203                  handle = self.listControl.GetNextItem(handle, commctrl.TVGN_NEXT)
204          except win32ui.error:
205              # out of children
206              pass
207          return ret

212      def Refresh(self, hparent=None):
213          # Attempt to refresh the given item's sub-entries, but maintain the tree state
214          # (ie, the selected item, expanded items, etc)
215          if hparent is None:
216              hparent = commctrl.TVI_ROOT
217          if hparent not in self.filledItemHandlesMap:
218              # This item has never been expanded, so no refresh can possibly be required.
219              return
220          root_item = self.itemHandleMap[hparent]
221          old_handles = self._GetChildHandles(hparent)
222          old_items = list(map(self.ItemFromHandle, old_handles))
223          new_items = self.GetSubList(root_item)
224          # Now an inefficient technique for synching the items.
225          inew = 0
226          hAfter = commctrl.TVI_FIRST
227          for iold in range(len(old_items)):
228              inewlook = inew
229              matched = 0
230              while inewlook < len(new_items):
231                  if old_items[iold] == new_items[inewlook]:
232                      matched = 1
233                      break
234                  inewlook = inewlook + 1
235              if matched:
236                  # Insert the new items.
237                  #               print "Inserting after", old_items[iold], old_handles[iold]
238                  for i in range(inew, inewlook):
239                      #                   print "Inserting index %d (%s)" % (i, new_items[i])
240                      hAfter = self.AddItem(hparent, new_items[i], hAfter)
241  
242                  inew = inewlook + 1
243                  # And recursively refresh iold
244                  hold = old_handles[iold]
245                  if hold in self.filledItemHandlesMap:
246                      self.Refresh(hold)
247              else:
248                  # Remove the deleted items.
249                  #               print "Deleting %d (%s)" % (iold, old_items[iold])
250                  hdelete = old_handles[iold]
251                  # First recurse and remove the children from the map.
252                  for hchild in self._GetChildHandles(hdelete):
253                      del self.itemHandleMap[hchild]
254                      if hchild in self.filledItemHandlesMap:
255                          del self.filledItemHandlesMap[hchild]
256                  self.listControl.DeleteItem(hdelete)
257              hAfter = old_handles[iold]
258          # Fill any remaining new items:
259          for newItem in new_items[inew:]:
260              #           print "Inserting new item", newItem
261              self.AddItem(hparent, newItem)

263      def AcceptRoot(self, root):
264          self.listControl.DeleteAllItems()
265          self.itemHandleMap = {commctrl.TVI_ROOT: root}
266          self.filledItemHandlesMap = {commctrl.TVI_ROOT: root}
267          subItems = self.GetSubList(root)
268          self.AddSubList(0, subItems)

303  class HierListWithItems(HierList):
304      def __init__(
305          self, root, bitmapID=win32ui.IDB_HIERFOLDERS, listBoxID=None, bitmapMask=None

309      def DelegateCall(self, fn):
310          return fn()
311  
312      def GetBitmapColumn(self, item):
313          rc = self.DelegateCall(item.GetBitmapColumn)
314          if rc is None:
315              rc = HierList.GetBitmapColumn(self, item)
316          return rc
317  
318      def GetSelectedBitmapColumn(self, item):
319          return self.DelegateCall(item.GetSelectedBitmapColumn)
320  
321      def IsExpandable(self, item):
322          return self.DelegateCall(item.IsExpandable)
323  
324      def GetText(self, item):
325          return self.DelegateCall(item.GetText)
326  
327      def GetSubList(self, item):
328          return self.DelegateCall(item.GetSubList)
329  
330      def PerformItemSelected(self, item):
331          func = getattr(item, "PerformItemSelected", None)
332          if func is None:
333              return HierList.PerformItemSelected(self, item)
334          else:
335              return self.DelegateCall(func)
336  
337      def TakeDefaultAction(self, item):
338          func = getattr(item, "TakeDefaultAction", None)
339          if func is None:
340              return HierList.TakeDefaultAction(self, item)
341          else:
342              return self.DelegateCall(func)
