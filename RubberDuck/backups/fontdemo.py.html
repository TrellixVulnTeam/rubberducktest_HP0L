13   import win32ui
14   import win32con
15   import win32api
16   
17   from pywin.mfc import docview
18   
19   
20   # font is a dictionary in which the following elements matter:
21   # (the best matching font to supplied parameters is returned)
22   #   name        string name of the font as known by Windows
23   #   size        point size of font in logical units
24   #   weight      weight of font (win32con.FW_NORMAL, win32con.FW_BOLD)
25   #   italic      boolean; true if set to anything but None
26   #   underline   boolean; true if set to anything but None
27   
28   
29   class FontView(docview.ScrollView):
30       def __init__(
31           self, doc, text="Python Rules!", font_spec={"name": "Arial", "height": 42}
32       ):
33           docview.ScrollView.__init__(self, doc)
34           self.font = win32ui.CreateFont(font_spec)
35           self.text = text
36           self.width = self.height = 0
37           # set up message handlers
38           self.HookMessage(self.OnSize, win32con.WM_SIZE)
39   
40       def OnAttachedObjectDeath(self):
41           docview.ScrollView.OnAttachedObjectDeath(self)
42           del self.font
43   
44       def SetFont(self, new_font):
45           # Change font on the fly
46           self.font = win32ui.CreateFont(new_font)
47           # redraw the entire client window
48           selfInvalidateRect(None)
49   
50       def OnSize(self, params):
51           lParam = params[3]
52           self.width = win32api.LOWORD(lParam)
53           self.height = win32api.HIWORD(lParam)
54   
55       def OnPrepareDC(self, dc, printinfo):
56           # Set up the DC for forthcoming OnDraw call
57           self.SetScrollSizes(win32con.MM_TEXT, (100, 100))
58           dc.SetTextColor(win32api.RGB(0, 0, 255))
59           dc.SetBkColor(win32api.GetSysColor(win32con.COLOR_WINDOW))
60           dc.SelectObject(self.font)
61           dc.SetTextAlign(win32con.TA_CENTER | win32con.TA_BASELINE)
62   
63       def OnDraw(self, dc):
64           if self.width == 0 and self.height == 0:
65               left, top, right, bottom = self.GetClientRect()
66               self.width = right - left
67               self.height = bottom - top
68           x, y = self.width // 2, self.height // 2
69           dc.TextOut(x, y, self.text)

72   def FontDemo():
73       # create doc/view
74       template = docview.DocTemplate(win32ui.IDR_PYTHONTYPE, None, None, FontView)
75       doc = template.OpenDocumentFile(None)
76       doc.SetTitle("Font Demo")
77       #   print "template is ", template, "obj is", template._obj_
78       template.close()
