125      def analyse_text(text):
126          """This is very difficult to guess relative to other business languages. 
127          -> in conjuction with BEGIN/END seems relatively rare though."""
128          result = 0
129  
130          if '->' in text:
131              result += 0.01
132          if 'BEGIN' in text:
133              result += 0.01
134          if 'END' in text:
135              result += 0.01
136  
137          return result
