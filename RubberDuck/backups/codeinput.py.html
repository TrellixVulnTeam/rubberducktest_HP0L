1    ''' 
2    Code Input 
3    ========== 
4     
5    .. versionadded:: 1.5.0 
6     
7    .. image:: images/codeinput.jpg 
8     
9    .. note:: 
10    
11       This widget requires ``pygments`` package to run. Install it with ``pip``. 
12    
13   The :class:`CodeInput` provides a box of editable highlighted text like the one 
14   shown in the image. 
15    
16   It supports all the features provided by the :class:`~kivy.uix.textinput` as 
17   well as code highlighting for `languages supported by pygments 
18   <http://pygments.org/docs/lexers/>`_ along with `KivyLexer` for 
19   :mod:`kivy.lang` highlighting. 
20    
21   Usage example 
22   ------------- 
23    
24   To create a CodeInput with highlighting for `KV language`:: 
25    
26       from kivy.uix.codeinput import CodeInput 
27       from kivy.extras.highlight import KivyLexer 
28       codeinput = CodeInput(lexer=KivyLexer()) 
29    
30   To create a CodeInput with highlighting for `Cython`:: 
31    
32       from kivy.uix.codeinput import CodeInput 
33       from pygments.lexers import CythonLexer 
34       codeinput = CodeInput(lexer=CythonLexer()) 
35    
36   '''
37   
38   __all__ = ('CodeInput', )
39   
40   from pygments import highlight
41   from pygments import lexers
42   from pygments import styles
43   from pygments.formatters import BBCodeFormatter
44   
45   from kivy.uix.textinput import TextInput
46   from kivy.core.text.markup import MarkupLabel as Label
47   from kivy.cache import Cache
48   from kivy.properties import ObjectProperty, OptionProperty
49   from kivy.utils import get_hex_from_color, get_color_from_hex
50   from kivy.uix.behaviors import CodeNavigationBehavior
51   
52   Cache_get = Cache.get
53   Cache_append = Cache.append
54   
55   # TODO: color chooser for keywords/strings/...

58   class CodeInput(CodeNavigationBehavior, TextInput):
59       '''CodeInput class, used for displaying highlighted code. 
60       '''
61   
62       lexer = ObjectProperty(None)
63       '''This holds the selected Lexer used by pygments to highlight the code. 
64    
65    
66       :attr:`lexer` is an :class:`~kivy.properties.ObjectProperty` and 
67       defaults to `PythonLexer`. 
68       '''
69   
70       style_name = OptionProperty(
71           'default', options=list(styles.get_all_styles())
72       )
73       '''Name of the pygments style to use for formatting. 
74    
75       :attr:`style_name` is an :class:`~kivy.properties.OptionProperty` 
76       and defaults to ``'default'``. 
77    
78       '''
79   
80       style = ObjectProperty(None)
81       '''The pygments style object to use for formatting. 
82    
83       When ``style_name`` is set, this will be changed to the 
84       corresponding style object. 
85    
86       :attr:`style` is a :class:`~kivy.properties.ObjectProperty` and 
87       defaults to ``None`` 
88    
89       '''
90   
91       def __init__(self, **kwargs):
92           stylename = kwargs.get('style_name', 'default')
93           style = kwargs['style'] if 'style' in kwargs \
94               else styles.get_style_by_name(stylename)
95           self.formatter = BBCodeFormatter(style=style)
96           self.lexer = lexers.PythonLexer()
97           self.text_color = '#000000'
98           self._label_cached = Label()
99           self.use_text_color = True
100  
101          super(CodeInput, self).__init__(**kwargs)
102  
103          self._line_options = kw = self._get_line_options()
104          self._label_cached = Label(**kw)
105          # use text_color as foreground color
106          text_color = kwargs.get('foreground_color')
107          if text_color:
108              self.text_color = get_hex_from_color(text_color)
109          # set foreground to white to allow text colors to show
110          # use text_color as the default color in bbcodes
111          self.use_text_color = False
112          self.foreground_color = [1, 1, 1, .999]
113          if not kwargs.get('background_color'):
114              self.background_color = [.9, .92, .92, 1]
115  
116      def on_style_name(self, *args):
117          self.style = styles.get_style_by_name(self.style_name)
118          self.background_color = get_color_from_hex(self.style.background_color)
119          self._trigger_refresh_text()
120  
121      def on_style(self, *args):
122          self.formatter = BBCodeFormatter(style=self.style)
123          self._trigger_update_graphics()
124  
125      def _create_line_label(self, text, hint=False):
126          # Create a label from a text, using line options
127          ntext = text.replace(u'\n', u'').replace(u'\t', u' ' * self.tab_width)
128          if self.password and not hint:  # Don't replace hint_text with *
129              ntext = u'*' * len(ntext)
130          ntext = self._get_bbcode(ntext)
131          kw = self._get_line_options()
132          cid = u'{}\0{}\0{}'.format(ntext, self.password, kw)
133          texture = Cache_get('textinput.label', cid)
134  
135          if texture is None:
136              # FIXME right now, we can't render very long line...
137              # if we move on "VBO" version as fallback, we won't need to
138              # do this.
139              # try to find the maximum text we can handle
140              label = Label(text=ntext, **kw)
141              if text.find(u'\n') > 0:
142                  label.text = u''
143              else:
144                  label.text = ntext
145              label.refresh()
146  
147              # ok, we found it.
148              texture = label.texture
149              Cache_append('textinput.label', cid, texture)
150              label.text = ''
151          return texture
152  
153      def _get_line_options(self):
154          kw = super(CodeInput, self)._get_line_options()
155          kw['markup'] = True
156          kw['valign'] = 'top'
157          kw['codeinput'] = repr(self.lexer)
158          return kw
159  
160      def _get_text_width(self, text, tab_width, _label_cached):
161          # Return the width of a text, according to the current line options.
162          cid = u'{}\0{}\0{}'.format(text, self.password,
163                                     self._get_line_options())
164          width = Cache_get('textinput.width', cid)
165          if width is not None:
166              return width
167          lbl = self._create_line_label(text)
168          width = lbl.width
169          Cache_append('textinput.width', cid, width)
170          return width
171  
172      def _get_bbcode(self, ntext):
173          # get bbcoded text for python
174          try:
175              ntext[0]
176              # replace brackets with special chars that aren't highlighted
177              # by pygment. can't use &bl; ... cause & is highlighted
178              ntext = ntext.replace(u'[', u'\x01').replace(u']', u'\x02')
179              ntext = highlight(ntext, self.lexer, self.formatter)
180              ntext = ntext.replace(u'\x01', u'&bl;').replace(u'\x02', u'&br;')
181              # replace special chars with &bl; and &br;
182              ntext = ''.join((u'[color=', str(self.text_color), u']',
183                               ntext, u'[/color]'))
184              ntext = ntext.replace(u'\n', u'')
185              # remove possible extra highlight options
186              ntext = ntext.replace(u'[u]', '').replace(u'[/u]', '')
187              return ntext
188          except IndexError:
189              return ''
190  
191      # overriden to prevent cursor position off screen
192      def _cursor_offset(self):
193          '''Get the cursor x offset on the current line 
194          '''
195          offset = 0
196          try:
197              if self.cursor_col:
198                  offset = self._get_text_width(
199                      self._lines[self.cursor_row][:self.cursor_col])
200                  return offset
201          except:
202              pass
203          finally:
204              return offset
205  
206      def on_lexer(self, instance, value):
207          self._trigger_refresh_text()
208  
209      def on_foreground_color(self, instance, text_color):
210          if not self.use_text_color:
211              self.use_text_color = True
212              return
213          self.text_color = get_hex_from_color(text_color)
214          self.use_text_color = False
215          self.foreground_color = (1, 1, 1, .999)
216          self._trigger_refresh_text()
