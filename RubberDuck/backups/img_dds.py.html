5    __all__ = ('ImageLoaderDDS', )
6    
7    from kivy.lib.ddsfile import DDSFile
8    from kivy.logger import Logger
9    from kivy.core.image import ImageLoaderBase, ImageData, ImageLoader
10   
11   
12   class ImageLoaderDDS(ImageLoaderBase):
13   
14       @staticmethod
15       def extensions():
16           return ('dds', )
17   
18       def load(self, filename):
19           try:
20               dds = DDSFile(filename=filename)
21           except:
22               Logger.warning('Image: Unable to load image <%s>' % filename)
23               raise
24   
25           self.filename = filename
26           width, height = dds.size
27           im = ImageData(width, height, dds.dxt, dds.images[0], source=filename,
28                          flip_vertical=False)
29           if len(dds.images) > 1:
30               images = dds.images
31               images_size = dds.images_size
32               for index in range(1, len(dds.images)):
33                   w, h = images_size[index]
34                   data = images[index]
35                   im.add_mipmap(index, w, h, data)
36           return [im]
37   
38   
39   # register
40   ImageLoader.register(ImageLoaderDDS)
