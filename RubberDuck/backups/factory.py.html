5    def RegisterClassFactories(clsids, flags=None, clsctx=None):
6        """Given a list of CLSID, create and register class factories. 
7     
8        Returns a list, which should be passed to RevokeClassFactories 
9        """
10       if flags is None:
11           flags = pythoncom.REGCLS_MULTIPLEUSE | pythoncom.REGCLS_SUSPENDED
12       if clsctx is None:
13           clsctx = pythoncom.CLSCTX_LOCAL_SERVER
14       ret = []
15       for clsid in clsids:
16           # Some server append '-Embedding' etc
17           if clsid[0] not in ["-", "/"]:
18               factory = pythoncom.MakePyFactory(clsid)
19               regId = pythoncom.CoRegisterClassObject(clsid, factory, clsctx, flags)
20               ret.append((factory, regId))
21       return ret
22   
23   
24   def RevokeClassFactories(infos):
25       for factory, revokeId in infos:
26           pythoncom.CoRevokeClassObject(revokeId)
