1    ''' 
2    Logger object 
3    ============= 
4     
5    The Kivy `Logger` class provides a singleton logger instance. This instance 
6    exposes a standard Python 
7    `logger <https://docs.python.org/3/library/logging.html>`_ object but adds 
8    some convenient features. 
9     
10   All the standard logging levels are available : `trace`, `debug`, `info`, 
11   `warning`, `error` and `critical`. 
12    
13   Example Usage 
14   ------------- 
15    
16   Use the `Logger` as you would a standard Python logger. :: 
17    
18       from kivy.logger import Logger 
19    
20       Logger.info('title: This is a info message.') 
21       Logger.debug('title: This is a debug message.') 
22    
23       try: 
24           raise Exception('bleh') 
25       except Exception: 
26           Logger.exception('Something happened!') 
27    
28   The message passed to the logger is split into two parts separated by a colon 
29   (:). The first part is used as a title and the second part is used as the 
30   message. This way, you can "categorize" your messages easily. :: 
31    
32       Logger.info('Application: This is a test') 
33    
34       # will appear as 
35    
36       [INFO   ] [Application ] This is a test 
37    
38   You can change the logging level at any time using the `setLevel` method. :: 
39    
40       from kivy.logger import Logger, LOG_LEVELS 
41    
42       Logger.setLevel(LOG_LEVELS["debug"]) 
43    
44    
45   Features 
46   -------- 
47    
48   Although you are free to use standard python loggers, the Kivy `Logger` offers 
49   some solid benefits and useful features. These include: 
50    
51   * simplied usage (single instance, simple configuration, works by default) 
52   * color-coded output 
53   * output to `stdout` by default 
54   * message categorization via colon separation 
55   * access to log history even if logging is disabled 
56   * built-in handling of various cross-platform considerations 
57    
58   Kivys' logger was designed to be used with kivy apps and makes logging from 
59   Kivy apps more convenient. 
60    
61   Logger Configuration 
62   -------------------- 
63    
64   The Logger can be controlled via the Kivy configuration file:: 
65    
66       [kivy] 
67       log_level = info 
68       log_enable = 1 
69       log_dir = logs 
70       log_name = kivy_%y-%m-%d_%_.txt 
71       log_maxfiles = 100 
72    
73   More information about the allowed values are described in the 
74   :mod:`kivy.config` module. 
75    
76   Logger History 
77   -------------- 
78    
79   Even if the logger is not enabled, you still have access to the last 100 
80   messages:: 
81    
82       from kivy.logger import LoggerHistory 
83    
84       print(LoggerHistory.history) 
85

115  def formatter_message(message, use_color=True):
116      if use_color:
117          message = message.replace("$RESET", RESET_SEQ)
118          message = message.replace("$BOLD", BOLD_SEQ)
119      else:
120          message = message.replace("$RESET", "").replace("$BOLD", "")
121      return message

142  class FileHandler(logging.Handler):
143      history = []
144      filename = 'log.txt'
145      fd = None
146      log_dir = ''
147      encoding = 'utf-8'
148  
149      def purge_logs(self):
150          '''Purge log is called randomly to prevent the log directory from being 
151          filled by lots and lots of log files. 
152          You've a chance of 1 in 20 that purge log will be fired. 
153          '''
154          if randint(0, 20) != 0:
155              return
156          if not self.log_dir:
157              return
158  
159          from kivy.config import Config
160          maxfiles = Config.getint('kivy', 'log_maxfiles')
161  
162          if maxfiles < 0:
163              return
164  
165          Logger.info('Logger: Purge log fired. Analysing...')
166          join = os.path.join
167          unlink = os.unlink
168  
169          # search all log files
170          lst = [join(self.log_dir, x) for x in os.listdir(self.log_dir)]
171          if len(lst) > maxfiles:
172              # get creation time on every files
173              lst = [{'fn': x, 'ctime': os.path.getctime(x)} for x in lst]
174  
175              # sort by date
176              lst = sorted(lst, key=lambda x: x['ctime'])
177  
178              # get the oldest (keep last maxfiles)
179              lst = lst[:-maxfiles] if maxfiles else lst
180              Logger.info('Logger: Purge %d log files' % len(lst))
181  
182              # now, unlink every file in the list
183              for filename in lst:
184                  try:
185                      unlink(filename['fn'])
186                  except PermissionError as e:
187                      Logger.info('Logger: Skipped file {0}, {1}'.
188                                  format(filename['fn'], e))
189  
190          Logger.info('Logger: Purge finished!')
191  
192      def _configure(self, *largs, **kwargs):
193          from time import strftime
194          from kivy.config import Config
195          log_dir = Config.get('kivy', 'log_dir')
196          log_name = Config.get('kivy', 'log_name')
197  
198          _dir = kivy.kivy_home_dir
199          if log_dir and os.path.isabs(log_dir):
200              _dir = log_dir
201          else:
202              _dir = os.path.join(_dir, log_dir)
203          if not os.path.exists(_dir):
204              os.makedirs(_dir)
205          self.log_dir = _dir
206  
207          pattern = log_name.replace('%_', '@@NUMBER@@')
208          pattern = os.path.join(_dir, strftime(pattern))
209          n = 0
210          while True:
211              filename = pattern.replace('@@NUMBER@@', str(n))
212              if not os.path.exists(filename):
213                  break
214              n += 1
215              if n > 10000:  # prevent maybe flooding ?
216                  raise Exception('Too many logfile, remove them')
217  
218          if FileHandler.filename == filename and FileHandler.fd is not None:
219              return
220          FileHandler.filename = filename
221          if FileHandler.fd is not None:
222              FileHandler.fd.close()
223          FileHandler.fd = open(filename, 'w', encoding=FileHandler.encoding)
224          Logger.info('Logger: Record log in %s' % filename)
225  
226      def _write_message(self, record):
227          if FileHandler.fd in (None, False):
228              return
229  
230          msg = self.format(record)
231          stream = FileHandler.fd
232          fs = "%s\n"
233          stream.write('[%-7s] ' % record.levelname)
234          stream.write(fs % msg)
235          stream.flush()
236  
237      def emit(self, message):
238          # during the startup, store the message in the history
239          if Logger.logfile_activated is None:
240              FileHandler.history += [message]
241              return
242  
243          # startup done, if the logfile is not activated, avoid history.
244          if Logger.logfile_activated is False:
245              FileHandler.history = []
246              return
247  
248          if FileHandler.fd is None:
249              try:
250                  self._configure()
251                  from kivy.config import Config
252                  Config.add_callback(self._configure, 'kivy', 'log_dir')
253                  Config.add_callback(self._configure, 'kivy', 'log_name')
254              except Exception:
255                  # deactivate filehandler...
256                  FileHandler.fd = False
257                  Logger.exception('Error while activating FileHandler logger')
258                  return
259              while FileHandler.history:
260                  _message = FileHandler.history.pop()
261                  self._write_message(_message)
262  
263          self._write_message(message)

266  class LoggerHistory(logging.Handler):
267  
268      history = []
269  
270      def emit(self, message):
271          LoggerHistory.history = [message] + LoggerHistory.history[:100]
272  
273      @classmethod
274      def clear_history(cls):
275          del cls.history[:]
276  
277      def flush(self):
278          super(LoggerHistory, self).flush()
279          self.clear_history()

282  class ColoredFormatter(logging.Formatter):
283  
284      def __init__(self, msg, use_color=True):
285          logging.Formatter.__init__(self, msg)
286          self.use_color = use_color
287  
288      def format(self, record):
289          try:
290              msg = record.msg.split(':', 1)
291              if len(msg) == 2:
292                  record.msg = '[%-12s]%s' % (msg[0], msg[1])
293          except:
294              pass
295          levelname = record.levelname
296          if record.levelno == logging.TRACE:
297              levelname = 'TRACE'
298              record.levelname = levelname
299          if self.use_color and levelname in COLORS:
300              levelname_color = (
301                  COLOR_SEQ % (30 + COLORS[levelname]) + levelname + RESET_SEQ)
302              record.levelname = levelname_color
303          return logging.Formatter.format(self, record)

306  class ConsoleHandler(logging.StreamHandler):
307  
308      def filter(self, record):
309          try:
310              msg = record.msg
311              k = msg.split(':', 1)
312              if k[0] == 'stderr' and len(k) == 2:
313                  previous_stderr.write(k[1] + '\n')
314                  return False
315          except:
316              pass
317          return True

320  class LogFile(object):
321  
322      def __init__(self, channel, func):
323          self.buffer = ''
324          self.func = func
325          self.channel = channel
326          self.errors = ''
327  
328      def write(self, s):
329          s = self.buffer + s
330          self.flush()
331          f = self.func
332          channel = self.channel
333          lines = s.split('\n')
334          for l in lines[:-1]:
335              f('%s: %s' % (channel, l))
336          self.buffer = lines[-1]
337  
338      def flush(self):
339          return
340  
341      def isatty(self):
342          return False

345  def logger_config_update(section, key, value):
346      if LOG_LEVELS.get(value) is None:
347          raise AttributeError('Loglevel {0!r} doesn\'t exists'.format(value))
348      Logger.setLevel(level=LOG_LEVELS.get(value))

362  if 'KIVY_NO_FILELOG' not in os.environ:
363      file_log_handler = FileHandler()
364      Logger.addHandler(file_log_handler)
365  
366  # Use the custom handler instead of streaming one.
367  if 'KIVY_NO_CONSOLELOG' not in os.environ:
368      if hasattr(sys, '_kivy_logging_handler'):
369          Logger.addHandler(getattr(sys, '_kivy_logging_handler'))
370      else:
371          use_color = (
372              (
373                  os.environ.get("WT_SESSION") or
374                  os.environ.get("COLORTERM") == 'truecolor' or
375                  os.environ.get('PYCHARM_HOSTED') == '1' or
376                  os.environ.get('TERM') in (
377                      'rxvt',
378                      'rxvt-256color',
379                      'rxvt-unicode',
380                      'rxvt-unicode-256color',
381                      'xterm',
382                      'xterm-256color',
383                  )
384              ) and os.environ.get('KIVY_BUILD') not in ('android', 'ios')
385          )
386          if not use_color:
387              # No additional control characters will be inserted inside the
388              # levelname field, 7 chars will fit "WARNING"
389              color_fmt = formatter_message(
390                  '[%(levelname)-7s] %(message)s', use_color)
391          else:
392              # levelname field width need to take into account the length of the
393              # color control codes (7+4 chars for bold+color, and reset)
394              color_fmt = formatter_message(
395                  '[%(levelname)-18s] %(message)s', use_color)
396          formatter = ColoredFormatter(color_fmt, use_color=use_color)
397          console = ConsoleHandler()
398          console.setFormatter(formatter)
399          Logger.addHandler(console)
