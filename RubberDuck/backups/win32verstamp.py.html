13   VS_FFI_SIGNATURE = -17890115  # 0xFEEF04BD
14   VS_FFI_STRUCVERSION = 0x00010000
15   VS_FFI_FILEFLAGSMASK = 0x0000003F
16   VOS_NT_WINDOWS32 = 0x00040004
17   
18   null_byte = "\0".encode("ascii")  # str in py2k, bytes in py3k
19   #
20   # Set VS_FF_PRERELEASE and DEBUG if Debug
21   #
22   def file_flags(debug):
23       if debug:
24           return 3  # VS_FF_DEBUG | VS_FF_PRERELEASE
25       return 0
26   
27   
28   def file_type(is_dll):
29       if is_dll:
30           return 2  # VFT_DLL
31       return 1  # VFT_APP

34   def VS_FIXEDFILEINFO(maj, min, sub, build, debug=0, is_dll=1):
35       return struct.pack(
36           "lllllllllllll",
37           VS_FFI_SIGNATURE,  # dwSignature
38           VS_FFI_STRUCVERSION,  # dwStrucVersion
39           (maj << 16) | min,  # dwFileVersionMS
40           (sub << 16) | build,  # dwFileVersionLS
41           (maj << 16) | min,  # dwProductVersionMS
42           (sub << 16) | build,  # dwProductVersionLS
43           VS_FFI_FILEFLAGSMASK,  # dwFileFlagsMask
44           file_flags(debug),  # dwFileFlags
45           VOS_NT_WINDOWS32,  # dwFileOS
46           file_type(is_dll),  # dwFileType
47           0x00000000,  # dwFileSubtype
48           0x00000000,  # dwFileDateMS
49           0x00000000,  # dwFileDateLS
50       )

53   def nullterm(s):
54       # get raw bytes for a NULL terminated unicode string.
55       if sys.version_info[:2] < (3, 7):
56           return (str(s) + "\0").encode("unicode-internal")
57       else:
58           return (str(s) + "\0").encode("utf-16le")

61   def pad32(s, extra=2):
62       # extra is normally 2 to deal with wLength
63       l = 4 - ((len(s) + extra) & 3)
64       if l < 4:
65           return s + (null_byte * l)
66       return s

69   def addlen(s):
70       return struct.pack("h", len(s) + 2) + s
71   
72   
73   def String(key, value):
74       key = nullterm(key)
75       value = nullterm(value)
76       result = struct.pack("hh", len(value) // 2, 1)  # wValueLength, wType
77       result = result + key
78       result = pad32(result) + value
79       return addlen(result)
80   
81   
82   def StringTable(key, data):
83       key = nullterm(key)
84       result = struct.pack("hh", 0, 1)  # wValueLength, wType
85       result = result + key
86       for k, v in data.items():
87           result = result + String(k, v)
88           result = pad32(result)
89       return addlen(result)
90   
91   
92   def StringFileInfo(data):
93       result = struct.pack("hh", 0, 1)  # wValueLength, wType
94       result = result + nullterm("StringFileInfo")
95       #  result = pad32(result) + StringTable('040904b0', data)
96       result = pad32(result) + StringTable("040904E4", data)
97       return addlen(result)
98   
99   
100  def Var(key, value):
101      result = struct.pack("hh", len(value), 0)  # wValueLength, wType
102      result = result + nullterm(key)
103      result = pad32(result) + value
104      return addlen(result)
105  
106  
107  def VarFileInfo(data):
108      result = struct.pack("hh", 0, 1)  # wValueLength, wType
109      result = result + nullterm("VarFileInfo")
110      result = pad32(result)
111      for k, v in data.items():
112          result = result + Var(k, v)
113      return addlen(result)
114

116  def VS_VERSION_INFO(maj, min, sub, build, sdata, vdata, debug=0, is_dll=1):
117      ffi = VS_FIXEDFILEINFO(maj, min, sub, build, debug, is_dll)
118      result = struct.pack("hh", len(ffi), 0)  # wValueLength, wType
119      result = result + nullterm("VS_VERSION_INFO")
120      result = pad32(result) + ffi
121      result = pad32(result) + StringFileInfo(sdata) + VarFileInfo(vdata)
122      return addlen(result)

125  def stamp(pathname, options):
126      # For some reason, the API functions report success if the file is open
127      # but doesnt work!  Try and open the file for writing, just to see if it is
128      # likely the stamp will work!
129      try:
130          f = open(pathname, "a+b")
131          f.close()
132      except IOError as why:
133          print("WARNING: File %s could not be opened - %s" % (pathname, why))
134  
135      ver = options.version
136      try:
137          bits = [int(i) for i in ver.split(".")]
138          vmaj, vmin, vsub, vbuild = bits
139      except (IndexError, TypeError, ValueError):
140          raise ValueError("--version must be a.b.c.d (all integers) - got %r" % ver)
141  
142      ifn = options.internal_name
143      if not ifn:
144          ifn = os.path.basename(pathname)
145      ofn = options.original_filename
146      if ofn is None:
147          ofn = os.path.basename(pathname)
148  
149      sdata = {
150          "Comments": options.comments,
151          "CompanyName": options.company,
152          "FileDescription": options.description,
153          "FileVersion": ver,
154          "InternalName": ifn,
155          "LegalCopyright": options.copyright,
156          "LegalTrademarks": options.trademarks,
157          "OriginalFilename": ofn,
158          "ProductName": options.product,
159          "ProductVersion": ver,
160      }
161      vdata = {
162          "Translation": struct.pack("hh", 0x409, 1252),
163      }
164      is_dll = options.dll
165      if is_dll is None:
166          is_dll = os.path.splitext(pathname)[1].lower() in ".dll .pyd".split()
167      is_debug = options.debug
168      if is_debug is None:
169          is_debug = os.path.splitext(pathname)[0].lower().endswith("_d")
170      # convert None to blank strings
171      for k, v in list(sdata.items()):
172          if v is None:
173              sdata[k] = ""
174      vs = VS_VERSION_INFO(vmaj, vmin, vsub, vbuild, sdata, vdata, is_debug, is_dll)
175  
176      h = BeginUpdateResource(pathname, 0)
177      UpdateResource(h, 16, 1, vs)
178      EndUpdateResource(h, 0)
179  
180      if options.verbose:
181          print("Stamped:", pathname)
182

208      parser.add_option("", "--comments", help="The comments string to embed.")
209      parser.add_option(

233      for g in args:
234          for f in glob.glob(g):
235              stamp(f, options)
