1    ''' 
2    reStructuredText renderer 
3    ========================= 
4     
5    .. versionadded:: 1.1.0 
6     
7    `reStructuredText <http://docutils.sourceforge.net/rst.html>`_ is an 
8    easy-to-read, what-you-see-is-what-you-get plaintext markup syntax and parser 
9    system. 
10    
11   .. note:: 
12    
13       This widget requires the ``docutils`` package to run. Install it with 
14       ``pip`` or include it as one of your deployment requirements. 
15    
16   .. warning:: 
17    
18       This widget is highly experimental. The styling and implementation should 
19       not be considered stable until this warning has been removed. 
20    
21   Usage with Text 
22   --------------- 
23    
24   :: 
25    
26       text = """ 
27       .. _top: 
28    
29       Hello world 
30       =========== 
31    
32       This is an **emphased text**, some ``interpreted text``. 
33       And this is a reference to top_:: 
34    
35           $ print("Hello world") 
36    
37       """ 
38       document = RstDocument(text=text) 
39    
40   The rendering will output: 
41    
42   .. image:: images/rstdocument.png 
43    
44   Usage with Source 
45   ----------------- 
46    
47   You can also render a rst file using the :attr:`~RstDocument.source` property:: 
48    
49       document = RstDocument(source='index.rst') 
50    
51   You can reference other documents using the role ``:doc:``. For example, in the 
52   document ``index.rst`` you can write:: 
53    
54       Go to my next document: :doc:`moreinfo.rst` 
55    
56   It will generate a link that, when clicked, opens the ``moreinfo.rst`` 
57   document. 
58    
59   '''

536      # internals.
537      content = ObjectProperty(None)
538      scatter = ObjectProperty(None)
539      anchors_widgets = ListProperty([])
540      refs_assoc = DictProperty({})
541  
542      def __init__(self, **kwargs):
543          self._trigger_load = Clock.create_trigger(self._load_from_text, -1)
544          self._parser = rst.Parser()
545          self._settings = frontend.OptionParser(
546              components=(rst.Parser, )).get_default_values()
547          super(RstDocument, self).__init__(**kwargs)
548  
549      def on_source(self, instance, value):
550          if not value:
551              return
552          if self.document_root is None:
553              # set the documentation root to the directory name of the
554              # first tile
555              self.document_root = abspath(dirname(value))
556          self._load_from_source()
557  
558      def on_text(self, instance, value):
559          self._trigger_load()
560  
561      def render(self):
562          '''Force document rendering. 
563          '''
564          self._load_from_text()
565  
566      def resolve_path(self, filename):
567          '''Get the path for this filename. If the filename doesn't exist, 
568          it returns the document_root + filename. 
569          '''
570          if exists(filename):
571              return filename
572          return join(self.document_root, filename)
573  
574      def preload(self, filename, encoding='utf-8', errors='strict'):
575          '''Preload a rst file to get its toctree and its title. 
576   
577          The result will be stored in :attr:`toctrees` with the ``filename`` as 
578          key. 
579          '''
580  
581          with open(filename, 'rb') as fd:
582              text = fd.read().decode(encoding, errors)
583          # parse the source
584          document = utils.new_document('Document', self._settings)
585          self._parser.parse(text, document)
586          # fill the current document node
587          visitor = _ToctreeVisitor(document)
588          document.walkabout(visitor)
589          self.toctrees[filename] = visitor.toctree
590          return text
591  
592      def _load_from_source(self):
593          filename = self.resolve_path(self.source)
594          self.text = self.preload(filename,
595                                   self.source_encoding,
596                                   self.source_error)
597  
598      def _load_from_text(self, *largs):
599          try:
600              # clear the current widgets
601              self.content.clear_widgets()
602              self.anchors_widgets = []
603              self.refs_assoc = {}
604  
605              # parse the source
606              document = utils.new_document('Document', self._settings)
607              text = self.text
608              if PY2 and type(text) is str:
609                  text = text.decode('utf-8')
610              self._parser.parse(text, document)
611  
612              # fill the current document node
613              visitor = _Visitor(self, document)
614              document.walkabout(visitor)
615  
616              self.title = visitor.title or 'No title'
617          except:
618              Logger.exception('Rst: error while loading text')
619  
620      def on_ref_press(self, node, ref):
621          self.goto(ref)
622  
623      def goto(self, ref, *largs):
624          '''Scroll to the reference. If it's not found, nothing will be done. 
625   
626          For this text:: 
627   
628              .. _myref: 
629   
630              This is something I always wanted. 
631   
632          You can do:: 
633   
634              from kivy.clock import Clock 
635              from functools import partial 
636   
637              doc = RstDocument(...) 
638              Clock.schedule_once(partial(doc.goto, 'myref'), 0.1) 
639   
640          .. note:: 
641   
642              It is preferable to delay the call of the goto if you just loaded 
643              the document because the layout might not be finished or the 
644              size of the RstDocument has not yet been determined. In 
645              either case, the calculation of the scrolling would be 
646              wrong. 
647   
648              You can, however, do a direct call if the document is already 
649              loaded. 
650   
651          .. versionadded:: 1.3.0 
652          '''
653          # check if it's a file ?
654          if ref.endswith('.rst'):
655              # whether it's a valid or invalid file, let source deal with it
656              self.source = ref
657              return
658  
659          # get the association
660          ref = self.refs_assoc.get(ref, ref)
661  
662          # search into all the nodes containing anchors
663          ax = ay = None
664          for node in self.anchors_widgets:
665              if ref in node.anchors:
666                  ax, ay = node.anchors[ref]
667                  break
668  
669          # not found, stop here
670          if ax is None:
671              return
672  
673          # found, calculate the real coordinate
674  
675          # get the anchor coordinate inside widget space
676          ax += node.x
677          ay = node.top - ay
678          # ay += node.y
679  
680          # what's the current coordinate for us?
681          sx, sy = self.scatter.x, self.scatter.top
682          # ax, ay = self.scatter.to_parent(ax, ay)
683  
684          ay -= self.height
685  
686          dx, dy = self.convert_distance_to_scroll(0, ay)
687          dy = max(0, min(1, dy))
688          Animation(scroll_y=dy, d=.25, t='in_out_expo').start(self)
689  
690      def add_anchors(self, node):
691          self.anchors_widgets.append(node)

904      def brute_refs(self, node):
905          # get foot/cit refs manually because the output from
906          # docutils' parser doesn't contain any of these:
907          # node's refid, refname, backref, ... and/or are just ''/[]
908  
909          def get_refs(condition, backref=False):
910              # backref=True is used in nodes.footnote
911              autonum = autosym = 0
912              _nodes = node.traverse(condition=condition, ascend=False)
913  
914              for f in _nodes:
915                  id = f['ids'][0]
916                  auto = ''
917                  if 'auto' in f:
918                      auto = f['auto']
919  
920                  # auto is either 1(int) or '*'
921                  if auto == 1:
922                      autonum += 1
923                      key = 'backref' + str(autonum) if backref else str(autonum)
924                      self.root.refs_assoc[key] = id
925                  elif auto == '*':
926                      sym = self.footlist[
927                          autosym % 10
928                      ] * (int(autosym / 10) + 1)
929                      key = 'backref' + sym if backref else sym
930                      self.root.refs_assoc[key] = id
931                      autosym += 1
932                  else:
933                      if not backref:
934                          key = f['names'][0]
935                          if key:
936                              self.root.refs_assoc[key] = id
937                          continue
938  
939                      key = 'backref' + f['refname'][0]
940  
941                      if key in self.root.refs_assoc:
942                          self.root.refs_assoc[key].append(id)
943                      else:
944                          self.root.refs_assoc[key] = [id, ]
945  
946          # these are unique and need to go FIRST
947          get_refs(nodes.footnote, backref=False)
948  
949          # autonum & autosym are unique
950          get_refs(nodes.footnote_reference, backref=True)

952      def dispatch_visit(self, node):
953          cls = node.__class__
954          if cls is nodes.document:
955              self.push(self.root.content)
956              self.brute_refs(node)
957  
958          elif cls is nodes.comment:
959              return
960  
961          elif cls is nodes.section:
962              self.section += 1
963  
964          elif cls is nodes.substitution_definition:
965              name = node.attributes['names'][0]
966              self.substitution[name] = node.children[0]
967  
968          elif cls is nodes.substitution_reference:
969              node = self.substitution[node.attributes['refname']]
970              # it can be e.g. image or something else too!
971              if isinstance(node, nodes.Text):
972                  self.text += node
973  
974          elif cls is nodes.footnote:
975              # .. [x] footnote
976              text = ''
977              foot = RstFootnote()
978              ids = node.attributes['ids']
979              self.current.add_widget(foot)
980              self.push(foot)
981  
982              # check if its autonumbered
983              auto = ''
984              if 'auto' in node.attributes:
985                  auto = node.attributes['auto']
986  
987              # auto is either 1(int) or '*'
988              if auto == 1:
989                  self.footnotes['autonum'] += 1
990                  name = str(self.footnotes['autonum'])
991                  node_id = node.attributes['ids'][0]
992              elif auto == '*':
993                  autosym = self.footnotes['autosym']
994                  name = self.footlist[
995                      autosym % 10
996                  ] * (int(autosym / 10) + 1)
997                  self.footnotes['autosym'] += 1
998                  node_id = node.attributes['ids'][0]
999              else:
1000                 # can have multiple refs:
1001                 # [8] (1, 2) Footnote ref
1002                 name = node.attributes['names'][0]
1003                 node_id = node['ids'][0]
1004 
1005             # we can have a footnote without any link or ref
1006             # .. [1] Empty footnote
1007             link = self.root.refs_assoc.get(name, '')
1008 
1009             # handle no refs
1010             ref = self.root.refs_assoc.get('backref' + name, '')
1011 
1012             # colorize only with refs
1013             colorized = self.colorize(name, 'link') if ref else name
1014 
1015             # has no refs
1016             if not ref:
1017                 text = '&bl;%s&br;' % (colorized)
1018             # list of refs
1019             elif ref and isinstance(ref, list):
1020                 ref_block = [
1021                     '[ref=%s][u]%s[/u][/ref]' % (r, i + 1)
1022                     for i, r in enumerate(ref)
1023                 ]
1024                 # [1] ( 1, 2, ...) Footnote
1025                 self.foot_refblock = ''.join([
1026                     '[i]( ', ', '.join(ref_block), ' )[/i]'
1027                 ])
1028 
1029                 text = '[anchor=%s]&bl;%s&br;' % (
1030                     node['ids'][0], colorized
1031                 )
1032             # single ref
1033             else:
1034                 text = '[anchor=%s][ref=%s]&bl;%s&br;[/ref]' % (
1035                     node['ids'][0], ref, colorized
1036                 )
1037 
1038             name = RstFootName(
1039                 document=self.root,
1040                 text=text,
1041             )
1042             self.current.add_widget(name)
1043             # give it anchor + event manually
1044             self.root.add_anchors(name)
1045             name.bind(on_ref_press=self.root.on_ref_press)

1047         elif cls is nodes.footnote_reference:
1048             self.text += '&bl;'
1049             text = ''
1050             name = ''
1051 
1052             # check if its autonumbered
1053             auto = ''
1054             if 'auto' in node.attributes:
1055                 auto = node.attributes['auto']
1056 
1057             # auto is either 1(int) or '*'
1058             if auto == 1:
