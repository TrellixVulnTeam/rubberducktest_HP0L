1    """Utilities for the win32 Performance Data Helper module 
2     
3    Example: 
4      To get a single bit of data: 
5      >>> import win32pdhutil 
6      >>> win32pdhutil.GetPerformanceAttributes("Memory", "Available Bytes") 
7      6053888 
8      >>> win32pdhutil.FindPerformanceAttributesByName("python", counter="Virtual Bytes") 
9      [22278144] 
10    
11     First example returns data which is not associated with any specific instance. 
12    
13     The second example reads data for a specific instance - hence the list return - 
14     it would return one result for each instance of Python running. 
15    
16     In general, it can be tricky finding exactly the "name" of the data you wish to query. 
17     Although you can use <om win32pdh.EnumObjectItems>(None,None,(eg)"Memory", -1) to do this, 
18     the easiest way is often to simply use PerfMon to find out the names. 
19   """
20

26   # see http://support.microsoft.com/default.aspx?scid=http://support.microsoft.com:80/support/kb/articles/Q287/1/59.asp&NoWebContent=1
27   # Build a map of english_counter_name: counter_id
28   counter_english_map = {}

31   def find_pdh_counter_localized_name(english_name, machine_name=None):
32       if not counter_english_map:
33           import win32api, win32con
34   
35           counter_reg_value = win32api.RegQueryValueEx(
36               win32con.HKEY_PERFORMANCE_DATA, "Counter 009"
37           )
38           counter_list = counter_reg_value[0]
39           for i in range(0, len(counter_list) - 1, 2):
40               try:
41                   counter_id = int(counter_list[i])
42               except ValueError:
43                   continue
44               counter_english_map[counter_list[i + 1].lower()] = counter_id
45       return win32pdh.LookupPerfNameByIndex(
46           machine_name, counter_english_map[english_name.lower()]
47       )

50   def GetPerformanceAttributes(
51       object, counter, instance=None, inum=-1, format=win32pdh.PDH_FMT_LONG, machine=None
52   ):
53       # NOTE: Many counters require 2 samples to give accurate results,
54       # including "% Processor Time" (as by definition, at any instant, a
55       # thread's CPU usage is either 0 or 100).  To read counters like this,
56       # you should copy this function, but keep the counter open, and call
57       # CollectQueryData() each time you need to know.
58       # See http://support.microsoft.com/default.aspx?scid=kb;EN-US;q262938
59       # and http://msdn.microsoft.com/library/en-us/dnperfmo/html/perfmonpt2.asp
60       # My older explanation for this was that the "AddCounter" process forced
61       # the CPU to 100%, but the above makes more sense :)
62       path = win32pdh.MakeCounterPath((machine, object, instance, None, inum, counter))
63       hq = win32pdh.OpenQuery()
64       try:
65           hc = win32pdh.AddCounter(hq, path)
66           try:
67               win32pdh.CollectQueryData(hq)
68               type, val = win32pdh.GetFormattedCounterValue(hc, format)
69               return val
70           finally:
71               win32pdh.RemoveCounter(hc)
72       finally:
73           win32pdh.CloseQuery(hq)
74

76   def FindPerformanceAttributesByName(
77       instanceName,
78       object=None,
79       counter=None,
80       format=win32pdh.PDH_FMT_LONG,
81       machine=None,
82       bRefresh=0,
83   ):
84       """Find peformance attributes by (case insensitive) instance name. 
85    
86       Given a process name, return a list with the requested attributes. 
87       Most useful for returning a tuple of PIDs given a process name. 
88       """
89       if object is None:
90           object = find_pdh_counter_localized_name("Process", machine)
91       if counter is None:
92           counter = find_pdh_counter_localized_name("ID Process", machine)
93       if bRefresh:  # PDH docs say this is how you do a refresh.
94           win32pdh.EnumObjects(None, machine, 0, 1)
95       instanceName = instanceName.lower()
96       items, instances = win32pdh.EnumObjectItems(None, None, object, -1)
97       # Track multiple instances.
98       instance_dict = {}
99       for instance in instances:
100          try:
101              instance_dict[instance] = instance_dict[instance] + 1
102          except KeyError:
103              instance_dict[instance] = 0
104  
105      ret = []
106      for instance, max_instances in instance_dict.items():
107          for inum in range(max_instances + 1):
108              if instance.lower() == instanceName:
109                  ret.append(
110                      GetPerformanceAttributes(
111                          object, counter, instance, inum, format, machine
112                      )
113                  )
114      return ret

117  def ShowAllProcesses():
118      object = find_pdh_counter_localized_name("Process")
119      items, instances = win32pdh.EnumObjectItems(
120          None, None, object, win32pdh.PERF_DETAIL_WIZARD
121      )
122      # Need to track multiple instances of the same name.
123      instance_dict = {}
124      for instance in instances:
125          try:
126              instance_dict[instance] = instance_dict[instance] + 1
127          except KeyError:
128              instance_dict[instance] = 0
129  
130      # Bit of a hack to get useful info.
131      items = [find_pdh_counter_localized_name("ID Process")] + items[:5]
132      print("Process Name", ",".join(items))
133      for instance, max_instances in instance_dict.items():
134          for inum in range(max_instances + 1):
135              hq = win32pdh.OpenQuery()
136              hcs = []
137              for item in items:
138                  path = win32pdh.MakeCounterPath(
139                      (None, object, instance, None, inum, item)
140                  )
141                  hcs.append(win32pdh.AddCounter(hq, path))
142              win32pdh.CollectQueryData(hq)
143              # as per http://support.microsoft.com/default.aspx?scid=kb;EN-US;q262938, some "%" based
144              # counters need two collections
145              time.sleep(0.01)
146              win32pdh.CollectQueryData(hq)
147              print("%-15s\t" % (instance[:15]), end=" ")
148              for hc in hcs:
149                  type, val = win32pdh.GetFormattedCounterValue(hc, win32pdh.PDH_FMT_LONG)
150                  print("%5d" % (val), end=" ")
151                  win32pdh.RemoveCounter(hc)
152              print()
153              win32pdh.CloseQuery(hq)

160  def BrowseCallBackDemo(counters):
161      ## BrowseCounters can now return multiple counter paths
162      for counter in counters:
163          (
164              machine,
165              object,
166              instance,
167              parentInstance,
168              index,
169              counterName,
170          ) = win32pdh.ParseCounterPath(counter)
171  
172          result = GetPerformanceAttributes(
173              object, counterName, instance, index, win32pdh.PDH_FMT_DOUBLE, machine
174          )
175          print("Value of '%s' is" % counter, result)
176          print(
177              "Added '%s' on object '%s' (machine %s), instance %s(%d)-parent of %s"
178              % (counterName, object, machine, instance, index, parentInstance)
179          )
180      return 0
