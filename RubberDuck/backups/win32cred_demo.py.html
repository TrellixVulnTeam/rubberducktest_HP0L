win32cred_demo.py 
1    """ 
2    Demonstrates prompting for credentials, saving, and loggging on with marshalled credential. 
3    Also shows how to load user's profile 
4    """
5    
6    import win32net, win32security, win32api, win32con
7    import win32profile, win32cred
8    
9    ## Prompt for a username/pwd for local computer
10   uiinfo = {
11       "MessageText": "Enter credentials for local machine",
12       "CaptionText": "win32cred_demo.py",
13   }
14   target, pwd, save = win32cred.CredUIPromptForCredentials(
15       TargetName=win32api.GetComputerName(),
16       AuthError=0,
17       Flags=win32cred.CREDUI_FLAGS_DO_NOT_PERSIST
18       | win32cred.CREDUI_FLAGS_SHOW_SAVE_CHECK_BOX,
19       Save=False,
20       UiInfo=uiinfo,
21   )
22   
23   attrs = [
24       {"Keyword": "attr1", "Flags": 0, "Value": "unicode data"},
25       {"Keyword": "attr2", "Flags": 0, "Value": b"character data"},
26   ]
27   cred = {
28       "Comment": "Created by win32cred_demo.py",
29       "UserName": target,
30       "TargetAlias": None,
31       "TargetName": target,
32       "CredentialBlob": pwd,
33       "Flags": win32cred.CRED_FLAGS_USERNAME_TARGET,
34       "Persist": win32cred.CRED_PERSIST_ENTERPRISE,
35       "Type": win32cred.CRED_TYPE_DOMAIN_PASSWORD,
36       "Attributes": attrs,
37   }
38   win32cred.CredWrite(cred)
39   pwd = None
40   print(win32cred.CredRead(target, win32cred.CRED_TYPE_DOMAIN_PASSWORD))
41   
42   ## Marshal saved credential and use it to log on
43   mc = win32cred.CredMarshalCredential(win32cred.UsernameTargetCredential, target)
44   
45   # As of pywin32 301 this no longer works for markh and unclear when it stopped, or
46   # even if it ever did! # Fails in Python 2.7 too, so not a 3.x regression.
47   try:
48       th = win32security.LogonUser(
49           mc,
50           None,
51           "",
52           win32con.LOGON32_LOGON_INTERACTIVE,
53           win32con.LOGON32_PROVIDER_DEFAULT,
54       )
55       win32security.ImpersonateLoggedOnUser(th)
56       print("GetUserName:", win32api.GetUserName())
57       win32security.RevertToSelf()
58   
59       ## Load user's profile.  (first check if user has a roaming profile)
60       username, domain = win32cred.CredUIParseUserName(target)
61       user_info_4 = win32net.NetUserGetInfo(None, username, 4)
62       profilepath = user_info_4["profile"]
63       ## LoadUserProfile apparently doesn't like an empty string
64       if not profilepath:
65           profilepath = None
66   
67       ## leave Flags in since 2.3 still chokes on some types of optional keyword args
68       hk = win32profile.LoadUserProfile(
69           th, {"UserName": username, "Flags": 0, "ProfilePath": profilepath}
70       )
71       ## Get user's environment variables in a form that can be passed to win32process.CreateProcessAsUser
72       env = win32profile.CreateEnvironmentBlock(th, False)
73   
74       ## Cleanup should probably be in a finally block
75       win32profile.UnloadUserProfile(th, hk)
76       th.Close()
77   except win32security.error as exc:
78       print("Failed to login for some reason", exc)
