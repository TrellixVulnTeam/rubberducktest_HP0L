1    # Dump lots of info about BITS jobs.
2    from win32com.bits import bits
3    import pythoncom
4    
5    states = dict(
6        [
7            (val, (name[13:]))
8            for name, val in vars(bits).items()
9            if name.startswith("BG_JOB_STATE_")
10       ]
11   )
12   
13   job_types = dict(
14       [
15           (val, (name[12:]))
16           for name, val in vars(bits).items()
17           if name.startswith("BG_JOB_TYPE_")
18       ]
19   )
20   
21   bcm = pythoncom.CoCreateInstance(
22       bits.CLSID_BackgroundCopyManager,
23       None,
24       pythoncom.CLSCTX_LOCAL_SERVER,
25       bits.IID_IBackgroundCopyManager,
26   )
27   
28   try:
29       enum = bcm.EnumJobs(bits.BG_JOB_ENUM_ALL_USERS)
30   except pythoncom.error:
31       print("Failed to get jobs for all users - trying for current user")
32       enum = bcm.EnumJobs(0)
33   
34   for job in enum:
35       print("Job:", job.GetDisplayName())
36       print("Description:", job.GetDescription())
37       print("Id:", job.GetId())
38       print("State:", states.get(job.GetState()))
39       print("Type:", job_types.get(job.GetType()))
40       print("Owner:", job.GetOwner())
41       print("Errors:", job.GetErrorCount())
42       print("Created/Modified/Finished times:", [str(t) for t in job.GetTimes()])
43       bytes_tot, bytes_xf, files_tot, files_xf = job.GetProgress()
44       print("Bytes: %d complete of %d total" % (bytes_xf, bytes_tot))
45       print("Files: %d complete of %d total" % (files_xf, files_tot))
46       for f in job.EnumFiles():
47           bytes, total, done = f.GetProgress()
48           print("  Remote:", f.GetRemoteName())
49           print("  Local:", f.GetLocalName())
50           print("  Progress: %d of %d bytes - completed=%s)" % (bytes, total, done))
51           print()
52       print()
