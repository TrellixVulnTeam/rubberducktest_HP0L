8    from kivy.base import EventLoop
9    from kivy.input.motionevent import MotionEvent
10   from kivy.uix.relativelayout import RelativeLayout

12   
13   # https://gist.github.com/tito/f111b6916aa6a4ed0851
14   # subclass for touch event in unit test
15   class UTMotionEvent(MotionEvent):
16   
17       def depack(self, args):
18           self.is_touch = True
19           self.sx = args['x']
20           self.sy = args['y']
21           self.profile = ['pos']
22           super(UTMotionEvent, self).depack(args)
23   
24   
25   class UixRelativeLayoutTest(unittest.TestCase):
26   
27       def test_relativelayout_on_touch_move(self):
28           EventLoop.ensure_window()
29           rl = RelativeLayout()
30           EventLoop.window.add_widget(rl)
31           touch = UTMotionEvent("unittest", 1, {"x": .5, "y": .5})
32           EventLoop.post_dispatch_input("begin", touch)
33           touch.move({"x": .6, "y": .4})
34           EventLoop.post_dispatch_input("update", touch)
35           EventLoop.post_dispatch_input("end", touch)
36   
37       def test_relativelayout_coordinates(self):
38           EventLoop.ensure_window()
39           rl = RelativeLayout(pos=(100, 100))
40           EventLoop.window.add_widget(rl)  # do_layout() called
41           self.assertEqual(rl.to_parent(50, 50), (150, 150))
42           self.assertEqual(rl.to_local(50, 50), (-50, -50))
