438      """ 
439      `RSL <http://en.wikipedia.org/wiki/RAISE>`_ is the formal specification 
440      language used in RAISE (Rigorous Approach to Industrial Software Engineering)

442   
443      .. versionadded:: 2.0

445      name = 'RSL'
446      aliases = ['rsl']
447      filenames = ['*.rsl']
448      mimetypes = ['text/rsl']
449  
450      flags = re.MULTILINE | re.DOTALL
451  
452      tokens = {
453          'root': [
454              (words((
455                  'Bool', 'Char', 'Int', 'Nat', 'Real', 'Text', 'Unit', 'abs',
456                  'all', 'always', 'any', 'as', 'axiom', 'card', 'case', 'channel',
457                  'chaos', 'class', 'devt_relation', 'dom', 'elems', 'else', 'elif',
458                  'end', 'exists', 'extend', 'false', 'for', 'hd', 'hide', 'if',
459                  'in', 'is', 'inds', 'initialise', 'int', 'inter', 'isin', 'len',
460                  'let', 'local', 'ltl_assertion', 'object', 'of', 'out', 'post',
461                  'pre', 'read', 'real', 'rng', 'scheme', 'skip', 'stop', 'swap',
462                  'then', 'theory', 'test_case', 'tl', 'transition_system', 'true',
463                  'type', 'union', 'until', 'use', 'value', 'variable', 'while',
464                  'with', 'write', '~isin', '-inflist', '-infset', '-list',
465                  '-set'), prefix=r'\b', suffix=r'\b'),
466               Keyword),
467              (r'(variable|value)\b', Keyword.Declaration),
468              (r'--.*?\n', Comment),
469              (r'<:.*?:>', Comment),
470              (r'\{!.*?!\}', Comment),
471              (r'/\*.*?\*/', Comment),
472              (r'^[ \t]*([\w]+)[ \t]*:[^:]', Name.Function),
473              (r'(^[ \t]*)([\w]+)([ \t]*\([\w\s,]*\)[ \t]*)(is|as)',
474               bygroups(Text, Name.Function, Text, Keyword)),
475              (r'\b[A-Z]\w*\b', Keyword.Type),
476              (r'(true|false)\b', Keyword.Constant),
477              (r'".*"', String),
478              (r'\'.\'', String.Char),
479              (r'(><|->|-m->|/\\|<=|<<=|<\.|\|\||\|\^\||-~->|-~m->|\\/|>=|>>|'
480               r'\.>|\+\+|-\\|<->|=>|:-|~=|\*\*|<<|>>=|\+>|!!|\|=\||#)',
481               Operator),
482              (r'[0-9]+\.[0-9]+([eE][0-9]+)?[fd]?', Number.Float),

484              (r'[0-9]+', Number.Integer),
485              (r'.', Text),
486          ],
487      }

489      def analyse_text(text):
490          """ 
491          Check for the most common text in the beginning of a RSL file. 
492          """
493          if re.search(r'scheme\s*.*?=\s*class\s*type', text, re.I) is not None:
494              return 1.0

854      """ 
855      Lexer for `Snowball <http://snowballstem.org/>`_ source code. 
856   
857      .. versionadded:: 2.2 
858      """
859  
860      name = 'Snowball'
861      aliases = ['snowball']
862      filenames = ['*.sbl']
863  
864      _ws = r'\n\r\t '
865  
866      def __init__(self, **options):
867          self._reset_stringescapes()
868          ExtendedRegexLexer.__init__(self, **options)
869  
870      def _reset_stringescapes(self):
871          self._start = "'"
872          self._end = "'"
873  
874      def _string(do_string_first):
875          def callback(lexer, match, ctx):
876              s = match.start()
877              text = match.group()
878              string = re.compile(r'([^%s]*)(.)' % re.escape(lexer._start)).match
879              escape = re.compile(r'([^%s]*)(.)' % re.escape(lexer._end)).match
880              pos = 0
881              do_string = do_string_first
882              while pos < len(text):
883                  if do_string:
884                      match = string(text, pos)
885                      yield s + match.start(1), String.Single, match.group(1)
886                      if match.group(2) == "'":
887                          yield s + match.start(2), String.Single, match.group(2)
888                          ctx.stack.pop()
889                          break
890                      yield s + match.start(2), String.Escape, match.group(2)
891                      pos = match.end()
892                  match = escape(text, pos)
893                  yield s + match.start(), String.Escape, match.group()
894                  if match.group(2) != lexer._end:
895                      ctx.stack[-1] = 'escape'
896                      break
897                  pos = match.end()
898                  do_string = True
899              ctx.pos = s + match.end()
900          return callback
901  
902      def _stringescapes(lexer, match, ctx):
903          lexer._start = match.group(3)
904          lexer._end = match.group(5)
905          return bygroups(Keyword.Reserved, Text, String.Escape, Text,
906                          String.Escape)(lexer, match, ctx)
