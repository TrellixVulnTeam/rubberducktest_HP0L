texturecompress.py 
1    ''' 
2    Texture compression tool 
3    ======================== 
4     
5    This tool is designed to compress images into: 
6     
7    - PVRTC (PowerVR Texture Compression), mostly iOS devices 
8    - ETC1 (Ericson compression), working on all GLES2/Android devices 
9     
10   Usage 
11   ----- 
12    
13   In order to compress a texture:: 
14    
15       texturecompress.py [--dir <directory>] <format> <image.png> 
16    
17   This will create a `image.tex` file with a json header that contains all the 
18   image information and the compressed data. 
19    
20   TODO 
21   ---- 
22    
23   Support more format, such as: 
24    
25   - S3TC (already supported in Kivy) 
26   - DXT1 (already supported in Kivy) 
27   '''
28   
29   import json
30   from struct import pack
31   from pprint import pprint
32   from subprocess import Popen
33   from PIL import Image
34   from argparse import ArgumentParser
35   from sys import exit
36   from os.path import join, exists, dirname, basename
37   from os import environ, unlink
38   
39   
40   class Tool(object):
41       def __init__(self, options):
42           super(Tool, self).__init__()
43           self.options = options
44           self.source_fn = options.image
45           self.dest_dir = options.dir or dirname(options.image)
46   
47       @property
48       def tex_fn(self):
49           fn = basename(self.source_fn).rsplit('.', 1)[0] + '.tex'
50           return join(self.dest_dir, fn)
51   
52       def compress(self):
53           pass
54   
55       def nearest_pow2(self, v):
56           # Credits: Sean Anderson
57           v -= 1
58           v |= v >> 1
59           v |= v >> 2
60           v |= v >> 4
61           v |= v >> 8
62           v |= v >> 16
63           return v + 1
64   
65       def runcmd(self, cmd):
66           print('Run: {}'.format(' '.join(cmd)))
67           Popen(cmd).communicate()
68   
69       def write_tex(self, data, fmt, image_size, texture_size, mipmap=False,
70               formatinfo=None):
71           infos = {
72               'datalen': len(data),
73               'image_size': image_size,
74               'texture_size': texture_size,
75               'mipmap': mipmap,
76               'format': fmt}
77           if formatinfo:
78               infos['formatinfo'] = formatinfo
79           header = json.dumps(infos, indent=0, separators=(',', ':'))
80           header = header.replace('\n', '')
81           with open(self.tex_fn, 'wb') as fd:
82               fd.write('KTEX')
83               fd.write(pack('I', len(header)))
84               fd.write(header)
85               fd.write(data)
86   
87           print('Done! Compressed texture written at {}'.format(self.tex_fn))
88           pprint(infos)
89   
90       @staticmethod
91       def run():
92           parser = ArgumentParser(
93                   description='Convert images to compressed texture')
94           parser.add_argument('--mipmap', type=bool, default=False,
95                   help='Auto generate mipmaps')
96           parser.add_argument('--dir', type=str, default=None,
97                   help='Output directory to generate the compressed texture')
98           parser.add_argument('format', type=str, choices=['pvrtc', 'etc1'],
99                   help='Format of the final texture')
100          parser.add_argument('image', type=str,
101                  help='Image filename')
102          args = parser.parse_args()
103  
104          if args.format == 'pvrtc':
105              PvrtcTool(args).compress()
106          elif args.format == 'etc1':
107              Etc1Tool(args).compress()
108          else:
109              print('Unknown compression format')
110              exit(1)
111

113  class Etc1Tool(Tool):
114      def __init__(self, options):
115          super(Etc1Tool, self).__init__(options)
116          self.etc1tool = None
117          self.locate_etc1tool()
118  
119      def locate_etc1tool(self):
120          search_directories = [environ.get('ANDROIDSDK', '/')]
121          search_directories += environ.get('PATH', '').split(':')
122          for directory in search_directories:
123              fn = join(directory, 'etc1tool')
124              if not exists(fn):
125                  fn = join(directory, 'tools', 'etc1tool')
126                  if not exists(fn):
127                      continue
128              print('Found texturetool at {}'.format(directory))
129              self.etc1tool = fn
130              return
131  
132          if self.etc1tool is None:
133              print('Error: Unable to locate "etc1tool".\n'
134                    'Make sure that "etc1tool" is available in your PATH.\n'
135                    'Or export the path of your Android SDK to ANDROIDSDK')
136              exit(1)
137  
138      def compress(self):
139          # 1. open the source image, and get the dimensions
140          image = Image.open(self.source_fn)
141          w, h = image.size
142          print('Image size is {}x{}'.format(*image.size))
143  
144          # 2. search the nearest 2^
145          w2 = self.nearest_pow2(w)
146          h2 = self.nearest_pow2(h)
147          print('Nearest power-of-2 size is {}x{}'.format(w2, h2))
148  
149          # 3. invoke etc1tool
150          raw_tex_fn = self.tex_fn + '.raw'
151          cmd = [self.etc1tool, self.source_fn, '--encodeNoHeader', '-o',
152                 raw_tex_fn]
153          try:
154              self.runcmd(cmd)
155              with open(raw_tex_fn, 'rb') as fd:
156                  data = fd.read()
157          finally:
158              if exists(raw_tex_fn):
159                  unlink(raw_tex_fn)
160  
161          # 5. write texture info
162          self.write_tex(data, 'etc1_rgb8', (w, h), (w2, h2),
163                          self.options.mipmap)

166  class PvrtcTool(Tool):
167      def __init__(self, options):
168          super(PvrtcTool, self).__init__(options)
169          self.texturetool = None
170          self.locate_texturetool()
171  
172      def locate_texturetool(self):
173          search_directories = [
174              ('/Applications/Xcode.app/Contents/Developer/Platforms/'
175               'iPhoneOS.platform/Developer/usr/bin/'),
176              '/Developer/Platforms/iPhoneOS.platform/Developer/usr/bin/']
177          search_directories += environ.get('PATH', '').split(':')
178  
179          for directory in search_directories:
180              fn = join(directory, 'texturetool')
181              if not exists(fn):
182                  continue
183              print('Found texturetool at {}'.format(directory))
184              self.texturetool = fn
185              return
186  
187          print('Error: Unable to locate "texturetool".\n'
188                'Please install the iPhone SDK, or the PowerVR SDK.\n'
189                'Then make sure that "texturetool" is available in your PATH.')
190          exit(1)
191  
192      def compress(self):
193          # 1. open the source image, and get the dimensions
194          image = Image.open(self.source_fn)
195          w, h = image.size
196          print('Image size is {}x{}'.format(*image.size))
197  
198          # 2. search the nearest 2^
199          w2 = self.nearest_pow2(w)
200          h2 = self.nearest_pow2(h)
201          print('Nearest power-of-2 size is {}x{}'.format(w2, h2))
202  
203          # 3. for PVR, the image MUST be a square. use the bigger size then
204          s2 = max(w2, h2)
205          print('PVR need a square image, the texture will be {0}x{0}'
206                  .format(s2))
207  
208          ext = self.source_fn.rsplit('.', 1)[-1]
209          tmpfile = '/tmp/ktexturecompress.{}'.format(ext)
210          image = image.resize((s2, s2))
211          image.save(tmpfile)
212  
213          # 4. invoke texture tool
214          raw_tex_fn = self.tex_fn + '.raw'
215          cmd = [self.texturetool]
216          if self.options.mipmap:
217              cmd += ['-m']
218          cmd += ['-e', 'PVRTC', '-o', raw_tex_fn, '-f', 'RAW', tmpfile]
219          try:
220              self.runcmd(cmd)
221              with open(raw_tex_fn, 'rb') as fd:
222                  data = fd.read()
223          finally:
224              if exists(raw_tex_fn):
225                  unlink(raw_tex_fn)
226  
227          # 5. write texture info
228          self.write_tex(data, 'pvrtc_rgba4', (w, h), (s2, s2),
229                         self.options.mipmap)
230  
231  
232  if __name__ == '__main__':
233      Tool.run()
