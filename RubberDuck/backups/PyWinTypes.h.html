31   // Helpers for our modules.
32   // Some macros to help the pywin32 modules co-exist in py2x and py3k.
33   // Creates and initializes local variables called 'module' and 'dict'.
34   
35   // Maybe these should all be removed - they existed to help in the py2->3
36   // transition.
37   // On one hand: the code would be cleaner if they were all just re-inlined?
38   // On the other: high confidence everything uses the exact same patterns?
39   // (Regardless, *some*, eg, PYWIN_MODULE_INIT_RETURN_* should be re-inlined!)
40   
41   // Use to define the function itself (ie, its name, linkage, params)
42   #define PYWIN_MODULE_INIT_FUNC(module_name) extern "C" __declspec(dllexport) PyObject *PyInit_##module_name(void)
43   
44   // If the module needs to early-exit on an error condition.
45   #define PYWIN_MODULE_INIT_RETURN_ERROR return NULL;
46   
47   // When the module has successfully initialized.
48   #define PYWIN_MODULE_INIT_RETURN_SUCCESS return module;
49   
50   // To setup the module object itself and the module's dictionary.
51   #define PYWIN_MODULE_INIT_PREPARE(module_name, functions, docstring)                                        \
52       PyObject *dict, *module;                                                                                \
53       static PyModuleDef module_name##_def = {PyModuleDef_HEAD_INIT, #module_name, docstring, -1, functions}; \
54       if (PyWinGlobals_Ensure() == -1)                                                                        \
55           return NULL;                                                                                        \
56       if (!(module = PyModule_Create(&module_name##_def)))                                                    \
57           return NULL;                                                                                        \
58       if (!(dict = PyModule_GetDict(module)))                                                                 \
59           return NULL;

156  // Automatically freed WCHAR that can be used anywhere WCHAR * is required
157  class TmpWCHAR {
158     public:
159      WCHAR *tmp;
160      TmpWCHAR() { tmp = NULL; }
161      TmpWCHAR(WCHAR *t) { tmp = t; }
162      WCHAR *operator=(WCHAR *t)
163      {
164          PyWinObject_FreeWCHAR(tmp);
165          tmp = t;
166          return t;
167      }
168      WCHAR **operator&() { return &tmp; }
169      boolean operator==(WCHAR *t) { return tmp == t; }
170      operator WCHAR *() { return tmp; }
171      ~TmpWCHAR() { PyWinObject_FreeWCHAR(tmp); }
172  };

515  // WARNING - NEVER EVER USE new() ON THIS CLASS
516  // This class can be used as a local variable, typically in a Python/C
517  // function, and can be passed whereever a TCHAR/WCHAR is expected.
518  // Typical Usage:
519  // PyWin_AutoFreeBstr arg;
520  // PyArg_ParseTuple("O", &obStr);
521  // PyWinObject_AsAutoFreeBstr(obStr, &arg);
522  // CallTheFunction(arg); // Will correctly pass BSTR/OLECHAR
523  // -- when the function goes out of scope, the string owned by "arg" will
524  // -- automatically be freed.
525  // ----------------------------------------------------------------------
526  class PYWINTYPES_EXPORT PyWin_AutoFreeBstr {
527     public:
528      PyWin_AutoFreeBstr(BSTR bstr = NULL);
529      ~PyWin_AutoFreeBstr();
530      void SetBstr(BSTR bstr);
531      operator BSTR() { return m_bstr; }
532  
533     private:
534      BSTR m_bstr;
535  };

537  inline BOOL PyWinObject_AsAutoFreeBstr(PyObject *stringObject, PyWin_AutoFreeBstr *pResult, BOOL bNoneOK = FALSE)
538  {
539      if (bNoneOK && stringObject == Py_None) {
540          pResult->SetBstr(NULL);
541          return TRUE;
542      }
543      BSTR bs;
544      if (!PyWinObject_AsBstr(stringObject, &bs, bNoneOK))
545          return FALSE;
546      pResult->SetBstr(bs);
547      return TRUE;
548  }

557  // We have 2 discrete locks in use (when no free-threaded is used, anyway).
558  // The first type of lock is the global Python lock.  This is the standard lock
559  // in use by Python, and must be used as documented by Python.  Specifically, no
560  // 2 threads may _ever_ call _any_ Python code (including INCREF/DECREF) without
561  // first having this thread lock.
562  //
563  // The second type of lock is a "global framework lock".  This lock is simply a
564  // critical section, and used whenever 2 threads of C code need access to global
565  // data.  This is different than the Python lock - this lock is used when no Python
566  // code can ever be called by the threads, but the C code still needs thread-safety.
567  
568  // We also supply helper classes which make the usage of these locks a one-liner.
569  
570  // The "framework" lock, implemented as a critical section.
571  PYWINTYPES_EXPORT void PyWin_AcquireGlobalLock(void);
572  PYWINTYPES_EXPORT void PyWin_ReleaseGlobalLock(void);

625  class CEnterLeavePython {
626     public:
627      CEnterLeavePython() : released(TRUE) { acquire(); }
628      void acquire(void)
629      {
630          if (!released)
631              return;
632          state = PyGILState_Ensure();
633          released = FALSE;
634      }
635      ~CEnterLeavePython() { release(); }
636      void release(void)
637      {
638          if (!released) {
639              PyGILState_Release(state);
640              released = TRUE;
641          }
642      }
643  
644     private:
645      PyGILState_STATE state;
646      BOOL released;
647  };

670      TmpPyObject(PyObject *ob) { tmp = ob; }
