1    '''Pygments lexer for kv language 
2    '''
3    from pygments.lexer import RegexLexer, bygroups, using
4    from pygments.lexers.agile import PythonLexer
5    from pygments import highlight
6    from pygments.token import Comment, Text, Name, Punctuation, Operator
7    from pygments.formatters import get_formatter_by_name
8    import sys
9    
10   
11   class KivyLexer(RegexLexer):
12       name = 'Kivy'
13       aliases = ['kivy', 'kv']
14       filenames = ['*.kv']
15       tokens = {
16           'root': [
17               (r'#:.*?$', Comment.Preproc),
18               (r'#.*?$', using(PythonLexer)),
19               (r'\s+', Text),
20               (r'<.+>', Name.Namespace),
21               (r'(\[)(\s*)(.*?)(\s*)(@)',
22                   bygroups(Punctuation, Text, Name.Class, Text, Operator),
23                   'classList'),
24               (r'[A-Za-z][A-Za-z0-9]*$', Name.Attribute),
25               (r'(.*?)(\s*)(:)(\s*)$',
26                   bygroups(Name.Class, Text, Punctuation, Text)),
27               (r'(.*?)(\s*)(:)(\s*)(.*?)$',
28                   bygroups(Name.Attribute, Text, Punctuation, Text,
29                   using(PythonLexer))),
30               (r'[^:]+?$', using(PythonLexer))],
31           'classList': [
32               (r'(,)(\s*)([A-Z][A-Za-z0-9]*)',
33                   bygroups(Punctuation, Text, Name.Class)),
34               (r'(\+)(\s*)([A-Z][A-Za-z0-9]*)',
35                   bygroups(Operator, Text, Name.Class)),
36               (r'\s+', Text),
37               (r'[A-Z][A-Za-z0-9]*', Name.Class),
38               (r'\]', Punctuation, '#pop')]}
39   
40   
41   if __name__ == '__main__':
42       ''' This lexer will highlight .kv file. The first argument is the source 
43       file, the second argument is the format of the destination and the third 
44       argument is the output filename 
45       '''
46       if len(sys.argv) != 4:
47           raise Exception('Three arguments expected, found %s' %
48               (len(sys.argv) - 1))
49       k = KivyLexer()
50       with open(sys.argv[1], 'r') as fd:
51           with open(sys.argv[3], 'w') as out:
52               highlight(fd.read(), k, get_formatter_by_name(sys.argv[2]), out)
