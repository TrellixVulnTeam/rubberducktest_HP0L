1    ''' 
2    ModalView 
3    ========= 
4     
5    .. versionadded:: 1.4.0 
6     
7    The :class:`ModalView` widget is used to create modal views. By default, the 
8    view will cover the whole "parent" window. 
9     
10   Remember that the default size of a Widget is size_hint=(1, 1). If you don't 
11   want your view to be fullscreen, either use size hints with values lower than 
12   1 (for instance size_hint=(.8, .8)) or deactivate the size_hint and use fixed 
13   size attributes. 
14    
15   Examples 
16   -------- 
17    
18   Example of a simple 400x400 Hello world view:: 
19    
20       view = ModalView(size_hint=(None, None), size=(400, 400)) 
21       view.add_widget(Label(text='Hello world')) 
22    
23   By default, any click outside the view will dismiss it. If you don't 
24   want that, you can set :attr:`ModalView.auto_dismiss` to False:: 
25    
26       view = ModalView(auto_dismiss=False) 
27       view.add_widget(Label(text='Hello world')) 
28       view.open() 
29    
30   To manually dismiss/close the view, use the :meth:`ModalView.dismiss` method of 
31   the ModalView instance:: 
32    
33       view.dismiss() 
34    
35   Both :meth:`ModalView.open` and :meth:`ModalView.dismiss` are bindable. That 
36   means you can directly bind the function to an action, e.g. to a button's 
37   on_press :: 
38    
39       # create content and add it to the view 
40       content = Button(text='Close me!') 
41       view = ModalView(auto_dismiss=False) 
42       view.add_widget(content) 
43    
44       # bind the on_press event of the button to the dismiss function 
45       content.bind(on_press=view.dismiss) 
46    
47       # open the view 
48       view.open() 
49    
50    
51   ModalView Events 
52   ---------------- 
53    
54   There are two events available: `on_open` which is raised when the view is 
55   opening, and `on_dismiss` which is raised when the view is closed. 
56   For `on_dismiss`, you can prevent the view from closing by explictly returning 
57   True from your callback. :: 
58    
59       def my_callback(instance): 
60           print('ModalView', instance, 'is being dismissed, but is prevented!') 
61           return True 
62       view = ModalView() 
63       view.add_widget(Label(text='Hello world')) 
64       view.bind(on_dismiss=my_callback) 
65       view.open() 
66    
67    
68   .. versionchanged:: 1.5.0 
69       The ModalView can be closed by hitting the escape key on the 
70       keyboard if the :attr:`ModalView.auto_dismiss` property is True (the 
71       default). 
72    
73   '''

77   from kivy.logger import Logger
78   from kivy.animation import Animation
79   from kivy.uix.anchorlayout import AnchorLayout
80   from kivy.properties import StringProperty, BooleanProperty, ObjectProperty, \

84   class ModalView(AnchorLayout):
85       '''ModalView class. See module documentation for more information. 
86    
87       :Events: 
88           `on_pre_open`: 
89               Fired before the ModalView is opened. When this event is fired 
90               ModalView is not yet added to window. 
91           `on_open`: 
92               Fired when the ModalView is opened. 
93           `on_pre_dismiss`: 
94               Fired before the ModalView is closed. 
95           `on_dismiss`: 
96               Fired when the ModalView is closed. If the callback returns True, 
97               the dismiss will be canceled. 
98    
99       .. versionchanged:: 1.11.0 
100          Added events `on_pre_open` and `on_pre_dismiss`. 
101   
102      .. versionchanged:: 2.0.0 
103          Added property 'overlay_color'. 
104   
105      '''
106  
107      auto_dismiss = BooleanProperty(True)
108      '''This property determines if the view is automatically 
109      dismissed when the user clicks outside it. 
110   
111      :attr:`auto_dismiss` is a :class:`~kivy.properties.BooleanProperty` and 
112      defaults to True. 
113      '''
114  
115      attach_to = ObjectProperty(None)
116      '''If a widget is set on attach_to, the view will attach to the nearest 
117      parent window of the widget. If none is found, it will attach to the 
118      main/global Window. 
119   
120      :attr:`attach_to` is an :class:`~kivy.properties.ObjectProperty` and 
121      defaults to None. 
122      '''
123  
124      background_color = ColorProperty([1, 1, 1, 1])
125      '''Background color, in the format (r, g, b, a). 
126   
127      This acts as a *multiplier* to the texture colour. The default 
128      texture is grey, so just setting the background color will give 
129      a darker result. To set a plain color, set the 
130      :attr:`background_normal` to ``''``. 
131   
132      The :attr:`background_color` is a 
133      :class:`~kivy.properties.ColorProperty` and defaults to [1, 1, 1, 1]. 
134   
135      .. versionchanged:: 2.0.0 
136          Changed behavior to affect the background of the widget itself, not 
137          the overlay dimming. 
138          Changed from :class:`~kivy.properties.ListProperty` to 
139          :class:`~kivy.properties.ColorProperty`. 
140      '''
141  
142      background = StringProperty(
143          'atlas://data/images/defaulttheme/modalview-background')
144      '''Background image of the view used for the view background. 
145   
146      :attr:`background` is a :class:`~kivy.properties.StringProperty` and 
147      defaults to 'atlas://data/images/defaulttheme/modalview-background'. 
148      '''
149  
150      border = ListProperty([16, 16, 16, 16])
151      '''Border used for :class:`~kivy.graphics.vertex_instructions.BorderImage` 
152      graphics instruction. Used for the :attr:`background_normal` and the 
153      :attr:`background_down` properties. Can be used when using custom 
154      backgrounds. 
155   
156      It must be a list of four values: (bottom, right, top, left). Read the 
157      BorderImage instructions for more information about how to use it. 
158   
159      :attr:`border` is a :class:`~kivy.properties.ListProperty` and defaults to 
160      (16, 16, 16, 16). 
161      '''
162  
163      overlay_color = ColorProperty([0, 0, 0, .7])
164      '''Overlay color in the format (r, g, b, a). 
165      Used for dimming the window behind the modal view. 
166   
167      :attr:`overlay_color` is a :class:`~kivy.properties.ColorProperty` and 
168      defaults to [0, 0, 0, .7]. 
169   
170      .. versionadded:: 2.0.0 
171      '''
172  
173      # Internals properties used for graphical representation.
174  
175      _anim_alpha = NumericProperty(0)
176  
177      _anim_duration = NumericProperty(.1)
178  
179      _window = ObjectProperty(None, allownone=True, rebind=True)
180  
181      _touch_started_inside = None
182  
183      __events__ = ('on_pre_open', 'on_open', 'on_pre_dismiss', 'on_dismiss')
184  
185      def __init__(self, **kwargs):
186          self._parent = None
187          super(ModalView, self).__init__(**kwargs)
188  
189      def _search_window(self):
190          # get window to attach to
191          window = None
192          if self.attach_to is not None:
193              window = self.attach_to.get_parent_window()
194              if not window:
195                  window = self.attach_to.get_root_window()
196          if not window:
197              from kivy.core.window import Window
198              window = Window
199          return window
200  
201      def open(self, *largs, **kwargs):
202          '''Show the view window from the :attr:`attach_to` widget. If set, it 
203          will attach to the nearest window. If the widget is not attached to any 
204          window, the view will attach to the global 
205          :class:`~kivy.core.window.Window`. 
206   
207          When the view is opened, it will be faded in with an animation. If you 
208          don't want the animation, use:: 
209   
210              view.open(animation=False) 
211   
212          '''
213          if self._window is not None:
214              Logger.warning('ModalView: you can only open once.')
215              return
216          # search window
217          self._window = self._search_window()
218          if not self._window:
219              Logger.warning('ModalView: cannot open view, no window found.')
220              return
221          self.dispatch('on_pre_open')
222          self._window.add_widget(self)
223          self._window.bind(
224              on_resize=self._align_center,
225              on_keyboard=self._handle_keyboard)
226          self.center = self._window.center
227          self.fbind('center', self._align_center)
228          self.fbind('size', self._align_center)
229          if kwargs.get('animation', True):
230              a = Animation(_anim_alpha=1., d=self._anim_duration)
231              a.bind(on_complete=lambda *x: self.dispatch('on_open'))
232              a.start(self)
233          else:
234              self._anim_alpha = 1.
235              self.dispatch('on_open')
236  
237      def dismiss(self, *largs, **kwargs):
238          '''Close the view if it is open. If you really want to close the 
239          view, whatever the on_dismiss event returns, you can use the *force* 
240          argument: 
241          :: 
242   
243              view = ModalView() 
244              view.dismiss(force=True) 
245   
246          When the view is dismissed, it will be faded out before being 
247          removed from the parent. If you don't want animation, use:: 
248   
249              view.dismiss(animation=False) 
250   
251          '''
252          if self._window is None:
253              return
254          self.dispatch('on_pre_dismiss')
255          if self.dispatch('on_dismiss') is True:
256              if kwargs.get('force', False) is not True:
257                  return
258          if kwargs.get('animation', True):
259              Animation(_anim_alpha=0., d=self._anim_duration).start(self)
260          else:
261              self._anim_alpha = 0
262              self._real_remove_widget()
263  
264      def _align_center(self, *l):
265          if self._window:
266              self.center = self._window.center
267  
268      def on_touch_down(self, touch):
269          self._touch_started_inside = self.collide_point(*touch.pos)
270          if not self.auto_dismiss or self._touch_started_inside:
271              super(ModalView, self).on_touch_down(touch)
272          return True
273  
274      def on_touch_move(self, touch):
275          if not self.auto_dismiss or self._touch_started_inside:
276              super(ModalView, self).on_touch_move(touch)
277          return True
278  
279      def on_touch_up(self, touch):
280          # Explicitly test for False as None occurs when shown by on_touch_down
281          if self.auto_dismiss and self._touch_started_inside is False:
282              self.dismiss()
283          else:
284              super(ModalView, self).on_touch_up(touch)
285          self._touch_started_inside = None
286          return True
287  
288      def on__anim_alpha(self, instance, value):
289          if value == 0 and self._window is not None:
290              self._real_remove_widget()
291  
292      def _real_remove_widget(self):
293          if self._window is None:
294              return
295          self._window.remove_widget(self)
296          self._window.unbind(
297              on_resize=self._align_center,
298              on_keyboard=self._handle_keyboard)
299          self._window = None
300  
301      def on_pre_open(self):
302          pass
303  
304      def on_open(self):
305          pass
306  
307      def on_pre_dismiss(self):
308          pass
309  
310      def on_dismiss(self):
311          pass
312  
313      def _handle_keyboard(self, window, key, *largs):
314          if key == 27 and self.auto_dismiss:
315              self.dismiss()
316              return True
