4    import pythoncom
5    from win32com.shell import shell, shellcon
6    from win32com.server.policy import DesignatedWrapPolicy
7    
8    tsf_flags = list(
9        (k, v) for k, v in list(shellcon.__dict__.items()) if k.startswith("TSF_")
10   )
11   
12   
13   def decode_flags(flags):
14       if flags == 0:
15           return "TSF_NORMAL"
16       flag_txt = ""
17       for k, v in tsf_flags:
18           if flags & v:
19               if flag_txt:
20                   flag_txt = flag_txt + "|" + k
21               else:
22                   flag_txt = k
23       return flag_txt

26   class FileOperationProgressSink(DesignatedWrapPolicy):
27       _com_interfaces_ = [shell.IID_IFileOperationProgressSink]
28       _public_methods_ = [
29           "StartOperations",
30           "FinishOperations",
31           "PreRenameItem",
32           "PostRenameItem",
33           "PreMoveItem",
34           "PostMoveItem",
35           "PreCopyItem",
36           "PostCopyItem",
37           "PreDeleteItem",
38           "PostDeleteItem",
39           "PreNewItem",
40           "PostNewItem",
41           "UpdateProgress",
42           "ResetTimer",
43           "PauseTimer",
44           "ResumeTimer",
45       ]
46   
47       def __init__(self):
48           self._wrap_(self)
49   
50       def StartOperations(self):
51           print("StartOperations")
52   
53       def FinishOperations(self, Result):
54           print("FinishOperations: HRESULT ", Result)
55   
56       def PreRenameItem(self, Flags, Item, NewName):
57           print(
58               "PreRenameItem: Renaming "
59               + Item.GetDisplayName(shellcon.SHGDN_FORPARSING)
60               + " to "
61               + NewName
62           )
63   
64       def PostRenameItem(self, Flags, Item, NewName, hrRename, NewlyCreated):
65           if NewlyCreated is not None:
66               newfile = NewlyCreated.GetDisplayName(shellcon.SHGDN_FORPARSING)
67           else:
68               newfile = "not renamed, HRESULT " + str(hrRename)
69           print(
70               "PostRenameItem: renamed "
71               + Item.GetDisplayName(shellcon.SHGDN_FORPARSING)
72               + " to "
73               + newfile
74           )
75   
76       def PreMoveItem(self, Flags, Item, DestinationFolder, NewName):
77           print(
78               "PreMoveItem: Moving "
79               + Item.GetDisplayName(shellcon.SHGDN_FORPARSING)
80               + " to "
81               + DestinationFolder.GetDisplayName(shellcon.SHGDN_FORPARSING)
82               + "\\"
83               + str(NewName)
84           )
85   
86       def PostMoveItem(
87           self, Flags, Item, DestinationFolder, NewName, hrMove, NewlyCreated
88       ):
89           if NewlyCreated is not None:
90               newfile = NewlyCreated.GetDisplayName(shellcon.SHGDN_FORPARSING)
91           else:
92               newfile = "not copied, HRESULT " + str(hrMove)
93           print(
94               "PostMoveItem: Moved "
95               + Item.GetDisplayName(shellcon.SHGDN_FORPARSING)
96               + " to "
97               + newfile
98           )
99   
100      def PreCopyItem(self, Flags, Item, DestinationFolder, NewName):
101          if not NewName:
102              NewName = ""
103          print(
104              "PreCopyItem: Copying "
105              + Item.GetDisplayName(shellcon.SHGDN_FORPARSING)
106              + " to "
107              + DestinationFolder.GetDisplayName(shellcon.SHGDN_FORPARSING)
108              + "\\"
109              + NewName
110          )
111          print("Flags: ", decode_flags(Flags))
