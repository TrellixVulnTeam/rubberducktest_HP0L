35       def expand_word_event(self, event):
36           curinsert = self.text.index("insert")
37           curline = self.text.get("insert linestart", "insert lineend")
38           if not self.state:
39               words = self.getwords()
40               index = 0
41           else:
42               words, index, insert, line = self.state
43               if insert != curinsert or line != curline:
44                   words = self.getwords()

51           newword = words[index]

61       def getwords(self):
62           word = self.getprevword()
63           if not word:
64               return []
65           before = self.text.get("1.0", "insert wordstart")
66           wbefore = re.findall(r"\b" + word + r"\w+\b", before)
67           del before
68           after = self.text.get("insert wordend", "end")
69           wafter = re.findall(r"\b" + word + r"\w+\b", after)
70           del after
71           if not wbefore and not wafter:
72               return []
73           words = []
74           dict = {}
75           # search backwards through words before
76           wbefore.reverse()
77           for w in wbefore:
78               if dict.get(w):
79                   continue
80               words.append(w)
81               dict[w] = w
82           # search onwards through words after
83           for w in wafter:
84               if dict.get(w):
85                   continue
86               words.append(w)
87               dict[w] = w
88           words.append(word)
89           return words

91       def getprevword(self):
92           line = self.text.get("insert linestart", "insert")
93           i = len(line)
94           while i > 0 and line[i - 1] in self.wordchars:
95               i = i - 1
96           return line[i:]
