CreateFileTransacted_MiniVersion.py 
1    """ 
2    This demonstrates the creation of miniversions of a file during a transaction. 
3    The FSCTL_TXFS_CREATE_MINIVERSION control code saves any changes to a new 
4    miniversion (effectively a savepoint within a transaction). 
5    """
6    
7    import win32file, win32api, win32transaction, winerror
8    import win32con, winioctlcon
9    import struct
10   import os
11   from pywin32_testutil import str2bytes  # py3k-friendly helper
12   
13   
14   def demo():
15       """ 
16       Definition of buffer used with FSCTL_TXFS_CREATE_MINIVERSION: 
17       typedef struct _TXFS_CREATE_MINIVERSION_INFO{ 
18           USHORT StructureVersion; 
19           USHORT StructureLength; 
20           ULONG BaseVersion; 
21           USHORT MiniVersion;} 
22       """
23       buf_fmt = "HHLH0L"  ## buffer size must include struct padding
24       buf_size = struct.calcsize(buf_fmt)
25   
26       tempdir = win32api.GetTempPath()
27       tempfile = win32api.GetTempFileName(tempdir, "cft")[0]
28       print("Demonstrating transactions on tempfile", tempfile)
29       f = open(tempfile, "w")
30       f.write("This is original file.\n")
31       f.close()
32   
33       trans = win32transaction.CreateTransaction(
34           Description="Test creating miniversions of a file"
35       )
36       hfile = win32file.CreateFileW(
37           tempfile,
38           win32con.GENERIC_READ | win32con.GENERIC_WRITE,
39           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
40           None,
41           win32con.OPEN_EXISTING,
42           0,
43           None,
44           Transaction=trans,
45       )
46   
47       win32file.WriteFile(hfile, str2bytes("This is first miniversion.\n"))
48       buf = win32file.DeviceIoControl(
49           hfile, winioctlcon.FSCTL_TXFS_CREATE_MINIVERSION, None, buf_size, None
50       )
51       struct_ver, struct_len, base_ver, ver_1 = struct.unpack(buf_fmt, buf)
52   
53       win32file.SetFilePointer(hfile, 0, win32con.FILE_BEGIN)
54       win32file.WriteFile(hfile, str2bytes("This is second miniversion!\n"))
55       buf = win32file.DeviceIoControl(
56           hfile, winioctlcon.FSCTL_TXFS_CREATE_MINIVERSION, None, buf_size, None
57       )
58       struct_ver, struct_len, base_ver, ver_2 = struct.unpack(buf_fmt, buf)
59       hfile.Close()
60   
61       ## miniversions can't be opened with write access
62       hfile_0 = win32file.CreateFileW(
63           tempfile,
64           win32con.GENERIC_READ,
65           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
66           None,
67           win32con.OPEN_EXISTING,
68           0,
69           None,
70           Transaction=trans,
71           MiniVersion=base_ver,
72       )
73       print("version:", base_ver, win32file.ReadFile(hfile_0, 100))
74       hfile_0.Close()
75   
76       hfile_1 = win32file.CreateFileW(
77           tempfile,
78           win32con.GENERIC_READ,
79           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
80           None,
81           win32con.OPEN_EXISTING,
82           0,
83           None,
84           Transaction=trans,
85           MiniVersion=ver_1,
86       )
87       print("version:", ver_1, win32file.ReadFile(hfile_1, 100))
88       hfile_1.Close()
89   
90       hfile_2 = win32file.CreateFileW(
91           tempfile,
92           win32con.GENERIC_READ,
93           win32con.FILE_SHARE_READ | win32con.FILE_SHARE_WRITE,
94           None,
95           win32con.OPEN_EXISTING,
96           0,
97           None,
98           Transaction=trans,
99           MiniVersion=ver_2,
100      )
101      print("version:", ver_2, win32file.ReadFile(hfile_2, 100))
102      hfile_2.Close()
103  
104      ## MiniVersions are destroyed when transaction is committed or rolled back
105      win32transaction.CommitTransaction(trans)
106  
107      os.unlink(tempfile)
108  
109  
110  if __name__ == "__main__":
111      # When run on CI, this fails with NOT_SUPPORTED, so don't have that cause "failure"
112      try:
113          demo()
114      except win32file.error as e:
115          if e.winerror == winerror.ERROR_NOT_SUPPORTED:
116              print("These features are not supported by this filesystem.")
117          else:
118              raise
