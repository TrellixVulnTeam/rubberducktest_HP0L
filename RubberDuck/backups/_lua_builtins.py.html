189      # you can't generally find out what module a function belongs to if you
190      # have only its name. Because of this, here are some callback functions
191      # that recognize if a gioven function belongs to a specific module
192      def module_callbacks():
193          def is_in_coroutine_module(name):
194              return name.startswith('coroutine.')
195  
196          def is_in_modules_module(name):
197              if name in ['require', 'module'] or name.startswith('package'):
198                  return True
199              else:
200                  return False
201  
202          def is_in_string_module(name):
203              return name.startswith('string.')
204  
205          def is_in_table_module(name):
206              return name.startswith('table.')
207  
208          def is_in_math_module(name):
209              return name.startswith('math')
210  
211          def is_in_io_module(name):
212              return name.startswith('io.')
213  
214          def is_in_os_module(name):
215              return name.startswith('os.')
216  
217          def is_in_debug_module(name):
218              return name.startswith('debug.')
219  
220          return {'coroutine': is_in_coroutine_module,
221                  'modules': is_in_modules_module,
222                  'string': is_in_string_module,
223                  'table': is_in_table_module,
224                  'math': is_in_math_module,
225                  'io': is_in_io_module,
226                  'os': is_in_os_module,
227                  'debug': is_in_debug_module}

231      def get_newest_version():
232          f = urlopen('http://www.lua.org/manual/')
233          r = re.compile(r'^<A HREF="(\d\.\d)/">(Lua )?\1</A>')
234          for line in f:
235              m = r.match(line)
236              if m is not None:
237                  return m.groups()[0]
238  
239      def get_lua_functions(version):
240          f = urlopen('http://www.lua.org/manual/%s/' % version)
241          r = re.compile(r'^<A HREF="manual.html#pdf-(?!lua|LUA)([^:]+)">\1</A>')
242          functions = []
243          for line in f:
244              m = r.match(line)
245              if m is not None:
246                  functions.append(m.groups()[0])
247          return functions
248  
249      def get_function_module(name):
250          for mod, cb in module_callbacks().items():
251              if cb(name):
252                  return mod
253          if '.' in name:
254              return name.split('.')[0]
255          else:
256              return 'basic'

258      def regenerate(filename, modules):
259          with open(filename) as fp:
260              content = fp.read()
261  
262          header = content[:content.find('MODULES = {')]
263          footer = content[content.find("if __name__ == '__main__':"):]
264  
265  
266          with open(filename, 'w') as fp:
267              fp.write(header)
268              fp.write('MODULES = %s\n\n' % pprint.pformat(modules))
269              fp.write(footer)
270  
271      def run():
272          version = get_newest_version()
273          functions = set()
274          for v in ('5.2', version):
275              print('> Downloading function index for Lua %s' % v)
276              f = get_lua_functions(v)
277              print('> %d functions found, %d new:' %
278                    (len(f), len(set(f) - functions)))
279              functions |= set(f)
280  
281          functions = sorted(functions)
282  
283          modules = {}
284          for full_function_name in functions:
285              print('>> %s' % full_function_name)
286              m = get_function_module(full_function_name)
287              modules.setdefault(m, []).append(full_function_name)
288          modules = {k: tuple(v) for k, v in modules.items()}
289  
290          regenerate(__file__, modules)
291  
292      run()
