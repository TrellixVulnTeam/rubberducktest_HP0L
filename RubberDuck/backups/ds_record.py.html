8    def wav_header_pack(wfx, datasize):
9        return struct.pack(
10           "<4sl4s4slhhllhh4sl",
11           "RIFF",
12           36 + datasize,
13           "WAVE",
14           "fmt ",
15           16,
16           wfx.wFormatTag,
17           wfx.nChannels,
18           wfx.nSamplesPerSec,
19           wfx.nAvgBytesPerSec,
20           wfx.nBlockAlign,
21           wfx.wBitsPerSample,
22           "data",
23           datasize,
24       )

33   sdesc.lpwfxFormat.nChannels = 2

35   sdesc.lpwfxFormat.nAvgBytesPerSec = 176400

45   
46   notify.SetNotificationPositions((ds.DSBPN_OFFSETSTOP, event))
47   
48   buffer.Start(0)
