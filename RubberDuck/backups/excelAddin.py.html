1    # A demo plugin for Microsoft Excel
2    #
3    # This addin simply adds a new button to the main Excel toolbar,
4    # and displays a message box when clicked.  Thus, it demonstrates
5    # how to plug in to Excel itself, and hook Excel events.
6    #
7    #
8    # To register the addin, simply execute:
9    #   excelAddin.py
10   # This will install the COM server, and write the necessary
11   # AddIn key to Excel
12   #
13   # To unregister completely:
14   #   excelAddin.py --unregister
15   #
16   # To debug, execute:
17   #   excelAddin.py --debug
18   #
19   # Then open Pythonwin, and select "Tools->Trace Collector Debugging Tool"
20   # Restart excel, and you should see some output generated.
21   #
22   # NOTE: If the AddIn fails with an error, Excel will re-register
23   # the addin to not automatically load next time Excel starts.  To
24   # correct this, simply re-register the addin (see above)

71   class ButtonEvent:
72       def OnClick(self, button, cancel):
73           import win32ui  # Possible, but not necessary, to use a Pythonwin GUI
74           import win32con
75   
76           win32ui.MessageBox("Hello from Python", "Python Test", win32con.MB_OKCANCEL)

80   class ExcelAddin:
81       _com_interfaces_ = ["_IDTExtensibility2"]
82       _public_methods_ = []
83       _reg_clsctx_ = pythoncom.CLSCTX_INPROC_SERVER
84       _reg_clsid_ = "{C5482ECA-F559-45A0-B078-B2036E6F011A}"
85       _reg_progid_ = "Python.Test.ExcelAddin"
86       _reg_policy_spec_ = "win32com.server.policy.EventHandlerPolicy"
87   
88       def __init__(self):
89           self.appHostApp = None
90   
91       def OnConnection(self, application, connectMode, addin, custom):
92           print("OnConnection", application, connectMode, addin, custom)
93           try:
94               self.appHostApp = application
95               cbcMyBar = self.appHostApp.CommandBars.Add(
96                   Name="PythonBar",
97                   Position=constants.msoBarTop,
98                   MenuBar=constants.msoBarTypeNormal,
99                   Temporary=True,
100              )
101              btnMyButton = cbcMyBar.Controls.Add(
102                  Type=constants.msoControlButton, Parameter="Greetings"
103              )
104              btnMyButton = self.toolbarButton = DispatchWithEvents(
105                  btnMyButton, ButtonEvent
106              )
107              btnMyButton.Style = constants.msoButtonCaption
108              btnMyButton.BeginGroup = True
109              btnMyButton.Caption = "&Python"
110              btnMyButton.TooltipText = "Python rules the World"
111              btnMyButton.Width = "34"
112              cbcMyBar.Visible = True
113          except pythoncom.com_error as xxx_todo_changeme:
114              (hr, msg, exc, arg) = xxx_todo_changeme.args
115              print("The Excel call failed with code %d: %s" % (hr, msg))
116              if exc is None:
117                  print("There is no extended error information")
118              else:
119                  wcode, source, text, helpFile, helpId, scode = exc
120                  print("The source of the error is", source)
121                  print("The error message is", text)
122                  print("More info can be found in %s (id=%d)" % (helpFile, helpId))
123  
124      def OnDisconnection(self, mode, custom):
125          print("OnDisconnection")
126          self.appHostApp.CommandBars("PythonBar").Delete
127          self.appHostApp = None
128  
129      def OnAddInsUpdate(self, custom):
130          print("OnAddInsUpdate", custom)
131  
132      def OnStartupComplete(self, custom):
133          print("OnStartupComplete", custom)
134  
135      def OnBeginShutdown(self, custom):
136          print("OnBeginShutdown", custom)

139  def RegisterAddin(klass):
140      import winreg
141  
142      key = winreg.CreateKey(
143          winreg.HKEY_CURRENT_USER, "Software\\Microsoft\\Office\\Excel\\Addins"
144      )
145      subkey = winreg.CreateKey(key, klass._reg_progid_)
146      winreg.SetValueEx(subkey, "CommandLineSafe", 0, winreg.REG_DWORD, 0)
147      winreg.SetValueEx(subkey, "LoadBehavior", 0, winreg.REG_DWORD, 3)
148      winreg.SetValueEx(subkey, "Description", 0, winreg.REG_SZ, "Excel Addin")
149      winreg.SetValueEx(subkey, "FriendlyName", 0, winreg.REG_SZ, "A Simple Excel Addin")
150  
151  
152  def UnregisterAddin(klass):
153      import winreg
154  
155      try:
156          winreg.DeleteKey(
157              winreg.HKEY_CURRENT_USER,
158              "Software\\Microsoft\\Office\\Excel\\Addins\\" + klass._reg_progid_,
159          )
160      except WindowsError:
161          pass
