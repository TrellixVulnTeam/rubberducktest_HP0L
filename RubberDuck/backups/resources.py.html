resources.py 
1    ''' 
2    Resources management 
3    ==================== 
4     
5    Resource management can be a pain if you have multiple paths and projects. 
6    Kivy offers 2 functions for searching for specific resources across a list 
7    of paths. 
8     
9    Resource lookup 
10   --------------- 
11    
12   When Kivy looks for a resource e.g. an image or a kv file, it searches through 
13   a predetermined set of folders. You can modify this folder list using the 
14   :meth:`resource_add_path` and :meth:`resource_remove_path` functions. 
15    
16   Customizing Kivy 
17   ---------------- 
18    
19   These functions can also be helpful if you want to replace standard Kivy 
20   resources with your own. For example, if you wish to customize or re-style 
21   Kivy, you can force your *style.kv* or *data/defaulttheme-0.png* files to be 
22   used in preference to the defaults simply by adding the path to your preferred 
23   alternatives via the :meth:`resource_add_path` method. 
24    
25   As almost all Kivy resources are looked up using the :meth:`resource_find`, so 
26   you can use this approach to add fonts and keyboard layouts and to replace 
27   images and icons. 
28    
29   '''
30   
31   __all__ = ('resource_find', 'resource_add_path', 'resource_remove_path')
32   
33   from os.path import join, dirname, exists, abspath
34   from kivy import kivy_data_dir
35   from kivy.utils import platform
36   from kivy.logger import Logger
37   import sys
38   import kivy
39   
40   resource_paths = ['.', dirname(sys.argv[0])]
41   if platform == 'ios':
42       resource_paths += [join(dirname(sys.argv[0]), 'YourApp')]
43   resource_paths += [dirname(kivy.__file__), join(kivy_data_dir, '..')]
44   
45   
46   def resource_find(filename):
47       '''Search for a resource in the list of paths. 
48       Use resource_add_path to add a custom path to the search. 
49       '''
50       if not filename:
51           return
52       if filename[:8] == 'atlas://':
53           return filename
54       if exists(abspath(filename)):
55           return abspath(filename)
56       for path in reversed(resource_paths):
57           output = abspath(join(path, filename))
58           if exists(output):
59               return output
60       if filename[:5] == 'data:':
61           return filename
62   
63   
64   def resource_add_path(path):
65       '''Add a custom path to search in. 
66       '''
67       if path in resource_paths:
68           return
69       Logger.debug('Resource: add <%s> in path list' % path)
70       resource_paths.append(path)
71   
72   
73   def resource_remove_path(path):
74       '''Remove a search path. 
75    
76       .. versionadded:: 1.0.8 
77       '''
78       if path not in resource_paths:
79           return
80       Logger.debug('Resource: remove <%s> from path list' % path)
81       resource_paths.remove(path)
