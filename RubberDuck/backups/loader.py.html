1    ''' 
2    Asynchronous data loader 
3    ======================== 
4     
5    This is the Asynchronous Loader. You can use it to load an image 
6    and use it, even if data are not yet available. You must specify a default 
7    loading image when using the loader:: 
8     
9        from kivy.loader import Loader 
10       image = Loader.image('mysprite.png') 
11    
12   You can also load an image from a url:: 
13    
14       image = Loader.image('http://mysite.com/test.png') 
15    
16   If you want to change the default loading image, you can do:: 
17    
18       Loader.loading_image = Image('another_loading.png') 
19    
20   Tweaking the asynchronous loader 
21   -------------------------------- 
22    
23   .. versionadded:: 1.6.0 
24    
25   You can tweak the loader to provide a better user experience or more 
26   performance, depending of the images you are going to load. Take a look at the 
27   parameters: 
28    
29   - :attr:`Loader.num_workers` - define the number of threads to start for 
30     loading images. 
31   - :attr:`Loader.max_upload_per_frame` - define the maximum image uploads in 
32     GPU to do per frame. 
33    
34   '''

88   class LoaderBase(object):
89       '''Common base for the Loader and specific implementations. 
90       By default, the Loader will be the best available loader implementation. 
91    
92       The _update() function is called every 1 / 25.s or each frame if we have 
93       less than 25 FPS. 
94       '''
95       _trigger_update = None
96   
97       '''Alias for mimetype extensions. 
98    
99       If you have trouble to have the right extension to be detected, 
100      you can either add #.EXT at the end of the url, or use this array 
101      to correct the detection. 
102      For example, a zip-file on Windows can be detected as pyz. 
103   
104      By default, '.pyz' is translated to '.zip' 
105   
106      .. versionadded:: 1.11.0 
107      '''
108      EXT_ALIAS = {
109          '.pyz': '.zip'
110      }

115          self._num_workers = 2

127          if platform in ['android', 'ios']:

135      def _set_num_workers(self, num):
136          if num < 2:
137              raise Exception('Must have at least 2 workers')

140      def _get_num_workers(self):

143      num_workers = property(_get_num_workers, _set_num_workers)
144      '''Number of workers to use while loading (used only if the loader 
145      implementation supports it). This setting impacts the loader only on 
146      initialization. Once the loader is started, the setting has no impact:: 
147   
148          from kivy.loader import Loader 
149          Loader.num_workers = 4 
150   
151      The default value is 2 for giving a smooth user experience. You could 
152      increase the number of workers, then all the images will be loaded faster, 
153      but the user will not been able to use the application while loading. 
154      Prior to 1.6.0, the default number was 20, and loading many full-hd images 
155      was completly blocking the application. 
156   
157      .. versionadded:: 1.6.0 
158      '''

160      def _set_max_upload_per_frame(self, num):
161          if num is not None and num < 1:
162              raise Exception('Must have at least 1 image processing per image')
163          self._max_upload_per_frame = num
164  
165      def _get_max_upload_per_frame(self):
166          return self._max_upload_per_frame
167  
168      max_upload_per_frame = property(_get_max_upload_per_frame,
169                                      _set_max_upload_per_frame)
170      '''The number of images to upload per frame. By default, we'll 
171      upload only 2 images to the GPU per frame. If you are uploading many 
172      small images, you can easily increase this parameter to 10 or more. 
173      If you are loading multiple full HD images, the upload time may have 
174      consequences and block the application. If you want a 
175      smooth experience, use the default. 
176   
177      As a matter of fact, a Full-HD RGB image will take ~6MB in memory, 
178      so it may take time. If you have activated mipmap=True too, then the 
179      GPU must calculate the mipmap of these big images too, in real time. 
180      Then it may be best to reduce the :attr:`max_upload_per_frame` to 1 
181      or 2. If you want to get rid of that (or reduce it a lot), take a 
182      look at the DDS format. 
183   
184      .. versionadded:: 1.6.0 
185      '''

251      def resume(self):
252          '''Resume the loader, after a :meth:`pause`. 
253   
254          .. versionadded:: 1.6.0 
255          '''
256          self._paused = False
257          self._resume_cond.acquire()
258          self._resume_cond.notify_all()
259          self._resume_cond.release()
260  
261      def _wait_for_resume(self):
262          while self._running and self._paused:
263              self._resume_cond.acquire()
264              self._resume_cond.wait(0.25)
265              self._resume_cond.release()

267      def _load(self, kwargs):
268          '''(internal) Loading function, called by the thread. 
269          Will call _load_local() if the file is local, 
270          or _load_urllib() if the file is on Internet. 
271          '''
272  
273          while len(self._q_done) >= (
274                  self.max_upload_per_frame * self._num_workers):
275              sleep(0.1)
276  
277          self._wait_for_resume()
278  
279          filename = kwargs['filename']
280          load_callback = kwargs['load_callback']
281          post_callback = kwargs['post_callback']
282          try:
283              proto = filename.split(':', 1)[0]
284          except:
285              # if blank filename then return
286              return
287          if load_callback is not None:
288              data = load_callback(filename)
289          elif proto in ('http', 'https', 'ftp', 'smb'):
290              data = self._load_urllib(filename, kwargs['kwargs'])
291          else:
292              data = self._load_local(filename, kwargs['kwargs'])
293  
294          if post_callback:
295              data = post_callback(data)
296  
297          self._q_done.appendleft((filename, data))
298          self._trigger_update()
299  
300      def _load_local(self, filename, kwargs):
301          '''(internal) Loading a local file'''
302          # With recent changes to CoreImage, we must keep data otherwise,
303          # we might be unable to recreate the texture afterwise.
304          return ImageLoader.load(filename, keep_data=True, **kwargs)
305  
306      def _load_urllib(self, filename, kwargs):
307          '''(internal) Loading a network file. First download it, save it to a 
308          temporary file, and pass it to _load_local().'''
309          if PY2:
310              import urllib2 as urllib_request
311  
312              def gettype(info):
313                  return info.gettype()
314          else:
315              import urllib.request as urllib_request
316  
317              def gettype(info):
318                  return info.get_content_type()
319          proto = filename.split(':', 1)[0]
320          if proto == 'smb':
321              try:
322                  # note: it's important to load SMBHandler every time
323                  # otherwise the data is occasionally not loaded
324                  from smb.SMBHandler import SMBHandler
325              except ImportError:
326                  Logger.warning(
327                      'Loader: can not load PySMB: make sure it is installed')
328                  return
329          import tempfile
330          data = fd = _out_osfd = None
331          try:
332              _out_filename = ''
333  
334              if proto == 'smb':
335                  # read from samba shares
336                  fd = urllib_request.build_opener(SMBHandler).open(filename)
337              else:
338                  # read from internet
339                  request = urllib_request.Request(filename)
340                  if Config.has_option('network', 'useragent'):
341                      useragent = Config.get('network', 'useragent')
342                      if useragent:
343                          request.add_header('User-Agent', useragent)
344                  opener = urllib_request.build_opener()
345                  fd = opener.open(request)
346  
347              if '#.' in filename:
348                  # allow extension override from URL fragment
349                  suffix = '.' + filename.split('#.')[-1]
350              else:
351                  ctype = gettype(fd.info())
352                  suffix = mimetypes.guess_extension(ctype)
353                  suffix = LoaderBase.EXT_ALIAS.get(suffix, suffix)
354                  if not suffix:
355                      # strip query string and split on path
356                      parts = filename.split('?')[0].split('/')[1:]
357                      while len(parts) > 1 and not parts[0]:
358                          # strip out blanks from '//'
359                          parts = parts[1:]
360                      if len(parts) > 1 and '.' in parts[-1]:
361                          # we don't want '.com', '.net', etc. as the extension
362                          suffix = '.' + parts[-1].split('.')[-1]
363              _out_osfd, _out_filename = tempfile.mkstemp(
364                  prefix='kivyloader', suffix=suffix)
365  
366              idata = fd.read()
367              fd.close()
368              fd = None
369  
370              # write to local filename
371              write(_out_osfd, idata)
372              close(_out_osfd)
373              _out_osfd = None
374  
375              # load data
376              data = self._load_local(_out_filename, kwargs)
377  
378              # FIXME create a clean API for that
379              for imdata in data._data:
380                  imdata.source = filename
381          except Exception as ex:
382              Logger.exception('Loader: Failed to load image <%s>' % filename)
383              # close file when remote file not found or download error
384              try:
385                  if _out_osfd:
386                      close(_out_osfd)
387              except OSError:
388                  pass
389  
390              # update client
391              for c_filename, client in self._client[:]:
392                  if filename != c_filename:
393                      continue
394                  # got one client to update
395                  client.image = self.error_image
396                  client.dispatch('on_error', error=ex)
397                  self._client.remove((c_filename, client))
398  
399              return self.error_image
400          finally:
401              if fd:
402                  fd.close()
403              if _out_osfd:
404                  close(_out_osfd)
405              if _out_filename != '':
406                  unlink(_out_filename)
407  
408          return data
409  
410      def _update(self, *largs):
411          '''(internal) Check if a data is loaded, and pass to the client.'''
412          # want to start it ?
413          if self._start_wanted:
414              if not self._running:
415                  self.start()
416              self._start_wanted = False
417  
418          # in pause mode, don't unqueue anything.
419          if self._paused:
420              self._trigger_update()
421              return
422  
423          for x in range(self.max_upload_per_frame):
424              try:
425                  filename, data = self._q_done.pop()
426              except IndexError:
427                  return
428  
429              # create the image
430              image = data  # ProxyImage(data)
431              if not image.nocache:
432                  Cache.append('kv.loader', filename, image)
433  
434              # update client
435              for c_filename, client in self._client[:]:
436                  if filename != c_filename:
437                      continue
438                  # got one client to update
439                  client.image = image
440                  client.loaded = True
441                  client.dispatch('on_load')
442                  self._client.remove((c_filename, client))
443  
444          self._trigger_update()

446      def image(self, filename, load_callback=None, post_callback=None,
447                **kwargs):
448          '''Load a image using the Loader. A ProxyImage is returned with a 
449          loading image. You can use it as follows:: 
450   
451              from kivy.app import App 
452              from kivy.uix.image import Image 
453              from kivy.loader import Loader 
454   
455              class TestApp(App): 
456                  def _image_loaded(self, proxyImage): 
457                      if proxyImage.image.texture: 
458                          self.image.texture = proxyImage.image.texture 
459   
460                  def build(self): 
461                      proxyImage = Loader.image("myPic.jpg") 
462                      proxyImage.bind(on_load=self._image_loaded) 
463                      self.image = Image() 
464                      return self.image 
465   
466              TestApp().run() 
467   
468          In order to cancel all background loading, call *Loader.stop()*. 
469          '''
470          data = Cache.get('kv.loader', filename)
471          if data not in (None, False):
472              # found image, if data is not here, need to reload.
473              return ProxyImage(data,
474                                loading_image=self.loading_image,
475                                loaded=True, **kwargs)
476  
477          client = ProxyImage(self.loading_image,
478                              loading_image=self.loading_image, **kwargs)
479          self._client.append((filename, client))
480  
481          if data is None:
482              # if data is None, this is really the first time
483              self._q_load.appendleft({
484                  'filename': filename,
485                  'load_callback': load_callback,
486                  'post_callback': post_callback,
487                  'kwargs': kwargs})
488              if not kwargs.get('nocache', False):
489                  Cache.append('kv.loader', filename, False)
490              self._start_wanted = True
491              self._trigger_update()
492          else:
493              # already queued for loading
494              pass
495  
496          return client

519      class _Worker(Thread):
520          '''Thread executing tasks from a given tasks queue 
521          '''
522          def __init__(self, pool, tasks):
523              Thread.__init__(self)
524              self.tasks = tasks
525              self.daemon = True
526              self.pool = pool
527              self.start()
528  
529          def run(self):
530              while self.pool.running:
531                  func, args, kargs = self.tasks.get()
532                  try:
533                      func(*args, **kargs)
534                  except Exception as e:
535                      print(e)
536                  self.tasks.task_done()
537  
538      class _ThreadPool(object):
539          '''Pool of threads consuming tasks from a queue 
540          '''
541          def __init__(self, num_threads):
542              super(_ThreadPool, self).__init__()
543              self.running = True
544              self.tasks = queue.Queue()
545              for _ in range(num_threads):
546                  _Worker(self, self.tasks)
547  
548          def add_task(self, func, *args, **kargs):
549              '''Add a task to the queue 
550              '''
551              self.tasks.put((func, args, kargs))
552  
553          def stop(self):
554              self.running = False
555              self.tasks.join()
556  
557      class LoaderThreadPool(LoaderBase):
558          def __init__(self):
559              super(LoaderThreadPool, self).__init__()
560              self.pool = None
561  
562          def start(self):
563              super(LoaderThreadPool, self).start()
564              self.pool = _ThreadPool(self._num_workers)
565              Clock.schedule_interval(self.run, 0)
566  
567          def stop(self):
568              super(LoaderThreadPool, self).stop()
569              Clock.unschedule(self.run)
570              self.pool.stop()
571  
572          def run(self, *largs):
573              while self._running:
574                  try:
575                      parameters = self._q_load.pop()
576                  except:
577                      return
578                  self.pool.add_task(self._load, parameters)
579  
580      Loader = LoaderThreadPool()
581      Logger.info('Loader: using a thread pool of {} workers'.format(
582          Loader.num_workers))
