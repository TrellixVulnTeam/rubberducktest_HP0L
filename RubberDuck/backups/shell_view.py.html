4    # * Execute this script to register the namespace.
5    # * Open Windows Explorer, and locate the new "Python Path Shell Browser"
6    #   folder off "My Computer"
7    # * Browse this tree - .py files are shown expandable, with classes and
8    #   methods selectable.  Selecting a Python file, or a class/method, will
9    #   display the file using Scintilla.

11   # * Classes and methods don't have icons - this is a demo, so we keep it small
12   #   See icon_handler.py for examples of how to work with icons.
13   #
14   #
15   # Notes on PIDLs
16   # PIDLS are complicated, but fairly well documented in MSDN.  If you need to
17   # do much with these shell extensions, you must understand their concept.
18   # Here is a short-course, as it applies to this sample:
19   # A PIDL identifies an item, much in the same way that a filename does
20   # (however, the shell is not limited to displaying "files").
21   # An "ItemID" is a single string, each being an item in the hierarchy.
22   # A "PIDL" is a list of these strings.
23   # All shell etc functions work with PIDLs, so even in the case where
24   # an ItemID is conceptually used, a 1-item list is always used.
25   # Conceptually, think of:
26   #    pidl = pathname.split("\\") # pidl is a list of "parent" items.
27   #    # each item is a string 'item id', but these are ever used directly
28   # As there is no concept of passing a single item, to open a file using only
29   # a relative filename, conceptually you would say:
30   #   open_file([filename]) # Pass a single-itemed relative "PIDL"
31   # and continuing the analogy, a "listdir" type function would return a list
32   # of single-itemed lists - each list containing the relative PIDL of the child.
33   #
34   # Each PIDL entry is a binary string, and may contain any character.  For
35   # PIDLs not created by you, they can not be interpreted - they are just
36   # blobs.  PIDLs created by you (ie, children of your IShellFolder) can
37   # store and interpret the string however makes most sense for your application.
38   # (but within PIDL rules - they must be persistable, etc)
39   # There is no reason that pickled strings, for example, couldn't be used
40   # as an EntryID.
41   # This application takes a simple approach - each PIDL is a string of form
42   # "directory\0directory_name", "file\0file_name" or
43   # "object\0file_name\0class_name[.method_name"
44   # The first string in each example is literal (ie, the word 'directory',
45   # 'file' or 'object', and every other string is variable.  We use '\0' as
46   # a field sep just 'cos we can (and 'cos it can't possibly conflict with the
47   # string content)

71   # Helper function to get a system IShellFolder interface, and the PIDL within
72   # that folder for an existing file/directory.
73   def GetFolderAndPIDLForPath(filename):
74       desktop = shell.SHGetDesktopFolder()
75       info = desktop.ParseDisplayName(0, None, os.path.abspath(filename))
76       cchEaten, pidl, attr = info
77       # We must walk the ID list, looking for one child at a time.
78       folder = desktop
79       while len(pidl) > 1:
80           this = pidl.pop(0)
81           folder = folder.BindToObject([this], None, shell.IID_IShellFolder)
82       # We are left with the pidl for the specific item.  Leave it as
83       # a list, so it remains a valid PIDL.
84       return folder, pidl

91   def get_clbr_for_file(path):
92       try:
93           objects = clbr_modules[path]
94       except KeyError:
95           dir, filename = os.path.split(path)
96           base, ext = os.path.splitext(filename)
97           objects = pyclbr.readmodule_ex(base, [dir])
98           clbr_modules[path] = objects
99       return objects

132      def BindToObject(self, pidl, bc, iid):
133          # We may be passed a set of relative PIDLs here - ie
134          # [pidl_of_dir, pidl_of_child_dir, pidl_of_file, pidl_of_function]
135          # But each of our PIDLs keeps the fully qualified name anyway - so
136          # just jump directly to the last.
137          final_pidl = pidl[-1]
138          typ, extra = final_pidl.split("\0", 1)
139          if typ == "directory":
140              klass = ShellFolderDirectory
141          elif typ == "file":
142              klass = ShellFolderFile
143          elif typ == "object":
144              klass = ShellFolderObject
145          else:
146              raise RuntimeError("What is " + repr(typ))
147          ret = wrap(klass(extra), iid, useDispatcher=(debug > 0))
148          return ret

259      def GetAttributesOf(self, pidls, attrFlags):
260          ret_flags = -1
261          for pidl in pidls:
262              assert len(pidl) == 1, "Expecting relative pidls"
263              pidl = pidl[0]
264              typ, filename, obname = pidl.split("\0")
265              obs = get_clbr_for_file(filename)
266              ob = obs[obname]
267              flags = (
268                  shellcon.SFGAO_BROWSABLE
269                  | shellcon.SFGAO_FOLDER
270                  | shellcon.SFGAO_FILESYSANCESTOR
271              )
272              if hasattr(ob, "methods"):
273                  flags |= shellcon.SFGAO_HASSUBFOLDER
274              ret_flags &= flags

281          if flags & shellcon.SHGDN_INFOLDER:
282              ret = obname
283          else:  # SHGDN_NORMAL is the default
284              ret = fqname
285          # No need to look at the SHGDN_FOR* modifiers.

292  # A ShellFolder for our Python objects
293  class ShellFolderObject(ShellFolderBase):
294      def __init__(self, details):
295          self.path, details = details.split("\0")
296          if details.find(".") > 0:
297              self.class_name, self.method_name = details.split(".")
298          else:
299              self.class_name = details
300              self.method_name = None
301  
302      def CreateViewObject(self, hwnd, iid):
303          mod_objects = get_clbr_for_file(self.path)
304          object = mod_objects[self.class_name]
305          if self.method_name is None:
306              lineno = object.lineno
307          else:
308              lineno = object.methods[self.method_name]
309              return wrap(
310                  ScintillaShellView(hwnd, self.path, lineno),
311                  iid,
312                  useDispatcher=debug > 0,
313              )
314  
315      def EnumObjects(self, hwndOwner, flags):
316          assert self.method_name is None, "Should not be enuming methods!"
317          mod_objects = get_clbr_for_file(self.path)
318          my_objects = mod_objects[self.class_name]
319          pidls = []
320          for func_name, lineno in my_objects.methods.items():
321              pidl = ["object\0" + self.path + "\0" + self.class_name + "." + func_name]
322              pidls.append(pidl)
323          return NewEnum(pidls, iid=shell.IID_IEnumIDList, useDispatcher=(debug > 0))
324  
325      def GetDisplayNameOf(self, pidl, flags):
326          assert len(pidl) == 1, "Expecting relative PIDL"
327          typ, fname, obname = pidl[0].split("\0")
328          class_name, method_name = obname.split(".")
329          fqname = os.path.splitext(fname)[0] + "." + obname
330          if flags & shellcon.SHGDN_INFOLDER:
331              ret = method_name
332          else:  # SHGDN_NORMAL is the default
333              ret = fqname
334          # No need to look at the SHGDN_FOR* modifiers.
335          return ret

337      def GetAttributesOf(self, pidls, attrFlags):
338          ret_flags = -1
339          for pidl in pidls:

542      def _OnActivate(self, activate_state):
543          if self.activate_state == activate_state:
544              return
545          self._OnDeactivate()  # restore menu's first, if necessary.
546          if activate_state != shellcon.SVUIA_DEACTIVATE:
547              assert self.hmenu is None, "Should have destroyed it!"
548              self.hmenu = win32gui.CreateMenu()
549              widths = 0, 0, 0, 0, 0, 0
550              # Ask explorer to add its standard items.
551              self.browser.InsertMenusSB(self.hmenu, widths)
552              # Merge with these standard items
553              self._MergeMenus(activate_state)
554              self.browser.SetMenuSB(self.hmenu, 0, self.hwnd)
555          self.activate_state = activate_state
556  
557      def _OnDeactivate(self):
558          if self.browser is not None and self.hmenu is not None:
559              self.browser.SetMenuSB(0, 0, 0)
560              self.browser.RemoveMenusSB(self.hmenu)
561              win32gui.DestroyMenu(self.hmenu)
562              self.hmenu = None
563          self.hsubmenus = None
564          self.activate_state = shellcon.SVUIA_DEACTIVATE

572          mid = shellcon.FCIDM_MENU_FILE

574          buf, extras = win32gui_struct.EmptyMENUITEMINFO(win32con.MIIM_SUBMENU)
575          win32gui.GetMenuItemInfo(self.hmenu, mid, False, buf)
576          data = win32gui_struct.UnpackMENUITEMINFO(buf)

581          stateMask = commctrl.LVIS_SELECTED | commctrl.LVIS_DROPHILITED
582          state = 0
583          self.children = []
584          # Enumerate and store the child PIDLs

590              typ, path = data[0].split("\0")
591              desc = os.path.exists(path) and "Yes" or "No"
592              prop_vals = (path, desc)
593              # first col
594              data, extras = win32gui_struct.PackLVITEM(
595                  item=row_index,
596                  subItem=0,
597                  text=prop_vals[0],
598                  state=state,
599                  stateMask=stateMask,
600              )
601              win32gui.SendMessage(
602                  self.hwnd_child, commctrl.LVM_INSERTITEM, row_index, data
603              )
604              # rest of the cols.
605              col_index = 1
606              for prop_val in prop_vals[1:]:
607                  data, extras = win32gui_struct.PackLVITEM(
608                      item=row_index, subItem=col_index, text=prop_val
609                  )
610  
611                  win32gui.SendMessage(self.hwnd_child, commctrl.LVM_SETITEM, 0, data)
612                  col_index += 1

638      def OnNotify(self, hwnd, msg, wparam, lparam):

640          # print "OnNotify code=0x%x (0x%x, 0x%x)" % (code, wparam, lparam)

642              # Control got focus - Explorer may not know - tell it
643              if self.browser is not None:
644                  self.browser.OnViewWindowActive(None)
645              # And do our menu thang
646              self._OnActivate(shellcon.SVUIA_ACTIVATE_FOCUS)
647          elif code == commctrl.NM_KILLFOCUS:
648              self._OnDeactivate()
649          elif code == commctrl.NM_DBLCLK:
650              # This DblClick implementation leaves a little to be desired :)
651              # It demonstrates some useful concepts, such as asking the
652              # folder for its context-menu and invoking a command from it.
653              # However, as our folder delegates IContextMenu to the shell

656              # our shell-extension.
657              # Determine the selected items.

660              while 1:
661                  n = win32gui.SendMessage(
662                      self.hwnd_child, commctrl.LVM_GETNEXTITEM, n, commctrl.LVNI_SELECTED

664                  if n == -1:
665                      break
666                  sel.append(self.children[n][-1:])
667              print("Selection is", sel)
668              hmenu = win32gui.CreateMenu()
669              try:
670                  # Get the IContextMenu for the items.
671                  inout, cm = self.folder.GetUIObjectOf(
672                      self.hwnd_parent, sel, shell.IID_IContextMenu, 0
673                  )
674

687                  # shows a dialog with text of E_INVALID_PARAM, and new
688                  # Explorer window opens with desktop view. Vista: function

725      def OnContextMenu(self, hwnd, msg, wparam, lparam):
726          # Get the selected items.
727          pidls = []
728          n = -1
729          while 1:
730              n = win32gui.SendMessage(
731                  self.hwnd_child, commctrl.LVM_GETNEXTITEM, n, commctrl.LVNI_SELECTED
732              )
733              if n == -1:
734                  break
735              pidls.append(self.children[n][-1:])
736  
737          spt = win32api.GetCursorPos()
738          if not pidls:
739              print("Ignoring background click")
740              return
741          # Get the IContextMenu for the items.
742          inout, cm = self.folder.GetUIObjectOf(
743              self.hwnd_parent, pidls, shell.IID_IContextMenu, 0
744          )
745          hmenu = win32gui.CreatePopupMenu()
746          sel = None
747          # As per 'Q179911', we need to determine if the default operation
748          # should be 'open' or 'explore'
749          try:
750              flags = 0
751              try:
752                  self.browser.GetControlWindow(shellcon.FCW_TREE)
753                  flags |= shellcon.CMF_EXPLORE
754              except pythoncom.com_error:
755                  pass
756              id_cmd_first = 1  # TrackPopupMenu makes it hard to use 0
757              cm.QueryContextMenu(hmenu, 0, id_cmd_first, -1, flags)
758              tpm_flags = (
759                  win32con.TPM_LEFTALIGN
760                  | win32con.TPM_RETURNCMD
761                  | win32con.TPM_RIGHTBUTTON
762              )
763              sel = win32gui.TrackPopupMenu(
764                  hmenu, tpm_flags, spt[0], spt[1], 0, self.hwnd, None
765              )
766              print("TrackPopupMenu returned", sel)
767          finally:
768              win32gui.DestroyMenu(hmenu)
769          if sel:
770              ci = 0, self.hwnd_parent, sel - id_cmd_first, None, None, 0, 0, 0
771              cm.InvokeCommand(ci)

926      )

929          shellcon.SFGAO_FOLDER | shellcon.SFGAO_HASSUBFOLDER | shellcon.SFGAO_BROWSABLE
930      )
931      import struct
932  
933      s = struct.pack("i", attr)
934      winreg.SetValueEx(key, "Attributes", 0, winreg.REG_BINARY, s)
935      print(ShellFolderRoot._reg_desc_, "registration complete.")
