1    """ 
2    Helper classes for SSPI authentication via the win32security module. 
3     
4    SSPI authentication involves a token-exchange "dance", the exact details 
5    of which depends on the authentication provider used.  There are also 
6    a number of complex flags and constants that need to be used - in most 
7    cases, there are reasonable defaults. 
8     
9    These classes attempt to hide these details from you until you really need 
10   to know.  They are not designed to handle all cases, just the common ones. 
11   If you need finer control than offered here, just use the win32security 
12   functions directly. 
13   """
14   # Based on Roger Upole's sspi demos.
15   # $Id$
16   import win32security, sspicon
17   
18   error = win32security.error
19

21   class _BaseAuth(object):
22       def __init__(self):
23           self.reset()
24   
25       def reset(self):
26           """Reset everything to an unauthorized state"""
27           self.ctxt = None
28           self.authenticated = False
29           self.initiator_name = None
30           self.service_name = None
31   
32           # The next seq_num for an encrypt/sign operation
33           self.next_seq_num = 0
34   
35       def _get_next_seq_num(self):
36           """Get the next sequence number for a transmission.  Default 
37           implementation is to increment a counter 
38           """
39           ret = self.next_seq_num
40           self.next_seq_num = self.next_seq_num + 1
41           return ret
42   
43       def encrypt(self, data):
44           """Encrypt a string, returning a tuple of (encrypted_data, trailer). 
45           These can be passed to decrypt to get back the original string. 
46           """
47           pkg_size_info = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
48           trailersize = pkg_size_info["SecurityTrailer"]
49   
50           encbuf = win32security.PySecBufferDescType()
51           encbuf.append(win32security.PySecBufferType(len(data), sspicon.SECBUFFER_DATA))
52           encbuf.append(
53               win32security.PySecBufferType(trailersize, sspicon.SECBUFFER_TOKEN)
54           )
55           encbuf[0].Buffer = data
56           self.ctxt.EncryptMessage(0, encbuf, self._get_next_seq_num())
57           return encbuf[0].Buffer, encbuf[1].Buffer
58   
59       def decrypt(self, data, trailer):
60           """Decrypt a previously encrypted string, returning the orignal data"""
61           encbuf = win32security.PySecBufferDescType()
62           encbuf.append(win32security.PySecBufferType(len(data), sspicon.SECBUFFER_DATA))
63           encbuf.append(
64               win32security.PySecBufferType(len(trailer), sspicon.SECBUFFER_TOKEN)
65           )
66           encbuf[0].Buffer = data
67           encbuf[1].Buffer = trailer
68           self.ctxt.DecryptMessage(encbuf, self._get_next_seq_num())
69           return encbuf[0].Buffer
70   
71       def sign(self, data):
72           """sign a string suitable for transmission, returning the signature. 
73           Passing the data and signature to verify will determine if the data 
74           is unchanged. 
75           """
76           pkg_size_info = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
77           sigsize = pkg_size_info["MaxSignature"]
78           sigbuf = win32security.PySecBufferDescType()
79           sigbuf.append(win32security.PySecBufferType(len(data), sspicon.SECBUFFER_DATA))
80           sigbuf.append(win32security.PySecBufferType(sigsize, sspicon.SECBUFFER_TOKEN))
81           sigbuf[0].Buffer = data
82   
83           self.ctxt.MakeSignature(0, sigbuf, self._get_next_seq_num())
84           return sigbuf[1].Buffer
85   
86       def verify(self, data, sig):
87           """Verifies data and its signature.  If verification fails, an sspi.error 
88           will be raised. 
89           """
90           sigbuf = win32security.PySecBufferDescType()
91           sigbuf.append(win32security.PySecBufferType(len(data), sspicon.SECBUFFER_DATA))
92           sigbuf.append(win32security.PySecBufferType(len(sig), sspicon.SECBUFFER_TOKEN))
93   
94           sigbuf[0].Buffer = data
95           sigbuf[1].Buffer = sig
96           self.ctxt.VerifySignature(sigbuf, self._get_next_seq_num())
97   
98       def unwrap(self, token):
99           """ 
100          GSSAPI's unwrap with SSPI. 
101          https://docs.microsoft.com/en-us/windows/win32/secauthn/sspi-kerberos-interoperability-with-gssapi 
102   
103          Usable mainly with Kerberos SSPI package, but this is not enforced. 
104   
105          Return the clear text, and a boolean that is True if the token was encrypted. 
106          """
107          buffer = win32security.PySecBufferDescType()
108          # This buffer will contain a "stream", which is the token coming from the other side
109          buffer.append(
110              win32security.PySecBufferType(len(token), sspicon.SECBUFFER_STREAM)
111          )
112          buffer[0].Buffer = token
113  
114          # This buffer will receive the clear, or just unwrapped text if no encryption was used.
115          # Will be resized by the lib.
116          buffer.append(win32security.PySecBufferType(0, sspicon.SECBUFFER_DATA))
117  
118          pfQOP = self.ctxt.DecryptMessage(buffer, self._get_next_seq_num())
119  
120          r = buffer[1].Buffer
121          return r, not (pfQOP == sspicon.SECQOP_WRAP_NO_ENCRYPT)
122  
123      def wrap(self, msg, encrypt=False):
124          """ 
125          GSSAPI's wrap with SSPI. 
126          https://docs.microsoft.com/en-us/windows/win32/secauthn/sspi-kerberos-interoperability-with-gssapi 
127   
128          Usable mainly with Kerberos SSPI package, but this is not enforced. 
129   
130          Wrap a message to be sent to the other side. Encrypted if encrypt is True. 
131          """
132  
133          size_info = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_SIZES)
134          trailer_size = size_info["SecurityTrailer"]
135          block_size = size_info["BlockSize"]
136  
137          buffer = win32security.PySecBufferDescType()
138  
139          # This buffer will contain unencrypted data to wrap, and maybe encrypt.
140          buffer.append(win32security.PySecBufferType(len(msg), sspicon.SECBUFFER_DATA))
141          buffer[0].Buffer = msg
142  
143          # Will receive the token that forms the beginning of the msg
144          buffer.append(
145              win32security.PySecBufferType(trailer_size, sspicon.SECBUFFER_TOKEN)
146          )
147  
148          # The trailer is needed in case of block encryption
149          buffer.append(
150              win32security.PySecBufferType(block_size, sspicon.SECBUFFER_PADDING)
151          )
152  
153          fQOP = 0 if encrypt else sspicon.SECQOP_WRAP_NO_ENCRYPT
154          self.ctxt.EncryptMessage(fQOP, buffer, self._get_next_seq_num())
155  
156          # Sec token, then data, then padding
157          r = buffer[1].Buffer + buffer[0].Buffer + buffer[2].Buffer
158          return r
159  
160      def _amend_ctx_name(self):
161          """Adds initiator and service names in the security context for ease of use"""
162          if not self.authenticated:
163              raise ValueError("Sec context is not completely authenticated")
164  
165          try:
166              names = self.ctxt.QueryContextAttributes(sspicon.SECPKG_ATTR_NATIVE_NAMES)
167          except error:
168              # The SSP doesn't provide these attributes.
169              pass
170          else:
171              self.initiator_name, self.service_name = names

174  class ClientAuth(_BaseAuth):
175      """Manages the client side of an SSPI authentication handshake"""
176  
177      def __init__(
178          self,
179          pkg_name,  # Name of the package to used.
180          client_name=None,  # User for whom credentials are used.
181          auth_info=None,  # or a tuple of (username, domain, password)
182          targetspn=None,  # Target security context provider name.
183          scflags=None,  # security context flags
184          datarep=sspicon.SECURITY_NETWORK_DREP,
185      ):
186          if scflags is None:
187              scflags = (
188                  sspicon.ISC_REQ_INTEGRITY
189                  | sspicon.ISC_REQ_SEQUENCE_DETECT
190                  | sspicon.ISC_REQ_REPLAY_DETECT
191                  | sspicon.ISC_REQ_CONFIDENTIALITY
192              )
193          self.scflags = scflags
194          self.datarep = datarep
195          self.targetspn = targetspn
196          self.pkg_info = win32security.QuerySecurityPackageInfo(pkg_name)
197          (
198              self.credentials,
199              self.credentials_expiry,
200          ) = win32security.AcquireCredentialsHandle(
201              client_name,
202              self.pkg_info["Name"],
203              sspicon.SECPKG_CRED_OUTBOUND,
204              None,
205              auth_info,
206          )
207          _BaseAuth.__init__(self)
208  
209      def authorize(self, sec_buffer_in):
210          """Perform *one* step of the client authentication process. Pass None for the first round"""
211          if (
212              sec_buffer_in is not None
213              and type(sec_buffer_in) != win32security.PySecBufferDescType
214          ):
215              # User passed us the raw data - wrap it into a SecBufferDesc
216              sec_buffer_new = win32security.PySecBufferDescType()
217              tokenbuf = win32security.PySecBufferType(
218                  self.pkg_info["MaxToken"], sspicon.SECBUFFER_TOKEN
219              )
220              tokenbuf.Buffer = sec_buffer_in
221              sec_buffer_new.append(tokenbuf)
222              sec_buffer_in = sec_buffer_new
223          sec_buffer_out = win32security.PySecBufferDescType()
224          tokenbuf = win32security.PySecBufferType(
225              self.pkg_info["MaxToken"], sspicon.SECBUFFER_TOKEN
226          )
227          sec_buffer_out.append(tokenbuf)
228          ## input context handle should be NULL on first call
229          ctxtin = self.ctxt
230          if self.ctxt is None:
231              self.ctxt = win32security.PyCtxtHandleType()
232          err, attr, exp = win32security.InitializeSecurityContext(
233              self.credentials,
234              ctxtin,
235              self.targetspn,
236              self.scflags,
237              self.datarep,
238              sec_buffer_in,
239              self.ctxt,
240              sec_buffer_out,
241          )
242          # Stash these away incase someone needs to know the state from the
243          # final call.
244          self.ctxt_attr = attr
245          self.ctxt_expiry = exp
246  
247          if err in (sspicon.SEC_I_COMPLETE_NEEDED, sspicon.SEC_I_COMPLETE_AND_CONTINUE):
248              self.ctxt.CompleteAuthToken(sec_buffer_out)
249  
250          self.authenticated = err == 0
251          if self.authenticated:
252              self._amend_ctx_name()
253  
254          return err, sec_buffer_out

257  class ServerAuth(_BaseAuth):
258      """Manages the server side of an SSPI authentication handshake"""
259  
260      def __init__(
261          self, pkg_name, spn=None, scflags=None, datarep=sspicon.SECURITY_NETWORK_DREP
262      ):
263          self.spn = spn
264          self.datarep = datarep
265  
266          if scflags is None:
267              scflags = (
268                  sspicon.ASC_REQ_INTEGRITY
269                  | sspicon.ASC_REQ_SEQUENCE_DETECT
270                  | sspicon.ASC_REQ_REPLAY_DETECT
271                  | sspicon.ASC_REQ_CONFIDENTIALITY
272              )
273          # Should we default to sspicon.KerbAddExtraCredentialsMessage
274          # if pkg_name=='Kerberos'?
275          self.scflags = scflags
276  
277          self.pkg_info = win32security.QuerySecurityPackageInfo(pkg_name)
278  
279          (
280              self.credentials,
281              self.credentials_expiry,
282          ) = win32security.AcquireCredentialsHandle(
283              spn, self.pkg_info["Name"], sspicon.SECPKG_CRED_INBOUND, None, None
284          )
285          _BaseAuth.__init__(self)
286  
287      def authorize(self, sec_buffer_in):
288          """Perform *one* step of the server authentication process."""
289          if (
290              sec_buffer_in is not None
291              and type(sec_buffer_in) != win32security.PySecBufferDescType
292          ):
293              # User passed us the raw data - wrap it into a SecBufferDesc
294              sec_buffer_new = win32security.PySecBufferDescType()
295              tokenbuf = win32security.PySecBufferType(
296                  self.pkg_info["MaxToken"], sspicon.SECBUFFER_TOKEN
297              )
298              tokenbuf.Buffer = sec_buffer_in
299              sec_buffer_new.append(tokenbuf)
300              sec_buffer_in = sec_buffer_new
301  
302          sec_buffer_out = win32security.PySecBufferDescType()
303          tokenbuf = win32security.PySecBufferType(
304              self.pkg_info["MaxToken"], sspicon.SECBUFFER_TOKEN
305          )
306          sec_buffer_out.append(tokenbuf)
307          ## input context handle is None initially, then handle returned from last call thereafter
308          ctxtin = self.ctxt
309          if self.ctxt is None:
310              self.ctxt = win32security.PyCtxtHandleType()
311          err, attr, exp = win32security.AcceptSecurityContext(
312              self.credentials,
313              ctxtin,
314              sec_buffer_in,
315              self.scflags,
316              self.datarep,
317              self.ctxt,
318              sec_buffer_out,
319          )
320  
321          # Stash these away incase someone needs to know the state from the
322          # final call.
323          self.ctxt_attr = attr
324          self.ctxt_expiry = exp
325  
326          if err in (sspicon.SEC_I_COMPLETE_NEEDED, sspicon.SEC_I_COMPLETE_AND_CONTINUE):
327              self.ctxt.CompleteAuthToken(sec_buffer_out)
328  
329          self.authenticated = err == 0
330          if self.authenticated:
331              self._amend_ctx_name()
332  
333          return err, sec_buffer_out

336  if __name__ == "__main__":
337      # This is the security package (the security support provider / the security backend)
338      # we want to use for this example.
339      ssp = "Kerberos"  # or "NTLM" or "Negotiate" which enable negotiation between
340      # Kerberos (prefered) and NTLM (if not supported on the other side).
341  
342      flags = (
343          sspicon.ISC_REQ_MUTUAL_AUTH
344          | sspicon.ISC_REQ_INTEGRITY  # mutual authentication
345          | sspicon.ISC_REQ_SEQUENCE_DETECT  # check for integrity
346          | sspicon.ISC_REQ_CONFIDENTIALITY  # enable out-of-order messages
347          | sspicon.ISC_REQ_REPLAY_DETECT  # request confidentiality  # request replay detection
348      )
349  
350      # Get our identity, mandatory for the Kerberos case *for this example*
351      # Kerberos cannot be used if we don't tell it the target we want
352      # to authenticate to.
353      cred_handle, exp = win32security.AcquireCredentialsHandle(
354          None, ssp, sspicon.SECPKG_CRED_INBOUND, None, None
355      )
356      cred = cred_handle.QueryCredentialsAttributes(sspicon.SECPKG_CRED_ATTR_NAMES)
357      print("We are:", cred)
358  
359      # Setup the 2 contexts. In real life, only one is needed: the other one is
360      # created in the process we want to communicate with.
361      sspiclient = ClientAuth(ssp, scflags=flags, targetspn=cred)
362      sspiserver = ServerAuth(ssp, scflags=flags)
363  
364      print(
365          "SSP : %s (%s)" % (sspiclient.pkg_info["Name"], sspiclient.pkg_info["Comment"])
366      )
367  
368      # Perform the authentication dance, each loop exchanging more information
369      # on the way to completing authentication.
370      sec_buffer = None
371      client_step = 0
372      server_step = 0
373      while not (sspiclient.authenticated) or len(sec_buffer[0].Buffer):
374          client_step += 1
375          err, sec_buffer = sspiclient.authorize(sec_buffer)
376          print("Client step %s" % client_step)
377          if sspiserver.authenticated and len(sec_buffer[0].Buffer) == 0:
378              break
379  
380          server_step += 1
381          err, sec_buffer = sspiserver.authorize(sec_buffer)
382          print("Server step %s" % server_step)
383  
384      # Authentication process is finished.
385      print("Initiator name from the service side:", sspiserver.initiator_name)
386      print("Service name from the client side:   ", sspiclient.service_name)
387  
388      data = "hello".encode("ascii")  # py3k-friendly
389  
390      # Simple signature, not compatible with GSSAPI.
391      sig = sspiclient.sign(data)
392      sspiserver.verify(data, sig)
393  
394      # Encryption
395      encrypted, sig = sspiclient.encrypt(data)
396      decrypted = sspiserver.decrypt(encrypted, sig)
397      assert decrypted == data
398  
399      # GSSAPI wrapping, no encryption (NTLM always encrypts)
400      wrapped = sspiclient.wrap(data)
401      unwrapped, was_encrypted = sspiserver.unwrap(wrapped)
402      print("encrypted ?", was_encrypted)
403      assert data == unwrapped
404  
405      # GSSAPI wrapping, with encryption
406      wrapped = sspiserver.wrap(data, encrypt=True)
407      unwrapped, was_encrypted = sspiclient.unwrap(wrapped)
408      print("encrypted ?", was_encrypted)
409      assert data == unwrapped
410  
411      print("cool!")
