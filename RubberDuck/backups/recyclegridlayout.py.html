24   class RecycleGridLayout(RecycleLayout, GridLayout):
25   
26       _cols_pos = None
27       _rows_pos = None

29       def __init__(self, **kwargs):
30           super(RecycleGridLayout, self).__init__(**kwargs)
31           self.funbind('children', self._trigger_layout)

33       def on_children(self, instance, value):
34           pass

36       def _fill_rows_cols_sizes(self):
37           cols, rows = self._cols, self._rows
38           cols_sh, rows_sh = self._cols_sh, self._rows_sh
39           cols_sh_min, rows_sh_min = self._cols_sh_min, self._rows_sh_min
40           cols_sh_max, rows_sh_max = self._cols_sh_max, self._rows_sh_max
41           self._cols_count = cols_count = [defaultdict(int) for _ in cols]
42           self._rows_count = rows_count = [defaultdict(int) for _ in rows]
43   
44           # calculate minimum size for each columns and rows
45           idx_iter = self._create_idx_iter(len(cols), len(rows))
46           has_bound_y = has_bound_x = False
47           for opt, (col, row) in zip(self.view_opts, idx_iter):
48               (shw, shh), (w, h) = opt['size_hint'], opt['size']
49               shw_min, shh_min = opt['size_hint_min']
50               shw_max, shh_max = opt['size_hint_max']
51   
52               if shw is None:
53                   cols_count[col][w] += 1
54               if shh is None:
55                   rows_count[row][h] += 1
56   
57               # compute minimum size / maximum stretch needed
58               if shw is None:
59                   cols[col] = nmax(cols[col], w)
60               else:
61                   cols_sh[col] = nmax(cols_sh[col], shw)
62                   if shw_min is not None:
63                       has_bound_x = True
64                       cols_sh_min[col] = nmax(cols_sh_min[col], shw_min)
65                   if shw_max is not None:
66                       has_bound_x = True
67                       cols_sh_max[col] = nmin(cols_sh_max[col], shw_max)
68   
69               if shh is None:
70                   rows[row] = nmax(rows[row], h)
71               else:
72                   rows_sh[row] = nmax(rows_sh[row], shh)
73                   if shh_min is not None:
74                       has_bound_y = True
75                       rows_sh_min[row] = nmax(rows_sh_min[row], shh_min)
76                   if shh_max is not None:
77                       has_bound_y = True
78                       rows_sh_max[row] = nmin(rows_sh_max[row], shh_max)
79           self._has_hint_bound_x = has_bound_x
80           self._has_hint_bound_y = has_bound_y

82       def _update_rows_cols_sizes(self, changed):
83           cols_count, rows_count = self._cols_count, self._rows_count
84           cols, rows = self._cols, self._rows
85           remove_view = self.remove_view
86           n_cols = len(cols)
87           n_rows = len(rows)
88           orientation = self.orientation
89   
90           # this can be further improved to reduce re-comp, but whatever...
91           for index, widget, (w, h), (wn, hn), sh, shn, sh_min, shn_min, \
92                   sh_max, shn_max, _, _ in changed:
93               if sh != shn or sh_min != shn_min or sh_max != shn_max:
94                   return True
95               elif (sh[0] is not None and w != wn and
96                     (h == hn or sh[1] is not None) or
97                     sh[1] is not None and h != hn and
98                     (w == wn or sh[0] is not None)):
99                   remove_view(widget, index)
100              else:  # size hint is None, so check if it can be resized inplace
101                  col, row = self._calculate_idx_from_a_view_idx(
102                      n_cols, n_rows, index)
103                  if w != wn:
104                      col_w = cols[col]
105                      cols_count[col][w] -= 1
106                      cols_count[col][wn] += 1
107                      was_last_w = cols_count[col][w] <= 0
108                      if was_last_w and col_w == w or wn > col_w:
109                          return True
110                      if was_last_w:
111                          del cols_count[col][w]
112  
113                  if h != hn:
114                      row_h = rows[row]
115                      rows_count[row][h] -= 1
116                      rows_count[row][hn] += 1
117                      was_last_h = rows_count[row][h] <= 0
118                      if was_last_h and row_h == h or hn > row_h:
119                          return True
120                      if was_last_h:
121                          del rows_count[row][h]
122  
123          return False

125      def compute_layout(self, data, flags):
126          super(RecycleGridLayout, self).compute_layout(data, flags)
127  
128          n = len(data)
129          smax = self.get_max_widgets()
130          if smax and n > smax:
131              raise GridLayoutException(
132                  'Too many children ({}) in GridLayout. Increase rows/cols!'.
133                  format(n))
134  
135          changed = self._changed_views
136          if (changed is None or
137                  changed and not self._update_rows_cols_sizes(changed)):
138              return
139  
140          self.clear_layout()
141          if not self._init_rows_cols_sizes(n):
142              self._cols_pos = None
143              l, t, r, b = self.padding
144              self.minimum_size = l + r, t + b
145              return
146          self._fill_rows_cols_sizes()
147          self._update_minimum_size()
148          self._finalize_rows_cols_sizes()
149  
150          view_opts = self.view_opts
151          for widget, x, y, w, h in self._iterate_layout(n):
152              opt = view_opts[n - widget - 1]
153              shw, shh = opt['size_hint']
154              opt['pos'] = x, y
155              wo, ho = opt['size']
156              # layout won't/shouldn't change previous size if size_hint is None
157              # which is what w/h being None means.
158              opt['size'] = [(wo if shw is None else w),
159                             (ho if shh is None else h)]
160  
161          spacing_x, spacing_y = self.spacing
162          cols, rows = self._cols, self._rows
163  
164          cols_pos = self._cols_pos = [None, ] * len(cols)
165          rows_pos = self._rows_pos = [None, ] * len(rows)
166  
167          cols_pos[0] = self.x
168          last = cols_pos[0] + self.padding[0] + cols[0] + spacing_x / 2.
169          for i, val in enumerate(cols[1:], 1):
170              cols_pos[i] = last
171              last += val + spacing_x
172  
173          last = rows_pos[-1] = \
174              self.y + self.height - self.padding[1] - rows[0] - spacing_y / 2.
175          n = len(rows)
176          for i, val in enumerate(rows[1:], 1):
177              last -= spacing_y + val
178              rows_pos[n - 1 - i] = last
179  
180      def get_view_index_at(self, pos):
181          if self._cols_pos is None:
182              return 0
183  
184          x, y = pos
185          col_pos = self._cols_pos
186          row_pos = self._rows_pos
187          cols, rows = self._cols, self._rows
188          if not col_pos or not row_pos:
189              return 0
190  
191          if x >= col_pos[-1]:
192              ix = len(cols) - 1
193          else:
194              ix = 0
195              for val in col_pos[1:]:
196                  if x < val:
197                      break
198                  ix += 1
199  
200          if y >= row_pos[-1]:
201              iy = len(rows) - 1
202          else:
203              iy = 0
204              for val in row_pos[1:]:
205                  if y < val:
206                      break
207                  iy += 1
208  
209          if not self._fills_from_left_to_right:
210              ix = len(cols) - ix - 1
211          if self._fills_from_top_to_bottom:
212              iy = len(rows) - iy - 1
213          return (iy * len(cols) + ix) if self._fills_row_first else \
214              (ix * len(rows) + iy)

216      def compute_visible_views(self, data, viewport):
217          if self._cols_pos is None:
218              return []
219          x, y, w, h = viewport
220          right = x + w
221          top = y + h
222          at_idx = self.get_view_index_at
223          # 'tl' is not actually 'top-left' unless 'orientation' is 'lr-tb'.
224          # But we can pretend it always is. Same for 'bl' and 'br'.
225          tl, __, bl, br = sorted((
226              at_idx((x, y)),
227              at_idx((right, y)),
228              at_idx((x, top)),
229              at_idx((right, top)),
230          ))
231  
232          n = len(data)
233          indices = []
234          stride = len(self._cols) if self._fills_row_first else len(self._rows)
235          if stride:
236              x_slice = br - bl + 1
237              for s in range(tl, bl + 1, stride):
238                  indices.extend(range(min(s, n), min(n, s + x_slice)))
239          return indices

241      def _calculate_idx_from_a_view_idx(self, n_cols, n_rows, view_idx):
242          '''returns a tuple of (column-index, row-index) from a view-index'''
243          if self._fills_row_first:
244              row_idx, col_idx = divmod(view_idx, n_cols)
245          else:
246              col_idx, row_idx = divmod(view_idx, n_rows)
247          if not self._fills_from_left_to_right:
248              col_idx = n_cols - col_idx - 1
249          if not self._fills_from_top_to_bottom:
250              row_idx = n_rows - row_idx - 1
251          return (col_idx, row_idx, )
