23   def MakeEmptyEnum():
24       # create the Python enumerator object as a real COM object
25       o = win32com.server.util.wrap(win32com.server.util.Collection())
26       return win32com.client.Dispatch(o)

29   def MakeTestEnum():
30       # create a sub-collection, just to make sure it works :-)
31       sub = win32com.server.util.wrap(
32           win32com.server.util.Collection(["Sub1", 2, "Sub3"])
33       )
34       # create the Python enumerator object as a real COM object
35       o = win32com.server.util.wrap(win32com.server.util.Collection([1, "Two", 3, sub]))
36       return win32com.client.Dispatch(o)

39   def TestEnumAgainst(o, check):
40       for i in range(len(check)):
41           if o(i) != check[i]:
42               raise error(
43                   "Using default method gave the incorrect value - %s/%s"
44                   % (repr(o(i)), repr(check[i]))
45               )
46   
47       for i in range(len(check)):
48           if o.Item(i) != check[i]:
49               raise error(
50                   "Using Item method gave the incorrect value - %s/%s"
51                   % (repr(o(i)), repr(check[i]))
52               )
53   
54       # First try looping.
55       cmp = []
56       for s in o:
57           cmp.append(s)
58   
59       if cmp[: len(check)] != check:
60           raise error(
61               "Result after looping isnt correct - %s/%s"
62               % (repr(cmp[: len(check)]), repr(check))
63           )
64   
65       for i in range(len(check)):
66           if o[i] != check[i]:
67               raise error("Using indexing gave the incorrect value")
